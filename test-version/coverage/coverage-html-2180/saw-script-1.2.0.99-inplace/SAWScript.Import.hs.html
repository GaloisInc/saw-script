   24 loadFile :: Options -> FilePath -> IO [Stmt]
   25 loadFile opts fname = do
   26   printOutLn opts Info $ "Loading file " ++ show fname
   27   ftext <- readFile fname
   28   either throwIO return (parseFile fname ftext)
   29 
   30 parseFile :: FilePath -> String -> Either ParseError [Stmt]
   31 parseFile fname input =
   32   case parseModule (lexSAW fname input) of
   33     Left err -> Left err
   34     Right stmts -> Right stmts
   35 
   36 findAndLoadFile :: Options -> FilePath -> IO [Stmt]
   37 findAndLoadFile opts fp = do
   38   let paths = importPath opts
   39   mfname <- findFile paths fp
   40   case mfname of
   41     Nothing -> fail $ unlines $
   42         [ "Couldn't find file: " ++ show fp
   43         , "  Searched in directories:"
   44         ] ++ map ("    " ++) paths
   45     Just fname -> loadFile opts fname
   46 
   47 #if __GLASGOW_HASKELL__ < 706
   48 findFile :: [FilePath] -> String -> IO (Maybe FilePath)
   49 findFile paths fileName = search paths
   50   where
   51     search :: [FilePath] -> IO (Maybe FilePath)
   52     search [] = return Nothing
   53     search (d:ds) = do
   54         let path = d </> fileName
   55         b <- doesFileExist path
   56         if b then return (Just path)
   57              else search ds
   58 #endif
