   51 -- | A representation of a refinement proof goal, i.e., a term of the form:
   52 -- > (a1:A1) -> ... -> (an:An) -> refinesS ev rtp1 rtp2 t1 t2
   53 data RefinesS = RefinesS {
   54   -- | The context of the refinement, i.e. @[(a1,A1), ..., (an,An)]@
   55   -- from the term above
   56   refnCtx :: [(LocalName, Term)],
   57   -- | The event type of the refinement, i.e. @ev@ above
   58   refnEv :: Term,
   59   -- | The LHS return type of the refinement, i.e. @rtp1@ above
   60   refnRType1 :: Term,
   61   -- | The RHS return type of the refinement, i.e. @rtp2@ above
   62   refnRType2 :: Term,
   63   -- | The LHS term of the refinement, i.e. @t1@ above
   64   refnLHS :: Term,
   65   -- | The RHS term of the refinement, i.e. @t2@ above
   66   refnRHS :: Term
   67 }
   68 
   69 -- | Recognizes a term of the form:
   70 -- @(a1:A1) -> ... -> (an:An) -> refinesS ev1 ev2 stack1 stack2 rtp1 rtp2 t1 t2@
   71 -- and returns:
   72 -- @RefinesS [(a1,A1), ..., (an,An)] ev1 ev2 stack1 stack2 rtp1 rtp2 t1 t2@
   73 asRefinesS :: Recognizer Term RefinesS
   74 asRefinesS (asPiList -> (args, asApplyAll ->
   75                          (asGlobalDef -> Just "SpecM.refinesS",
   76                           [ev, rtp1, rtp2,
   77                            asApplyAll -> (asGlobalDef -> Just "SpecM.eqRR", _),
   78                            t1, t2]))) =
   79   Just $ RefinesS args ev rtp1 rtp2 t1 t2
   80 asRefinesS (asPiList -> (_, asApplyAll -> (asGlobalDef -> Just "SpecM.refinesS", _))) =
   81   error "FIXME: MRSolver does not yet accept refinesS goals with non-trivial return relation"
   82 asRefinesS _ = Nothing
   83 
   84 -- | The right-hand-side of a 'FunAssump': either a 'FunName' and arguments, if
   85 -- it is an opaque 'FunAsump', or a 'NormComp', if it is a rewrite 'FunAssump'
   86 data FunAssumpRHS = OpaqueFunAssump FunName [Term]
   87                   | RewriteFunAssump Term
   88 
   89 -- | An assumption that a named function refines some specification. This has
   90 -- the form
   91 --
   92 -- > forall x1, ..., xn. F e1 ... ek |= m
   93 --
   94 -- for some universal context @x1:T1, .., xn:Tn@, some list of argument
   95 -- expressions @ei@ over the universal @xj@ variables, and some right-hand side
   96 -- computation expression @m@.
   97 data FunAssump t = FunAssump {
   98   -- | The uvars that were in scope when this assumption was created
   99   fassumpCtx :: MRVarCtx,
  100   -- | The function on the left-hand-side
  101   fassumpFun :: FunName,
  102   -- | The argument expressions @e1, ..., en@ over the 'fassumpCtx' uvars
  103   fassumpArgs :: [Term],
  104   -- | The right-hand side upper bound @m@ over the 'fassumpCtx' uvars
  105   fassumpRHS :: FunAssumpRHS,
  106   -- | An optional annotation, which in the case of SAWScript, is always a
  107   -- 'TheoremNonce' indicating from which 'Theorem' this assumption comes
  108   fassumpAnnotation :: Maybe t
  109 }
  110 
  111 -- | Recognizes a term of the form:
  112 -- @(a1:A1) -> ... -> (an:An) -> refinesS ev rtp rtp eqRR (f b1 ... bm) t2@,
  113 -- and returns: @FunAssump f [a1,...,an] [b1,...,bm] rhs ann@,
  114 -- where @ann@ is the given argument and @rhs@ is either
  115 -- @OpaqueFunAssump g [c1,...,cl]@ if @t2@ is @g c1 ... cl@,
  116 -- or @RewriteFunAssump t2@ otherwise
  117 asFunAssump :: Maybe t -> Recognizer Term (FunAssump t)
  118 asFunAssump ann (asRefinesS -> Just (RefinesS args
  119                                      (asGlobalDef -> Just "SpecM.VoidEv")
  120                                      _ _ (asApplyAll -> (asGlobalFunName -> Just f1, args1))
  121                                      t2@(asApplyAll -> (asGlobalFunName -> mb_f2, args2)))) =
  122   let rhs = maybe (RewriteFunAssump t2) (\f2 -> OpaqueFunAssump f2 args2) mb_f2
  123    in Just $ FunAssump { fassumpCtx = mrVarCtxFromOuterToInner args,
  124                          fassumpFun = f1, fassumpArgs = args1,
  125                          fassumpRHS = rhs,
  126                          fassumpAnnotation = ann }
  127 asFunAssump _ _ = Nothing

  130 ----------------------------------------------------------------------
  131 -- * Refinement Sets
  132 ----------------------------------------------------------------------
  133 
  134 -- | A set of refinements whose left-hand-sides are function applications,
  135 -- represented as 'FunAssump's. Internally, a map from the 'VarIndex'es of the
  136 -- LHS functions to 'FunAssump's describing the complete refinement.
  137 type Refnset t = HashMap VarIndex (Map [TermProj] (FunAssump t))
  138 
  139 -- | The 'Refnset' with no refinements
  140 emptyRefnset :: Refnset t
  141 emptyRefnset = HashMap.empty
  142 
  143 -- | Given a 'FunName' and a 'Refnset', return the 'FunAssump' which has
  144 -- the given 'FunName' as its LHS function, if possible
  145 lookupFunAssump :: FunName -> Refnset t -> Maybe (FunAssump t)
  146 lookupFunAssump (GlobalName (GlobalDef _ ix _ _ _) projs) refSet =
  147     HashMap.lookup ix refSet >>= Map.lookup projs
  148 lookupFunAssump _ _ = Nothing
  149 
  150 -- | Add a 'FunAssump' to a 'Refnset'
  151 addFunAssump :: FunAssump t -> Refnset t -> Refnset t
  152 addFunAssump fa@(fassumpFun -> GlobalName (GlobalDef _ ix _ _ _) projs) =
  153     HashMap.insertWith (\_ -> Map.insert projs fa) ix
  154                        (Map.singleton projs fa)
  155 addFunAssump _ = error "Cannot insert a non-global name into a Refnset"
  156 
  157 -- | Return the list of 'FunAssump's in a given 'Refnset'
  158 listFunAssumps :: Refnset t -> [FunAssump t]
  159 listFunAssumps = concatMap Map.elems . HashMap.elems

  166 -- | A global MR Solver environment
  167 data MREnv = MREnv {
  168   -- | The debug level, which controls debug printing
  169   mreDebugLevel :: Int,
  170   -- | The options for pretty-printing to be used by MRSolver in error messages
  171   -- and debug printing
  172   mrePPOpts :: PPOpts
  173 }
  174 
  175 -- | The empty 'MREnv'
  176 emptyMREnv :: MREnv
  177 emptyMREnv = MREnv { mreDebugLevel = 0, mrePPOpts = defaultPPOpts }
  178 
  179 -- | Set the debug level of a Mr Solver environment
  180 mrEnvSetDebugLevel :: Int -> MREnv -> MREnv
  181 mrEnvSetDebugLevel dlvl env = env { mreDebugLevel = dlvl }

  188 -- | An entry in 'MREvidence' indicating a usage of an SMT solver or a
  189 -- 'FunAssump'
  190 data MREvidenceEntry t = MREUsedSolver !SolverStats !Term
  191                        | MREUsedFunAssump !t
  192 
  193 -- | Records evidence for the truth of a refinement proposition proved by
  194 -- MRSolver. Currently, this is just a list of 'MREvidenceEntry's indicating
  195 -- each instance where MRSolver needed to use an SMT solver or a 'FunAssump'.
  196 -- FIXME: Have this data type actually provide evidence! i.e. have it keep
  197 -- track of each refinement theorem used by MRSolver along the way.
  198 type MREvidence t = [MREvidenceEntry t]
  199 
  200 -- | Verify that the given evidence in fact supports the given refinement
  201 -- proposition. Returns the identifiers of all the theorems depended on while
  202 -- checking evidence.
  203 -- FIXME: Actually take in a refinement to check against!
  204 checkMREvidence :: Ord t => MREvidence t -> IO (Set t, SolverStats)
  205 checkMREvidence = return . foldMap' checkEntry
  206   where checkEntry (MREUsedSolver stats _) = (mempty, stats)
  207         checkEntry (MREUsedFunAssump t) = (Set.singleton t, mempty)
