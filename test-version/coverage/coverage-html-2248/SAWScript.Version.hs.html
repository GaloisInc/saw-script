    9 {-# LANGUAGE Safe #-}
   10 
   11 module SAWScript.Version (
   12     versionText
   13   , shortVersionText
   14   ) where
   15 
   16 import Paths_saw_script (version)
   17 import GitRev (foundGit, hash, branch)
   18 import Data.Version (showVersion)
   19 
   20 gitText :: String
   21 gitText =
   22     if foundGit then
   23         case (hash, branch) of
   24             (Nothing, Nothing) -> "<non-dev build>"
   25             (Just h, Nothing) -> h ++ " <unknown-branch>"
   26             (Nothing, Just b) -> "<unknown-hash> " ++ b
   27             (Just h, Just b) -> h ++ " " ++ b
   28     else
   29         "<VCS-less build>"
   30 
   31 shortVersionText :: String
   32 shortVersionText = showVersion version ++ " (" ++ gitText ++ ")"
   33 
   34 versionText :: String
   35 versionText = "version " ++ shortVersionText
