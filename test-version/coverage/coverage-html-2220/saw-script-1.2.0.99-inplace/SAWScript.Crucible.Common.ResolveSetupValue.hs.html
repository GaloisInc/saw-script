   18 -- | Resolve a SAWCore 'Term' into a What4 'W4.Pred'.
   19 resolveBoolTerm :: Sym -> Term -> IO (W4.Pred Sym)
   20 resolveBoolTerm sym tm =
   21   do st <- sawCoreState sym
   22      let sc = saw_ctx st
   23      mx <- case getAllExts tm of
   24              -- concretely evaluate if it is a closed term
   25              [] ->
   26                do modmap <- scGetModuleMap sc
   27                   let v = Concrete.evalSharedTerm modmap mempty mempty tm
   28                   pure (Just (Concrete.toBool v))
   29              _ -> return Nothing
   30      case mx of
   31        Just x  -> return (W4.backendPred sym x)
   32        Nothing -> bindSAWTerm sym st W4.BaseBoolRepr tm
