   18 -- | Parse a Java class into a list of declarations
   19 --   Yields an Interaction so that we can talk to the user about what went wrong
   20 getDeclsJVM :: JVM.Class -> IO (Interaction (Maybe [Decl]))
   21 getDeclsJVM cls = return $ do
   22 
   23    let (untranslateable, translations) =
   24          partitionEithers . for (JVM.classMethods cls) $ \method ->
   25             maybe (Left $ JVM.methodName method) Right $ do
   26                returnType <- jssTypeToStdType =<< JVM.methodReturnType method
   27                argTypes <- mapM jssTypeToStdType $ JVM.methodParameterTypes method
   28                let argIndices = map (JVM.localIndexOfParameter method) [0 .. length argTypes - 1]
   29                tableEntries <- forM argIndices $ JVM.lookupLocalVariableByIdx method 0
   30                let args = zipWith Arg (map JVM.localName tableEntries) argTypes
   31                return $ Decl (JVM.methodName method) returnType args
   32 
   33    when (not . null $ untranslateable) $ do
   34       separator ThinSep
   35       liftF . flip Warning () $ "No translation for the following signatures in " ++ JVM.unClassName (JVM.className cls) ++ ".class:"
   36       bulleted $ map (("'" ++) . (++ "'")) untranslateable
   37 
   38    return $ Just translations
   39 
   40    where
   41 
   42       jssTypeToStdType = \case
   43          JVM.ArrayType _    -> Nothing -- TODO: Handle array types
   44          JVM.BooleanType    -> Just $ TCon (TC TCBit) []
   45          JVM.ByteType       -> Just $ bitSeqType 8
   46          JVM.CharType       -> Just $ bitSeqType 16 -- TODO: Not sure about this equivalence
   47          JVM.ClassType _str -> Nothing -- TODO: Support classes as records?
   48          JVM.DoubleType     -> Nothing -- We don't support floating point types
   49          JVM.FloatType      -> Nothing -- We don't support floating point types
   50          JVM.IntType        -> Just $ bitSeqType 32
   51          JVM.LongType       -> Just $ bitSeqType 64
   52          JVM.ShortType      -> Just $ bitSeqType 16
