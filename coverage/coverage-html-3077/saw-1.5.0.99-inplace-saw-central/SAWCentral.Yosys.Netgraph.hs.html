   43 --------------------------------------------------------------------------------
   44 -- ** Building a network graph from a Yosys module
   45 
   46 -- | A 'Netgraph' represents the data dependencies between 'Cell's in
   47 -- a module. The graph has 'Text' keys which are the cell instance
   48 -- names.
   49 data Netgraph = Netgraph
   50   { _netgraphGraph :: Graph.Graph
   51   , _netgraphNodeFromVertex :: Graph.Vertex -> (Cell [Bitrep], CellInstName, [CellInstName])
   52   }
   53 makeLenses ''Netgraph
   54 
   55 moduleNetgraph :: Map CellTypeName ConvertedModule -> Module -> Netgraph
   56 moduleNetgraph env m =
   57   let
   58     sources :: Map Bitrep CellInstName
   59     sources =
   60       Map.fromList $
   61       [ (b, cname)
   62       | (cname, c) <- Map.assocs (m ^. moduleCells)
   63       , b <- concat (Map.elems (cellOutputConnections c)) ]
   64 
   65     isMooreMachine :: CellType -> Bool
   66     isMooreMachine ct =
   67       case ct of
   68         CellTypeUserType t ->
   69           case Map.lookup t env of
   70             Nothing -> False
   71             Just cm ->
   72               case _convertedModuleState cm of
   73                 Nothing -> False
   74                 Just (mt, _) -> mt == Moore
   75         _ -> False
   76 
   77     cellDeps :: Cell [Bitrep] -> [CellInstName]
   78     cellDeps c
   79       | cellIsRegister c = []
   80       | isMooreMachine (c ^. cellType) = []
   81       | otherwise =
   82         Set.toAscList $ Set.fromList $
   83         Maybe.mapMaybe (flip Map.lookup sources) $
   84         concat $ Map.elems $ cellInputConnections c
   85 
   86     nodes :: [(Cell [Bitrep], CellInstName, [CellInstName])]
   87     nodes = [ (c, cname, cellDeps c) | (cname, c) <- Map.assocs (m ^. moduleCells) ]
   88 
   89     (_netgraphGraph, _netgraphNodeFromVertex, _netgraphVertexFromKey) =
   90       Graph.graphFromEdges nodes
   91   in
   92     Netgraph{..}

   94 --------------------------------------------------------------------------------
   95 -- ** Building a SAWCore term from a network graph
   96 
   97 -- | A Mealy machine allows same-cycle data flow from inputs to
   98 -- outputs, while a Moore machine only allows the outputs to depend on
   99 -- the internal state.
  100 data MachineType = Mealy | Moore
  101   deriving Eq
  102 
  103 -- | A SAWCore translation of a module imported from the Yosys JSON format.
  104 data ConvertedModule = ConvertedModule
  105   { _convertedModuleTerm :: SC.Term
  106     -- ^ A SAWCore term representing the module semantics, of type
  107     -- @Input -> Output@ for combinational modules, @{step : Input ->
  108     -- State -> State, out : Input -> State -> Output}@ for Mealy
  109     -- machines, or @{step : Input -> State -> State, out : State ->
  110     -- Output}@ for Moore machines.
  111   , _convertedModuleState :: Maybe (MachineType, SC.Term)
  112     -- ^ The module's state type represented in both SAWCore and
  113     -- Cryptol, or 'Nothing' if the module is purely combinational.
  114   }
  115 makeLenses ''ConvertedModule
  116 
  117 -- | A mapping from wires to their corresponding terms.
  118 type WireEnv = Map [Bitrep] Preterm
  119 
  120 lookupPatternTerm ::
  121   SC.SharedContext ->
  122   YosysBitvecConsumer ->
  123   [Bitrep] ->
  124   WireEnv ->
  125   IO SC.Term
  126 lookupPatternTerm sc loc pat ts =
  127   case Map.lookup pat ts of
  128     -- if we can find that pattern exactly, great! use that term
  129     Just t -> scPreterm sc t
  130     -- otherwise, find each individual bit and append the terms
  131     Nothing ->
  132       do bits <-
  133            forM pat $ \b ->
  134            case Map.lookup [b] ts of
  135              Just t -> pure t
  136              Nothing -> yosysError $ YosysErrorNoSuchOutputBitvec (Text.pack $ show b) loc
  137          -- Yosys lists bits in little-endian order, while scVector expects big-endian, so reverse
  138          let ps = fusePreterms (reverse bits)
  139          scPreterms sc ps

  141 -- | Given a netgraph and an initial map from bit patterns to terms, populate that map with terms
  142 -- generated from the rest of the netgraph.
  143 netgraphToTerms ::
  144   SC.SharedContext ->
  145   Map CellTypeName ConvertedModule ->
  146   Netgraph ->
  147   WireEnv ->
  148   Map CellInstName SC.Term {- ^ state inputs -} ->
  149   IO WireEnv
  150 netgraphToTerms sc env ng inputs states
  151   | length (Graph.scc $ ng ^. netgraphGraph) /= length (ng ^. netgraphGraph)
  152   = yosysError $ YosysError "Network graph contains a cycle after splitting on DFFs; SAW does not currently support analysis of this circuit"
  153   | otherwise =
  154       let sorted = reverseTopSort $ ng ^. netgraphGraph
  155       in foldM doVertex inputs sorted
  156   where
  157     doVertex :: WireEnv -> Graph.Vertex -> IO WireEnv
  158     doVertex acc v =
  159       do let (c, cnm, _deps) = ng ^. netgraphNodeFromVertex $ v
  160          let outputFields = Map.filter isOutput (c ^. cellPortDirections)
  161          let lookupConn portname =
  162                case Map.lookup portname (c ^. cellConnections) of
  163                  Nothing ->
  164                    yosysError $
  165                    YosysError $
  166                    "Malformed Yosys file: Missing port " <> portname <> " for cell " <> cnm
  167                  Just bs ->
  168                    pure bs
  169 
  170          case c ^. cellType of
  171            CellTypeCombinational ctc ->
  172              do let doInput inm i =
  173                       do t <- lookupPatternTerm sc (YosysBitvecConsumerCell cnm inm) i acc
  174                          let w = fromIntegral (length i)
  175                          let s =
  176                                case Map.lookup (inm <> "_SIGNED") (c ^. cellParameters) of
  177                                  Just (Aeson.Number n) -> n > 0
  178                                  Just (Aeson.String x) -> textBinNat x > 0
  179                                  Just _ -> False
  180                                  Nothing -> False
  181                          pure (CellTerm t w s)
  182                 args <- Map.traverseWithKey doInput (cellInputConnections c)
  183                 bs <- lookupConn "Y"
  184                 let ywidth = fromIntegral (length bs)
  185                 t <- combCellToTerm sc ctc args ywidth
  186                 ts <- deriveTermsByIndices sc bs t
  187                 pure $ Map.union ts acc
  188            CellTypeDff ->
  189              do r <-
  190                   case Map.lookup cnm states of
  191                     Nothing ->
  192                       panic "netgraphToTerms" ["missing state for dff cell " <> cnm]
  193                     Just r -> pure r
  194                 bs <- lookupConn "Q"
  195                 ts <- deriveTermsByIndices sc bs r
  196                 pure $ Map.union ts acc
  197            CellTypeFf ->
  198              do r <-
  199                   case Map.lookup cnm states of
  200                     Nothing -> panic "netgraphToTerms" ["missing state for ff cell " <> cnm]
  201                     Just r -> pure r
  202                 bs <- lookupConn "Q"
  203                 ts <- deriveTermsByIndices sc bs r
  204                 pure $ Map.union ts acc
  205            CellTypeUnsupportedPrimitive nm
  206              | Map.null outputFields ->
  207                  -- Cells with no output ports are debugging cells, which we can simply ignore.
  208                  pure acc
  209              | otherwise ->
  210                  yosysError $ YosysErrorUnsupportedFF nm
  211            CellTypeUserType submoduleName ->
  212              case Map.lookup submoduleName env of
  213                Nothing ->
  214                  yosysError $ YosysErrorNoSuchSubmodule submoduleName cnm
  215                Just cm ->
  216                  do inputArgs <-
  217                       case cm ^. convertedModuleState of
  218                         Just (Moore, _) -> pure []
  219                         _ ->
  220                           do let doInput inm i = lookupPatternTerm sc (YosysBitvecConsumerCell cnm inm) i acc
  221                              args <- Map.traverseWithKey doInput (cellInputConnections c)
  222                              rin <- cryptolRecord sc args
  223                              pure [rin]
  224                     let stateArgs = Maybe.maybeToList (Map.lookup cnm states)
  225                     f <-
  226                       case cm ^. convertedModuleState of
  227                         Nothing -> pure (cm ^. convertedModuleTerm)
  228                         Just _ -> SC.scRecordSelect sc (cm ^. convertedModuleTerm) "out"
  229                     r <- SC.scApplyAll sc f (inputArgs ++ stateArgs)
  230 
  231                     -- once we've built a term, insert it along with each of its bits
  232                     ts <-
  233                       forM (Map.assocs $ cellOutputConnections c) $ \(o, out) ->
  234                       do t <- cryptolRecordSelect sc outputFields r o
  235                          deriveTermsByIndices sc out t
  236                     pure $ Map.union (Map.unions ts) acc

  238 -- | Compute the new state value for a stateful cell type.
  239 cellNewState ::
  240   SC.SharedContext ->
  241   Map CellTypeName ConvertedModule ->
  242   WireEnv ->
  243   CellInstName ->
  244   (Cell [Bitrep], SC.Term) ->
  245   IO SC.Term
  246 cellNewState sc env terms cnm (c, prevState) =
  247   case c ^. cellType of
  248     CellTypeDff ->
  249       do bs <- lookupConn "D"
  250          lookupPatternTerm sc (YosysBitvecConsumerCell cnm "D") bs terms
  251     CellTypeFf ->
  252       do bs <- lookupConn "D"
  253          lookupPatternTerm sc (YosysBitvecConsumerCell cnm "D") bs terms
  254     CellTypeCombinational _ ->
  255       panic "cellNewState" ["unexpected combinational cell"]
  256     CellTypeUnsupportedPrimitive _ ->
  257       panic "cellNewState" ["unexpected unsupported cell"]
  258     CellTypeUserType submoduleName ->
  259       case Map.lookup submoduleName env of
  260         Nothing ->
  261           yosysError $ YosysErrorNoSuchSubmodule submoduleName cnm
  262         Just cm ->
  263           do -- Both Mealy and Moore machines have a "step" field of type `Input -> State -> State`
  264              f <- SC.scRecordSelect sc (cm ^. convertedModuleTerm) "step"
  265              let doInput inm i = lookupPatternTerm sc (YosysBitvecConsumerCell cnm inm) i terms
  266              args <- Map.traverseWithKey doInput (cellInputConnections c)
  267              rin <- cryptolRecord sc args
  268              SC.scApplyAll sc f [rin, prevState]
  269   where
  270     lookupConn portname =
  271       case Map.lookup portname (c ^. cellConnections) of
  272         Nothing ->
  273           yosysError $
  274           YosysError $
  275           "Malformed Yosys file: Missing port " <> portname <> " for cell " <> cnm
  276         Just bs ->
  277           pure bs

  279 convertModule ::
  280   SC.SharedContext ->
  281   Map CellTypeName ConvertedModule ->
  282   Module ->
  283   IO ConvertedModule
  284 convertModule sc env m0 =
  285   do let m = renameDffInstances m0
  286      let ng = moduleNetgraph env m
  287 
  288      let inputPorts = moduleInputPorts m
  289      let outputPorts = moduleOutputPorts m
  290 
  291      let
  292        f :: [Bitrep] -> IO SC.Term
  293        f = SC.scBitvector sc . fromIntegral . length
  294 
  295      inputFields <- traverse f inputPorts
  296      inputRecordType <- cryptolRecordType sc inputFields
  297      inputVarName <- SC.scFreshVarName sc "input"
  298      inputRecord <- SC.scVariable sc inputVarName inputRecordType
  299 
  300      derivedInputs <-
  301        forM (Map.assocs inputPorts) $ \(nm, inp) ->
  302        do t <- cryptolRecordSelect sc inputFields inputRecord nm
  303           deriveTermsByIndices sc inp t
  304 
  305      -- Collect state types from all register cells in this module
  306      let
  307        registerCells :: Map CellInstName (Cell [Bitrep])
  308        registerCells = Map.filter cellIsRegister (m ^. moduleCells)
  309        registerPorts :: Map CellInstName [Bitrep]
  310        registerPorts = Map.mapMaybe (\c -> Map.lookup "Q" (c ^. cellConnections)) registerCells
  311      (stateFields1 :: Map CellInstName SC.Term) <- traverse f registerPorts
  312 
  313      -- Collect state types from all submodules
  314      let
  315        getSubmodule :: Cell [Bitrep] -> Maybe ConvertedModule
  316        getSubmodule c =
  317          case c ^. cellType of
  318            CellTypeUserType t -> Map.lookup t env
  319            _ -> Nothing
  320        submodules :: Map CellInstName ConvertedModule
  321        submodules = Map.mapMaybe getSubmodule (m ^. moduleCells)
  322        stateSubmodules :: Map CellInstName (MachineType, SC.Term)
  323        stateSubmodules = Map.mapMaybe (\cm -> cm ^. convertedModuleState) submodules
  324        stateFields2 :: Map CellInstName SC.Term
  325        stateFields2 = fmap snd stateSubmodules
  326        stateFields :: Map CellInstName SC.Term
  327        stateFields = stateFields1 <> stateFields2
  328 
  329      stateRecordType <- cryptolRecordType sc stateFields
  330      stateVarName <- SC.scFreshVarName sc "state"
  331      stateRecord <- SC.scVariable sc stateVarName stateRecordType
  332      oldstates <-
  333        Map.traverseWithKey (\nm _inp -> cryptolRecordSelect sc stateFields stateRecord nm) stateFields
  334 
  335      oneBitType <- SC.scBitvector sc 1
  336      xMsg <- SC.scString sc "Attempted to read X bit"
  337      xTerm <- SC.scGlobalApply sc (SC.mkIdent SC.preludeName "error") [oneBitType, xMsg]
  338      zMsg <- SC.scString sc "Attempted to read Z bit"
  339      zTerm <- SC.scGlobalApply sc (SC.mkIdent SC.preludeName "error") [oneBitType, zMsg]
  340      let inputs = Map.unions $ mconcat
  341            [ [ Map.fromList
  342                [ ( [BitrepZero], PretermBvNat 1 0 )
  343                , ( [BitrepOne], PretermBvNat 1 1 )
  344                , ( [BitrepX], PretermSlice 0 1 0 xTerm )
  345                , ( [BitrepZ], PretermSlice 0 1 0 zTerm )
  346                ]
  347              ]
  348            , derivedInputs
  349            ]
  350 
  351      -- translate outputs of all cells in dependency order
  352      terms <- netgraphToTerms sc env ng inputs oldstates
  353      -- assemble the final output
  354      outputRecord <- cryptolRecord sc =<< mapForWithKeyM outputPorts
  355        (\onm out -> lookupPatternTerm sc (YosysBitvecConsumerOutputPort onm) out terms)
  356 
  357      -- translate the new state components of all stateful cells (if any)
  358      newstates <-
  359        Map.traverseWithKey (cellNewState sc env terms) $
  360        Map.intersectionWith (,) (m ^. moduleCells) oldstates
  361      newstateRecord <- cryptolRecord sc newstates
  362 
  363      case Map.null newstates of
  364        True ->
  365          -- No state; term for combinational module is function `Input -> Output`
  366          do t <- SC.scAbstractTerms sc [inputRecord] outputRecord
  367             pure $
  368               ConvertedModule
  369               { _convertedModuleTerm = t
  370               , _convertedModuleState = Nothing
  371               }
  372        False ->
  373          do step <- SC.scAbstractTerms sc [inputRecord, stateRecord] newstateRecord
  374             case IntSet.member (SC.vnIndex inputVarName) (SC.freeVars outputRecord) of
  375               True ->
  376                 -- Output depends on input, so we have a Mealy machine
  377                 do out <- SC.scAbstractTerms sc [inputRecord, stateRecord] outputRecord
  378                    t <- SC.scRecordValue sc [("out", out), ("step", step)]
  379                    pure $
  380                      ConvertedModule
  381                      { _convertedModuleTerm = t
  382                      , _convertedModuleState = Just (Mealy, stateRecordType)
  383                      }
  384               False ->
  385                 -- Output does not depend on input, so we have a Moore machine
  386                 do out <- SC.scAbstractTerms sc [stateRecord] outputRecord
  387                    t <- SC.scRecordValue sc [("out", out), ("step", step)]
  388                    pure $
  389                      ConvertedModule
  390                      { _convertedModuleTerm = t
  391                      , _convertedModuleState = Just (Moore, stateRecordType)
  392                      }
