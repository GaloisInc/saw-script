  104 -- *** JVMMethodId
  105 
  106 data JVMMethodId =
  107   JVMMethodId
  108     { _jvmMethodKey :: J.MethodKey
  109     , _jvmClassName  :: J.ClassName
  110     }
  111   deriving (Eq, Ord, Show)
  112 
  113 makeLenses ''JVMMethodId
  114 
  115 -- TODO: avoid intermediate 'String' values
  116 instance PPL.Pretty JVMMethodId where
  117   pretty (JVMMethodId methKey className) =
  118     PPL.pretty (concat [J.unClassName className ,".", J.methodKeyName methKey])
  119 
  120 type instance MS.MethodId CJ.JVM = JVMMethodId
  121

  125 data Allocation
