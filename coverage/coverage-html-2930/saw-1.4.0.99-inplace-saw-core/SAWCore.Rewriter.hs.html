   95 data RewriteRule a
   96   = RewriteRule
   97     { ctxt :: [(VarName, Term)]
   98     , lhs :: Term
   99     , rhs :: Term
  100     , permutative :: Bool
  101     , shallow :: Bool
  102     , convertible :: Bool -- ^ flag is true if the rule's LHS and RHS are convertible in SAWcore type system
  103     , annotation :: Maybe a
  104     }
  105   deriving (Show)
  106 -- ^ Invariant: The set of loose variables in @lhs@ must be exactly
  107 -- @[0 .. length ctxt - 1]@. The @rhs@ may contain a subset of these.
  108 
  109 -- NB, exclude the annotation from equality tests
  110 instance Eq (RewriteRule a) where
  111   RewriteRule c1 l1 r1 p1 s1 co1 _a1 == RewriteRule c2 l2 r2 p2 s2 co2 _a2 =
  112     c1 == c2 && l1 == l2 && r1 == r2 && p1 == p2 && s1 == s2 && co1 == co2
  113 
  114 ctxtRewriteRule :: RewriteRule a -> [(VarName, Term)]
  115 ctxtRewriteRule = ctxt

  117 lhsRewriteRule :: RewriteRule a -> Term
  118 lhsRewriteRule = lhs

  120 rhsRewriteRule :: RewriteRule a -> Term
  121 rhsRewriteRule = rhs

  126 -- | Convert a rewrite rule to a proposition (a 'Term' of SAWCore type
  127 -- @Prop@) representing the meaning of the rewrite rule.
  128 propOfRewriteRule :: SharedContext -> RewriteRule a -> IO Term
  129 propOfRewriteRule sc rule =
  130   do ty <- scTypeOf sc (lhs rule)
  131      eq <- scGlobalApply sc "Prelude.Eq" [ty, lhs rule, rhs rule]
  132      scPiList sc (ctxt rule) eq

  150 -- | Equivalent to @(lookup k t, insert k x t)@.
  151 insertLookup :: VarIndex -> a -> IntMap a -> (Maybe a, IntMap a)
  152 insertLookup k x t = IntMap.insertLookupWithKey (\_ a _ -> a) k x t
  153 
  154 firstOrderMatch :: [(VarName, Term)] -> Term -> Term -> Maybe (IntMap Term)
  155 firstOrderMatch ctxt pat term = match pat term IntMap.empty
  156   where
  157     ixs :: IntSet
  158     ixs = IntSet.fromList (map (vnIndex . fst) ctxt)
  159     match :: Term -> Term -> IntMap Term -> Maybe (IntMap Term)
  160     match x y m =
  161       case (unwrapTermF x, unwrapTermF y) of
  162         (Variable (vnIndex -> i) _, _) | IntSet.member i ixs ->
  163             case my' of
  164               Nothing -> Just m'
  165               Just y' -> if alphaEquiv y y' then Just m' else Nothing
  166             where (my', m') = insertLookup i y m
  167         (App x1 x2, App y1 y2) ->
  168             match x1 y1 m >>= match x2 y2
  169         (FTermF xf, FTermF yf) ->
  170             do zf <- zipWithFlatTermF match xf yf
  171                Foldable.foldl (>=>) Just zf m
  172         (_, _) ->
  173             if alphaEquiv x y then Just m else Nothing
  174 -- ^ Precondition: Every loose variable in the pattern @pat@ must
  175 -- occur as the 2nd argument of an @App@ constructor. This ensures
  176 -- that instantiations are well-typed.
  177 
  178 -- | Test if a term is a constant natural number
  179 asConstantNat :: Term -> Maybe Natural
  180 asConstantNat t =
  181   case t of
  182     (R.asGlobalApply preludeZeroIdent -> Just []) -> Just 0
  183     (R.asGlobalApply preludeSuccIdent -> Just [x]) -> (+ 1) <$> asConstantNat x
  184     _ ->
  185       do let (f, xs) = R.asApplyAll t
  186          i <- R.asGlobalDef f
  187          case xs of
  188            [x, y]
  189              | i == "Prelude.addNat" -> (+) <$> asConstantNat x <*> asConstantNat y
  190              | i == "Prelude.mulNat" -> (*) <$> asConstantNat x <*> asConstantNat y
  191              | i == "Prelude.expNat" -> (^) <$> asConstantNat x <*> asConstantNat y
  192              | i == "Prelude.subNat" ->
  193                  do x' <- asConstantNat x
  194                     y' <- asConstantNat y
  195                     guard (x' >= y')
  196                     return (x' - y')
  197              | i == "Prelude.divNat" ->
  198                  do x' <- asConstantNat x
  199                     y' <- asConstantNat y
  200                     guard (y' > 0)
  201                     return (x' `div` y')
  202              | i == "Prelude.remNat" ->
  203                  do x' <- asConstantNat x
  204                     y' <- asConstantNat y
  205                     guard (y' > 0)
  206                     return (x' `rem` y')
  207            _ -> Nothing

  209 -- | An enhanced matcher that can handle higher-order patterns.
  210 --
  211 --   This matching procedure will attempt to find an instantiation
  212 --   for the dangling variables appearing in @pattern@.
  213 --   The resulting instantation will return terms that are in the same
  214 --   variable-scoping context as @term@.  In particular, if @term@
  215 --   is closed, then the terms in the instantiation will also be closed.
  216 scMatch ::
  217   SharedContext ->
  218   [(VarName, Term)] {- ^ context of unification variables in pattern -} ->
  219   Term {- ^ pattern -} ->
  220   Term {- ^ term -} ->
  221   IO (Maybe (IntMap Term))
  222 scMatch sc ctxt pat term =
  223   runMaybeT $
  224   do -- lift $ putStrLn $ "********** scMatch **********"
  225      MatchState inst cs <- match IntSet.empty pat term emptyMatchState
  226      mapM_ (check inst) cs
  227      return inst
  228   where
  229     -- The set of VarIndexes of the unification variables
  230     ixs :: IntSet
  231     ixs = IntSet.fromList (map (vnIndex . fst) ctxt)
  232     -- Check that a constraint of the form pat = n for natural number literal n
  233     -- is satisfied by the supplied substitution (aka instantiation) inst
  234     check :: IntMap Term -> (Term, Natural) -> MaybeT IO ()
  235     check inst (t, n) = do
  236       --lift $ putStrLn $ "checking: " ++ show (t, n)
  237       -- apply substitution to the term
  238       t' <- lift $ scInstantiate sc inst t
  239       --lift $ putStrLn $ "t': " ++ show t'
  240       -- constant-fold nat operations
  241       -- ensure that it evaluates to the same number
  242       case asConstantNat t' of
  243         Just i | i == n -> return ()
  244         _ -> mzero
  245 
  246     -- Check if a term is a higher-order variable pattern, i.e., a free variable
  247     -- (meaning one that can match anything) applied to 0 or more bound variable
  248     -- arguments.
  249     asVarPat :: IntSet -> Term -> Maybe (VarIndex, [(VarName, Term)])
  250     asVarPat locals = go []
  251       where
  252         go js x =
  253           case unwrapTermF x of
  254             Variable nm _tp
  255               | IntSet.member (vnIndex nm) ixs -> Just (vnIndex nm, js)
  256               | otherwise  -> Nothing
  257             App t (unwrapTermF -> Variable nm tp)
  258               | IntSet.member (vnIndex nm) locals -> go ((nm, tp) : js) t
  259             _ -> Nothing
  260 
  261     -- Test if term y matches pattern x, meaning whether there is a substitution
  262     -- to the free variables of x to make it equal to y.
  263     -- The IntSet contains the VarIndexes named variables that are locally bound.
  264     match :: IntSet -> Term -> Term -> MatchState -> MaybeT IO MatchState
  265     match _ x y s
  266       | closedTerm x && termIndex x == termIndex y = pure s
  267     match locals x y s@(MatchState m cs) =
  268       -- (lift $ putStrLn $ "matching (lhs): " ++ ppTermPure PPS.defaultOpts x) >>
  269       -- (lift $ putStrLn $ "matching (rhs): " ++ ppTermPure PPS.defaultOpts y) >>
  270       case asVarPat locals x of
  271         -- If the lhs pattern is of the form (?u b1..bk) where ?u is a
  272         -- unification variable and b1..bk are all locally bound
  273         -- variables: First check whether the rhs contains any locally
  274         -- bound variables *not* in the list b1..bk. If it contains any
  275         -- others, then there is no match. If it only uses a subset of
  276         -- b1..bk, then we can instantiate ?u to (\b1..bk -> rhs).
  277         Just (i, vs) ->
  278           do -- ensure parameter variables are distinct
  279              guard (Set.size (Set.fromList vs) == length vs)
  280              -- ensure y mentions only variables that are in vs
  281              let vset = IntSet.fromList (map (vnIndex . fst) vs)
  282              guard (IntSet.disjoint (IntSet.difference locals vset) (freeVars y))
  283              y2 <- lift $ scLambdaList sc vs y
  284              let (my3, m') = insertLookup i y2 m
  285              case my3 of
  286                Nothing -> return (MatchState m' cs)
  287                Just y3 -> if y2 == y3 then return (MatchState m' cs) else mzero
  288         Nothing ->
  289           case (unwrapTermF x, unwrapTermF y) of
  290             -- check that neither x nor y contains bound variables less than `depth`
  291             (FTermF xf, FTermF yf) ->
  292               case zipWithFlatTermF (match locals) xf yf of
  293                 Nothing -> mzero
  294                 Just zf -> Foldable.foldl (>=>) return zf s
  295             (App x1 x2, App y1 y2) ->
  296               match locals x1 y1 s >>= match locals x2 y2
  297             (Lambda nm t1 x1, Lambda _ t2 x2) ->
  298               match locals t1 t2 s >>= match (IntSet.insert (vnIndex nm) locals) x1 x2
  299             (Pi nm t1 x1, Pi _ t2 x2) ->
  300               match locals t1 t2 s >>= match (IntSet.insert (vnIndex nm) locals) x1 x2
  301             (_, _) ->
  302               -- other possible matches are local vars and constants
  303               if x == y then return s else mzero

  338 -- | Converts a universally quantified equality proposition from a
  339 -- Term representation to a RewriteRule.
  340 ruleOfTerm :: Term -> Maybe a -> RewriteRule a
  341 ruleOfTerm t ann =
  342   do let (vars, body) = R.asPiList t
  343      case R.asGlobalApply eqIdent body of
  344        Just [_, x, y] -> mkRewriteRule vars x y False False ann
  345        _ -> panic "ruleOfTerm" ["Illegal argument"]
  346 
  347 -- Test whether a rewrite rule is permutative
  348 -- this is a rule that immediately loops whether used forwards or backwards.
  349 rulePermutes :: [(VarName, Term)] -> Term -> Term -> Bool
  350 rulePermutes ctxt lhs rhs =
  351     case firstOrderMatch ctxt lhs rhs of
  352         Nothing -> False -- rhs is not an instance of lhs
  353         Just _ ->
  354           case firstOrderMatch ctxt rhs lhs of
  355             Nothing -> False -- but here we have a looping rule, not good!
  356             Just _ -> True
  357 
  358 mkRewriteRule :: [(VarName, Term)] -> Term -> Term -> Bool -> Bool -> Maybe a -> RewriteRule a
  359 mkRewriteRule c l r shallow convFlag ann =
  360     RewriteRule
  361     { ctxt = c
  362     , lhs = l
  363     , rhs = r
  364     , permutative = rulePermutes c l r
  365     , shallow = shallow
  366     , convertible = convFlag
  367     , annotation = ann
  368     }

  370 -- | Converts a universally quantified equality proposition between the
  371 -- two given terms to a RewriteRule.
  372 ruleOfTerms :: Term -> Term -> RewriteRule a
  373 ruleOfTerms l r = mkRewriteRule [] l r False False Nothing

  375 -- | Converts a parameterized equality predicate to a RewriteRule,
  376 -- returning 'Nothing' if the predicate is not an equation.
  377 ruleOfProp :: SharedContext -> Term -> Maybe a -> IO (Maybe (RewriteRule a))
  378 ruleOfProp sc term ann =
  379   case R.asPi term of
  380   Just (nm, tp, body) ->
  381     do rule <- ruleOfProp sc body ann
  382        pure $ (\r -> r { ctxt = (nm, tp) : ctxt r}) <$> rule
  383   Nothing ->
  384     case R.asLambda term of
  385     Just (nm, tp, body) ->
  386       do rule <- ruleOfProp sc body ann
  387          pure $ (\r -> r { ctxt = (nm, tp) : ctxt r}) <$> rule
  388     Nothing ->
  389       case term of
  390         (R.asGlobalApply ecEqIdent -> Just [_, _, x, y]) -> eqRule x y
  391         (R.asGlobalApply bvEqIdent -> Just [_, x, y]) -> eqRule x y
  392         (R.asGlobalApply equalNatIdent -> Just [x, y]) -> eqRule x y
  393         (R.asGlobalApply boolEqIdent -> Just [x, y]) -> eqRule x y
  394         (R.asGlobalApply vecEqIdent -> Just [_, _, _, x, y]) -> eqRule x y
  395         (R.asGlobalApply pairEqIdent -> Just [_, _, _, _, x, y]) -> eqRule x y
  396         (R.asGlobalApply arrayEqIdent -> Just [_, _, x, y]) -> eqRule x y
  397         (R.asGlobalApply intEqIdent -> Just [x, y]) -> eqRule x y
  398         (R.asGlobalApply intModEqIdent -> Just [_, x, y]) -> eqRule x y
  399         (unwrapTermF -> Constant nm) ->
  400           do mres <- lookupVarIndexInMap (nameIndex nm) <$> scGetModuleMap sc
  401              case mres of
  402                Just (ResolvedDef (defBody -> Just body)) -> ruleOfProp sc body ann
  403                _ -> pure Nothing
  404         (R.asEq -> Just (_, x, y)) -> eqRule x y
  405         (R.asEqTrue -> Just body) -> ruleOfProp sc body ann
  406         (R.asApplyAll -> (R.asConstant -> Just nm, args)) ->
  407           do mres <- lookupVarIndexInMap (nameIndex nm) <$> scGetModuleMap sc
  408              case mres of
  409                Just (ResolvedDef (defBody -> Just body)) ->
  410                  do app <- scApplyAllBeta sc body args
  411                     ruleOfProp sc app ann
  412                _ -> pure Nothing
  413         _ -> pure Nothing
  414 
  415   where
  416     eqRule x y = pure $ Just $ mkRewriteRule [] x y False False ann

  426 -- | Transform the given rewrite rule to a set of one or more
  427 -- equivalent rewrite rules, if possible.
  428 --
  429 -- * If the rhs is a lambda, then add an argument to the lhs.
  430 -- * If the rhs is a recursor, then split into a separate rule for each constructor.
  431 -- * If the rhs is a record, then split into a separate rule for each accessor.
  432 scExpandRewriteRule :: SharedContext -> RewriteRule a -> IO (Maybe [RewriteRule a])
  433 scExpandRewriteRule sc (RewriteRule ctxt lhs rhs _ shallow convFlag ann) =
  434   case R.asLambda rhs of
  435   Just (nm, tp, body) ->
  436     do let ctxt' = ctxt ++ [(nm, tp)]
  437        var0 <- scVariable sc nm tp
  438        lhs' <- scApply sc lhs var0
  439        pure $ Just [mkRewriteRule ctxt' lhs' body shallow convFlag ann]
  440   Nothing ->
  441     case rhs of
  442     (R.asRecordValue -> Just m) ->
  443       do let mkRule (k, x) =
  444                do l <- scRecordSelect sc lhs k
  445                   return (mkRewriteRule ctxt l x shallow convFlag ann)
  446          Just <$> traverse mkRule (Map.assocs m)
  447     (R.asApplyAll ->
  448      (R.asRecursorApp -> Just (r, crec),
  449       splitAt (recursorNumParams crec) ->
  450       (params,
  451        motive :
  452        (splitAt (length (recursorCtorOrder crec)) ->
  453         (elims,
  454          splitAt (recursorNumIxs crec) ->
  455          (_ixs, (R.asVariable -> Just (x, xt)) : more))))))
  456       | (ctxt1, _ : ctxt2) <- break ((== x) . fst) ctxt ->
  457       do -- ti is the type of the value being scrutinized
  458          ti <- scWhnf sc xt
  459          -- The datatype parameters are also in context @ctxt1@.
  460          let (_d, (params1, _ixs)) = fmap (splitAt (recursorNumParams crec)) (R.asApplyAll ti)
  461          let ctorRule ctor =
  462                do -- Compute the argument types @argTs@.
  463                   ctorT <- piAppType (ctorType ctor) params1
  464                   argCtx <- fst <$> asFreshPiList sc ctorT
  465                   -- Build a fully-applied constructor @c@.
  466                   args <- scVariables sc argCtx
  467                   c <- scConstApply sc (ctorName ctor) (params1 ++ args)
  468                   -- Define function to substitute the constructor @c@
  469                   -- in for the old local variable @x@.
  470                   let subst = IntMap.singleton (vnIndex x) c
  471                   let adjust t = scInstantiate sc subst t
  472                   -- Build the list of types of the new context.
  473                   ctxt2' <- traverse (traverse adjust) ctxt2
  474                   let ctxt' = ctxt1 ++ argCtx ++ ctxt2'
  475                   -- Substitute the new constructor value to make the
  476                   -- new lhs and rhs in context @ctxt'@.
  477                   lhs' <- adjust lhs
  478 
  479                   r'  <- adjust r
  480                   more' <- traverse adjust more
  481 
  482                   rhs1 <- scReduceRecursor sc r' crec params motive elims (ctorName ctor) args
  483                   rhs2 <- scApplyAll sc rhs1 more'
  484                   rhs3 <- betaReduce rhs2
  485                   -- re-fold recursive occurrences of the original rhs
  486                   let ss = addRule (mkRewriteRule ctxt rhs lhs shallow convFlag Nothing) emptySimpset
  487                   (_,rhs') <- rewriteSharedTerm sc (ss :: Simpset ()) rhs3
  488                   return (mkRewriteRule ctxt' lhs' rhs' shallow convFlag ann)
  489          let d = recursorDataType crec
  490          mm <- scGetModuleMap sc
  491          dt <-
  492            case lookupVarIndexInMap (nameIndex d) mm of
  493              Just (ResolvedDataType dt) -> pure dt
  494              _ -> panic "scExpandRewriteRule" ["Datatype not found: " <> toAbsoluteName (nameInfo d)]
  495          rules <- traverse ctorRule (dtCtors dt)
  496          return (Just rules)
  497     _ -> return Nothing
  498   where
  499     piAppType :: Term -> [Term] -> IO Term
  500     piAppType funtype [] = return funtype
  501     piAppType funtype (arg : args) =
  502       do funtype' <- reducePi sc funtype arg
  503          piAppType funtype' args
  504 
  505     betaReduce :: Term -> IO Term
  506     betaReduce t =
  507       case R.asApp t of
  508         Nothing -> return t
  509         Just (f, arg) ->
  510           do f' <- betaReduce f
  511              case R.asLambda f' of
  512                Nothing -> scApply sc f' arg
  513                Just (vn, _, body) ->
  514                  scInstantiate sc (IntMap.singleton (vnIndex vn) arg) body
  515 
  516 -- | Like 'R.asPiList', but freshen all variables in the context.
  517 asFreshPiList :: SharedContext -> Term -> IO ([(VarName, Term)], Term)
  518 asFreshPiList sc t =
  519   case R.asPi t of
  520     Nothing -> pure ([], t)
  521     Just (x, t1, t2) ->
  522       do -- never use "_" as the base name
  523          let basename = if vnName x == "_" then "_x" else vnName x
  524          x' <- scFreshVarName sc basename
  525          var <- scVariable sc x' t1
  526          t2' <- scInstantiate sc (IntMap.singleton (vnIndex x) var) t2
  527          (ctx, body) <- asFreshPiList sc t2'
  528          pure ((x', t1) : ctx, body)

  530 -- | Repeatedly apply the rule transformations in 'scExpandRewriteRule'.
  531 scExpandRewriteRules :: SharedContext -> [RewriteRule a] -> IO [RewriteRule a]
  532 scExpandRewriteRules sc rs =
  533   case rs of
  534     [] -> return []
  535     r : rs2 ->
  536       do m <- scExpandRewriteRule sc r
  537          case m of
  538            Nothing -> (r :) <$> scExpandRewriteRules sc rs2
  539            Just rs1 -> scExpandRewriteRules sc (rs1 ++ rs2)
  540 
  541 -- | Create a rewrite rule for a definition that expands the definition, if it
  542 -- has a body to expand to, otherwise return the empty list
  543 scDefRewriteRules :: SharedContext -> Def -> IO [RewriteRule a]
  544 scDefRewriteRules sc d =
  545   case defBody d of
  546     Just rhs ->
  547       do lhs <- scConst sc (defName d)
  548          scExpandRewriteRules sc [mkRewriteRule [] lhs rhs False True Nothing]
  549     Nothing ->
  550       pure []

  552 -- | A "shallow" rule is one where further
  553 --   rewrites are not applied to the result
  554 --   of a rewrite.
  555 shallowRule :: RewriteRule a -> RewriteRule a
  556 shallowRule r = r{ shallow = True }

  568 addRule :: RewriteRule a -> Simpset a -> Simpset a
  569 addRule rule | lhs rule /= rhs rule = Net.insert_term (termPat (lhs rule), Left rule)
  570              | otherwise = id
  571 
  572 delRule :: RewriteRule a -> Simpset a -> Simpset a
  573 delRule rule = Net.delete_term (termPat (lhs rule), Left rule)
  574 
  575 addRules :: [RewriteRule a] -> Simpset a -> Simpset a
  576 addRules rules ss = foldr addRule ss rules
  577 
  578 addSimp :: Term -> Maybe a -> Simpset a -> Simpset a
  579 addSimp prop ann = addRule (ruleOfTerm prop ann)
  580 
  581 delSimp :: Term -> Simpset a -> Simpset a
  582 delSimp prop = delRule (ruleOfTerm prop Nothing)
  583 
  584 addConv :: Conversion -> Simpset a -> Simpset a
  585 addConv conv = Net.insert_term (conversionPat conv, Right conv)
  586 
  587 addConvs :: [Conversion] -> Simpset a -> Simpset a
  588 addConvs convs ss = foldr addConv ss convs
  589 
  590 scSimpset :: SharedContext -> [Def] -> [Ident] -> [Conversion] -> IO (Simpset a)
  591 scSimpset sc defs eqIdents convs = do
  592   defRules <- concat <$> traverse (scDefRewriteRules sc) defs
  593   eqRules <- mapM (scEqRewriteRule sc) eqIdents
  594   return $ addRules defRules $ addRules eqRules $ addConvs convs $ emptySimpset
  595 
  596 listRules :: Simpset a -> [RewriteRule a]
  597 listRules ss = [ r | Left r <- Net.content ss ]

  602 asBetaRedex :: R.Recognizer Term (VarName, Term, Term, Term)
  603 asBetaRedex t =
  604     do (f, arg) <- R.asApp t
  605        (s, ty, body) <- R.asLambda f
  606        return (s, ty, body, arg)
  607 
  608 asPairRedex :: R.Recognizer Term Term
  609 asPairRedex t =
  610     do (u, b) <- R.asPairSelector t
  611        (x, y) <- R.asPairValue u
  612        return (if b then y else x)
  613 
  614 asRecordRedex :: R.Recognizer Term Term
  615 asRecordRedex t =
  616     do (x, i) <- R.asRecordSelector t
  617        ts <- R.asRecordValue x
  618        case Map.lookup i ts of
  619          Just t' -> return t'
  620          Nothing -> fail "Record field not found"

  623 -- Bottom-up rewriting
  624 
  625 -- | Term ordering
  626 -- The existing "<" on terms is not adequate for deciding how to handle permutative rules,
  627 -- as then associativity and commutativity can loop.
  628 -- The following rather unsophisticated functions *might* prevent looping.
  629 -- More analysis is needed!
  630 --
  631 -- here we get the "fringe" of arguments in an application, looking at either curried or
  632 -- tupled arguments.  That is
  633 --   for `f x y z`, return [x,y,z]
  634 --   for `f (x,y)` return [x,y]
  635 --   for `f (f x y) z`, return [x,y,z]
  636 --   for `f (x, f (y,z))`, return [x,y,z]
  637 appCollectedArgs :: Term -> [Term]
  638 appCollectedArgs t = step0 t []
  639   where
  640     -- step 0: accumulate curried args, find the function
  641     step0 :: Term -> [Term] -> [Term]
  642     step0 (R.asApp -> Just (f, a)) args = step0 f (a : args)
  643     step0 other args = step1 other args
  644     -- step 1: analyse each arg, knowing the called function, append together
  645     step1 :: Term -> [Term] -> [Term]
  646     step1 f args = foldl (++) [] (map (step2 f) args)
  647     -- step2: analyse an arg.  look inside tuples, sequences (TBD), more calls to f
  648     step2 :: Term -> Term -> [Term]
  649     step2 f (R.asPairValue -> Just (x, y)) = step2 f x ++ step2 f y
  650     step2 f (s@(R.asApp -> Just (g, a))) = possibly_curried_args s f g (step2 f a)
  651     step2 _ a = [a]
  652     --
  653     possibly_curried_args :: Term -> Term -> Term -> [Term] -> [Term]
  654     possibly_curried_args s f (R.asApp -> Just (g, a)) args =
  655       possibly_curried_args s f g (step2 f a ++ args)
  656     possibly_curried_args s f h args = if f == h then args else [s]
  657 
  658 
  659 termWeightLt :: Term -> Term -> Bool
  660 termWeightLt t t' =
  661   (appCollectedArgs t) < (appCollectedArgs t')
  662 
  663 -- | Do a single reduction step (beta, record or tuple selector) at top
  664 -- level, if possible.
  665 reduceSharedTerm :: SharedContext -> Term -> IO (Maybe Term)
  666 reduceSharedTerm sc (asBetaRedex -> Just (vn, _, body, arg)) =
  667   Just <$> scInstantiate sc (IntMap.singleton (vnIndex vn) arg) body
  668 reduceSharedTerm _ (asPairRedex -> Just t) = pure (Just t)
  669 reduceSharedTerm _ (asRecordRedex -> Just t) = pure (Just t)
  670 reduceSharedTerm sc
  671   (R.asApp -> Just (R.asApplyAll -> (R.asRecursorApp -> Just (r, crec),
  672                                      splitAt (recursorNumParams crec) -> (params, motive : elims_ixs)), arg))
  673   | length (recursorCtorOrder crec) + recursorNumIxs crec == length elims_ixs =
  674   do let (f, args) = R.asApplyAll arg
  675      let elims = take (length (recursorCtorOrder crec)) elims_ixs
  676      mm <- scGetModuleMap sc
  677      case R.asConstant f of
  678        Nothing -> pure Nothing
  679        Just c ->
  680          case lookupVarIndexInMap (nameIndex c) mm of
  681            Just (ResolvedCtor ctor) ->
  682              Just <$> scReduceRecursor sc r crec params motive elims c (drop (ctorNumParams ctor) args)
  683            _ -> pure Nothing
  684 reduceSharedTerm _ _ = pure Nothing

  686 data Convertibility = AllRules | ConvertibleRulesOnly

  688 type RewriterCache = IntCache IO Term
  689 
  690 -- | Rewriter for shared terms.  The annotations of any used rules are collected
  691 --   and returned in the result set.
  692 rewriteSharedTerm :: forall a. Ord a => SharedContext -> Simpset a -> Term -> IO (Set a, Term)
  693 rewriteSharedTerm sc ss t0 =
  694     do cache <- newIntCache
  695        let ?cache = cache
  696        setRef <- newIORef mempty
  697        let ?annSet = setRef
  698        t <- rewriteAll AllRules t0
  699        anns <- readIORef setRef
  700        pure (anns, t)
  701 
  702   where
  703 
  704     rewriteAll :: (?cache :: RewriterCache, ?annSet :: IORef (Set a)) => Convertibility -> Term  -> IO Term
  705     rewriteAll convertibleFlag t =
  706         useIntCache ?cache (termIndex t)
  707         (rewriteTermF convertibleFlag (unwrapTermF t) >>= scTermF sc >>= rewriteTop convertibleFlag)
  708 
  709     rewriteTermF :: (?cache :: RewriterCache, ?annSet :: IORef (Set a)) =>
  710                     Convertibility -> TermF Term -> IO (TermF Term)
  711     rewriteTermF convertibleFlag tf =
  712         case tf of
  713           FTermF ftf -> FTermF <$> rewriteFTermF convertibleFlag ftf
  714           App e1 e2 ->
  715               do t1 <- scTypeOf sc e1
  716                  t1' <- scWhnf sc t1
  717                  case unwrapTermF t1' of
  718                    -- If type of e1 is not a dependent type, we can use any rule to rewrite e2
  719                    -- otherwise, we only rewrite using convertible rules
  720                    -- This prevents rewriting e2 from changing type of @App e1 e2@.
  721                    Pi x _ t
  722                      | IntSet.notMember (vnIndex x) (freeVars t) ->
  723                          App <$> rewriteAll convertibleFlag e1 <*> rewriteAll convertibleFlag e2
  724                    _ -> App <$> rewriteAll convertibleFlag e1 <*> rewriteAll ConvertibleRulesOnly e2
  725           Lambda x t1 t2 ->
  726             do var <- scVariable sc x t1 -- we don't rewrite t1 which represents types
  727                t2' <- scInstantiate sc (IntMap.singleton (vnIndex x) var) t2
  728                t2'' <- rewriteAll convertibleFlag t2'
  729                pure (Lambda x t1 t2'')
  730           Constant{}     -> return tf
  731           Variable{}     -> return tf
  732           Pi x t1 t2 ->
  733             do t1' <- rewriteAll convertibleFlag t1
  734                var <- scVariable sc x t1'
  735                t2' <- scInstantiate sc (IntMap.singleton (vnIndex x) var) t2
  736                t2'' <- rewriteAll convertibleFlag t2'
  737                pure (Pi x t1' t2'')
  738 
  739     rewriteFTermF :: (?cache :: RewriterCache, ?annSet :: IORef (Set a)) =>
  740                      Convertibility -> FlatTermF Term -> IO (FlatTermF Term)
  741     rewriteFTermF convertibleFlag ftf =
  742         case ftf of
  743           UnitValue        -> return ftf
  744           UnitType         -> return ftf
  745           PairValue{}      -> traverse (rewriteAll convertibleFlag) ftf
  746           PairType{}       -> traverse (rewriteAll convertibleFlag) ftf
  747           PairLeft{}       -> traverse (rewriteAll convertibleFlag) ftf
  748           PairRight{}      -> traverse (rewriteAll convertibleFlag) ftf
  749           Recursor{}       -> return ftf
  750           RecordType{}     -> traverse (rewriteAll convertibleFlag) ftf
  751           RecordValue{}    -> traverse (rewriteAll convertibleFlag) ftf
  752           RecordProj{}     -> traverse (rewriteAll convertibleFlag) ftf
  753           Sort{}           -> return ftf
  754           ArrayValue t es  -> ArrayValue t <$> traverse (rewriteAll convertibleFlag) es -- specifically NOT rewriting type, only elts
  755           StringLit{}      -> return ftf
  756 
  757     filterRulesFlag :: Convertibility -> Bool -> Bool
  758     filterRulesFlag convertibleFlag isConvertible =
  759       case convertibleFlag of
  760         ConvertibleRulesOnly -> isConvertible
  761         AllRules -> True
  762 
  763     filterRules :: Convertibility -> Either (RewriteRule a) Conversion -> Bool
  764     filterRules convertibleFlag (Left RewriteRule{convertible = ruleConvFlag}) =
  765       filterRulesFlag convertibleFlag ruleConvFlag
  766     filterRules convertibleFlag (Right (Conversion convConvFlag _)) =
  767       filterRulesFlag convertibleFlag convConvFlag
  768 
  769     rewriteTop :: (?cache :: RewriterCache, ?annSet :: IORef (Set a)) => Convertibility -> Term -> IO Term
  770     rewriteTop convertibleFlag t =
  771       do mt <- reduceSharedTerm sc t
  772          case mt of
  773            Nothing -> let filteredRules = filter (filterRules convertibleFlag) (Net.match_term ss (termPat t)) in
  774               apply convertibleFlag filteredRules t
  775            Just t' -> rewriteAll convertibleFlag t'
  776 
  777     recordAnn :: (?annSet :: IORef (Set a)) => Maybe a -> IO ()
  778     recordAnn Nothing  = return ()
  779     recordAnn (Just a) = modifyIORef' ?annSet (Set.insert a)
  780 
  781     apply :: (?cache :: RewriterCache, ?annSet :: IORef (Set a)) =>
  782              Convertibility -> [Either (RewriteRule a) Conversion] -> Term -> IO Term
  783     apply _ [] t = return t
  784     apply convertibleFlag (Left (RewriteRule {ctxt, lhs, rhs, permutative, shallow, annotation}) : rules) t = do
  785       -- if rewrite rule
  786       result <- scMatch sc ctxt lhs t
  787       case result of
  788         Nothing -> apply convertibleFlag rules t
  789         Just inst
  790           | lhs == rhs ->
  791             -- This should never happen because we avoid inserting
  792             -- reflexive rules into simp sets in the first place.
  793             do
  794                lhs' <- ppTerm sc PPS.defaultOpts lhs
  795                putStrLn $ "rewriteSharedTerm: skipping reflexive rule " ++
  796                           "(THE IMPOSSIBLE HAPPENED!): " ++ lhs'
  797                apply convertibleFlag rules t
  798           | IntMap.keysSet inst /= IntSet.fromList (map (vnIndex . fst) ctxt) ->
  799             do
  800                lhs' <- ppTerm sc PPS.defaultOpts lhs
  801                putStrLn $ "rewriteSharedTerm: invalid lhs does not contain all variables: "
  802                  ++ lhs'
  803                apply convertibleFlag rules t
  804           | permutative ->
  805             do
  806               t' <- scInstantiate sc inst rhs
  807               case termWeightLt t' t of
  808                 True -> recordAnn annotation >> rewriteAll convertibleFlag t' -- keep the result only if it is "smaller"
  809                 False -> apply convertibleFlag rules t
  810           | shallow ->
  811             -- do not to further rewriting to the result of a "shallow" rule
  812             do recordAnn annotation
  813                scInstantiate sc inst rhs
  814           | otherwise ->
  815             do recordAnn annotation
  816                rewriteAll convertibleFlag =<< scInstantiate sc inst rhs
  817     -- instead of syntactic rhs, has a bit of code that rewrites lhs (Term -> Maybe Term)
  818     apply convertibleFlag (Right conv : rules) t =
  819         do 
  820           case runConversion conv t of
  821              Nothing -> apply convertibleFlag rules t
  822              Just tb -> rewriteAll convertibleFlag =<< OT.complete sc tb
  823

  827 replaceTerm :: Ord a =>
  828   SharedContext ->
  829   Simpset a    {- ^ A simpset of rewrite rules to apply along with the replacement -} ->
  830   (Term, Term) {- ^ (pat,repl) is a tuple of a pattern term to replace and a replacement term -} ->
  831   Term         {- ^ the term in which to perform the replacement -} ->
  832   IO (Set a, Term)
  833 replaceTerm sc ss (pat, repl) t = do
  834     let rule = ruleOfTerms pat repl
  835     let ss' = addRule rule ss
  836     rewriteSharedTerm sc ss' t

  840 -- If/then/else hoisting
  841 
  842 -- | Find all instances of Prelude.ite in the given term and hoist them
  843 --   higher.  An if-then-else floats upward until it hits a binder that
  844 --   binds one of its free variables, or until it bubbles to the top of
  845 --   the term.  When multiple if-then-else branches bubble to the same
  846 --   place, they will be nested via a canonical term ordering.  This transformation
  847 --   also does rewrites by basic boolean identities.
  848 hoistIfs :: SharedContext
  849          -> Term
  850          -> IO Term
  851 hoistIfs sc t = do
  852    cache <- newIntCache
  853 
  854    rules <- map (\rt -> ruleOfTerm rt Nothing) <$> mapM (scTypeOfIdent sc)
  855               [ "Prelude.ite_true"
  856               , "Prelude.ite_false"
  857               , "Prelude.ite_not"
  858               , "Prelude.ite_nest1"
  859               , "Prelude.ite_nest2"
  860               , "Prelude.ite_eq"
  861               , "Prelude.ite_bit_false_1"
  862               , "Prelude.ite_bit_true_1"
  863               , "Prelude.ite_bit"
  864               , "Prelude.not_not"
  865               , "Prelude.and_True1"
  866               , "Prelude.and_False1"
  867               , "Prelude.and_True2"
  868               , "Prelude.and_False2"
  869               , "Prelude.and_idem"
  870               , "Prelude.or_True1"
  871               , "Prelude.or_False1"
  872               , "Prelude.or_True2"
  873               , "Prelude.or_False2"
  874               , "Prelude.or_idem"
  875               , "Prelude.not_or"
  876               , "Prelude.not_and"
  877               ]
  878    let ss :: Simpset () = addRules rules emptySimpset
  879 
  880    (t', conds) <- doHoistIfs sc ss cache . snd =<< rewriteSharedTerm sc ss t
  881 
  882    -- remove duplicate conditions from the list, as muxing in SAW can result in
  883    -- many copies of the same condition, which cause a performance issue
  884    splitConds sc ss (nubOrd $ map fst conds) t'
  885 
  886 
  887 splitConds :: Ord a => SharedContext -> Simpset a -> [Term] -> Term -> IO Term
  888 splitConds sc ss = go
  889  where
  890    go [] t = return t
  891    go (c:cs) t = go cs =<< splitCond sc ss c t
  892 
  893 splitCond :: Ord a => SharedContext -> Simpset a -> Term -> Term -> IO Term
  894 splitCond sc ss c t = do
  895    ty <- scTypeOf sc t
  896    trueTerm  <- scBool sc True
  897    falseTerm <- scBool sc False
  898 
  899    (_,then_branch) <- replaceTerm sc ss (c, trueTerm) t
  900    (_,else_branch) <- replaceTerm sc ss (c, falseTerm) t
  901    scGlobalApply sc "Prelude.ite" [ty, c, then_branch, else_branch]
  902 
  903 
  904 type HoistIfs s = (Term, [(Term, Map VarName Term)])
  905 
  906 
  907 orderTerms :: SharedContext -> [Term] -> IO [Term]
  908 orderTerms _sc xs = return $ List.sort xs
  909 
  910 doHoistIfs :: Ord a =>
  911   SharedContext ->
  912   Simpset a ->
  913   IntCache IO (HoistIfs s) ->
  914   Term ->
  915   IO (HoistIfs s)
  916 doHoistIfs sc ss hoistCache = go
  917 
  918  where go :: Term -> IO (HoistIfs s)
  919        go t = useIntCache hoistCache (termIndex t) $ top t (unwrapTermF t)
  920 
  921        top :: Term -> TermF Term -> IO (HoistIfs s)
  922        top t tf =
  923          case R.asGlobalApply "Prelude.ite" t of
  924            Just [branch_tp, cond, then_branch, else_branch] ->
  925              do (then_branch',conds1) <- go then_branch
  926                 (else_branch',conds2) <- go else_branch
  927                 t' <- scGlobalApply sc "Prelude.ite" [branch_tp, cond, then_branch', else_branch']
  928                 let vars = getAllVarsMap cond
  929                 return (t', (cond, vars) : conds1 ++ conds2)
  930            _ ->
  931              goF t tf
  932 
  933        goF :: Term -> TermF Term -> IO (HoistIfs s)
  934 
  935        goF t (Constant {}) = return (t, [])
  936        goF t (Variable {}) = return (t, [])
  937 
  938        goF _ (FTermF ftf) = do
  939                 (ftf', conds) <- runWriterT $ traverse WriterT $ (fmap go ftf)
  940                 t' <- scFlatTermF sc ftf'
  941                 return (t', conds)
  942 
  943        goF _ (App f x) = do
  944            (f', conds1) <- go f
  945            (x', conds2) <- go x
  946            t' <- scApply sc f' x'
  947            return (t', conds1 ++ conds2)
  948 
  949        goF _ (Lambda nm tp body) = goBinder scLambda nm tp body
  950        goF _ (Pi nm tp body) = goBinder scPi nm tp body
  951 
  952        goBinder close nm tp body =
  953           do (body'', conds) <- go body
  954              let (stuck, float) = List.partition (Map.member nm . snd) conds
  955 
  956              stuck' <- orderTerms sc (map fst stuck)
  957              body''' <- splitConds sc ss stuck' body''
  958 
  959              t' <- close sc nm tp body'''
  960              return (t', float)
