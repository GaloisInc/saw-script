  127 -- | A specification of a constructor
  128 data Ctor =
  129   Ctor
  130   { ctorNameInfo :: !NameInfo
  131     -- ^ The name of this constructor
  132   , ctorVarIndex :: !VarIndex
  133     -- ^ Unique var index for this constructor
  134   , ctorArgStruct :: CtorArgStruct
  135     -- ^ Arguments to the constructor
  136   , ctorDataType :: !Name
  137     -- ^ The datatype this constructor belongs to
  138   , ctorType :: Term
  139     -- ^ Cached type of the constructor, which should always be equal to
  140     --
  141     -- > (p1::P1) -> .. -> (pn::Pn) -> (x1::arg1) -> .. -> (xm::argm) ->
  142     -- >  d p1 .. pn ix1 .. ixk
  143     --
  144     -- where the @pi@ are the 'ctorParams', the @argi@ are the types specified
  145     -- by the 'ctorArgs', and the @ixi@ are the 'ctorDataTypeIndices'. Note that
  146     -- this type should always be top-level, i.e., have no free variables.
  147   , ctorElimTypeFun :: [Term] -> Term -> IO Term
  148     -- ^ Cached function for generating the type of an eliminator for this
  149     -- constructor by passing it a list of parameters and a @p_ret@ function,
  150     -- also known as the "motive function", which itself must have type
  151     --
  152     -- > (x1::ix1) -> .. -> (xm::ixm) -> d p1 .. pn x1 .. xm -> Type i
  153     --
  154     -- where the @ps@ are the parameters and the @ix@s are the indices of
  155     -- datatype @d@
  156   , ctorIotaReduction ::
  157        Term {- ^ recursor term -} ->
  158        Map VarIndex Term {- ^ constructor eliminators -} ->
  159        [Term] {- ^ constructor arguments -} ->
  160        IO Term
  161     -- ^ Cached function for computing the result of one step of iota
  162     --   reduction of the term
  163     --
  164     -- > RecursorApp rec ixs (c params args)
  165     --
  166     --   The arguments to this function are the recusor value, the
  167     --   the map from the recursor that maps constructors to eliminator
  168     --   functions, and the arguments to the constructor.
  169 
  170   , ctorIotaTemplate :: Term
  171     -- ^ Cached term used for computing iota reductions.  It has free variables
  172     --   @rec@, @elim@ and @args@, in that order so that the last @arg@ is the
  173     --   most recently-bound variable with deBruijn index 0.  The @rec@ variable
  174     --   represents the recursor value, @elim@ represents the eliminator function
  175     --   for the constructor, and @args@ represent the arguments to this constructor.
  176   }
  177 
  178 -- | Return the number of parameters of a constructor
  179 ctorNumParams :: Ctor -> Int
  180 ctorNumParams (Ctor { ctorArgStruct = CtorArgStruct {..}}) =
  181   length ctorParams
  182 
  183 -- | Return the number of non-parameter arguments of a constructor
  184 ctorNumArgs :: Ctor -> Int
  185 ctorNumArgs (Ctor { ctorArgStruct = CtorArgStruct {..}}) =
  186   length ctorArgs
  187 
  188 -- | Compute the 'Name' that uniquely references a constructor
  189 ctorName :: Ctor -> Name
  190 ctorName ctor = Name (ctorVarIndex ctor) (ctorNameInfo ctor)
  191 
  192 lift2 :: (a -> b) -> (b -> b -> c) -> a -> a -> c
  193 lift2 f h x y = h (f x) (f y)
  194 
  195 instance Eq Ctor where
  196   (==) = lift2 ctorName (==)
  197 
  198 instance Ord Ctor where
  199   compare = lift2 ctorName compare
  200 
  201 instance Show Ctor where
  202   show = show . toAbsoluteName . ctorNameInfo

  207 -- | An inductively-defined datatype
  208 data DataType =
  209   DataType
  210   { dtNameInfo :: NameInfo
  211     -- ^ The name of this datatype
  212   , dtVarIndex :: !VarIndex
  213     -- ^ Unique var index for this data type
  214   , dtParams :: [(LocalName, Term)]
  215     -- ^ The context of parameters of this datatype
  216   , dtIndices :: [(LocalName, Term)]
  217     -- ^ The context of indices of this datatype
  218   , dtSort :: Sort
  219     -- ^ The universe of this datatype
  220   , dtCtors :: [Ctor]
  221     -- ^ The list of constructors of this datatype
  222   , dtType :: Term
  223     -- ^ The cached type of this datatype, which should always be equal to
  224     --
  225     -- > (p1::P1) -> .. -> (pn::Pn) -> (i1::I1) -> .. -> (im::Im) -> 'dtSort'
  226     --
  227     -- where the @pi@ are the 'dtParams' and the @ii@ are the 'dtIndices'. Note
  228     -- that this type should always be top-level, i.e., have no free variables.
  229   }
  230 
  231 -- | Return the number of parameters of a datatype
  232 dtNumParams :: DataType -> Int
  233 dtNumParams dt = length $ dtParams dt
  234 
  235 -- | Return the number of indices of a datatype
  236 dtNumIndices :: DataType -> Int
  237 dtNumIndices dt = length $ dtIndices dt
  238 
  239 -- | Compute the ExtCns that uniquely references a datatype
  240 dtExtCns :: DataType -> ExtCns Term
  241 dtExtCns dt = EC (dtVarIndex dt) (dtNameInfo dt) (dtType dt)
  242 
  243 -- | Compute the 'Name' that uniquely references a datatype
  244 dtName :: DataType -> Name
  245 dtName dt = Name (dtVarIndex dt) (dtNameInfo dt)
  246 
  247 instance Eq DataType where
  248   (==) = lift2 dtName (==)
  249 
  250 instance Ord DataType where
  251   compare = lift2 dtName compare
  252 
  253 instance Show DataType where
  254   show = show . dtNameInfo

  268 -- | The different sorts of things that a 'Text' name can be resolved to
  269 data ResolvedName
  270   = ResolvedCtor Ctor
  271   | ResolvedDataType DataType
  272   | ResolvedDef Def
  273 
  274 -- | Get the 'NameInfo' for a 'ResolvedName'
  275 resolvedNameInfo :: ResolvedName -> NameInfo
  276 resolvedNameInfo (ResolvedCtor ctor) = ctorNameInfo ctor
  277 resolvedNameInfo (ResolvedDataType dt) = dtNameInfo dt
  278 resolvedNameInfo (ResolvedDef d) = defNameInfo d
  279 
  280 -- | Get the type of a 'ResolvedName' as a 'Term'.
  281 resolvedNameType :: ResolvedName -> Term
  282 resolvedNameType r =
  283   case r of
  284     ResolvedCtor ctor -> ctorType ctor
  285     ResolvedDataType dt -> dtType dt
  286     ResolvedDef def -> defType def
  287 
  288 -- | Get the 'VarIndex' for a 'ResolvedName'.
  289 resolvedNameVarIndex :: ResolvedName -> VarIndex
  290 resolvedNameVarIndex r =
  291   case r of
  292     ResolvedCtor ctor -> ctorVarIndex ctor
  293     ResolvedDataType dt -> dtVarIndex dt
  294     ResolvedDef def -> defVarIndex def

  296 -- | Modules define namespaces of datatypes, constructors, and definitions,
  297 -- i.e., mappings from 'Text' names to these objects. A module is allowed to
  298 -- map a 'Text' name to an object defined in a different module. Modules also
  299 -- keep a record of the top-level declarations and the imports that were used to
  300 -- build them.
  301 data Module = Module {
  302           moduleName    :: !ModuleName
  303         , moduleResolveMap :: !(Map Text ResolvedName)
  304         , moduleRDecls   :: [ModuleDecl] -- ^ All declarations in reverse order they were added.
  305         }
  306 
  307 emptyModule :: ModuleName -> Module
  308 emptyModule nm =
  309   Module { moduleName = nm
  310          , moduleResolveMap = Map.empty
  311          , moduleRDecls = []
  312          }
  313

  315 -- | Resolve a 'Text' name in the namespace defined by a 'Module', to either a
  316 -- 'Ctor', 'DataType', or 'Def'
  317 resolveName :: Module -> Text -> Maybe ResolvedName
  318 resolveName m str = Map.lookup str (moduleResolveMap m)
  319 
  320 asResolvedCtor :: ResolvedName -> Maybe Ctor
  321 asResolvedCtor = \case { ResolvedCtor ctor -> Just ctor; _ -> Nothing }
  322 
  323 asResolvedDataType :: ResolvedName -> Maybe DataType
  324 asResolvedDataType = \case { ResolvedDataType d -> Just d; _ -> Nothing }
  325 
  326 asResolvedDef :: ResolvedName -> Maybe Def
  327 asResolvedDef = \case { ResolvedDef d -> Just d; _ -> Nothing }
  328 
  329 -- | Resolve a 'Text' name to a 'Ctor'
  330 findCtor :: Module -> Text -> Maybe Ctor
  331 findCtor m str = resolveName m str >>= asResolvedCtor
  332 
  333 -- | Resolve a 'Text' name to a 'DataType'
  334 findDataType :: Module -> Text -> Maybe DataType
  335 findDataType m str = resolveName m str >>= asResolvedDataType
  336 
  337 -- | Resolve a 'Text' name to a 'Def'
  338 findDef :: Module -> Text -> Maybe Def
  339 findDef m str = resolveName m str >>= asResolvedDef

  342 -- | Insert a 'ResolvedName' into a 'Module', adding a mapping from the 'Text'
  343 -- name of that resolved name to it. Signal an error in the case of a name
  344 -- clash, i.e., an existing binding for the same 'Text' name.
  345 insResolvedName :: Module -> ResolvedName -> Module
  346 insResolvedName m nm =
  347   let str = toShortName $ resolvedNameInfo nm in
  348   if Map.member str (moduleResolveMap m) then
  349     panic "insResolvedName" [
  350         "inserting duplicate name " <> str <> " into module " <>
  351             moduleNameText (moduleName m)
  352     ]
  353   else
  354     m { moduleResolveMap = Map.insert str nm (moduleResolveMap m) }
  355 
  356 -- | @insImport i m@ returns the module obtained by importing @i@ into @m@,
  357 -- using a predicate to specify which names are imported from @i@ into @m@. In
  358 -- the case of name clashes, an error is signaled.
  359 insImport :: (ResolvedName -> Bool) -> Module -> Module -> Module
  360 insImport name_p i m =
  361   (foldl' insResolvedName m $ Map.elems $
  362    Map.filter name_p (moduleResolveMap i))
  363 
  364 -- | Insert an \"incomplete\" datatype, used as part of building up a
  365 -- 'DataType' to typecheck its constructors. This incomplete datatype
  366 -- must have no constructors, and it will not be added to the
  367 -- declaration list until it is completed by 'completeDataType'.
  368 -- Return 'Left' in the case of a name clash, i.e., an existing
  369 -- binding for the same 'Ident' name.
  370 beginDataType :: DataType -> ModuleMap -> Either Ident ModuleMap
  371 beginDataType dt =
  372   insResolvedNameInMap (ResolvedDataType dt')
  373   where dt' = dt { dtCtors = [] }
  374 
  375 -- | Complete a datatype, by adding its constructors. Return 'Left' if
  376 -- there is a name clash with any constructor name.
  377 completeDataType :: Ident -> [Ctor] -> ModuleMap -> Either Ident ModuleMap
  378 completeDataType ident ctors mm0 =
  379   let str = identBaseName ident in
  380   case resolveNameInMap mm0 ident of
  381     Just (ResolvedDataType dt)
  382       | null (dtCtors dt) ->
  383         do let dt' = dt {dtCtors = ctors}
  384            let r = ResolvedDataType dt'
  385            let mm1 = insDeclInMap (identModule ident) (TypeDecl dt') mm0
  386            let mm2 = mm1 { mmIndexMap = IntMap.insert (dtVarIndex dt) r (mmIndexMap mm1) }
  387            foldM (flip insResolvedNameInMap) mm2 (map ResolvedCtor ctors)
  388     Just (ResolvedDataType _) ->
  389       panic "completeDataType" ["datatype already completed: " <> str]
  390     Just _ ->
  391       panic "completeDataType" ["not a datatype: " <> str]
  392     Nothing ->
  393       panic "completeDataType" ["datatype not found: " <> str]

  396 -- | Get the resolved names that are local to a module
  397 localResolvedNames :: Module -> [ResolvedName]
  398 localResolvedNames m =
  399   filter isLocal (Map.elems (moduleResolveMap m))
  400   where
  401     isLocal :: ResolvedName -> Bool
  402     isLocal r =
  403       case resolvedNameInfo r of
  404         ModuleIdentifier i -> identModule i == moduleName m
  405         ImportedName{} -> False

  407 -- | Get all definitions defined in a module
  408 moduleDefs :: Module -> [Def]
  409 moduleDefs =
  410   foldr' (\case { ResolvedDef d -> (d :); _ -> id } ) [] .
  411   localResolvedNames

  413 -- | Get all declarations that are local to a module, i.e., that defined names
  414 -- that were not imported from some other module
  415 moduleDecls :: Module -> [ModuleDecl]
  416 moduleDecls = reverse . moduleRDecls

  418 -- | The type of mappings from module names to modules
  419 data ModuleMap =
  420   ModuleMap
  421   { mmNameEnv :: !(Map ModuleName DisplayNameEnv)
  422   , mmIndexMap :: !(IntMap ResolvedName) -- keyed by VarIndex
  423   , mmRDecls :: !(Map ModuleName [ModuleDecl])
  424   }
  425 
  426 emptyModuleMap :: ModuleMap
  427 emptyModuleMap =
  428   ModuleMap
  429   { mmNameEnv = Map.empty
  430   , mmIndexMap = IntMap.empty
  431   , mmRDecls = Map.empty
  432   }

  434 -- | Test whether a 'ModuleName' is in a 'ModuleMap'.
  435 moduleIsLoaded :: ModuleName -> ModuleMap -> Bool
  436 moduleIsLoaded mn mm = Map.member mn (mmRDecls mm)
  437 
  438 loadModule :: Module -> ModuleMap -> ModuleMap
  439 loadModule m mm =
  440   ModuleMap
  441   { mmNameEnv = Map.insert (moduleName m) env (mmNameEnv mm)
  442   , mmIndexMap = IntMap.union indexMap (mmIndexMap mm)
  443   , mmRDecls = Map.insert (moduleName m) (moduleRDecls m) (mmRDecls mm)
  444   }
  445   where
  446     env =
  447       foldr' (\(x, r) -> extendDisplayNameEnv (resolvedNameVarIndex r) [x]) emptyDisplayNameEnv $
  448       Map.assocs (moduleResolveMap m)
  449     indexMap = IntMap.fromList [ (resolvedNameVarIndex r, r) | r <- Map.elems (moduleResolveMap m) ]
  450 
  451 findModule :: ModuleName -> ModuleMap -> Maybe Module
  452 findModule mnm mm =
  453   do decls <- Map.lookup mnm (mmRDecls mm)
  454      env <- Map.lookup mnm (mmNameEnv mm)
  455      let rmap =
  456            Map.mapMaybe (\vi -> IntMap.lookup vi (mmIndexMap mm)) $
  457            Map.fromList [ (x, i) | (x, s) <- Map.assocs (displayIndexes env), i <- IntSet.elems s ]
  458      Just $ Module { moduleName = mnm, moduleResolveMap = rmap, moduleRDecls = decls }

  460 lookupVarIndexInMap :: VarIndex -> ModuleMap -> Maybe ResolvedName
  461 lookupVarIndexInMap vi mm = IntMap.lookup vi (mmIndexMap mm)

  463 requireNameInMap :: Name -> ModuleMap -> ResolvedName
  464 requireNameInMap nm mm =
  465   case lookupVarIndexInMap (nameIndex nm) mm of
  466     Just r -> r
  467     Nothing -> panic "requireNameInMap" ["Constant not found: " <> toAbsoluteName (nameInfo nm)]

  469 resolveNameInMap :: ModuleMap -> Ident -> Maybe ResolvedName
  470 resolveNameInMap mm i =
  471   do env <- Map.lookup (identModule i) (mmNameEnv mm)
  472      vi <-
  473        case resolveDisplayName env (identBaseName i) of
  474          [vi] -> Just vi
  475          _ -> Nothing
  476      IntMap.lookup vi (mmIndexMap mm)

  490 -- | Get all definitions defined in any module in an entire module map. Note
  491 -- that the returned list might have redundancies if a definition is visible /
  492 -- imported in multiple modules in the module map.
  493 allModuleDefs :: ModuleMap -> [Def]
  494 allModuleDefs mm = mapMaybe asResolvedDef (IntMap.elems (mmIndexMap mm))
  495 
  496 -- | Get all local declarations from all modules in an entire module map
  497 allModuleDecls :: ModuleMap -> [ModuleDecl]
  498 allModuleDecls mm = concat (Map.elems (mmRDecls mm))

  500 -- | Get all local declarations from all modules in an entire module map that
  501 -- are marked as primitives
  502 allModulePrimitives :: ModuleMap -> [Def]
  503 allModulePrimitives modmap =
  504     [ def
  505     | DefDecl def <- allModuleDecls modmap
  506     , defQualifier def == PrimQualifier
  507     ]
  508 
  509 -- | Get all local declarations from all modules in an entire module map that
  510 -- are marked as axioms
  511 allModuleAxioms :: ModuleMap -> [Def]
  512 allModuleAxioms modmap =
  513     [ def
  514     | DefDecl def <- allModuleDecls modmap
  515     , defQualifier def == AxiomQualifier
  516     ]
  517 
  518 -- | Get all local declarations from all modules in an entire module map that
  519 -- are marked as neither primitives nor axioms
  520 allModuleActualDefs :: ModuleMap -> [Def]
  521 allModuleActualDefs modmap =
  522     [ def
  523     | DefDecl def <- allModuleDecls modmap
  524     , defQualifier def == NoQualifier
  525     ]

  527 -- | Get all datatypes in all modules in a module map
  528 allModuleDataTypes :: ModuleMap -> [DataType]
  529 allModuleDataTypes mm = mapMaybe asResolvedDataType (IntMap.elems (mmIndexMap mm))

  531 -- | Get all constructors in all modules in a module map

  535 insertResolvedName :: ResolvedName -> ModuleMap -> ModuleMap
  536 insertResolvedName r mm =
  537   mm { mmIndexMap = IntMap.insert vi r (mmIndexMap mm) }
  538   where
  539     vi = resolvedNameVarIndex r

  541 -- | Insert a 'ResolvedName' into a 'ModuleMap', adding a mapping from
  542 -- the 'Ident' name of that resolved name to it. Return 'Left' in the
  543 -- case of a name clash, i.e., an existing binding for the same
  544 -- 'Ident' name.
  545 insResolvedNameInMap :: ResolvedName -> ModuleMap -> Either Ident ModuleMap
  546 insResolvedNameInMap r mm =
  547   let mm' = insertResolvedName r mm in
  548   case resolvedNameInfo r of
  549     ModuleIdentifier ident ->
  550       if Map.member base (displayIndexes env)
  551       then Left ident
  552       else Right $ mm' { mmNameEnv = Map.insert mname env' (mmNameEnv mm) }
  553       where
  554         vi = resolvedNameVarIndex r
  555         mname = identModule ident
  556         base = identBaseName ident
  557         env = fromMaybe emptyDisplayNameEnv $ Map.lookup mname (mmNameEnv mm)
  558         env' = extendDisplayNameEnv vi [base] env
  559     ImportedName{} -> Right mm'
  560 
  561 insDeclInMap :: ModuleName -> ModuleDecl -> ModuleMap -> ModuleMap
  562 insDeclInMap mname decl mm =
  563   mm { mmRDecls = Map.alter (Just . (decl :) . fromMaybe []) mname (mmRDecls mm) }
  564 
  565 -- | Insert a definition into a 'ModuleMap'.
  566 insDefInMap :: Def -> ModuleMap -> Either Ident ModuleMap
  567 insDefInMap d mm =
  568   insResolvedNameInMap (ResolvedDef d) $
  569   case defNameInfo d of
  570     ModuleIdentifier i ->
  571       insDeclInMap (identModule i) (DefDecl d) mm
  572     ImportedName{} -> mm
  573 
  574 -- | Insert an injectCode declaration into a 'ModuleMap'.
  575 insInjectCodeInMap :: ModuleName -> Text -> Text -> ModuleMap -> ModuleMap
  576 insInjectCodeInMap mname ns txt = insDeclInMap mname (InjectCodeDecl ns txt)
  577 
  578 -- | @insImportInMap p m1 m2@ adds all the names from module @m1@ into
  579 -- module @m2@ in the name table of the 'ModuleMap'. The predicate @p@
  580 -- determines which names to import.
  581 insImportInMap :: (Text -> Bool) -> ModuleName -> ModuleName -> ModuleMap -> ModuleMap
  582 insImportInMap p name1 name2 mm =
  583   mm { mmNameEnv = Map.insert name2 env2' (mmNameEnv mm) }
  584   where
  585     env1 = fromMaybe emptyDisplayNameEnv $ Map.lookup name1 (mmNameEnv mm)
  586     env2 = fromMaybe emptyDisplayNameEnv $ Map.lookup name2 (mmNameEnv mm)
  587     env2' = mergeDisplayNameEnv (filterDisplayNameEnv p env1) env2
