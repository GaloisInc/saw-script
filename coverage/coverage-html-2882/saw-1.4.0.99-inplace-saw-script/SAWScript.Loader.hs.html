   47 -- | Type shorthand for an operation that can return warnings and/or
   48 --   errors. On error it returns Left and a list of messages, at least
   49 --   one of which is an error and therefore fatal. On success, it
   50 --   returns a (possibly empty) list of messages, which are all
   51 --   warnings and not fatal, and a result of type a.
   52 type WithMsgs a = Either [Text] ([Text], a)
   53 
   54 -- | The messages produced by the typechecker are pairs of position and
   55 --   String. Convert to Text. XXX: the error infrastructure is supposed
   56 --   to be what knows how to print source positions.
   57 convertTypeMsg :: (Pos, String) -> Text
   58 convertTypeMsg (pos, msg) =
   59     let pos' = Text.pack (show pos)
   60         msg' = Text.pack msg
   61     in
   62     pos' <> ": " <> msg'

   64 -- | Type shorthand for an include path.
   65 --
   66 --   The LHS is the directory the current file was included from
   67 --   (starts as ".") and the RHS is the user's search path.
   68 type IncludePath = (FilePath, [FilePath])

   70 -- | Type shorthand for the set of includes already seen.
   71 --
   72 --   We pass it around as an IORef because it needs to accumulate
   73 --   as we go.
   74 type SeenSet = IORef (Set FilePath)
   75 
   76 -- | Construct an empty SeenSet.
   77 emptySeenSet :: IO SeenSet
   78 emptySeenSet = IORef.newIORef Set.empty
   79 
   80 -- | Check for membership in a SeenSet.
   81 seenSetMember :: FilePath -> SeenSet -> IO Bool
   82 seenSetMember x ref = do
   83     set <- IORef.readIORef ref
   84     pure $ Set.member x set
   85 
   86 -- | Insert into a SeenSet.
   87 seenSetInsert :: FilePath -> SeenSet -> IO ()
   88 seenSetInsert x ref = do
   89     set <- IORef.readIORef ref
   90     IORef.writeIORef ref (Set.insert x set)
   91     pure ()

   93 -- | Wrap statements in a pushd/popd pair.
   94 --
   95 --   See the notes in the interpreter for why we do this.
   96 --
   97 --   FUTURE: if we keep this thing, it would be nice to attach the
   98 --   source position of the include to the push and pop statements.
   99 --
  100 wrapDir :: FilePath -> WithMsgs [Stmt] -> WithMsgs [Stmt]
  101 wrapDir dir result =
  102     case result of
  103         Left errs -> Left errs
  104         Right (msgs, stmts) ->
  105             let stmts' = 
  106                   let pos = PosInternal "pushd/popd derived from include" in
  107                   [StmtPushdir pos dir] ++ stmts ++ [StmtPopdir pos]
  108             in
  109             Right (msgs, stmts')

  111 -- | Read some SAWScript text, using the selected parser entry point.
  112 --
  113 --   Returns only in Either; caller is responsible for handling the
  114 --   resulting warnings and errors.
  115 --
  116 --   The FilePath argument is the name of the file the text comes
  117 --   from. If the text didn't come from a file, it should have a
  118 --   suitable placeholder string; that's what ends up in the source
  119 --   positions in the results.
  120 --
  121 readAny :: FilePath -> Text.Text -> ([Token Pos] -> Either ParseError a) -> WithMsgs a
  122 readAny fileName str parser =
  123     -- XXX printing of positions properly belongs to the
  124     -- message-printing infrastructure.
  125     let positionMsg pos msg =
  126           Text.pack (show pos) <> ": " <> msg
  127     in
  128     case lexSAW fileName str of
  129         Left (_verbosity, pos, msg) ->
  130             Left [positionMsg pos msg]
  131         Right (tokens, optmsg) ->
  132             let msgs = case optmsg of
  133                   Nothing -> []
  134                   Just (Options.Error, _, _) ->
  135                       panic "readAny" ["Lexer returned an error in the warning slot"]
  136                   Just (_verbosity, pos, msg) ->
  137                       [positionMsg pos msg]
  138             in
  139             case parser tokens of
  140                 Left err ->
  141                     let err' = Text.pack (show err) in
  142                     Left (msgs ++ [err'])
  143                 Right tree ->
  144                     -- Note: there's no such things as warnings in a
  145                     -- Happy parser, so there are no more messages to
  146                     -- add in this branch.
  147                     Right (msgs, tree)
  148 
  149 -- | Use the readAny result to panic if any messages were generated.
  150 panicOnMsgs :: Text -> WithMsgs a -> a
  151 panicOnMsgs whoAmI result = case result of
  152    Left errs ->
  153        panic whoAmI ("Unexpected errors:" : errs)
  154    Right ([], tree) ->
  155        tree
  156    Right (warns, _) ->
  157        panic whoAmI ("Unexpected warnings:" : warns)
  158 
  159 -- | Handle the readAny result in IO.
  160 --
  161 --   Also return in Either, to be consistent with the current interface
  162 --   into this file. Prints any warnings, and return just errors or a
  163 --   result.
  164 --
  165 dispatchMsgs :: Options -> WithMsgs a -> IO (Either [Text] a)
  166 dispatchMsgs opts result =
  167     let printMsg vrb msg =
  168           Options.printOutLn opts vrb (Text.unpack msg)
  169     in
  170     case result of
  171         Left errs ->
  172             pure $ Left errs
  173         Right (msgs, tree) -> do
  174             mapM_ (printMsg Options.Warn) msgs
  175             pure $ Right tree
  176 
  177 -- | Run the readAny result through the `Include` module to resolve
  178 --   @include@ statements.
  179 --
  180 --   If we fail after collecting warnings, print the warnings before
  181 --   returning the failure.
  182 --
  183 resolveIncludes ::
  184     Int -> SeenSet -> IncludePath -> Options -> Inc.Processor a ->
  185     WithMsgs a ->
  186     IO (WithMsgs a)
  187 resolveIncludes depth seen incpath opts process result =
  188     let printMsg vrb msg =
  189           Options.printOutLn opts vrb (Text.unpack msg)
  190     in
  191     case result of
  192         Left errs ->
  193             pure $ Left errs
  194         Right (msgs, tree) -> do
  195             result' <- process (includeFile depth seen incpath opts) tree
  196             case result' of
  197                 Left errs -> do
  198                     mapM_ (printMsg Options.Warn) msgs
  199                     pure $ Left errs
  200                 Right tree' ->
  201                     pure $ Right (msgs, tree')

  203 -- | Read a type schema from a string. This is used to digest the type
  204 --   signatures for builtins, and the expansions for builtin typedefs.
  205 --
  206 -- Pure function that panics on any error or warning. This is
  207 -- appropriate since we use it only to handle the builtins; any
  208 -- glitches there are properly panics.
  209 --
  210 -- The first argument (fileName) is a string to pass as the
  211 -- filename for the lexer, which (complete with line and column
  212 -- numbering of dubious value) will go into the positions of the
  213 -- elements of the resulting type.
  214 --
  215 -- The fake file name names the builtin object involved, so we'll also
  216 -- use it as the panic location string if we need to panic.
  217 --
  218 -- FUTURE: we should figure out how to generate more meaningful
  219 -- positions (like "third argument of concat") but this at least
  220 -- allows telling the user which builtin the type came from.
  221 --
  222 -- Note: there is no way to reach @include@ statements from type
  223 -- schemas, so no need to process includes in what we read.
  224 --
  225 readSchemaPure ::
  226     FilePath ->
  227     PrimitiveLifecycle ->
  228     ScopedMap Name (PrimitiveLifecycle, NamedType) ->
  229     Text ->
  230     Schema
  231 readSchemaPure fakeFileName lc tyenv str = do
  232     let result = readAny fakeFileName str parseSchema
  233     let result' = case result of
  234           Left errs -> Left errs
  235           Right (msgs, schema) ->
  236               let (errs_or_results, warns) = checkSchema lc tyenv schema
  237                   warns' = map convertTypeMsg warns
  238               in
  239               case errs_or_results of
  240                   Left errs -> Left (msgs ++ warns' ++ map convertTypeMsg errs)
  241                   Right () -> Right (msgs ++ warns', schema)
  242     panicOnMsgs (Text.pack fakeFileName) result'

  244 -- | Read a schema pattern from a string. This is used by the
  245 --   :search REPL command.
  246 --
  247 --   Also runs the typechecker to check the pattern.
  248 --
  249 -- Note: there is no way to reach @include@ statements from schema
  250 -- patterns, so no need to process includes in what we read.
  251 --
  252 readSchemaPattern ::
  253     Options ->
  254     FilePath -> Environ -> RebindableEnv -> Set PrimitiveLifecycle -> Text ->
  255     IO (Either [Text] SchemaPattern)
  256 readSchemaPattern opts fileName environ rbenv avail str = do
  257   let result = readAny fileName str parseSchemaPattern
  258   let result' = case result of
  259         Left errs -> Left errs
  260         Right (msgs, pat) ->
  261           let Environ varenv tyenv _cryenv = environ in
  262 
  263           -- XXX it should not be necessary to do this munging
  264           --
  265           -- Note that we need to flatten the scopes we have here
  266           -- (outside the typechecker) in order to have the union with
  267           -- the rebindable env to work right.
  268           let squash (pos, lc, ty, _val, _doc) = (pos, lc, ReadOnlyVar, ty)
  269               varenv' = Map.map squash $ ScopedMap.flatten varenv
  270               rbsquash (pos, ty, _val) = (pos, Current, RebindableVar, ty)
  271               rbenv' = Map.map rbsquash rbenv
  272               varenv'' = Map.union varenv' rbenv'
  273               varenv''' = ScopedMap.seed varenv''
  274           in
  275           let (errs_or_results, warns) = checkSchemaPattern avail varenv''' tyenv pat
  276               warns' = map convertTypeMsg warns
  277           in
  278           case errs_or_results of
  279               Left errs -> Left (msgs ++ warns' ++ map convertTypeMsg errs)
  280               Right results -> Right (msgs ++ warns', results)
  281   dispatchMsgs opts result'

  283 -- | Read an expression from a string. This is used by the
  284 --   :type REPL command.
  285 --
  286 -- It is possible for someone to provide an expression that includes
  287 -- an @include@ in a do-block, so just in case call into `Include`
  288 -- to resolve any that appear.
  289 --
  290 readExpression ::
  291     Options ->
  292     FilePath -> Environ -> RebindableEnv -> Set PrimitiveLifecycle -> Text ->
  293     IO (Either [Text] (Schema, Expr))
  294 readExpression opts fileName environ rbenv avail str = do
  295   seen <- emptySeenSet
  296   let incpath = (".", Options.importPath opts)
  297 
  298   let result = readAny fileName str parseExpression
  299   result' <- resolveIncludes 0{-depth-} seen incpath opts Inc.processExpr result
  300   let result'' = case result' of
  301         Left errs -> Left errs
  302         Right (msgs, expr) ->
  303            let Environ varenv tyenv _cryenvs = environ in
  304 
  305            -- XXX it should not be necessary to do this munging
  306            let squash (defpos, lc, ty, _val, _doc) = (defpos, lc, ReadOnlyVar, ty)
  307                varenv' = Map.map squash $ ScopedMap.flatten varenv
  308                rbsquash (defpos, ty, _val) = (defpos, Current, RebindableVar, ty)
  309                rbenv' = Map.map rbsquash rbenv
  310                varenv'' = Map.union varenv' rbenv'
  311                varenv''' = ScopedMap.seed varenv''
  312            in
  313            -- XXX: also it shouldn't be necessary to do this wrappery
  314            let pos = getPos expr
  315                decl = Decl pos (PWild pos Nothing) Nothing expr
  316            in
  317            let (errs_or_results, warns) = checkDecl avail varenv''' tyenv decl
  318                warns' = map convertTypeMsg warns
  319            in
  320            case errs_or_results of
  321                Left errs -> Left (msgs ++ warns' ++ map convertTypeMsg errs)
  322                Right decl' ->
  323                    let expr' = dDef decl'
  324                        schema = case dType decl' of
  325                          Just sch -> sch
  326                          Nothing ->
  327                              -- If the typechecker didn't insert a type, it's bust,
  328                              -- so panic. Not much point in printing the expression
  329                              -- or position in panic, since it's what the user just
  330                              -- typed.
  331                              panic "readExpressionChecked" [
  332                                  "Typechecker failed to produce a type"
  333                              ]
  334                    in
  335                    Right (msgs ++ warns', (schema, expr'))
  336   dispatchMsgs opts result''

  352 -- | Find a file, potentially looking in a list of multiple search paths (as
  353 -- specified via the @SAW_IMPORT_PATH@ environment variable or
  354 -- @-i@/@--import-path@ command-line options). If the file was successfully
  355 -- found, return the full path. If not, fail by returning `Left`.
  356 --
  357 locateFile :: IncludePath -> FilePath -> IO (Either [Text] FilePath)
  358 locateFile (current, rawdirs) file = do
  359   -- Replace "." in the search path with the directory the current
  360   -- file came from, which might or might not be ".".
  361   let dirs = map adjust rawdirs
  362       adjust "." = current
  363       adjust dir = dir
  364 
  365   mfname <- findFile dirs file
  366   case mfname of
  367     Nothing -> do
  368         let msgs = [
  369                 "Couldn't find file: " <> Text.pack file,
  370                 "  Searched in directories:"
  371              ] ++ map (\p -> "    " <> Text.pack p) dirs
  372         return $ Left msgs
  373     Just fname ->
  374       -- NB: Normalise the path name. The default SAW_IMPORT_PATH contains ".",
  375       -- and the behavior of filepath's 'normalise' function is to prepend a
  376       -- search path to the front of the file path that is found, which can
  377       -- cause paths like "./foo.saw" to be returned. This looks ugly in error
  378       -- messages, where we would rather display "foo.saw" instead.
  379       return $ Right (normalise fname)
  380 
  381 -- | Load the 'Stmt's in a @.saw@ file.
  382 --   Doesn't run the typechecker (yet).
  383 includeFile ::
  384     Int -> SeenSet -> IncludePath -> Options -> FilePath -> Bool ->
  385     IO (Either [Text] [Stmt])
  386 includeFile depth seen incpath opts fname once = do
  387   result <- locateFile incpath fname
  388   case result of
  389     Left errs -> return $ Left errs
  390     Right fname' -> do
  391       alreadySeen <- seenSetMember fname' seen 
  392       if depth > 128 then
  393           pure $ Left ["Maximum include depth exceeded"]
  394       else if once && alreadySeen then do
  395           -- include_once and we've already seen this file
  396           Options.printOutLn opts Options.Info $ "Skipping already-included file " ++
  397                                                  show fname'
  398           pure $ Right []
  399       else do
  400           seenSetInsert fname' seen
  401           let (_current, dirs) = incpath
  402               current' = takeDirectory fname'
  403               incpath' = (current', dirs)
  404 
  405           Options.printOutLn opts Options.Info $ "Loading file " ++ show fname'
  406           ftext <- TextIO.readFile fname'
  407 
  408           let result' = wrapDir current' $ readAny fname ftext parseModule
  409           result'' <- resolveIncludes (depth + 1) seen incpath' opts Inc.processStmts result'
  410           dispatchMsgs opts result''

  412 -- | Find a file, potentially looking in a list of multiple search paths (as
  413 -- specified via the @SAW_IMPORT_PATH@ environment variable or
  414 -- @-i@/@--import-path@ command-line options). If the file was successfully
  415 -- found, load it. If not, fail by returning `Left`.
  416 --
  417 -- Doesn't run the typechecker (yet).
  418 findAndLoadFileUnchecked :: Options -> FilePath -> IO (Either [Text] [Stmt])
  419 findAndLoadFileUnchecked opts fp = do
  420   let depth = 0
  421   seen <- emptySeenSet
  422   let incpath = (".", Options.importPath opts)
  423       once = False
  424   includeFile depth seen incpath opts fp once
