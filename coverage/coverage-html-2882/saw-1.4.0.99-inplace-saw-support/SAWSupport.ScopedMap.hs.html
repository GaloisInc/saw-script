    3 -- |
    4 -- Module      : SAWSupport.ScopedMap
    5 -- Description : Scoped-aware finite map library
    6 -- License     : BSD3
    7 -- Maintainer  : saw@galois.com
    8 -- Stability   : provisional
    9 --
   10 -- Scope-aware map for use as an interpreter or typechecker
   11 -- environment.
   12 --
   13 -- The `ScopedMap` type is a stack of maps; each layer corresponds
   14 -- to a scope as the client moves into and out of nested naming
   15 -- environments.
   16 --
   17 -- The current (innermost) scope is at the top of the stack / head of
   18 -- the list.
   19 --
   20 -- Update operations act on the current scope. Lookup operations
   21 -- inspect the whole stack, starting at the innermost layer and
   22 -- working out.
   23 --
   24 -- The `push` and `pop` operations add and drop scopes respectively,
   25 -- and should be used when the client enters and leaves a nested
   26 -- naming environment.
   27 --
   28 -- The `flatten` operation produces a single-layer Data.Map that
   29 -- contains all the visible mappings. (That is, duplicate keys in
   30 -- outer scopes are suppressed.)
   31 --
   32 -- Other operations correspond to the ordinary operations on
   33 -- Data.Map. Operations that generate collections (e.g. @keys@) come
   34 -- in two versions: one that preserves the scope structure
   35 -- (generally called @scoped@), and one that flattens it (generally
   36 -- called @all@).
   37 --
   38 -- In principle we should provide all the operations on Data.Map.
   39 -- However, the implementation has been lazily evaluated and the
   40 -- operations present are the ones that have been needed so far.
   41 -- Add more as desired.
   42

   75 -- | The type of scoped maps.
   76 --
   77 --   The list of scopes is a `NonEmpty` to avoid unnecessary
   78 --   degenerate cases.
   79 newtype ScopedMap k v = ScopedMap (NonEmpty (Map k v))
   80 
   81 -- | Push a new scope.
   82 push :: Ord k => ScopedMap k v -> ScopedMap k v
   83 push (ScopedMap scopes) =
   84     ScopedMap $ N.cons Map.empty scopes
   85 
   86 -- | Pop and discard the most recent scope.
   87 --
   88 --   Popping the last scope is an error and will panic.
   89 pop :: Ord k => ScopedMap k v -> ScopedMap k v
   90 pop (ScopedMap (_ :| more)) = case more of
   91     [] ->
   92         panic "pop" ["Popping topmost scope"]
   93     s : more' ->
   94         ScopedMap (s :| more')
   95 
   96 -- | An empty ScopedMap, with a single empty scope.
   97 empty :: Ord k => ScopedMap k v
   98 empty =
   99     ScopedMap (Map.empty :| [])
  100 
  101 -- | Initialize a ScopedMap from a Map, which becomes the current (and
  102 --   only) scope.
  103 seed :: Ord k => Map k v -> ScopedMap k v
  104 seed m =
  105     ScopedMap (m :| [])
  106 
  107 -- | Insert into a ScopedMap. Always inserts in the innermost (most
  108 --   recent) scope.
  109 insert :: Ord k => k -> v -> ScopedMap k v -> ScopedMap k v
  110 insert k v (ScopedMap (s :| more)) =
  111     let s' = Map.insert k v s in
  112     ScopedMap (s' :| more)
  113 
  114 -- | Look up a key in a ScopedMap. Checks all the scopes with the most
  115 --   recent first.
  116 lookup :: Ord k => k -> ScopedMap k v -> Maybe v
  117 lookup k (ScopedMap (s0 :| more0)) =
  118     let visit s more = case Map.lookup k s of
  119           Just result -> Just result
  120           Nothing -> case more of
  121               [] -> Nothing
  122               s' : more' -> visit s' more'
  123     in
  124     visit s0 more0
  125 
  126 -- | Drop entries that don't match a predicate.
  127 filter :: Ord k => (v -> Bool) -> ScopedMap k v -> ScopedMap k v
  128 filter keep (ScopedMap (s :| more)) =
  129     ScopedMap (Map.filter keep s :| Prelude.map (Map.filter keep) more)
  130 
  131 -- | Call a function on each entry. Processes all entries, including
  132 --   hidden entries in outer scopes.
  133 map :: Ord k => (v -> v') -> ScopedMap k v -> ScopedMap k v'
  134 map f (ScopedMap (s :| more)) =
  135     ScopedMap (Map.map f s :| Prelude.map (Map.map f) more)
  136 
  137 -- | Union. Takes the union of all layers; this may result in inner
  138 --   entries from one side hiding outer entries in the other. Any
  139 --   overlaps in the same scope are handled as per Map.union; that is,
  140 --   the entry from the left-hand map is preferred.
  141 union :: Ord k => ScopedMap k v -> ScopedMap k v -> ScopedMap k v
  142 union (ScopedMap (s1 :| more1)) (ScopedMap (s2 :| more2)) =
  143     ScopedMap (Map.union s1 s2 :| zipWith Map.union more1 more2)
  144 
  145 -- | Return Map.assocs for each scope, preserving the scope boundaries.
  146 --   The head of the returned list is the most recent scope.
  147 scopedAssocs :: Ord k => ScopedMap k v -> [[(k, v)]]
  148 scopedAssocs (ScopedMap scopes) =
  149     Prelude.map Map.assocs $ toList scopes
  150

  153 -- | Convert to a plain map by folding the scopes together. Duplicate
  154 --   bindings in more recent scopes take priority.
  155 flatten :: Ord k => ScopedMap k v -> Map k v
  156 flatten (ScopedMap (s0 :| more0)) =
  157     let visit s more =
  158           let result = case more of
  159                 [] -> Map.empty
  160                 s' : more' -> visit s' more'
  161           in
  162           -- Map.union gives its left argument priority given a
  163           -- duplicate key.
  164           Map.union s result
  165     in
  166     visit s0 more0
  167

  170 -- | Return all values, ignoring scope.
  171 allElems :: Ord k => ScopedMap k v -> [v]
  172 allElems m =
  173     Map.elems $ flatten m
  174 
  175 -- | Return all keys, ignoring scope.
  176 allKeys :: Ord k => ScopedMap k v -> [k]
  177 allKeys m =
  178     Map.keys $ flatten m
  179 
  180 -- | Return all keys as a set, ignoring scope.
  181 allKeysSet :: Ord k => ScopedMap k v -> Set k
  182 allKeysSet m =
  183     Map.keysSet $ flatten m
