  101 parseType :: LLVM.Type -> IO TypeSkeleton
  102 parseType (LLVM.PtrTo t) = pure $ TypeSkeleton t True [SizeGuess 1 True "default guess of size 1"]
  103 -- It is unclear how to combine opaque pointers with type skeletons due to the
  104 -- lack of a pointee type. For now, we simply fail if we encounter one
  105 -- (see #1877).
  106 parseType LLVM.PtrOpaque =
  107   panic "SAWCentral.Crucible.LLVM.Skeleton.parseType"
  108         [ "Skeleton generation does not support opaque pointers"
  109         , "Please report this at: https://github.com/GaloisInc/saw-script/issues/1877"
  110         ]
  111 parseType (LLVM.Array i t) = pure $ TypeSkeleton t True
  112   [ SizeGuess (fromIntegral i) True $ "default guess of size " <> Text.pack (show i)
  113   ]
  114 parseType t = pure $ TypeSkeleton t False []
  115 
  116 parseGlobal :: Map String Int -> LLVM.Global -> IO GlobalSkeleton
  117 parseGlobal ls LLVM.Global
  118   { LLVM.globalSym = LLVM.Symbol s
  119   , LLVM.globalType = t
  120   , LLVM.globalValue = v
  121   , LLVM.globalAttrs = LLVM.GlobalAttrs { LLVM.gaConstant = c }
  122   } = do
  123   ty <- parseType t
  124   pure GlobalSkeleton
  125     { _globSkelName = Text.pack s
  126     , _globSkelLoc = flip Location Nothing <$> Map.lookup s ls
  127     , _globSkelType = ty
  128     , _globSkelMutable = not c
  129     , _globSkelInitialized = isJust v
  130     }

  132 parseArg :: LLVM.Typed LLVM.Ident -> (Maybe String, Maybe Location) -> IO ArgSkeleton
  133 parseArg LLVM.Typed { LLVM.typedType = t } (nm, loc) = do
  134   ty <- parseType t
  135   pure ArgSkeleton
  136     { _argSkelName = Text.pack <$> nm
  137     , _argSkelLoc = loc
  138     , _argSkelType = ty
  139     }

  141 stmtCalls :: [LLVM.Stmt] -> Set Text
  142 stmtCalls [] = Set.empty
  143 stmtCalls (LLVM.Result _ (LLVM.Call _ _ (LLVM.ValSymbol (LLVM.Symbol s)) _) _ _:stmts) =
  144   Set.insert (Text.pack s) $ stmtCalls stmts
  145 stmtCalls (LLVM.Effect (LLVM.Call _ _ (LLVM.ValSymbol (LLVM.Symbol s)) _) _ _:stmts) =
  146   Set.insert (Text.pack s) $ stmtCalls stmts
  147 stmtCalls (_:stmts) = stmtCalls stmts
  148 
  149 -- | Infer locations of declarations by looking at either debug records (for
  150 -- recent versions of LLVM) or debug-related intrinsic functions (for older
  151 -- versions of LLVM.
  152 stmtDebugDeclares :: [LLVM.Stmt] -> Map Int Location
  153 stmtDebugDeclares [] = Map.empty
  154 stmtDebugDeclares (LLVM.Result _ instr drs md:stmts)
  155   | LLVM.Call _ _
  156       (LLVM.ValSymbol (LLVM.Symbol s))
  157       [ _
  158       , LLVM.Typed
  159         { LLVM.typedValue =
  160           LLVM.ValMd (LLVM.ValMdDebugInfo (LLVM.DebugInfoLocalVariable LLVM.DILocalVariable { LLVM.dilvArg = a }))
  161         }
  162       , _
  163       ] <- instr
  164   , s == "llvm.dbg.declare" || s == "llvm.dbg.value"
  165   , Just (LLVM.ValMdLoc LLVM.DebugLoc { LLVM.dlLine = line, LLVM.dlCol = col }) <- lookup "dbg" md
  166   = Map.insert (fromIntegral a) (Location (fromIntegral line) . Just $ fromIntegral col) $ stmtDebugDeclares stmts
  167   | otherwise
  168   = Map.union (debugRecordDeclares drs) (stmtDebugDeclares stmts)
  169 stmtDebugDeclares (LLVM.Effect instr drs md:stmts)
  170   | LLVM.Call _ _
  171       (LLVM.ValSymbol (LLVM.Symbol s))
  172       [ _
  173       , LLVM.Typed
  174         { LLVM.typedValue =
  175           LLVM.ValMd (LLVM.ValMdDebugInfo (LLVM.DebugInfoLocalVariable LLVM.DILocalVariable { LLVM.dilvArg = a }))
  176         }
  177       , _
  178       ] <- instr
  179   , s == "llvm.dbg.declare" || s == "llvm.dbg.value"
  180   , Just (LLVM.ValMdLoc LLVM.DebugLoc { LLVM.dlLine = line, LLVM.dlCol = col }) <- lookup "dbg" md
  181   = Map.insert (fromIntegral a) (Location (fromIntegral line) . Just $ fromIntegral col) $ stmtDebugDeclares stmts
  182   | otherwise
  183   = Map.union (debugRecordDeclares drs) (stmtDebugDeclares stmts)
  184 
  185 -- | Infer locations of declarations by looking at @#dbg_declare@ and
  186 -- @#dbg_value@ debug records.
  187 debugRecordDeclares :: [LLVM.DebugRecord] -> Map Int Location
  188 debugRecordDeclares [] = Map.empty
  189 debugRecordDeclares
  190   (LLVM.DebugRecordDeclare
  191     (LLVM.DbgRecDeclare
  192       { LLVM.drdLocation =
  193           LLVM.ValMdLoc
  194             LLVM.DebugLoc { LLVM.dlLine = line, LLVM.dlCol = col }
  195       , LLVM.drdLocalVariable =
  196           LLVM.ValMdDebugInfo
  197             (LLVM.DebugInfoLocalVariable
  198               LLVM.DILocalVariable { LLVM.dilvArg = a })
  199       }):drs) =
  200   Map.insert
  201     (fromIntegral a)
  202     (Location (fromIntegral line) (Just (fromIntegral col)))
  203     (debugRecordDeclares drs)
  204 debugRecordDeclares
  205   (LLVM.DebugRecordValue
  206     (LLVM.DbgRecValue
  207       { LLVM.drvLocation =
  208           LLVM.ValMdLoc
  209             LLVM.DebugLoc { LLVM.dlLine = line, LLVM.dlCol = col }
  210       , LLVM.drvLocalVariable =
  211           LLVM.ValMdDebugInfo
  212             (LLVM.DebugInfoLocalVariable
  213               LLVM.DILocalVariable { LLVM.dilvArg = a })
  214       }):drs) =
  215   Map.insert
  216     (fromIntegral a)
  217     (Location (fromIntegral line) (Just (fromIntegral col)))
  218     (debugRecordDeclares drs)
  219 debugRecordDeclares (_:drs) = debugRecordDeclares drs

  221 defineName :: LLVM.Define -> Text
  222 defineName LLVM.Define { LLVM.defName = LLVM.Symbol s } = Text.pack s
  223 
  224 parseDefine :: Map String Int -> LLVM.Module -> LLVM.Define -> IO FunctionSkeleton
  225 parseDefine _ _ LLVM.Define { LLVM.defVarArgs = True } =
  226   fail "Skeleton generation does not support varargs"
  227 parseDefine ls m d@LLVM.Define
  228   { LLVM.defName = LLVM.Symbol s
  229   , LLVM.defArgs = args
  230   , LLVM.defBody = body
  231   , LLVM.defRetType = ret
  232   } = do
  233   let stmts = mconcat $ LLVM.bbStmts <$> body
  234   let argNames = LLVM.debugInfoArgNames m d
  235   let debugDeclares = stmtDebugDeclares stmts
  236   argSkels <- zipWithM parseArg args $ (flip IntMap.lookup argNames &&& flip Map.lookup debugDeclares) <$> [1, 2..]
  237   retTy <- parseType ret
  238   pure FunctionSkeleton
  239     { _funSkelName = Text.pack s
  240     , _funSkelLoc = flip Location Nothing <$> Map.lookup s ls
  241     , _funSkelArgs = argSkels
  242     , _funSkelRet = retTy
  243     , _funSkelCalls = Set.intersection
  244       (Set.fromList $ defineName <$> LLVM.modDefines m)
  245       (stmtCalls stmts)
  246     }
  247 
  248 moduleSkeleton :: LLVM.Module -> IO ModuleSkeleton
  249 moduleSkeleton ast = do
  250   globs <- mapM (parseGlobal $ LLVM.debugInfoGlobalLines ast) $ LLVM.modGlobals ast
  251   funs <- mapM (parseDefine (LLVM.debugInfoDefineLines ast) ast) $ LLVM.modDefines ast
  252   pure $ ModuleSkeleton
  253     { _modSkelGlobals = Map.fromList $ (\g -> (g ^. globSkelName, g)) <$> globs
  254     , _modSkelFunctions = Map.fromList $ (\f -> (f ^. funSkelName, f)) <$> funs
  255     }
