   64 -- | The type of state kept in the 'CrucibleSetup' monad

   81 makeCrucibleSetupState ::
   82   MS.ResolvedState ext ->
   83   MS.CrucibleContext ext ->
   84   MS.CrucibleMethodSpecIR ext ->
   85   CrucibleSetupState ext
   86 makeCrucibleSetupState rs cc mspec =
   87   CrucibleSetupState
   88     { _csVarCounter      = MS.AllocIndex 0
   89     , _csPrePost         = MS.PreState
   90     , _csResolvedState   = rs
   91     , _csMethodSpec      = mspec
   92     , _csCrucibleContext = cc
   93     }

  104 currentState :: Lens' (CrucibleSetupState ext) (MS.StateSpec ext)
  105 currentState f x = case x^. csPrePost of
  106   MS.PreState  -> csMethodSpec (MS.csPreState f) x
  107   MS.PostState -> csMethodSpec (MS.csPostState f) x
  108 
  109 addPointsTo :: Monad m => MS.PointsTo ext -> CrucibleSetupT ext m ()
  110 addPointsTo pt = currentState . MS.csPointsTos %= (pt : )
  111 
  112 addAllocGlobal :: Monad m => MS.AllocGlobal ext -> CrucibleSetupT ext m ()
  113 addAllocGlobal ag = csMethodSpec . MS.csGlobalAllocs %= (ag : )
  114 
  115 addCondition :: Monad m => MS.SetupCondition ext -> CrucibleSetupT ext m ()
  116 addCondition cond = currentState . MS.csConditions %= (cond : )

  118 -- | Allocate a fresh variable in the form of a 'TypedVariable' and
  119 -- record this allocation in the setup state.
  120 freshTypedVariable ::
  121   MonadIO m =>
  122   SharedContext {- ^ shared context -} ->
  123   Text          {- ^ variable name  -} ->
  124   Cryptol.Type  {- ^ variable type  -} ->
  125   CrucibleSetupT arch m TypedVariable
  126 freshTypedVariable sc name cty =
  127   do ty <- liftIO $ Cryptol.importType sc Cryptol.emptyEnv cty
  128      vn <- liftIO $ scFreshVarName sc name
  129      let tt = TypedVariable cty vn ty
  130      currentState . MS.csFreshVars %= cons tt
  131      return tt
  132 
  133 -- | Allocate a fresh variable in the form of a 'TypedTerm' and record
  134 -- this allocation in the setup state.
  135 freshVariable ::
  136   MonadIO m =>
  137   SharedContext {- ^ shared context -} ->
  138   Text          {- ^ variable name  -} ->
  139   Cryptol.Type  {- ^ variable type  -} ->
  140   CrucibleSetupT arch m TypedTerm
  141 freshVariable sc name cty =
  142   do tec <- freshTypedVariable sc name cty
  143      liftIO $ typedTermOfVariable sc tec

  145 setupWithTag :: Text -> CrucibleSetupT arch m a -> CrucibleSetupT arch m a
  146 setupWithTag tag m =
  147   withReaderT (croTags %~ Set.insert (Text.unpack tag)) m
