   56 module_skeleton ::
   57   Some LLVMModule ->
   58   TopLevel ModuleSkeleton
   59 module_skeleton m = do
   60   let ast = viewSome modAST m
   61   liftIO $ moduleSkeleton ast
   62 
   63 function_skeleton ::
   64   ModuleSkeleton ->
   65   Text ->
   66   TopLevel FunctionSkeleton
   67 function_skeleton mskel nm =
   68   case mskel ^. modSkelFunctions . at nm of
   69     Just fskel -> pure fskel
   70     Nothing -> throwTopLevel $ Text.unpack $ "No skeleton exists for function \"" <> nm <> "\""
   71 
   72 -- | Manually add a new guess for the size of the argument at the given index
   73 skeleton_resize_arg_index ::
   74   FunctionSkeleton ->
   75   Int ->
   76   Int ->
   77   Bool ->
   78   TopLevel FunctionSkeleton
   79 skeleton_resize_arg_index skel idx sz initialized =
   80   pure (skel & funSkelArgs . ix idx . argSkelType . typeSkelSizeGuesses %~ (guess:))
   81   where
   82     guess :: SizeGuess
   83     guess = SizeGuess
   84       { _sizeGuessElems = sz
   85       , _sizeGuessInitialized = initialized
   86       , _sizeGuessSource = "user guess"
   87       }
   88 
   89 skelArgIndex ::
   90   FunctionSkeleton ->
   91   Text ->
   92   Maybe Int
   93 skelArgIndex skel nm = flip findIndex (skel ^. funSkelArgs) $ \arg ->
   94   arg ^. argSkelName == Just nm
   95 
   96 -- | Manually add a new guess for the size of the argument with the given name
   97 skeleton_resize_arg ::
   98   FunctionSkeleton ->
   99   Text ->
  100   Int ->
  101   Bool ->
  102   TopLevel FunctionSkeleton
  103 skeleton_resize_arg skel nm sz initialized
  104   | Just idx <- skelArgIndex skel nm
  105   = skeleton_resize_arg_index skel idx sz initialized
  106   | otherwise = throwTopLevel $ Text.unpack $
  107       "No argument named \"" <> nm <>
  108       "\" (enabling debug symbols when compiling might help)"

  110 llvmTypeSize :: C.LLVM.TypeContext -> LLVM.Type -> Int
  111 llvmTypeSize tc t =
  112   let ?lc = tc in
  113     case C.LLVM.liftMemType t of
  114       Left _ -> 1
  115       Right m -> fromIntegral $ C.LLVM.memTypeSize (C.LLVM.llvmDataLayout ?lc) m

  117 skeleton_guess_arg_sizes ::
  118   ModuleSkeleton ->
  119   Some LLVMModule ->
  120   [(Text, [FunctionProfile])] ->
  121   TopLevel ModuleSkeleton
  122 skeleton_guess_arg_sizes mskel (Some m) profiles = do
  123   let (_, tc) = C.LLVM.typeContextFromModule $ modAST m
  124   fskels <- forM (mskel ^. modSkelFunctions) $ \skel -> do
  125     case lookup (skel ^. funSkelName) profiles of
  126       Just (prof:_) -> let
  127         updateArg (a, p)
  128           | a ^. argSkelType . typeSkelIsPointer
  129           , Just s <- p ^. argProfileSize
  130           = a & argSkelType . typeSkelSizeGuesses
  131             %~ (SizeGuess
  132                  (quot s $ llvmTypeSize tc $ a ^. argSkelType . typeSkelLLVMType)
  133                  (p ^. argProfileInitialized)
  134                  "checking sizes in the simulator":)
  135           | otherwise = a
  136         uargs args = updateArg <$> zip args (prof ^. funProfileArgs)
  137         in pure (skel & funSkelArgs %~ uargs)
  138       _ -> pure skel
  139   pure $ mskel & modSkelFunctions .~ fskels

  144 skeleton_globals_pre ::
  145   ModuleSkeleton ->
  146   LLVMCrucibleSetupM ()
  147 skeleton_globals_pre mskel =
  148   forM_ (mskel ^. modSkelGlobals) $ \gskel ->
  149     when (gskel ^. globSkelMutable) $ do
  150       let gname = gskel ^. globSkelName
  151       llvm_alloc_global gname
  152       when (gskel ^. globSkelInitialized)
  153         . llvm_points_to True (anySetupGlobal gname)
  154         $ anySetupGlobalInitializer gname
  155 
  156 skeleton_globals_post ::
  157   ModuleSkeleton ->
  158   LLVMCrucibleSetupM ()
  159 skeleton_globals_post mskel =
  160   forM_ (mskel ^. modSkelGlobals) $ \gskel -> do
  161     when (gskel ^. globSkelMutable && gskel ^. globSkelInitialized) $ do
  162       let gname = gskel ^. globSkelName
  163       llvm_points_to True (anySetupGlobal gname)
  164         $ anySetupGlobalInitializer gname

  166 buildArg ::
  167   ArgSkeleton ->
  168   Int ->
  169   LLVMCrucibleSetupM (Maybe TypedTerm, Maybe (AllLLVM SetupValue), Maybe Text)
  170 buildArg arg idx
  171   | arg ^. argSkelType . typeSkelIsPointer
  172   = let
  173       pt = arg ^. argSkelType . typeSkelLLVMType
  174       (t, initialized) = case arg ^. argSkelType . typeSkelSizeGuesses of
  175         (s:_)
  176           | s ^. sizeGuessElems > 1
  177             -> (LLVM.Array (fromIntegral $ s ^. sizeGuessElems) pt, s ^. sizeGuessInitialized)
  178           | otherwise -> (pt, s ^. sizeGuessInitialized)
  179         _ -> (pt, False)
  180     in do
  181       ptr <- llvm_alloc t
  182       mval <- if initialized
  183         then do
  184         val <- llvm_fresh_var ident t
  185         llvm_points_to True ptr (anySetupTerm val)
  186         pure $ Just val
  187         else pure Nothing
  188       pure (mval, Just ptr, arg ^. argSkelName)
  189   | otherwise
  190   = do
  191       val <- llvm_fresh_var ident
  192         $ arg ^. argSkelType . typeSkelLLVMType
  193       pure (Just val, Nothing, arg ^. argSkelName)
  194   where
  195     ident = maybe ("arg" <> Text.pack (show idx)) id $ arg ^. argSkelName
  196 
  197 skeleton_prestate ::
  198   FunctionSkeleton ->
  199   LLVMCrucibleSetupM SkeletonState
  200 skeleton_prestate skel = do
  201   _skelArgs <- mapM (uncurry buildArg) $ zip (skel ^. funSkelArgs) [1,2..]
  202   pure $ SkeletonState{..}
  203 
  204 skeleton_exec ::
  205   SkeletonState ->
  206   LLVMCrucibleSetupM ()
  207 skeleton_exec prestate = do
  208   args <- forM (prestate ^. skelArgs) $ \(mval, mptr, _) ->
  209     case (mval, mptr) of
  210       (_, Just ptr) -> pure ptr
  211       (Just val, Nothing) -> pure $ anySetupTerm val
  212       (Nothing, Nothing) -> throwLLVMFun "skeleton_exec" "Invalid pointer-pointee combination on skeleton argument"
  213   llvm_execute_func args
  214 
  215 rebuildArg ::
  216   (ArgSkeleton, (Maybe TypedTerm, Maybe (AllLLVM SetupValue), Maybe Text))  ->
  217   Int ->
  218   LLVMCrucibleSetupM (Maybe TypedTerm, Maybe (AllLLVM SetupValue), Maybe Text)
  219 rebuildArg (arg, prearg) idx
  220   | arg ^. argSkelType . typeSkelIsPointer
  221   , (_, Just ptr, nm) <- prearg
  222   = let
  223       pt = arg ^. argSkelType . typeSkelLLVMType
  224       t = case arg ^. argSkelType . typeSkelSizeGuesses of
  225         (s:_)
  226           | s ^. sizeGuessElems > 1
  227             -> LLVM.Array (fromIntegral $ s ^. sizeGuessElems) pt
  228           | otherwise -> pt
  229         _ -> pt
  230       ident = maybe ("arg" <> Text.pack (show idx)) id nm
  231     in do
  232       val' <- llvm_fresh_var ident t
  233       llvm_points_to True ptr $ anySetupTerm val'
  234       pure (Just val', Just ptr, nm)
  235   | otherwise = pure prearg
  236 
  237 skeleton_poststate ::
  238   FunctionSkeleton ->
  239   SkeletonState ->
  240   LLVMCrucibleSetupM SkeletonState
  241 skeleton_poststate skel prestate = do
  242   _skelArgs <- zipWithM rebuildArg
  243     (zip (skel ^. funSkelArgs) (prestate ^. skelArgs))
  244     [1,2..]
  245   case skel ^. funSkelRet . typeSkelLLVMType of
  246     LLVM.PrimType LLVM.Void -> pure ()
  247     t -> do
  248       ret <- llvm_fresh_var ("return value of " <> (skel ^. funSkelName)) t
  249       llvm_return $ anySetupTerm ret
  250   pure $ SkeletonState{..}

  252 skeleton_arg_index ::
  253   SkeletonState ->
  254   Int ->
  255   LLVMCrucibleSetupM TypedTerm
  256 skeleton_arg_index state idx
  257   | idx < length (state ^. skelArgs)
  258   , (Just t, _, _) <- (state ^. skelArgs) !! idx
  259   = pure t
  260   | otherwise = throwLLVMFun "skeleton_arg_index" $ mconcat
  261     [ "No initialized argument at index "
  262     , show idx
  263     ]
  264 
  265 stateArgIndex ::
  266   SkeletonState ->
  267   Text ->
  268   Maybe Int
  269 stateArgIndex state nm = flip findIndex (state ^. skelArgs) $ \(_, _, mnm) ->
  270   mnm == Just nm
  271 
  272 skeleton_arg ::
  273   SkeletonState ->
  274   Text ->
  275   LLVMCrucibleSetupM TypedTerm
  276 skeleton_arg state nm
  277   | Just idx <- stateArgIndex state nm
  278   = skeleton_arg_index state idx
  279   | otherwise = throwLLVMFun "skeleton_arg" $ Text.unpack $
  280       "No initialized argument named \"" <> nm <>
  281       "\" (enabling debug symbols when compiling might help)"

  283 skeleton_arg_index_pointer ::
  284   SkeletonState ->
  285   Int ->
  286   LLVMCrucibleSetupM (AllLLVM SetupValue)
  287 skeleton_arg_index_pointer state idx
  288   | idx < length (state ^. skelArgs)
  289   , (_, mp, _) <- (state ^. skelArgs) !! idx
  290   = case mp of
  291       Just p -> pure p
  292       Nothing -> throwLLVMFun "skeleton_arg_index_pointer" $ mconcat
  293         [ "Argument at index "
  294         , show idx
  295         , " is not a pointer or array"
  296         ]
  297   | otherwise = throwLLVMFun "skeleton_arg_index_pointer" $ mconcat
  298     [ "No argument at index "
  299     , show idx
  300     ]
  301 
  302 skeleton_arg_pointer ::
  303   SkeletonState ->
  304   Text ->
  305   LLVMCrucibleSetupM (AllLLVM SetupValue)
  306 skeleton_arg_pointer state nm
  307   | Just idx <- stateArgIndex state nm
  308   = skeleton_arg_index_pointer state idx
  309   | otherwise = do
  310       throwLLVMFun "skeleton_arg_pointer" $ Text.unpack $
  311         "No argument named \"" <> nm <>
  312         "\" (enabling debug symbols when compiling might help)"
