    6 -- | An 'IntervalSet' represents a set of disjoint contiguous
    7 -- intervals of type 'Int'.
    8 -- Each interval @(i, j)@ with @i <= j@ represents the set of values
    9 -- @[i..j]@.
   10 newtype IntervalSet = IntervalSet (IntMap Int)
   11   deriving Eq
   12 
   13 instance Show IntervalSet where
   14   show (IntervalSet m) = show m
   15 
   16 -- | The empty set.
   17 empty :: IntervalSet
   18 empty = IntervalSet IntMap.empty
   19 
   20 -- | A set containing a single interval @[i..j]@.
   21 singleton :: (Int, Int) -> IntervalSet
   22 singleton (i, j)
   23   | i <= j = IntervalSet (IntMap.singleton i j)
   24   | otherwise = empty
   25 
   26 -- | Add an interval @[i..j]@ to the set.
   27 insert :: (Int, Int) -> IntervalSet -> IntervalSet
   28 insert (i, j) (IntervalSet m)
   29   | i > j = IntervalSet m
   30   | otherwise =
   31     let -- m1 contains all intervals strictly below i
   32         -- also possibly one interval starting at i'
   33         m1 = IntMap.takeWhileAntitone (< i) m
   34         -- m2 contains all intervals strictly above j
   35         m2 = IntMap.dropWhileAntitone (<= j+1) m
   36         i' = case IntMap.lookupLE i m of
   37                Nothing -> i
   38                Just (u, v) -> if i <= v+1 then u else i
   39         j' = case IntMap.lookupLE (j+1) m of
   40                Nothing -> j
   41                Just (_, v) -> max v j
   42     in IntervalSet (IntMap.insert i' j' (IntMap.union m1 m2))
   43     -- Any interval [u..v] with v+1 < i stays unchanged.
   44     -- Any interval [u..v] with u <= i <= v+1 is overwritten.
   45     -- Any interval [u..v] with i < u <= j is removed.
   46     -- Any interval [u..v] with j < u stays unchanged.
   47     -- If i falls inside (or is adjacent to the upper bound of) an
   48     -- interval [u..v], set i' = u; else i' = i.
   49     -- If j falls inside (or is adjacent to the lower bound of) an
   50     -- interval [u..v], let j' = v; else j' = j.
   51     -- Always insert a new interval [i'..j'].
   52 
   53 -- | Remove an interval @[i..j]@ from the set.
   54 delete :: (Int, Int) -> IntervalSet -> IntervalSet
   55 delete (i, j) (IntervalSet m)
   56   | i > j = IntervalSet m
   57   | otherwise =
   58     let m1 = IntMap.takeWhileAntitone (< i) m
   59         m2 = IntMap.dropWhileAntitone (<= j) m
   60         m1' = case IntMap.lookupLT i m1 of
   61                 Nothing -> m1
   62                 Just (u, v) -> if i <= v then IntMap.insert u (i-1) m1 else m1
   63         m2' = case IntMap.lookupLE j m of
   64                 Nothing -> m2
   65                 Just (_, v) -> if j < v then IntMap.insert (j+1) v m2 else m2
   66     in IntervalSet (IntMap.union m1' m2')
   67     -- Any interval [u..v] with v < i stays unchanged.
   68     -- Any interval [u..v] with u < i <= v is overwritten.
   69     -- Any interval [u..v] with i <= u <= j is removed.
   70     -- Any interval [u..v] with j < u stays unchanged.
   71     -- If i falls inside an interval [u..v], replace it with interval [u..i-1].
   72     -- If j falls inside an interval [u..v], insert a new interval [j+1..v].
   73 
   74 -- | Is a specific value in the set?
   75 member :: Int -> IntervalSet -> Bool
   76 member x (IntervalSet m) =
   77   case IntMap.lookupLE x m of
   78     Nothing -> False
   79     Just (_, j) -> x <= j
   80 
   81 -- | The union of two sets.
   82 union :: IntervalSet -> IntervalSet -> IntervalSet
   83 union s1 s2 = foldr insert s1 (toList s2)
   84 
   85 -- | The difference of two sets: Remove all members of the second set
   86 -- from the first set.
   87 difference :: IntervalSet -> IntervalSet -> IntervalSet
   88 difference s1 s2 = foldr delete s1 (toList s2)
   89 
   90 -- | Create a set from a list of intervals.
   91 fromList :: [(Int, Int)] -> IntervalSet
   92 fromList = foldr insert empty
   93 
   94 -- | Convert the set to a list of disjoint intervals in ascending
   95 -- order.
   96 toList :: IntervalSet -> [(Int, Int)]
   97 toList (IntervalSet m) = IntMap.toList m
