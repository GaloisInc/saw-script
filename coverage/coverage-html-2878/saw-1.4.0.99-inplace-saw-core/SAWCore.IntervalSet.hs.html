    6 -- | An 'IntervalSet' represents a set of disjoint contiguous
    7 -- intervals of type 'Int'.
    8 -- Each interval @(i, j)@ with @i < j@ represents the set of values
    9 -- @[i..j-1]@.
   10 newtype IntervalSet = IntervalSet (IntMap Int)
   11 
   12 instance Show IntervalSet where
   13   show (IntervalSet m) = show m
   14 
   15 -- | The empty set.
   16 empty :: IntervalSet
   17 empty = IntervalSet IntMap.empty
   18 
   19 -- | A set containing a single half-open interval @[i..j-1]@.
   20 singleton :: (Int, Int) -> IntervalSet
   21 singleton (i, j)
   22   | i < j = IntervalSet (IntMap.singleton i j)
   23   | otherwise = empty
   24 
   25 -- | Add an interval @[i..j-1]@ to the set.
   26 insert :: (Int, Int) -> IntervalSet -> IntervalSet
   27 insert (i, j) (IntervalSet m)
   28   | i >= j = IntervalSet m
   29   | otherwise =
   30     let -- m1 contains all intervals strictly below i
   31         -- also possibly one interval starting at i'
   32         m1 = IntMap.takeWhileAntitone (< i) m
   33         -- m2 contains all intervals strictly above j
   34         m2 = IntMap.dropWhileAntitone (<= j) m
   35         i' = case IntMap.lookupLE i m of
   36                Nothing -> i
   37                Just (u, v) -> if i <= v then u else i
   38         j' = case IntMap.lookupLE j m of
   39                Nothing -> j
   40                Just (_, v) -> max v j
   41     in IntervalSet (IntMap.insert i' j' (IntMap.union m1 m2))
   42     -- Any interval [u, v) with v < i stays unchanged.
   43     -- Any interval [u, v) with u <= i <= v is overwritten.
   44     -- Any interval [u, v) with i < u <= j is removed.
   45     -- Any interval [u, v) with j < u stays unchanged.
   46     -- If i falls inside (or coincides with the upper bound of) an
   47     -- interval [u, v), set i' = u; else i' = i.
   48     -- If j falls inside an interval [u, v), let j' = v; else j' = j.
   49     -- Always insert a new interval [i', j').
   50 
   51 -- | Remove an interval @[i..j-1]@ from the set.
   52 delete :: (Int, Int) -> IntervalSet -> IntervalSet
   53 delete (i, j) (IntervalSet m)
   54   | i >= j = IntervalSet m
   55   | otherwise =
   56     let m1 = IntMap.takeWhileAntitone (< i) m
   57         m2 = IntMap.dropWhileAntitone (<= j) m
   58         m1' = case IntMap.lookupLT i m1 of
   59                 Nothing -> m1
   60                 Just (u, v) -> if i < v then IntMap.insert u i m1 else m1
   61         m2' = case IntMap.lookupLE j m of
   62                 Nothing -> m2
   63                 Just (_, v) -> if j < v then IntMap.insert j v m2 else m2
   64     in IntervalSet (IntMap.union m1' m2')
   65     -- Any interval [u, v) with v <= i stays unchanged.
   66     -- Any interval [u, v) with u < i < v is overwritten.
   67     -- Any interval [u, v) with i <= u <= j is removed.
   68     -- Any interval [u, v) with j < u stays unchanged.
   69     -- If i falls inside an interval [u, v), replace it with interval [u, i).
   70     -- If j falls inside an interval [u, v), insert a new interval [j, v).
   71 
   72 -- | Is a specific value in the set?
   73 member :: Int -> IntervalSet -> Bool
   74 member x (IntervalSet m) =
   75   case IntMap.lookupLE x m of
   76     Nothing -> False
   77     Just (_, j) -> x < j
   78 
   79 -- | The union of two sets.
   80 union :: IntervalSet -> IntervalSet -> IntervalSet
   81 union s1 s2 = foldr insert s1 (toList s2)
   82 
   83 -- | The difference of two sets: Remove all members of the second set
   84 -- from the first set.
   85 difference :: IntervalSet -> IntervalSet -> IntervalSet
   86 difference s1 s2 = foldr delete s1 (toList s2)
   87 
   88 -- | Create a set from a list of half-open intervals.
   89 fromList :: [(Int, Int)] -> IntervalSet
   90 fromList = foldr insert empty
   91 
   92 -- | Convert the set to a list of disjoint half-open intervals in
   93 -- ascending order.
   94 toList :: IntervalSet -> [(Int, Int)]
   95 toList (IntervalSet m) = IntMap.toList m
