  135 -- | Try to translate the spec\'s 'SetupValue' into an 'LLVMVal', pretty-print
  136 --   the 'LLVMVal'.
  137 mkStructuralMismatch ::
  138   Options              {- ^ output/verbosity options -} ->
  139   JVMCrucibleContext ->
  140   SharedContext {- ^ context for constructing SAW terms -} ->
  141   CrucibleMethodSpecIR {- ^ for name and typing environments -} ->
  142   JVMVal {- ^ the value from the simulator -} ->
  143   SetupValue {- ^ the value from the spec -} ->
  144   J.Type     {- ^ the expected type -} ->
  145   OverrideMatcher CJ.JVM w (OverrideFailureReason CJ.JVM)
  146 mkStructuralMismatch opts cc sc spec jvmval setupval jty = do
  147   setupTy <- typeOfSetupValueJVM cc spec setupval
  148   setupJVal <- resolveSetupValueJVM opts cc sc spec setupval
  149   pure $ StructuralMismatch
  150             (ppJVMVal jvmval)
  151             (ppJVMVal setupJVal)
  152             (Just setupTy)
  153             jty

  158 -- | This function is responsible for implementing the \"override\" behavior
  159 --   of method specifications.  The main work done in this function to manage
  160 --   the process of selecting between several possible different override
  161 --   specifications that could apply.  We want a proof to succeed if /any/
  162 --   choice of method spec allows the proof to go through, which is a slightly
  163 --   awkward thing to fit into the symbolic simulation framework.
  164 --
  165 --   The main work of determining the preconditions, postconditions, memory
  166 --   updates and return value for a single specification is done by
  167 --   the @methodSpecHandler_prestate@ and @methodSpecHandler_poststate@ functions.
  168 --
  169 --   In a first phase, we attempt to apply the precondition portion of each of
  170 --   the given method specifications.  Each of them that might apply generate
  171 --   a substitution for the setup variables and a collection of preconditions
  172 --   that guard the specification.  We use these preconditions to compute
  173 --   a multiway symbolic branch, one for each override which might apply.
  174 --
  175 --   In the body of each of the individual branches, we compute the postcondition
  176 --   actions of the corresponding method specification.  This will update memory
  177 --   and compute function return values, in addition to assuming postcondition
  178 --   predicates.
  179 methodSpecHandler ::
  180   forall rtp args ret.
  181   Options                  {- ^ output/verbosity options                     -} ->
  182   SharedContext            {- ^ context for constructing SAW terms           -} ->
  183   JVMCrucibleContext          {- ^ context for interacting with Crucible        -} ->
  184   W4.ProgramLoc            {- ^ Location of the call site for error reporting-} ->
  185   IORef MetadataMap {- ^ metadata map -} ->
  186   NonEmpty CrucibleMethodSpecIR {- ^ specification for current function override  -} ->
  187   Crucible.FnHandle args ret {- ^ a handle for the function -} ->
  188   Crucible.OverrideSim (SAWCruciblePersonality Sym) Sym CJ.JVM rtp args ret
  189      (Crucible.RegValue Sym ret)
  190 methodSpecHandler opts sc cc top_loc _mdMap css h =
  191   -- TODO, special case for single-override situations,
  192   --  and use the mdMap to keep track of obligations arising
  193   --  from override preconditions.
  194 
  195   jccWithBackend cc $ \bak -> do
  196   let sym = backendGetSym bak
  197   Crucible.RegMap args <- Crucible.getOverrideArgs
  198 
  199   -- First, run the precondition matcher phase.  Collect together a list of the results.
  200   -- For each override, this will either be an error message, or a matcher state and
  201   -- a method spec.
  202   prestates <-
  203     do g0 <- Crucible.readGlobals
  204        forM css $ \cs -> liftIO $
  205          let initialFree =
  206                Set.fromList (cs ^.. MS.csPreState. MS.csFreshVars . each . to tecExt . to ecVarIndex)
  207           in runOverrideMatcher sym g0 Map.empty Map.empty initialFree (view MS.csLoc cs)
  208                       (do methodSpecHandler_prestate opts sc cc args cs
  209                           return cs)
  210 
  211   -- Print a failure message if all overrides failed to match.  Otherwise, collect
  212   -- all the override states that might apply, and compute the conjunction of all
  213   -- the preconditions.  We'll use these to perform symbolic branches between the
  214   -- various overrides.
  215   branches <- case partitionEithers (NE.toList prestates) of
  216                 (e, []) ->
  217                   fail $ show $
  218                   PP.vcat
  219                   [ "All overrides failed during structural matching:"
  220                   , PP.vcat (map (\x -> "*" <> PP.indent 2 (ppOverrideFailure x)) e)
  221                   ]
  222                 (_, ss) -> liftIO $
  223                   forM ss $ \(cs,st) ->
  224                     do precond <- W4.andAllOf sym (folded._2.labeledPred) (st^.osAsserts)
  225                        return ( precond, cs, st )
  226 
  227   -- Now use crucible's symbolic branching machinery to select between the branches.
  228   -- Essentially, we are doing an n-way if statement on the precondition predicates
  229   -- for each override, and selecting the first one whose preconditions hold.
  230   --
  231   -- Then, in the body of the branch, we run the poststate handler to update the
  232   -- memory state, compute return values and compute postcondition predicates.
  233   --
  234   -- For each override branch that doesn't fail outright, we assume the relevant
  235   -- postconditions, update the crucible global variable state, and return the
  236   -- computed return value.
  237   --
  238   -- We add a final default branch that simply fails unless some previous override
  239   -- branch has already succeeded.
  240   let retTy = Crucible.handleReturnType h
  241   Crucible.regValue <$> Crucible.callOverride h
  242      (Crucible.mkOverride' "overrideBranches" retTy
  243        (Crucible.symbolicBranches Crucible.emptyRegMap $
  244          [ ( precond
  245            , do g <- Crucible.readGlobals
  246                 res <- liftIO $ runOverrideMatcher sym g
  247                    (st^.setupValueSub)
  248                    (st^.termSub)
  249                    (st^.osFree)
  250                    (st^.osLocation)
  251                    (methodSpecHandler_poststate opts sc cc retTy cs)
  252                 case res of
  253                   Left (OF loc rsn)  ->
  254                     -- TODO, better pretty printing for reasons
  255                     liftIO
  256                       $ Crucible.abortExecBecause
  257                       $ Crucible.AssertionFailure
  258                       $ Crucible.SimError loc
  259                       $ Crucible.AssertFailureSimError "assumed false" (show rsn)
  260                   Right (ret,st') ->
  261                     do liftIO $ forM_ (st'^.osAssumes) $ \(_md,asum) ->
  262                          Crucible.addAssumption bak
  263                           $ Crucible.GenericAssumption (st^.osLocation) "override postcondition" asum
  264                        Crucible.writeGlobals (st'^.overrideGlobals)
  265                        Crucible.overrideReturn' (Crucible.RegEntry retTy ret)
  266            , Just (W4.plSourceLoc (cs ^. MS.csLoc))
  267            )
  268          | (precond, cs, st) <- branches
  269          ] ++
  270         [
  271 
  272             let fnName = case branches of
  273                            (_, cs, _) : _  -> cs ^. MS.csMethod . jvmMethodName
  274                            _               -> "unknown function"
  275             in
  276             ( W4.truePred sym
  277             , liftIO $ Crucible.addFailedAssertion bak (Crucible.GenericSimError $ "no override specification applies for " ++ fnName)
  278             , Just (W4.plSourceLoc top_loc)
  279             )
  280         ]
  281        ))
  282      (Crucible.RegMap args)

  286 -- | Use a method spec to override the behavior of a function.
  287 --   This function computes the pre-state portion of the override,
  288 --   which involves reading values from arguments and memory and computing
  289 --   substitutions for the setup value variables, and computing precondition
  290 --   predicates.
  291 methodSpecHandler_prestate ::
  292   forall ctx w.
  293   Options                  {- ^ output/verbosity options                     -} ->
  294   SharedContext            {- ^ context for constructing SAW terms           -} ->
  295   JVMCrucibleContext          {- ^ context for interacting with Crucible        -} ->
  296   Ctx.Assignment (Crucible.RegEntry Sym) ctx
  297                            {- ^ the arguments to the function -} ->
  298   CrucibleMethodSpecIR     {- ^ specification for current function override  -} ->
  299   OverrideMatcher CJ.JVM w ()
  300 methodSpecHandler_prestate opts sc cc args cs =
  301   do let expectedArgTypes = Map.elems (cs ^. MS.csArgBindings)
  302      let aux ::
  303            (J.Type, SetupValue) -> Crucible.AnyValue Sym ->
  304            IO (JVMVal, J.Type, SetupValue)
  305          aux (argTy, setupVal) val =
  306            case decodeJVMVal argTy val of
  307              Just val' -> return (val', argTy, setupVal)
  308              Nothing -> fail "unexpected type"
  309 
  310      -- todo: fail if list lengths mismatch
  311      xs <- liftIO (zipWithM aux expectedArgTypes (assignmentToList args))
  312 
  313      let md = MS.ConditionMetadata
  314               { MS.conditionLoc = cs ^. MS.csLoc
  315               , MS.conditionTags = mempty
  316               , MS.conditionType = "formal argument matching"
  317               , MS.conditionContext = ""
  318               }
  319 
  320      sequence_ [ matchArg opts sc cc cs PreState md x y z | (x, y, z) <- xs]
  321 
  322      learnCond opts sc cc cs PreState (cs ^. MS.csPreState)
  323

  325 -- | Use a method spec to override the behavior of a function.
  326 --   This function computes the post-state portion of the override,
  327 --   which involves writing values into memory, computing the return value,
  328 --   and computing postcondition predicates.
  329 methodSpecHandler_poststate ::
  330   forall ret.
  331   Options                  {- ^ output/verbosity options                     -} ->
  332   SharedContext            {- ^ context for constructing SAW terms           -} ->
  333   JVMCrucibleContext          {- ^ context for interacting with Crucible        -} ->
  334   Crucible.TypeRepr ret    {- ^ type representation of function return value -} ->
  335   CrucibleMethodSpecIR     {- ^ specification for current function override  -} ->
  336   OverrideMatcher CJ.JVM RW (Crucible.RegValue Sym ret)
  337 methodSpecHandler_poststate opts sc cc retTy cs =
  338   do executeCond opts sc cc cs (cs ^. MS.csPostState)
  339      computeReturnValue opts cc sc cs retTy (cs ^. MS.csRetValue)

  341 -- learn pre/post condition
  342 learnCond ::
  343   Options ->
  344   SharedContext ->
  345   JVMCrucibleContext ->
  346   CrucibleMethodSpecIR ->
  347   PrePost ->
  348   StateSpec ->
  349   OverrideMatcher CJ.JVM w ()
  350 learnCond opts sc cc cs prepost ss =
  351   do let loc = cs ^. MS.csLoc
  352      matchPointsTos opts sc cc cs prepost (ss ^. MS.csPointsTos)
  353      traverse_ (learnSetupCondition opts sc cc cs prepost) (ss ^. MS.csConditions)
  354      assertTermEqualities sc cc
  355      enforceDisjointness cc loc ss
  356      enforceCompleteSubstitution loc ss
  357 
  358 
  359 assertTermEqualities ::
  360   SharedContext ->
  361   JVMCrucibleContext ->
  362   OverrideMatcher CJ.JVM md ()
  363 assertTermEqualities sc cc = do
  364   let sym = cc ^. jccSym
  365   let assertTermEquality (cond, t, md, e) = do
  366         p <- instantiateExtResolveSAWPred sc cc t
  367         p' <- liftIO $ W4.impliesPred sym cond p
  368         addAssert p' md e
  369   traverse_ assertTermEquality =<< OM (use termEqs)

  372 -- execute a pre/post condition
  373 executeCond ::
  374   Options ->
  375   SharedContext ->
  376   JVMCrucibleContext ->
  377   CrucibleMethodSpecIR ->
  378   StateSpec ->
  379   OverrideMatcher CJ.JVM RW ()
  380 executeCond opts sc cc cs ss =
  381   do refreshTerms sc ss
  382      traverse_ (executeAllocation opts cc) (Map.assocs (ss ^. MS.csAllocs))
  383      traverse_ (executePointsTo opts sc cc cs) (ss ^. MS.csPointsTos)
  384      traverse_ (executeSetupCondition opts sc cc cs) (ss ^. MS.csConditions)

  389 -- | Generate assertions that all of the memory allocations matched by
  390 -- an override's precondition are disjoint.
  391 enforceDisjointness ::
  392   JVMCrucibleContext -> W4.ProgramLoc -> StateSpec -> OverrideMatcher CJ.JVM w ()
  393 enforceDisjointness cc loc ss =
  394   do let sym = cc^.jccSym
  395      sub <- OM (use setupValueSub)
  396      let mems = Map.elems $ Map.intersectionWith (,) (view MS.csAllocs ss) sub
  397 
  398      let md = MS.ConditionMetadata
  399               { MS.conditionLoc = loc
  400               , MS.conditionTags = mempty
  401               , MS.conditionType = "memory region disjointness"
  402               , MS.conditionContext = ""
  403               }
  404      -- Ensure that all regions are disjoint from each other.
  405      sequence_
  406         [ do c <- liftIO $ W4.notPred sym =<< CJ.refIsEqual sym p q
  407              addAssert c md a
  408 
  409         | let a = Crucible.SimError loc $
  410                     Crucible.AssertFailureSimError "Memory regions not disjoint" ""
  411         , ((_ploc, _pty), p) : ps <- tails mems
  412         , ((_qloc, _qty), q)      <- ps
  413         ]

  417 -- | For each points-to statement read the memory value through the
  418 -- given pointer (lhs) and match the value against the given pattern
  419 -- (rhs).  Statements are processed in dependency order: a points-to
  420 -- statement cannot be executed until bindings for any/all lhs
  421 -- variables exist.
  422 matchPointsTos ::
  423   Options          {- ^ saw script print out opts -} ->
  424   SharedContext    {- ^ term construction context -} ->
  425   JVMCrucibleContext  {- ^ simulator context     -}     ->
  426   CrucibleMethodSpecIR                               ->
  427   PrePost                                            ->
  428   [JVMPointsTo]       {- ^ points-tos                -} ->
  429   OverrideMatcher CJ.JVM w ()
  430 matchPointsTos opts sc cc spec prepost = go False []
  431   where
  432     go ::
  433       Bool       {- progress indicator -} ->
  434       [JVMPointsTo] {- delayed conditions -} ->
  435       [JVMPointsTo] {- queued conditions  -} ->
  436       OverrideMatcher CJ.JVM w ()
  437 
  438     -- all conditions processed, success
  439     go _ [] [] = return ()
  440 
  441     -- not all conditions processed, no progress, failure
  442     go False delayed [] = failure (spec ^. MS.csLoc) (AmbiguousPointsTos delayed)
  443 
  444     -- not all conditions processed, progress made, resume delayed conditions
  445     go True delayed [] = go False [] delayed
  446 
  447     -- progress the next points-to in the work queue
  448     go progress delayed (c:cs) =
  449       do ready <- checkPointsTo c
  450          if ready then
  451            do learnPointsTo opts sc cc spec prepost c
  452               go True delayed cs
  453          else
  454            do go progress (c:delayed) cs
  455 
  456     -- determine if a precondition is ready to be checked
  457     checkPointsTo :: JVMPointsTo -> OverrideMatcher CJ.JVM w Bool
  458     checkPointsTo (JVMPointsToField _loc p _ _) = checkAllocIndex p
  459     checkPointsTo (JVMPointsToStatic _loc _ _) = pure True
  460     checkPointsTo (JVMPointsToElem _loc p _ _) = checkAllocIndex p
  461     checkPointsTo (JVMPointsToArray _loc p _) = checkAllocIndex p
  462 
  463     checkAllocIndex :: AllocIndex -> OverrideMatcher CJ.JVM w Bool
  464     checkAllocIndex i =
  465       do m <- OM (use setupValueSub)
  466          return (Map.member i m)

  471 computeReturnValue ::
  472   Options               {- ^ saw script debug and print options     -} ->
  473   JVMCrucibleContext       {- ^ context of the crucible simulation     -} ->
  474   SharedContext         {- ^ context for generating saw terms       -} ->
  475   CrucibleMethodSpecIR  {- ^ method specification                   -} ->
  476   Crucible.TypeRepr ret {- ^ representation of function return type -} ->
  477   Maybe SetupValue      {- ^ optional symbolic return value         -} ->
  478   OverrideMatcher CJ.JVM w (Crucible.RegValue Sym ret)
  479                         {- ^ concrete return value                  -}
  480 
  481 computeReturnValue _opts _cc _sc spec ty Nothing =
  482   case ty of
  483     Crucible.UnitRepr -> return ()
  484     _ -> failure (spec ^. MS.csLoc) (BadReturnSpecification (Some ty))
  485 
  486 computeReturnValue opts cc sc spec ty (Just val) =
  487   do val' <- resolveSetupValueJVM opts cc sc spec val
  488      let fail_ = failure (spec ^. MS.csLoc) (BadReturnSpecification (Some ty))
  489      case val' of
  490        IVal i ->
  491          case testEquality ty CJ.intRepr of
  492            Just Refl -> return i
  493            Nothing -> fail_
  494        LVal l ->
  495          case testEquality ty CJ.longRepr of
  496            Just Refl -> return l
  497            Nothing -> fail_
  498        RVal r ->
  499          case testEquality ty CJ.refRepr of
  500            Just Refl -> return r
  501            Nothing -> fail_

  506 -- | Assign the given pointer value to the given allocation index in
  507 -- the current substitution. If there is already a binding for this
  508 -- index, then add a pointer-equality constraint.
  509 assignVar ::
  510   JVMCrucibleContext {- ^ context for interacting with Crucible -} ->
  511   MS.ConditionMetadata ->
  512   AllocIndex {- ^ variable index -} ->
  513   JVMRefVal  {- ^ concrete value -} ->
  514   OverrideMatcher CJ.JVM w ()
  515 
  516 assignVar cc md var ref =
  517   do old <- OM (setupValueSub . at var <<.= Just ref)
  518      let loc = MS.conditionLoc md
  519      let sym = cc ^. jccSym
  520      for_ old $ \ref' ->
  521        do p <- liftIO (CJ.refIsEqual sym ref ref')
  522           addAssert p md (Crucible.SimError loc (Crucible.AssertFailureSimError "equality of aliased pointers" ""))

  526 -- | Match the value of a function argument with a symbolic 'SetupValue'.
  527 matchArg ::
  528   Options          {- ^ saw script print out opts -} ->
  529   SharedContext      {- ^ context for constructing SAW terms    -} ->
  530   JVMCrucibleContext    {- ^ context for interacting with Crucible -} ->
  531   CrucibleMethodSpecIR {- ^ specification for current function override  -} ->
  532   PrePost                                                          ->
  533   MS.ConditionMetadata ->
  534   JVMVal             {- ^ concrete simulation value             -} ->
  535   J.Type             {- ^ expected memory type                  -} ->
  536   SetupValue         {- ^ expected specification value          -} ->
  537   OverrideMatcher CJ.JVM w ()
  538 
  539 matchArg opts sc cc cs prepost md actual expectedTy expected@(MS.SetupTerm expectedTT)
  540   | TypedTermSchema (Cryptol.Forall [] [] tyexpr) <- ttType expectedTT
  541   , Right tval <- Cryptol.evalType mempty tyexpr
  542   = do sym <- Ov.getSymInterface
  543        failMsg  <- mkStructuralMismatch opts cc sc cs actual expected expectedTy
  544        realTerm <- valueToSC sym md failMsg tval actual
  545        matchTerm sc md prepost realTerm (ttTerm expectedTT)
  546 
  547 matchArg opts sc cc cs prepost md actual@(RVal ref) expectedTy setupval =
  548   case setupval of
  549     MS.SetupVar var ->
  550       do assignVar cc md var ref
  551 
  552     MS.SetupNull () ->
  553       do sym <- Ov.getSymInterface
  554          p   <- liftIO (CJ.refIsNull sym ref)
  555          addAssert p md (Crucible.SimError (cs ^. MS.csLoc) (Crucible.AssertFailureSimError ("null-equality " ++ MS.stateCond prepost) ""))
  556 
  557     MS.SetupGlobal empty _ -> absurd empty
  558     MS.SetupEnum   empty   -> absurd empty
  559     MS.SetupTuple  empty _ -> absurd empty
  560     MS.SetupSlice  empty   -> absurd empty
  561 
  562     _ -> failure (cs ^. MS.csLoc) =<<
  563            mkStructuralMismatch opts cc sc cs actual setupval expectedTy
  564 
  565 matchArg opts sc cc cs _prepost md actual expectedTy expected =
  566   failure (MS.conditionLoc md) =<<
  567     mkStructuralMismatch opts cc sc cs actual expected expectedTy

  571 valueToSC ::
  572   Sym ->
  573   MS.ConditionMetadata ->
  574   OverrideFailureReason CJ.JVM ->
  575   Cryptol.TValue ->
  576   JVMVal ->
  577   OverrideMatcher CJ.JVM w Term
  578 valueToSC sym _ _ Cryptol.TVBit (IVal x) =
  579   do b <- liftIO $ W4.bvIsNonzero sym x
  580       -- TODO: assert that x is 0 or 1
  581      st <- liftIO (sawCoreState sym)
  582      liftIO (toSC sym st b)
  583 
  584 valueToSC sym _ _ (Cryptol.TVSeq 8 Cryptol.TVBit) (IVal x) =
  585   do st <- liftIO (sawCoreState sym)
  586      liftIO (toSC sym st =<< W4.bvTrunc sym (W4.knownNat @8) x)
  587 
  588 valueToSC sym _ _ (Cryptol.TVSeq 16 Cryptol.TVBit) (IVal x) =
  589   do st <- liftIO (sawCoreState sym)
  590      liftIO (toSC sym st =<< W4.bvTrunc sym (W4.knownNat @16) x)
  591 
  592 valueToSC sym _ _ (Cryptol.TVSeq 32 Cryptol.TVBit) (IVal x) =
  593   do st <- liftIO (sawCoreState sym)
  594      liftIO (toSC sym st x)
  595 
  596 valueToSC sym _ _ (Cryptol.TVSeq 64 Cryptol.TVBit) (LVal x) =
  597   do st <- liftIO (sawCoreState sym)
  598      liftIO (toSC sym st x)
  599 
  600 valueToSC _sym md failMsg _tval _val =
  601   failure (MS.conditionLoc md) failMsg

  623 -- | Process a "points_to" statement from the precondition section of
  624 -- the CrucibleSetup block. First, load the value from the address
  625 -- indicated by 'ptr', and then match it against the pattern 'val'.
  626 learnPointsTo ::
  627   Options                    ->
  628   SharedContext              ->
  629   JVMCrucibleContext            ->
  630   CrucibleMethodSpecIR       ->
  631   PrePost                    ->
  632   JVMPointsTo                   ->
  633   OverrideMatcher CJ.JVM w ()
  634 learnPointsTo opts sc cc spec prepost pt =
  635   jccWithBackend cc $ \bak -> do
  636   let sym = backendGetSym bak
  637   let tyenv = MS.csAllocations spec
  638   let nameEnv = MS.csTypeNames spec
  639   let jc = cc ^. jccJVMContext
  640   globals <- OM (use overrideGlobals)
  641   case pt of
  642 
  643     JVMPointsToField md ptr fid (Just val) ->
  644       do ty <- typeOfSetupValue cc tyenv nameEnv val
  645          rval <- resolveAllocIndexJVM ptr
  646          dyn <- liftIO $ CJ.doFieldLoad bak globals rval fid
  647          v <- liftIO $ projectJVMVal bak ty ("field load " ++ J.fieldIdName fid ++ ", " ++ show (MS.conditionLoc md)) dyn
  648          matchArg opts sc cc spec prepost md v ty val
  649 
  650     JVMPointsToStatic md fid (Just val) ->
  651       do ty <- typeOfSetupValue cc tyenv nameEnv val
  652          dyn <- liftIO $ CJ.doStaticFieldLoad bak jc globals fid
  653          v <- liftIO $ projectJVMVal bak ty ("static field load " ++ J.fieldIdName fid ++ ", " ++ show (MS.conditionLoc md)) dyn
  654          matchArg opts sc cc spec prepost md v ty val
  655 
  656     JVMPointsToElem md ptr idx (Just val) ->
  657       do ty <- typeOfSetupValue cc tyenv nameEnv val
  658          rval <- resolveAllocIndexJVM ptr
  659          dyn <- liftIO $ CJ.doArrayLoad bak globals rval idx
  660          v <- liftIO $ projectJVMVal bak ty ("array load " ++ show idx ++ ", " ++ show (MS.conditionLoc md)) dyn
  661          matchArg opts sc cc spec prepost md v ty val
  662 
  663     JVMPointsToArray md ptr (Just tt) ->
  664       do (len, ety) <-
  665            case ttIsMono (ttType tt) of
  666              Nothing -> fail "jvm_array_is: invalid polymorphic value"
  667              Just cty ->
  668                case Cryptol.tIsSeq cty of
  669                  Nothing -> fail "jvm_array_is: expected array type"
  670                  Just (lty, ety) ->
  671                    case Cryptol.tIsNum lty of
  672                      Nothing -> fail "jvm_array_is: expected finite-sized array"
  673                      Just len -> pure (len, ety)
  674          jty <-
  675            case toJVMType (Cryptol.evalValType mempty ety) of
  676              Nothing -> fail "jvm_array_is: invalid element type"
  677              Just jty -> pure jty
  678          rval <- resolveAllocIndexJVM ptr
  679          let tval = Cryptol.evalValType mempty ety
  680          let
  681            load idx =
  682              do dyn <- liftIO $ CJ.doArrayLoad bak globals rval idx
  683                 let msg = "array load " ++ show idx ++ ", " ++ show (MS.conditionLoc md)
  684                 jval <- liftIO $ projectJVMVal bak jty msg dyn
  685                 let failMsg = StructuralMismatch (ppJVMVal jval) mempty (Just jty) jty -- REVISIT
  686                 valueToSC sym md failMsg tval jval
  687 
  688          when (len > toInteger (maxBound :: Int)) $ fail "jvm_array_is: array length too long"
  689          ety_tm <- liftIO $ Cryptol.importType sc Cryptol.emptyEnv ety
  690          ts <- traverse load [0 .. fromInteger len - 1]
  691          realTerm <- liftIO $ scVector sc ety_tm ts
  692          matchTerm sc md prepost realTerm (ttTerm tt)
  693 
  694     -- If the right-hand-side is 'Nothing', this is indicates a "modifies" declaration,
  695     -- which should probably not appear in the pre-state section, and has no effect.
  696     _ -> pure ()
  697

  701 -- section of the CrucibleSetup block.
  702 learnEqual ::
  703   Options                                          ->
  704   SharedContext                                    ->
  705   JVMCrucibleContext                                  ->

  794 -- | Process a "points_to" statement from the postcondition section of
  795 -- the CrucibleSetup block. First we compute the value indicated by
  796 -- 'val', and then write it to the address indicated by 'ptr'.
  797 executePointsTo ::
  798   Options                    ->
  799   SharedContext              ->
  800   JVMCrucibleContext            ->
  801   CrucibleMethodSpecIR       ->
  802   JVMPointsTo                   ->
  803   OverrideMatcher CJ.JVM w ()
  804 executePointsTo opts sc cc spec pt =
  805   jccWithBackend cc $ \bak -> do
  806   let sym = backendGetSym bak
  807   globals <- OM (use overrideGlobals)
  808   let jc = cc ^. jccJVMContext
  809   case pt of
  810 
  811     JVMPointsToField _loc ptr fid val ->
  812       do dyn <- maybe (pure CJ.unassignedJVMValue) (injectSetupValueJVM sym opts cc sc spec) val
  813          rval <- resolveAllocIndexJVM ptr
  814          globals' <- liftIO $ CJ.doFieldStore bak globals rval fid dyn
  815          OM (overrideGlobals .= globals')
  816 
  817     JVMPointsToStatic _loc fid val ->
  818       do dyn <- maybe (pure CJ.unassignedJVMValue) (injectSetupValueJVM sym opts cc sc spec) val
  819          globals' <- liftIO $ CJ.doStaticFieldStore bak jc globals fid dyn
  820          OM (overrideGlobals .= globals')
  821 
  822     JVMPointsToElem _loc ptr idx val ->
  823       do dyn <- maybe (pure CJ.unassignedJVMValue) (injectSetupValueJVM sym opts cc sc spec) val
  824          rval <- resolveAllocIndexJVM ptr
  825          globals' <- liftIO $ CJ.doArrayStore bak globals rval idx dyn
  826          OM (overrideGlobals .= globals')
  827 
  828     JVMPointsToArray _loc ptr (Just tt) ->
  829       do (_ety, tts) <-
  830            liftIO (destVecTypedTerm sc tt) >>=
  831            \case
  832              Nothing -> fail "jvm_array_is: not a monomorphic sequence type"
  833              Just x -> pure x
  834          rval <- resolveAllocIndexJVM ptr
  835          vs <- traverse (injectSetupValueJVM sym opts cc sc spec . MS.SetupTerm) tts
  836          globals' <- liftIO $ doEntireArrayStore bak globals rval vs
  837          OM (overrideGlobals .= globals')
  838 
  839     JVMPointsToArray _loc ptr Nothing ->
  840       case Map.lookup ptr (MS.csAllocations spec) of
  841         Just (_, AllocArray len _) ->
  842           do let vs = replicate len CJ.unassignedJVMValue
  843              rval <- resolveAllocIndexJVM ptr
  844              globals' <- liftIO $ doEntireArrayStore bak globals rval vs
  845              OM (overrideGlobals .= globals')
  846         _ -> panic "JVMSetup" ["executePointsTo", "expected array allocation"]

  859 doEntireArrayStore ::
  860   (Crucible.IsSymBackend sym bak) =>
  861   bak ->
  862   Crucible.SymGlobalState sym ->
  863   Crucible.RegValue sym CJ.JVMRefType ->
  864   [Crucible.RegValue sym CJ.JVMValueType] ->
  865   IO (Crucible.SymGlobalState sym)
  866 doEntireArrayStore bak glob ref vs = foldM store glob (zip [0..] vs)
  867   where store g (i, v) = CJ.doArrayStore bak g ref i v

  869 -- | Given a 'TypedTerm' with a vector type, return the element type
  870 -- along with a list of its projected components. Return 'Nothing' if
  871 -- the 'TypedTerm' does not have a vector type.
  872 destVecTypedTerm :: SharedContext -> TypedTerm -> IO (Maybe (Cryptol.Type, [TypedTerm]))
  873 destVecTypedTerm sc (TypedTerm ttp t) =
  874   case asVec of
  875     Nothing -> pure Nothing
  876     Just (len, ety) ->
  877       do len_tm <- scNat sc (fromInteger len)
  878          ty_tm <- Cryptol.importType sc Cryptol.emptyEnv ety
  879          idxs <- traverse (scNat sc) (map fromInteger [0 .. len-1])
  880          ts <- traverse (scAt sc len_tm ty_tm t) idxs
  881          pure $ Just (ety, map (TypedTerm (TypedTermSchema (Cryptol.tMono ety))) ts)
  882   where
  883     asVec =
  884       do ty <- ttIsMono ttp
  885          (n, a) <- Cryptol.tIsSeq ty
  886          n' <- Cryptol.tIsNum n
  887          Just (n', a)

  925 -- | Map the given substitution over all 'SetupTerm' constructors in
  926 -- the given 'SetupValue'.
  927 instantiateSetupValue ::
  928   SharedContext     ->
  929   Map VarIndex Term ->
  930   SetupValue        ->
  931   IO SetupValue
  932 instantiateSetupValue sc s v =
  933   case v of
  934     MS.SetupVar _                     -> return v
  935     MS.SetupTerm tt                   -> MS.SetupTerm <$> doTerm tt
  936     MS.SetupNull ()                   -> return v
  937     MS.SetupGlobal empty _            -> absurd empty
  938     MS.SetupStruct empty _            -> absurd empty
  939     MS.SetupEnum empty                -> absurd empty
  940     MS.SetupTuple empty _             -> absurd empty
  941     MS.SetupSlice empty               -> absurd empty
  942     MS.SetupArray empty _             -> absurd empty
  943     MS.SetupElem empty _ _            -> absurd empty
  944     MS.SetupField empty _ _           -> absurd empty
  945     MS.SetupCast empty _              -> absurd empty
  946     MS.SetupUnion empty _ _           -> absurd empty
  947     MS.SetupGlobalInitializer empty _ -> absurd empty
  948     MS.SetupMux empty _ _ _           -> absurd empty
  949   where
  950     doTerm (TypedTerm schema t) = TypedTerm schema <$> scInstantiateExt sc s t

  992     RVal x -> Crucible.injectVariant sym W4.knownRepr CJ.tagR x

  996 projectJVMVal :: OnlineSolver solver =>
  997   Backend solver -> J.Type -> String -> Crucible.RegValue Sym CJ.JVMValueType -> IO JVMVal
  998 projectJVMVal bak ty msg' v =
  999   case ty of
 1000     J.BooleanType -> IVal <$> proj v CJ.tagI
 1001     J.ByteType    -> IVal <$> proj v CJ.tagI
 1002     J.CharType    -> IVal <$> proj v CJ.tagI
 1003     J.ShortType   -> IVal <$> proj v CJ.tagI
 1004     J.IntType     -> IVal <$> proj v CJ.tagI
 1005     J.LongType    -> LVal <$> proj v CJ.tagL
 1006     J.FloatType   -> err -- FIXME
 1007     J.DoubleType  -> err -- FIXME
 1008     J.ArrayType{} -> RVal <$> proj v CJ.tagR
 1009     J.ClassType{} -> RVal <$> proj v CJ.tagR
 1010   where
 1011     proj ::
 1012       forall tp.
 1013       Crucible.RegValue Sym CJ.JVMValueType ->
 1014       Ctx.Index CJ.JVMValueCtx tp ->
 1015       IO (Crucible.RegValue Sym tp)
 1016     proj val idx = Crucible.readPartExpr bak (Crucible.unVB (val Ctx.! idx)) msg
 1017 
 1018     msg = Crucible.GenericSimError $ "Ill-formed value for type " ++ show ty ++ " (" ++ msg' ++ ")"
 1019     err = Crucible.addFailedAssertion bak msg
