   31 ----------------------------------------------------------------------
   32 -- Monads for tracking whether values have changed
   33 
   34 class (Monad m, Applicative m) => ChangeMonad m where
   35   preserve :: a -> m a -> m a
   36   taint :: m a -> m a
   37   taint m = m >>= modified
   38   modified :: a -> m a
   39   modified x = taint (pure x)
   40   -- Laws (not a minimal set):
   41   -- taint (taint m) = taint m
   42   -- taint (pure x) = modified x
   43   -- fmap f (taint m) = taint (fmap f m)
   44   -- taint m1 <*> m2 = taint (m1 <*> m2)
   45   -- m1 <*> taint m2 = taint (m1 <*> m2)
   46   -- fmap f (modified x) = modified (f x)
   47   -- modified x >>= k = taint (k x)
   48   -- m >>= modified = taint m
   49   -- taint (modified x) = modified x
   50   -- taint (return x) = modified x
   51   -- taint (m >>= k) = taint m >>= k
   52   -- taint (m >>= k) = m >>= (taint . k)
   53   -- preserve x (pure _) = pure x
   54   -- preserve _ (modified y) = modified y
   55   -- preserve _ (taint m) = taint m
   56 
   57 change :: ChangeMonad m => (a -> Maybe a) -> a -> m a
   58 change f a =
   59   case f a of
   60     Nothing -> pure a
   61     Just x  -> modified x
   62 
   63 changeList :: ChangeMonad m => (a -> m a) -> [a] -> m [a]
   64 changeList f xs =
   65   preserve xs $
   66     case xs of
   67       [] -> pure []
   68       y : ys -> (:) <$> f y <*> changeList f ys
   69 
   70 ----------------------------------------------------------------------
   71 -- Change monad
   72 
   73 data Change a = Original a | Modified a
   74   deriving (Show, Functor)
   75 
   76 instance Applicative Change where
   77   pure x = Original x
   78   Original f <*> Original x = Original (f x)
   79   Original f <*> Modified x = Modified (f x)
   80   Modified f <*> Original x = Modified (f x)
   81   Modified f <*> Modified x = Modified (f x)
   82 
   83 instance Monad Change where
   84   return = pure
   85   Original x >>= k = k x
   86   Modified x >>= k = taint (k x)
   87 
   88 instance ChangeMonad Change where
   89   preserve x (Original _) = Original x
   90   preserve _ c@(Modified _) = c
   91   taint (Original x) = Modified x
   92   taint c@(Modified _) = c
   93   modified x = Modified x
   94 
   95 commitChange :: Change a -> a
   96 commitChange (Original x) = x
   97 commitChange (Modified x) = x
   98 -- ^ Satisfies the following laws:
   99 -- @commitChange (fmap f m) = f (commitChange m)@
  100 -- @commitChange (taint m) = commitChange m@
  101 -- @commitChange (m >>= k) = commitChange (k (commitChange m))@
  102 
  103 ----------------------------------------------------------------------
  104 -- Change monad transformer
  105 
  106 newtype ChangeT m a = ChangeT { runChangeT :: m (Change a) }
  107 
  108 instance Functor m => Functor (ChangeT m) where
  109   fmap f (ChangeT m) = ChangeT (fmap (fmap f) m)
  110 
  111 instance Applicative m => Applicative (ChangeT m) where
  112   pure x = ChangeT (pure (Original x))
  113   ChangeT m1 <*> ChangeT m2 = ChangeT (App.liftA2 (<*>) m1 m2)
  114 
  115 instance Monad m => Monad (ChangeT m) where
  116   return = pure
  117   ChangeT m >>= k = ChangeT (m >>= f)
  118     where f (Original x) = runChangeT (k x)
  119           f (Modified x) = runChangeT (k x) >>= (return . taint)
  120 
  121 instance MonadTrans ChangeT where
  122   lift m = ChangeT (liftM Original m)
  123 
  124 instance MonadIO m => MonadIO (ChangeT m) where
  125   liftIO m = lift (liftIO m)
  126 
  127 instance Monad m => ChangeMonad (ChangeT m) where
  128   preserve x (ChangeT m) = ChangeT (liftM (preserve x) m)
  129   taint (ChangeT m) = ChangeT (liftM taint m)
  130   modified x = ChangeT (return (modified x))
  131 
  132 commitChangeT :: Monad m => ChangeT m a -> m a
  133 commitChangeT (ChangeT m) = liftM commitChange m
  134 -- ^ Is a natural transformation from @ChangeT m@ to @m@:
  135 -- @commitChangeT (fmap f m) = fmap f (commitChangeT m)@
  136 -- @commitChangeT (lift m) = m@
  137 -- @commitChangeT (m >>= k) = commitChangeT m >>= (commitChangeT . k)@
  138 -- @commitChangeT (return x) = return x@
  139 -- @commitChangeT (taint m) = commitChangeT m@
  140 
  141 preserveChangeT :: Monad m => a -> ChangeT m (m a) -> ChangeT m a
  142 preserveChangeT x (ChangeT c) = ChangeT (c >>= k)
  143   where k (Original _) = return (Original x)
  144         k (Modified m) = liftM Modified m
  145 -- ^ Satisfies @preserveChangeT x (return _) = return x@ and
  146 -- @preserveChangeT _ (modified m) = taint (lift m)@.
  147 
  148 flatten :: Monad m => (a -> ChangeT m (m a)) -> a -> ChangeT m a
  149 flatten f x = ChangeT (runChangeT (f x) >>= k)
  150   where k (Original _) = return (Original x)
  151         k (Modified m) = liftM Modified m
  152 -- ^ @flatten f x = preserveChangeT x (f x)@
