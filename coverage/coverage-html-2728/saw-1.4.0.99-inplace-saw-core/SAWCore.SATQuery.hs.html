   17 -- | This datatype represents a satisfiability query that might
   18 --   be dispatched to a solver.  It carries a series of assertions
   19 --   to be made to a solver, together with a collection of
   20 --   variables we expect the solver to report models over,
   21 --   and a collection of @VarIndex@ values identifying
   22 --   subterms that should be considered uninterpreted.
   23 --
   24 --   All the free variables in the query should
   25 --   appear either in @satVariables@ or @satUninterp@.
   26 --   Constant values for which definitions are provided
   27 --   may also appear in @satUninterp@, in which case
   28 --   they will be treated as uninterpreted.  Otherwise,
   29 --   their definitions will be unfolded.
   30 --
   31 --   Solve solvers do not support uninterpreted values
   32 --   and will fail if presented a query that requests them.
   33 data SATQuery =
   34   SATQuery
   35   { satVariables :: Map VarName FirstOrderType
   36       -- ^ The variables in the query, for which we
   37       --   expect the solver to find values in satisfiable
   38       --   cases.
   39 
   40   , satUninterp  :: Set VarIndex
   41       -- ^ A set indicating which variables and constant
   42       --   values should be considered uninterpreted by
   43       --   the solver. Models will not report values
   44       --   for uninterpreted values.
   45 
   46   , satAsserts   :: [SATAssert]
   47       -- ^ A collection of assertions. The overall
   48       --   query should be understood as the conjunction
   49       --   of these terms.
   50   }
   51 
   52 -- | The type of assertions we can make to a solver. These
   53 --   are either boolean terms, or universally-quantified
   54 --   statements. At present, only the What4 backends can
   55 --   handle universally quantified statments, and only
   56 --   some of the solvers will accept them without errors.
   57 data SATAssert
   58    = BoolAssert Term -- ^ A boolean term to be asserted
   59    | UniversalAssert [(VarName, FirstOrderType)] [Term] Term
   60           -- ^ A universally-quantified assertion, consisting of a
   61           --   collection of first-order variables, a sequence
   62           --   of boolean hypotheses, and a boolean conclusion
   63 
   64 -- | The result of a sat query. In the event a model is found,
   65 --   return a mapping from variable names to values.
   66 data SATResult
   67   = Unsatisfiable
   68   | Satisfiable (VarName -> IO FirstOrderValue)
   69   | Unknown
   70 
   71 -- | Compute the conjunction of all the assertions
   72 --   in this SAT query as a single term of type Bool.
   73 --
   74 --   This method of reducing a sat query to a boolean
   75 --   cannot be used for universally-quantified assertions,
   76 --   and will raise an error if it encounters one.
   77 satQueryAsTerm :: SharedContext -> SATQuery -> IO Term
   78 satQueryAsTerm sc satq =
   79   case satAsserts satq of
   80          [] -> scBool sc True
   81          (BoolAssert x:xs) -> loop x xs
   82          (UniversalAssert{} : _) -> univFail
   83  where
   84    univFail = fail "satQueryAsTerm : Solver backend cannot handle universally-quantified assertions"
   85 
   86    loop x [] = return x
   87    loop x (BoolAssert y:xs) =
   88      do x' <- scAnd sc x y
   89         loop x' xs
   90    loop _ (UniversalAssert{} : _) = univFail
   91 
   92 -- | Compute the conjunction of all the assertions
   93 --   in this SAT query as a single term of type Prop.
   94 satQueryAsPropTerm :: SharedContext -> SATQuery -> IO Term
   95 satQueryAsPropTerm sc satq =
   96   scTupleType sc =<< mapM assertAsPropTerm (satAsserts satq)
   97   where assertAsPropTerm (BoolAssert b) = scEqTrue sc b
   98         assertAsPropTerm (UniversalAssert vars hs g) =
   99           do vars' <- traverse (traverse (scFirstOrderType sc)) vars
  100              scPiList sc vars' =<<
  101                scEqTrue sc =<< foldrM (scImplies sc) g hs
