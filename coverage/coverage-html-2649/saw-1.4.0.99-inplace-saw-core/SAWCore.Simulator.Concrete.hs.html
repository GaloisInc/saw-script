   45 -- | Evaluator for shared terms.
   46 evalSharedTerm :: ModuleMap -> Map Ident CPrim -> Map VarIndex CValue -> Term -> CValue
   47 evalSharedTerm m addlPrims ecVals t =
   48   runIdentity $ do
   49     cfg <- Sim.evalGlobal m (Map.union constMap addlPrims) extcns (\_ _ -> Nothing) primHandler lazymux
   50     Sim.evalSharedTerm cfg t
   51   where
   52     lazymux = Prims.lazyMuxValue prims
   53     extcns ec =
   54       case Map.lookup (ecVarIndex ec) ecVals of
   55         Just v  -> return v
   56         Nothing -> return $ Prim.userError $ "Unimplemented: external constant " ++ show (ecName ec)
   57     primHandler nm msg env =
   58       return $ Prim.userError $ unlines
   59         [ "Could not evaluate primitive " ++ Text.unpack (toAbsoluteName (nameInfo nm))
   60         , "On argument " ++ show (length env)
   61         , Text.unpack msg
   62         ]

   67 data Concrete
   68 
   69 type instance EvalM Concrete = Identity
   70 type instance VBool Concrete = Bool
   71 type instance VWord Concrete = BitVector
   72 type instance VInt  Concrete = Integer
   73 type instance VArray Concrete = ()
   74 type instance Extra Concrete = CExtra

   76 type CValue = Value Concrete -- (WithM Identity Concrete)
   77 type CPrim  = Prims.Prim Concrete

   79 data CExtra
   80   = CStream (IntTrie CValue)
   81 
   82 instance Show CExtra where
   83   show (CStream _) = "<stream>"

   85 toBool :: CValue -> Bool
   86 toBool (VBool b) = b
   87 toBool x = error $ unwords ["SAWCore.Simulator.Concrete.toBool", show x]
   88 
   89 vWord :: BitVector -> CValue
   90 vWord x = VWord x
   91 
   92 toWord :: CValue -> BitVector
   93 toWord (VWord x) = x
   94 toWord (VVector vv) = Prim.packBitVector (fmap (toBool . runIdentity . force) vv)
   95 toWord x = error $ unwords ["SAWCore.Simulator.Concrete.toWord", show x]
   96 
   97 vStream :: IntTrie CValue -> CValue
   98 vStream x = VExtra (CStream x)
   99 
  100 toStream :: CValue -> IntTrie CValue
  101 toStream (VExtra (CStream x)) = x
  102 toStream x = error $ unwords ["SAWCore.Simulator.Concrete.toStream", show x]
  103 
  104 wordFun :: (BitVector -> CPrim) -> CPrim
  105 wordFun f = Prims.strictFun (\x -> f (toWord x))
  106 
  107 -- | op : (n : Nat) -> Vec n Bool -> Nat -> Vec n Bool
  108 bvShiftOp :: (BitVector -> Int -> BitVector) -> CPrim
  109 bvShiftOp natOp =
  110   Prims.constFun $
  111   wordFun $ \x ->
  112   Prims.strictFun $ \y -> Prims.PrimValue $
  113     case y of
  114       VNat n | toInteger n < toInteger (maxBound :: Int) -> vWord (natOp x (fromIntegral n))
  115       _      -> error $ unwords ["SAWCore.Simulator.Concrete.shiftOp", show y]

  119 pure1 :: Applicative f => (a -> b) -> a -> f b
  120 pure1 f x = pure (f x)
  121 
  122 pure2 :: Applicative f => (a -> b -> c) -> a -> b -> f c
  123 pure2 f x y = pure (f x y)
  124 
  125 pure3 :: Applicative f => (a -> b -> c -> d) -> a -> b -> c -> f d
  126 pure3 f x y z = pure (f x y z)
  127 
  128 divOp :: (a -> b -> Maybe c) -> a -> b -> Identity c
  129 divOp f x y = maybe Prim.divideByZero pure (f x y)
  130 
  131 ite :: Bool -> a -> a -> a
  132 ite b x y = if b then x else y

  259 -- primitive bvToNat : (n : Nat) -> Vec n Bool -> Nat;
  260 bvToNatOp :: CPrim
  261 bvToNatOp = Prims.constFun $ wordFun $ Prims.PrimValue . VNat . fromInteger . unsigned

  263 -- primitive bvToInt : (n : Nat) -> Vec n Bool -> Integer;
  264 bvToIntOp :: CPrim
  265 bvToIntOp = Prims.constFun $ wordFun $ Prims.PrimValue . VInt . unsigned

  267 -- primitive sbvToInt : (n : Nat) -> Vec n Bool -> Integer;
  268 sbvToIntOp :: CPrim
  269 sbvToIntOp = Prims.constFun $ wordFun $ Prims.PrimValue . VInt . signed

  271 -- primitive intToBv : (n : Nat) -> Integer -> Vec n Bool;
  272 intToBvOp :: CPrim
  273 intToBvOp =
  274   Prims.natFun $ \n ->
  275   Prims.intFun $ \x ->
  276     Prims.PrimValue $ VWord $
  277      if n >= 0 then bv (fromIntegral n) x
  278                else bvNeg n $ bv (fromIntegral n) $ negate x

  282 toIntModOp :: CPrim
  283 toIntModOp =
  284   Prims.natFun $ \n ->
  285   Prims.intFun $ \x ->
  286     Prims.PrimValue $ VIntMod n (x `mod` toInteger n)
  287 
  288 fromIntModOp :: CPrim
  289 fromIntModOp =
  290   Prims.constFun $
  291   Prims.intModFun $ \x ->
  292     Prims.PrimValue $ VInt x
  293 
  294 intModEqOp :: CPrim
  295 intModEqOp =
  296   Prims.constFun $
  297   Prims.intModFun $ \x ->
  298   Prims.intModFun $ \y ->
  299     Prims.PrimValue $ VBool (x == y)
  300 
  301 intModBinOp :: (Integer -> Integer -> Integer) -> CPrim
  302 intModBinOp f =
  303   Prims.natFun $ \n ->
  304   Prims.intModFun $ \x ->
  305   Prims.intModFun $ \y ->
  306     Prims.PrimValue $ VIntMod n (f x y `mod` toInteger n)
  307 
  308 intModUnOp :: (Integer -> Integer) -> CPrim
  309 intModUnOp f =
  310   Prims.natFun $ \n ->
  311   Prims.intModFun $ \x ->
  312     Prims.PrimValue $ VIntMod n (f x `mod` toInteger n)

  316 -- MkStream :: (a :: sort 0) -> (Nat -> a) -> Stream a;
  317 mkStreamOp :: CPrim
  318 mkStreamOp =
  319   Prims.constFun $
  320   Prims.strictFun $ \f -> Prims.PrimValue $
  321     vStream (fmap (\n -> runIdentity (apply f (ready (VNat n)))) IntTrie.identity)
  322 
  323 -- streamGet :: (a :: sort 0) -> Stream a -> Nat -> a;
  324 streamGetOp :: CPrim
  325 streamGetOp =
  326   Prims.constFun $
  327   Prims.strictFun $ \xs ->
  328   Prims.strictFun $ \ix -> Prims.Prim $ case ix of
  329     VNat n -> return $ IntTrie.apply (toStream xs) (toInteger n)
  330     VIntToNat (VInt i) -> return $ IntTrie.apply (toStream xs) i
  331     VBVToNat _ w -> return $ IntTrie.apply (toStream xs) (unsigned (toWord w))
  332     n -> panic "streamGetOp" ["Expected Nat value; found " <> Text.pack (show n)]
