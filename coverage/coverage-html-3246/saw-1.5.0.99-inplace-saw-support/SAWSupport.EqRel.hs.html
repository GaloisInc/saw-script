    8 -- An 'EqRel' is a collection of sets representing equivalence classes.
    9 -- Two values are considered equivalent with respect to this relation if
   10 -- they are both in the same equivalence class.
   11 --
   12 -- The relation is defined pointwise: pairs of values are declared
   13 -- equivalent via 'insert', which implicitly extends the data structure
   14 -- such that the result is a still a valid equivalence relation
   15 -- (symmetric, transitive, reflexive).
   16 
   17 module SAWSupport.EqRel
   18   ( EqRel
   19   , empty
   20   , eq
   21   , insert
   22   ) where
   23 
   24 import Data.HashMap.Strict (HashMap)
   25 import qualified Data.HashMap.Strict as HashMap
   26 import Data.Hashable (Hashable)
   27 import Data.HashSet (HashSet)
   28 import qualified Data.HashSet as HashSet
   29 
   30 -- | Represents an equivalence relation that is defined pointwise.
   31 --   Values marked as equivalent via 'insert' are considered to
   32 --   belong to the same equivalence class.
   33 --   The relation is implicitly transitive, symmetric and reflexive.
   34 --   Specifically, for any @r@:
   35 --     * if @eq r x y@ is @True@
   36 --       then @eq (insert a b r) x y@ is @True@
   37 --     * if @x == y@ is @True@ then @eq r x y@ is @True@
   38 --     * @eq (insert x y r) x y @ is @True@
   39 --     * if @eq r x y@ is @True@,
   40 --       then @eq r y x@ is @True@
   41 --     * if @eq r x y@ and @eq r y z@ are @True@,
   42 --       then @eq r x z@ is @True@
   43 --     * in all other cases, @eq r x y@ is @False@
   44 --
   45 --   Invariant: for each @s@ in the codomain of the map @m@ and each
   46 --   @x@ in @s@: @HashMap.lookup x m == Just s@
   47 newtype EqRel a = EqRel (HashMap a (HashSet a))
   48 
   49 -- | An empty 'EqRel', where values are equivalent iff they
   50 --   are exactly equal.
   51 empty :: EqRel a
   52 empty = EqRel HashMap.empty
   53 
   54 -- | Test if two values are equivalent up to the given equivalence
   55 --   relation 'EqRel'.
   56 eq :: Hashable a => EqRel a -> a -> a -> Bool
   57 eq (EqRel m) x y =
   58   case HashMap.lookup y m of
   59     Just ys -> HashSet.member x ys
   60     Nothing -> x == y
   61 
   62 -- | Extend an equivalence relation 'EqRel' to combine the equivalence
   63 --   classes of the two given values (i.e. consider the
   64 --   two values to be equivalent).
   65 insert :: Hashable a => a -> a -> EqRel a -> EqRel a
   66 insert x y (EqRel m) = EqRel (HashMap.union (const s <$> HashSet.toMap s) m)
   67   where
   68     s = case (HashMap.lookup x m, HashMap.lookup y m) of
   69       (Just xs, Just ys) -> HashSet.union xs ys
   70       (Nothing, Just ys) -> HashSet.insert x ys
   71       (Just xs, Nothing) -> HashSet.insert y xs
   72       (Nothing, Nothing) -> HashSet.insert x $ HashSet.singleton y
