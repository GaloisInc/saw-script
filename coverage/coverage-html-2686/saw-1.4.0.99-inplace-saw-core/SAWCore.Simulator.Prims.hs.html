   86 -- | A utility type for implementing primitive functions.
   87 --   This type allows primtives to more easily define
   88 --   functions that expect certain kinds of arguments,
   89 --   and allows the simulator to respond gracefully if
   90 --   the actual arguments don't match the expected filters.
   91 data Prim l
   92   = PrimFun    (Thunk l -> Prim l)
   93   | PrimStrict (Value l -> Prim l)
   94   | forall a. PrimFilterFun Text (Value l -> MaybeT (EvalM l) a) (a -> Prim l)
   95   | PrimExcept (ExceptT Text (EvalM l) (Value l))
   96   | Prim (EvalM l (Value l))
   97   | PrimValue (Value l)
   98 
   99 -- | A primitive that takes a nonstrict argument
  100 primFun :: (Thunk l -> Prim l) -> Prim l
  101 primFun = PrimFun
  102 
  103 -- | A primitive that takes a strict argument
  104 strictFun :: (Value l -> Prim l) -> Prim l
  105 strictFun = PrimStrict
  106 
  107 -- | A primitive that ignores an argument
  108 constFun :: Prim l -> Prim l
  109 constFun p = PrimFun (const p)
  110 
  111 -- | A primitive that requires a boolean argument
  112 boolFun :: VMonad l => (VBool l -> Prim l) -> Prim l
  113 boolFun = PrimFilterFun "expected Bool" r
  114   where r (VBool b) = pure b
  115         r _ = mzero
  116 
  117 -- | A primitive that requires a concrete natural argument
  118 natFun :: VMonad l => (Natural -> Prim l) -> Prim l
  119 natFun = PrimFilterFun "expected Nat" r
  120   where r (VNat n) = pure n
  121         r (VCtorApp (nameInfo -> ModuleIdentifier "Prelude.Zero") _ [] [])  = pure 0
  122         r (VCtorApp (nameInfo -> ModuleIdentifier "Prelude.Succ") _ [] [x]) = succ <$> (r =<< lift (force x))
  123         r _ = mzero
  124 
  125 -- | A primitive that requires an integer argument
  126 intFun :: VMonad l => (VInt l -> Prim l) -> Prim l
  127 intFun = PrimFilterFun "expected Integer" r
  128   where r (VInt i) = pure i
  129         r _ = mzero
  130 
  131 -- | A primitive that requires a (Z n) argument
  132 intModFun :: VMonad l => (VInt l -> Prim l) -> Prim l
  133 intModFun = PrimFilterFun "expected IntMod" r
  134   where r (VIntMod _ i) = pure i
  135         r _ = mzero
  136 
  137 -- | A primitive that requires a type argument
  138 tvalFun :: VMonad l => (TValue l -> Prim l) -> Prim l
  139 tvalFun = PrimFilterFun "expected type value" r
  140   where r (TValue tv) = pure tv
  141         r _ = mzero
  142 
  143 stringFun :: VMonad l => (Text -> Prim l) -> Prim l
  144 stringFun = PrimFilterFun "expected string value" r
  145   where r (VString x) = pure x
  146         r _ = mzero
  147 
  148 -- | A primitive that requires a packed word argument
  149 wordFun :: VMonad l => Pack l -> (VWord l -> Prim l) -> Prim l
  150 wordFun pack = PrimFilterFun "expected word" r
  151   where r (VWord w)    = pure w
  152         r (VVector xs) = lift . pack =<< V.mapM (\x -> r' =<< lift (force x)) xs
  153         r _ = mzero
  154 
  155         r' (VBool b)   = pure b
  156         r' _ = mzero
  157 
  158 -- | A primitive that requires a vector argument
  159 vectorFun :: (VMonad l, Show (Extra l)) =>
  160   Unpack l -> (Vector (Thunk l) -> Prim l) -> Prim l
  161 vectorFun unpack = PrimFilterFun "expected vector" r
  162   where r (VVector xs) = pure xs
  163         r (VWord w)    = fmap (ready . VBool) <$> lift (unpack w)
  164         r _ = mzero

  487 -- @selectV mux maxValue valueFn v@ treats the vector @v@ as an
  488 -- index, represented as a big-endian list of bits. It does a binary
  489 -- lookup, using @mux@ as an if-then-else operator. If the index is
  490 -- greater than @maxValue@, then it returns @valueFn maxValue@.
  491 selectV :: (b -> a -> a -> a) -> Int -> (Int -> a) -> Vector b -> a
  492 selectV mux maxValue valueFn v = impl len 0
  493   where
  494     len = V.length v
  495     err = panic "selectV" ["Guarded access is out of range"]
  496     impl _ x | x > maxValue || x < 0 = valueFn maxValue
  497     impl 0 x = valueFn x
  498     impl i x = mux (vecIdx err v (len - i)) (impl j (x `setBit` j)) (impl j x) where j = i - 1

  530 -- which should be a value of type Nat.

  539 -- | Return the number of bits necessary to represent the given value,
  540 -- which should be a value of type Nat, calling 'panic' if this cannot be done.
  541 natSize :: (HasCallStack, Show (Extra l)) => Value l -> Natural
  542 natSize val = fromMaybe (panic "natSize" ["expected Nat, got: " <> Text.pack (show val)])
  543                         (natSizeMaybe val)

  545 -- | A primitive that requires a natural argument, returning its value as a
  546 -- 'Natural' if the argument is concrete, or a pair of a size in bits and a
  547 -- 'Value', if 'natSizeMaybe' returns 'Just'
  548 natSizeFun :: (HasCallStack, VMonad l) =>
  549               (Either (Natural, Value l) Natural -> Prim l) -> Prim l
  550 natSizeFun = PrimFilterFun "expected Nat with a known size" r
  551   where r (VNat n) = pure (Right n)
  552         r (VCtorApp (nameInfo -> ModuleIdentifier "Prelude.Zero") _ [] []) = pure (Right 0)
  553         r v@(VCtorApp (nameInfo -> ModuleIdentifier "Prelude.Succ") _ [] [x]) =
  554           lift (force x) >>= r >>= bimapM (const (szPr v)) (pure . succ)
  555         r v = Left <$> szPr v
  556         szPr v = maybe mzero (pure . (,v)) (natSizeMaybe v)

  558 -- | Convert the given value (which should be of type Nat) to a word
  559 -- of the given bit-width. The bit-width must be at least as large as
  560 -- that returned by @natSize@.
  561 natToWord :: (HasCallStack, VMonad l, Show (Extra l)) => BasePrims l ->
  562              Natural -> Value l -> MWord l
  563 natToWord bp w val =
  564   unless (w <= fromIntegral (maxBound :: Int))
  565          (panic "natToWord" ["width too large: " <> Text.pack (show w)]) >>
  566   -- TODO, remove the calls to fromIntegral below
  567   case val of
  568     VNat n -> bpBvLit bp (fromIntegral w) (toInteger n)
  569     VIntToNat _i -> panic "natToWord" ["Reached VIntToNat: TODO!"]
  570     VBVToNat xsize v ->
  571       do x <- toWord (bpPack bp) v
  572          case compare xsize (fromIntegral w) of
  573            GT -> bpBvSlice bp (xsize - fromIntegral w) (fromIntegral w) x
  574            EQ -> return x
  575            LT -> -- zero-extend x to width w
  576              do pad <- bpBvLit bp (fromIntegral w - xsize) 0
  577                 bpBvJoin bp pad x
  578     _ -> panic "natToWord" ["expected Nat, got: " <> Text.pack (show val)]

  580 -- | A primitive which is a unary operation on a natural argument.
  581 -- The second argument gives how to modify the size in bits of this operation's
  582 -- argument so that no overflow occurs (e.g. 'succ' for 'succ').
  583 -- The third argument gives this operation for a concrete natural argument.
  584 -- The fourth argument gives this operation for a natural argument of the given
  585 -- size in bits.
  586 unaryNatOp :: (HasCallStack, VMonad l, Show (Extra l)) => BasePrims l ->
  587               (Natural -> Natural) ->
  588               (Natural -> MValue l) ->
  589               (Int -> VWord l -> MValue l) -> Prim l
  590 unaryNatOp bp fw fn fv = natSizeFun $ \case
  591   Right n -> Prim (fn n)
  592   Left (w1,v) -> Prim $ do let w = fw w1
  593                            x <- natToWord bp w v
  594                            fv (fromIntegral w) x

  596 -- | A primitive which is a unary operation on a natural argument and which
  597 -- returns a natural.
  598 -- The second argument gives how to modify the size in bits of this operation's
  599 -- argument so that no overflow occurs (e.g. 'succ' for 'succ').
  600 -- The third argument gives this operation for a concrete natural argument.
  601 -- The fourth argument gives this operation for a natural argument of the given
  602 -- size in bits.
  603 unaryNatToNatOp :: (HasCallStack, VMonad l, Show (Extra l)) => BasePrims l ->
  604                    (Natural -> Natural) ->
  605                    (Natural -> Natural) ->
  606                    (Int -> VWord l -> MWord l) -> Prim l
  607 unaryNatToNatOp bp fw fn fv =
  608   unaryNatOp bp fw (\n -> pure (vNat (fn n)))
  609                    (\w x -> VBVToNat w . VWord <$> fv w x)

  611 -- | A primitive which is a binary operation on natural arguments.
  612 -- The second argument gives how to combine the the sizes in bits of this
  613 -- operation's arguments so that no overflow occurs (e.g. 'max' for
  614 -- comparisons, @(\w1 w2 -> suc (max w1 w2))@ for addition, '(+)' for
  615 -- multiplication).
  616 -- The third argument gives this operation for concrete natural arguments.
  617 -- The fourth argument gives this operation for natural arguments of the size
  618 -- in bits given by the second argument.
  619 binNatOp :: (HasCallStack, VMonad l, Show (Extra l)) => BasePrims l ->
  620             (Natural -> Natural -> Natural) ->
  621             (Natural -> Natural -> MValue l) ->
  622             (Int -> VWord l -> VWord l -> MValue l) -> Prim l
  623 binNatOp bp fw fn fv = natSizeFun (natSizeFun . go)
  624   where go (Right m) (Right n) = Prim (fn m n)
  625         go (Right m) (Left pr) = go (Left (widthNat m, VNat m)) (Left pr)
  626         go (Left pr) (Right n) = go (Left pr) (Left (widthNat n, VNat n))
  627         go (Left (w1,v1)) (Left (w2,v2)) = Prim $
  628           do let w = fw w1 w2
  629              x1 <- natToWord bp w v1
  630              x2 <- natToWord bp w v2
  631              fv (fromIntegral w) x1 x2

  633 -- | A primitive which is a binary operation on natural arguments and which
  634 -- returns a natural.
  635 -- The second argument gives how to combine the the sizes in bits of this
  636 -- operation's arguments so that no overflow occurs (e.g. 'max' for
  637 -- comparisons, @(\w1 w2 -> suc (max w1 w2))@ for addition, '(+)' for
  638 -- multiplication).
  639 -- The third argument gives this operation for concrete natural arguments.
  640 -- The fourth argument gives this operation for natural arguments of the size
  641 -- in bits given by the second argument.
  642 binNatToNatOp :: (HasCallStack, VMonad l, Show (Extra l)) => BasePrims l ->
  643                  (Natural -> Natural -> Natural) ->
  644                  (Natural -> Natural -> Natural) ->
  645                  (Int -> VWord l -> VWord l -> MWord l) -> Prim l

  647   binNatOp bp fw (\m n -> pure (vNat (fn m n)))

  739 -- bytesToString : (n : Nat) -> Vec n (Vec 8 Bool) -> String;
  740 bytesToStringOp :: forall l. (VMonad l, Show (Extra l))
  741                 => BasePrims l -> Prim l

  761                           zeroBits
