  133 data Value
  134   = VBool Bool
  135   | VString Text
  136   | VInteger Integer
  137   | VArray [Value]
  138   | VTuple [Value]
  139   | VMaybe (Maybe Value)
  140   | VRecord (Map SS.Name Value)
  141   | VLambda (Value -> TopLevel Value)
  142   | VTerm TypedTerm
  143   | VType Cryptol.Schema
  144   | VReturn Value -- Returned value in unspecified monad
  145   | VBind SS.Pos Value Value
  146     -- ^ Monadic bind in unspecified monad. Requires a source position because
  147     -- operations in these monads can fail at runtime.
  148   | VTopLevel (TopLevel Value)
  149   | VProofScript (ProofScript Value)
  150   | VSimpset SAWSimpset
  151   | VRefnset SAWRefnset
  152   | VTheorem Theorem
  153   | VBisimTheorem BisimTheorem
  154   -----
  155   | VLLVMCrucibleSetup !(LLVMCrucibleSetupM Value)
  156   | VLLVMCrucibleMethodSpec (CMSLLVM.SomeLLVM CMS.ProvedSpec)
  157   | VLLVMCrucibleSetupValue (CMSLLVM.AllLLVM CMS.SetupValue)
  158   -----
  159   | VJVMSetup !(JVMSetupM Value)
  160   | VJVMMethodSpec !(CMS.ProvedSpec CJ.JVM)
  161   | VJVMSetupValue !(CMS.SetupValue CJ.JVM)
  162   -----
  163   | VMIRSetup !(MIRSetupM Value)
  164   | VMIRMethodSpec !(CMS.ProvedSpec MIR)
  165   | VMIRSetupValue !(CMS.SetupValue MIR)
  166   -----
  167   | VLLVMModuleSkeleton ModuleSkeleton
  168   | VLLVMFunctionSkeleton FunctionSkeleton
  169   | VLLVMSkeletonState SkeletonState
  170   | VLLVMFunctionProfile FunctionProfile
  171   -----
  172   | VJavaType JavaType
  173   | VLLVMType LLVM.Type
  174   | VMIRType MIR.Ty
  175   | VCryptolModule CryptolModule
  176   | VJavaClass JSS.Class
  177   | VLLVMModule (Some CMSLLVM.LLVMModule)

  243 showsProofResult :: PPS.Opts -> ProofResult -> ShowS
  244 showsProofResult opts r =
  245   case r of
  246     ValidProof _ _ -> showString "Valid"
  247     InvalidProof _ ts _ -> showString "Invalid: [" . showMulti "" ts
  248     UnfinishedProof st  -> showString "Unfinished: " . shows (length (psGoals st)) . showString " goals remaining"
  249   where
  250     showVal t = shows (ppFirstOrderValue opts t)
  251     showEqn (x, t) = showEC x . showString " = " . showVal t
  252     showEC ec = showString (unpack (toShortName (ecName ec)))
  253 
  254     showMulti _ [] = showString "]"
  255     showMulti s (eqn : eqns) = showString s . showEqn eqn . showMulti ", " eqns
  256 
  257 showsSatResult :: PPS.Opts -> SatResult -> ShowS
  258 showsSatResult opts r =
  259   case r of
  260     Unsat _ -> showString "Unsat"
  261     Sat _ ts -> showString "Sat: [" . showMulti "" ts
  262     SatUnknown  -> showString "Unknown"
  263   where
  264     showVal t = shows (ppFirstOrderValue opts t)
  265     showEC ec = showString (unpack (toShortName (ecName ec)))
  266     showEqn (x, t) = showEC x . showString " = " . showVal t
  267     showMulti _ [] = showString "]"
  268     showMulti s (eqn : eqns) = showString s . showEqn eqn . showMulti ", " eqns

  270 showSimpset :: PPS.Opts -> Simpset a -> String
  271 showSimpset opts ss =
  272   unlines ("Rewrite Rules" : "=============" : map (show . ppRule) (listRules ss))
  273   where
  274     ppRule r =
  275       PP.pretty '*' PP.<+>
  276       (PP.nest 2 $ PP.fillSep
  277        [ ppTerm (lhsRewriteRule r)
  278        , PP.pretty '=' PP.<+> ppTerm (rhsRewriteRule r) ])
  279     ppTerm t = SAWCorePP.ppTerm opts t
  280 
  281 -- | Pretty-print a 'Refnset' to a 'String'
  282 showRefnset :: PPS.Opts -> MRSolver.Refnset a -> String
  283 showRefnset opts ss =
  284   unlines ("Refinements" : "=============" : map (show . ppFunAssump)
  285                                                  (MRSolver.listFunAssumps ss))
  286   where
  287     ppFunAssump (MRSolver.FunAssump ctx f args rhs _) =
  288       PP.pretty '*' PP.<+>
  289       (PP.nest 2 $ PP.fillSep
  290        [ ppTermAppInCtx opts ctx (funNameTerm f) args
  291        , PP.pretty ("|=" :: String) PP.<+> ppFunAssumpRHS ctx rhs ])
  292     ppFunAssumpRHS ctx (OpaqueFunAssump f args) =
  293       ppTermAppInCtx opts ctx (funNameTerm f) args
  294     ppFunAssumpRHS ctx (RewriteFunAssump rhs) =
  295       SAWCorePP.ppTermInCtx opts (map fst $ mrVarCtxInnerToOuter ctx) rhs

  297 showsPrecValue :: PPS.Opts -> SAWNamingEnv -> Int -> Value -> ShowS
  298 showsPrecValue opts nenv p v =
  299   case v of
  300     VBool True -> showString "true"
  301     VBool False -> showString "false"
  302     VString s -> shows s
  303     VInteger n -> shows n
  304     VArray vs -> PPS.showBrackets $ PPS.showCommaSep $ map (showsPrecValue opts nenv 0) vs
  305     VTuple vs -> showParen True $ PPS.showCommaSep $ map (showsPrecValue opts nenv 0) vs
  306     VMaybe (Just v') -> showString "(Just " . showsPrecValue opts nenv 0 v' . showString ")"
  307     VMaybe Nothing -> showString "Nothing"
  308     VRecord m ->
  309       PPS.showBraces $ PPS.showCommaSep $ map showFld (M.toList m)
  310         where
  311           showFld (n, fv) =
  312             showString (unpack n) . showString "=" . showsPrecValue opts nenv 0 fv
  313 
  314     VLambda {} -> showString "<<function>>"
  315     VTerm t -> showString (SAWCorePP.showTermWithNames opts nenv (ttTerm t))
  316     VType sig -> showString (pretty sig)
  317     VReturn {} -> showString "<<monadic>>"
  318     VBind {} -> showString "<<monadic>>"
  319     VTopLevel {} -> showString "<<TopLevel>>"
  320     VSimpset ss -> showString (showSimpset opts ss)
  321     VRefnset ss -> showString (showRefnset opts ss)
  322     VProofScript {} -> showString "<<proof script>>"
  323     VTheorem thm ->
  324       showString "Theorem " .
  325       showParen True (showString (prettyProp opts nenv (thmProp thm)))
  326     VBisimTheorem _ -> showString "<<Bisimulation theorem>>"
  327     VLLVMCrucibleSetup{} -> showString "<<Crucible Setup>>"
  328     VLLVMCrucibleSetupValue{} -> showString "<<Crucible SetupValue>>"
  329     VLLVMCrucibleMethodSpec{} -> showString "<<Crucible MethodSpec>>"
  330     VLLVMModuleSkeleton s -> shows s
  331     VLLVMFunctionSkeleton s -> shows s
  332     VLLVMSkeletonState _ -> showString "<<Skeleton state>>"
  333     VLLVMFunctionProfile _ -> showString "<<Array sizes for function>>"
  334     VJavaType {} -> showString "<<Java type>>"
  335     VLLVMType t -> showString (show (Crucible.LLVM.ppType t))
  336     VMIRType t -> showString (show (PP.pretty t))
  337     VCryptolModule m -> showString (showCryptolModule m)
  338     VLLVMModule (Some m) -> showString (CMSLLVM.showLLVMModule m)
  339     VMIRModule m -> shows (PP.pretty (m^.rmCS^.collection))
  340     VMIRAdt adt -> shows (PP.pretty adt)
  341     VHeapsterEnv env -> showString (showHeapsterEnv env)
  342     VJavaClass c -> shows (prettyClass c)
  343     VProofResult r -> showsProofResult opts r
  344     VSatResult r -> showsSatResult opts r
  345     VUninterp u -> showString "Uninterp: " . shows u
  346     VAIG _ -> showString "<<AIG>>"
  347     VCFG (LLVM_CFG g) -> showString (show g)
  348     VCFG (JVM_CFG g) -> showString (show g)
  349     VGhostVar x -> showParen (p > 10)

  351     VYosysModule _ -> showString "<<Yosys module>>"
  352     VYosysImport _ -> showString "<<Yosys import>>"
  353     VYosysSequential _ -> showString "<<Yosys sequential>>"
  354     VYosysTheorem _ -> showString "<<Yosys theorem>>"
  355     VJVMSetup _      -> showString "<<JVM Setup>>"
  356     VJVMMethodSpec _ -> showString "<<JVM MethodSpec>>"

  358     VMIRSetup{} -> showString "<<MIR Setup>>"

  365 indexValue :: Value -> Value -> Value
  366 indexValue (VArray vs) (VInteger x)
  367     | i < length vs = vs !! i
  368     | otherwise = error "array index out of bounds"
  369     where i = fromInteger x
  370 indexValue _ _ = error "indexValue"

  372 lookupValue :: Value -> Text -> Value
  373 lookupValue (VRecord vm) name =
  374     case M.lookup name vm of
  375       Nothing -> error $ "no such record field: " ++ unpack name
  376       Just x -> x
  377 lookupValue _ _ = error "lookupValue"

  379 tupleLookupValue :: Value -> Integer -> Value

  381   | 0 <= i && fromIntegral i < length vs = vs !! fromIntegral i

  421 applyValue :: Value -> Value -> TopLevel Value
  422 applyValue (VLambda f) x = f x

  425 thenValue :: SS.Pos -> Value -> Value -> Value
  426 thenValue pos v1 v2 = VBind pos v1 (VLambda (const (return v2)))

  428 bindValue :: SS.Pos -> Value -> Value -> TopLevel Value
  429 bindValue pos v1 v2 = return (VBind pos v1 v2)

  431 forValue :: [Value] -> Value -> TopLevel Value
  432 forValue [] _ = return $ VReturn (VArray [])
  433 forValue (x : xs) f =
  434   do m1 <- applyValue f x
  435      m2 <- forValue xs f
  436      bindValue (SS.PosInternal "<for>") m1 (VLambda $ \v1 ->
  437        bindValue (SS.PosInternal "<for>") m2 (VLambda $ \v2 ->
  438          return $ VReturn (VArray (v1 : fromValue v2))))

  455 -- Note that the expansion type should have already been through the
  456 -- typechecker, so it's ok to panic if it turns out to be broken.
  457 addTypedef :: SS.Name -> SS.Type -> TopLevelRW -> TopLevelRW
  458 addTypedef name ty rw =
  459   let primsAvail = rwPrimsAvail rw
  460       typeInfo = rwTypeInfo rw
  461       ty' = SS.substituteTyVars primsAvail typeInfo ty
  462       typeInfo' = M.insert name (SS.Current, SS.ConcreteType ty') typeInfo
  463   in
  464   rw { rwTypeInfo = typeInfo' }

  466 mergeLocalEnv :: SharedContext -> LocalEnv -> TopLevelRW -> IO TopLevelRW
  467 mergeLocalEnv sc env rw = foldrM addBinding rw env
  468   where addBinding (LocalLet x ty md v) = extendEnv sc x ty md v
  469         addBinding (LocalTypedef n ty) = pure . addTypedef n ty

  583 io :: IO a -> TopLevel a
  584 io f = (TopLevel_ (liftIO f))
  585        `catches`
  586        [ Handler (\(ex :: X86Unsupported) -> handleX86Unsupported ex)
  587        , Handler (\(ex :: X86Error) -> handleX86Error ex)
  588        , Handler handleTopLevel
  589        , Handler handleIO
  590        ]
  591   where
  592     rethrow :: X.Exception ex => ex -> TopLevel a
  593     rethrow ex =
  594       do stk <- getStackTrace
  595          throwM (SS.TraceException stk (X.SomeException ex))

  597     handleTopLevel :: SS.TopLevelException -> TopLevel a
  598     handleTopLevel e = rethrow e
  599 
  600     handleIO :: X.IOException -> TopLevel a
  601     handleIO e
  602       | IOError.isUserError e =
  603           do pos <- getPosition
  604              rethrow (SS.TopLevelException pos (dropEnd 1 . drop 12 $ show e))
  605       | otherwise = rethrow e

  607     handleX86Unsupported (X86Unsupported path s) =
  608       do let pos = SS.FileOnlyPos path
  609          rethrow (SS.TopLevelException pos ("Unsupported x86 feature: " ++ s))

  611     handleX86Error (X86Error path optfunc s) =
  612       do let pos = case optfunc of
  613                Nothing -> SS.FileOnlyPos path
  614                Just func -> SS.FileAndFunctionPos path func
  615          rethrow (SS.TopLevelException pos ("Error in x86 code: " ++ s))

  777 extendEnv ::
  778   SharedContext ->
  779   SS.LName -> SS.Schema -> Maybe String -> Value -> TopLevelRW -> IO TopLevelRW
  780 extendEnv sc x ty md v rw =
  781   do ce' <-
  782        case v of
  783          VTerm t ->
  784            pure $ CEnv.bindTypedTerm (ident, t) ce
  785          VType s ->
  786            pure $ CEnv.bindType (ident, s) ce
  787          VInteger n ->
  788            pure $ CEnv.bindInteger (ident, n) ce
  789          VCryptolModule m ->
  790            pure $ CEnv.bindCryptolModule (modname, m) ce
  791          VString s ->
  792            do tt <- typedTermOfString sc (unpack s)
  793               pure $ CEnv.bindTypedTerm (ident, tt) ce
  794          VBool b ->
  795            do tt <- typedTermOfBool sc b
  796               pure $ CEnv.bindTypedTerm (ident, tt) ce
  797          _ ->
  798            pure ce
  799      pure $
  800       rw { rwValueInfo  = M.insert name (SS.Current, ty, v) (rwValueInfo rw)
  801          , rwDocs    = maybeInsert (SS.getVal name) md (rwDocs rw)
  802          , rwCryptol = ce'
  803          }
  804   where
  805     name = x
  806     -- XXX why is this using getOrig?
  807     ident = T.mkIdent (SS.getOrig x)
  808     modname = T.packModName [SS.getOrig x]
  809     ce = rwCryptol rw

  811 typedTermOfString :: SharedContext -> String -> IO TypedTerm
  812 typedTermOfString sc str =
  813   do let schema = Cryptol.tMono (Cryptol.tString (length str))
  814      bvNat <- scGlobalDef sc "Prelude.bvNat"
  815      bvNat8 <- scApply sc bvNat =<< scNat sc 8
  816      byteT <- scBitvector sc 8
  817      let scChar c = scApply sc bvNat8 =<< scNat sc (fromIntegral (fromEnum c))
  818      ts <- traverse scChar str
  819      trm <- scVector sc byteT ts
  820      pure (TypedTerm (TypedTermSchema schema) trm)
  821 
  822 typedTermOfBool :: SharedContext -> Bool -> IO TypedTerm
  823 typedTermOfBool sc b =
  824   do let schema = Cryptol.tMono Cryptol.tBit
  825      trm <- scBool sc b
  826      pure (TypedTerm (TypedTermSchema schema) trm)

 1366 -- | Implement stack tracing by adding error handlers that rethrow
 1367 -- user errors, prepended with the given string.
 1368 addTrace :: String -> Value -> Value
 1369 addTrace str val =
 1370   case val of
 1371     VLambda        f -> VLambda        (\x -> addTrace str `fmap` addTraceTopLevel str (f x))
 1372     VTopLevel      m -> VTopLevel      (addTrace str `fmap` addTraceTopLevel str m)
 1373     VProofScript   m -> VProofScript   (addTrace str `fmap` addTraceProofScript str m)
 1374     VBind pos v1 v2  -> VBind pos      (addTrace str v1) (addTrace str v2)
 1375     VLLVMCrucibleSetup (LLVMCrucibleSetupM m) -> VLLVMCrucibleSetup $ LLVMCrucibleSetupM $
 1376         addTrace str `fmap` underReaderT (underStateT (addTraceTopLevel str)) m
 1377   -- TODO? JVM setup blocks too?
 1378     _                -> val
 1379 
 1380 addTraceProofScript :: String -> ProofScript a -> ProofScript a
 1381 addTraceProofScript str (ProofScript m) = ProofScript (underExceptT (underStateT (addTraceTopLevel str)) m)
 1382 
 1383 addTraceTopLevel :: String -> TopLevel a -> TopLevel a
 1384 addTraceTopLevel str (TopLevel_ action) =
 1385   TopLevel_ (local (\ro -> ro{ roStackTrace = str : roStackTrace ro }) action)
