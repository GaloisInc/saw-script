   37 -- | Construct the prompt for the current environment.
   38 getPrompt :: REPL String
   39 getPrompt =
   40   do batch <- gets rIsBatch
   41      if batch then return ""
   42      else do
   43        mpst <- gets rProofState
   44        case mpst of
   45          Nothing ->
   46              return "sawscript> "
   47          Just pst ->
   48              return ("proof ("++show (length (psGoals pst))++")> ")
   49 
   50 -- | Haskeline-specific repl implementation.
   51 repl :: Maybe FilePath -> REPL ()
   52 repl mbBatchFile = do
   53     let style = case mbBatchFile of
   54           Nothing   -> defaultBehavior
   55           Just path -> useFile path
   56     settings <- liftIO (setHistoryFile replSettings)
   57     runInputTBehavior style settings $ withInterrupt loop
   58   where
   59 
   60   loop = do
   61     prompt <- lift getPrompt
   62     mb     <- handleInterrupt (return (Just "")) (getInputLines prompt [])
   63     case mb of
   64         Nothing ->
   65             -- EOF
   66             return ()
   67         Just txt -> do
   68             lift $ executeText (Text.pack txt)
   69             continue <- lift $ gets rContinue
   70             when continue
   71                 loop
   72 
   73   getInputLines prompt linesSoFar = do
   74       mbtext <- getInputLine prompt
   75       case mbtext of
   76           Nothing ->
   77               return Nothing
   78           Just text -> do
   79               let text' = filter (\c -> c /= '\r') text
   80               if not (null text') && last text' == '\\' then do
   81                   -- Get more lines
   82                   let newPrompt = map (\_ -> ' ') prompt
   83                   getInputLines newPrompt (init text' : linesSoFar)
   84               else do
   85                   -- We accumulated these backwards, so reverse
   86                   let linesAll = text : linesSoFar
   87                       linetext = unlines $ reverse linesAll
   88                   return $ Just linetext
   89 
   90 -- | Try to set the history file.
   91 setHistoryFile :: Settings REPL -> IO (Settings REPL)
   92 setHistoryFile ss =
   93   do dir <- getAppUserDataDirectory "saw"
   94      createDirectoryIfMissing True dir
   95      return ss { historyFile = Just (dir </> "history") }
   96    `X.catch` \(X.SomeException {}) -> return ss

   98 -- | Haskeline settings for the REPL.
   99 replSettings :: Settings REPL
  100 replSettings  = Settings
  101   { complete       = completeReplText
  102   , historyFile    = Nothing
  103   , autoAddHistory = True
  104   }

  107 -- Approximate lexer and parser
  108 -- (for use in completions)
  109 
  110 data ALPResult
  111     = HaveSAWScriptValue Text
  112     | HaveSAWScriptType Text
  113     | HaveCryptolValue Text
  114     | HaveCryptolType Text
  115     | HaveQuotedString
  116 
  117 isIdentChar :: Char -> Bool
  118 isIdentChar c = isAlphaNum c || c == '_' || c == '\''
  119 
  120 approxLexerParser :: Text -> ALPResult
  121 approxLexerParser text0 = nowhere text0 text0
  122   where
  123 
  124     -- State for nowhere in particular
  125     nowhere start here =
  126         case Text.uncons here of
  127             Nothing ->
  128                 HaveSAWScriptValue start
  129             Just ('{', here') -> case Text.uncons here' of
  130                 Nothing ->
  131                     -- End of string with a single left-brace.  In an
  132                     -- ideal world here we might complete from field
  133                     -- names of known record types, and add in another
  134                     -- left-brace as an option.  We don't have enough
  135                     -- infrastructure for that, though, so assume
  136                     -- we're still nowhere in particular.
  137                     nowhere here' here'
  138                 Just ('{', here'') ->
  139                     cryexpr here'' here''
  140                 Just ('|', here'') ->
  141                     crytype here'' here''
  142                 Just (c, here'') ->
  143                     nowhere (if isIdentChar c then start else here'') here''
  144             Just ('"', here') ->
  145                 qstring here'
  146             Just (':', here') ->
  147                 sawtypeinit here'
  148             Just (c, here') ->
  149                 nowhere (if isIdentChar c then start else here') here'
  150 
  151     -- State for when we've seen a colon so we probably want a type,
  152     -- but we haven't seen anything yet. Accept spaces.
  153     sawtypeinit here =
  154         case Text.uncons here of
  155             Nothing ->
  156                 HaveSAWScriptType ""
  157             Just (' ', here') ->
  158                 sawtypeinit here'
  159             Just (c, here') ->
  160                 if isIdentChar c then
  161                     sawtype here here'
  162                 else
  163                     nowhere here here
  164 
  165     -- State for when we've seen a colon so we probably want a type,
  166     -- and we're accumulating a word. Stop on spaces.
  167     sawtype start here =
  168         case Text.uncons here of
  169             Nothing ->
  170                 HaveSAWScriptType start
  171             Just (c, here') ->
  172                 if isIdentChar c then
  173                     sawtype start here'
  174                 else
  175                     nowhere here here
  176 
  177     -- State for Cryptol expressions in {{ }}.
  178     cryexpr start here =
  179         case Text.uncons here of
  180             Nothing ->
  181                 HaveCryptolValue start
  182             Just ('}', here') -> case Text.uncons here' of
  183                 Nothing ->
  184                     HaveCryptolValue ""
  185                 Just ('}', here'') ->
  186                     nowhere here'' here''
  187                 Just _ ->
  188                     cryexpr here' here'
  189             Just (c, here') ->
  190                 cryexpr (if isIdentChar c then start else here') here'
  191 
  192     -- State for Cryptol types in {| |}.
  193     crytype start here =
  194         case Text.uncons here of
  195             Nothing ->
  196                 HaveCryptolType start
  197             Just ('|', here') -> case Text.uncons here' of
  198                 Nothing ->
  199                     HaveCryptolType ""
  200                 Just ('}', here'') ->
  201                     nowhere here'' here''
  202                 Just _ ->
  203                     crytype here' here'
  204             Just (c, here') ->
  205                 crytype (if isIdentChar c then start else here') here'
  206 
  207     -- State for quoted strings.
  208     qstring here =
  209         case Text.uncons here of
  210             Nothing ->
  211                 HaveQuotedString
  212             Just ('\\', here') -> case Text.uncons here' of
  213                 Nothing ->
  214                     HaveQuotedString
  215                 Just (_, here'') ->
  216                     qstring here''
  217             Just ('"', here') ->
  218                 nowhere here' here'
  219             Just (_, here') ->
  220                 qstring here'

  224 -- Completion
  225 
  226 -- | Haskeline's completion cursors are string-zippers in String,
  227 --   pairs where the LHS is a reverse list of characters behind the
  228 --   cursor, and the RHS is a forward list of characters after the
  229 --   cursor.
  230 type Cursor = (String, String)
  231 
  232 -- | Extract the LHS from a Haskeline cursor in its raw form.
  233 cursorLeftRaw :: Cursor -> String
  234 cursorLeftRaw (l, _r) = l
  235 
  236 -- | Extract the LHS from a Haskeline cursor as Text.
  237 cursorLeftText :: Cursor -> Text
  238 cursorLeftText (l, _r) = sanitize (Text.pack $ reverse l)
  239 
  240 -- | Construct a completion that appends the rest of `word`
  241 --   having already seen `prefix`.
  242 appendCompletion :: (Int, Text) -> Completion
  243 appendCompletion (prefixlen, word) =
  244    Completion {
  245        replacement = Text.unpack $ Text.drop prefixlen word,
  246        display = Text.unpack word,
  247        isFinished = True
  248    }
  249 
  250 -- | Search a list of names by prefix
  251 searchNames :: [Text] -> Text -> [(Int, Text)]
  252 searchNames candidates prefix =
  253     let len = Text.length prefix
  254         visit candidate =
  255           if prefix `Text.isPrefixOf` candidate then
  256               Just (len, candidate)
  257           else
  258               Nothing
  259     in
  260     mapMaybe visit candidates
  261 
  262 -- | Search a list of names by prefix, where we get the
  263 --   list of names from a REPL action.
  264 searchNames' :: REPL [Text] -> Text -> REPL [(Int, Text)]
  265 searchNames' getCandidates prefix = do
  266     candidates <- getCandidates
  267     return $ searchNames candidates prefix
  268 
  269 -- | Search for SAWScript types by prefix
  270 searchSAWScriptTypes :: Text -> REPL [(Int, Text)]
  271 searchSAWScriptTypes prefix =
  272     searchNames' getSAWScriptTypeNames prefix

  274 -- | Search for SAWScript values/variables by prefix
  275 searchSAWScriptValues :: Text -> REPL [(Int, Text)]

  279 -- | Search for Cryptol types by prefix
  280 searchCryptolTypes :: Text -> REPL [(Int, Text)]

  289 -- | Search for SAWScript things based on context.
  290 searchSAWScriptText :: Text -> REPL (Maybe [(Int, Text)])
  291 searchSAWScriptText text =
  292    -- Run the approximate lexer/parser.
  293    --
  294    -- Maybe someday alex/happy will give us a completion mode so we
  295    -- can run the real lexer and parser...
  296    --
  297    case approxLexerParser text of
  298        HaveSAWScriptValue word -> Just <$> searchSAWScriptValues word
  299        HaveSAWScriptType word -> Just <$> searchSAWScriptTypes word
  300        HaveCryptolValue word -> Just <$> searchCryptolValues word
  301        HaveCryptolType word -> Just <$> searchCryptolTypes word
  302        HaveQuotedString -> return Nothing
  303 
  304 -- | Completion for SAWScript types
  305 completeSAWScriptType :: Text -> CompletionFunc REPL
  306 completeSAWScriptType text cursor = do
  307     candidates <- searchSAWScriptTypes text
  308     let completions = map appendCompletion candidates 
  309     return (cursorLeftRaw cursor, completions)
  310 
  311 -- | Completion for general SAWScript text / values
  312 completeSAWScriptValue :: Text -> CompletionFunc REPL
  313 completeSAWScriptValue text cursor = do
  314     mbCandidates <- searchSAWScriptText text
  315     case mbCandidates of
  316         Nothing ->
  317             completeFilename cursor
  318         Just candidates -> do
  319             let completions = map appendCompletion candidates
  320             return (cursorLeftRaw cursor, completions)
  321 
  322 -- | Completion for REPL :-commands
  323 completeReplCommand :: Text -> CompletionFunc REPL
  324 completeReplCommand text cursor =
  325     -- Split into words by spaces
  326     case Text.split isSpace text of
  327         [] ->
  328             -- Impossible: there's a colon in the string
  329             panic "replComp" ["The colon disappeared!"] 
  330         [cmdPrefix] -> do
  331             -- We have one word, which is a partial or full command
  332             -- name. Search for everything that begins with that
  333             -- prefix, then construct an append-completion for each.
  334             --
  335             -- Because we don't want to replace any of the existing
  336             -- text on the line, the LHS of the return value should
  337             -- always be the LHS of the input cursor.
  338 
  339             let completion cmd =
  340                     appendCompletion (Text.length cmdPrefix, cName cmd)
  341             let completions = map completion $ searchCommandsByPrefix cmdPrefix
  342             return (cursorLeftRaw cursor, completions)
  343         cmdName : args ->
  344             -- We have at least one partial argument
  345             case searchExactCommandByPrefix cmdName of
  346                 Nothing ->
  347                     -- It's not a valid command so there's no completions
  348                     return (cursorLeftRaw cursor, [])
  349                 Just cmd ->
  350                     -- Complete from whatever namespace the argument
  351                     -- type is in. XXX: this assumes that all commands
  352                     -- take either zero (NoArg) or one or more uniform
  353                     -- (other cases) of arguments. This isn't actually
  354                     -- true.  For example, completing for :cd will
  355                     -- cheerfully add any number of directory names,
  356                     -- but you can only actually give it one. Should
  357                     -- strengthen the argument schema.
  358                     case cBody cmd of
  359                         ExprArg _     -> completeSAWScriptValue (last args) cursor
  360                         VarArg _      -> completeSAWScriptValue (last args) cursor
  361                         TypeArgs _    -> completeSAWScriptType (last args) cursor
  362                         FilenameArg _ -> completeFilename cursor
  363                         NoArg       _ -> return (cursorLeftRaw cursor, [])
  364 
  365 -- | Top-level completion for the REPL.
  366 --
  367 completeReplText :: CompletionFunc REPL
  368 completeReplText cursor = do
  369   let text = cursorLeftText cursor
  370   if Text.null text then
  371       completeSAWScriptValue "" cursor
  372   else if Text.isPrefixOf ":" text then do
  373       completeReplCommand text cursor
  374   else
  375       completeSAWScriptValue text cursor
