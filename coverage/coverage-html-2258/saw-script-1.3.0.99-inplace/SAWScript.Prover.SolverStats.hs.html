    8 module SAWScript.Prover.SolverStats where
    9 
   10 import Data.Semigroup
   11 import Data.Set (Set)
   12 import qualified Data.Set as Set
   13 import Data.List (intercalate)
   14 
   15 import Prelude
   16 
   17 
   18 -- | Data structure for recording useful information about
   19 --   verification runs on goals.
   20 data SolverStats
   21  = SolverStats
   22    { -- | The names of the solvers that proved this goal.  If verification was
   23      --   skipped for some reason, this should be the empty list.
   24      solverStatsSolvers :: Set String
   25      -- | The size of the term(s) that represent the proof obligations
   26      --   involved in this verification goal.  Usually, this is the size
   27      --   of the SAWCore term representing the goal, but it might also
   28      --   be the size of an AIG network, etc.
   29    , solverStatsGoalSize :: Integer
   30    }
   31  deriving (Show)
   32 
   33 solverStats :: String -> Integer -> SolverStats
   34 solverStats nm sz = SolverStats (Set.singleton nm) sz
   35 
   36 instance Semigroup SolverStats where
   37   SolverStats xs n <> SolverStats ys m = SolverStats (mappend xs ys) (n+m)
   38 
   39 instance Monoid SolverStats where
   40   mempty = SolverStats mempty 0
   41   mappend = (<>)
   42 
   43 
   44 ppStats :: SolverStats -> String
   45 ppStats x = "Goal size " ++ show (solverStatsGoalSize x) ++
   46             ". Provers used: " ++
   47             intercalate "," (Set.toList (solverStatsSolvers x))
