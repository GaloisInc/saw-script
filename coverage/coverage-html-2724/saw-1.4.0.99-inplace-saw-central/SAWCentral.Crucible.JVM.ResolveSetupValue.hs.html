   85 data JVMTypeOfError
   86   = JVMPolymorphicType Cryptol.Schema
   87   | JVMNonRepresentableType Cryptol.Type
   88   | JVMInvalidTypedTerm TypedTermType
   89 
   90 instance Show JVMTypeOfError where
   91   show (JVMPolymorphicType s) =
   92     unlines
   93     [ "Expected monomorphic term"
   94     , "instead got:"
   95     , show (Cryptol.pp s)
   96     ]
   97   show (JVMNonRepresentableType ty) =
   98     unlines
   99     [ "Type not representable in JVM:"
  100     , show (Cryptol.pp ty)
  101     ]
  102   show (JVMInvalidTypedTerm tp) =
  103     unlines
  104     [ "Expected typed term with Cryptol represnentable type, but got"
  105     , show (ppTypedTermType tp)
  106     ]
  107 
  108 instance X.Exception JVMTypeOfError

  110 typeOfSetupValue ::
  111   X.MonadThrow m =>
  112   JVMCrucibleContext ->
  113   Map AllocIndex (MS.ConditionMetadata, Allocation) ->
  114   Map AllocIndex JIdent ->
  115   SetupValue ->
  116   m J.Type
  117 typeOfSetupValue _cc env _nameEnv val =
  118   case val of
  119     MS.SetupVar i ->
  120       case Map.lookup i env of
  121         Nothing ->
  122             panic "JVMSetup (in typeOfSetupValue)" [
  123                 "Unresolved prestate variable: " <> Text.pack (show i)
  124             ]
  125         Just (_, alloc) -> return (allocationType alloc)
  126     MS.SetupTerm tt ->
  127       case ttType tt of
  128         TypedTermSchema (Cryptol.Forall [] [] ty) ->
  129           case toJVMType (Cryptol.evalValType mempty ty) of
  130             Nothing -> X.throwM (JVMNonRepresentableType ty)
  131             Just jty -> return jty
  132         TypedTermSchema s -> X.throwM (JVMPolymorphicType s)
  133         tp -> X.throwM (JVMInvalidTypedTerm tp)
  134 
  135     MS.SetupNull () ->
  136       -- We arbitrarily set the type of NULL to java.lang.Object,
  137       -- because a) it is memory-compatible with any type that NULL
  138       -- can be used at, and b) it prevents us from doing any
  139       -- type-safe field accesses.
  140       return (J.ClassType (J.mkClassName "java/lang/Object"))
  141     MS.SetupGlobal empty _            -> absurd empty
  142     MS.SetupStruct empty _            -> absurd empty
  143     MS.SetupEnum empty                -> absurd empty
  144     MS.SetupTuple empty _             -> absurd empty
  145     MS.SetupSlice empty               -> absurd empty
  146     MS.SetupArray empty _             -> absurd empty
  147     MS.SetupElem empty _ _            -> absurd empty
  148     MS.SetupField empty _ _           -> absurd empty
  149     MS.SetupCast empty _              -> absurd empty
  150     MS.SetupUnion empty _ _           -> absurd empty
  151     MS.SetupGlobalInitializer empty _ -> absurd empty
  152     MS.SetupMux empty _ _ _           -> absurd empty
  153 
  154 lookupAllocIndex :: Map AllocIndex a -> AllocIndex -> a
  155 lookupAllocIndex env i =
  156   case Map.lookup i env of
  157     Just x -> x
  158     Nothing ->
  159       panic "JVMSetup (in lookupAllocIndex" [
  160           "Unresolved prestate variable: " <> Text.pack (show i)
  161       ]

  163 -- | Translate a SetupValue into a Crucible JVM value, resolving
  164 -- references
  165 resolveSetupVal ::
  166   JVMCrucibleContext ->
  167   Map AllocIndex JVMRefVal ->
  168   Map AllocIndex (MS.ConditionMetadata, Allocation) ->
  169   Map AllocIndex JIdent ->
  170   SetupValue ->
  171   IO JVMVal
  172 resolveSetupVal cc env _tyenv _nameEnv val =
  173   case val of
  174     MS.SetupVar i ->
  175       pure (RVal (lookupAllocIndex env i))
  176     MS.SetupTerm tm -> resolveTypedTerm cc tm
  177     MS.SetupNull () ->
  178       return (RVal (W4.maybePartExpr sym Nothing))
  179     MS.SetupGlobal empty _            -> absurd empty
  180     MS.SetupStruct empty _            -> absurd empty
  181     MS.SetupEnum empty                -> absurd empty
  182     MS.SetupTuple empty _             -> absurd empty
  183     MS.SetupSlice empty               -> absurd empty
  184     MS.SetupArray empty _             -> absurd empty
  185     MS.SetupElem empty _ _            -> absurd empty
  186     MS.SetupField empty _ _           -> absurd empty
  187     MS.SetupCast empty _              -> absurd empty
  188     MS.SetupUnion empty _ _           -> absurd empty
  189     MS.SetupGlobalInitializer empty _ -> absurd empty
  190     MS.SetupMux empty _ _ _           -> absurd empty
  191   where
  192     sym = cc^.jccSym
  193 
  194 resolveTypedTerm ::
  195   JVMCrucibleContext ->
  196   TypedTerm       ->
  197   IO JVMVal
  198 resolveTypedTerm cc tm =
  199   case ttType tm of
  200     TypedTermSchema (Cryptol.Forall [] [] ty) ->
  201       resolveSAWTerm cc (Cryptol.evalValType mempty ty) (ttTerm tm)
  202     tp -> fail $ unlines
  203             [ "resolveSetupVal: expected monomorphic term"
  204             , "but got a term of type"
  205             , show (ppTypedTermType tp)
  206             ]
  207 
  208 resolveSAWPred ::
  209   JVMCrucibleContext ->
  210   Term ->
  211   IO (W4.Pred Sym)
  212 resolveSAWPred cc = resolveBoolTerm (cc ^. jccSym) (cc ^. jccUninterp)
  213 
  214 resolveSAWTerm ::
  215   JVMCrucibleContext ->
  216   Cryptol.TValue ->
  217   Term ->
  218   IO JVMVal
  219 resolveSAWTerm cc tp tm =
  220   case tp of
  221     Cryptol.TVBit ->
  222       do b <- resolveBoolTerm sym (cc ^. jccUninterp) tm
  223          x0 <- W4.bvLit sym W4.knownNat (BV.zero W4.knownNat)
  224          x1 <- W4.bvLit sym W4.knownNat (BV.one  W4.knownNat)
  225          x <- W4.bvIte sym b x1 x0
  226          return (IVal x)
  227     Cryptol.TVInteger ->
  228       fail "resolveSAWTerm: unimplemented type Integer (FIXME)"
  229     Cryptol.TVIntMod _ ->
  230       fail "resolveSAWTerm: unimplemented type Z n (FIXME)"
  231     Cryptol.TVFloat{} ->
  232       fail "resolveSAWTerm: unimplemented type Float e p (FIXME)"
  233     Cryptol.TVArray{} ->
  234       fail "resolveSAWTerm: unimplemented type Array a b (FIXME)"
  235     Cryptol.TVRational ->
  236       fail "resolveSAWTerm: unimplemented type Rational (FIXME)"
  237     Cryptol.TVSeq sz Cryptol.TVBit ->
  238       let unint = cc ^. jccUninterp in
  239       case sz of
  240         8  -> do x <- resolveBitvectorTerm sym unint (W4.knownNat @8) tm
  241                  IVal <$> W4.bvSext sym W4.knownNat x
  242         16 -> do x <- resolveBitvectorTerm sym unint (W4.knownNat @16) tm
  243                  IVal <$> W4.bvSext sym W4.knownNat x
  244         32 -> IVal <$> resolveBitvectorTerm sym unint W4.knownNat tm
  245         64 -> LVal <$> resolveBitvectorTerm sym unint W4.knownNat tm
  246         _  -> fail ("Invalid bitvector width: " ++ show sz)
  247     Cryptol.TVSeq _sz _tp' ->
  248       fail "resolveSAWTerm: unimplemented sequence type"
  249     Cryptol.TVStream _tp' ->
  250       fail "resolveSAWTerm: unsupported infinite stream type"
  251     Cryptol.TVTuple _tps ->
  252       fail "resolveSAWTerm: unsupported tuple type"
  253     Cryptol.TVRec _flds ->
  254       fail "resolveSAWTerm: unsupported record type"
  255     Cryptol.TVFun _ _ ->
  256       fail "resolveSAWTerm: unsupported function type"
  257     Cryptol.TVNominal {} ->
  258       fail "resolveSAWTerm: unsupported nominal type"
  259   where
  260     sym = cc^.jccSym
  261 
  262 toJVMType :: Cryptol.TValue -> Maybe J.Type
  263 toJVMType tp =
  264   case tp of
  265     Cryptol.TVBit -> Just J.BooleanType
  266     Cryptol.TVInteger -> Nothing
  267     Cryptol.TVIntMod _ -> Nothing
  268     Cryptol.TVFloat{} -> Nothing
  269     Cryptol.TVArray{} -> Nothing
  270     Cryptol.TVRational -> Nothing
  271     Cryptol.TVSeq n Cryptol.TVBit ->
  272       case n of
  273         8  -> Just J.ByteType
  274         16 -> Just J.ShortType
  275         32 -> Just J.IntType
  276         64 -> Just J.LongType
  277         _  -> Nothing
  278     Cryptol.TVSeq _n t ->
  279       do t' <- toJVMType t
  280          Just (J.ArrayType t')
  281     Cryptol.TVStream _tp' -> Nothing
  282     Cryptol.TVTuple _tps -> Nothing
  283     Cryptol.TVRec _flds -> Nothing
  284     Cryptol.TVFun _ _ -> Nothing
  285     Cryptol.TVNominal {} -> Nothing

  287 equalValsPred ::
  288   JVMCrucibleContext ->
  289   JVMVal ->
  290   JVMVal ->
  291   IO (W4.Pred Sym)
  292 equalValsPred cc v1 v2 = go (v1, v2)
  293   where
  294   go :: (JVMVal, JVMVal) -> IO (W4.Pred Sym)
  295   go (RVal r1, RVal r2) = CJ.refIsEqual sym r1 r2
  296   go (IVal i1, IVal i2) = W4.bvEq sym i1 i2
  297   go (LVal l1, LVal l2) = W4.bvEq sym l1 l2
  298   go _ = return (W4.falsePred sym)
  299 
  300   sym = cc^.jccSym
