   45 -- | Type shorthand for an operation that can return warnings and/or
   46 --   errors. On error it returns Left and a list of messages, at least
   47 --   one of which is an error and therefore fatal. On success, it
   48 --   returns a (possibly empty) list of messages, which are all
   49 --   warnings and not fatal, and a result of type a.
   50 type WithMsgs a = Either [Text] ([Text], a)
   51 
   52 -- | The messages produced by the typechecker are pairs of position and
   53 --   String. Convert to Text. XXX: the error infrastructure is supposed
   54 --   to be what knows how to print source positions.
   55 convertTypeMsg :: (Pos, String) -> Text
   56 convertTypeMsg (pos, msg) =
   57     let pos' = Text.pack (show pos)
   58         msg' = Text.pack msg
   59     in
   60     pos' <> ": " <> msg'

   62 -- | Type shorthand for an include path.
   63 --
   64 --   The LHS is the directory the current file was included from
   65 --   (starts as ".") and the RHS is the user's search path.
   66 type IncludePath = (FilePath, [FilePath])

   68 -- | Wrap statements in a pushd/popd pair.
   69 --
   70 --   See the notes in the interpreter for why we do this.
   71 --
   72 --   FUTURE: if we keep this thing, it would be nice to attach the
   73 --   source position of the include to the push and pop statements.
   74 --
   75 wrapDir :: FilePath -> WithMsgs [Stmt] -> WithMsgs [Stmt]
   76 wrapDir dir result =
   77     case result of
   78         Left errs -> Left errs
   79         Right (msgs, stmts) ->
   80             let stmts' = 
   81                   let pos = PosInternal "pushd/popd derived from include" in
   82                   [StmtPushdir pos dir] ++ stmts ++ [StmtPopdir pos]
   83             in
   84             Right (msgs, stmts')

   86 -- | Read some SAWScript text, using the selected parser entry point.
   87 --
   88 --   Returns only in Either; caller is responsible for handling the
   89 --   resulting warnings and errors.
   90 --
   91 --   The FilePath argument is the name of the file the text comes
   92 --   from. If the text didn't come from a file, it should have a
   93 --   suitable placeholder string; that's what ends up in the source
   94 --   positions in the results.
   95 --
   96 readAny :: FilePath -> Text.Text -> ([Token Pos] -> Either ParseError a) -> WithMsgs a
   97 readAny fileName str parser =
   98     -- XXX printing of positions properly belongs to the
   99     -- message-printing infrastructure.
  100     let positionMsg pos msg =
  101           Text.pack (show pos) <> ": " <> msg
  102     in
  103     case lexSAW fileName str of
  104         Left (_verbosity, pos, msg) ->
  105             Left [positionMsg pos msg]
  106         Right (tokens, optmsg) ->
  107             let msgs = case optmsg of
  108                   Nothing -> []
  109                   Just (Options.Error, _, _) ->
  110                       panic "readAny" ["Lexer returned an error in the warning slot"]
  111                   Just (_verbosity, pos, msg) ->
  112                       [positionMsg pos msg]
  113             in
  114             case parser tokens of
  115                 Left err ->
  116                     let err' = Text.pack (show err) in
  117                     Left (msgs ++ [err'])
  118                 Right tree ->
  119                     -- Note: there's no such things as warnings in a
  120                     -- Happy parser, so there are no more messages to
  121                     -- add in this branch.
  122                     Right (msgs, tree)
  123 
  124 -- | Use the readAny result to panic if any messages were generated.
  125 panicOnMsgs :: Text -> WithMsgs a -> a
  126 panicOnMsgs whoAmI result = case result of
  127    Left errs ->
  128        panic whoAmI ("Unexpected errors:" : errs)
  129    Right ([], tree) ->
  130        tree
  131    Right (warns, _) ->
  132        panic whoAmI ("Unexpected warnings:" : warns)
  133 
  134 -- | Handle the readAny result in IO.
  135 --
  136 --   Also return in Either, to be consistent with the current interface
  137 --   into this file. Prints any warnings, and return just errors or a
  138 --   result.
  139 --
  140 dispatchMsgs :: Options -> WithMsgs a -> IO (Either [Text] a)
  141 dispatchMsgs opts result =
  142     let printMsg vrb msg =
  143           Options.printOutLn opts vrb (Text.unpack msg)
  144     in
  145     case result of
  146         Left errs ->
  147             pure $ Left errs
  148         Right (msgs, tree) -> do
  149             mapM_ (printMsg Options.Warn) msgs
  150             pure $ Right tree
  151 
  152 -- | Run the readAny result through the `Include` module to resolve
  153 --   @include@ statements.
  154 --
  155 --   If we fail after collecting warnings, print the warnings before
  156 --   returning the failure.
  157 --
  158 resolveIncludes ::
  159     Int -> IncludePath -> Options -> Inc.Processor a -> WithMsgs a ->
  160     IO (WithMsgs a)
  161 resolveIncludes depth incpath opts process result =
  162     let printMsg vrb msg =
  163           Options.printOutLn opts vrb (Text.unpack msg)
  164     in
  165     case result of
  166         Left errs ->
  167             pure $ Left errs
  168         Right (msgs, tree) -> do
  169             result' <- process (includeFile depth incpath opts) tree
  170             case result' of
  171                 Left errs -> do
  172                     mapM_ (printMsg Options.Warn) msgs
  173                     pure $ Left errs
  174                 Right tree' ->
  175                     pure $ Right (msgs, tree')

  177 -- | Read a type schema from a string. This is used to digest the type
  178 --   signatures for builtins, and the expansions for builtin typedefs.
  179 --
  180 -- Pure function that panics on any error or warning. This is
  181 -- appropriate since we use it only to handle the builtins; any
  182 -- glitches there are properly panics.
  183 --
  184 -- The first argument (fileName) is a string to pass as the
  185 -- filename for the lexer, which (complete with line and column
  186 -- numbering of dubious value) will go into the positions of the
  187 -- elements of the resulting type.
  188 --
  189 -- The fake file name names the builtin object involved, so we'll also
  190 -- use it as the panic location string if we need to panic.
  191 --
  192 -- FUTURE: we should figure out how to generate more meaningful
  193 -- positions (like "third argument of concat") but this at least
  194 -- allows telling the user which builtin the type came from.
  195 --
  196 -- Note: there is no way to reach @include@ statements from type
  197 -- schemas, so no need to process includes in what we read.
  198 --
  199 readSchemaPure ::
  200     FilePath ->
  201     PrimitiveLifecycle ->
  202     Map Name (PrimitiveLifecycle, NamedType) ->
  203     Text ->
  204     Schema
  205 readSchemaPure fakeFileName lc tyenv str = do
  206     let result = readAny fakeFileName str parseSchema
  207     let result' = case result of
  208           Left errs -> Left errs
  209           Right (msgs, schema) ->
  210               let (errs_or_results, warns) = checkSchema lc tyenv schema
  211                   warns' = map convertTypeMsg warns
  212               in
  213               case errs_or_results of
  214                   Left errs -> Left (msgs ++ warns' ++ map convertTypeMsg errs)
  215                   Right () -> Right (msgs ++ warns', schema)
  216     panicOnMsgs (Text.pack fakeFileName) result'

  218 -- | Read a schema pattern from a string. This is used by the
  219 --   :search REPL command.
  220 --
  221 --   Also runs the typechecker to check the pattern.
  222 --
  223 -- Note: there is no way to reach @include@ statements from schema
  224 -- patterns, so no need to process includes in what we read.
  225 --
  226 readSchemaPattern ::
  227     Options ->
  228     FilePath -> Environ -> RebindableEnv -> Set PrimitiveLifecycle -> Text ->
  229     IO (Either [Text] SchemaPattern)
  230 readSchemaPattern opts fileName environ rbenv avail str = do
  231   let result = readAny fileName str parseSchemaPattern
  232   let result' = case result of
  233         Left errs -> Left errs
  234         Right (msgs, pat) ->
  235           let Environ varenv tyenv _cryenv = environ in
  236 
  237           -- XXX it should not be necessary to do this munging
  238           let squash (pos, lc, ty, _val, _doc) = (pos, lc, ReadOnlyVar, ty)
  239               varenv' = Map.map squash $ ScopedMap.flatten varenv
  240               tyenv' = ScopedMap.flatten tyenv
  241               rbsquash (pos, ty, _val) = (pos, Current, RebindableVar, ty)
  242               rbenv' = Map.map rbsquash rbenv
  243               varenv'' = Map.union varenv' rbenv'
  244           in
  245           let (errs_or_results, warns) = checkSchemaPattern avail varenv'' tyenv' pat
  246               warns' = map convertTypeMsg warns
  247           in
  248           case errs_or_results of
  249               Left errs -> Left (msgs ++ warns' ++ map convertTypeMsg errs)
  250               Right results -> Right (msgs ++ warns', results)
  251   dispatchMsgs opts result'

  253 -- | Read an expression from a string. This is used by the
  254 --   :type REPL command.
  255 --
  256 -- It is possible for someone to provide an expression that includes
  257 -- an @include@ in a do-block, so just in case call into `Include`
  258 -- to resolve any that appear.
  259 --
  260 readExpression ::
  261     Options ->
  262     FilePath -> Environ -> RebindableEnv -> Set PrimitiveLifecycle -> Text ->
  263     IO (Either [Text] (Schema, Expr))
  264 readExpression opts fileName environ rbenv avail str = do
  265   let incpath = (".", Options.importPath opts)
  266 
  267   let result = readAny fileName str parseExpression
  268   result' <- resolveIncludes 0{-depth-} incpath opts Inc.processExpr result
  269   let result'' = case result' of
  270         Left errs -> Left errs
  271         Right (msgs, expr) ->
  272            let Environ varenv tyenv _cryenvs = environ in
  273 
  274            -- XXX it should not be necessary to do this munging
  275            let squash (defpos, lc, ty, _val, _doc) = (defpos, lc, ReadOnlyVar, ty)
  276                varenv' = Map.map squash $ ScopedMap.flatten varenv
  277                tyenv' = ScopedMap.flatten tyenv
  278                rbsquash (defpos, ty, _val) = (defpos, Current, RebindableVar, ty)
  279                rbenv' = Map.map rbsquash rbenv
  280                varenv'' = Map.union varenv' rbenv'
  281            in
  282            -- XXX: also it shouldn't be necessary to do this wrappery
  283            let pos = getPos expr
  284                decl = Decl pos (PWild pos Nothing) Nothing expr
  285            in
  286            let (errs_or_results, warns) = checkDecl avail varenv'' tyenv' decl
  287                warns' = map convertTypeMsg warns
  288            in
  289            case errs_or_results of
  290                Left errs -> Left (msgs ++ warns' ++ map convertTypeMsg errs)
  291                Right decl' ->
  292                    let expr' = dDef decl'
  293                        schema = case dType decl' of
  294                          Just sch -> sch
  295                          Nothing ->
  296                              -- If the typechecker didn't insert a type, it's bust,
  297                              -- so panic. Not much point in printing the expression
  298                              -- or position in panic, since it's what the user just
  299                              -- typed.
  300                              panic "readExpressionChecked" [
  301                                  "Typechecker failed to produce a type"
  302                              ]
  303                    in
  304                    Right (msgs ++ warns', (schema, expr'))
  305   dispatchMsgs opts result''

  320 -- | Find a file, potentially looking in a list of multiple search paths (as
  321 -- specified via the @SAW_IMPORT_PATH@ environment variable or
  322 -- @-i@/@--import-path@ command-line options). If the file was successfully
  323 -- found, return the full path. If not, fail by returning `Left`.
  324 --
  325 locateFile :: IncludePath -> FilePath -> IO (Either [Text] FilePath)
  326 locateFile (current, rawdirs) file = do
  327   -- Replace "." in the search path with the directory the current
  328   -- file came from, which might or might not be ".".
  329   let dirs = map adjust rawdirs
  330       adjust "." = current
  331       adjust dir = dir
  332 
  333   mfname <- findFile dirs file
  334   case mfname of
  335     Nothing -> do
  336         let msgs = [
  337                 "Couldn't find file: " <> Text.pack file,
  338                 "  Searched in directories:"
  339              ] ++ map (\p -> "    " <> Text.pack p) dirs
  340         return $ Left msgs
  341     Just fname ->
  342       -- NB: Normalise the path name. The default SAW_IMPORT_PATH contains ".",
  343       -- and the behavior of filepath's 'normalise' function is to prepend a
  344       -- search path to the front of the file path that is found, which can
  345       -- cause paths like "./foo.saw" to be returned. This looks ugly in error
  346       -- messages, where we would rather display "foo.saw" instead.
  347       return $ Right (normalise fname)
  348 
  349 -- | Load the 'Stmt's in a @.saw@ file.
  350 --   Doesn't run the typechecker (yet).
  351 includeFile :: Int -> IncludePath -> Options -> FilePath ->
  352     IO (Either [Text] [Stmt])
  353 includeFile depth incpath opts fname = do
  354   result <- locateFile incpath fname
  355   case result of
  356     Left errs -> return $ Left errs
  357     Right fname' ->
  358       if depth > 128 then
  359           pure $ Left ["Maximum include depth exceeded"]
  360       else do
  361           let (_current, dirs) = incpath
  362               current' = takeDirectory fname'
  363               incpath' = (current', dirs)
  364 
  365           Options.printOutLn opts Options.Info $ "Loading file " ++ show fname'
  366           ftext <- TextIO.readFile fname'
  367 
  368           let result' = wrapDir current' $ readAny fname ftext parseModule
  369           result'' <- resolveIncludes (depth + 1) incpath' opts Inc.processStmts result'
  370           dispatchMsgs opts result''

  372 -- | Find a file, potentially looking in a list of multiple search paths (as
  373 -- specified via the @SAW_IMPORT_PATH@ environment variable or
  374 -- @-i@/@--import-path@ command-line options). If the file was successfully
  375 -- found, load it. If not, fail by returning `Left`.
  376 --
  377 -- Doesn't run the typechecker (yet).
  378 findAndLoadFileUnchecked :: Options -> FilePath -> IO (Either [Text] [Stmt])
  379 findAndLoadFileUnchecked opts fp =
  380   let incpath = (".", Options.importPath opts) in
  381   includeFile 0{-depth-} incpath opts fp
