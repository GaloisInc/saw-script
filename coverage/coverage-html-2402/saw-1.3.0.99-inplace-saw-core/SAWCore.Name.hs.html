   78 -- Module Names ----------------------------------------------------------------
   79 
   80 newtype ModuleName = ModuleName Text
   81   deriving (Eq, Ord, Generic, TH.Lift)
   82 
   83 instance Hashable ModuleName -- automatically derived
   84 
   85 instance Show ModuleName where
   86   show (ModuleName s) = Text.unpack s
   87 
   88 
   89 moduleNameText :: ModuleName -> Text
   90 moduleNameText (ModuleName x) = x
   91 
   92 moduleNamePieces :: ModuleName -> [Text]
   93 moduleNamePieces (ModuleName x) = Text.splitOn "." x
   94 
   95 -- | Create a module name given a list of strings with the top-most
   96 -- module name given first.
   97 mkModuleName :: [Text] -> ModuleName
   98 mkModuleName nms = ModuleName $ Text.intercalate "." nms
   99 
  100 preludeName :: ModuleName
  101 preludeName = mkModuleName ["Prelude"]

  104 -- Identifiers -----------------------------------------------------------------
  105 
  106 data Ident =
  107   Ident
  108   { identModule :: ModuleName
  109   , identBaseName :: Text
  110   }
  111   deriving (Eq, Ord, Generic)
  112 
  113 instance Hashable Ident -- automatically derived
  114 
  115 instance Show Ident where
  116   show (Ident m s) = shows m ('.' : Text.unpack s)
  117 
  118 identText :: Ident -> Text
  119 identText i = moduleNameText (identModule i) <> Text.pack "." <> identBaseName i
  120 
  121 identPieces :: Ident -> NonEmpty Text
  122 identPieces i =
  123   case moduleNamePieces (identModule i) of
  124     [] -> identBaseName i :| []
  125     (x:xs) -> x :| (xs ++ [identBaseName i])
  126 
  127 identName :: Ident -> String
  128 identName = Text.unpack . identBaseName
  129 
  130 instance Read Ident where
  131   readsPrec _ str =
  132     let (str1, str2) = break (not . isIdChar) str in
  133     [(parseIdent str1, str2)]
  134 
  135 mkIdent :: ModuleName -> Text -> Ident
  136 mkIdent m s = Ident m s
  137 
  138 -- | Make a \"coq-safe\" identifier from a string that might contain
  139 -- non-identifier characters, where we use the SAW core notion of identifier
  140 -- characters as letters, digits, underscore and primes. Any disallowed
  141 -- character is mapped to the string @__xNN@, where @NN@ is the hexadecimal code
  142 -- for that character. Additionally, a SAW core identifier is not allowed to
  143 -- start with a prime, so a leading underscore is added in such a case.
  144 mkSafeIdent :: ModuleName -> String -> Ident
  145 mkSafeIdent _ [] = fromString "_"
  146 mkSafeIdent mnm nm =
  147   let is_safe_char c = isAlphaNum c || c == '_' || c == '\'' in
  148   mkIdent mnm $ Text.pack $
  149   (if nm!!0 == '\'' then ('_' :) else id) $
  150   concatMap
  151   (\c -> if is_safe_char c then [c] else
  152            "__x" ++ showHex (ord c) "")
  153   nm
  154 
  155 -- | Parse a fully qualified identifier.
  156 parseIdent :: String -> Ident
  157 parseIdent s0 =
  158     case reverse (breakEach s0) of
  159       (_:[]) -> panic "parseIdent" ["Empty module name"]
  160       (nm:rMod) -> mkIdent (mkModuleName (reverse rMod)) nm
  161       _ -> panic "parseIdent" ["Bad identifier " <> Text.pack s0]
  162   where breakEach s =
  163           case break (=='.') s of
  164             (h, []) -> [Text.pack h]
  165             (h, _ : r) -> Text.pack h : breakEach r
  166 
  167 instance IsString Ident where
  168   fromString = parseIdent
  169 
  170 isIdent :: String -> Bool
  171 isIdent (c:l) = isAlpha c && all isIdChar l
  172 isIdent [] = False
  173 
  174 -- | Returns true if character can appear in identifier.
  175 isIdChar :: Char -> Bool
  176 isIdChar c = isAlphaNum c || (c == '_') || (c == '\'') || (c == '.')

  180 -- NameInfo
  181 
  182 
  183 -- | Descriptions of the origins of names that may be in scope
  184 data NameInfo
  185   = -- | This name arises from an exported declaration from a module
  186     ModuleIdentifier Ident
  187 
  188   | -- | This name was imported from some other programming language/scope
  189     ImportedName
  190       URI      -- ^ An absolutely-qualified name, which is required to be unique
  191       [Text]   -- ^ A collection of aliases for this name.  Shorter or "less-qualified"
  192                --   aliases should be nearer the front of the list
  193 
  194  deriving (Eq,Ord,Show)
  195 
  196 nameURI :: NameInfo -> URI
  197 nameURI =
  198   \case
  199     ModuleIdentifier i -> moduleIdentToURI i
  200     ImportedName uri _ -> uri
  201 
  202 nameAliases :: NameInfo -> [Text]
  203 nameAliases =
  204   \case
  205     ModuleIdentifier i -> [identBaseName i, identText i]
  206     ImportedName _ aliases -> aliases
  207 
  208 toShortName :: NameInfo -> Text
  209 toShortName (ModuleIdentifier i) = identBaseName i
  210 toShortName (ImportedName uri []) = render uri
  211 toShortName (ImportedName _ (x:_)) = x
  212 
  213 toAbsoluteName :: NameInfo -> Text
  214 toAbsoluteName (ModuleIdentifier i) = identText i
  215 toAbsoluteName (ImportedName uri _) = render uri
  216 
  217 moduleIdentToURI :: Ident -> URI
  218 moduleIdentToURI ident = fromMaybe (panic "moduleIdentToURI" ["Failed to construct ident URI: " <> identText ident]) $
  219   do sch  <- mkScheme "sawcore"
  220      path <- mapM mkPathPiece (identPieces ident)
  221      pure URI
  222        { uriScheme = Just sch
  223        , uriAuthority = Left True -- absolute path
  224        , uriPath   = Just (False, path)
  225        , uriQuery  = []
  226        , uriFragment = Nothing
  227        }

  230 -- External Constants ----------------------------------------------------------
  231 
  232 type VarIndex = Int
  233 
  234 -- | An external constant with a name.
  235 -- Names are not necessarily unique, but the var index should be.
  236 data ExtCns e =
  237   EC
  238   { ecVarIndex :: !VarIndex
  239   , ecName :: !NameInfo
  240   , ecType :: !e
  241   }
  242   deriving (Show, Functor, Foldable, Traversable)
  243 
  244 instance Eq (ExtCns e) where
  245   x == y = ecVarIndex x == ecVarIndex y
  246 
  247 instance Ord (ExtCns e) where
  248   compare x y = compare (ecVarIndex x) (ecVarIndex y)
  249 
  250 instance Hashable (ExtCns e) where
  251   hashWithSalt x ec = hashWithSalt x (ecVarIndex ec)
  252

  254 -- Primitive Names ------------------------------------------------------------
  255 
  256 -- | Names of SAWCore primitives, data types and data type constructors.
  257 data PrimName e =
  258   PrimName
  259   { primVarIndex :: !VarIndex
  260   , primName     :: !Ident
  261   , primType     :: e
  262   }
  263   deriving (Show, Functor, Foldable, Traversable)
  264 
  265 instance Eq (PrimName e) where
  266   x == y = primName x == primName y
  267 
  268 instance Ord (PrimName e) where
  269   compare x y = compare (primName x) (primName y)
  270 
  271 instance Hashable (PrimName e) where
  272   hashWithSalt x pn = hashWithSalt x (primName pn)
  273 
  274 primNameToExtCns :: PrimName e -> ExtCns e
  275 primNameToExtCns (PrimName varIdx nm tp) = EC varIdx (ModuleIdentifier nm) tp
  276 
  277 scFreshNameURI :: Text -> VarIndex -> URI
  278 scFreshNameURI nm i = fromMaybe (panic "scFreshNameURI" ["Failed to construct name URI: <> " <> nm <> "  " <> Text.pack (show i)]) $
  279   do sch <- mkScheme "fresh"
  280      nm' <- mkPathPiece (if Text.null nm then "_" else nm)
  281      i'  <- mkFragment (Text.pack (show i))
  282      pure URI
  283        { uriScheme = Just sch
  284        , uriAuthority = Left False -- relative path
  285        , uriPath   = Just (False, (nm' :| []))
  286        , uriQuery  = []
  287        , uriFragment = Just i'
  288        }

  291 -- Display Name Environments --------------------------------------------------------
  292 
  293 -- | A 'DisplayNameEnv' encodes the mappings between display names (type
  294 -- 'Text') and internal names (type 'VarIndex'). Multiple display
  295 -- names may be associated with each internal name; these are stored
  296 -- in priority order, with preferred display names first.
  297 data DisplayNameEnv =
  298   DisplayNameEnv
  299   { displayNames :: !(IntMap [Text]) -- Keyed by VarIndex; preferred names come first.
  300   , displayIndexes :: !(Map Text IntSet)
  301   }
  302 -- Invariants: The 'displayNames' and 'displayIndexes' maps should be
  303 -- inverses of each other. That is, 'displayNames' maps @i@ to a list
  304 -- containing @x@ if and only if 'displayIndexes' maps @x@ to a set
  305 -- containing @i@.
  306 
  307 emptyDisplayNameEnv :: DisplayNameEnv
  308 emptyDisplayNameEnv = DisplayNameEnv mempty mempty
  309 
  310 -- | Extend a 'DisplayNameEnv' by providing new 'Text' display names
  311 -- for a given 'VarIndex'. Display names should be provided in
  312 -- priority order, with preferred names first. If the 'VarIndex'
  313 -- already exists in the map, then add the new display names with
  314 -- higher priority while keeping the old ones.
  315 extendDisplayNameEnv :: VarIndex -> [Text] -> DisplayNameEnv -> DisplayNameEnv
  316 extendDisplayNameEnv i aliases env =
  317   DisplayNameEnv
  318   { displayNames = IntMap.insertWith List.union i aliases (displayNames env)
  319   , displayIndexes = foldr insertAlias (displayIndexes env) aliases
  320   }
  321   where
  322     insertAlias :: Text -> Map Text IntSet -> Map Text IntSet
  323     insertAlias x m = Map.insertWith IntSet.union x (IntSet.singleton i) m
  324 
  325 -- | Filter display names in a 'DisplayNameEnv' that satisfy a predicate.
  326 filterDisplayNameEnv :: (Text -> Bool) -> DisplayNameEnv -> DisplayNameEnv
  327 filterDisplayNameEnv p env =
  328   DisplayNameEnv
  329   { displayNames = IntMap.filter (not . null) $ fmap (filter p) $ displayNames env
  330   , displayIndexes = Map.filterWithKey (\k _ -> p k) $ displayIndexes env
  331   }
  332 
  333 -- | Merge two 'DisplayNameEnv's, giving higher priority to display
  334 -- names from the first argument.
  335 mergeDisplayNameEnv :: DisplayNameEnv -> DisplayNameEnv -> DisplayNameEnv
  336 mergeDisplayNameEnv env1 env2 =
  337   DisplayNameEnv
  338   { displayNames = IntMap.unionWith List.union (displayNames env1) (displayNames env2)
  339   , displayIndexes = Map.unionWith IntSet.union (displayIndexes env1) (displayIndexes env2)
  340   }
  341 
  342 -- | Look up a 'Text' display name in an environment, returning the
  343 -- list of indexes it could resolve to.
  344 resolveDisplayName :: DisplayNameEnv -> Text -> [VarIndex]
  345 resolveDisplayName env x =
  346   case Map.lookup x (displayIndexes env) of
  347     Nothing -> []
  348     Just s -> IntSet.elems s
  349 
  350 -- | Return the first display name for the given 'VarIndex' that is
  351 -- unambiguous in the naming environment. If there is no unambiguous
  352 -- alias, then return 'Nothing'.
  353 bestDisplayName :: DisplayNameEnv -> VarIndex -> Maybe Text
  354 bestDisplayName env i =
  355   do aliases <- IntMap.lookup i (displayNames env)
  356      go aliases
  357   where
  358     go [] = Nothing
  359     go (x : xs) =
  360       case Map.lookup x (displayIndexes env) of
  361         Nothing -> panic "bestDisplayName" ["Invariant violated: Missing key " <> x]
  362         Just vs
  363           | IntSet.size vs == 1 -> Just x
  364           | otherwise -> go xs
