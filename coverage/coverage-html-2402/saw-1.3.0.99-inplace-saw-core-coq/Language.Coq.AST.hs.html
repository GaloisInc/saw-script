   10 module Language.Coq.AST where
   11 
   12 type Ident = String
   13 
   14 data Sort
   15   = Prop
   16   | Set
   17   | Type
   18   deriving (Show)
   19 
   20 data Term
   21   = Lambda [Binder] Term
   22   | Fix Ident [Binder] Term Term
   23   | Pi [PiBinder] Term
   24   | Let Ident [Binder] (Maybe Type) Term Term
   25   | If Term Term Term
   26   | App Term [Term]
   27   | Sort Sort
   28   | Var Ident
   29     -- | A variable that needs to be printed with a leading at sign in order to
   30     -- make all arguments explicit
   31   | ExplVar Ident
   32     -- | A ascription @tm : tp@ of a type to a term
   33   | Ascription Term Term
   34   | NatLit Integer
   35   | ZLit Integer
   36   | List [Term]
   37   | StringLit String
   38   | Scope Term String
   39   | Ltac String
   40   deriving (Show)
   41 
   42 -- | Type synonym useful for indicating when a term is used as a type.
   43 type Type = Term
   44 
   45 -- | An 'Ident' with an optional 'Type', which may be explicit or implicit.
   46 -- For use representing the bound variables in 'Lambda's, 'Let's, etc.
   47 data Binder
   48   = Binder Ident (Maybe Type)
   49   | ImplicitBinder Ident (Maybe Type)
   50     deriving (Show)
   51 
   52 -- | An 'Type' with an optional 'Ident', which may be explicit or implicit.
   53 -- For use representing arguments in 'Pi' types.
   54 data PiBinder
   55   = PiBinder (Maybe Ident) Type
   56   | PiImplicitBinder (Maybe Ident) Type
   57     deriving (Show)
   58 
   59 -- Because saw-core does not give very helpful access to the parameters and
   60 -- indices, we just follow their style and define the constructor by its fully
   61 -- applied return type.
   62 data Constructor = Constructor
   63   { constructorName    :: Ident
   64   , constructorType    :: Term
   65   }
   66   deriving (Show)
   67 
   68 data Inductive = Inductive
   69   { inductiveName         :: Ident
   70   , inductiveParameters   :: [Binder]
   71   , inductiveIndices      :: [PiBinder]
   72   , inductiveSort         :: Sort
   73   , inductiveConstructors :: [Constructor]
   74   }
   75   deriving (Show)
   76 
   77 data Decl
   78   = Axiom Ident Type
   79   | Comment String
   80   | Definition Ident [Binder] (Maybe Type) Term
   81   | Parameter Ident Type
   82   | Variable Ident Type
   83   | InductiveDecl Inductive
   84   | Section Ident [Decl]
   85   | Snippet String
   86   deriving (Show)
