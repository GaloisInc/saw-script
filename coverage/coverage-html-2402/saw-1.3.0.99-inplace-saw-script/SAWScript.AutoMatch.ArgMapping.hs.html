   27 -- | A bidirectional mapping representing what arguments have what types at what positions in a declaration
   28 --   Abstract, because we preserve the invariant that the maps contained are coherent.
   29 data ArgMapping =
   30    ArgMapping { typeBins_ :: Map Type (Set (Name, Int))
   31               , nameLocs_ :: Map Name (Int, Type) }
   32               deriving (Show)
   33 
   34 typeBins :: ArgMapping -> Map Type (Set (Name, Int))
   35 typeBins = typeBins_
   36 
   37 nameLocs :: ArgMapping -> Map Name (Int, Type)
   38 nameLocs = nameLocs_
   39 
   40 -- | Make an argMapping from a list of Args
   41 makeArgMapping :: [Arg] -> ArgMapping
   42 makeArgMapping =
   43    ArgMapping <$> foldr (uncurry (\k -> Map.insertWith Set.union k . Set.singleton)) Map.empty
   44                   . map (\(i, a) -> (argType a, (argName a, i)))
   45                   . zip [0..]
   46               <*> foldr (uncurry Map.insert) Map.empty
   47                   . map (\(i, a) -> (argName a, (i, argType a)))
   48                   . zip [0..]
   49 
   50 -- | Remove a name from an ArgMapping
   51 removeName :: Name -> ArgMapping -> ArgMapping
   52 removeName name mapping =
   53    fromMaybe mapping $ do
   54       (nameIndex, nameType) <- Map.lookup name (nameLocs mapping)
   55       return $ ArgMapping
   56          (deleteFromSetMap nameType (name, nameIndex) (typeBins mapping))
   57          (Map.delete name (nameLocs mapping))
   58 
   59 -- | Look up a name's type and position in the original declaration
   60 lookupName :: Name -> ArgMapping -> Maybe (Int, Type)
   61 lookupName name (ArgMapping _ nls) = Map.lookup name nls
   62 
   63 -- | Look up all the names of a given type and their indices
   64 lookupType :: Type -> ArgMapping -> Maybe (Set (Name, Int))
   65 lookupType typ (ArgMapping tbs _) = Map.lookup typ tbs
   66 
   67 -- | The empty ArgMapping
   68 emptyArgMapping :: ArgMapping
   69 emptyArgMapping = makeArgMapping []
   70 
   71 -- | Test if an ArgMapping is empty
   72 isEmptyArgMapping :: ArgMapping -> Bool
   73 isEmptyArgMapping (ArgMapping t n) =
   74   Map.null t && Map.null n
