   28 -- | Given an 'SBV.Solver' from @SBV@, attempt to query the solver for its
   29 -- version and return the result as a string.
   30 -- Adapted from @what4/test/ProbeSolvers.hs@
   31 getSolverVersion :: SBV.Solver -> IO (Maybe String)
   32 getSolverVersion s =
   33   let s' = SBV.solver $ SBV.defaultSolverConfig s
   34       nope what = panic "getSolverVersion" [
   35              "Version request for unsupported solver " <> what
   36            ]
   37 
   38       (args, pref) = case SBV.name s' of
   39         -- n.b. abc will return a non-zero exit code if asked for command usage.
   40         SBV.ABC       -> (["s", "-q", "version;quit"], "UC Berkeley, ABC ")
   41         SBV.Boolector -> (["--version"]              , "")
   42         SBV.Bitwuzla  -> (["--version"]              , "")
   43         SBV.CVC4      -> nope "CVC4"
   44         SBV.CVC5      -> (["--version"]              , "This is cvc5 version ")
   45         SBV.DReal     -> (["--version"]              , "dReal v")
   46         SBV.MathSAT   -> (["-version"]               , "MathSAT5 version ")
   47         SBV.OpenSMT   -> nope "OpenSMT"
   48         SBV.Yices     -> (["--version"]              , "Yices ")
   49         SBV.Z3        -> (["--version"]              , "Z3 version ")
   50   in
   51   try (readProcessWithExitCode (SBV.executable s') args "") >>= \case
   52     Right (ExitSuccess,o,_) | (l:_) <- lines o ->
   53       return $ Just $ dropPrefix pref l
   54     Right _                   -> return Nothing
   55     Left (_ :: SomeException) -> return Nothing
   56   where dropPrefix (x:xs) (y:ys) | x == y = dropPrefix xs ys
   57         dropPrefix _ ys = ys
   58 
   59 -- | Get the 'SolverBackendVersion' of a 'SolverBackend'
   60 getSolverBackendVersion :: SolverBackend -> IO (Maybe String)
   61 getSolverBackendVersion backend = case backend of
   62   What4     -> return what4Hash
   63   SBV       -> return (Just VERSION_sbv)
   64   AIG       -> return aigHash
   65   RME       -> return rmeHash
   66   -- We use individual cases for the remaining constructors to ensure that if
   67   -- a new backend is added, a warning is generated for this pattern match
   68   ABC       -> getSolverVersion SBV.ABC
   69   Bitwuzla  -> getSolverVersion SBV.Bitwuzla
   70   Boolector -> getSolverVersion SBV.Boolector
   71   CVC4      -> getSolverVersion SBV.CVC4
   72   CVC5      -> getSolverVersion SBV.CVC5
   73   DReal     -> getSolverVersion SBV.DReal
   74   MathSAT   -> getSolverVersion SBV.MathSAT
   75   OpenSMT   -> getSolverVersion SBV.OpenSMT
   76   Yices     -> getSolverVersion SBV.Yices
   77   Z3        -> getSolverVersion SBV.Z3
   78 
   79 -- | Get the set of 'SolverBackendVersions' of a list of 'SolverBackend's
   80 getSolverBackendVersions :: [SolverBackend] -> IO SolverBackendVersions
   81 getSolverBackendVersions backends = Map.fromList <$>
   82   mapM (\backend -> (backend,) <$> getSolverBackendVersion backend) backends
