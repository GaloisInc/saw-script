    8 module Heapster.NamedMb where

   10 import Data.Binding.Hobbits
   11 import Data.Binding.Hobbits.MonadBind
   12 import Data.Type.RList
   13 import Control.Lens
   14 
   15 -- | A constant type functor for 'String's
   16 newtype StringF a = StringF { unStringF :: String }
   17 
   18 mkNuMatching [t| forall a. StringF a |]
   19 
   20 -- | An 'Mb' multi-binding where each bound 'Name' has an associated 'String'
   21 -- for parsing and printing it
   22 data NamedMb ctx a = NamedMb
   23   { _mbNames :: RAssign StringF ctx
   24   , _mbBinding :: Mb ctx a
   25   }
   26   deriving Functor
   27 
   28 -- | A 'Binding' of a single 'Name' with a 'String'
   29 type NamedBinding c = NamedMb (RNil :> c)
   30 
   31 instance Liftable (StringF a) where
   32     mbLift (mbMatch -> [nuMP| StringF x |]) = StringF (mbLift x)
   33 
   34 instance LiftableAny1 StringF where
   35     mbLiftAny1 = mbLift
   36 
   37 mkNuMatching [t| forall ctx a. NuMatching a => NamedMb ctx a |]
   38 
   39 -- | Apply a binary function to the body of a 'NamedMb'; similar to 'mbMap2'
   40 mbMap2Named :: (a -> b -> c) -> NamedMb ctx a -> NamedMb ctx b -> NamedMb ctx c
   41 mbMap2Named f mb1 mb2 =
   42   NamedMb (_mbNames mb1) (mbMap2 f (_mbBinding mb1) (_mbBinding mb2))
   43 
   44 -- | A 'Lens' to get the binding ouf of a 'NamedMb'
   45 mbBinding :: Lens (NamedMb ctx a) (NamedMb ctx b) (Mb ctx a) (Mb ctx b)
   46 mbBinding f x = NamedMb (_mbNames x) <$> f (_mbBinding x)
   47 
   48 -- | Build a 'NamedMb' that binds multiple 'Name's with the given 'String's
   49 nuMultiNamed :: RAssign StringF ctx -> (RAssign Name ctx -> b) -> NamedMb ctx b
   50 nuMultiNamed tps f = NamedMb
   51   { _mbNames = tps
   52   , _mbBinding = nuMulti (mapRAssign (const Proxy) tps) f
   53   }
   54 
   55 -- | A version of 'nuMultiWithElim1' for 'NamedMb'
   56 nuMultiWithElim1Named :: (RAssign Name ctx -> arg -> b) ->
   57                          NamedMb ctx arg -> NamedMb ctx b
   58 nuMultiWithElim1Named k = over mbBinding (nuMultiWithElim1 k)
   59 
   60 -- | Commute a 'NamedMb' inside a strong binding monad
   61 strongMbMNamed :: MonadStrongBind m => NamedMb ctx (m a) -> m (NamedMb ctx a)
   62 strongMbMNamed = traverseOf mbBinding strongMbM
   63 
   64 -- | Commute a 'NamedMb' inside a binding monad
   65 mbMNamed :: (MonadBind m, NuMatching a) => NamedMb ctx (m a) -> m (NamedMb ctx a)
   66 mbMNamed = traverseOf mbBinding mbM
   67 
   68 -- | Swap the order of two nested named bindings
   69 mbSwapNamed :: RAssign Proxy ctx -> NamedMb ctx' (NamedMb ctx a) ->
   70                NamedMb ctx (NamedMb ctx' a)
   71 mbSwapNamed p (NamedMb names' body') =
   72     NamedMb
   73     { _mbNames = mbLift (_mbNames <$> body')
   74     , _mbBinding = NamedMb names' <$> mbSwap p (_mbBinding <$> body')
   75     }
   76 
   77 -- | Swap the order of a binding with 'String' names with one without
   78 mbSink :: RAssign Proxy ctx -> Mb ctx' (NamedMb ctx a) -> NamedMb ctx (Mb ctx' a)
   79 mbSink p m =
   80     NamedMb
   81     { _mbNames = mbLift (_mbNames <$> m)
   82     , _mbBinding = mbSwap p (_mbBinding <$> m)
   83     }
   84 
   85 -- | Lift a 'Liftable' value out of a 'NamedMb'
   86 mbLiftNamed :: Liftable a => NamedMb ctx a -> a
   87 mbLiftNamed = views mbBinding mbLift
   88 
   89 -- | Eliminate a 'NamedMb' that binds zero names
   90 elimEmptyNamedMb :: NamedMb RNil a -> a
   91 elimEmptyNamedMb = views mbBinding elimEmptyMb
   92 
   93 -- | Create a 'NamedMb' that binds zero names
   94 emptyNamedMb :: a -> NamedMb RNil a
   95 emptyNamedMb = NamedMb MNil . emptyMb
