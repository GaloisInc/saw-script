   64 -- | Like zipWithM_ on two lists, but on maps by aligning keys.
   65 --   This could also perhaps be called intersectionWithM, but
   66 --   since it doesn't actually compute a resultant map that seems
   67 --   misleading.
   68 zipByKeyWithM_ :: (Ord k, Monad m) =>
   69       (a -> b -> m ()) -> Map k a -> Map k b ->
   70       m ()
   71 zipByKeyWithM_ f xs ys =
   72     let xys = Map.intersectionWith (\x y -> (x, y)) xs ys in
   73     mapM_ (\(x, y) -> f x y) (Map.elems xys)

   75 -- | Drop a list of keys from xs.
   76 dropKeys :: Ord k => [k] -> Map k v -> Map k v
   77 dropKeys keys xs =
   78     foldr Map.delete xs keys

   82 -- UnifyVars
   83 
   84 --
   85 -- unifyVars is a type-class-polymorphic function for extracting
   86 -- unification vars from a type or type schema. It returns a set of
   87 -- TypeIndex (TypeIndex is just Integer) manifested as a map from
   88 -- those TypeIndexes to their positions/provenance.
   89 --
   90 
   91 class UnifyVars t where
   92     unifyVars :: t -> Map TypeIndex Pos
   93 
   94 instance (Ord k, UnifyVars a) => UnifyVars (Map k a) where
   95     unifyVars = unifyVars . Map.elems
   96 
   97 instance (UnifyVars a) => UnifyVars [a] where
   98     unifyVars = Map.unionsWith Pos.choosePos . map unifyVars
   99 
  100 instance (UnifyVars a) => UnifyVars (PrimitiveLifecycle, a) where
  101     unifyVars (_lc, t) = unifyVars t
  102 
  103 instance (UnifyVars a) => UnifyVars (Pos, PrimitiveLifecycle, Rebindable, a) where
  104     unifyVars (_pos, _lc, _rb, t) = unifyVars t
  105 
  106 instance UnifyVars Type where
  107     unifyVars t = case t of
  108         TyCon _ _ ts      -> unifyVars ts
  109         TyFunc _ _ params namedParams ret ->
  110             let paramsVars = unifyVars params
  111                 namedVars = unifyVars namedParams
  112                 retVars = unifyVars ret
  113             in
  114             let vars1 = Map.unionWith Pos.choosePos paramsVars namedVars in
  115             Map.unionWith Pos.choosePos vars1 retVars
  116         TyRecord _ tm     -> unifyVars tm
  117         TyVar _ _         -> Map.empty
  118         TyUnifyVar pos i  -> Map.singleton i pos
  119 
  120 instance UnifyVars Schema where
  121     unifyVars (Forall _ t) = unifyVars t
  122 
  123 instance UnifyVars NamedType where
  124     unifyVars nt = case nt of
  125         ConcreteType ty -> unifyVars ty
  126         AbstractType _kind -> Map.empty

  130 -- Substitutions
  131 
  132 -- | Subst is the type of a substitution map for unification vars.
  133 --
  134 --   For the substitution we accumulate in the `TI` context while
  135 --   working, we maintain the following invariant:
  136 --
  137 --      If one unification var @b@ points to another unification
  138 --      var @a@, @a < b@. That is, entries in the map that point
  139 --      to other entries always point downwards. This prevents
  140 --      generating cycles.
  141 --
  142 --   This does not apply to the smaller transient substitutions
  143 --   we use in `generalize`, whose entries always map unification
  144 --   vars to named vars.
  145 --
  146 type Subst = Map TypeIndex Type
  147 
  148 --
  149 -- appSubst is a type-class-polymorphic function for applying a
  150 -- substitution (of numbered unification vars) to AST elements.
  151 --
  152 
  153 class AppSubst t where
  154     appSubst :: Subst -> t -> t
  155 
  156 instance (AppSubst t) => AppSubst (Maybe t) where
  157     appSubst s = fmap $ appSubst s
  158 
  159 instance (AppSubst t) => AppSubst [t] where
  160     appSubst s = map $ appSubst s
  161 
  162 instance (AppSubst t1, AppSubst t2) => AppSubst (Pos, (Pos, t1, t2)) where
  163     appSubst s (pos1, (pos2, a, b)) = (pos1, (pos2, appSubst s a, appSubst s b))
  164 
  165 instance (AppSubst t) => AppSubst (Maybe (Pos, Text), t) where
  166     appSubst s (mbName, e) = (mbName, appSubst s e)
  167 
  168 instance (AppSubst t) => AppSubst (PrimitiveLifecycle, t) where
  169     appSubst s (lc, x) = (lc, appSubst s x)
  170 
  171 instance (AppSubst t) => AppSubst (Pos, PrimitiveLifecycle, Rebindable, t) where
  172     appSubst s (pos, lc, rb, x) = (pos, lc, rb, appSubst s x)
  173 
  174 instance (Ord k, AppSubst a) => AppSubst (Map k a) where
  175     appSubst s = fmap (appSubst s)
  176 
  177 instance AppSubst Expr where
  178     appSubst s expr = case expr of
  179         TSig pos e t           -> TSig pos (appSubst s e) (appSubst s t)
  180         Bool _ _               -> expr
  181         String _ _             -> expr
  182         Int _ _                -> expr
  183         Code _ _               -> expr
  184         CType _ _              -> expr
  185         Array pos es           -> Array pos (appSubst s es)
  186         Block pos (bs, e)      -> Block pos (appSubst s bs, appSubst s e)
  187         Tuple pos es           -> Tuple pos (appSubst s es)
  188         Record pos fs          -> Record pos (appSubst s fs)
  189         Index pos ar ix        -> Index pos (appSubst s ar) (appSubst s ix)
  190         Lookup pos rec fld     -> Lookup pos (appSubst s rec) fld
  191         TLookup pos tpl idx    -> TLookup pos (appSubst s tpl) idx
  192         Var _pos _x            -> expr
  193         Lambda pos mname pats ppats body ->
  194             let pats' = map (appSubst s) pats
  195                 ppats' = Map.map (appSubst s) ppats
  196                 body' = appSubst s body
  197             in
  198             Lambda pos mname pats' ppats' body'
  199         Application pos f v    ->
  200             Application pos (appSubst s f) (appSubst s v)
  201         Let pos dg e           ->
  202             Let pos (appSubst s dg) (appSubst s e)
  203         IfThenElse pos e e2 e3 ->
  204             IfThenElse pos (appSubst s e) (appSubst s e2) (appSubst s e3)
  205 
  206 instance AppSubst Pattern where
  207     appSubst s pat = case pat of
  208         PWild pos mt  -> PWild pos (appSubst s mt)
  209         PVar allpos xpos x mt -> PVar allpos xpos x (appSubst s mt)
  210         PTuple pos ps -> PTuple pos (appSubst s ps)
  211 
  212 instance AppSubst Stmt where
  213     appSubst s bst = case bst of
  214         StmtBind pos pat e       -> StmtBind pos (appSubst s pat) (appSubst s e)
  215         StmtLet pos rb dg        -> StmtLet pos rb (appSubst s dg)
  216         StmtCode allpos spos str -> StmtCode allpos spos str
  217         StmtImport pos imp       -> StmtImport pos imp
  218         StmtInclude pos file once -> StmtInclude pos file once
  219         StmtTypedef allpos apos a ty ->
  220             StmtTypedef allpos apos a (appSubst s ty)
  221         StmtPushdir pos dir      -> StmtPushdir pos dir
  222         StmtPopdir pos           -> StmtPopdir pos
  223 
  224 instance AppSubst DeclGroup where
  225     appSubst s dg = case dg of
  226         Recursive ds -> Recursive (appSubst s ds)
  227         NonRecursive d -> NonRecursive (appSubst s d)
  228 
  229 instance AppSubst Decl where
  230     appSubst s (Decl pos p mt e) =
  231         Decl pos (appSubst s p) (appSubst s mt) (appSubst s e)
  232 
  233 instance AppSubst Type where
  234     appSubst s t = case t of
  235         TyCon pos tc ts -> TyCon pos tc (appSubst s ts)
  236         TyFunc pos ninfo params namedParams ret ->
  237             let params' = appSubst s params
  238                 namedParams' = appSubst s namedParams
  239                 ret' = appSubst s ret
  240             in
  241             TyFunc pos ninfo params' namedParams' ret'
  242         TyRecord pos fs -> TyRecord pos (appSubst s fs)
  243         TyVar _ _  -> t
  244         TyUnifyVar _ i -> case Map.lookup i s of
  245             Nothing -> t
  246             Just t' ->
  247                 -- Apply any further substitutions needed in the
  248                 -- expansion of i.
  249                 --
  250                 -- This will always terminate because the expansion of
  251                 -- i cannot refer to i; the correctness arguments for
  252                 -- checkOccurs ensure that. They do in turn depend on
  253                 -- full expansion of existing substitutions, which
  254                 -- in general requires this call.
  255                 appSubst s t'
  256 
  257 instance AppSubst Schema where
  258     appSubst s (Forall ns t) =
  259         Forall ns (appSubst s t)
  260 
  261 instance AppSubst NamedType where
  262     appSubst s nt = case nt of
  263         ConcreteType ty -> ConcreteType $ appSubst s ty
  264         AbstractType kind -> AbstractType kind

  268 -- Pass context / monad
  269 
  270 -- | The monad for this pass is "TI", which is composed of a read-only
  271 --   part plus a read-write part that accumulates as we move through the
  272 --   code.
  273 --
  274 --   Note that, for the time being at least, this monad cannot be in
  275 --   IO, because we need to be able to use the typechecker in a pure
  276 --   context when constructing the builtins table during startup. If
  277 --   that becomes a prohibitive problem, we can rearrange the builtins
  278 --   table construction, but that will be expensive.
  279 --
  280 newtype TI a = TI { unTI :: ReaderT RO (StateT RW Identity) a }
  281     deriving (Functor, Applicative, Monad, MonadReader RO, MonadState RW)
  282 
  283 -- | The read-only portion
  284 data RO = RO {
  285     -- | The variable availability (lifecycle set)
  286     tiPrimsAvail :: Set PrimitiveLifecycle,
  287 
  288     -- | The prettyprinter options
  289     tiPPOpts :: PPS.Opts
  290 }
  291 
  292 -- | The read-write portion
  293 data RW = RW {
  294     -- | The variable typing environment (variable name to type scheme)
  295     tiVarEnv :: VarEnv,
  296 
  297     -- | The type environment: named type variables, which are either
  298     --   typedefs (map to ConcreteType) or abstract types (AbstractType)
  299     tiTyEnv :: TyEnv,
  300 
  301     -- | The next fresh unification var number
  302     tiNextTypeIndex :: TypeIndex,
  303 
  304     -- | The unification var substitution we're accumulating
  305     tiSubst :: Subst,
  306 
  307     -- | Any type errors and warnings we've generated so far
  308     --   These accumulate in reverse order; later messages are consed
  309     --   on the head of the list.
  310     tiErrors :: [(Pos, PPS.Doc)],
  311     tiWarnings :: [(Pos, PPS.Doc)]
  312 }
  313 
  314 -- | The result of a `TI` typechecker computation is either a list of
  315 --   errors or a result value, along with (always) a list of warnings.
  316 --   FUTURE: it would be better to preserve the ordering of warnings
  317 --   and errors in a single list...
  318 --
  319 type MsgList = [(Pos, PPS.Doc)]
  320 type Result a = (Either MsgList a, MsgList)
  321 
  322 -- | Run the TI monad.
  323 --
  324 --   Note that we don't return the updated environments! This is a
  325 --   manifestation of the unhealthy codependent relationship with the
  326 --   interpreter.
  327 --
  328 runTI :: PPS.Opts -> Set PrimitiveLifecycle -> VarEnv -> TyEnv -> TI a ->
  329          Result a
  330 runTI ppopts avail varenv tyenv m =
  331     let rw = RW {
  332             tiVarEnv = varenv,
  333             tiTyEnv = tyenv,
  334             tiNextTypeIndex = 0,
  335             tiSubst = Map.empty,
  336             tiErrors = [],
  337             tiWarnings = []
  338         }
  339         ro = RO {
  340             tiPrimsAvail = avail,
  341             tiPPOpts = ppopts
  342         }
  343         (result, rw') = runState (runReaderT (unTI m) ro) rw
  344         errs = reverse $ tiErrors rw'
  345         warns = reverse $ tiWarnings rw'
  346     in
  347     -- We succeed if and only if the error list is empty.
  348     case errs of
  349         [] -> (Right result, warns)
  350         _ -> (Left errs, warns)
  351 
  352 -- | Run a speculative sub-action in the `TI` monad. Throw away
  353 --   whatever it does, and return `True` if it succeeds (does
  354 --   not generate new errors).
  355 speculateTI :: TI a -> TI Bool
  356 speculateTI m = do
  357     rw <- get
  358     ro <- ask
  359     -- Micro-optimization: speculate with an empty error list, so we
  360     -- don't need to iterate over any existing errors afterwards to
  361     -- check for success.
  362     let (_result, rw') = runState (runReaderT (unTI m) ro) (rw {tiErrors = []})
  363     pure $ null (tiErrors rw')

  369 -- | Enter a scope
  370 pushScope :: TI ()
  371 pushScope = do
  372     varenv <- gets tiVarEnv
  373     tyenv <- gets tiTyEnv
  374     let varenv' = ScopedMap.push varenv
  375         tyenv' = ScopedMap.push tyenv
  376     modify (\rw -> rw { tiVarEnv = varenv', tiTyEnv = tyenv' })
  377 
  378 -- | Leave a scope
  379 popScope :: TI ()
  380 popScope = do
  381     varenv <- gets tiVarEnv
  382     tyenv <- gets tiTyEnv
  383     let varenv' = ScopedMap.pop varenv
  384         tyenv' = ScopedMap.pop tyenv
  385     modify (\rw -> rw { tiVarEnv = varenv', tiTyEnv = tyenv' })

  387 -- | Get a fresh unification var number.
  388 getFreshTypeIndex :: TI TypeIndex
  389 getFreshTypeIndex = do
  390     next <- gets tiNextTypeIndex
  391     modify $ (\rw -> rw { tiNextTypeIndex = next + 1 })
  392     return next

  394 -- | Construct a fresh type variable.
  395 --
  396 --   Collect the position that prompted us to make it; for example, if
  397 --   we're the element type of an empty list we get the position of the
  398 --   []. We haven't inferred anything, so use the InfFresh position.
  399 --   This will cause the position of anything more substantive that gets
  400 --   unified with it to be preferred. If no such thing happens though
  401 --   this will be the position that gets attached to the quantifier
  402 --   binding in generalize.
  403 getFreshTyVar :: Pos -> TI Type
  404 getFreshTyVar pos = TyUnifyVar (PosInferred InfFresh pos) <$> getFreshTypeIndex
  405

  408 --   variables, but I've split it out in case we want to distinguish
  409 --   it in the future.
  410 getErrorTyVar :: Pos -> TI Type

  437 -- | Apply the current typedef collection with `Util.substituteTyVars`.
  438 --
  439 --   The type t has already been checked, so it's ok to panic if it
  440 --   refers to something in the typedef collection that's not visible.
  441 resolveCurrentTypedefs :: Type -> TI Type
  442 resolveCurrentTypedefs t = do
  443     avail <- asks tiPrimsAvail
  444     s <- gets tiTyEnv
  445     return $ Util.substituteTyVars avail s t

  447 -- | Condense functions that return functions to a single function
  448 --   type with more parameters. The passed-in position is used for
  449 --   error reporting and should be the position of the expression
  450 --   or other object whose type we're working on. (The position of
  451 --   the type itself is not suitable; positions of types are meant
  452 --   as further annotations to errors.)
  453 condenseFunctions :: Pos -> Type -> TI Type
  454 condenseFunctions errPos ty = case ty of
  455     TyFunc pos1 names1 params1 namedParams1 ret1 ->
  456         case ret1 of
  457             TyFunc _pos2 names2 params2 namedParams2 ret2 ->
  458                 let dups = Map.intersection namedParams1 namedParams2 in
  459                 case null dups of
  460                     True -> do
  461                         -- FUTURE: we could/should splice pos1 and
  462                         -- pos2... but we don't have a way to
  463                         -- represent disjoint position spans, so it's
  464                         -- only a good idea if they're "near" each
  465                         -- other. But they won't necessarily be, and
  466                         -- we have no way to figure that for the time
  467                         -- being.
  468                         let pos = pos1
  469                             names = names1 <> names2
  470                             params = params1 ++ params2
  471                             namedParams = Map.union namedParams1 namedParams2
  472                             ret = ret2
  473                             ty' = TyFunc pos names params namedParams ret
  474                         -- Try again in case there's more functions hiding
  475                         condenseFunctions errPos ty'
  476                     False -> do
  477                         let dups' = PP.hsep $ map PP.pretty $ Map.keys dups
  478                         recordError errPos $ "Function has duplicate" <+>
  479                                              "parameter names:" <+> dups'
  480                         getErrorTyVar pos1
  481             _ ->
  482                 pure ty
  483     _ ->
  484         pure ty
  485 
  486 -- | Apply both the current substitution and the current typedef
  487 --   collection. Then if we get a function type, condense functions
  488 --   that return functions to a single function type with more
  489 --   parameters.
  490 --
  491 --   The position should be the position of the AST construct whose
  492 --   type we're handling, and is used for error reporting.
  493 expandFully :: Pos -> Type -> TI Type
  494 expandFully pos t = do
  495     t' <- resolveCurrentTypedefs t
  496     t'' <- applyCurrentSubst t'
  497     condenseFunctions pos t''

  503 -- | Get the unification vars that are used in the current variable typing
  504 --   and named type environments.
  505 --
  506 -- FIXME: This function may miss type variables that occur in the type
  507 -- of a binding that has been shadowed by another value with the same
  508 -- name. This could potentially cause a run-time type error if the
  509 -- type of a local function gets generalized too much. We can probably
  510 -- wait to fix it until someone finds a sawscript program that breaks.
  511 --
  512 -- dholland 20241220: I don't think that's a problem. If there's a
  513 -- loose unification var somewhere that's been shadowed to the point
  514 -- where it's not accessible, we can't have accessed it in order to
  515 -- generate a reference to it in the current block. If it is somewhere
  516 -- accessible, we'll find it there. This might have broken in the past
  517 -- when it didn't search the named type environment, but that leak has
  518 -- been corrected.
  519 --
  520 -- dholland 20251120: If it does turn out to be a problem, we can now
  521 -- also get shadowed values out of the environment by adding a suitable
  522 -- variant of @elems@ to `ScopedMap`.
  523 --
  524 -- Note that we apply the current substitution first. This means that
  525 -- the caller must also apply the current substitution before reasoning
  526 -- about what unification vars do and don't appear.
  527 --
  528 -- Returns a map of the index number to the occurrence position.
  529 unifyVarsInEnvs :: TI (Map TypeIndex Pos)
  530 unifyVarsInEnvs = do
  531     venv <- gets tiVarEnv
  532     tenv <- gets tiTyEnv
  533     vtys <- mapM applyCurrentSubst $ ScopedMap.allElems venv
  534     ttys <- mapM applyCurrentSubst $ ScopedMap.allElems tenv
  535     return $ Map.unionWith Pos.choosePos (unifyVars vtys) (unifyVars ttys)

  544 -- | Get all the bindings in a pattern.
  545 patternBindings :: Pattern -> [(Name, Pos, Maybe Type)]
  546 patternBindings pat = case pat of
  547     PWild _ _mt -> []
  548     PVar _ xpos x mt -> [(x, xpos, mt)]
  549     PTuple _ ps -> concatMap patternBindings ps
  550 
  551 -- | Get all the bindings in a pattern, using a separate passed-in
  552 --   schema to get the types. Ignore the types in the pattern.
  553 --
  554 -- XXX: is that reasonable? Should probably assert that the schema
  555 -- matches the types in the pattern, unless the pattern hasn't already
  556 -- been checked yet, and that seems like it would be a bug.
  557 --
  558 -- Note that if the pattern is a tuple and the schema is not a tuple
  559 -- type, we return nothing. Presumably in this case a type error has
  560 -- already been generated and we don't need another one? But it would
  561 -- probably be a good idea to check up on that. XXX
  562 --
  563 -- Alternatively if the pattern has had its types filled in, this
  564 -- should not be different from the plain patternBindings and should
  565 -- probably just be removed.
  566 --
  567 patternBindingsWithSchema :: Pattern -> Schema -> [(Name, Pos, Schema)]
  568 patternBindingsWithSchema pat sch = case pat of
  569     PWild _ _ -> []
  570     PVar _ xpos x _ -> [(x, xpos, sch)]
  571     PTuple _ ps ->
  572       case sch of
  573         Forall vs t -> case t of
  574             TyCon _pos (TupleCon _) ts' ->
  575                 let once pat' t' =
  576                       patternBindingsWithSchema pat' (Forall vs t')
  577                 in
  578                 concat $ zipWith once ps ts'
  579             _ -> []

  585 --
  586 -- Error reporting.
  587 --
  588 -- When we find a mismatch, we have potentially recursed arbitrarily
  589 -- deeply into the original type. We need to print the specific types
  590 -- we trip on (this is important if they are e.g. elements in a large
  591 -- system of nested records and typles) but we also want to print the
  592 -- rest of the original context as well.
  593 --
  594 -- Therefore, as we recurse into the type we accumulate a list of the
  595 -- expected/found pairs that we have seen; and when we get an error,
  596 -- we print all of them along with the rest of the error message.
  597 --
  598 -- Empty strings are inserted between pairs to make the output more
  599 -- readable.
  600 --
  601 -- Further notes on the message formatting:
  602 --
  603 -- Print the expected and found types on their own lines. They can be
  604 -- large; if they are the resulting lines can still be fairly
  605 -- illegible, but at least the user doesn't have to hunt for "found"
  606 -- in the middle of a multi-line print.

  608 -- Pad the prefix of the prints so that the types line up; this is
  609 -- helpful for longer types that still fit on one output line.
  610 --
  611 -- We'll indent each line with four spaces. What we send back gets
  612 -- printed underneath a message that's already (at least in some
  613 -- cases) indented by two spaces. It's important to make it clear that
  614 -- all the stuff we generate is part of that message and not, for
  615 -- example, an additional separate error. The indenting happens below.

  617 -- I keep going back and forth on whether the larger types should be

  622 -- | Print a list of enclosing types.
  623 prettyEnclosing :: PPS.Opts -> [(Type, Type)] -> PPS.Doc
  624 prettyEnclosing ppopts tys =
  625     let once (tyexp, tyfound) =
  626           let tyexp' = prettyType ppopts tyexp
  627               tyfound' = prettyType ppopts tyfound
  628               expectedShort  = "Expected:" <+> tyexp'
  629               foundShort     = "Found:   " <+> tyfound'
  630               -- Use of nest here allows the open-brace of records to
  631               -- go on the same line, which ends up looking better
  632               expectedLong   = "Expected:" <+> PP.nest 3 tyexp'
  633               foundLong      = "Found:" <+> PP.nest 3 tyfound'
  634               expected = PP.group $ PP.flatAlt expectedLong expectedShort
  635               found = PP.group $ PP.flatAlt foundLong foundShort
  636           in
  637           expected <> PP.hardline <> found <> PP.hardline <> ""
  638     in
  639     PP.vsep $ map once tys

  641 -- | Print details of a type. This prints the provenance info
  642 --   we carry in type positions.
  643 prettyTypeDetails :: PPS.Opts -> Type -> (Pos, PPS.Doc)
  644 prettyTypeDetails ppopts ty =
  645     let (pos, what) = case Pos.getPos ty of
  646            PosInferred InfFresh p ->
  647                (p, "a fresh type variable introduced here")
  648            PosInferred InfTerm p ->
  649                (p, "the type of this term")
  650            PosInferred InfContext p ->
  651                (p, "the context of the term")
  652            p ->
  653                (p, "this type annotation")
  654     in
  655     let ty' = prettyType ppopts ty
  656         what' = "arises from" <+> what
  657 
  658         -- Deliberately render and re-docify the type, and generate a
  659         -- multi-line message only if the type comes out as multiple
  660         -- lines when rendered on its own. This is kind of gross, but
  661         -- the prettyprinter library does not give much in the way of
  662         -- formatting control, and if we just do things its way we
  663         -- pretty much always get a multiline message, even for very
  664         -- short types like (), because what' coupled
  665         -- with the position text at the beginning of the line is long
  666         -- enough to make the prettyprinter library think the message
  667         -- ought to be multiline. Perhaps the right way to deal with
  668         -- this problem is to force it to use a different notion of
  669         -- what constitutes a "long" line when dealing with error
  670         -- messages rather than program text; but for the time being
  671         -- at least we have no useful infrastructure to support that.
  672         -- So instead generate our own faux "reactive" layout. XXX.
  673         --
  674         -- If you find a way to fix this better, please also fix the
  675         -- analogous code for "Too many arguments to function" below.
  676         --
  677         msg = case map PP.pretty $ Text.lines $ PPS.renderText ppopts ty' of
  678             [ty''] -> "The type" <+> ty'' <+> what'
  679             ty'' -> "The type" <+> PP.nest 3 (PP.vsep ty'' <> PP.line <> what')
  680     in
  681     (pos, msg)

  683 -- | Insert an entry in the substitution we're carrying around. Raw
  684 --   version; everyone except `resolveVar` should call `resolveVar`
  685 --   instead.
  686 addResolution :: TypeIndex -> Type -> TI ()
  687 addResolution i ty = do
  688     subst <- gets tiSubst
  689     let subst' = Map.insert i ty subst
  690     modify (\rw -> rw { tiSubst = subst' })

  692 -- | Resolve a unification var: update the table we're carrying around
  693 --   to hold the new definition for @i@.
  694 resolveVar :: Pos -> TypeIndex -> Type -> TI ()
  695 resolveVar pos'i i ty = do
  696     -- Check if we should prefer using t1 to t2 as an expansion.
  697     -- Return the unification variable ID inside t2.
  698     let prefer t1 t2 = case (t1, t2) of
  699           (TyUnifyVar _ j, TyUnifyVar pos'k k)
  700               | j < k -> Just (pos'k, k) -- prefer t1/j
  701               | otherwise -> Nothing
  702           (TyUnifyVar{}, _) -> Nothing
  703           (_, TyUnifyVar pos'j j) -> Just (pos'j, j) -- prefer t1
  704           (_, _) ->
  705               -- We should only ever get here for convertible types;
  706               -- we could check that; but we don't have an easy way
  707               -- to at the moment, so don't bother.
  708               Nothing
  709 
  710     -- Insert the new result. Shuffle around what's already there
  711     -- as needed to preserve the ordering invariant.
  712     case ty of
  713         TyUnifyVar pos'j j | j > i ->
  714             -- Maintain the ordering invariant for unification vars
  715             -- pointing at each other.
  716             resolveVar pos'j j (TyUnifyVar pos'i i)
  717         _ -> do
  718             -- Check what's in slot i.
  719             subst <- gets tiSubst
  720             case Map.lookup i subst of
  721                 Nothing ->
  722                     -- Nothing already there, resolve to what we've got
  723                     addResolution i ty
  724                 Just ty'already ->
  725                     -- There's already a type there. Check if it's a
  726                     -- unification var we should replace.
  727                     case prefer ty ty'already of
  728                         Nothing ->
  729                             -- There's a type already there, and we
  730                             -- shouldn't replace it with @ty@. If @ty@
  731                             -- is a unification var, resolve @ty@ to
  732                             -- the type already there. Otherwise, both
  733                             -- types are real types. We could assert
  734                             -- that they're convertible, but we don't
  735                             -- have an easy way to do that. There is
  736                             -- no need to do anything else.
  737                             case ty of
  738                                 TyUnifyVar pos'j j ->
  739                                     resolveVar pos'j j ty'already
  740                                 _ ->
  741                                     pure ()
  742                         Just (pos'j, j) -> do
  743                             -- There's a type already there, and we
  744                             -- should replace it because it's a
  745                             -- unification var. Do that, then resolve
  746                             -- its unification var too.
  747                             addResolution i ty
  748                             resolveVar pos'j j ty
  749 
  750 --
  751 -- | Unify two types.
  752 --
  753 -- When typechecking an expression the first type argument (@exp@)
  754 -- should be the type expected from the context, and the second
  755 -- (@found@) should be the type found in the expression appearing in
  756 -- that context. The position argument should be the position at which
  757 -- any error should be reported, typically the position of the program
  758 -- element whose type is t2.
  759 --
  760 -- If it doesn't work, the message generated will be of the form "pos:
  761 -- found t2, expected t1".
  762 --
  763 -- For example, when checking the second argument of a function
  764 -- application (Application _pos e1 e2) checking e1 gives rise to an
  765 -- expected type for e2. So when unifying that with the result of
  766 -- checking e2,
  767 --    - the t1 argument should be the expected type arising from e1,
  768 --    - the t2 argument should be the type returned by checking e2,
  769 --    - and the position argument should be the position of e2, not
  770 --      the position of the enclosing apply node.
  771 --
  772 -- Other cases should pass the arguments analogously. Some may still be
  773 -- backwards.
  774 --
  775 -- Error messages are indented (all but the first line) by four spaces
  776 -- because the first line sometimes ends up indented by two when it
  777 -- ultimately gets printed, and we want the grouping to be clearly
  778 -- recognizable.
  779 --
  780 unify :: Type -> Pos -> Type -> TI ()
  781 unify exp0 pos found0 = visit [] exp0 found0
  782   where
  783     visit :: [(Type, Type)] -> Type -> Type -> TI ()
  784     visit encs expectBase foundBase = do
  785         -- Use pos as the failure position for either type; they're the
  786         -- same type after all, and any position that gives rise to it
  787         -- should be good enough if expandFully croaks. (Hopefully.)
  788         expect <- expandFully pos expectBase
  789         found <- expandFully pos foundBase
  790 
  791         -- | Fail with expected/found types
  792         let reject msg more = do
  793               ppopts <- asks tiPPOpts
  794               encs' <- do
  795                   let once (t1, t2) = do
  796                         t1' <- expandFully pos t1
  797                         t2' <- expandFully pos t2
  798                         pure (t1', t2')
  799                   mapM once encs
  800               let body = PP.vsep $ more ++ [
  801                       prettyEnclosing ppopts ((expect, found) : encs')
  802                    ]
  803               recordError pos $ "Error:" <+> msg <> PP.line <> PP.indent 4 body
  804 
  805               let (pos'expect, expect') = prettyTypeDetails ppopts expect
  806                   (pos'found, found') = prettyTypeDetails ppopts found
  807               recordError pos'expect $ "Note:" <+> expect'
  808               -- Attach a blank line to this message so there's a separator
  809               -- between it and the next type error. XXX: we should have a
  810               -- less ad hoc way to do this.
  811               recordError pos'found $ "Note:" <+> found' <> PP.hardline <> ""
  812 
  813         -- | We would like to resolve unification var @i@ to type @ty@.
  814         --   Make sure this is well formed.
  815         --
  816         --   Does not handle the case where t _is_ TyUnifyVar i; there's
  817         --   a separate case for that.
  818         --
  819         let checkOccurs pos'i i ty =
  820               -- Collect the unification vars in ty, and check for an
  821               -- appearance of i. This is sufficient because ty has
  822               -- been fully expanded (it is either expect or found,
  823               -- and both were expanded above) so it cannot contain
  824               -- typedefs and all unification vars it contains are so
  825               -- far unresolved.
  826               --
  827               -- It is possible for some such unification var j to
  828               -- later be resolved to something that includes
  829               -- i. However, that will trigger this check then:
  830               -- because we've resolved i, it will check the candidate
  831               -- resolution of j for occurrences of j. That will
  832               -- contain the expansion of i, and by hypothesis there
  833               -- is an occurrence of j in the expansion of i.
  834               --
  835               -- It's thus impossible to get cycles of unification
  836               -- var references that bypass this check. This argument
  837               -- depends critically on every resolution attempt being
  838               -- able to see the complete set of previous resolutions
  839               -- in `tiSubst`, and on `expandFully` applying them all
  840               -- everywhere in ty, including recursively.
  841               --
  842               case Map.lookup i $ unifyVars ty of
  843                   Nothing -> pure ty
  844                   Just _otherpos -> do
  845                       ppopts <- asks tiPPOpts
  846                       let expect' = prettyType ppopts expect
  847                           found' = prettyType ppopts found
  848                           i' = prettyType ppopts $ TyUnifyVar pos'i i
  849                           ty' = prettyType ppopts ty
  850 
  851                       _ <- reject "Occurs check failure." [
  852                           "Cannot unify" <+> expect' <+>
  853                           "with" <+> found' <+> "because" <+> i' <+>
  854                           "appears within" <+> ty' <> "."
  855                        ]
  856                       getErrorTyVar pos'i
  857 
  858         -- recurse into one nested type
  859         let recOnce exp' found' =
  860               visit ((expect, found) : encs) exp' found'
  861 
  862         -- recurse into a list of types
  863         let recList exps' founds' =
  864               zipWithM_ (visit ((expect, found) : encs)) exps' founds'
  865 
  866 
  867         case (expect, found) of
  868             (TyUnifyVar _ i, TyUnifyVar _ j) | i == j ->
  869                 -- same unification var, nothing to do
  870                 pure ()
  871 
  872             (TyUnifyVar pos'i i, _) -> do
  873                 -- one side is a unification var, resolve it
  874                 found' <- checkOccurs (Pos.getPos found) i found
  875                 resolveVar pos'i i found'
  876 
  877             (_, TyUnifyVar pos'i i) -> do
  878                 -- the other side is a unification var, resolve it
  879                 expect' <- checkOccurs (Pos.getPos expect) i expect
  880                 resolveVar pos'i i expect'
  881 
  882             (TyFunc pos'expect _ expParams expNamedParams expRet,
  883              TyFunc pos'found _ foundParams foundNamedParams foundRet) -> do
  884                 -- First, unify the named parameters.
  885                 --
  886                 -- (We handle the named parameters first because because
  887                 -- they don't carry into the return value like curried
  888                 -- positional parameters.)
  889                 --
  890                 -- XXX: should we insist that both sides have the same named
  891                 -- parameters, or take the union of them? Allowing functions to
  892                 -- grow extra named (thus optional) parameters they ignore is
  893                 -- sound, but possibly unexpected/weird, so for now require them
  894                 -- to match exactly.
  895                 let expNames = Map.keysSet expNamedParams
  896                     foundNames = Map.keysSet foundNamedParams
  897                 if expNames /= foundNames then do
  898                     ppopts <- asks tiPPOpts
  899                     let expect' = prettyType ppopts expect
  900                         found' = prettyType ppopts found
  901                         expMissing = Map.difference foundNamedParams expNamedParams
  902                         foundMissing = Map.difference expNamedParams foundNamedParams
  903                         prettyMissing (name, ty) =
  904                             let ty' = prettyType ppopts ty in
  905                             PP.pretty name <+> ":" <+> ty'
  906                         prettyMissingList fty' ms = case ms of
  907                             [] ->
  908                                 []
  909                             _ ->
  910                                 let ms' = PP.vsep $ map prettyMissing ms
  911                                     line1 = "Missing from" <+> fty' <> ":"
  912                                     lines2 = PP.indent 3 ms'
  913                                 in
  914                                 [line1 <> PP.line <> lines2]
  915                         expMissing' = prettyMissingList expect' $ Map.toList expMissing
  916                         foundMissing' = prettyMissingList found' $ Map.toList foundMissing
  917                         missing' = expMissing' ++ foundMissing'
  918                     reject "Mismatched named parameters." missing'
  919 
  920                 else do
  921                     -- In principle when you have checked that the keys
  922                     -- match, you can do zip (Map.toList namedParams1)
  923                     -- (Map.toList namedParams2) and have things match up
  924                     -- correctly, but it makes me nervous, so do it like
  925                     -- this instead.
  926                     let allNamedParams =
  927                             Map.intersectionWith (,) expNamedParams foundNamedParams
  928                         (expNP, foundNP) = unzip $ Map.elems allNamedParams
  929                     recList expNP foundNP
  930 
  931                 -- Now unify as many positional params as possible. This
  932                 -- also produces the remainder types, basically the return
  933                 -- type. If one function is a -> b, and the other function
  934                 -- is a -> c -> d, the remainder types are b and c -> d
  935                 -- respectively.
  936 
  937                 let nExp = length expParams
  938                     nFound = length foundParams
  939                 (expRemainder, foundRemainder) <- do
  940                     if nExp < nFound then do
  941                         let foundParamsL = take nExp foundParams
  942                             foundParamsR = drop nExp foundParams
  943                         recList expParams foundParamsL
  944                         -- we've used up expParams.
  945                         let ty' = TyFunc pos'found noNames foundParamsR Map.empty foundRet
  946                         pure (expRet, ty')
  947                     else if nFound > nExp then do
  948                         -- unfortunately we need two copies of this because
  949                         -- left vs. right side is semantically significant :-(
  950                         let expParamsL = take nFound expParams
  951                             expParamsR = drop nFound expParams
  952                         recList expParamsL foundParams
  953                         -- we've used up foundParams.
  954                         let ty' = TyFunc pos'expect noNames expParamsR Map.empty expRet
  955                         pure (ty', foundRet)
  956                     else do
  957                         recList expParams foundParams
  958                         -- we've used up both expParams and foundParams.
  959                         pure (expRet, foundRet)
  960 
  961                 -- now unify the remainders / return types
  962                 recOnce expRemainder foundRemainder
  963 
  964             (TyRecord _ expFields, TyRecord _ foundFields)
  965               | Map.keys expFields /= Map.keys foundFields ->
  966                 -- records with different keys
  967                 reject "Record field names do not match." []
  968 
  969               | otherwise ->
  970                 -- records with the same field names, try unifying the field types
  971                 recList (Map.elems expFields) (Map.elems foundFields)
  972 
  973             (TyCon _ expTC expTS, TyCon _ foundTC foundTS) | expTC == foundTC -> do
  974                 -- same type constructor, unify the args
  975                 when (length expTS /= length foundTS) $ do
  976                     -- This case is unreachable.
  977                     --
  978                     -- Every distinct type constructor has a definite
  979                     -- arity (tuples of different lengths are not the same
  980                     -- type constructor) and every type is supposed to
  981                     -- pass `checkType` before we do anything more
  982                     -- significant with it; that does a kind check, and on
  983                     -- failure produces a fresh unification var that can't
  984                     -- cause further trouble.
  985                     --
  986                     -- Therefore, if we get here, something's broked and we should
  987                     -- panic.
  988                     --
  989                     ppopts <- asks tiPPOpts
  990                     let expTS'   = "LHS:" : map (\t -> "   " <> ppType ppopts t) expTS
  991                         foundTS' = "RHS:" : map (\t -> "   " <> ppType ppopts t) foundTS
  992                     let nexpect'   = Text.pack $ show $ length expTS
  993                         nfound' = Text.pack $ show $ length foundTS
  994                         heading = "Mismatched type constructor arguments: " <>
  995                                   "expected " <> nexpect' <> ", found " <> nfound'
  996                     panic "unify" (heading : expTS' ++ foundTS')
  997 
  998                 recList expTS foundTS
  999 
 1000             (TyVar _ a, TyVar _ b) | a == b ->
 1001                 -- Same named variable, nothing to do
 1002                 pure ()
 1003 
 1004             (_, TyFunc{}) ->
 1005                 -- If we expected a scalar and found a function, speculate that
 1006                 -- someone forgot a function argument earlier.
 1007                 reject "Type mismatch."
 1008                     ["Perhaps a function was not given enough arguments?"]
 1009 
 1010             (_, _) ->
 1011                 -- Did not work
 1012                 reject "Type mismatch." []

 1028 -- Inspect for free type variables
 1029 
 1030 -- We want to allow declaring polymorphic functions by introducing
 1031 -- type variables in the function header (rather than requiring an
 1032 -- explicit forall binding), like Haskell does.
 1033 --
 1034 -- This means that free type variables in a function header (but not
 1035 -- elsewhere) should be accepted, collected, and handed off to
 1036 -- generalize for insertion in the resultant type scheme.
 1037 --
 1038 -- It turns out that because of the way the AST represents functions
 1039 -- in let-bindings that this is highly unpleasant to do on the fly
 1040 -- while typechecking. So instead extract the free type variables
 1041 -- separately.
 1042 --
 1043 -- A function header comes through like this:
 1044 --    Decl _pos <function-name-pattern> Nothing <expr>
 1045 --
 1046 -- where <expr> is
 1047 --    zero or more times, Lambda _pos <params> <named-params> <expr'>
 1048 --    then optionally, TSig _pos <expr''> <return-type>
 1049 --
 1050 -- so we need any free type variables in
 1051 --    - <function-name-pattern>
 1052 --    - <return-type>
 1053 --    - all <params> and <named-params>
 1054 --
 1055 -- On the plus side this will also then work when people write
 1056 -- otherwise annoying things like
 1057 --    let f (x: a) = \(y: b) -> (a, b)
 1058 --
 1059 -- We extract the type variables with the position of their
 1060 -- initial mention, and the kind that appears to apply.
 1061 --
 1062 -- If the kind usage is inconsistent, the declaration involved will
 1063 -- fail kind-checking downstream. So it doesn't matter which of the
 1064 -- multiple usages we return, and we'll leave that unspecified.
 1065 
 1066 -- Get the free type variables found in a Type.
 1067 inspectTypeFTVs :: Kind -> Type -> TI (Map Name (Pos, Kind))
 1068 inspectTypeFTVs kind ty = case ty of
 1069     TyCon _pos ctor args -> do
 1070         let kinds = lookupTyCon ctor
 1071         Map.unions <$> zipWithM inspectTypeFTVs kinds args
 1072     TyFunc _pos _ params namedParams ret ->
 1073         let np = Map.elems namedParams in
 1074         Map.unions <$> mapM (inspectTypeFTVs kindStar) (ret : params ++ np)
 1075     TyRecord _pos fields ->
 1076         Map.unions <$> traverse (inspectTypeFTVs kindStar) fields
 1077     TyUnifyVar _pos _x ->
 1078         return Map.empty
 1079     TyVar pos x -> do
 1080         tyenv <- gets tiTyEnv
 1081         case ScopedMap.lookup x tyenv of
 1082             Nothing -> return $ Map.singleton x (pos, kind)
 1083             Just _ -> return $ Map.empty
 1084 
 1085 -- Get the free type variables found in a Maybe Type.
 1086 inspectMaybeTypeFTVs :: Kind -> Maybe Type -> TI (Map Name (Pos, Kind))
 1087 inspectMaybeTypeFTVs kind mty = case mty of
 1088     Nothing -> return Map.empty
 1089     Just ty -> inspectTypeFTVs kind ty
 1090 
 1091 -- Get the free type variables found in a Pattern.
 1092 inspectPatternFTVs :: Pattern -> TI (Map Name (Pos, Kind))
 1093 inspectPatternFTVs pat = case pat of
 1094     PWild _pos mty -> inspectMaybeTypeFTVs kindStar mty
 1095     PVar _allpos _xpos _x mty -> inspectMaybeTypeFTVs kindStar mty
 1096     PTuple _pos subpats ->
 1097         Map.unions <$> mapM inspectPatternFTVs subpats
 1098 
 1099 -- Get the free type variables found in a parameter list, aka [Pattern].
 1100 inspectParamsFTVs :: [Pattern] -> TI (Map Name (Pos, Kind))
 1101 inspectParamsFTVs pats =
 1102     Map.unions <$> mapM inspectPatternFTVs pats
 1103 
 1104 -- Get the free type variables found in a named parameter list.
 1105 inspectNamedParamsFTVs ::
 1106       Map Text (Pos, (Pos, Expr, Pattern)) ->
 1107       TI (Map Name (Pos, Kind))
 1108 inspectNamedParamsFTVs params =
 1109     --
 1110     -- Note: it's possible we should also allow type variables to
 1111     -- appear in the default value expressions for optional arguments.
 1112     -- For the time being we don't.
 1113     --
 1114     -- That means that you can't write
 1115     --   let foo x?=(undefined : a) = ...
 1116     -- ...you should instead write that as
 1117     --   let foo (x?=undefined : a) = ...
 1118     -- that is, put the type on the parameter and not the value.
 1119     --
 1120     let extract (_x, (_xpos, (_allpos, _e, pat))) = pat
 1121         pats = map extract $ Map.toList params
 1122     in
 1123     inspectParamsFTVs pats
 1124 
 1125 -- Get the free type variables found in a chain of Lambda Exprs.
 1126 -- Also return the body expression found on the inside of the chain
 1127 -- for possible further analysis.
 1128 inspectLambdaFTVs :: Expr -> TI (Expr, Map Name (Pos, Kind))
 1129 inspectLambdaFTVs e0 = case e0 of
 1130     Lambda _fpos _mname params namedParams e1 -> do
 1131         paramFTVs <- inspectParamsFTVs params
 1132         namedFTVs <- inspectNamedParamsFTVs namedParams
 1133         (e1', bodyFTVs) <- inspectLambdaFTVs e1
 1134         return (e1', Map.union (Map.union paramFTVs namedFTVs) bodyFTVs)
 1135     _ ->
 1136         return (e0, Map.empty)
 1137 
 1138 -- Get the free type variables found in a Decl.
 1139 inspectDeclFTVs :: Decl -> TI (Map Name (Pos, Kind))
 1140 inspectDeclFTVs (Decl _dpos pat _mty e0) = do
 1141     nameFTVs <- inspectPatternFTVs pat
 1142     (e1, argFTVs) <- inspectLambdaFTVs e0
 1143     retFTVs <- case e1 of
 1144         TSig _tspos _e2 ty -> inspectTypeFTVs kindStar ty
 1145         _ -> return Map.empty
 1146     return $ Map.unions [nameFTVs, argFTVs, retFTVs]

 1168 -- | Add x with type ty to the environment.
 1169 addVar :: Name -> Pos -> Rebindable -> Schema -> TI ()

 1418         -- The way this works is:
 1419         --    - First we infer a type for the thing we're calling (the
 1420         --      expression f). In the common case, it's just a function
 1421         --      name and we'll get a function type back from the
 1422         --      environment.
 1423         --    - Then we resolve that type fully by substituting etc so
 1424         --      we can look at it.
 1425         --    - Then we infer types for all the arguments.
 1426         --    - Then we use checkCall to check the actual call, passing
 1427         --      the function type and the arguments/argument types.  (We
 1428         --      want the argument expressions too, because we want their
 1429         --      positions for some of the failure cases.)
 1430         --    - checkCall examines the type of the callee, and if it's a
 1431         --      function type unifies the parameter types with the
 1432         --      argument type we got, and then:
 1433         --         - if there are more parameters left, returns a
 1434         --           partially applied function type;
 1435         --         - if there are more arguments left, recurses to try
 1436         --           calling the result type with the rest of the
 1437         --           arguments;
 1438         --         - if we exactly consumed all the arguments, just
 1439         --           returns the result type.
 1440         --    - If it _wasn't_ a function type but an unresolved
 1441         --      unification var, cons up a function type from the
 1442         --      arguments and unify that with the unification var.
 1443         --    - If it was anything else, we tried to call a
 1444         --      non-function, so complain. Handle the error paths for
 1445         --      this being the first go vs. not differently, because one
 1446         --      is calling a non-function and the other is giving too
 1447         --      many arguments.
 1448         --
 1449         -- There's an oddity that comes up when you have a function that
 1450         -- polymorphically returns its last parameter and you pass it
 1451         -- too many arguments. This causes us to infer a type t1 -> t2
 1452         -- for that last parameter (so that it can accept the excess
 1453         -- argument) and then when it isn't actually a function the
 1454         -- resulting error message readily becomes confusing. This can
 1455         -- be staved off by doing the unifications just so (in
 1456         -- particular, if we have f x y = y, unify the type of (f x)
 1457         -- with a -> b, and then unify b with the type of y _before_
 1458         -- unifying it with a type a' -> b' to accept the next argument,
 1459         -- we hit the expected path for too many arguments) but in the
 1460         -- current formulation with multiple-argument functions we can't
 1461         -- readily do it that way. What happens below is that we make a
 1462         -- second go through checkCall and it goes to the TyUnifyVar
 1463         -- case, and by being careful about the positions it reports we
 1464         -- manage to produce an ok message. There is a test in
 1465         -- test_type_errors to make sure the message for this particular

 1684 -- | Infer types for a pattern, producing fresh type variables as needed.
 1685 --
 1686 --   There may already be types in the pattern if there were explicit
 1687 --   type annotations in the input; if so don't throw them away.
 1688 --
 1689 --   If the enclosing context says "rebindable", either
 1690 --      - the variable is not already present in the environment
 1691 --      - or it is present and already declared "rebindable", in which
 1692 --        case it must have the same type.
 1693 inferPattern :: Rebindable -> Pattern -> TI (Type, Pattern)
 1694 inferPattern rebindable pat = do
 1695     let resolveType pos mt = case mt of
 1696           Nothing -> getFreshTyVar pos
 1697           Just t -> checkType kindStar t
 1698 
 1699     case pat of
 1700         PWild pos mt -> do
 1701             t <- resolveType pos mt
 1702             return (t, PWild pos (Just t))
 1703         PVar allpos xpos x mt -> do
 1704             t <- resolveType allpos mt
 1705             env <- gets tiVarEnv
 1706             case ScopedMap.lookup x env of
 1707                 Nothing -> pure ()
 1708                 Just (prevpos, lc, prevrb, tyscheme) -> case rebindable of
 1709                     RebindableVar -> do
 1710                         let croak msg =
 1711                               recordError xpos $ "Cannot rebind" <+>
 1712                                                  PP.pretty x <>
 1713                                                  ":" <+> msg
 1714                         avail <- asks tiPrimsAvail
 1715                         when (not $ Set.member lc avail) $
 1716                             croak "A previous binding exists but is hidden"
 1717                         oldt <- case tyscheme of
 1718                             Forall [] oldt' -> pure oldt'
 1719                             Forall (_ : _) _ -> do
 1720                                 croak "Polymorphic objects cannot be rebound"
 1721                                 getErrorTyVar xpos
 1722                         when (prevrb == ReadOnlyVar) $
 1723                             croak "Previous binding was not tagged 'rebindable'"
 1724                         unify oldt prevpos t
 1725                     ReadOnlyVar -> do
 1726                         -- The ocaml-style behavior of being able to do
 1727                         --    let x = 3;
 1728                         --    let x = foo x;
 1729                         --    let x = bar x;
 1730                         -- to create successive versions of x is often
 1731                         -- convenient, and an unconditional
 1732                         -- warning defeats that. However, the historical
 1733                         -- behavior of SAWScript (in certain contexts)
 1734                         -- is to mutate x, such that
 1735                         --     let x = 3;
 1736                         --     let y = x;
 1737                         --     let x = 4;
 1738                         --     print y;

 1817 -- | Type inference for a single statement.
 1818 --
 1819 --   The boolean is whether we're at the syntactic top level, which is used
 1820 --   for workaround logic for issue #2162.
 1821 --
 1822 --   The passed-in position should be the position associated with the monad type
 1823 --   the first type argument (ctx) is the monad type for any binds that occur.
 1824 --
 1825 -- Updates the environment and returns an updated statement.
 1826 inferStmt :: Bool -> Pos -> Type -> Stmt -> TI Stmt
 1827 inferStmt atSyntacticTopLevel blockpos ctx s = do
 1828     ppopts <- asks tiPPOpts
 1829     case s of
 1830         StmtBind spos pat e -> do
 1831             (pty, pat') <- inferPattern ReadOnlyVar pat
 1832             -- The expression should be of monad type. The
 1833             -- straightforward way to proceed here is to unify both
 1834             -- the monad type (ctx) and the result type expected by
 1835             -- the pattern (pty), like this:
 1836             --    e' <- checkExpr e (tBlock blockpos ctx pty)
 1837             --
 1838             -- However, historically when at the syntactic top level
 1839             -- (only), the monad type was left off, meaning that
 1840             -- various incorrect forms were silently accepted. Fixing
 1841             -- this in Dec 2024 triggered a lot of fallout, so we want
 1842             -- to specifically check for the following cases. (Again,
 1843             -- only when at the syntactic top level. Which is not when
 1844             -- in the TopLevel monad.)
 1845             --    x <- e for non-monadic e
 1846             --    x <- e for e in the wrong monad
 1847             --
 1848             -- These are now errors again, but the explicit messages
 1849             -- should be kept for at least one further release. See
 1850             -- #2167 and #2162. They are warnings in SAW 1.5; they
 1851             -- will be errors in 1.6; they can be removed in 1.7.
 1852             --
 1853             -- To accomplish this, call inferExpr to get a type for
 1854             -- the expression, and examine it. If the special cases
 1855             -- apply, issue special-case errors with explanations,
 1856             -- unify the type with only the pattern type, and patch up
 1857             -- the expression by wrapping it in "return".  (The latter
 1858             -- will restore the old behavior for both cases, so we
 1859             -- don't need to also gunk up the interpreter to handle
 1860             -- this problem.)
 1861             --
 1862             -- If the special cases don't apply, unify the result type
 1863             -- with the complete type.
 1864             (e', ty) <- inferExpr e
 1865             ty' <- expandFully (Pos.getPos e) ty
 1866 
 1867             -- The correct, restricted case
 1868             let restrictToCorrect = do
 1869                   -- unify the type of e with the expected monad and
 1870                   -- pattern types
 1871                   unify (tBlock blockpos ctx pty) (Pos.getPos e') ty
 1872                   return e'
 1873 
 1874             -- The special case for non-monadic values
 1875             let allowNonMonadic = do
 1876                   recordError spos $ "Monadic bind of non-monadic value;" <+>
 1877                                      "rewrite as let-binding or use return"
 1878                   unify pty (Pos.getPos e') ty
 1879                   -- Wrap the expression in "return" to correct the type
 1880                   return $ wrapReturn e'
 1881 
 1882             -- The special case for the wrong monad
 1883             let allowWrongMonad ctx' valty' = do
 1884                   let pctx =  prettyType ppopts ctx
 1885                       pctx' = prettyType ppopts ctx'
 1886                   recordError spos $ "Monadic bind with the wrong monad;" <+>
 1887                                      "found" <+> pctx' <+>
 1888                                      "but expected" <+> pctx
 1889                   recordError spos $ "This creates the action but does" <+>
 1890                                      "not execute it; if you meant to do" <+>
 1891                                      "that, prefix the" <+>
 1892                                      "expression with return"
 1893 
 1894                   -- The historic behavior is that the pattern gets bound
 1895                   -- to a value of type m t instead of type t. This means:
 1896                   --    - we should unify pty, which is the type of the
 1897                   --      pattern, with m t, which is tBlock ctx' valty'
 1898                   --      (rather than tBlock ctx valty', which is the
 1899                   --      type we should be getting)
 1900                   --    - this will fail if the pattern includes a type
 1901                   --      signature with a non-monad type, but that's ok
 1902                   --      because that case also fails in old SAW
 1903                   --    - we do _not_ need to update pty before returning
 1904                   --      it out of inferStmt
 1905                   --    - we _do_ need to wrap the expression in "return"
 1906                   --      so that the ultimate results are well-typed and
 1907                   --      happen in the TopLevel monad
 1908                   unify pty (Pos.getPos e') (tBlock spos ctx' valty')
 1909 
 1910                   -- Wrap the expression in "return" to produce an
 1911                   -- expression of type TopLevel (m t).
 1912                   return $ wrapReturn e'
 1913 
 1914             -- Figure out which case applies.
 1915             e'' <-
 1916                 if not atSyntacticTopLevel then
 1917                     restrictToCorrect
 1918                 else do
 1919                     ok <- matches blockpos (tBlock blockpos ctx pty) ty
 1920                     if ok then
 1921                         restrictToCorrect
 1922                     else
 1923                         case monadType ty' of
 1924                             Just (ctx', valty') ->
 1925                                -- Allow it only for _ and a single var.
 1926                                -- Binding elements of a tuple this way
 1927                                -- failed typecheck in the old saw and
 1928                                -- doesn't need to be allowed now.
 1929                                case pat of
 1930                                    PTuple _ _ -> restrictToCorrect
 1931                                    _ -> allowWrongMonad ctx' valty'
 1932                             Nothing ->
 1933                                -- allow it only if actually binding something
 1934                                -- (just proclaiming a value by itself is not a
 1935                                -- case we need to worry about)
 1936                                case pat of
 1937                                    PWild _ _ -> restrictToCorrect
 1938                                    _ -> allowNonMonadic
 1939 
 1940             let s' = StmtBind spos pat' e''
 1941             addPattern pat'
 1942             return s'
 1943         StmtLet spos rebindable dg -> do
 1944             when (rebindable == RebindableVar && not atSyntacticTopLevel) $ do
 1945                 recordError spos "Invalid use of 'rebindable'"
 1946                 recordError spos "It is only allowed at the syntactic top level"
 1947             dg' <- inferDeclGroup rebindable dg
 1948             let s' = StmtLet spos rebindable dg'
 1949             addDeclGroup rebindable dg'
 1950             return s'
 1951         StmtCode _allpos _spos _txt ->
 1952             return s
 1953         StmtImport _spos _ ->
 1954             return s
 1955         StmtInclude spos _ _ -> do
 1956             -- Restrict include to TopLevel. This matches the prior
 1957             -- behavior when it was a builtin function rather than
 1958             -- syntax. FUTURE: consider relaxing the requirement.
 1959             let spos' = PosInferred InfTerm spos
 1960             let tm = TyCon spos' (ContextCon TopLevel) []
 1961             tx <- getFreshTyVar spos
 1962             unify (tBlock blockpos ctx tx) spos (tBlock spos tm tx)
 1963             return s
 1964         StmtTypedef allpos apos a ty -> do
 1965             ty' <- checkType kindStar ty
 1966             tyenv <- gets tiTyEnv
 1967             case ScopedMap.lookup a tyenv of
 1968                 Nothing -> do
 1969                     let s' = StmtTypedef allpos apos a ty'
 1970                     addTypedef a ty'
 1971                     return s'
 1972                 Just (lc, _expansion) -> do
 1973                     -- Prohibit redefining any type, even ones that
 1974                     -- aren't visible. In principle it is ok to
 1975                     -- redefine a type that isn't visible, since
 1976                     -- existing references to it shouldn't be visible
 1977                     -- either. But (a) that's not always true, there
 1978                     -- have been bugs in the builtin list before; and
 1979                     -- (b) it would require remembering permanently
 1980                     -- that any corresponding future use of
 1981                     -- enable_experimental or enable_deprecated must
 1982                     -- be blocked.
 1983                     let a' = PP.pretty a
 1984                     avail <- asks tiPrimsAvail
 1985                     let addendum =
 1986                           if Set.member lc avail then ""
 1987                           else " (which is not currently visible)"
 1988                     recordError allpos $ "Redefinition of type" <+> a' <>
 1989                                          addendum
 1990                     -- FUTURE: print the position of the previous definition
 1991                     -- (currently we don't keep it around)
 1992                     return s
 1993         StmtPushdir _spos _ ->
 1994             return s
 1995         StmtPopdir _spos ->
 1996             return s

 1998 -- | Inference for a do-block.
 1999 --
 2000 --   The passed-in position should be the position for the whole
 2001 --   statement block.
 2002 --
 2003 --   The first type argument (ctx) is the monad type for the block.
 2004 --
 2005 --   The second type argument (ty) is the expected full result type for
 2006 --   the block (including the monad) to be unified with the result type
 2007 --   found.
 2008 --
 2009 inferBlock :: Pos -> Type -> Type -> ([Stmt], Expr) -> TI ([OutStmt], OutExpr)
 2010 inferBlock blockpos ctx ty (stmts, lastexpr) = do
 2011     let atSyntacticTopLevel = False
 2012 
 2013     -- Check the statements in order, left first.
 2014     stmts' <- mapM (inferStmt atSyntacticTopLevel blockpos ctx) stmts
 2015 
 2016     -- Check the final expression.
 2017     -- This produces the result type for the block.
 2018     (lastexpr', ty') <- inferExpr lastexpr
 2019     unify ty (Pos.getPos lastexpr) ty'
 2020 
 2021     return (stmts', lastexpr')

 2052 -- | Create a type schema for a list of mutually referential
 2053 --   declarations out of their free vars.
 2054 --
 2055 --   (This creates names for any remaining unification vars, so
 2056 --   potentially updates the expression.)
 2057 --
 2058 --   The "foralls" argument is a set of tyvars that were mentioned
 2059 --   explicitly and should be forall-bound.
 2060 generalize ::
 2061     Map Name Pos -> [Pattern] -> [OutExpr] -> [Type] ->
 2062     TI [(Pattern, OutExpr, Schema)]
 2063 generalize foralls pats0 es0 ts0 = do
 2064     -- first, substitute away any resolved unification variables
 2065     -- in both the expressions and types.
 2066     pats <- applyCurrentSubst pats0
 2067     es <- applyCurrentSubst es0
 2068     ts <- applyCurrentSubst ts0
 2069 
 2070     -- Extract lists of any unification vars and named type vars that
 2071     -- still appear.
 2072     let is0 = unifyVars ts
 2073     let bs0 = Util.namedTyVars ts
 2074 
 2075     -- Drop any unification vars and named type vars that we
 2076     -- shouldn't forall-bind.
 2077     --
 2078     -- For unification vars, any whose scope reaches beyond the
 2079     -- current declaration should be left alone; they should only be
 2080     -- bound when they eventually move out of scope. Get these by
 2081     -- examining the types used in the right-hand sides of both the
 2082     -- variable environment and the type environment.
 2083     --
 2084     -- For named vars, exclude any that appear that appear as keys
 2085     -- (on the left-hand side) of the type environment. Those are
 2086     -- already defined.
 2087     --
 2088     -- The only other named variables involved should be the set we
 2089     -- explicitly intend to be forall-bound as passed in. Insert
 2090     -- those, and favor their positions.
 2091     --
 2092     -- It would be handy for scaling if we didn't have to examine
 2093     -- the entire variable environment (on the grounds that there
 2094     -- should be no loose unification vars at the top level where
 2095     -- most definitions will come from) but (a) we don't have the
 2096     -- structure to support that and (b) it is not absolutely clear
 2097     -- that there isn't a way to get such loose unification vars,
 2098     -- in which case we'd have to do something about it.
 2099     --
 2100     -- This code also used to exclude named vars used on the
 2101     -- right-hand side of the variable environment; this was to allow
 2102     -- the use of otherwise undefined type names in the primitives
 2103     -- table. There is no longer any need for such hackery, and
 2104     -- undefined type names are not allowed to appear in the variable
 2105     -- environment.
 2106     --
 2107     -- FUTURE: we end up replacing the user's forall-bound names with
 2108     -- generated names, and I'm not sure why. It seems like it
 2109     -- shouldn't be possible the way the code is structured. But the
 2110     -- type signatures are coming out correct (which they wouldn't if
 2111     -- something were seriously wrong) and we aren't inappropriately
 2112     -- unifying these vars with each other or with other things, so
 2113     -- I'm not going to stress over it right now.
 2114 
 2115     envUnifyVars <- unifyVarsInEnvs
 2116     knownNamedVars <- namedVarDefinitions
 2117     let is1 = is0 Map.\\ envUnifyVars
 2118     let bs1 = Map.union foralls $ Map.withoutKeys bs0 knownNamedVars
 2119 
 2120     -- convert to lists
 2121     let is2 = Map.toList is1
 2122     let bs2 = Map.toList bs1
 2123 
 2124     -- if the position is "fresh" turn it into "inferred from term"
 2125     let adjustPos pos = case pos of
 2126           PosInferred InfFresh pos' -> PosInferred InfTerm pos'
 2127           _ -> pos
 2128 
 2129     -- generate names for the unification vars
 2130     let is3 = [ (i, adjustPos pos, "a." <> Text.pack (show i)) | (i, pos) <- is2 ]
 2131 
 2132     -- build a substitution
 2133     let s = Map.fromList [ (i, TyVar pos n) | (i, pos, n) <- is3 ]
 2134 
 2135     -- get the names for the Forall
 2136     let inames = [ (pos, n) | (_i, pos, n) <- is3 ]
 2137     let bnames = [ (pos, x) | (x, pos) <- bs2 ]
 2138 
 2139     let mk pat e t =
 2140           let pat' = appSubst s pat
 2141               e' = appSubst s e
 2142               t' = appSubst s t
 2143           in
 2144           (pat', e', Forall (inames ++ bnames) t')
 2145 
 2146     return $ zipWith3 mk pats es ts

 2149 -- | Check that a type is a function and isn't a plain value, in order
 2150 --   to reject recursive values in "rec" definitions. Otherwise they
 2151 --   crash the interpreter downstream. See issue #2203.
 2152 --
 2153 --   There are cases where it might be convenient to include a plain
 2154 --   value within a system of recursive declarations. For example, if
 2155 --   you have something like
 2156 --      rec foo x = ...
 2157 --      and foo0 = foo 0
 2158 --      and foo1 = foo 1
 2159 --      and bar x = ...
 2160 --      and bar0 = bar 0
 2161 --      and bar1 = bar 1
 2162 --      and baz x = ...
 2163 --      and baz0 = baz 0
 2164 --      and baz1 = baz 1
 2165 --   then depending on what the code is, it might be logically
 2166 --   reasonable to place the values like this and ugly to need to move
 2167 --   them out of the flow. If this ever comes up it might make sense to
 2168 --   loosen this check (e.g. to check whether the value is actually
 2169 --   recursive) and also fix the interpreter to not choke. However,
 2170 --   provided the values actually aren't recursive it is _possible_ to
 2171 --   move them out, so this is only worth chasing after given a fairly
 2172 --   compelling use case.
 2173 --
 2174 --   Note that actual recursively defined values are always bottom (in
 2175 --   the absence of mutable variables) and are best not allowed.
 2176 --
 2177 requireFunction :: Pos -> Type -> TI ()
 2178 requireFunction pos ty = do
 2179     ty' <- expandFully pos ty
 2180     case ty' of
 2181         TyFunc _pos _ninfo _params _namedParams _ret ->
 2182             return ()
 2183         _ ->
 2184             recordError pos $ "Only functions may be recursive."

 2186 -- | Type inference for a declaration.
 2187 --
 2188 --   Note that the type schema slot in Decl is always Nothing the way it
 2189 --   comes from the parser; if there's an explicit type annotation on
 2190 --   the declaration, it shows up as a type signature in the expression.
 2191 --
 2192 --   This function does _not_ update the variable environment to reflect
 2193 --   the declaration. The caller does that. XXX: this seems messy. (But
 2194 --   note that checkDecl is used by the :type REPL command, which
 2195 --   shouldn't update anything, so it's not open and shut.)
 2196 inferDecl :: Rebindable -> Decl -> TI Decl
 2197 inferDecl rebindable d@(Decl pos pat _ e) = do
 2198     -- collect the free type variables
 2199     foralls <- inspectDeclFTVs d
 2200     let foralls' = Map.map (\(typos, _kind) -> typos) foralls
 2201 
 2202     -- Add abstract type variables for the foralls while we check the body.
 2203     -- Note: this is a variable declaration. It doesn't add types; the types
 2204     -- get forall-bound in the type scheme by the `generalize` call.
 2205     pushScope
 2206     addAbstractTyVars foralls
 2207 
 2208     -- Check the body and check the pattern against the body.
 2209     (e', t) <- inferExpr e
 2210     pat' <- checkPattern rebindable t pat
 2211 
 2212     -- Use `generalize` to build the type scheme.
 2213     ~[(pat'', e1, s)] <- generalize foralls' [pat'] [e'] [t]
 2214 
 2215     -- Drop the abstract type variables
 2216     popScope
 2217 
 2218     -- Return the updated `Decl`
 2219     return (Decl pos pat'' (Just s) e1)

 2221 -- | Type inference for a system of mutually recursive declarations.
 2222 --
 2223 --   Note that the type schema slot in the Decls is always Nothing as we
 2224 --   get them from the parser; if there's an explicit type annotation on
 2225 --   some or all of the declarations those shows up as type signatures
 2226 --   in the expressions.
 2227 --
 2228 --   This function does _not_ update the variable environment to reflect
 2229 --   the declaration. The caller does that. XXX: this is messy.
 2230 inferRecDecls :: [Decl] -> TI [Decl]
 2231 inferRecDecls ds = do
 2232     -- Get the patterns out of the decls.
 2233     let pats = map dPat ds
 2234 
 2235     -- Collect the free type variables.
 2236     foralls <- Map.unions <$> mapM inspectDeclFTVs ds
 2237     let foralls' = Map.map (\(typos, _kind) -> typos) foralls
 2238 
 2239     -- Add abstract type variables for the foralls while we check the
 2240     -- bodies.
 2241     pushScope
 2242     addAbstractTyVars foralls
 2243 
 2244     -- Check the patterns first to get types.
 2245     (_ts, pats') <- unzip <$> mapM (inferPattern ReadOnlyVar) pats
 2246 
 2247     -- Check all the expressions in an environment that includes all
 2248     -- the bound variables.
 2249     --
 2250     -- FUTURE: we only need a second nested scope here because we run

 2255     -- twice.

 2257     addPatterns pats'
 2258     let checkOneExpr (Decl _pos _p _ e) = inferExpr e
 2259     (es, tys) <- unzip <$> mapM checkOneExpr ds
 2260     popScope

 2269     sequence_ $ zipWith (checkPattern ReadOnlyVar) tys pats'
 2270 
 2271     -- Run generalize and get back a list of updated expressions and
 2272     -- type schemes.
 2273     patetys <- generalize foralls' pats' es tys
 2274 
 2275     -- Drop the abstract type variables.
 2276     popScope
 2277 
 2278     -- Generate the updated declarations.
 2279     let rebuild pos (pat, e1, ty) = Decl pos pat (Just ty) e1
 2280         ds' = zipWith rebuild (map Pos.getPos ds) patetys
 2281 
 2282     return ds'

 2520 --   (second argument) and return True if they can be unified.
 2521 --
 2522 --   Both types are schemes because that's what we need upstream.
 2523 --
 2524 --   (This is an external interface.)
 2525 typesMatch ::
 2526       PPS.Opts ->
 2527       Set PrimitiveLifecycle ->
 2528       TyEnv ->
 2529       Schema ->
 2530       Schema ->
 2531       Bool
 2532 typesMatch ppopts avail tenv schema'found schema'expected =
 2533   let unpack (Forall as ty) = do
 2534         -- Generate unification vars for all the forall-bindings
 2535         let generate (pos'a, a) = do
 2536               ty'a <- getFreshTyVar pos'a
 2537               return (a, (Current, ConcreteType ty'a))
 2538         substs <- mapM generate as
 2539         -- Substitute them into the type
 2540         let ty' = Util.substituteTyVars' avail (Map.fromList substs) ty
 2541         return ty'
 2542       match = do
 2543         -- Unpack the schemas and check if they match
 2544         ty'found <- unpack schema'found
 2545         ty'expected <- unpack schema'expected
 2546         matches (Pos.getPos ty'found) ty'found ty'expected
 2547   in
 2548   case runTI ppopts avail ScopedMap.empty tenv match of
 2549       (Left _errors, _warnings) -> False        -- not actually reachable
 2550       (Right b, _warnings) -> b                 -- return match success/failure

 2552 -- | Check a schema (type) as used when constructing the builtins
 2553 --   table. (This is an external interface.)
 2554 --
 2555 --   The first argument is the lifecycle context the type is being used
 2556 --   in. More on that below. The second is the typedef environment to
 2557 --   use. The third argument is the schema to check.
 2558 --
 2559 --   All types found should be of kind *.
 2560 --
 2561 --   Purely a validity check; there's no updates it can make to the
 2562 --   schema that are of use to the caller, on the assumption that the
 2563 --   caller doesn't want to do stuff with the schema before exiting on
 2564 --   errors, which it doesn't. Thus, return the original schema and not
 2565 --   one with the potentially updated type.
 2566 --
 2567 --   (Otherwise we'd need to rerun `generalize` to build a new schema,
 2568 --   and that's a headache and not worthwhile given that the result
 2569 --   isn't going to be used.)
 2570 --
 2571 --   Do however return the original schema and not unit to make sure
 2572 --   the code actually gets evaluated.
 2573 --
 2574 --   This is called for the types of objects that may themselves not be
 2575 --   visible, so rather than using the caller's set of visible lifecycle
 2576 --   states, construct the set based on the lifecycle state of the
 2577 --   declaration context. Deprecated objects can see equally or less
 2578 --   deprecated types; experimental objects can see experimental types;
 2579 --   everything can see current types.
 2580 --
 2581 checkSchema ::
 2582       PPS.Opts ->
 2583       PrimitiveLifecycle ->
 2584       TyEnv ->
 2585       Schema ->
 2586       Result Schema
 2587 checkSchema ppopts contextLC tyenv schema = do
 2588     let check = do
 2589           let Forall tyvars ty = schema
 2590           -- Generate unification vars for all the forall-bindings
 2591           let generate (pos'a, a) = do
 2592                 ty'a <- getFreshTyVar pos'a
 2593                 return (a, (Current, ConcreteType ty'a))
 2594           substs <- mapM generate tyvars
 2595           -- Substitute them into the type
 2596           let ty' = Util.substituteTyVars' everythingAvailable (Map.fromList substs) ty
 2597           -- The only way checking can return an updated type is if
 2598           -- there's also an error, so discard the type
 2599           _ <- checkType kindStar ty'
 2600           return schema
 2601 
 2602     let avail = Set.fromList $ case contextLC of
 2603           Current -> [Current]
 2604           WarnDeprecated -> [Current, WarnDeprecated]
 2605           HideDeprecated -> [Current, WarnDeprecated, HideDeprecated, Experimental]
 2606           Experimental -> [Current, Experimental]
 2607     runTI ppopts avail ScopedMap.empty tyenv check

 2639 Note [-Wincomplete-uni-patterns and irrefutable patterns]
 2640 ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
 2641 Various parts of SAW use irrefutable patterns in functions that assume that
 2642 their arguments have particular shapes. For example, inferDecl in this module
 2643 matches on ~[(e1,s)] with an irrefutable pattern because it assumes the
 2644 invariant that the list will have exactly one element. This lets inferDecl be
 2645 slightly lazier when evaluated.
 2646 
 2647 Unfortunately, this use of irrefutable patterns is at odds with the
 2648 -Wincomplete-uni-patterns warning. At present, -Wincomplete-uni-patterns will
 2649 produce a warning for any irrefutable pattern that does not cover all possible
 2650 data constructors. While we could rewrite functions like `inferDecl` to
 2651 explicitly provide a fall-through case, that would change its strictness
 2652 properties. As a result, we simply disable -Wincomplete-uni-patterns warnings
 2653 in each part of SAW that uses irrefutable patterns.
 2654 
 2655 Arguably, -Wincomplete-uni-patterns shouldn't be producing warnings for
 2656 irrefutable patterns at all. GHC issue #14800
 2657 (https://gitlab.haskell.org/ghc/ghc/-/issues/14800) proposes this idea.
 2658 If that issue is fixed in the future, we may want to reconsider whether we want
 2659 to disable -Wincomplete-uni-patterns.
 2660 -}
