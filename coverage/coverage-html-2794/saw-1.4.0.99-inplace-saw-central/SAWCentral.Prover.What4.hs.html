   46 setupWhat4_sym ::
   47   Bool ->
   48   Bool ->
   49   IO (B.ExprBuilder
   50       GlobalNonceGenerator
   51       St
   52       (B.Flags B.FloatReal))
   53 setupWhat4_sym hashConsing what4PushMuxOps =
   54   do -- TODO: get rid of GlobalNonceGenerator ???
   55      sym <- B.newExprBuilder B.FloatRealRepr St globalNonceGenerator
   56      let cfg = getConfiguration sym
   57      cacheTermsSetting <- getOptionSetting B.cacheTerms cfg
   58      _ <- setOpt cacheTermsSetting hashConsing
   59      pushMuxOpsSetting <- getOptionSetting B.pushMuxOpsOption cfg
   60      _ <- setOpt pushMuxOpsSetting what4PushMuxOps
   61      return sym

   63 what4Theories ::
   64   Set VarIndex ->
   65   Bool ->
   66   Sequent ->
   67   TopLevel [Text]
   68 what4Theories unintSet hashConsing goal = do
   69   sc <- getSharedContext
   70   what4PushMuxOps <- gets rwWhat4PushMuxOps
   71   io $ do
   72      sym <- setupWhat4_sym hashConsing what4PushMuxOps
   73      satq <- sequentToSATQuery sc unintSet goal
   74      (_varMap, lits) <- W.w4Solve sym sc satq
   75      let pf lit = (predicateVarInfo lit)^.problemFeatures
   76      return (nub (concatMap evalTheories (map pf lits)))
   77 
   78 evalTheories :: ProblemFeatures -> [Text]
   79 evalTheories pf = [ nm | (nm,f) <- xs, hasProblemFeature pf f ]
   80  where
   81   xs = [ ("LinearArithmetic", useLinearArithmetic)
   82        , ("NonlinearArithmetic", useNonlinearArithmetic)
   83        , ("TranscendentalFunctions", useComputableReals)
   84        , ("Integers", useIntegerArithmetic)
   85        , ("Bitvectors", useBitvectors)
   86        , ("ExistsForall", useExistForall)
   87        , ("FirstOrderQuantifiers", useQuantifiers)
   88        , ("Arrays", useSymbolicArrays)
   89        , ("Structs", useStructs)
   90        , ("Strings", useStrings)
   91        , ("FloatingPoint", useFloatingPoint)
   92        ]

   94 proveWhat4_sym ::
   95   SolverAdapter St ->
   96   Bool ->
   97   SATQuery ->
   98   TopLevel (Maybe CEX, Text)
   99 proveWhat4_sym solver hashConsing satq = do
  100   sc <- getSharedContext
  101   what4PushMuxOps <- gets rwWhat4PushMuxOps
  102   io $ do
  103      sym <- setupWhat4_sym hashConsing what4PushMuxOps
  104      proveWhat4_solver solver sym sc satq (return ())

  106 proveExportWhat4_sym ::
  107   SolverAdapter St ->
  108   Bool ->
  109   FilePath ->
  110   SATQuery ->
  111   TopLevel (Maybe CEX, Text)
  112 proveExportWhat4_sym solver hashConsing outFilePath satq = do
  113   sc <- getSharedContext
  114   what4PushMuxOps <- gets rwWhat4PushMuxOps
  115   io $ do
  116      sym <- setupWhat4_sym hashConsing what4PushMuxOps
  117 
  118      -- Write smt out
  119      (_, _, lits, solver_name) <- setupWhat4_solver solver sym sc satq
  120      withFile outFilePath WriteMode $ \handle ->
  121        solver_adapter_write_smt2 solver sym handle lits
  122 
  123      -- Assume unsat
  124      return (Nothing, solver_name)

  180 setupWhat4_solver :: forall st t ff.
  181   SolverAdapter st   {- ^ Which solver to use -} ->
  182   B.ExprBuilder t st ff {- ^ The glorious sym -}  ->
  183   SharedContext      {- ^ Context for working with terms -} ->
  184   SATQuery           {- ^ The query to be proved/checked. -} ->
  185   IO ( [VarName]
  186      , [W.Labeler (B.ExprBuilder t st ff)]
  187      , [Pred (B.ExprBuilder t st ff)]
  188      , Text)
  189 setupWhat4_solver solver sym sc satq =
  190   do
  191      -- symbolically evaluate
  192      let varList  = Map.toList (satVariables satq)
  193      let argNames = map fst varList
  194      (varMap, lits) <- W.w4Solve sym sc satq
  195      let bvs = map (fst . snd) varMap
  196 
  197      extendConfig (solver_adapter_config_options solver)
  198                   (getConfiguration sym)
  199 
  200      let solver_name = "W4 ->" <> (Text.pack $ solver_adapter_name solver)
  201 
  202      return (argNames, bvs, lits, solver_name)

  205 -- | Check the validity of a proposition using What4.
  206 proveWhat4_solver :: forall st t ff.
  207   SolverAdapter st   {- ^ Which solver to use -} ->
  208   B.ExprBuilder t st ff {- ^ The glorious sym -}  ->
  209   SharedContext      {- ^ Context for working with terms -} ->
  210   SATQuery           {- ^ The query to be proved/checked. -} ->
  211   IO ()              {- ^ Extra setup actions -} ->
  212   IO (Maybe CEX, Text)
  213   -- ^ (example/counter-example, solver statistics)
  214 proveWhat4_solver solver sym sc satq extraSetup =
  215   do
  216      (argNames, bvs, lits, solver_name) <- setupWhat4_solver solver sym sc satq
  217      extraSetup
  218 
  219      -- log to stdout
  220      let logger _ str = putStrLn str
  221      let logData = defaultLogData { logCallbackVerbose = logger
  222                                   , logReason = "SAW proof" }
  223 
  224      -- run solver
  225      solver_adapter_check_sat solver sym logData lits $ \ r -> case r of
  226          Sat (gndEvalFcn,_) -> do
  227            mvals <- mapM (getValues @(B.ExprBuilder t st ff) gndEvalFcn)
  228                          (zip bvs argNames)
  229            return (Just mvals, solver_name) where
  230 
  231          Unsat _ -> return (Nothing, solver_name)
  232 
  233          Unknown -> fail "Prover returned Unknown"

  237 getValues :: forall sym gt a. (SymExpr sym ~ B.Expr gt) => GroundEvalFn gt ->
  238   (W.Labeler sym, a) -> IO (a, FirstOrderValue)
  239 getValues f (labeler, orig) = do
  240   fov <- W.getLabelValues f labeler
  241   return (orig, fov)

  244 -- | For debugging
  245 printValue :: (B.ExprBuilder t st ff) -> GroundEvalFn t ->
  246   (Maybe (W.TypedExpr (B.ExprBuilder t st ff)), String) -> IO ()
  247 printValue _ _ (Nothing, _) = return ()
  248 printValue _ f (Just (W.TypedExpr (ty :: BaseTypeRepr ty) (bv :: B.Expr t ty)), orig) = do
  249   gv <- groundEval f @ty bv
  250   putStr $ orig ++ "=?"
  251   print (groundToFOV ty gv)
