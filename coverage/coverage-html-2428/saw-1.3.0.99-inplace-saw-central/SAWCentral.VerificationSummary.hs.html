   54 vsVerifSolvers :: VerificationSummary -> Set.Set Text
   55 vsVerifSolvers vs =
   56   Set.unions $
   57   map (\ms -> solverStatsSolvers (ms ^. psSolverStats)) (vsJVMMethodSpecs vs) ++
   58   map (\(CMSLLVM.SomeLLVM ms) -> solverStatsSolvers (ms ^. psSolverStats)) (vsLLVMMethodSpecs vs)
   59 
   60 vsTheoremSolvers :: VerificationSummary -> Set.Set Text
   61 vsTheoremSolvers = Set.unions . map getSolvers . vsTheorems
   62   where getSolvers thm = solverStatsSolvers (thmStats thm)
   63 
   64 vsAllSolvers :: VerificationSummary -> Set.Set Text
   65 vsAllSolvers vs = Set.union (vsVerifSolvers vs) (vsTheoremSolvers vs)

   67 computeVerificationSummary :: TheoremDB -> [JVMTheorem] -> [LLVMTheorem] -> [Theorem] -> VerificationSummary
   68 computeVerificationSummary db js ls thms =
   69   let roots = mconcat (
   70                 [ vcDeps vc | j <- js, vc <- j^.psVCStats ] ++
   71                 [ vcDeps vc | CMSLLVM.SomeLLVM l <- ls, vc <- l^.psVCStats ] ++
   72                 [ Set.singleton (thmNonce t) | t <- thms ])
   73       thms' = Map.elems (reachableTheorems db roots)
   74   in  VerificationSummary js ls thms'

   76 -- TODO: we could make things instances of a ToJSON typeclass instead of using the two methods below.
   77 msToJSON :: forall ext . Pretty (MethodId ext) => CMS.ProvedSpec ext -> Value
   78 msToJSON cms = object [
   79     ("type" .= ("method" :: String))
   80     , ("id" .= (indexValue $ cms ^. psSpecIdent))
   81     , ("method" .= (show $ pretty $ cms ^. psSpec.csMethod))
   82     , ("loc" .= (show $ pretty $ plSourceLoc $ cms ^. psSpec.csLoc))
   83     , ("status" .= case cms ^. psProofMethod of
   84                      SpecAdmitted -> "assumed" :: String
   85                      SpecProved   -> "verified")
   86 --    , ("specification" .= ("unknown" :: String)) -- TODO
   87     , ("dependencies" .= toJSON (map indexValue (Set.toList (cms ^. psSpecDeps))))
   88     , ("vcs"          .= toJSON (map vcToJSON (cms ^. psVCStats)))
   89     , ("elapsedtime"  .= toJSON (cms ^. psElapsedTime))
   90     ]
   91 
   92 vcToJSON :: CMS.VCStats -> Value
   93 vcToJSON vc = object ([
   94   ("type"  .= ("vc" :: String))
   95   , ("id"  .= indexValue (vcIdent vc))
   96   , ("loc" .= show (conditionLoc (vcMetadata vc)))
   97   , ("reason" .= conditionType (vcMetadata vc))
   98   , ("elapsedtime" .= toJSON (vcElapsedTime vc))
   99   , ("dependencies" .= toJSON (map indexValue (Set.toList (vcDeps vc))))
  100   , ("tags" .= toJSON (Set.toList (conditionTags (vcMetadata vc))))
  101   ] ++ theoremStatus (vcThmSummary vc)
  102   )
  103 
  104 thmToJSON :: Theorem -> Value
  105 thmToJSON thm = object ([
  106     ("type" .= ("property" :: String))
  107     , ("id" .= (indexValue $ thmNonce thm))
  108     , ("loc" .= (show $ thmLocation thm))
  109     , ("reason" .= (thmReason thm))
  110     , ("dependencies" .= toJSON (map indexValue (Set.toList (thmDepends thm))))
  111     , ("elapsedtime"  .= toJSON (thmElapsedTime thm))
  112   ] ++ theoremStatus (thmSummary thm)
  113     ++ case thmProgramLoc thm of
  114          Nothing -> []
  115          Just ploc -> [("ploc" .= plocToJSON ploc)]
  116   )
  117 
  118 theoremStatus :: TheoremSummary -> [(Key,Value)]
  119 theoremStatus summary = case summary of
  120       ProvedTheorem stats ->
  121         [ ("status"  .= ("verified" :: String))
  122         , ("provers" .= toJSON (Set.toList (solverStatsSolvers stats)))
  123         ]
  124       TestedTheorem n ->
  125         [ ("status"   .= ("tested" :: String))
  126         , ("numtests" .= toJSON n)
  127         ]
  128       AdmittedTheorem msg ->
  129         [ ("status"   .= ("assumed" :: String))
  130         , ("admitmsg" .= msg)
  131         ]
  132 
  133 --    , ("term" .= (show $ ppProp PPS.defaultOpts $ thmProp thm))
  134 
  135 plocToJSON :: ProgramLoc -> Value
  136 plocToJSON ploc = object
  137   [ "function" .= toJSON (functionName (plFunction ploc))
  138   , "loc"      .= show (plSourceLoc ploc)
  139   ]

  142 jsonVerificationSummary :: VerificationSummary -> String
  143 jsonVerificationSummary (VerificationSummary jspecs lspecs thms) =
  144   BLU.toString $ encode vals where
  145     vals = foldr (++) [] [jvals, lvals, thmvals]
  146     jvals = msToJSON <$> jspecs
  147     lvals = (\(CMSLLVM.SomeLLVM ls) -> msToJSON ls) <$> lspecs -- TODO: why is the type annotation required here?
  148     thmvals = thmToJSON <$> thms

  150 prettyVerificationSummary :: PPS.Opts -> DisplayNameEnv -> VerificationSummary -> String
  151 prettyVerificationSummary ppOpts nenv vs@(VerificationSummary jspecs lspecs thms) =
  152   show $ vsep
  153   [ prettyJVMSpecs jspecs
  154   , prettyLLVMSpecs lspecs
  155   , prettyTheorems thms
  156   , prettySolvers (Set.toList (vsAllSolvers vs))
  157   ] where
  158       section nm = "#" <+> nm
  159       item txt = "*" <+> txt
  160       code txt = vsep ["~~~~", txt, "~~~~"]
  161       subitem = indent 4 . item
  162       sectionWithItems _ _ [] = mempty
  163       sectionWithItems nm prt items =
  164         vsep [section nm, "", vsep (map prt items)]
  165       verifStatus s = case s ^. CMS.psProofMethod of
  166                         CMS.SpecAdmitted -> "assumed"
  167                         CMS.SpecProved   -> "verified"
  168       -- TODO: ultimately the goal is for the following to summarize all
  169       -- preconditions made by this verification, but we need to extract
  170       -- a bunch more information for that to be meaningful.
  171       {-
  172       condStatus s = (if null (s ^. (CMS.csPreState . CMS.csConditions))
  173                       then "without"
  174                       else "with") <+> "conditions"
  175                       -}
  176       prettyJVMSpecs ss =
  177         sectionWithItems "JVM Methods Analyzed" prettyJVMSpec ss
  178       prettyJVMSpec s =
  179         vsep [ item (fromString (s ^. CMS.psSpec.CMSJVM.csMethodName))
  180              -- , subitem (condStatus s)
  181              , subitem (verifStatus s)
  182              ]
  183       prettyLLVMSpecs ss =
  184         sectionWithItems "LLVM Functions Analyzed" prettyLLVMSpec ss
  185       prettyLLVMSpec (CMSLLVM.SomeLLVM s) =
  186         vsep [ item (fromString $ Text.unpack (s ^. CMS.psSpec.CMSLLVM.csName))
  187              -- , subitem (condStatus s)
  188              , subitem (verifStatus s)
  189              ]
  190       prettyTheorems ts =
  191         sectionWithItems "Theorems Proved or Assumed" (item . prettyTheorem) ts
  192       prettyTheorem t =
  193         vsep [ case thmSummary t of
  194                  ProvedTheorem{}   -> "Theorem:"
  195                  TestedTheorem n   -> "Theorem (randomly tested on" <+> viaShow n <+> "samples):"
  196                  AdmittedTheorem{} -> "Axiom:"
  197              , code (indent 2 (ppProp ppOpts nenv (thmProp t)))
  198              , ""
  199              ]
  200       prettySolvers ss =
  201         sectionWithItems "Solvers Used" (item . pretty) ss
