   26 import           Control.Monad (liftM)
   27 import qualified Data.IntMap as IntMap

   31 import           Data.Ref
   32 import           Prelude hiding (lookup)

   34 data Cache m k a = forall t. Cache (T m t) (k -> t -> Maybe a) (k -> a -> t -> t)
   35 
   36 useCache :: C m => Cache m k a -> k -> m a -> m a
   37 useCache (Cache ref lookup update) k action = do
   38   result <- liftM (lookup k) (Data.Ref.read ref)
   39   case result of
   40     Just x -> return x
   41     Nothing -> do
   42       x <- action
   43       modify ref (update k x)
   44       return x
   45 
   46 newCache :: (C m, Ord k) => m (Cache m k a)
   47 newCache = newCacheMap
   48 
   49 newCacheMap :: (C m, Ord k) => m (Cache m k a)
   50 newCacheMap = newCacheMap' Map.empty
   51 
   52 newCacheMap' :: (C m, Ord k) => Map k a -> m (Cache m k a)
   53 newCacheMap' initialMap = do
   54   ref <- new initialMap
   55   return (Cache ref Map.lookup Map.insert)
   56 
   57 newCacheIntMap :: (C m) => m (Cache m Int a)
   58 newCacheIntMap = newCacheIntMap' IntMap.empty
   59 
   60 newCacheIntMap' :: (C m) => IntMap a -> m (Cache m Int a)
   61 newCacheIntMap' initialMap = do
   62   ref <- new initialMap
   63   return (Cache ref IntMap.lookup IntMap.insert)
   64 
   65 newtype IntCache m a = IntCache (T m (IntMap a))
   66 
   67 useIntCache :: C m => IntCache m a -> Int -> m a -> m a
   68 useIntCache (IntCache ref) k action =
   69   do result <- IntMap.lookup k <$> Data.Ref.read ref
   70      case result of
   71        Just x -> pure x
   72        Nothing ->
   73          do x <- action
   74             modify ref (IntMap.insert k x)
   75             pure x
   76 
   77 newIntCache :: (C m) => m (IntCache m a)
   78 newIntCache = IntCache <$> new IntMap.empty
