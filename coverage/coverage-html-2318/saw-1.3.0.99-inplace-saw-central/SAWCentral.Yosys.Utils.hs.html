   51 data YosysError
   52   = YosysError Text
   53   | YosysErrorTypeError Text Text
   54   | YosysErrorNoSuchOutputBitvec Text YosysBitvecConsumer
   55   | YosysErrorNoSuchSubmodule Text Text
   56   | YosysErrorUnsupportedFF Text
   57   | YosysErrorInvalidOverrideTarget
   58   | YosysErrorOverrideNameNotFound Text
   59   | YosysErrorInvalidStateFieldWidth Text
   60   | YosysErrorTransitionSystemMissingField Text
   61   | YosysErrorTransitionSystemBadType
   62 instance Exception YosysError
   63 instance Show YosysError where
   64   show (YosysError msg) = Text.unpack $ "Error: " <> msg <> "\n" <> reportBugText
   65   show (YosysErrorTypeError msg err) = Text.unpack $ mconcat
   66     [ "Error: An internal term failed to type-check.\n"
   67     , "This occured while ", msg, ".\n"
   68     , "The type error was:\n", err
   69     , "This may represent a bug in SAW.\n"
   70     , reportBugText
   71     ]
   72   show (YosysErrorNoSuchOutputBitvec bvec (YosysBitvecConsumerOutputPort onm)) = Text.unpack $ mconcat
   73     [ "Error: Could not find the output bitvector ", bvec
   74     , ", which is connected to a module output port \"", onm
   75     , "\".\n"
   76     , "This may represent a bug in SAW.\n"
   77     , reportBugText
   78     ]
   79   show (YosysErrorNoSuchOutputBitvec bvec (YosysBitvecConsumerCell cnm inm)) = Text.unpack $ mconcat
   80     [ "Error: Could not find the output bitvector ", bvec
   81     , ", which is connected to the input \"", inm
   82     , "\" of the cell with name \"", cnm
   83     , "\".\n"
   84     , "It is possible that this represents an undetected cycle in the netgraph.\n"
   85     , reportBugText
   86     ]
   87   show (YosysErrorNoSuchSubmodule submoduleName cellName) =
   88     Text.unpack $ mconcat
   89       [ "Error: Encountered a cell \""
   90       , cellName
   91       , "\" with type \""
   92       , submoduleName
   93       , "\", but could not find a submodule named \""
   94       , submoduleName
   95       , "\"."
   96       ]
   97   show (YosysErrorUnsupportedFF mnm) = Text.unpack $ mconcat
   98     [ "Error: The circuit contains cells with type \"", mnm, "\".\n"
   99     , "These cells are not currently supported by SAW.\n"
  100     , "It may be helpful to replace certain stateful cells using the \"memory\", \"dffunmap\", and \"async2sync\" tactics within Yosys.\n"
  101     , consultYosysManual
  102     ]
  103   show YosysErrorInvalidOverrideTarget = Text.unpack $ mconcat
  104     [ "Error: The first argument to \"yosys_verify\" could not be identified as a Yosys module.\n"
  105     , "This argument should typically take the form {{ m.module_name }}, where \"m\" is a record term returned by \"yosys_import\""
  106     ]
  107   show (YosysErrorOverrideNameNotFound nm) = Text.unpack $ mconcat
  108     [ "Error: The name \"", nm, "\" could not be found while applying overrides.\n"
  109     , "This may represent a bug in SAW.\n"
  110     , reportBugText
  111     ]
  112   show (YosysErrorInvalidStateFieldWidth nm) = Text.unpack $ mconcat
  113     [ "Error: The state field \"", nm, "\" has an invalid width.\n"
  114     , "This may represent a bug in SAW.\n"
  115     , reportBugText
  116     ]
  117   show (YosysErrorTransitionSystemMissingField nm) = Text.unpack $ mconcat
  118     [ "Error: While translating a sequential circuit to a Sally transition system, could not find the field \"", nm, "\".\n"
  119     , "This may represent a bug in SAW.\n"
  120     , reportBugText
  121     ]
  122   show YosysErrorTransitionSystemBadType = Text.unpack $ mconcat
  123     [ "Error: While translating a sequential circuit to a Sally transition system, an intermediate What4 predicate was not a boolean.\n"
  124     , "This may represent a bug in SAW.\n"
  125     , reportBugText
  126     ]

  139 -- | Check that a SAWCore term is well-typed, throwing an exception otherwise.
  140 validateTerm :: MonadIO m => SC.SharedContext -> Text -> SC.Term -> m SC.Term
  141 validateTerm sc msg t = liftIO (SC.TC.scTypeCheck sc Nothing t) >>= \case
  142   Right _ -> pure t
  143   Left err ->
  144     throw
  145     . YosysErrorTypeError msg
  146     . Text.pack
  147     . unlines
  148     $ SC.TC.prettyTCError err
  149 
  150 -- | Check that a SAWCore term is well-typed and has a specific type
  151 validateTermAtType :: MonadIO m => SC.SharedContext -> Text ->
  152                       SC.Term -> SC.Term -> m ()
  153 validateTermAtType sc msg trm tp =
  154   liftIO (SC.TC.runTCM (SC.TC.typeInferComplete trm >>= \tp_trm ->
  155                          SC.TC.checkSubtype tp_trm tp) sc Nothing []) >>= \case
  156   Right _ -> return ()
  157   Left err ->
  158     throw
  159     . YosysErrorTypeError msg
  160     . Text.pack
  161     . unlines
  162     $ SC.TC.prettyTCError err

  164 -- | Produce a SAWCore tuple type corresponding to a Cryptol record type with the given fields.
  165 cryptolRecordType ::
  166   MonadIO m =>
  167   SC.SharedContext ->
  168   Map Text SC.Term ->
  169   m SC.Term
  170 cryptolRecordType sc fields =
  171   liftIO $ SC.scTupleType sc (fmap snd . C.canonicalFields . C.recordFromFields $ Map.assocs fields)

  173 -- | Produce a SAWCore tuple corresponding to a Cryptol record with the given fields.
  174 cryptolRecord ::
  175   MonadIO m =>

  177   Map Text SC.Term ->
  178   m SC.Term
  179 cryptolRecord sc fields =
  180   liftIO $ SC.scTuple sc (fmap snd . C.canonicalFields . C.recordFromFields $ Map.assocs fields)

  182 -- | Produce a SAWCore tuple index corresponding to a lookup in a Cryptol record with the given fields.
  183 cryptolRecordSelect ::
  184   MonadIO m =>
  185   SC.SharedContext ->
  186   Map Text a ->
  187   SC.Term ->
  188   Text ->
  189   m SC.Term
  190 cryptolRecordSelect sc fields r nm =
  191   case List.elemIndex nm ord of
  192     Just i -> liftIO $ SC.scTupleSelector sc r (i + 1) (length ord)
  193     Nothing -> throw . YosysError $ mconcat
  194       [ "Could not build record selector term for field name \""
  195       , nm
  196       , "\" on record term: "
  197       , Text.pack $ SC.showTerm r
  198       , "\nFields are: "
  199       , Text.pack . show $ Map.keys fields
  200       ]
  201   where ord = fmap fst . C.canonicalFields . C.recordFromFields $ Map.assocs fields
  202 
  203 -- | Produce a SAWCore tuple index corresponding to a lookup in a Cryptol record.
  204 -- The record fields are inferred from the Cryptol type attached to the `TypedTerm`.
  205 cryptolRecordSelectTyped ::
  206   MonadIO m =>
  207   SC.SharedContext ->
  208   SC.TypedTerm ->
  209   Text ->
  210   m SC.TypedTerm
  211 cryptolRecordSelectTyped sc r nm = do
  212   fields <- Map.mapKeys C.identText . Map.fromList . C.canonicalFields <$> case SC.ttType r of
  213     SC.TypedTermSchema (C.Forall [] [] (C.TRec fs)) -> pure fs
  214     _ -> throw . YosysError $ mconcat
  215       [ "Type\n"
  216       , Text.pack . show $ SC.ttType r
  217       , "\nis not a record type"
  218       ]
  219   cty <- case Map.lookup nm fields of
  220     Just cty -> pure cty
  221     _ -> throw . YosysError $ mconcat
  222       [ "Record type\n"
  223       , Text.pack . show $ SC.ttType r
  224       , "\ndoes not have field "
  225       , nm
  226       ]
  227   t <- cryptolRecordSelect sc fields (SC.ttTerm r) nm
  228   pure $ SC.TypedTerm (SC.TypedTermSchema $ C.tMono cty) t

  230 -- | Construct a SAWCore expression asserting equality between each field of two records.
  231 -- Both records should be tuples corresponding to the specified Cryptol record type.
  232 eqBvRecords ::
  233   (MonadIO m, MonadThrow m) =>
  234   SC.SharedContext ->
  235   C.Type ->
  236   SC.Term ->
  237   SC.Term ->
  238   m SC.Term
  239 eqBvRecords sc cty a b = do
  240   fields <- Map.mapKeys C.identText . Map.fromList . C.canonicalFields <$> case cty of
  241     C.TRec fs -> pure fs
  242     _ -> throw . YosysError $ mconcat
  243       [ "Type\n"
  244       , Text.pack $ show cty
  245       , "\nis not a record type"
  246       ]
  247   eqs <- forM (Map.assocs fields) $ \(nm, fcty) -> do
  248     w <- case fcty of
  249       C.TCon (C.TC C.TCSeq) [C.TCon (C.TC (C.TCNum wint)) [], C.TCon (C.TC C.TCBit) []] ->
  250         liftIO . SC.scNat sc $ fromIntegral wint
  251       _ -> throw . YosysError $ mconcat
  252         [ "Type\n"
  253         , Text.pack $ show fcty
  254         , "\nis not a bitvector type"
  255         ]
  256     fa <- cryptolRecordSelect sc fields a nm
  257     fb <- cryptolRecordSelect sc fields b nm
  258     liftIO $ SC.scBvEq sc w fa fb
  259   case eqs of
  260     [] -> throw . YosysError $ mconcat
  261       [ "Record type\n"
  262       , Text.pack $ show cty
  263       , "\nhas no fields"
  264       ]
  265     (e:es) -> foldM (\x y -> liftIO $ SC.scAnd sc x y) e es

  267 -- | Encode the given string such that is a valid Cryptol identifier. 
  268 -- Since Yosys cell names often look like "\42", this makes it much easier to manipulate state records,
  269 -- which are keyed by cell name.
  270 cellIdentifier :: Text -> Text
  271 cellIdentifier = Text.pack . zEncodeString . Text.unpack

  281 -- | Build a Cryptol record type with the given field types

  283   MonadIO m =>

  288 -- | Given a bit pattern ([b]) and a term, construct a map associating that output pattern with
  289 -- the term, and each bit of that pattern with the corresponding bit of the term.
  290 deriveTermsByIndices :: (MonadIO m, Ord b) => SC.SharedContext -> [b] -> SC.Term -> m (Map [b] SC.Term)
  291 deriveTermsByIndices sc rep t = do
  292   boolty <- liftIO $ SC.scBoolType sc
  293   -- Yosys uses little-endian order while saw uses big-endian (msb at index 0), so reverse indices
  294   telems <- forM (reverse [0..length rep - 1]) $ \index -> do
  295     tlen <- liftIO . SC.scNat sc . fromIntegral $ length rep
  296     idx <- liftIO . SC.scNat sc $ fromIntegral index
  297     bit <- liftIO $ SC.scAt sc tlen boolty t idx
  298     liftIO $ SC.scSingle sc boolty bit
  299   pure . Map.fromList $ mconcat
  300     [ [(rep, t)]
  301     , zip ((:[]) <$> rep) telems
  302     ]
