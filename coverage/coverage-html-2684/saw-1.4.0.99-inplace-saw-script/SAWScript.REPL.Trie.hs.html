    9 module SAWScript.REPL.Trie where
   10 
   11 import           Cryptol.Utils.Panic (panic)
   12 import qualified Data.Map as Map
   13 import           Data.Map (Map)
   14 import           Data.Maybe (fromMaybe,maybeToList)
   15 
   16 
   17 -- | Maps string names to values, allowing for partial key matches and querying.
   18 data Trie a = Node (Map Char (Trie a)) (Maybe a)
   19     deriving (Show)
   20 
   21 emptyTrie :: Trie a
   22 emptyTrie  = Node Map.empty Nothing
   23 
   24 -- | Insert a value into the Trie.  Will call `panic` if a value already exists
   25 -- with that key.
   26 insertTrie :: String -> a -> Trie a -> Trie a
   27 insertTrie k a = loop k
   28   where
   29   loop key (Node m mb) = case key of
   30     c:cs -> Node (Map.alter (Just . loop cs . fromMaybe emptyTrie) c m) mb
   31     []   -> case mb of
   32       Nothing -> Node m (Just a)
   33       Just _  -> panic "[REPL] Trie" ["key already exists:", "\t" ++ k]
   34 
   35 -- | Return all matches with the given prefix.
   36 lookupTrie :: String -> Trie a -> [a]
   37 lookupTrie key t@(Node mp _) = case key of
   38 
   39   c:cs -> case Map.lookup c mp of
   40     Just m' -> lookupTrie cs m'
   41     Nothing -> []
   42 
   43   [] -> leaves t
   44 
   45 -- | Return all matches with the given prefix. However, if an exact match
   46 -- exists, return just that match.
   47 lookupTrieWithExact :: String -> Trie a -> [a]
   48 lookupTrieWithExact key t@(Node mp mb) = case key of
   49 
   50   c:cs -> case Map.lookup c mp of
   51     Just m' -> lookupTrieWithExact cs m'
   52     Nothing -> []
   53 
   54   [] -> case mb of
   55     Just b -> [b]
   56     Nothing -> leaves t
   57 
   58 -- | Return all of the values from a Trie.
   59 leaves :: Trie a -> [a]
   60 leaves (Node mp mb) = maybeToList mb ++ concatMap leaves (Map.elems mp)
