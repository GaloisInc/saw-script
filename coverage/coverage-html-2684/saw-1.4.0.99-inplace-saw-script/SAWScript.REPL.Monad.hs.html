  113 -- REPL Environment.
  114 data Refs = Refs
  115   { rContinue   :: IORef Bool
  116   , rIsBatch    :: Bool
  117   , rTopLevelRO :: TopLevelRO
  118   , rTopLevelRW :: IORef TopLevelRW
  119   , rProofState :: Maybe (IORef ProofState)
  120   }
  121 
  122 -- | Initial, empty environment.
  123 defaultRefs :: Bool -> Options -> IO Refs
  124 defaultRefs isBatch opts =
  125   do (_biContext, ro, rw) <- buildTopLevelEnv (AIGProxy AIG.compactProxy) opts []
  126      contRef <- newIORef True
  127      rwRef <- newIORef rw
  128      return Refs
  129        { rContinue   = contRef
  130        , rIsBatch    = isBatch
  131        , rTopLevelRO = ro
  132        , rTopLevelRW = rwRef
  133        , rProofState  = Nothing
  134        }
  135 
  136 
  137 mkTitle :: Refs -> String
  138 mkTitle _refs = "sawscript"

  143 -- | REPL_ context with InputT handling.
  144 newtype REPL a = REPL { unREPL :: Refs -> IO a }
  145 
  146 -- | Run a REPL action with a fresh environment.
  147 runREPL :: Bool -> Options -> REPL a -> IO a
  148 runREPL isBatch opts m =
  149   do refs <- defaultRefs isBatch opts
  150      unREPL m refs
  151 
  152 subshell :: REPL () -> TopLevel ()
  153 subshell (REPL m) = TopLevel_ $
  154   do ro <- ask
  155      rw <- get
  156      rw' <- liftIO $
  157        do contRef <- newIORef True
  158           rwRef <- newIORef rw
  159           let refs = Refs
  160                      { rContinue = contRef
  161                      , rIsBatch  = False
  162                      , rTopLevelRO = ro
  163                      , rTopLevelRW = rwRef
  164                      , rProofState  = Nothing
  165                      }
  166           m refs
  167           readIORef rwRef
  168      put rw'
  169 
  170 proof_subshell :: REPL () -> ProofScript ()
  171 proof_subshell (REPL m) =
  172   ProofScript $ ExceptT $ StateT $ \proofSt ->
  173   do ro <- ask
  174      rw <- get
  175      (rw', outProofSt) <- liftIO $
  176        do contRef <- newIORef True
  177           rwRef <- newIORef rw
  178           proofRef <- newIORef proofSt
  179           let refs = Refs
  180                      { rContinue = contRef
  181                      , rIsBatch  = False
  182                      , rTopLevelRO = ro
  183                      , rTopLevelRW = rwRef
  184                      , rProofState  = Just proofRef
  185                      }
  186           m refs
  187           (,) <$> readIORef rwRef <*> readIORef proofRef
  188      put rw'
  189      return (Right (), outProofSt)
  190 
  191 instance Functor REPL where
  192   {-# INLINE fmap #-}
  193   fmap f m = REPL (\ ref -> fmap f (unREPL m ref))
  194 
  195 instance Monad REPL where
  196   {-# INLINE return #-}
  197   return = pure
  198 
  199   {-# INLINE (>>=) #-}
  200   m >>= f = REPL $ \ref -> do
  201     x <- unREPL m ref
  202     unREPL (f x) ref
  203 
  204 instance Fail.MonadFail REPL where
  205   {-# INLINE fail #-}
  206   fail msg = REPL (\_ -> fail msg)
  207 
  208 instance Applicative REPL where
  209   {-# INLINE pure #-}
  210   pure x = REPL (\_ -> pure x)
  211   {-# INLINE (<*>) #-}
  212   (<*>) = ap

  216 -- | REPL exceptions.
  217 data REPLException
  218   = ParseError ParseError
  219   | FileNotFound FilePath
  220   | DirectoryNotFound FilePath
  221   | NoPatError [Error]
  222   | NoIncludeError [IncludeError]
  223   | EvalError EvalError
  224   | ModuleSystemError M.ModuleError
  225   | EvalPolyError T.Schema
  226   | TypeNotTestable T.Type
  227     deriving (Show,Typeable)
  228 
  229 instance X.Exception REPLException
  230 
  231 instance PP REPLException where
  232   ppPrec _ re = case re of
  233     ParseError e         -> ppError e
  234     FileNotFound path    -> sep [ text "File"
  235                                 , text ("`" ++ path ++ "'")
  236                                 , text"not found"
  237                                 ]
  238     DirectoryNotFound path -> sep [ text "Directory"
  239                                   , text ("`" ++ path ++ "'")
  240                                   , text"not found or not a directory"
  241                                   ]
  242     NoPatError es        -> vcat (map pp es)
  243     NoIncludeError es    -> vcat (map ppIncludeError es)
  244     ModuleSystemError me -> pp me
  245     EvalError e          -> pp e
  246     EvalPolyError s      -> text "Cannot evaluate polymorphic value."
  247                          $$ text "Type:" <+> pp s
  248     TypeNotTestable t    -> text "The expression is not of a testable type."
  249                          $$ text "Type:" <+> pp t
  250 
  251 -- | Raise an exception.
  252 raise :: REPLException -> REPL a
  253 raise exn = io (X.throwIO exn)
  254 
  255 -- | Handle any exception type in 'REPL' actions.
  256 catchEx :: X.Exception e => REPL a -> (e -> REPL a) -> REPL a
  257 catchEx m k = REPL (\ ref -> unREPL m ref `X.catch` \ e -> unREPL (k e) ref)
  258 
  259 -- | Handle 'REPLException' exceptions in 'REPL' actions.
  260 catch :: REPL a -> (REPLException -> REPL a) -> REPL a
  261 catch = catchEx
  262 
  263 -- | Similar to 'catch' above, but catches generic IO exceptions from 'fail'.
  264 catchFail :: REPL a -> (String -> REPL a) -> REPL a
  265 catchFail m k = REPL (\ ref -> X.catchJust sel (unREPL m ref) (\s -> unREPL (k s) ref))
  266   where
  267     sel :: X.IOException -> Maybe String
  268     sel e | isUserError e = Just (ioeGetErrorString e)
  269           | otherwise     = Nothing
  270 
  271 -- | Handle any other exception (except that we ignore async exceptions and exitWith)
  272 catchOther :: REPL a -> (X.SomeException -> REPL a) -> REPL a
  273 catchOther m k = REPL (\ref -> X.catchJust flt (unREPL m ref) (\s -> unREPL (k s) ref))
  274  where
  275   flt e
  276     | Just (_ :: X.AsyncException) <- X.fromException e = Nothing
  277     | Just (_ :: ExitCode)       <- X.fromException e = Nothing
  278     | otherwise = Just e
  279 
  280 rethrowEvalError :: IO a -> IO a
  281 rethrowEvalError m = run `X.catch` rethrow
  282   where
  283   run = do
  284     a <- m
  285     return $! a
  286 
  287   rethrow :: EvalErrorEx -> IO a
  288   rethrow (EvalErrorEx _ exn) = X.throwIO (EvalError exn)

  291 exceptionProtect :: REPL () -> REPL ()
  292 exceptionProtect cmd =
  293       do chk <- io . makeCheckpoint =<< getTopLevelRW
  294          cmd `catch`      (handlerPP chk)
  295              `catchFail`  (handlerFail chk)
  296              `catchOther` (handlerPrint chk)
  297 
  298     where
  299     handlerPP chk re =
  300       do io (putStrLn "" >> print (pp re))
  301          void $ liftTopLevel (restoreCheckpoint chk)
  302          return ()
  303     handlerPrint chk e =
  304       do io (putStrLn "" >> putStrLn (X.displayException e))
  305          void $ liftTopLevel (restoreCheckpoint chk)
  306 
  307     handlerFail chk s =
  308       do io (putStrLn "" >> putStrLn s)
  309          void $ liftTopLevel (restoreCheckpoint chk)
  310 
  311 liftTopLevel :: TopLevel a -> REPL a
  312 liftTopLevel m =
  313   do ro  <- getTopLevelRO
  314      ref <- getTopLevelRWRef
  315      io $ do rw <- readIORef ref
  316              (a, rw') <- runTopLevel m ro rw
  317              writeIORef ref rw'
  318              return a
  319 
  320 liftProofScript :: ProofScript a -> IORef ProofState -> REPL a
  321 liftProofScript m ref =
  322   liftTopLevel $
  323   do st <- liftIO $ readIORef ref
  324      (res, st') <- runStateT (runExceptT (unProofScript m)) st
  325      liftIO $ writeIORef ref st'
  326      case res of
  327        Left (stats, cex) ->
  328          do ppOpts <- rwPPOpts <$> get
  329             fail (showsProofResult ppOpts (InvalidProof stats cex st') "")
  330        Right x -> return x

  346 -- | Construct the prompt for the current environment.
  347 getPrompt :: REPL String
  348 getPrompt =
  349   do batch <- REPL (return . rIsBatch)
  350      if batch then return ""
  351      else
  352        getProofStateRef >>= \case
  353          Nothing -> return "sawscript> "
  354          Just psr ->
  355            do ps <- io (readIORef psr)
  356               return ("proof ("++show (length (psGoals ps))++")> ")
  357 
  358 shouldContinue :: REPL Bool
  359 shouldContinue = readRef rContinue

  361 stop :: REPL ()
  362 stop = modifyRef rContinue (const False)

  364 unlessBatch :: REPL () -> REPL ()
  365 unlessBatch body =
  366   do batch <- REPL (return . rIsBatch)

  375 getVars :: REPL (Map T.Name M.IfaceDecl)
  376 getVars  = do
  377   me <- getModuleEnv
  378   let decls = getAllIfaceDecls me
  379   let vars1 = M.ifDecls decls
  380   extras <- getExtraTypes
  381   let vars2 = Map.mapWithKey (\q s -> M.IfaceDecl { M.ifDeclName = q
  382                                                   , M.ifDeclSig = s
  383                                                   , M.ifDeclIsPrim = False
  384                                                   , M.ifDeclPragmas = []
  385                                                   , M.ifDeclInfix = False
  386                                                   , M.ifDeclFixity = Nothing
  387                                                   , M.ifDeclDoc = Nothing
  388                                                   }) extras
  389   return (Map.union vars1 vars2)

  490 -- | Get visible variable names for Haskeline completion.
  491 getSAWScriptValueNames :: REPL [String]
  492 getSAWScriptValueNames = do
  493   rw <- getTopLevelRW
  494   let avail = rwPrimsAvail rw
  495       visible (_, lc, _, _, _) = Set.member lc avail
  496       Environ valenv _tyenv _cryenv = rwEnviron rw
  497       rbenv = rwRebindables rw
  498   let rnames1 = ScopedMap.allKeys $ ScopedMap.filter visible valenv
  499       rnames2 = Map.keys rbenv
  500   return (map Text.unpack (rnames1 ++ rnames2))
  501 
  502 -- | Get visible type names for Haskeline completion.
  503 getSAWScriptTypeNames :: REPL [String]
  504 getSAWScriptTypeNames = do
  505   rw <- getTopLevelRW
  506   let avail = rwPrimsAvail rw
  507       visible (lc, _) = Set.member lc avail
  508       Environ _valenv tyenv _cryenv = rwEnviron rw
  509   let rnames = ScopedMap.allKeys $ ScopedMap.filter visible tyenv
  510   return (map Text.unpack rnames)
