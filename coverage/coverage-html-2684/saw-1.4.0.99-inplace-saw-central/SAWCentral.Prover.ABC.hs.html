   49 -- | Bit-blast a proposition and check its validity using ABC.
   50 proveABC ::
   51   (AIG.IsAIG l g) =>
   52   AIG.Proxy l g ->
   53   SATQuery ->
   54   TopLevel (Maybe CEX, Text)
   55 proveABC proxy satq = getSharedContext >>= \sc -> liftIO $
   56   BBSim.withBitBlastedSATQuery proxy sc mempty satq $ \be lit shapes ->
   57     do let (ecs,fts) = unzip shapes
   58        res <- getModel ecs fts =<< AIG.checkSat be lit
   59        return (res, "ABC")
   60 
   61 
   62 getModel ::
   63   Show name =>
   64   [name] ->
   65   [FiniteType] ->
   66   AIG.SatResult ->
   67   IO (Maybe [(name, FirstOrderValue)])
   68 getModel argNames shapes satRes =
   69   case satRes of
   70     AIG.Unsat -> return Nothing
   71 
   72     AIG.Sat cex -> do
   73       case liftCexBB shapes cex of
   74         Left err -> fail ("Can't parse counterexample: " ++ err)
   75         Right vs
   76 
   77           | length argNames == length vs ->
   78             return (Just (zip argNames (map toFirstOrderValue vs)))
   79 
   80           | otherwise ->
   81               fail $ unwords [ "ABC SAT results do not match expected arguments"
   82                              , show argNames, show vs]
   83 
   84     AIG.SatUnknown -> fail "Unknown result from ABC"

  104 w4AbcExternal ::
  105   (FilePath -> SATQuery -> TopLevel [(VarName, FiniteType)]) ->
  106   (String -> String -> String) ->
  107   Bool ->
  108   SATQuery ->
  109   TopLevel (Maybe CEX, Text)
  110 w4AbcExternal exporter argFn _hashcons satq =
  111        -- Create a temporary directory to put input and output files
  112     do canonTmpDir <- liftIO getCanonicalTemporaryDirectory
  113        tmpDir <- liftIO $ createTempDirectory canonTmpDir "saw_abc_external"
  114        let tmp = tmpDir </> "abc_verilog.v"
  115            tmpCex = tmpDir </> "abc_verilog.cex"
  116 
  117        (argNames, argTys) <- unzip <$> exporter tmp satq
  118 
  119        -- Run ABC and remove temporaries
  120        let execName = "abc"
  121            args = ["-q", argFn tmp tmpCex]
  122        (_out, _err) <- liftIO $ readProcessExitIfFailure execName args
  123        cexExists <- liftIO $ doesFileExist tmpCex
  124        res <-
  125          if cexExists
  126            then do
  127              -- If a counterexample file exists, parse and report the results.
  128              cexText <- liftIO $ C8.unpack <$> C8.readFile tmpCex
  129              cex <- liftIO $ parseAigerCex cexText argTys
  130              case cex of
  131                Left parseErr -> fail parseErr
  132                Right vs -> return $ Just model
  133                  where model = zip argNames (map toFirstOrderValue vs)
  134            else
  135              -- Otherwise, there is no counterexample.
  136              return Nothing
  137        liftIO $ removeDirectoryRecursive tmpDir
  138        return (res, "abc_verilog")

  140 parseAigerCex :: String -> [FiniteType] -> IO (Either String [FiniteValue])
  141 parseAigerCex text tys =
  142   case lines text of
  143     -- Output from `write_cex`
  144     [cex] ->
  145       case words cex of
  146         -- Queries with no variables will result in a blank line of output.
  147         [] -> pure $ liftCexBB tys []
  148         [bits] -> liftCexBB tys <$> mapM bitToBool bits
  149         _ -> fail $ "invalid counterexample line: " ++ cex
  150     -- Output from `write_aiger_cex`
  151     [_, cex] ->
  152       case words cex of
  153         -- Queries with no variables will result in an empty vector of output.
  154         [_] -> pure $ liftCexBB tys []
  155         [bits, _] -> liftLECexBB tys <$> mapM bitToBool bits
  156         _ -> fail $ "invalid counterexample line: " ++ cex
  157     _ -> fail $ "invalid counterexample text: " ++ text
  158   where
  159     bitToBool '0' = return False
  160     bitToBool '1' = return True
  161     bitToBool c   = fail ("invalid bit: " ++ [c])

  164 abcSatExternal :: MonadIO m =>
  165   (AIG.IsAIG l g) =>
  166   AIG.Proxy l g ->
  167   SharedContext ->
  168   Bool ->
  169   String ->
  170   [String] ->
  171   ProofGoal ->
  172   m (Maybe CEX, SolverStats)
  173 abcSatExternal proxy sc doCNF execName args g = liftIO $
  174   do satq <- sequentToSATQuery sc mempty (goalSequent g)
  175      let cnfName = goalType g ++ show (goalNum g) ++ ".cnf"
  176      (path, fh) <- openTempFile "." cnfName
  177      hClose fh -- Yuck. TODO: allow writeCNF et al. to work on handles.
  178 
  179      let args' = map replaceFileName args
  180          replaceFileName "%f" = path
  181          replaceFileName a = a
  182 
  183      BBSim.withBitBlastedSATQuery proxy sc mempty satq $ \be l shapes -> do
  184        variables <- (if doCNF then AIG.writeCNF else writeAIGWithMapping) be l path
  185        (_ec, out, err) <- readProcessWithExitCode execName args' ""
  186        removeFile path
  187        unless (null err) $
  188          print $ unlines ["Standard error from SAT solver:", err]
  189        let ls = lines out
  190            sls = filter ("s " `isPrefixOf`) ls
  191            vls = filter ("v " `isPrefixOf`) ls
  192        let stats = solverStats ("external SAT: " <> Text.pack execName) (sequentSharedSize (goalSequent g))
  193        case (sls, vls) of
  194          (["s SATISFIABLE"], _) -> do
  195            let bs = parseDimacsSolution variables vls
  196            let r = liftCexBB (map snd shapes) bs
  197                argNames = map (Text.unpack . vnName . fst) shapes
  198                ecs = map fst shapes
  199            case r of
  200              Left msg -> fail $ "Can't parse counterexample: " ++ msg
  201              Right vs
  202                | length ecs == length vs -> do
  203                  return (Just (zip ecs (map toFirstOrderValue vs)), stats)
  204                | otherwise -> fail $ unwords ["external SAT results do not match expected arguments", show argNames, show vs]
  205          (["s UNSATISFIABLE"], []) ->
  206            return (Nothing, stats)
  207          _ -> fail $ "Unexpected result from SAT solver:\n" ++ out
  208 
  209 parseDimacsSolution :: [Int]    -- ^ The list of CNF variables to return
  210                     -> [String] -- ^ The value lines from the solver
  211                     -> [Bool]
  212 parseDimacsSolution vars ls = map lkup vars
  213   where
  214     vs :: [Int]
  215     vs = concatMap (filter (/= 0) . mapMaybe readMaybe . parseLine) ls
  216     varToPair n | n < 0 = (-n, False)
  217                 | otherwise = (n, True)
  218     assgnMap = Map.fromList (map varToPair vs)
  219     lkup v = Map.findWithDefault False v assgnMap
  220 
  221     parseLine :: String -> [String]
  222     parseLine line =
  223       case words line of
  224         _:ws -> ws
  225         [] -> panic "parseDimacsSolution" [
  226                   "DIMACS solution has line with no values"
  227               ]

  229 writeAIGWithMapping :: AIG.IsAIG l g => g s -> l s -> FilePath -> IO [Int]
  230 writeAIGWithMapping be l path = do
  231   nins <- AIG.inputCount be
  232   AIG.writeAiger path (AIG.Network be [l])
  233   return [1..nins]
