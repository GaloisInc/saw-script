   95 -- Simulator configuration

   97 data SimulatorConfig l =
   98   SimulatorConfig
   99   { simPrimitive :: Name -> MValue l
  100   -- ^ Interpretation of 'Primitive' terms.
  101   , simVariable :: Term -> VarName -> TValue l -> MValue l
  102   -- ^ Interpretation of free 'Variable' terms.
  103   , simConstant :: Name -> TValue l -> Maybe (MValue l)
  104   -- ^ Interpretation of 'Constant' terms. 'Nothing' indicates that
  105   -- the body of the constant should be evaluated. 'Just' indicates
  106   -- that the constant's definition should be overridden.
  107   , simRecursor :: Name -> Sort -> Maybe (MValue l)
  108   -- ^ Interpretation of 'Recursor' terms. 'Nothing' indicates that
  109   -- the generic recursor implementation should be used, while 'Just'
  110   -- indicates that the recursor should be overridden.
  111   , simModMap :: ModuleMap
  112   , simLazyMux :: VBool l -> MValue l -> MValue l -> MValue l
  113   }

  140 -- | Generic evaluator for TermFs.
  141 evalTermF :: forall l. (VMonadLazy l, Show (Extra l)) =>
  142              SimulatorConfig l          -- ^ Evaluator for global constants
  143           -> (Term -> OpenValue l)      -- ^ Evaluator for subterms under binders
  144           -> (Term -> MValue l)         -- ^ Evaluator for subterms in the same bound variable context
  145           -> TermF Term -> OpenValue l
  146 evalTermF cfg lam recEval tf env =
  147   case tf of
  148     App t1 t2               -> recEval t1 >>= \case
  149                                  VFun f ->
  150                                    do x <- recEvalDelay t2
  151                                       f x
  152                                  _ -> panic "evalTermF" ["Expected VFun"]
  153     Lambda nm _tp t         -> pure $ VFun (\x -> lam t (IntMap.insert (vnIndex nm) x env))
  154     Pi nm t1 t2             -> do v <- evalType t1
  155                                   body <-
  156                                     if IntSet.member (vnIndex nm) (freeVars t2) then
  157                                       pure (VDependentPi (\x -> toTValue <$> lam t2 (IntMap.insert (vnIndex nm) x env)))
  158                                     else
  159                                       VNondependentPi . toTValue <$> lam t2 env
  160                                   return $ TValue $ VPiType v body
  161 
  162     Constant nm             -> do let r = requireNameInMap nm (simModMap cfg)
  163                                   ty' <- evalType (resolvedNameType r)
  164                                   case simConstant cfg nm ty' of
  165                                     Just override -> override
  166                                     Nothing ->
  167                                       case r of
  168                                         ResolvedCtor ctor ->
  169                                           ctorValue nm ty' (ctorNumParams ctor) (ctorNumArgs ctor)
  170                                         ResolvedDataType dt ->
  171                                           dtValue nm (dtNumParams dt) (dtNumIndices dt)
  172                                         ResolvedDef d ->
  173                                           case defBody d of
  174                                             Just t -> recEval t
  175                                             Nothing -> simPrimitive cfg nm
  176 
  177     Variable nm tp          -> do tp' <- evalType tp
  178                                   case IntMap.lookup (vnIndex nm) env of
  179                                     Nothing -> simVariable cfg tp nm tp'
  180                                     Just x -> force x
  181     FTermF ftf              ->
  182       case ftf of
  183         UnitValue           -> return VUnit
  184 
  185         UnitType            -> return $ TValue VUnitType
  186 
  187         PairValue x y       -> do tx <- recEvalDelay x
  188                                   ty <- recEvalDelay y
  189                                   return $ VPair tx ty
  190 
  191         PairType x y        -> do vx <- evalType x
  192                                   vy <- evalType y
  193                                   return $ TValue $ VPairType vx vy
  194 
  195         PairLeft x          -> recEval x >>= \case
  196                                  VPair l _r -> force l
  197                                  _ -> panic "evalTermF" ["Expected VPair"]
  198 
  199         PairRight x         -> recEval x >>= \case
  200                                  VPair _l r -> force r
  201                                  _ -> panic "evalTermF" ["Expected VPair"]
  202 
  203         Recursor r ->
  204           case simRecursor cfg (recursorDataType r) (recursorSort r) of
  205             Just v -> v
  206             Nothing ->
  207               do let dname = recursorDataType r
  208                  let nparams = recursorNumParams r
  209                  let nixs = recursorNumIxs r
  210                  let cnames = recursorCtorOrder r
  211                  vFunList nparams $ \_ps_thunks ->
  212                    pure $ VFun $ \_motive ->
  213                    vFunList (length cnames) $ \elim_thunks ->
  214                    do let es = Map.fromList (zip (map nameIndex cnames) elim_thunks)
  215                       let vrec = VRecursor dname nixs es
  216                       vFunList nixs (\_ixs -> pure (evalRecursor vrec))
  217 
  218         RecordType elem_tps ->
  219           TValue . VRecordType <$> traverse (traverse evalType) elem_tps
  220 
  221         RecordValue elems   ->
  222           VRecordValue <$> mapM (\(fld,t) -> (fld,) <$> recEvalDelay t) elems
  223 
  224         RecordProj t fld    -> recEval t >>= \case
  225                                  v@VRecordValue{} -> valRecordProj v fld
  226                                  _ -> panic "evalTermF" ["Expected VRecordValue"]
  227 
  228         Sort s _h           -> return $ TValue (VSort s)
  229 
  230         NatLit n            -> return $ VNat n
  231 
  232         ArrayValue _ tv     -> liftM VVector $ mapM recEvalDelay tv
  233 
  234         StringLit s         -> return $ VString s
  235   where
  236     evalType :: Term -> EvalM l (TValue l)
  237     evalType t = toTValue <$> recEval t
  238 
  239     toTValue :: HasCallStack => Value l -> TValue l
  240     toTValue (TValue x) = x
  241     toTValue t = panic "evalTermF / toTValue" ["Not a type value: " <> Text.pack (show t)]
  242 
  243     evalRecursor :: VRecursor l -> Value l
  244     evalRecursor vrec@(VRecursor d _nixs ps_fs) =
  245       vStrictFun $ \argv ->
  246       case evalConstructor argv of
  247         Just (ctor, args)
  248           | Just elim <- Map.lookup (nameIndex (ctorName ctor)) ps_fs ->
  249               do elimv <- force elim
  250                  reduceRecursor (evalRecursor vrec) elimv args (ctorArgStruct ctor)
  251 
  252           | otherwise ->
  253               panic "evalTermF / evalRecursor"
  254               ["Could not find info for constructor: " <> Text.pack (show ctor)]
  255         Nothing ->
  256           case argv of
  257             VCtorMux _ps branches ->
  258               do alts <- traverse (evalCtorMuxBranch vrec) (IntMap.elems branches)
  259                  combineAlts alts
  260             _ ->
  261               panic "evalTermF / evalRecursor"
  262               ["Expected constructor for datatype: " <> toAbsoluteName (nameInfo d)]
  263 
  264     evalCtorMuxBranch ::
  265       VRecursor l ->
  266       (VBool l, Name, TValue l, [Thunk l]) ->
  267       EvalM l (VBool l, EvalM l (Value l))
  268     evalCtorMuxBranch r (p, c, _ct, args) =
  269       case r of
  270         VRecursor _d _nixs ps_fs ->
  271           do let i = nameIndex c
  272              case (lookupVarIndexInMap i (simModMap cfg), Map.lookup i ps_fs) of
  273                (Just (ResolvedCtor ctor), Just elim) ->
  274                  do elimv <- force elim
  275                     pure (p, reduceRecursor (evalRecursor r) elimv args (ctorArgStruct ctor))
  276                _ -> panic "evalTermF / evalCtorMuxBranch"
  277                     ["could not find info for constructor: " <> toAbsoluteName (nameInfo c)]
  278 
  279     combineAlts :: [(VBool l, EvalM l (Value l))] -> EvalM l (Value l)
  280     combineAlts [] = panic "evalTermF / combineAlts" ["no alternatives"]
  281     combineAlts [(_, x)] = x
  282     combineAlts ((p, x) : alts) = simLazyMux cfg p x (combineAlts alts)
  283 
  284     evalConstructor :: Value l -> Maybe (Ctor, [Thunk l])
  285     evalConstructor (VCtorApp c _tv _ps args) =
  286       case lookupVarIndexInMap (nameIndex c) (simModMap cfg) of
  287         Just (ResolvedCtor ctor) -> Just (ctor, args)
  288         _ -> Nothing
  289     evalConstructor (VNat 0) =
  290        do ctor <- findCtorInMap preludeZeroIdent (simModMap cfg)
  291           Just (ctor, [])
  292     evalConstructor (VNat n) =
  293        do ctor <- findCtorInMap preludeSuccIdent (simModMap cfg)
  294           Just (ctor, [ ready (VNat (pred n)) ])
  295     evalConstructor _ =
  296        Nothing
  297 
  298     recEvalDelay :: Term -> EvalM l (Thunk l)
  299     recEvalDelay = delay . recEval
  300 
  301     ctorValue :: Name -> TValue l -> Int -> Int -> MValue l
  302     ctorValue nm tv i j =
  303       vFunList i $ \params ->
  304       vFunList j $ \args ->
  305       pure $ VCtorApp nm tv params args
  306 
  307     dtValue :: Name -> Int -> Int -> MValue l
  308     dtValue nm i j =
  309       vStrictFunList i $ \params ->
  310       vStrictFunList j $ \idxs ->
  311       pure $ TValue $ VDataType nm params idxs

  313 -- | Evaluate a recursor applied to a specific data constructor.
  314 reduceRecursor ::
  315   forall l. (VMonadLazy l, Show (Extra l)) =>
  316   Value l {- ^ recursor function expecting datatype argument -} ->
  317   Value l {- ^ constructor eliminator function -} ->
  318   [Thunk l] {- ^ constructor arguments -} ->
  319   CtorArgStruct {- ^ constructor formal argument descriptor -} ->
  320   MValue l
  321 reduceRecursor r elim c_args argstruct = go elim c_args (map snd (ctorArgs argstruct))
  322   where
  323     go :: Value l -> [Thunk l] -> [CtorArg] -> MValue l
  324     go e [] [] = pure e
  325     go e (x : xs) (arg : args) =
  326       case arg of
  327         ConstArg _ ->
  328           do e' <- apply e x
  329              go e' xs args
  330         RecursiveArg zs _ixs ->
  331           do e1 <- apply e x
  332              recx <- delay (mk_rec_arg (length zs) x)
  333              e2 <- apply e1 recx
  334              go e2 xs args
  335     go _ _ _ = panic "reduceRecursor" ["Wrong number of constructor arguments"]
  336 
  337     -- For a recursive argument, we need a value of the form
  338     -- > \z1 .. zk -> r (x z1 .. zk)
  339     mk_rec_arg :: Int -> Thunk l -> MValue l
  340     mk_rec_arg k x =
  341       vFunList k $ \zs ->
  342       do x_zs <- delay (force x >>= \v -> applyAll v zs)
  343          apply r x_zs

  405 -- | A variant of 'evalGlobal' that lets the uninterpreted function
  406 -- symbol and external-constant callbacks have access to the 'TermF'.
  407 evalGlobal' ::
  408   forall l. (VMonadLazy l, Show (Extra l)) =>
  409   ModuleMap ->
  410   -- | Implementations of 'Primitive' terms, plus overrides for 'Constant' and 'CtorApp' terms
  411   Map Ident (Prims.Prim l) ->
  412   -- | Implementations of free 'Variable' terms
  413   (Term -> VarName -> TValue l -> MValue l) ->
  414   -- | Overrides for Constant terms (e.g. uninterpreted functions)
  415   (Name -> TValue l -> Maybe (MValue l)) ->
  416   -- | Overrides for Recursor terms
  417   (Name -> Sort -> Maybe (MValue l)) ->
  418   -- | Handler for stuck primitives
  419   (Name -> Text -> [Thunk l] -> MValue l) ->
  420   -- | Lazy mux operation
  421   (VBool l -> MValue l -> MValue l -> MValue l) ->
  422   EvalM l (SimulatorConfig l)
  423 evalGlobal' modmap prims variable constant recursor primHandler lazymux =
  424   do checkPrimitives modmap prims
  425      return (SimulatorConfig primitive variable constant' recursor modmap lazymux)
  426   where
  427     constant' :: Name -> TValue l -> Maybe (MValue l)
  428     constant' nm tv =
  429       case constant nm tv of
  430         Just v -> Just v
  431         Nothing ->
  432           case nameInfo nm of
  433             ModuleIdentifier ident ->
  434               evalPrim (primHandler nm) <$> Map.lookup ident prims
  435             ImportedName{} -> Nothing
  436 
  437     primitive :: Name -> MValue l
  438     primitive nm =
  439       case nameInfo nm of
  440         ImportedName {} ->
  441           panic "evalGlobal'" ["Unimplemented global: " <> toAbsoluteName (nameInfo nm)]
  442         ModuleIdentifier ident ->
  443           case Map.lookup ident prims of
  444             Just v  -> evalPrim (primHandler nm) v
  445             Nothing -> panic "evalGlobal'" ["Unimplemented global: " <> identText ident]

  447 -- | Check that all the primitives declared in the given module
  448 --   are implemented, and that terms with implementations are not
  449 --   overridden.
  450 checkPrimitives :: forall l. (VMonadLazy l, Show (Extra l))
  451                 => ModuleMap
  452                 -> Map Ident (Prims.Prim l)
  453                 -> EvalM l ()
  454 checkPrimitives modmap prims = do
  455    -- FIXME this is downgraded to a warning temporarily while we work out a
  456    -- solution to issue GaloisInc/saw-script#48
  457    --   when (not $ null unimplementedPrims) (panic $ unimplementedMsg)
  458    -- (if null unimplementedPrims then id else Debug.trace (unimplementedMsg++"\n")) $
  459 --   (if null overridePrims then id else Debug.trace (overrideMsg++"\n")) $
  460      return ()
  461 
  462   where _unimplementedMsg = unwords $
  463             ("WARNING unimplemented primitives:" : (map show unimplementedPrims))
  464         _overrideMsg = unwords $
  465             ("WARNING overridden definitions:" : (map show overridePrims))
  466 
  467         primSet = Set.fromList $ mapMaybe defIdent $ allModulePrimitives modmap
  468         defSet  = Set.fromList $ mapMaybe defIdent $ allModuleActualDefs modmap
  469         implementedPrims = Map.keysSet prims
  470 
  471         unimplementedPrims = Set.toList $ Set.difference primSet implementedPrims
  472         overridePrims = Set.toList $ Set.intersection defSet implementedPrims

  481 -- The evaluation strategy for SharedTerms involves two memo tables:
  482 -- The first, @memoClosed@, is precomputed and contains the result of
  483 -- evaluating all _closed_ subterms. The same @memoClosed@ table is
  484 -- used for evaluation under lambdas, since the meaning of a closed
  485 -- term does not depend on the local variable context. The second memo
  486 -- table is @memoLocal@, which additionally includes the result of
  487 -- evaluating _open_ terms in the current variable context. It is
  488 -- reinitialized to @memoClosed@ whenever we descend under a lambda
  489 -- binder.
  490 
  491 {-# SPECIALIZE evalSharedTerm ::
  492   Show (Extra l) => SimulatorConfigIn Id l -> Term -> MValueIn Id l #-}
  493 {-# SPECIALIZE evalSharedTerm ::
  494   Show (Extra l) => SimulatorConfigIn IO l -> Term -> MValueIn IO l #-}
  495 
  496 -- | Evaluator for shared terms.
  497 evalSharedTerm :: (VMonadLazy l, MonadFix (EvalM l), Show (Extra l)) =>
  498                   SimulatorConfig l -> Term -> MValue l
  499 evalSharedTerm cfg t = do
  500   memoClosed <- mkMemoClosed cfg t
  501   evalOpen cfg memoClosed t IntMap.empty
  502 
  503 {-# SPECIALIZE mkMemoClosed ::
  504   Show (Extra l) =>
  505   SimulatorConfigIn Id l -> Term -> Id (IntMap (ThunkIn Id l)) #-}
  506 {-# SPECIALIZE mkMemoClosed ::
  507   Show (Extra l) =>
  508   SimulatorConfigIn IO l -> Term -> IO (IntMap (ThunkIn IO l)) #-}
  509 
  510 -- | Precomputing the memo table for closed subterms.
  511 mkMemoClosed :: forall l. (VMonadLazy l, MonadFix (EvalM l), Show (Extra l)) =>
  512                 SimulatorConfig l -> Term -> EvalM l (IntMap (Thunk l))
  513 mkMemoClosed cfg t =
  514   mfix $ \memoClosed -> mapM (delay . evalClosedTermF cfg memoClosed) subterms
  515   where
  516     -- | Map of all closed subterms of t.
  517     subterms :: IntMap (TermF Term)
  518     subterms = fmap fst $ IntMap.filter (IntSet.null . snd) $ State.execState (go t) IntMap.empty
  519 
  520     go :: Term -> State.State (IntMap (TermF Term, IntSet)) IntSet
  521     go (Unshared tf) = termf tf
  522     go (STApp{ stAppIndex = i, stAppTermF = tf }) =
  523       do memo <- State.get
  524          case IntMap.lookup i memo of
  525            Just (_, b) -> pure b
  526            Nothing ->
  527              do b <- termf tf
  528                 State.modify (IntMap.insert i (tf, b))
  529                 pure b
  530 
  531     termf :: TermF Term -> State.State (IntMap (TermF Term, IntSet)) IntSet
  532     termf tf =
  533       do -- if tf is a defined constant, traverse the definition body and type
  534          case tf of
  535            Constant nm ->
  536              do let r = requireNameInMap nm (simModMap cfg)
  537                 void $ go (resolvedNameType r)
  538                 case r of
  539                   ResolvedDef (defBody -> Just body) -> void $ go body
  540                   _ -> pure ()
  541            _ -> pure ()
  542          freesTermF <$> traverse go tf
  543

  557 -- | Evaluator for closed terms, used to populate @memoClosed@.
  558 evalClosedTermF :: (VMonadLazy l, Show (Extra l)) =>
  559                    SimulatorConfig l
  560                 -> IntMap (Thunk l)
  561                 -> TermF Term -> MValue l
  562 evalClosedTermF cfg memoClosed tf = evalTermF cfg lam recEval tf IntMap.empty
  563   where
  564     lam = evalOpen cfg memoClosed
  565     recEval (Unshared tf') = evalTermF cfg lam recEval tf' IntMap.empty
  566     recEval (STApp{ stAppIndex = i }) =
  567       case IntMap.lookup i memoClosed of
  568         Just x -> force x
  569         Nothing -> panic "evalClosedTermF" ["internal error"]

  571 {-# SPECIALIZE mkMemoLocal ::
  572   Show (Extra l) =>
  573   SimulatorConfigIn Id l ->
  574   IntMap (ThunkIn Id l) ->
  575   Term ->
  576   EnvIn Id l ->
  577   Id (IntMap (ThunkIn Id l)) #-}
  578 {-# SPECIALIZE mkMemoLocal ::
  579   Show (Extra l) =>
  580   SimulatorConfigIn IO l ->
  581   IntMap (ThunkIn IO l) ->
  582   Term ->
  583   EnvIn IO l ->
  584   IO (IntMap (ThunkIn IO l)) #-}
  585 
  586 -- | Precomputing the memo table for open subterms in the current context.
  587 mkMemoLocal :: forall l. (VMonadLazy l, Show (Extra l)) =>
  588                SimulatorConfig l -> IntMap (Thunk l) ->
  589                Term -> Env l -> EvalM l (IntMap (Thunk l))
  590 mkMemoLocal cfg memoClosed t env = go mempty t
  591   where
  592     go :: IntMap (Thunk l) -> Term -> EvalM l (IntMap (Thunk l))
  593     go memo (Unshared tf) = goTermF memo tf
  594     go memo (t'@STApp{ stAppIndex = i, stAppTermF = tf })
  595       | closedTerm t' = pure memo
  596       | otherwise =
  597         case IntMap.lookup i memo of
  598           Just _ -> pure memo
  599           Nothing ->
  600             do memo' <- goTermF memo tf
  601                thunk <- delay (evalLocalTermF cfg memoClosed memo' tf env)
  602                pure (IntMap.insert i thunk memo')
  603     goTermF :: IntMap (Thunk l) -> TermF Term -> EvalM l (IntMap (Thunk l))
  604     goTermF memo tf =
  605       case tf of
  606         FTermF ftf      -> foldlM go memo ftf
  607         App t1 t2       -> do memo' <- goTermF memo (unwrapTermF t1)
  608                               go memo' t2
  609         Lambda _ t1 _   -> go memo t1
  610         Pi _ t1 _       -> go memo t1
  611         Constant{}      -> pure memo
  612         Variable _nm tp -> go memo tp
  613 
  614 {-# SPECIALIZE evalLocalTermF ::
  615   Show (Extra l) =>
  616   SimulatorConfigIn Id l ->
  617   IntMap (ThunkIn Id l) ->
  618   IntMap (ThunkIn Id l) ->
  619   TermF Term ->
  620   OpenValueIn Id l #-}
  621 {-# SPECIALIZE evalLocalTermF ::
  622   Show (Extra l) =>
  623   SimulatorConfigIn IO l ->
  624   IntMap (ThunkIn IO l) ->
  625   IntMap (ThunkIn IO l) ->
  626   TermF Term ->
  627   OpenValueIn IO l #-}
  628 -- | Evaluator for open terms, used to populate @memoLocal@.
  629 evalLocalTermF :: (VMonadLazy l, Show (Extra l)) =>
  630                    SimulatorConfig l
  631                 -> IntMap (Thunk l) -> IntMap (Thunk l)
  632                 -> TermF Term -> OpenValue l
  633 evalLocalTermF cfg memoClosed memoLocal tf0 env = evalTermF cfg lam recEval tf0 env
  634   where
  635     lam = evalOpen cfg memoClosed
  636     recEval (Unshared tf) = evalTermF cfg lam recEval tf env
  637     recEval (t@STApp{ stAppIndex = i, stAppTermF = tf }) =
  638       case IntMap.lookup i memo of
  639         Just x -> force x
  640         Nothing -> evalTermF cfg lam recEval tf env
  641       where memo = if closedTerm t then memoClosed else memoLocal

  643 {-# SPECIALIZE evalOpen ::
  644   Show (Extra l) =>
  645   SimulatorConfigIn Id l ->
  646   IntMap (ThunkIn Id l) ->
  647   Term ->
  648   OpenValueIn Id l #-}

  650 {-# SPECIALIZE evalOpen ::
  651   Show (Extra l) =>
  652   SimulatorConfigIn IO l ->
  653   IntMap (ThunkIn IO l) ->
  654   Term ->
  655   OpenValueIn IO l #-}
  656 
  657 -- | Evaluator for open terms; parameterized by a precomputed table @memoClosed@.
  658 evalOpen :: forall l. (VMonadLazy l, Show (Extra l)) =>
  659             SimulatorConfig l
  660          -> IntMap (Thunk l)
  661          -> Term -> OpenValue l
  662 evalOpen cfg memoClosed t env = do
  663   memoLocal <- mkMemoLocal cfg memoClosed t env
  664   let eval :: Term -> MValue l
  665       eval (Unshared tf) = evalF tf
  666       eval (t'@STApp{ stAppIndex = i, stAppTermF = tf }) =
  667         case IntMap.lookup i memo of
  668           Just x -> force x
  669           Nothing -> evalF tf
  670         where memo = if closedTerm t' then memoClosed else memoLocal
  671       evalF :: TermF Term -> MValue l
  672       evalF tf = evalTermF cfg (evalOpen cfg memoClosed) eval tf env
  673   eval t

  676 {-# SPECIALIZE evalPrim ::
  677   Show (Extra l) =>
  678   (Text -> [ThunkIn Id l] -> MValueIn Id l) ->
  679   PrimIn Id l ->
  680   MValueIn Id l
  681  #-}
  682 {-# SPECIALIZE evalPrim ::
  683   Show (Extra l) =>
  684   (Text -> [ThunkIn IO l] -> MValueIn IO l) ->
  685   PrimIn IO l ->
  686   MValueIn IO l
  687  #-}
  688 evalPrim :: forall l. (VMonadLazy l, Show (Extra l)) =>
  689   (Text -> [Thunk l] -> MValue l) ->
  690   Prims.Prim l ->
  691   MValue l
  692 evalPrim fallback = loop []
  693   where
  694     loop :: [Thunk l] -> Prims.Prim l -> MValue l
  695     loop env (Prims.PrimFun f) =
  696       pure $ VFun $ \x ->
  697         loop (x : env) (f x)
  698 
  699     loop env (Prims.PrimStrict f) =
  700       pure $ vStrictFun $ \x ->
  701         loop (ready x : env) (f x)
  702 
  703     loop env (Prims.PrimFilterFun msg r f) =
  704       pure $ vStrictFun $ \x ->
  705         runMaybeT (r x) >>= \case
  706           Just v -> loop (ready x : env) (f v)
  707           _ -> fallback msg (ready x : env)
  708 
  709     loop env (Prims.PrimExcept m) =
  710       runExceptT m >>= \case
  711         Right v  -> pure v
  712         Left msg -> fallback msg env
  713 
  714     loop _env (Prims.Prim m) = m
  715     loop _env (Prims.PrimValue v) = pure v
  716 
  717 -- | A basic handler for stuck primitives.
  718 defaultPrimHandler ::
  719   (VMonadLazy l, MonadFail (EvalM l)) =>
  720   Name -> Text -> [Thunk l] -> MValue l
  721 defaultPrimHandler nm msg env =
  722   fail $ unlines
  723   [ "Could not evaluate primitive " ++ Text.unpack (toAbsoluteName (nameInfo nm))
  724   , "On argument " ++ show (length env)
  725   , Text.unpack msg
