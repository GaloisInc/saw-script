   33 -- | Given two terms, compute a term representing a decidable
   34 --   equality test between them.  The terms are assumed to
   35 --   be of the same type, which must be a first-order type.
   36 --   The returned term will be of type @Bool@.
   37 scEq :: SharedContext -> Term -> Term -> IO Term
   38 scEq sc x y = do
   39   xty <- scTypeOf sc x
   40   mmap <- scGetModuleMap sc
   41   case asFirstOrderTypeValue (evalSharedTerm mmap mempty mempty xty) of
   42     Just fot -> scDecEq sc fot (Just (x,y))
   43     Nothing  -> fail ("scEq: expected first order type, but got: " ++ showTerm xty)
   44 
   45 -- | Given a first-order type, return the decidable equality
   46 --   operation on that type.  If arguments are provided, they
   47 --   will be applied, returning a value of type @Bool@.  If no
   48 --   arguments are provided a function of type @tp -> tp -> Bool@
   49 --   will be returned.
   50 scDecEq ::
   51   SharedContext ->
   52   FirstOrderType    {- ^ Type of elements to test for equality -} ->
   53   Maybe (Term,Term) {- ^ optional arguments to apply -} ->
   54   IO Term
   55 scDecEq sc fot args = case fot of
   56   FOTBit ->
   57     do fn <- scGlobalDef sc "Prelude.boolEq"
   58        case args of
   59          Nothing    -> return fn
   60          Just (x,y) -> scApplyAll sc fn [x,y]
   61 
   62   FOTInt ->
   63     do fn <- scGlobalDef sc "Prelude.intEq"
   64        case args of
   65          Nothing    -> return fn
   66          Just (x,y) -> scApplyAll sc fn [x,y]
   67 
   68   FOTIntMod m ->
   69     do fn <- scGlobalDef sc "Prelude.intModEq"
   70        m' <- scNat sc m
   71        case args of
   72          Nothing    -> scApply sc fn m'
   73          Just (x,y) -> scApplyAll sc fn [m',x,y]
   74 
   75   FOTVec w FOTBit ->
   76     do fn <- scGlobalDef sc "Prelude.bvEq"
   77        w' <- scNat sc w
   78        case args of
   79          Nothing    -> scApply sc fn w'
   80          Just (x,y) -> scApplyAll sc fn [w',x,y]
   81 
   82   FOTVec w t ->
   83     do fn <- scGlobalDef sc "Prelude.vecEq"
   84        w' <- scNat sc w
   85        t' <- scFirstOrderType sc t
   86        subFn <- scDecEq sc t Nothing
   87        case args of
   88          Nothing    -> scApplyAll sc fn [w',t',subFn]
   89          Just (x,y) -> scApplyAll sc fn [w',t',subFn,x,y]
   90 
   91   FOTArray a b ->
   92     do a' <- scFirstOrderType sc a
   93        b' <- scFirstOrderType sc b
   94        fn <- scGlobalDef sc "Prelude.arrayEq"
   95        case args of
   96          Nothing    -> scApplyAll sc fn [a',b']
   97          Just (x,y) -> scApplyAll sc fn [a',b',x,y]
   98 
   99   FOTTuple []  ->
  100     case args of
  101       Nothing -> scGlobalDef sc "Prelude.unitEq"
  102       Just _  -> scBool sc True
  103 
  104   FOTTuple [t] -> scDecEq sc t args
  105 
  106   FOTTuple (t:ts) ->
  107     do fnLeft  <- scDecEq sc t Nothing
  108        fnRight <- scDecEq sc (FOTTuple ts) Nothing
  109        fn      <- scGlobalDef sc "Prelude.pairEq"
  110        t'      <- scFirstOrderType sc t
  111        ts'     <- scFirstOrderType sc (FOTTuple ts)
  112        case args of
  113          Nothing    -> scApplyAll sc fn [t',ts',fnLeft,fnRight]
  114          Just (x,y) -> scApplyAll sc fn [t',ts',fnLeft,fnRight,x,y]
  115 
  116   FOTRec fs ->
  117     case args of
  118       Just (x,y) ->
  119            mkRecordEqBody (Map.toList fs) x y
  120 
  121       Nothing ->
  122         do tp <- scFirstOrderType sc fot
  123            x <- scFreshVariable sc "x" tp
  124            y <- scFreshVariable sc "y" tp
  125            body <- mkRecordEqBody (Map.toList fs) x y
  126            scAbstractTerms sc [x, y] body
  127 
  128  where
  129   mkRecordEqBody [] _x _y = scBool sc True
  130   mkRecordEqBody [(f,tp)] x y =
  131      do xf <- scRecordSelect sc x f
  132         yf <- scRecordSelect sc y f
  133         scDecEq sc tp (Just (xf,yf))
  134   mkRecordEqBody ((f,tp):fs) x y =
  135      do xf   <- scRecordSelect sc x f
  136         yf   <- scRecordSelect sc y f
  137         fp  <- scDecEq sc tp (Just (xf,yf))
  138         fsp <- mkRecordEqBody fs x y
  139         scAnd sc fp fsp
