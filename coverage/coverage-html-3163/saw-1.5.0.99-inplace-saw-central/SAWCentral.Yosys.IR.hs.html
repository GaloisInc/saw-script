   65 -- | The direction of a module port.
   66 data Direction
   67   = DirectionInput
   68   | DirectionOutput
   69   | DirectionInout
   70   deriving (Show, Eq, Ord)
   71 
   72 isInput :: Direction -> Bool
   73 isInput d =
   74   case d of
   75     DirectionInput -> True
   76     DirectionOutput -> False
   77     DirectionInout -> True
   78 
   79 isOutput :: Direction -> Bool
   80 isOutput d =
   81   case d of
   82     DirectionInput -> False
   83     DirectionOutput -> True
   84     DirectionInout -> True
   85 
   86 instance Aeson.FromJSON Direction where
   87   parseJSON (Aeson.String "input") = pure DirectionInput
   88   parseJSON (Aeson.String "output") = pure DirectionOutput
   89   parseJSON (Aeson.String "inout") = pure DirectionInout
   90   parseJSON v = fail $ "Failed to parse direction: " <> show v

   92 -- | The value of a connection
   93 data Bitrep
   94   = BitrepZero -- ^ Constant zero bit
   95   | BitrepOne -- ^ Constant one bit
   96   | BitrepX -- ^ Undefined bit X
   97   | BitrepZ -- ^ Undefined bit Z
   98   | Bitrep Integer -- ^ The signal bit with the given index
   99   deriving (Show, Eq, Ord)
  100 
  101 instance Aeson.FromJSON Bitrep where
  102   parseJSON (Aeson.String "0") = pure BitrepZero
  103   parseJSON (Aeson.String "1") = pure BitrepOne
  104   parseJSON (Aeson.String "x") = pure BitrepX
  105   parseJSON (Aeson.String "z") = pure BitrepZ
  106   parseJSON vn@Aeson.Number{} = Bitrep <$> Aeson.parseJSON vn
  107   parseJSON v = fail $ "Failed to parse bits: " <> show v

  123 -- | Return 'True' iff a given cell type is a primitive type
  124 cellTypeIsPrimitive :: Text -> Bool
  125 cellTypeIsPrimitive cellType =
  126   case Text.uncons cellType of
  127     Just ('$', _) -> True
  128     _ -> False
  129 
  130 -- | Mapping from 'Text' to combinational primitive cell types.
  131 textToCellTypeCombinational :: Map Text CellTypeCombinational
  132 textToCellTypeCombinational =
  133   Map.fromList [ (ppCellTypeCombinational t, t) | t <- [minBound .. maxBound] ]
  134 
  135 -- | Mapping from 'Text' to primitive register cell types.
  136 textToCellTypeRegister :: Map Text CellTypeRegister
  137 textToCellTypeRegister =
  138   Map.fromList [ (ppCellTypeRegister t, t) | t <- [minBound .. maxBound] ]
  139 
  140 -- | Mapping from 'Text' to primitive cell types.
  141 textToPrimitiveCellType :: Map Text CellType
  142 textToPrimitiveCellType =
  143   fmap CellTypeRegister textToCellTypeRegister <>
  144   fmap CellTypeCombinational textToCellTypeCombinational

  146 -- | All supported primitive combinational cell types.
  147 data CellTypeCombinational
  148   = CellTypeNot
  149   | CellTypePos
  150   | CellTypeNeg
  151   | CellTypeAnd
  152   | CellTypeOr
  153   | CellTypeXor
  154   | CellTypeXnor
  155   | CellTypeReduceAnd
  156   | CellTypeReduceOr
  157   | CellTypeReduceXor
  158   | CellTypeReduceXnor
  159   | CellTypeReduceBool
  160   | CellTypeShl
  161   | CellTypeShr
  162   | CellTypeSshl
  163   | CellTypeSshr
  164   | CellTypeShift
  165   | CellTypeShiftx
  166   | CellTypeLt
  167   | CellTypeLe
  168   | CellTypeGt
  169   | CellTypeGe
  170   | CellTypeEq
  171   | CellTypeNe
  172   | CellTypeEqx
  173   | CellTypeNex
  174   | CellTypeAdd
  175   | CellTypeSub
  176   | CellTypeMul
  177   | CellTypeDiv
  178   | CellTypeMod
  179   | CellTypeLogicNot
  180   | CellTypeLogicAnd
  181   | CellTypeLogicOr
  182   | CellTypeMux
  183   | CellTypePmux
  184   | CellTypeBmux
  185   | CellTypeBUF
  186   deriving (Eq, Ord, Enum, Bounded)

  188 -- | All supported primitive register cell types.
  189 data CellTypeRegister
  190   = CellTypeAdff
  191   | CellTypeAldff
  192   | CellTypeDff
  193   | CellTypeDffe
  194   | CellTypeFf
  195   deriving (Eq, Ord, Enum, Bounded)
  196 
  197 -- | All supported cell types.
  198 -- All types are primitives except for 'CellTypeUserType' which
  199 -- represents user-defined submodules.
  200 -- Invariants: 'CellTypeUnsupportedPrimitive' should only be applied
  201 -- to names satisfying 'cellTypeIsPrimitive', and 'CellTypeUserType'
  202 -- should only be applied to names *not* satisfying
  203 -- 'cellTypeIsPrimitive'.
  204 data CellType
  205   = CellTypeCombinational CellTypeCombinational
  206   | CellTypeRegister CellTypeRegister
  207   | CellTypeUnsupportedPrimitive CellTypeName
  208   | CellTypeUserType CellTypeName
  209   deriving (Eq, Ord)
  210 
  211 instance Aeson.FromJSON CellType where
  212   parseJSON (Aeson.String s) =
  213     case s of
  214       "$sdff"        -> fail $ show $ YosysErrorUnsupportedFF "$sdff"
  215       "$dffsr"       -> fail $ show $ YosysErrorUnsupportedFF "$dffsr"
  216       "$adffe"       -> fail $ show $ YosysErrorUnsupportedFF "$adffe"
  217       "$sdffe"       -> fail $ show $ YosysErrorUnsupportedFF "$sdffe"
  218       "$sdffce"      -> fail $ show $ YosysErrorUnsupportedFF "$sdffce"
  219       "$aldffe"      -> fail $ show $ YosysErrorUnsupportedFF "$aldffe"
  220       "$dffsre"      -> fail $ show $ YosysErrorUnsupportedFF "$dffsre"
  221       _ | cellTypeIsPrimitive s ->
  222           case Map.lookup s textToPrimitiveCellType of
  223             Just cellType -> pure cellType
  224             -- XXX We should probably log a warning when generating CellTypeUnsupportedPrimitive,
  225             -- we can't do that here however.
  226             Nothing -> pure $ CellTypeUnsupportedPrimitive s
  227         | otherwise -> pure $ CellTypeUserType s
  228   parseJSON v = fail $ "Failed to parse cell type: " <> show v
  229 
  230 ppCellTypeCombinational :: CellTypeCombinational -> Text
  231 ppCellTypeCombinational ctc =
  232   case ctc of
  233     CellTypeNot        -> "$not"
  234     CellTypePos        -> "$pos"
  235     CellTypeNeg        -> "$neg"
  236     CellTypeAnd        -> "$and"
  237     CellTypeOr         -> "$or"
  238     CellTypeXor        -> "$xor"
  239     CellTypeXnor       -> "$xnor"
  240     CellTypeReduceAnd  -> "$reduce_and"
  241     CellTypeReduceOr   -> "$reduce_or"
  242     CellTypeReduceXor  -> "$reduce_xor"
  243     CellTypeReduceXnor -> "$reduce_xnor"
  244     CellTypeReduceBool -> "$reduce_bool"
  245     CellTypeShl        -> "$shl"
  246     CellTypeShr        -> "$shr"
  247     CellTypeSshl       -> "$sshl"
  248     CellTypeSshr       -> "$sshr"
  249     CellTypeShift      -> "$shift"
  250     CellTypeShiftx     -> "$shiftx"
  251     CellTypeLt         -> "$lt"
  252     CellTypeLe         -> "$le"
  253     CellTypeGt         -> "$gt"
  254     CellTypeGe         -> "$ge"
  255     CellTypeEq         -> "$eq"
  256     CellTypeNe         -> "$ne"
  257     CellTypeEqx        -> "$eqx"
  258     CellTypeNex        -> "$nex"
  259     CellTypeAdd        -> "$add"
  260     CellTypeSub        -> "$sub"
  261     CellTypeMul        -> "$mul"
  262     CellTypeDiv        -> "$div"
  263     CellTypeMod        -> "$mod"
  264     CellTypeLogicNot   -> "$logic_not"
  265     CellTypeLogicAnd   -> "$logic_and"
  266     CellTypeLogicOr    -> "$logic_or"
  267     CellTypeMux        -> "$mux"
  268     CellTypePmux       -> "$pmux"
  269     CellTypeBmux       -> "$bmux"
  270     CellTypeBUF        -> "$_BUF_"
  271 
  272 instance Show CellTypeCombinational where
  273   show ctc = Text.unpack (ppCellTypeCombinational ctc)
  274 
  275 ppCellTypeRegister :: CellTypeRegister -> Text
  276 ppCellTypeRegister ctr =
  277   case ctr of
  278     CellTypeAdff -> "$adff"
  279     CellTypeAldff -> "$aldff"
  280     CellTypeDff -> "$dff"
  281     CellTypeDffe -> "$dffe"
  282     CellTypeFf -> "$ff"
  283 
  284 ppCellType :: CellType -> Text
  285 ppCellType ct =
  286   case ct of
  287     CellTypeCombinational ctc -> ppCellTypeCombinational ctc
  288     CellTypeRegister ctr -> ppCellTypeRegister ctr
  289     CellTypeUnsupportedPrimitive t -> t
  290     CellTypeUserType t -> t
  291 
  292 instance Show CellType where
  293   show ct = Text.unpack (ppCellType ct)

  295 -- | A cell within an HDL module.
  296 data Cell =
  297   Cell
  298   { _cellHideName :: Bool -- ^ Whether the cell's name is human-readable (default: False)
  299   , _cellType :: CellType -- ^ The cell type
  300     -- NB: Yosys's documentation for write_json doesn't impose any restrictions
  301     -- on what type parameter values may take on, so we opt to be as permissive
  302     -- as possible by using Aeson Values. Most of the time, these Values will
  303     -- be strings, but they can also be numbers on occasion (e.g., if you call
  304     -- write_json using the -compat-int flag).
  305   , _cellParameters :: Map Text Aeson.Value -- ^ Metadata parameters
  306   , _cellAttributes :: Maybe Aeson.Value -- currently unused
  307   , _cellPortDirections :: Map PortName Direction -- ^ Direction for each cell connection
  308   , _cellConnections :: Map PortName [Bitrep] -- ^ Bitrep for each cell connection
  309   }
  310   deriving (Show, Eq, Ord)
  311 
  312 makeLenses ''Cell
  313 
  314 instance Aeson.FromJSON Cell where
  315   parseJSON = Aeson.withObject "cell" $ \o -> do
  316     _cellHideName <- Maybe.maybe False (/= (0::Int)) <$> o Aeson..:? "hide_name"
  317     _cellType <- o Aeson..: "type"
  318     _cellParameters <- o Aeson..: "parameters"
  319     _cellAttributes <- o Aeson..:? "attributes"
  320     _cellPortDirections <- o Aeson..: "port_directions"
  321     _cellConnections <- o Aeson..: "connections"
  322     pure Cell{..}

  372 -- | Read a collection of HDL modules from a file produced by Yosys' write_json command.
  373 loadYosysIR :: FilePath -> IO YosysIR
  374 loadYosysIR p = Aeson.eitherDecodeFileStrict p >>= \case
  375   Left err -> yosysError $ YosysError $ Text.pack err
  376   Right ir -> pure ir

  378 -- | Return the patterns for all of the input ports of a module
  379 moduleInputPorts :: Module -> Map PortName [Bitrep]
  380 moduleInputPorts m =
  381   Map.mapMaybe
  382   ( \ip ->
  383       if ip ^. portDirection == DirectionInput || ip ^. portDirection == DirectionInout
  384       then Just (ip ^. portBits)
  385       else Nothing
  386   )
  387   $ m ^. modulePorts
  388 
  389 -- | Return the patterns for all of the output ports of a module
  390 moduleOutputPorts :: Module -> Map PortName [Bitrep]
  391 moduleOutputPorts m =
  392   Map.mapMaybe
  393   ( \ip ->
  394       if isOutput (ip ^. portDirection)
  395       then Just (ip ^. portBits)
  396       else Nothing
  397   )
  398   $ m ^. modulePorts
  399 
  400 -- | Return the patterns for all of the input connections of a cell
  401 cellInputConnections :: Cell -> Map PortName [Bitrep]
  402 cellInputConnections c = Map.intersection (c ^. cellConnections) inp
  403   where
  404     inp = Map.filter isInput (c ^. cellPortDirections)
  405 
  406 -- | Return the patterns for all of the output connections of a cell
  407 cellOutputConnections :: Cell -> Map PortName [Bitrep]
  408 cellOutputConnections c = Map.intersection (c ^. cellConnections) out
  409   where
  410     out = Map.filter isOutput (c ^. cellPortDirections)

  412 -- | Test whether a 'Cell' is a state element ('CellTypeDff' or 'CellTypeFf').
  413 cellIsRegister :: Cell -> Bool
  414 cellIsRegister c =
  415   case c ^. cellType of
  416     CellTypeRegister _ -> True
  417     _ -> False

  419 -- | Swap out machine-generated names of DFF cells for user-provided
  420 -- names from the netnames section of the module, wherever possible.
  421 -- If no suitable name exists in the netnames table, then use function
  422 -- 'cellIdentifier' to produce a lexically-valid field name.
  423 renameDffInstances :: Module -> Module
  424 renameDffInstances m = set moduleCells cells' m
  425   where
  426     cells' :: Map CellInstName Cell
  427     cells' =
  428       Map.fromList $
  429       map (\(t, c) -> (bestName t c, c)) $
  430       Map.toList (m ^. moduleCells)
  431 
  432     netnames :: Map [Bitrep] CellInstName
  433     netnames =
  434       Map.fromList
  435       [ (n ^. netnameBits, t)
  436       | (t, n) <- Map.assocs (m ^. moduleNetnames), not (n ^. netnameHideName) ]
  437 
  438     bestName :: CellInstName -> Cell -> CellInstName
  439     bestName t c
  440       | cellIsRegister c =
  441           Maybe.fromMaybe (cellIdentifier t) $
  442           do bs <- Map.lookup "Q" (c ^. cellConnections)
  443              Map.lookup bs netnames
  444       | otherwise =
  445           t
