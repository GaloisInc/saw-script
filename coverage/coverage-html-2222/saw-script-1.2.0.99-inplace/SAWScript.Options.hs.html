   42 -- | Verbosity is currently a linear setting (vs a mask or tree).  Any given
   43 -- level includes the outputs of all lower levels.
   44 data Verbosity
   45   = Silent
   46   | OnlyCounterExamples
   47   | Error
   48   | Warn
   49   | Info
   50   | Debug
   51     deriving (Show,Eq,Ord)

   79 printOutWith :: Verbosity -> Verbosity -> String -> IO ()
   80 printOutWith setting level msg
   81     | setting >= level = do
   82       t <- formatTime defaultTimeLocale "%T.%3q" <$> getCurrentTime
   83       putStr $ "[" ++ t ++ "] " ++ msg
   84     | otherwise        = return ()
   85 
   86 printOutLn :: Options -> Verbosity -> String -> IO ()
   87 printOutLn o v s = printOutFn o v (s ++ "\n")

   89 -- | Perform some additional post-processing on an 'Options' value based on
   90 -- whether certain environment variables are defined.
   91 --
   92 -- XXX: this does not belong in Options.hs; it is a top-level
   93 -- operation that's really part of main. But it's shared between saw
   94 -- and saw-remote-api so it has to live somewhere shared, and for the
   95 -- time being there isn't anywhere else.
   96 processEnv :: Options -> IO Options
   97 processEnv opts = do
   98   curEnv <- getEnvironment
   99   opts' <- addJavaBinDirInducedOpts opts
  100   return $ foldr addSawOpt opts' curEnv
  101   where
  102     -- If a Java executable's path is specified (either by way of
  103     -- --java-bin-dirs or PATH, see the Haddocks for findJavaIn), then use that
  104     -- to detect the path to Java's standard rt.jar file and add it to the
  105     -- jarList on Java 8 or earlier. (Later versions of Java do not use
  106     -- rt.jar—see Note [Loading classes from JIMAGE files] in
  107     -- Lang.JVM.Codebase in crucible-jvm.)
  108     -- If Java's path is not specified, return the Options unchanged.
  109     addJavaBinDirInducedOpts :: Options -> IO Options
  110     addJavaBinDirInducedOpts os@Options{javaBinDirs} = do
  111       mbJavaPath <- findJavaIn javaBinDirs
  112       case mbJavaPath of
  113         Nothing       -> pure os
  114         Just javaPath -> do
  115           javaMajorVersion <- findJavaMajorVersion javaPath
  116           if javaMajorVersion >= 9
  117              then pure os
  118              else addRTJar javaPath os
  119 
  120     -- rt.jar lives in a standard location relative to @java.home@. At least,
  121     -- this is the case on every operating system I've tested.
  122     addRTJar :: FilePath -> Options -> IO Options
  123     addRTJar javaPath os = do
  124       mbJavaHome <- findJavaProperty javaPath "java.home"
  125       case mbJavaHome of
  126         Nothing -> fail $ "Could not find where rt.jar lives for " ++ javaPath
  127         Just javaHome ->
  128           let rtJarPath = javaHome </> "lib" </> "rt.jar" in
  129           pure $ os{ jarList = rtJarPath : jarList os }
  130 
  131     addSawOpt :: (String, String) -> Options -> Options
  132     addSawOpt ("SAW_IMPORT_PATH", p) os =
  133       os { importPath = importPath os ++ splitSearchPath p }
  134     addSawOpt ("SAW_JDK_JAR", p) os = os { jarList = p : jarList os }
  135     addSawOpt ("CLASSPATH", p) os = os { jarList = jarList os ++ jars
  136                                        , classPath = classPath os ++ dirs
  137                                        }
  138                                     where
  139                                       es = splitSearchPath p
  140                                       (jars, dirs) = partition (".jar" `isExtensionOf`) es
  141     addSawOpt _ os = os
