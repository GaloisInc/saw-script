   99 newSAWCoreBackendWithTimeout :: PathSatSolver -> Sym -> Integer -> IO SomeOnlineBackend
  100 newSAWCoreBackendWithTimeout PathSat_Yices sym timeout =
  101   do bak <- newOnlineBackend sym Yices.yicesDefaultFeatures
  102      W4.extendConfig Z3.z3Options (W4.getConfiguration sym)
  103      W4.extendConfig Yices.yicesOptions (W4.getConfiguration sym)
  104      W4.extendConfig CVC5.cvc5Options (W4.getConfiguration sym)
  105      yicesTimeoutSetting <- W4.getOptionSetting Yices.yicesGoalTimeout (W4.getConfiguration sym)
  106      _ <- W4.setOpt yicesTimeoutSetting timeout
  107      return (SomeOnlineBackend (bak :: Backend Yices.Connection))
  108 
  109 newSAWCoreBackendWithTimeout PathSat_Z3 sym timeout =
  110   do bak <- newOnlineBackend sym (SMT2.defaultFeatures Z3.Z3)
  111      W4.extendConfig Z3.z3Options (W4.getConfiguration sym)
  112      W4.extendConfig Yices.yicesOptions (W4.getConfiguration sym)
  113      W4.extendConfig CVC5.cvc5Options (W4.getConfiguration sym)
  114      z3TimeoutSetting <- W4.getOptionSetting Z3.z3Timeout (W4.getConfiguration sym)
  115      _ <- W4.setOpt z3TimeoutSetting timeout
  116      return (SomeOnlineBackend (bak :: Backend (SMT2.Writer Z3.Z3)))

  118 ppAbortedResult :: (forall l args. GlobalPair Sym (SimFrame Sym ext l args) -> PP.Doc ann)
  119                 -> AbortedResult Sym ext
  120                 -> PP.Doc ann
  121 ppAbortedResult _ (AbortedExec InfeasibleBranch{} _) =
  122   "Infeasible branch"
  123 ppAbortedResult ppGP (AbortedExec abt gp) = do
  124   PP.vcat
  125     [ ppAbortExecReason abt
  126     , ppGP gp
  127     ]
  128 ppAbortedResult ppGP (AbortedBranch loc _predicate trueBranch falseBranch) =
  129   PP.vcat
  130     [ "Both branches aborted after a symbolic branch."
  131     , "Location of control-flow branching:"
  132     , PP.indent 2 (PP.pretty (W4.plSourceLoc loc))
  133     , "Message from the true branch:"
  134     , PP.indent 2 (ppAbortedResult ppGP trueBranch)
  135     , "Message from the false branch:"
  136     , PP.indent 2 (ppAbortedResult ppGP falseBranch)
  137     ]
  138 ppAbortedResult _ (AbortedExit ec _gp) =
  139   "Branch exited:" PP.<+> PP.viaShow ec

  141 setupProfiling ::
  142   IsSymInterface sym =>
  143   sym -> String -> Maybe FilePath ->
  144   IO (IO (), [GenericExecutionFeature sym])
  145 setupProfiling _ _ Nothing = return (return (), [])
  146 setupProfiling sym profSource (Just dir) =
  147   do tbl <- newProfilingTable
  148 
  149      createDirectoryIfMissing True dir
  150 
  151      startRecordingSolverEvents sym tbl
  152 
  153      let profOutFile = dir </> "report_data.js"
  154          saveProf = writeProfileReport profOutFile "Crucible profile" profSource
  155          profOpts = ProfilingOptions
  156                       { periodicProfileInterval = 5
  157                       , periodicProfileAction = saveProf
  158                       }
  159 
  160      pfs <- profilingFeature tbl profilingEventFilter (Just profOpts)
  161      return (saveProf tbl, [pfs])

  163 -- | Given a base Crucible type, compute the corresponding Cryptol type. This
  164 -- should return reasonable results for most language backends, but it may not
  165 -- cover everything. (For instance, MIR arrays use a custom intrinsic type,
  166 -- not 'Crucible.BaseArrayRepr', so those would need to be handled separately.)
  167 baseCryptolType :: Crucible.BaseTypeRepr tp -> Maybe Cryptol.Type
  168 baseCryptolType bt =
  169   case bt of
  170     Crucible.BaseBoolRepr -> pure $ Cryptol.tBit
  171     Crucible.BaseBVRepr w -> pure $ Cryptol.tWord (Cryptol.tNum (natValue w))
  172     Crucible.BaseIntegerRepr -> pure $ Cryptol.tInteger
  173     Crucible.BaseArrayRepr {} -> Nothing
  174     Crucible.BaseFloatRepr _ -> Nothing
  175     Crucible.BaseStringRepr _ -> Nothing
  176     Crucible.BaseComplexRepr  -> Nothing
  177     Crucible.BaseRealRepr     -> Nothing
  178     Crucible.BaseStructRepr ts ->
  179       Cryptol.tTuple <$> baseCryptolTypes ts
  180   where
  181     baseCryptolTypes :: Ctx.Assignment Crucible.BaseTypeRepr args -> Maybe [Cryptol.Type]
  182     baseCryptolTypes Ctx.Empty = pure []
  183     baseCryptolTypes (xs Ctx.:> x) =
  184       do ts <- baseCryptolTypes xs
  185          t <- baseCryptolType x
  186          pure (ts ++ [t])

  188 getGlobalPair ::
  189   Options ->
  190   Crucible.PartialResult sym ext v ->
  191   IO (Crucible.GlobalPair sym v)
  192 getGlobalPair opts pr =
  193   case pr of
  194     Crucible.TotalRes gp -> return gp
  195     Crucible.PartialRes _ _ gp _ -> do
  196       printOutLn opts Info "Symbolic simulation completed with side conditions."
  197       return gp

  199 runCFG ::
  200   (Crucible.IsSyntaxExtension ext, IsSymInterface sym) =>
  201   Crucible.SimContext p sym ext ->
  202   Crucible.SymGlobalState sym ->
  203   Crucible.FnHandle args a ->
  204   Crucible.CFG ext blocks init a ->
  205   Crucible.RegMap sym init ->
  206   IO (Crucible.ExecResult p sym ext (Crucible.RegEntry sym a))
  207 runCFG simCtx globals h cfg args =
  208   do let initExecState =
  209            Crucible.InitialState simCtx globals Crucible.defaultAbortHandler (Crucible.handleReturnType h) $
  210            Crucible.runOverrideSim (Crucible.handleReturnType h)
  211                     (Crucible.regValue <$> (Crucible.callCFG cfg args))
  212      Crucible.executeCrucible [] initExecState

  214 -- | Given a 'Crucible.TypeRepr', compute the corresponding Cryptol type
  215 -- ('Cryptol.Type') and SAWCore type ('Term'). If the 'Crucible.TypeRepr' is
  216 -- unsupported for extraction, throw an error message.
  217 --
  218 -- This is a language backend-agnostic implementation that only supports
  219 -- Crucible base types. Individual language backends will want to add
  220 -- additional cases on top of this function in order to support intrinsic
  221 -- types.
  222 typeReprToSAWTypes ::
  223   Sym ->
  224   SharedContext ->
  225   Crucible.TypeRepr tp ->
  226   IO (Cryptol.Type, Term)
  227 typeReprToSAWTypes sym sc tp =
  228   case Crucible.asBaseType tp of
  229     Crucible.AsBaseType btp -> do
  230       cty <-
  231         case baseCryptolType btp of
  232           Just cty -> pure cty
  233           Nothing ->
  234             fail $ unwords ["Unsupported type for Crucible extraction:", show btp]
  235       scTp <- baseSCType sym sc btp
  236       pure (cty, scTp)
  237 
  238     Crucible.NotBaseType ->
  239       fail $ unwords ["Unsupported type for Crucible extraction:", show tp]

  241 -- | Convert a fresh SAWCore 'Term' to a 'Crucible.RegValue', also binding a
  242 -- fresh What4 constant and associating it to the 'Term'.
  243 --
  244 -- This is a language backend-agnostic implementation that only supports
  245 -- Crucible base types. Individual language backends will want to add
  246 -- additional cases on top of this function in order to support intrinsic
  247 -- types.
  248 termToRegValue ::
  249   Sym ->
  250   -- | The Crucible type of the SAWCore term.
  251   Crucible.TypeRepr tp ->
  252   -- | The SAWCore term.
  253   Term ->
  254   IO (Crucible.RegValue Sym tp)
  255 termToRegValue sym tp t =
  256   case Crucible.asBaseType tp of
  257     Crucible.AsBaseType btp -> do
  258       st <- sawCoreState sym
  259       bindSAWTerm sym st btp t
  260 
  261     Crucible.NotBaseType ->
  262       fail $ unwords ["Unsupported type for Crucible extraction:", show tp]
