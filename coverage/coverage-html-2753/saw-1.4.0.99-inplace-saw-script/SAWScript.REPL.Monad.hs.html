   88 -- REPL Environment.
   89 data REPLState = REPLState
   90   { rContinue   :: Bool
   91   , rIsBatch    :: Bool
   92   , rTopLevelRO :: TopLevelRO
   93   , rTopLevelRW :: TopLevelRW
   94   , rProofState :: Maybe ProofState
   95   }
   96 
   97 -- | Create an initial, empty environment.
   98 initREPL :: Bool -> Options -> TopLevelShellHook -> ProofScriptShellHook -> IO REPLState
   99 initREPL isBatch opts tlhook pshook =
  100   do (ro, rw) <- buildTopLevelEnv opts [] tlhook pshook
  101      return REPLState
  102        { rContinue   = True
  103        , rIsBatch    = isBatch
  104        , rTopLevelRO = ro
  105        , rTopLevelRW = rw
  106        , rProofState  = Nothing
  107        }
  108 
  109 -- | Create an environment from an existing interpreter state.
  110 --
  111 -- FUTURE: it might be nice to be able to read subshell input from a
  112 -- file in batch mode. Or from the same file that's feeding the parent
  113 -- shell, though that's probably difficult. For now, assume any
  114 -- subshells are intended to be interactive.
  115 resumeREPL :: TopLevelRO -> TopLevelRW -> Maybe ProofState -> REPLState
  116 resumeREPL ro rw mpst =
  117     REPLState {
  118         rContinue   = True,
  119         rIsBatch    = False,
  120         rTopLevelRO = ro,
  121         rTopLevelRW = rw,
  122         rProofState = mpst
  123     }

  128 -- | REPL monad context.
  129 newtype REPL a = REPL { unREPL :: StateT REPLState IO a }
  130   deriving (Applicative, Functor, Monad, MonadThrow, MonadCatch, MonadMask, MonadFail)
  131 
  132 deriving instance MonadState REPLState REPL
  133 
  134 -- | Run a REPL action on a REPL state.
  135 runREPL :: REPL a -> REPLState -> IO (a, REPLState)
  136 runREPL m st = do
  137     runStateT (unREPL m) st

  142 -- | REPL exceptions.
  143 data REPLException
  144   = ParseError ParseError
  145   | FileNotFound FilePath
  146   | DirectoryNotFound FilePath
  147   | NoPatError [Error]
  148   | NoIncludeError [IncludeError]
  149   | EvalError EvalError
  150   | ModuleSystemError M.ModuleError
  151   | EvalPolyError T.Schema
  152   | TypeNotTestable T.Type
  153     deriving (Show,Typeable)
  154 
  155 instance X.Exception REPLException
  156 
  157 instance PP REPLException where
  158   ppPrec _ re = case re of
  159     ParseError e         -> ppError e
  160     FileNotFound path    -> sep [ text "File"
  161                                 , text ("`" ++ path ++ "'")
  162                                 , text"not found"
  163                                 ]
  164     DirectoryNotFound path -> sep [ text "Directory"
  165                                   , text ("`" ++ path ++ "'")
  166                                   , text"not found or not a directory"
  167                                   ]
  168     NoPatError es        -> vcat (map pp es)
  169     NoIncludeError es    -> vcat (map ppIncludeError es)
  170     ModuleSystemError me -> pp me
  171     EvalError e          -> pp e
  172     EvalPolyError s      -> text "Cannot evaluate polymorphic value."
  173                          $$ text "Type:" <+> pp s
  174     TypeNotTestable t    -> text "The expression is not of a testable type."
  175                          $$ text "Type:" <+> pp t
  176 
  177 -- | Raise an exception.
  178 raise :: REPLException -> REPL a
  179 raise exn = liftIO $ X.throwIO exn
  180 
  181 -- | Handle any exception type in 'REPL' actions.
  182 catchEx :: X.Exception e => REPL a -> (e -> REPL a) -> REPL a
  183 catchEx m k = m `catch` k
  184 
  185 -- | Handle 'REPLException' exceptions in 'REPL' actions.
  186 catchREPL :: REPL a -> (REPLException -> REPL a) -> REPL a
  187 catchREPL = catchEx
  188 
  189 -- | Similar to 'catchREPL' above, but catches generic IO exceptions from 'fail'.
  190 catchFail :: REPL a -> (String -> REPL a) -> REPL a
  191 catchFail m k = catchJust sel m k
  192   where
  193     sel :: X.IOException -> Maybe String
  194     sel e | isUserError e = Just (ioeGetErrorString e)
  195           | otherwise     = Nothing
  196 
  197 -- | Handle any other exception (except that we ignore async exceptions and exitWith)
  198 catchOther :: REPL a -> (X.SomeException -> REPL a) -> REPL a
  199 catchOther m k = catchJust flt m k
  200  where
  201   flt e
  202     | Just (_ :: X.AsyncException) <- X.fromException e = Nothing
  203     | Just (_ :: ExitCode)       <- X.fromException e = Nothing
  204     | otherwise = Just e
  205 
  206 exceptionProtect :: REPL () -> REPL ()
  207 exceptionProtect cmd =
  208       do chk <- liftIO . makeCheckpoint =<< getTopLevelRW
  209          cmd `catchREPL`  (handlerPP chk)
  210              `catchFail`  (handlerFail chk)
  211              `catchOther` (handlerPrint chk)
  212 
  213     where
  214     handlerPP chk re =
  215       do liftIO (putStrLn "" >> print (pp re))
  216          void $ liftTopLevel (restoreCheckpoint chk)
  217          return ()
  218     handlerPrint chk e =
  219       do liftIO (putStrLn "" >> putStrLn (X.displayException e))
  220          void $ liftTopLevel (restoreCheckpoint chk)
  221 
  222     handlerFail chk s =
  223       do liftIO (putStrLn "" >> putStrLn s)
  224          void $ liftTopLevel (restoreCheckpoint chk)
  225 
  226 liftTopLevel :: TopLevel a -> REPL a
  227 liftTopLevel m = do
  228     ro <- getTopLevelRO
  229     rw <- getTopLevelRW
  230     (result, rw') <- liftIO $ runTopLevel m ro rw
  231     modify (\st -> st { rTopLevelRW = rw' })
  232     return result
  233 
  234 liftProofScript :: ProofScript a -> REPL a
  235 liftProofScript m = do
  236     mpst <- gets rProofState
  237     let pst = case mpst of
  238           Nothing -> panic "liftProofScript" ["Not in ProofScript mode"]
  239           Just ps -> ps
  240     (result, pst') <- liftTopLevel $ runStateT (runExceptT (unProofScript m)) pst
  241     modify (\st -> st { rProofState = Just pst' })
  242     liftTopLevel $ case result of
  243        Left (stats, cex) ->
  244          do ppOpts <- rwPPOpts <$> get
  245             fail (showsProofResult ppOpts (InvalidProof stats cex pst') "")
  246        Right x -> return x

  253 -- | Construct the prompt for the current environment.
  254 getPrompt :: REPL String
  255 getPrompt =
  256   do batch <- gets rIsBatch
  257      if batch then return ""
  258      else do
  259        mpst <- gets rProofState
  260        case mpst of
  261          Nothing ->
  262              return "sawscript> "
  263          Just pst ->
  264              return ("proof ("++show (length (psGoals pst))++")> ")
  265 
  266 shouldContinue :: REPL Bool
  267 shouldContinue =
  268     gets rContinue

  270 stop :: REPL ()
  271 stop =
  272     modify (\st -> st { rContinue = False })

  275 getCryptolExprNames :: REPL [String]

  277   do fNames <- fmap getNamingEnv getCryptolEnv
  278      return (map (show . pp) (Map.keys (MN.namespaceMap NSValue fNames)))

  300 -- | Get visible variable names for Haskeline completion.
  301 getSAWScriptValueNames :: REPL [String]
  302 getSAWScriptValueNames = do
  303   rw <- getTopLevelRW
  304   let avail = rwPrimsAvail rw
  305       visible (_, lc, _, _, _) = Set.member lc avail
  306       Environ valenv _tyenv _cryenv = rwEnviron rw
  307       rbenv = rwRebindables rw
  308   let rnames1 = ScopedMap.allKeys $ ScopedMap.filter visible valenv
  309       rnames2 = Map.keys rbenv
  310   return (map Text.unpack (rnames1 ++ rnames2))
  311 
  312 -- | Get visible type names for Haskeline completion.
  313 getSAWScriptTypeNames :: REPL [String]
  314 getSAWScriptTypeNames = do
  315   rw <- getTopLevelRW
  316   let avail = rwPrimsAvail rw
  317       visible (lc, _) = Set.member lc avail
  318       Environ _valenv tyenv _cryenv = rwEnviron rw
  319   let rnames = ScopedMap.allKeys $ ScopedMap.filter visible tyenv
  320   return (map Text.unpack rnames)
