   46 -- | Get visible Cryptol variable names.
   47 getCryptolExprNames :: REPL [String]
   48 getCryptolExprNames =
   49   do fNames <- fmap getNamingEnv getCryptolEnv
   50      return (map (show . pp) (Map.keys (MN.namespaceMap NSValue fNames)))
   51 
   52 -- | Get visible Cryptol type names.
   53 getCryptolTypeNames :: REPL [String]
   54 getCryptolTypeNames =
   55   do fNames <- fmap getNamingEnv getCryptolEnv
   56      return (map (show . pp) (Map.keys (MN.namespaceMap NSType fNames)))

   58 -- | Get visible variable names for Haskeline completion.
   59 getSAWScriptValueNames :: REPL [String]
   60 getSAWScriptValueNames = do
   61   rw <- getTopLevelRW
   62   let avail = rwPrimsAvail rw
   63       visible (_, lc, _, _, _) = Set.member lc avail
   64       Environ valenv _tyenv _cryenv = rwEnviron rw
   65       rbenv = rwRebindables rw
   66   let rnames1 = ScopedMap.allKeys $ ScopedMap.filter visible valenv
   67       rnames2 = Map.keys rbenv
   68   return (map Text.unpack (rnames1 ++ rnames2))
   69 
   70 -- | Get visible type names for Haskeline completion.
   71 getSAWScriptTypeNames :: REPL [String]
   72 getSAWScriptTypeNames = do

   83 -- | Get names and printed types from the SAWScript variable
   84 --   environment.
   85 --
   86 --   Returns the rebindable environment and a list of scopes, with the
   87 --   innermost scope last. (That's the normal print order.)
   88 --
   89 getSAWScriptVarEnv :: REPL (VarEnvList, [VarEnvList])
   90 getSAWScriptVarEnv = do
   91   rw <- getTopLevelRW
   92   let Environ varenv _tyenv _cryenv = rwEnviron rw
   93       rbenv = rwRebindables rw
   94       avail = rwPrimsAvail rw
   95 
   96   -- Get the rebindable globals. These are always visible.
   97   let extractRB (x, (_pos, ty, _v)) = (x, ty)
   98       rebindables = map extractRB $ Map.assocs rbenv
   99 
  100   -- Get the scopes.
  101   --
  102   -- Reverse the list so the innermost is last, because that's what
  103   -- people will expect to see when it's printed.
  104   --
  105   let extractVar (x, (_pos, lc, ty, _v, _doc)) =
  106         if Set.member lc avail then Just (x, ty)
  107         else Nothing
  108       extractScope entries = mapMaybe extractVar entries
  109       scopes = map extractScope $ reverse $ ScopedMap.scopedAssocs varenv
  110 
  111   return (rebindables, scopes)
  112 
  113 -- Type shorthand for list of pairs of names and type expansions.
  114 type TypeEnvList = [(Text, AST.NamedType)]
  115 
  116 -- | Get names and printed types from the SAWScript type environment.
  117 --
  118 --   Returns a list of scopes, with the innermost scope last. (That's
  119 --   the normal print order.)
  120 --
  121 getSAWScriptTypeEnv :: REPL [TypeEnvList]
  122 getSAWScriptTypeEnv = do
  123   rw <- getTopLevelRW
  124   let Environ _varenv tyenv _cryenv = rwEnviron rw
  125       avail = rwPrimsAvail rw
  126 
  127   -- Get the scopes.
  128   --
  129   -- Reverse the list so the innermost is last, because that's what
  130   -- people will expect to see when it's printed.
  131   --
  132   let extractTyVar (x, (lc, ty)) =
  133         if Set.member lc avail then Just (x, ty)
  134         else Nothing
  135       extractScope entries = mapMaybe extractTyVar entries
  136       scopes = map extractScope $ reverse $ ScopedMap.scopedAssocs tyenv
  137 
  138   return scopes
