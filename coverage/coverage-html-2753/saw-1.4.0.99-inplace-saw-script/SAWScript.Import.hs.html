   29 parseFile :: [Token Pos] -> Either ParseError [Stmt]
   30 parseFile tokens = do
   31   case parseModule tokens of
   32     Left err -> Left err
   33     Right stmts -> Right stmts
   34 
   35 loadFile :: Options -> FilePath -> IO (Either [Text] [Stmt])
   36 loadFile opts fname = do
   37   printOutLn opts Info $ "Loading file " ++ show fname
   38   ftext <- TextIO.readFile fname
   39   -- XXX: the print functions should take care of printing the position
   40   -- (clean this up when we clean out the printing infrastructure)
   41   -- XXX: the print functions should also take care of exiting on an
   42   -- error (immediately or later). For now, print warnings right away
   43   -- and return errors in Either.
   44   case lexSAW fname ftext of
   45     Left (_, pos, txt) -> do
   46       let txt' = Text.pack (show pos) <> ": " <> txt
   47       return $ Left [txt']
   48     Right (tokens, optmsg) -> do
   49       case optmsg of
   50         Nothing -> return ()
   51         -- The types make it possible to get an error back in the
   52         -- optmsg field. Panic if we do, because the lexer isn't
   53         -- supposed to do that, and currently does not.
   54         Just (Error, _, _) ->
   55           panic "loadFile" ["Lexer returned an error in the warning slot"]
   56         Just (vrb, pos, txt) -> do
   57           let txt' = show pos ++ ": " ++ Text.unpack txt
   58           printOutLn opts vrb txt'
   59       case parseFile tokens of
   60         Left err -> return $ Left [Text.pack $ show err]
   61         Right result -> return $ Right result
   62 
   63 findAndLoadFile :: Options -> FilePath -> IO (Either [Text] [Stmt])
   64 findAndLoadFile opts fp = do
   65   let paths = importPath opts
   66   mfname <- findFile paths fp
   67   case mfname of
   68     Nothing -> do
   69         let msgs = [
   70                 "Couldn't find file: " <> Text.pack fp,
   71                 "  Searched in directories:"
   72              ] ++ map (\p -> "    " <> Text.pack p) paths
   73         return $ Left msgs
   74     Just fname ->
   75         loadFile opts fname
