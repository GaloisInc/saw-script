   30 -- | Haskeline-specific repl implementation.
   31 --
   32 -- XXX this needs to handle Ctrl-C, which at the moment will just cause
   33 -- haskeline to exit.  See the function 'withInterrupt' for more info on how to
   34 -- handle this.
   35 repl :: Maybe FilePath -> REPL ()
   36 repl mbBatchFile =
   37   replBody mbBatchFile
   38 
   39 
   40 replBody :: Maybe FilePath -> REPL ()
   41 replBody mbBatchFile =
   42   do settings <- MTL.liftIO (setHistoryFile replSettings)
   43      runInputTBehavior style settings body
   44   where
   45   body = withInterrupt loop
   46 
   47   style = case mbBatchFile of
   48     Nothing   -> defaultBehavior
   49     Just path -> useFile path
   50 
   51   loop = do
   52     prompt <- MTL.lift getPrompt
   53     mb     <- handleInterrupt (return (Just "")) (getInputLines prompt [])
   54     case mb of
   55       Just line
   56         | Just cmd <- parseCommand findCommand (Text.pack line) -> do
   57           continue <- MTL.lift $ do
   58             handleInterrupt handleCtrlC (runCommand cmd)
   59             shouldContinue
   60           when continue loop
   61 
   62         | otherwise -> loop
   63 
   64       Nothing -> return ()
   65 
   66   getInputLines prompt ls =
   67     do mb <- fmap (filter (/= '\r')) <$> getInputLine prompt
   68        let newPrompt = map (\_ -> ' ') prompt
   69        case mb of
   70           Nothing -> return Nothing
   71           Just l | not (null l) && last l == '\\' ->
   72                                       getInputLines newPrompt (init l : ls)
   73                  | otherwise -> return $ Just $ unlines $ reverse $ l : ls

   77 -- | Try to set the history file.
   78 setHistoryFile :: Settings REPL -> IO (Settings REPL)
   79 setHistoryFile ss =
   80   do dir <- getAppUserDataDirectory "saw"
   81      createDirectoryIfMissing True dir
   82      return ss { historyFile = Just (dir </> "history") }
   83    `X.catch` \(X.SomeException {}) -> return ss
   84 
   85 -- | Haskeline settings for the REPL.
   86 replSettings :: Settings REPL
   87 replSettings  = Settings
   88   { complete       = replComp
   89   , historyFile    = Nothing
   90   , autoAddHistory = True
   91   }

   96 -- | Top-level completion for the REPL.
   97 replComp :: CompletionFunc REPL
   98 replComp cursor@(l,r)
   99   | ":" `Text.isPrefixOf` l'
  100   , Just (cmd,rest) <- splitCommand l' = case findCommand cmd of
  101 
  102       [c] | Text.null rest && not (Text.any isSpace l') -> do
  103             return (l, [cmdComp cmd c])
  104           | otherwise -> do
  105             (rest',cs) <- cmdArgument (cBody c) (reverse $ Text.unpack $ sanitize rest, r)
  106             -- XXX this drops any leading spaces in l', which is ok but untidy
  107             return (unwords [rest', reverse $ Text.unpack cmd], cs)
  108 
  109       cmds ->
  110         return (l, [ cmdComp l' c | c <- cmds ])
  111 
  112   | l' == ":" =
  113       return (l, [ cmdComp l' c | c <- findCommand "" ])
  114   | otherwise = completeSAWScriptValue cursor
  115   where
  116   l' = sanitize (Text.pack $ reverse l)
  117 
  118 -- | Generate a completion from a REPL command definition.
  119 cmdComp :: Text -> CommandDescr -> Completion
  120 cmdComp prefix c = Completion
  121   { replacement = Text.unpack $ Text.drop (Text.length prefix) (cName c)
  122   , display     = Text.unpack $ cName c
  123   , isFinished  = True
  124   }
  125 
  126 -- | Dispatch to a completion function based on the kind of completion the
  127 -- command is expecting.
  128 cmdArgument :: CommandBody -> CompletionFunc REPL
  129 cmdArgument ct cursor@(l,_) = case ct of
  130   ExprArg _     -> completeSAWScriptValue cursor
  131   TypeArg _     -> completeSAWScriptType cursor
  132   FilenameArg _ -> completeFilename cursor
  133   ShellArg _    -> completeFilename cursor
  134   NoArg       _ -> return (l,[])
  135 
  136 data LexerMode = ModeNormal | ModeCryptol | ModeCryType | ModeQuote
  137 
  138 lexerMode :: String -> LexerMode
  139 lexerMode = normal
  140   where
  141     normal [] = ModeNormal
  142     normal ('{' : '{' : s) = cryptol s
  143     normal ('{' : '|' : s) = crytype s
  144     normal ('\"' : s) = quote s
  145     normal (_ : s) = normal s
  146 
  147     cryptol [] = ModeCryptol
  148     cryptol ('}' : '}' : s) = normal s
  149     cryptol (_ : s) = cryptol s
  150 
  151     crytype [] = ModeCryType
  152     crytype ('|' : '}' : s) = normal s
  153     crytype (_ : s) = crytype s
  154 
  155     quote [] = ModeQuote
  156     quote ('\"' : s) = normal s
  157     quote (_ : s) = quote s
  158 
  159 isIdentChar :: Char -> Bool
  160 isIdentChar c = isAlphaNum c || c == '_' || c == '\''
  161 
  162 -- | Complete a name from the SAWScript value environment.
  163 completeSAWScriptValue :: CompletionFunc REPL
  164 completeSAWScriptValue cursor@(l, _) = do
  165   ns1 <- getSAWScriptValueNames
  166   ns2 <- getCryptolExprNames
  167   ns3 <- getCryptolTypeNames
  168   let n = reverse (takeWhile isIdentChar l)
  169       nameComps prefix ns = map (nameComp prefix) (filter (prefix `isPrefixOf`) ns)
  170   case lexerMode (reverse l) of
  171     ModeNormal  -> return (l, nameComps n ns1)
  172     ModeCryptol -> return (l, nameComps n ns2)
  173     ModeCryType -> return (l, nameComps n ns3)
  174     ModeQuote   -> completeFilename cursor
  175 
  176 -- | Complete a name from the SAWScript type environment.
  177 completeSAWScriptType :: CompletionFunc REPL
  178 completeSAWScriptType (l, _) = do
  179   ns <- getSAWScriptTypeNames
  180   let prefix = reverse (takeWhile isIdentChar l)
  181       nameComps = map (nameComp prefix) (filter (prefix `isPrefixOf`) ns)
  182   return (l, nameComps)
  183 
  184 -- | Generate a completion from a prefix and a name.
  185 nameComp :: String -> String -> Completion
  186 nameComp prefix c = Completion
  187   { replacement = drop (length prefix) c
  188   , display     = c
  189   , isFinished  = True
  190   }
