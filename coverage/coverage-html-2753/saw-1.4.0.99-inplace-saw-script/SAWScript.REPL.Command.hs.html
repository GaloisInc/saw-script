   79 -- | Commands.
   80 data Command
   81   = Command (REPL ())         -- ^ Successfully parsed command
   82   | Ambiguous Text [Text]     -- ^ Ambiguous command, list of conflicting
   83                               --   commands
   84   | Unknown Text              -- ^ The unknown command
   85 
   86 -- | Command builder.
   87 data CommandDescr = CommandDescr
   88   { cName :: Text
   89   , cAliases :: [Text]
   90   , cBody :: CommandBody
   91   , cHelp :: Text
   92   }
   93 
   94 instance Show CommandDescr where
   95   show cd = Text.unpack $ cName cd
   96 
   97 instance Eq CommandDescr where
   98   (==) = (==) `on` cName
   99 
  100 instance Ord CommandDescr where
  101   compare = compare `on` cName
  102 
  103 data CommandBody
  104   = ExprArg     (Text     -> REPL ())
  105   | TypeArg     (Text     -> REPL ())
  106   | FilenameArg (FilePath -> REPL ())
  107   | ShellArg    (Text     -> REPL ())
  108   | NoArg       (REPL ())
  109 
  110 
  111 -- | Convert the command list to a Trie, expanding aliases.
  112 makeCommands :: [CommandDescr] -> CommandMap
  113 makeCommands list  = foldl insert Trie.empty (concatMap expandAliases list)
  114   where
  115   insert m (name, d) = Trie.insert name d m
  116   expandAliases :: CommandDescr -> [(Text, CommandDescr)]
  117   expandAliases d = (cName d, d) : zip (cAliases d) (repeat d)

  119 -- | REPL command parsing.

  121 commands = makeCommands commandList

  151 genHelp :: [CommandDescr] -> [Text]
  152 genHelp cs = map cmdHelp cs
  153   where
  154   cmdHelp cmd = Text.concat [ "  ", cName cmd, pad (cName cmd), cHelp cmd ]
  155   padding     = 2 + maximum (map (Text.length . cName) cs)
  156   pad n       = Text.replicate (max 0 (padding - Text.length n)) " "

  161 -- | Run a command.
  162 runCommand :: Command -> REPL ()
  163 runCommand c = case c of
  164 
  165   Command cmd -> exceptionProtect cmd
  166 
  167   Unknown cmd -> liftIO (TextIO.putStrLn ("Unknown command: " <> cmd))
  168 
  169   Ambiguous cmd cmds -> liftIO $ do
  170     TextIO.putStrLn (cmd <> " is ambiguous; it could mean one of:")
  171     TextIO.putStrLn ("\t" <> Text.intercalate ", " cmds)

  174 lexSAW :: FilePath -> Text.Text -> REPL [Token Pos]
  175 lexSAW fileName str = do
  176   -- XXX wrap printing of positions in the message-printing infrastructure
  177   case SAWScript.Lexer.lexSAW fileName str of
  178     Left (_, pos, msg) ->
  179          fail $ show pos ++ ": " ++ Text.unpack msg
  180     Right (tokens', Nothing) -> pure tokens'
  181     Right (_, Just (SAWCentral.Options.Error, pos, msg)) ->
  182          fail $ show pos ++ ": " ++ Text.unpack msg
  183     Right (tokens', Just _) -> pure tokens'

  185 typeOfCmd :: Text -> REPL ()
  186 typeOfCmd str
  187   | Text.null str = liftIO $ putStrLn "[error] :type requires an argument"
  188   | otherwise =
  189   do tokens <- lexSAW replFileName str
  190      expr <- case SAWScript.Parser.parseExpression tokens of
  191        Left err -> fail (show err)
  192        Right expr -> return expr
  193      let pos = getPos expr
  194          decl = SS.Decl pos (SS.PWild pos Nothing) Nothing expr
  195      rw <- getTopLevelRW
  196      decl' <- do
  197        let primsAvail = rwPrimsAvail rw
  198            -- XXX it should not be necessary to do this munging
  199            Environ varenv tyenv _cryenvs = rwEnviron rw
  200            squash (defpos, lc, ty, _val, _doc) = (defpos, lc, ReadOnlyVar, ty)
  201            varenv' = Map.map squash $ ScopedMap.flatten varenv
  202            tyenv' = ScopedMap.flatten tyenv
  203            rbenv = rwRebindables rw
  204            rbsquash (defpos, ty, _val) = (defpos, Current, RebindableVar, ty)
  205            rbenv' = Map.map rbsquash rbenv
  206            varenv'' = Map.union varenv' rbenv'
  207        let (errs_or_results, warns) = checkDecl primsAvail varenv'' tyenv' decl
  208        let issueWarning (msgpos, msg) =
  209              -- XXX the print functions should be what knows how to show positions...
  210              putStrLn (show msgpos ++ ": Warning: " ++ msg)
  211        liftIO $ mapM_ issueWarning warns
  212        either failTypecheck return errs_or_results
  213      let schema = case SS.dType decl' of
  214            Just sch -> sch
  215            Nothing ->
  216                -- If the typechecker didn't insert a type, it's bust,
  217                -- so panic. Not much point in printing the expression
  218                -- or position in panic, since it's what the user just
  219                -- typed.
  220                panic "typeOfCmd" ["Typechecker failed to produce a type"]
  221      liftIO $ TextIO.putStrLn $ PPS.pShowText schema
  222 
  223 searchCmd :: Text -> REPL ()
  224 searchCmd str
  225   | Text.null str = liftIO $ putStrLn $ "[error] :search requires at least one argument"
  226   | otherwise =
  227   do tokens <- lexSAW replFileName str
  228      pat <- case SAWScript.Parser.parseSchemaPattern tokens of
  229        Left err -> fail (show err)
  230        Right pat -> return pat
  231      rw <- getTopLevelRW
  232 
  233      -- Always search the entire environment and recognize all type
  234      -- names in the user's pattern, regardless of whether
  235      -- enable_experimental or enable_deprecated is in effect. It is
  236      -- definitely confusing to search for a hidden type and have
  237      -- search treat that as a free type variable and match all kinds
  238      -- of things you didn't intend. It is also better to retrieve
  239      -- invisible/deprecated items and report their existence than to
  240      -- hide them from the search.
  241 
  242      -- FUTURE: it would be nice to be able to use the words
  243      -- "experimental" and "deprecated" in the search term to match
  244      -- against the lifecycle, to allow doing stuff like searching
  245      -- for deprecated functions that take Terms.
  246 
  247      let primsAvail = rwPrimsAvail rw
  248          -- XXX it should not be necessary to do this munging
  249          Environ varenv tyenv _cryenv = rwEnviron rw
  250          rbenv = rwRebindables rw
  251          squash (pos, lc, ty, _val, _doc) = (pos, lc, ReadOnlyVar, ty)
  252          varenv' = Map.map squash $ ScopedMap.flatten varenv
  253          tyenv' = ScopedMap.flatten tyenv
  254          rbsquash (pos, ty, _val) = (pos, Current, RebindableVar, ty)
  255          rbenv' = Map.map rbsquash rbenv
  256          varenv'' = Map.union varenv' rbenv'
  257          (errs_or_results, warns) = checkSchemaPattern everythingAvailable varenv'' tyenv' pat
  258      let issueWarning (msgpos, msg) =
  259            -- XXX the print functions should be what knows how to show positions...
  260            putStrLn (show msgpos ++ ": Warning: " ++ msg)
  261      liftIO $ mapM_ issueWarning warns
  262      pat' <- either failTypecheck return $ errs_or_results
  263      let search = compileSearchPattern tyenv pat'
  264          keep (_pos, _lc, _rb, ty) = matchSearchPattern search ty
  265          allMatches = Map.filter keep varenv'
  266 
  267          -- Divide the results into visible, experimental-not-visible,
  268          -- and deprecated-not-visible.
  269          inspect ::
  270              SS.Name ->
  271              (SS.Pos, PrimitiveLifecycle, SS.Rebindable, SS.Schema) ->
  272              (Map SS.Name (PrimitiveLifecycle, SS.Schema),
  273               Map SS.Name (PrimitiveLifecycle, SS.Schema),
  274               Map SS.Name (PrimitiveLifecycle, SS.Schema)) ->
  275              (Map SS.Name (PrimitiveLifecycle, SS.Schema),
  276               Map SS.Name (PrimitiveLifecycle, SS.Schema),
  277               Map SS.Name (PrimitiveLifecycle, SS.Schema))
  278          inspect name (_pos, lc, _rb, ty) (vis, ex, dep) =
  279              if Set.member lc primsAvail then
  280                  (Map.insert name (lc, ty) vis, ex, dep)
  281              else case lc of
  282                Current -> oops
  283                WarnDeprecated -> oops
  284                HideDeprecated ->
  285                  (vis, ex, Map.insert name (lc, ty) dep)
  286                Experimental ->
  287                  (vis, Map.insert name (lc, ty) ex, dep)
  288              where
  289                oops =
  290                  panic "searchCmd" [
  291                      "Found non-visible object " <> Text.pack (show name) <>
  292                      " with unexpected lifecycle " <> Text.pack (show lc)
  293                  ]
  294          (visMatches, expMatches, depMatches) =
  295              Map.foldrWithKey inspect (Map.empty, Map.empty, Map.empty) allMatches
  296 
  297          printMatch (name, (lc, ty)) = do
  298            let ty' = PPS.pShowText ty
  299                lc' = case lc of
  300                    Current -> ""
  301                    WarnDeprecated -> "  (DEPRECATED AND WILL WARN)"
  302                    HideDeprecated -> "  (DEPRECATED AND UNAVAILABLE BY DEFAULT)"
  303                    Experimental -> "  (EXPERIMENTAL)"
  304            TextIO.putStrLn (name <> " : " <> ty' <> lc')
  305          printMatches matches =
  306            liftIO $ mapM_ printMatch (Map.assocs matches)
  307 
  308          moreMatches matches =
  309              let n = Map.size matches in
  310              if n == 1 then "1 more match"
  311              else show n ++ " more matches"
  312          alsoExperimental =
  313              if not (Map.null expMatches) then
  314                  liftIO $ putStrLn $ moreMatches expMatches ++ " tagged " ++
  315                                  "experimental; use enable_experimental to " ++
  316                                  "see them"
  317              else
  318                  pure ()
  319          alsoDeprecated =
  320              if not (Map.null depMatches) then
  321                  liftIO $ putStrLn $ moreMatches depMatches ++ " tagged " ++
  322                                  "deprecated; use enable_deprecated to " ++
  323                                  "see them"
  324              else
  325                  pure ()
  326 
  327      if not (Map.null visMatches) then do
  328          printMatches visMatches
  329          alsoExperimental
  330          alsoDeprecated
  331      else do
  332          liftIO $ putStrLn "No matches."
  333          if not (Map.null expMatches) then do
  334              liftIO $ putStrLn $ "The following experimental matches require " ++
  335                              "enable_experimental:"
  336              printMatches expMatches
  337              alsoDeprecated
  338          else if not (Map.null depMatches) then do
  339              liftIO $ putStrLn $ "The following deprecated matches require " ++
  340                              "enable_deprecated:"
  341              printMatches depMatches
  342          else
  343              pure ()

  350 envCmd :: REPL ()
  351 envCmd = do
  352   rw <- getTopLevelRW
  353   let Environ varenv _tyenv _cryenv = rwEnviron rw
  354       rbenv = rwRebindables rw
  355       avail = rwPrimsAvail rw
  356 
  357   -- print the rebindable globals first, if any
  358   unless (Map.null rbenv) $ do
  359       let rbsay (x, (_pos, ty, _v)) = do
  360               let ty' = PPS.pShowText ty
  361               TextIO.putStrLn (x <> " : rebindable " <> ty')
  362       liftIO $ mapM_ rbsay $ Map.assocs rbenv
  363       liftIO $ TextIO.putStrLn ""
  364 
  365   -- print the normal environment
  366   let say (x, (_pos, _lc, ty, _v, _doc)) = do
  367           let ty' = PPS.pShowText ty
  368           TextIO.putStrLn (x <> " : " <> ty')
  369       -- Print only the visible objects
  370       keep (_x, (_pos, lc, _ty, _v, _doc)) = Set.member lc avail
  371       -- Insert a blank line in the output where there's a scope boundary
  372       printScope mItems = case mItems of
  373           Nothing -> TextIO.putStrLn ""
  374           Just items -> mapM_ say $ filter keep items
  375       -- Reverse the list of scopes so the innermost prints last,
  376       -- because that's what people will expect to see.
  377       itemses = reverse $ ScopedMap.scopedAssocs varenv
  378   liftIO $ mapM_ printScope $ intersperse Nothing $ map Just itemses
  379 
  380 tenvCmd :: REPL ()
  381 tenvCmd = do
  382   rw <- getTopLevelRW
  383   let avail = rwPrimsAvail rw

  385       say (x, (_lc, ty)) = do
  386           let ty' = PPS.pShowText ty
  387           TextIO.putStrLn (x <> " : " <> ty')
  388       -- Print only the visible objects

  390       -- Insert a blank line in the output where there's a scope boundary

  399 helpCmd :: Text -> REPL ()
  400 helpCmd cmd
  401   | Text.null cmd = liftIO (mapM_ TextIO.putStrLn (genHelp commandList))
  402   | otherwise =
  403     do rw <- getTopLevelRW
  404        -- Note: there's no rebindable builtins and thus no way to
  405        -- attach help text to anything rebindable, so we can ignore
  406        -- the rebindables.
  407        let Environ varenv _tyenv _cryenvs = rwEnviron rw
  408        case ScopedMap.lookup cmd varenv of
  409          Just (_pos, _lc, _ty, _v, Just doc) ->
  410            liftIO $ mapM_ TextIO.putStrLn doc
  411          Just (_pos, _lc, _ty, _v, Nothing) -> do
  412            liftIO $ putStrLn $ "// No documentation is available."
  413            typeOfCmd cmd
  414          Nothing ->
  415            liftIO $ putStrLn "// No such command."

  432 -- SAWScript commands ----------------------------------------------------------
  433 
  434 {- Evaluation is fairly straightforward; however, there are a few important
  435 caveats:
  436 
  437   1. 'return' is type-polymorphic.  This means that 'return "foo"' will produce
  438      a type-polymorphic function 'm -> m String', for any monad 'm'.  It also
  439      means that if you type 'return "foo"' into a naively-written interpreter,
  440      you won't see 'foo'!  The solution is to force each statement that comes
  441      into the REPL to have type 'TopLevel t' ('TopLevel' is the SAWScript
  442      version of 'IO').  This gets done as soon as the statement is parsed.
  443 
  444   2. Handling binding variables to values is somewhat tricky.  When you're
  445      allowed to bind variables in the REPL, you're basically working in the
  446      context of a partially evaluated module: all the results of all the old
  447      computations are in scope, and you're allowed to add new computations that
  448      depend on them.  It's insufficient to merely hang onto the AST for the
  449      old computations, as that could cause them to be evaluated multiple times;
  450      it could also cause their type to be inferred differently several times,
  451      which is bad.  Instead, we hang onto the inferred types of previous
  452      computations and use them to seed the name resolver and the typechecker;
  453      we also hang onto the results and use them to seed the interpreter. -}
  454 sawScriptCmd :: Text -> REPL ()
  455 sawScriptCmd str = do
  456   tokens <- lexSAW replFileName str
  457   case SAWScript.Parser.parseStmtSemi tokens of
  458     Left err -> liftIO $ print err
  459     Right stmt ->
  460       do mpst <- getProofState
  461          case mpst of
  462            Nothing -> void $ liftTopLevel (interpretTopStmt True stmt)
  463            Just _  -> void $ liftProofScript (interpretTopStmt True stmt)
  464 
  465 replFileName :: FilePath
  466 replFileName = "<stdin>"

  482 -- | Strip leading space.
  483 sanitize :: Text -> Text
  484 sanitize  = Text.dropWhile isSpace
  485 
  486 -- | Strip trailing space.
  487 sanitizeEnd :: Text -> Text
  488 sanitizeEnd = Text.dropWhileEnd isSpace
  489 
  490 -- | Split at the first word boundary.
  491 splitCommand :: Text -> Maybe (Text, Text)
  492 splitCommand txt = do
  493   (c, more) <- Text.uncons (sanitize txt) 
  494   case c of
  495     ':'
  496       | (as,bs) <- Text.span (\x -> isPunctuation x || isSymbol x) more
  497       , not (Text.null as) -> Just (Text.cons ':' as, sanitize bs)
  498 
  499       | (as,bs) <- Text.break isSpace more
  500       , not (Text.null as) -> Just (Text.cons ':' as, sanitize bs)
  501 
  502       | otherwise -> Nothing
  503 
  504     _ -> Just (Text.cons c more, "")
  505 
  506 -- | Look up a string in a command list. If given a string that's both
  507 -- itself a command and a prefix of something else, choose that
  508 -- command; otherwise such commands are inaccessible. Also, deduplicate
  509 -- the list of results to avoid silliness with command aliases.
  510 findSomeCommand :: Text -> CommandMap -> [CommandDescr]
  511 findSomeCommand str commandTable = nub $ Trie.lookupWithExact str commandTable
  512 
  513 -- | Look up a string in the command list.
  514 findCommand :: Text -> [CommandDescr]
  515 findCommand str = findSomeCommand str commands
  516 
  517 -- | Parse a line as a command.
  518 parseCommand :: (Text -> [CommandDescr]) -> Text -> Maybe Command
  519 parseCommand findCmd line = do
  520   (cmd,args) <- splitCommand line
  521   let args' = sanitizeEnd args
  522   case findCmd cmd of
  523     -- nothing matched; if it doesn't begin with a colon, eval it
  524     [] -> case Text.uncons cmd of
  525       Nothing -> Nothing
  526       Just (':', _) -> Just (Unknown cmd)
  527       Just _ -> Just (Command (sawScriptCmd line))
  528 
  529     -- matched exactly one command; run it
  530     [c] -> case cBody c of
  531       ExprArg     body -> Just (Command (body args'))
  532       TypeArg     body -> Just (Command (body args'))
  533       FilenameArg body -> Just (Command (body =<< expandHome args'))
  534       ShellArg    body -> Just (Command (body args'))
  535       NoArg       body -> Just (Command  body)
  536 
  537     -- matched several things; complain
  538     cs -> Just (Ambiguous cmd (map cName cs))
  539 
  540   where
  541   expandHome path =
  542     case Text.unpack path of
  543       '~' : c : more | isPathSeparator c -> do dir <- liftIO getHomeDirectory
  544                                                return (dir </> more)
  545       path' -> pure path'
