    9 module SAWScript.REPL.Logo (displayLogo) where
   10 
   11 import SAWScript.Panic (panic)
   12 import SAWVersion.Version (versionText)
   13 import System.Console.ANSI
   14 
   15 type Logo = [String]
   16 
   17 logo :: Bool -> Logo
   18 logo useColor =
   19      [ sgr [SetColor Foreground Dull  White] ++ l | l <- ws ]
   20   ++ [ sgr [SetColor Foreground Vivid Blue ] ++ l | l <- vs ]
   21   ++ [ sgr [SetColor Foreground Dull  Blue ] ++ l | l <- ds ]
   22   ++ [ sgr [Reset] ]
   23   where
   24   sgr | useColor  = setSGRCode
   25       | otherwise = const []
   26   ver = sgr [SetColor Foreground Dull White]
   27         ++ replicate (lineLen - 20 - length versionText) ' '
   28         ++ versionText
   29   ls = (if useColor then logo2 else logo1) ver
   30   line      = case ls of
   31                 line':_ -> line'
   32                 [] -> panic "logo" ["empty lines"]
   33   lineLen   = length line
   34   slen      = length ls `div` 3
   35   (ws,rest) = splitAt slen ls
   36   (vs,ds)   = splitAt slen rest
   37 
   38 logo1 :: String -> [String]
   39 logo1 ver =
   40     [ "   ___  __ _ _ _ _"
   41     , "  / __|/ _\' | | | |"
   42     , "  \\__ \\ (_| | | | |"
   43     , "  |___/\\__,_|\\_,_/ " ++ ver
   44     ]
   45 
   46 logo2 :: String -> [String]
   47 logo2 ver =
   48     [ " ┏━━━┓━━━┓━┓━┓━┓"
   49     , " ┃ ━━┓ ╻ ┃ ┃ ┃ ┃"
   50     , " ┣━━ ┃ ╻ ┃┓ ╻ ┏┛"
   51     , " ┗━━━┛━┛━┛┗━┛━┛ " ++ ver
   52     ]
   53 
   54 displayLogo :: Bool -> IO ()
   55 displayLogo useColor = mapM_ putStrLn (logo useColor)
