   22 -- | Test if a 'Term' is an application of a specific datatype with the
   23 -- supplied context of parameters and indices
   24 ctxAsDataTypeApp :: Name -> [param] ->
   25                     [index] -> Term ->
   26                     Maybe ([Term], [Term])
   27 ctxAsDataTypeApp d params ixs t =
   28   do let (f, args) = asApplyAll t
   29      d' <- asConstant f
   30      guard (d == d')
   31      guard (length args == length params + length ixs)
   32      let (params', ixs') = splitAt (length params) args
   33      pure (params', ixs')
   34 
   35 -- | Test whether a specific datatype occurs in a term.
   36 usesDataType :: Name -> Term -> Bool
   37 usesDataType d t =
   38   case unwrapTermF t of
   39     Constant d'
   40       | d' == d -> True
   41     FTermF (Recursor (recursorDataType -> d'))
   42       | d' == d -> True
   43     tf -> any (usesDataType d) tf
   44 
   45 -- | Check that a type is a valid application of datatype @d@ for use in
   46 -- specific ways in the type of a constructor for @d@. This requires that this
   47 -- application of @d@ be of the form
   48 --
   49 -- > d p1 .. pn x1 .. xm
   50 --
   51 -- where the @pi@ are the distinct bound variables bound in the @params@
   52 -- context, given as argument, and that the @xj@ have no occurrences of @d@. If
   53 -- the given type is of this form, return the @xj@.
   54 asCtorDTApp :: Name -> [(VarName, Term)] -> [index] -> Term -> Maybe [Term]
   55 asCtorDTApp d params dt_ixs (ctxAsDataTypeApp d params dt_ixs ->
   56                                        Just (param_vars, ixs))
   57   | isVarList (map fst params) param_vars && not (any (usesDataType d) ixs)
   58   = Just ixs
   59   where
   60     -- Check that the given list of terms is a list of named
   61     -- variables, one for each parameter
   62     isVarList :: [VarName] -> [Term] -> Bool
   63     isVarList _ [] = True
   64     isVarList (p : ps) ((asVariable -> Just (x, _)) : ts) =
   65       x == p && isVarList ps ts
   66     isVarList _ _ = False
   67 asCtorDTApp _ _ _ _ = Nothing

   70 -- | Check that an argument for a constructor has one of the allowed forms
   71 asCtorArg ::
   72   Name ->
   73   [(VarName, Term)] ->
   74   [index] ->
   75   Term ->
   76   Maybe CtorArg
   77 asCtorArg d _ _ tp
   78   | not (usesDataType d tp)
   79   = Just (ConstArg tp)
   80 asCtorArg d params dt_ixs tp =
   81   do let (zs, ret) = asPiList tp
   82      case asCtorDTApp d params dt_ixs ret of
   83        Just ixs
   84          | not (any (usesDataType d . snd) zs) ->
   85            Just (RecursiveArg zs ixs)
   86        _ ->
   87          Nothing
   88 
   89 -- | Check that a constructor type is a pi-abstraction that takes as input an
   90 -- argument of one of the allowed forms described by 'CtorArg'
   91 asPiCtorArg ::
   92   Name ->
   93   [(VarName, Term)] ->
   94   [index] ->
   95   Term ->
   96   Maybe (VarName, CtorArg, Term)
   97 asPiCtorArg d params dt_ixs t =
   98   case asPi t of
   99     Nothing -> Nothing
  100     Just (nm, tp, rest) ->
  101       case asCtorArg d params dt_ixs tp of
  102         Nothing -> Nothing
  103         Just arg -> Just (nm, arg, rest)
  104 
  105 -- | Helper function for 'mkCtorArgStruct'
  106 mkCtorArgsIxs ::
  107   Name ->
  108   [(VarName, Term)] ->
  109   [index] ->
  110   Term ->
  111   Maybe ([(VarName, CtorArg)], [Term])
  112 mkCtorArgsIxs d params dt_ixs (asCtorDTApp d params dt_ixs -> Just ixs) =
  113   Just ([], ixs)
  114 mkCtorArgsIxs d params dt_ixs ty =
  115   case asPiCtorArg d params dt_ixs ty of
  116     Nothing -> Nothing
  117     Just (x, arg, rest) ->
  118       case mkCtorArgsIxs d params dt_ixs rest of
  119         Nothing -> Nothing
  120         Just (args, ixs) -> Just ((x, arg) : args, ixs)
  121 
  122 -- | Take in a datatype and bindings lists for its parameters and indices, and
  123 -- also a prospective type of a constructor for that datatype, where the
  124 -- constructor type is allowed to have the parameters but not the indices free.
  125 -- Test that the constructor type is an allowed type for a constructor of this
  126 -- datatype, and, if so, build a 'CtorArgStruct' for it.
  127 mkCtorArgStruct ::
  128   Name ->
  129   [(VarName, Term)] ->
  130   [(VarName, Term)] ->
  131   Term ->
  132   Maybe CtorArgStruct
  133 mkCtorArgStruct d params dt_ixs ctor_tp =
  134   case mkCtorArgsIxs d params dt_ixs ctor_tp of
  135     Nothing -> Nothing
  136     Just (args, ctor_ixs) ->
  137       Just (CtorArgStruct params args ctor_ixs)
