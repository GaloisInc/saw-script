   42 -- | Evaluator for shared terms.
   43 evalSharedTerm :: ModuleMap -> Map Ident CPrim -> Map VarIndex CValue -> Term -> CValue
   44 evalSharedTerm m addlPrims varVals t =
   45   runIdentity $ do
   46     cfg <-
   47       Sim.evalGlobal m (Map.union constMap addlPrims) variable
   48       (\_ _ -> Nothing) recursor primHandler lazymux
   49     Sim.evalSharedTerm cfg t
   50   where
   51     lazymux = Prims.lazyMuxValue prims
   52     variable vn _tp =
   53       case Map.lookup (vnIndex vn) varVals of
   54         Just v  -> return v
   55         Nothing -> return $ Prim.userError $ "Unimplemented: free variable " ++ show (vnName vn)
   56     recursor nm _sort =
   57       case nameInfo nm of
   58         ModuleIdentifier "Prelude.Stream" -> Just (pure streamRecOp)
   59         _ -> Nothing
   60     primHandler nm msg env =
   61       return $ Prim.userError $ unlines
   62         [ "Could not evaluate primitive " ++ Text.unpack (toAbsoluteName (nameInfo nm))
   63         , "On argument " ++ show (length env)
   64         , Text.unpack msg
   65         ]

   70 data Concrete

   72 type instance EvalM Concrete = Identity
   73 type instance VBool Concrete = Bool
   74 type instance VWord Concrete = BitVector
   75 type instance VInt  Concrete = Integer

   88 toBool :: CValue -> Bool
   89 toBool (VBool b) = b
   90 toBool x = error $ unwords ["SAWCore.Simulator.Concrete.toBool", show x]
   91 
   92 vWord :: BitVector -> CValue
   93 vWord x = VWord x
   94 
   95 toWord :: CValue -> BitVector
   96 toWord (VWord x) = x
   97 toWord (VVector vv) = Prim.packBitVector (fmap (toBool . runIdentity . force) vv)
   98 toWord x = error $ unwords ["SAWCore.Simulator.Concrete.toWord", show x]
   99 
  100 vStream :: IntTrie CValue -> CValue
  101 vStream x = VExtra (CStream x)
  102 
  103 toStream :: CValue -> IntTrie CValue
  104 toStream (VExtra (CStream x)) = x
  105 toStream x = error $ unwords ["SAWCore.Simulator.Concrete.toStream", show x]
  106 
  107 wordFun :: (BitVector -> CPrim) -> CPrim
  108 wordFun f = Prims.strictFun (\x -> f (toWord x))

  110 -- | op : (n : Nat) -> Vec n Bool -> Nat -> Vec n Bool
  111 bvShiftOp :: (BitVector -> Int -> BitVector) -> CPrim
  112 bvShiftOp natOp =
  113   Prims.constFun $
  114   wordFun $ \x ->
  115   Prims.strictFun $ \y -> Prims.PrimValue $
  116     case y of
  117       VNat n | toInteger n < toInteger (maxBound :: Int) -> vWord (natOp x (fromIntegral n))
  118       _      -> error $ unwords ["SAWCore.Simulator.Concrete.shiftOp", show y]

  122 pure1 :: Applicative f => (a -> b) -> a -> f b
  123 pure1 f x = pure (f x)
  124 
  125 pure2 :: Applicative f => (a -> b -> c) -> a -> b -> f c
  126 pure2 f x y = pure (f x y)
  127 
  128 pure3 :: Applicative f => (a -> b -> c -> d) -> a -> b -> c -> f d
  129 pure3 f x y z = pure (f x y z)
  130 
  131 divOp :: (a -> b -> Maybe c) -> a -> b -> Identity c
  132 divOp f x y = maybe Prim.divideByZero pure (f x y)
  133 
  134 ite :: Bool -> a -> a -> a
  135 ite b x y = if b then x else y

  262 -- primitive bvToNat : (n : Nat) -> Vec n Bool -> Nat;
  263 bvToNatOp :: CPrim
  264 bvToNatOp = Prims.constFun $ wordFun $ Prims.PrimValue . VNat . fromInteger . unsigned

  274 -- primitive intToBv : (n : Nat) -> Integer -> Vec n Bool;
  275 intToBvOp :: CPrim
  276 intToBvOp =
  277   Prims.natFun $ \n ->
  278   Prims.intFun $ \x ->
  279     Prims.PrimValue $ VWord $
  280      if n >= 0 then bv (fromIntegral n) x
  281                else bvNeg n $ bv (fromIntegral n) $ negate x

  285 toIntModOp :: CPrim
  286 toIntModOp =
  287   Prims.natFun $ \n ->
  288   Prims.intFun $ \x ->
  289     Prims.PrimValue $ VIntMod n (x `mod` toInteger n)

  293   Prims.constFun $

  298 intModEqOp =
  299   Prims.constFun $

  301   Prims.intModFun $ \y ->

  304 intModBinOp :: (Integer -> Integer -> Integer) -> CPrim
  305 intModBinOp f =
  306   Prims.natFun $ \n ->
  307   Prims.intModFun $ \x ->
  308   Prims.intModFun $ \y ->
  309     Prims.PrimValue $ VIntMod n (f x y `mod` toInteger n)

  311 intModUnOp :: (Integer -> Integer) -> CPrim

  313   Prims.natFun $ \n ->
  314   Prims.intModFun $ \x ->
  315     Prims.PrimValue $ VIntMod n (f x `mod` toInteger n)

  319 -- MkStream :: (a :: sort 0) -> (Nat -> a) -> Stream a;
  320 mkStreamOp :: CPrim
  321 mkStreamOp =
  322   Prims.constFun $
  323   Prims.strictFun $ \f -> Prims.PrimValue $
  324     vStream (fmap (\n -> runIdentity (apply f (ready (VNat n)))) IntTrie.identity)
  325 
  326 -- streamGet :: (a :: sort 0) -> Stream a -> Nat -> a;
  327 streamGetOp :: CPrim
  328 streamGetOp =
  329   Prims.constFun $
  330   Prims.strictFun $ \xs ->
  331   Prims.strictFun $ \ix ->
  332   Prims.Prim $ pure $ streamGet (toStream xs) ix
  333 
  334 streamGet :: IntTrie CValue -> CValue -> CValue
  335 streamGet xs ix =
  336   case ix of
  337     VNat n -> IntTrie.apply xs (toInteger n)
  338     VIntToNat (VInt i) -> IntTrie.apply xs i
  339     VBVToNat _ w -> IntTrie.apply xs (unsigned (toWord w))
  340     n -> panic "streamGet" ["Expected Nat value; found " <> Text.pack (show n)]
  341 
  342 -- Stream#rec :
  343 --   (a : sort 0) -> (p : Stream a -> sort 0) ->
  344 --   ((f : Nat -> a) -> p (MkStream a f)) -> (str : Stream a) -> p str
  345 streamRecOp :: CValue
  346 streamRecOp =
  347   VFun $ \_a -> pure $
  348   VFun $ \_p -> pure $
  349   vStrictFun $ \f1 -> pure $
  350   vStrictFun $ \xs ->
  351   do let f = vStrictFun $ \ix -> pure (streamGet (toStream xs) ix)
  352      apply f1 (ready f)
