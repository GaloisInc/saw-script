   31 -- | Maps string names to values, allowing for partial key matches and
   32 --   querying.
   33 --
   34 --   Each node contains a map from characters to subnodes, and
   35 --   possibly also a value corresponding to the characters seen so
   36 --   far.
   37 data Trie a = Node (Map Char (Trie a)) (Maybe a)
   38     deriving (Show)
   39 
   40 -- | The empty `Trie`.
   41 empty :: Trie a
   42 empty  = Node Map.empty Nothing
   43 
   44 -- | Insert a value into a `Trie`. Will panic if a value already exists
   45 --   with the same key.
   46 insert :: Text -> a -> Trie a -> Trie a
   47 insert k v node0 = insert' k node0
   48   where
   49   -- the insertion value is invariant so recurse without it
   50   insert' key (Node tbl mbValue) = case Text.uncons key of
   51       Nothing -> case mbValue of
   52           Nothing ->
   53               -- Reached the end of the key and there's nothing here yet;
   54               -- insert the value.
   55               Node tbl (Just v)
   56               -- There's already something here; croak
   57           Just _  ->
   58               panic "insert" ["Key already exists: " <> k]
   59       Just (c, key') ->
   60           let subnode = case Map.lookup c tbl of
   61                 Nothing -> empty
   62                 Just n -> n
   63               subnode' = insert' key' subnode
   64           in
   65           Node (Map.insert c subnode' tbl) mbValue
   66 
   67 -- | Return all values in the `Trie` whose keys match the given
   68 --   prefix.
   69 lookup :: Text -> Trie a -> [a]
   70 lookup key node@(Node tbl _) = case Text.uncons key of
   71     Nothing ->
   72         leaves node
   73     Just (c, key') -> case Map.lookup c tbl of
   74         Nothing -> []
   75         Just node' -> lookup key' node'
   76 
   77 -- | Return all values in the `Trie` whose keys match the given
   78 --   prefix. However, if an exact match to the key exists, return just
   79 --   that match.
   80 lookupWithExact :: Text -> Trie a -> [a]
   81 lookupWithExact key node@(Node tbl mbValue) = case Text.uncons key of
   82     Nothing -> case mbValue of
   83         Nothing -> leaves node
   84         Just val -> [val]
   85     Just (c, key') -> case Map.lookup c tbl of
   86         Nothing -> []
   87         Just node' -> lookupWithExact key' node'
   88 
   89 -- | Return all of the values from a Trie.
   90 leaves :: Trie a -> [a]
   91 leaves (Node tbl mbValue) =
   92     maybeToList mbValue ++ concatMap leaves (Map.elems tbl)
