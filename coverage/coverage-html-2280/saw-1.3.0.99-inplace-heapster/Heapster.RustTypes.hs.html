  135 -- | Reassociate a binding list
  136 mbAssoc :: prx1 ctx1 -> RAssign prx2 ctx2 -> RAssign prx3 ctx3 ->
  137            Mb (ctx1 :++: (ctx2 :++: ctx3)) a ->
  138            Mb ((ctx1 :++: ctx2) :++: ctx3) a
  139 mbAssoc ctx1 ctx2 ctx3 mb_a | Refl <- RL.appendAssoc ctx1 ctx2 ctx3 = mb_a
  140 
  141 -- | Reassociate a binding list in the reverse direction of 'mbAssoc'
  142 mbUnAssoc :: prx1 ctx1 -> RAssign prx2 ctx2 -> RAssign prx3 ctx3 ->
  143              Mb ((ctx1 :++: ctx2) :++: ctx3) a ->
  144              Mb (ctx1 :++: (ctx2 :++: ctx3)) a
  145 mbUnAssoc ctx1 ctx2 ctx3 mb_a | Refl <- RL.appendAssoc ctx1 ctx2 ctx3 = mb_a
  146 
  147 -- | Prove type-level equality by reassociating four type lists
  148 appendAssoc4 :: RAssign Proxy ctx1 -> RAssign Proxy ctx2 ->
  149                 RAssign Proxy ctx3 -> RAssign Proxy ctx4 ->
  150                 ctx1 :++: ((ctx2 :++: ctx3) :++: ctx4) :~:
  151                 ((ctx1 :++: ctx2) :++: ctx3) :++: ctx4
  152 appendAssoc4 ctx1 ctx2 ctx3 ctx4
  153   | Refl <- RL.appendAssoc ctx1 (RL.append ctx2 ctx3) ctx4
  154   , Refl <- RL.appendAssoc ctx1 ctx2 ctx3
  155   = Refl
  156 
  157 -- | Reassociate a binding list of four contexts
  158 mbAssoc4 :: RAssign Proxy ctx1 -> RAssign Proxy ctx2 ->
  159             RAssign Proxy ctx3 -> RAssign Proxy ctx4 ->
  160             Mb (ctx1 :++: ((ctx2 :++: ctx3) :++: ctx4)) a ->
  161             Mb (((ctx1 :++: ctx2) :++: ctx3) :++: ctx4) a
  162 mbAssoc4 ctx1 ctx2 ctx3 ctx4 mb_a
  163   | Refl <- appendAssoc4 ctx1 ctx2 ctx3 ctx4 = mb_a

  165 -- | Combine bindings lists using 'mbCombine' and reassociate them
  166 mbCombineAssoc ::
  167   prx1 ctx1 ->
  168   RAssign prx2 ctx2 ->
  169   RAssign prx3 ctx3 ->
  170   Mb ctx1 (Mb (ctx2 :++: ctx3) a) ->
  171   Mb ((ctx1 :++: ctx2) :++: ctx3) a
  172 mbCombineAssoc _ ctx2 ctx3
  173   = mbCombine (RL.mapRAssign (const Proxy) ctx3)
  174   . mbCombine (RL.mapRAssign (const Proxy) ctx2)
  175   . fmap (mbSeparatePrx ctx2 ctx3)

  177 -- | Combine bindings lists using 'mbCombine' and reassociate them
  178 mbCombineAssoc4 ::
  179   RAssign Proxy ctx1 -> RAssign Proxy ctx2 ->
  180   RAssign Proxy ctx3 -> RAssign Proxy ctx4 ->
  181   Mb ctx1 (Mb ((ctx2 :++: ctx3) :++: ctx4) a) ->
  182   Mb (((ctx1 :++: ctx2) :++: ctx3) :++: ctx4) a
  183 mbCombineAssoc4 ctx1 ctx2 ctx3 ctx4 mb_mb_a
  184   | Refl <- appendAssoc4 ctx1 ctx2 ctx3 ctx4

  187 -- | Prepend and reassociate an 'RAssign'
  188 assocAppend :: RAssign f ctx1 -> prx2 ctx2 -> RAssign prx3 ctx3 ->

  190                RAssign f ((ctx1 :++: ctx2) :++: ctx3)

  216 -- | The Rust conversion monad is just a state-error monad
  217 newtype RustConvM a =
  218   RustConvM { unRustConvM :: ReaderT RustConvInfo (Except String) a }
  219   deriving (Functor, Applicative, Monad,
  220             MonadError String, MonadReader RustConvInfo)

  222 instance Fail.MonadFail RustConvM where
  223   fail = RustConvM . throwError

  225 instance MonadBind RustConvM where
  226   mbM mb_m = RustConvM $ mbM $ fmap unRustConvM mb_m

  228 -- | Prefix any error message with location information
  229 atSpan :: Span -> RustConvM a -> RustConvM a
  230 atSpan span m =
  231   catchError m (\msg -> fail ("At " ++ show span ++ ": " ++ msg))

  233 -- | Run a Rust conversion computation with the given 'RustConvInfo', lifting
  234 -- any errors into the outer monad
  235 runLiftRustConvM :: Fail.MonadFail m => RustConvInfo -> RustConvM a -> m a
  236 runLiftRustConvM info (RustConvM m) =
  237   case runExcept (runReaderT m info) of
  238     Left err -> fail err
  239     Right x -> return x

  279 -- | Run a 'RustConvM' computation in a context of bound type-level variables,

  334 -- | Build a 'ShapeFun' for the given 'RustName' from a function on permission
  335 -- expressions of the supplied types
  336 mkShapeFun :: RustName -> CruCtx ctx ->
  337               (PermExprs ctx -> PermExpr (LLVMShapeType w)) -> ShapeFun w
  338 mkShapeFun nm ctx f = \some_exprs exprs_str -> case some_exprs of
  339   Some exprs
  340     | Just Refl <- testEquality ctx (typedPermExprsCtx exprs) ->
  341       return $ f (typedPermExprsExprs exprs)
  342   _ ->
  343     fail $ renderDoc $ fillSep
  344     [ pretty "Converting application of type:" <+> pretty (show nm)
  345     , pretty "To arguments:" <+> pretty exprs_str
  346     , pretty "Arguments not of expected types:" <+> pretty ctx ]

  349 constShapeFun :: RustName -> PermExpr (LLVMShapeType w) -> ShapeFun w
  350 constShapeFun nm sh = mkShapeFun nm CruCtxNil (const sh)

  352 -- | Test if a shape is \"option-like\", meaning it is a tagged union shape with
  353 -- two tags, one of which has contents and one which has no contents; i.e., it
  354 -- is of the form
  355 --
  356 -- > (fieldsh(eq(llvmword(bv1)));sh) orsh (fieldsh(eq(llvmword(bv2))))
  357 --
  358 -- or
  359 --
  360 -- > (fieldsh(eq(llvmword(bv1)))) orsh (fieldsh(eq(llvmword(bv2)));sh)
  361 --
  362 -- where @sh@ is non-empty. If so, return the non-empty shape @sh@, called the
  363 -- \"payload\" shape.
  364 matchOptionLikeShape :: PermExpr (LLVMShapeType w) ->
  365                         Maybe (PermExpr (LLVMShapeType w))
  366 matchOptionLikeShape top_sh = case asTaggedUnionShape top_sh of
  367   Just (SomeTaggedUnionShape (taggedUnionDisjsNoTags ->
  368                               [PExpr_EmptyShape, PExpr_EmptyShape])) -> Nothing
  369   Just (SomeTaggedUnionShape (taggedUnionDisjsNoTags ->
  370                               [PExpr_EmptyShape, sh])) -> Just sh
  371   Just (SomeTaggedUnionShape (taggedUnionDisjsNoTags ->
  372                               [sh, PExpr_EmptyShape])) -> Just sh
  373   _ -> Nothing
  374 
  375 -- | Test if a shape-in-binding is \"option-like\" as per 'matchOptionLikeShape'
  376 matchMbOptionLikeShape :: Mb ctx (PermExpr (LLVMShapeType w)) ->
  377                           Maybe (Mb ctx (PermExpr (LLVMShapeType w)))
  378 matchMbOptionLikeShape =
  379   mbMaybe . mbMapCl $(mkClosed [| matchOptionLikeShape |])
  380 
  381 -- | Test if a shape is a sum type where one branch is empty, i.e., a tagged
  382 -- union shape with two tags, one of which has the false shape as contents. If
  383 -- so, return the non-false shape @sh@.
  384 matchSumFalseShape :: PermExpr (LLVMShapeType w) ->
  385                       Maybe (PermExpr (LLVMShapeType w))
  386 matchSumFalseShape top_sh = case asTaggedUnionShape $ simplifyShape top_sh of
  387   Just (SomeTaggedUnionShape (taggedUnionDisjsNoTags -> [sh1, sh2]))
  388     | PExpr_FalseShape <- simplifyShape sh1 -> Just sh2
  389   Just (SomeTaggedUnionShape (taggedUnionDisjsNoTags -> [sh1, sh2]))
  390     | PExpr_FalseShape <- simplifyShape sh2 -> Just sh1
  391   _ -> Nothing

  394 -- | Build a `ShapeFun` from a `SomeNamedShape`
  395 namedShapeShapeFun :: (1 <= w, KnownNat w) => RustName -> NatRepr w ->
  396                       SomeNamedShape -> RustConvM (ShapeFun w)
  397 -- For an option-like shape, try to do discriminant elision
  398 namedShapeShapeFun nm w (SomeNamedShape nmsh)
  399   | Just Refl <- testEquality w (natRepr nmsh)
  400   , DefinedShapeBody mb_sh <- namedShapeBody nmsh
  401   , Just mb_payload_sh <- matchMbOptionLikeShape mb_sh =
  402     return $ mkShapeFun nm (namedShapeArgs nmsh) $ \exprs ->
  403     case simplifyShape (subst (substOfExprs exprs) mb_payload_sh) of
  404 
  405       -- If the payload is a pointer shape, return payload orsh eq(0)
  406       payload_sh@(isLLVMPointerShape -> True) ->
  407         PExpr_OrShape payload_sh $ llvmEqWordShape w 0
  408 
  409       -- If the payload is a tagged union shape, add an extra tag with an empty
  410       -- shape for its argument
  411       (asTaggedUnionShape -> Just (SomeTaggedUnionShape tag_u)) ->
  412         let new_tag =
  413               foldr max 0 $ map ((1+) . BV.asUnsigned) $
  414               taggedUnionTags tag_u in
  415         taggedUnionToShape $
  416         taggedUnionCons (BV.mkBV knownNat new_tag) (llvmEqWordShape w new_tag)
  417         tag_u
  418 
  419       -- Otherwise, just use the named shape itself
  420       _ -> PExpr_NamedShape Nothing Nothing nmsh exprs
  421 
  422 namedShapeShapeFun nm w (SomeNamedShape nmsh)
  423   | Just Refl <- testEquality w (natRepr nmsh) =
  424     return $ mkShapeFun nm (namedShapeArgs nmsh) $ \exprs ->
  425     case namedShapeBody nmsh of
  426       -- Test if nmsh applied to exprs unfolds to a sum with false, and if so,
  427       -- return just the non-false payload shape
  428       DefinedShapeBody mb_sh
  429         | unfolded_sh <- simplifyShape (subst (substOfExprs exprs) mb_sh)
  430         , Just sh <- matchSumFalseShape unfolded_sh ->
  431           sh
  432       -- Otherwise just return the named shape applied to exprs
  433       _ -> PExpr_NamedShape Nothing Nothing nmsh exprs
  434 
  435 namedShapeShapeFun _ w (SomeNamedShape nmsh) =
  436   fail $ renderDoc $ fillSep
  437   [pretty "Incorrect size of shape" <+> pretty (namedShapeName nmsh),
  438    pretty "Expected:" <+> pretty (intValue w),
  439    pretty "Actual:" <+> pretty (intValue (natRepr nmsh))]

  487 -- | Decide whether a Rust type definition is polymorphic and \"Option-like\";
  488 -- that is, it contains only one data-bearing variant, and the data is of the
  489 -- polymorphic type
  490 isPolyOptionLike :: Item Span -> Bool
  491 isPolyOptionLike (Enum _ _ _ variants (Generics _ [TyParam _ t _ _ _] _ _) _) =
  492   -- Short-circuit if no variant carries the type parameter. Otherwise check
  493   -- that all other variants carry nothing
  494   case find containsT variants of
  495     Nothing -> False
  496     Just v -> all isUnitVariant (delete v variants)
  497   where
  498     containsT (Variant _ _ (TupleD [StructField _ _ (PathTy _ (Path _ [PathSegment t' _ _] _) _) _ _] _) _ _) =
  499       name t == name t'
  500     containsT (Variant _ _ (StructD [StructField _ _ (PathTy _ (Path _ [PathSegment t' _ _] _) _) _ _] _) _ _) =
  501       name t == name t'
  502     containsT _ = False
  503 
  504     isUnitVariant (Variant _ _ (UnitD _) _ _) = True
  505     isUnitVariant _ = False

  565 instance RsConvert w (Ty Span) (PermExpr (LLVMShapeType w)) where
  566   rsConvert w (Slice tp _) =
  567     do sh <- rsConvert w tp
  568        case llvmShapeLength sh of
  569          Just (bvMatchConstInt -> Just stride) ->
  570            return (PExpr_ExShape $ nu $ \n ->
  571                     PExpr_ArrayShape (PExpr_Var n) (fromInteger stride) sh)
  572          _ ->
  573            rsPPInfo >>= \ppInfo ->
  574            fail ("rsConvert: slices not supported for dynamically-sized type: "
  575                  ++ show (RustPP.pretty tp) ++ " with translation:\n"
  576                  ++ renderDoc (permPretty ppInfo sh))
  577   rsConvert _ (Rptr Nothing _ _ _) =
  578     fail "rsConvert: lifetimes must be supplied for reference types"
  579   rsConvert w (Rptr (Just rust_l) mut tp' _) =
  580     do l <- rsConvert w rust_l
  581        sh <- rsConvert w tp'
  582        rw <- rsConvert w mut
  583        case sh of
  584          -- Test if sh is a slice type = an array of existential length
  585          (matchSliceShape -> Just (stride,mb_sh)) ->
  586            -- If so, build a "fat pointer" = a pair of a pointer to our array
  587            -- shape plus a length value
  588            return $ PExpr_ExShape $ nu $ \n ->
  589            PExpr_SeqShape (PExpr_PtrShape rw (Just l) $
  590                            PExpr_ArrayShape (PExpr_Var n) stride $
  591                            subst1 (PExpr_Var n) mb_sh)
  592            (PExpr_FieldShape $ LLVMFieldShape $ ValPerm_Eq $
  593             PExpr_LLVMWord $ PExpr_Var n)
  594 
  595          -- If it's not a slice, make sure it has a known size
  596          _ | Just len <- llvmShapeLength sh
  597            , isJust (bvMatchConst len) ->
  598              return $ PExpr_PtrShape rw (Just l) sh
  599 
  600          -- Otherwise, it's a non-standard dynamically-sized type, which we
  601          -- don't quite know how to handle yet...
  602          _ -> fail "rsConvert: pointer to non-slice dynamically-sized type"
  603   rsConvert w (PathTy Nothing path _) =
  604     do mrec <- asks rciRecType
  605        case mrec of
  606          Just (rec_n, rec_arg_ns, sh_nm)
  607            | rec_n == rsPathName path &&
  608              all isNamedParams (rsPathParams path) &&
  609              rec_arg_ns == concatMap pParamNames (rsPathParams path) ->
  610              PExpr_Var <$> castTypedM "TypedName" (LLVMShapeRepr (natRepr w)) sh_nm
  611          Just (rec_n, _, _)
  612            | rec_n == rsPathName path -> fail "Arguments do not match"
  613          _ ->
  614            do shapeFn <- rsConvert w (rsPathName path)
  615               someTypedArgs <- rsConvert w (rsPathParams path)
  616               shapeFn someTypedArgs $ show (RustPP.prettyUnresolved path)
  617   rsConvert (w :: prx w) (BareFn _ abi rust_ls2 fn_tp span) =
  618     do Some3FunPerm fun_perm <- rsConvertMonoFun w span abi rust_ls2 fn_tp
  619        let args = funPermArgs fun_perm
  620        case cruCtxToReprEq args of
  621          Refl ->
  622            return $ PExpr_FieldShape $ LLVMFieldShape @w @w $ ValPerm_Conj1 $
  623            Perm_LLVMFunPtr
  624            (FunctionHandleRepr (cruCtxToRepr args) (funPermRet fun_perm)) $
  625            ValPerm_Conj1 $ Perm_Fun fun_perm
  626   rsConvert w (TupTy tys _) =
  627     do tyShs <- mapM (rsConvert w) tys
  628        return $ foldr PExpr_SeqShape PExpr_EmptyShape tyShs
  629   rsConvert _ (Never _) =
  630     return $ PExpr_FalseShape
  631   rsConvert _ tp = fail ("Rust type not supported: " ++ show tp)

  647 -- | Return true if and only if the provided Rust type definition is recursive
  648 isRecursiveDef :: Item Span -> Bool
  649 isRecursiveDef item =
  650   case item of
  651     Enum _ _ n variants _ _ -> any (containsName n . getVD) variants
  652     StructItem _ _ n vd _ _ -> containsName n vd
  653     _ -> False
  654 
  655   where
  656     tyContainsName :: Ident -> Ty Span -> Bool
  657     tyContainsName i ty =
  658       case ty of
  659         Slice t _                  -> tyContainsName i t
  660         Language.Rust.Syntax.Array t _ _ -> tyContainsName i t
  661         Ptr _ t _                  -> tyContainsName i t
  662         Rptr _ _ t _               -> tyContainsName i t
  663         TupTy ts _                 -> any (tyContainsName i) ts
  664         PathTy _ (Path _ segs _) _ -> any (segContainsName i) segs
  665         ParenTy t _                -> tyContainsName i t
  666         _                          -> False
  667 
  668     segContainsName :: Ident -> PathSegment Span -> Bool
  669     segContainsName i (PathSegment i' mParams _) =
  670       i == i' || case mParams of
  671                    Nothing -> False
  672                    Just params -> paramsContainName i params
  673 
  674     paramsContainName :: Ident -> PathParameters Span -> Bool
  675     paramsContainName i (AngleBracketed _ tys _ _) = any (tyContainsName i) tys
  676     paramsContainName _ (Parenthesized _ _ _) = error "Parenthesized types not supported"
  677 
  678     typeOf :: StructField Span -> Ty Span
  679     typeOf (StructField _ _ t _ _) = t
  680 
  681     getVD :: Variant Span -> VariantData Span
  682     getVD (Variant _ _ vd _ _) = vd
  683 
  684     containsName :: Ident -> VariantData Span -> Bool
  685     containsName i (StructD fields _) = any (tyContainsName i) $ typeOf <$> fields
  686     containsName i (TupleD fields _) = any (tyContainsName i) $ typeOf <$> fields
  687     containsName _ (UnitD _) = False

 1026 -- | Try to compute the layout of a structure of the given shape as a value,
 1027 -- over 1 or more registers, if this is possible
 1028 layoutArgShapeByVal :: (1 <= w, KnownNat w) => Abi ->
 1029                        PermExpr (LLVMShapeType w) ->
 1030                        MaybeT RustConvM ArgLayout
 1031 
 1032 -- The empty shape --> no values
 1033 layoutArgShapeByVal Rust PExpr_EmptyShape = return argLayout0
 1034 
 1035 -- Named shapes that unfold --> layout their unfoldings
 1036 layoutArgShapeByVal Rust (PExpr_NamedShape rw l nmsh args)
 1037   | TrueRepr <- namedShapeCanUnfoldRepr nmsh
 1038   , Just sh' <- unfoldModalizeNamedShape rw l nmsh args =
 1039     layoutArgShapeByVal Rust sh'
 1040 
 1041 -- Opaque named shapes that are bigger than 16 bytes --> not laid out by value
 1042 --
 1043 -- FIXME: if an opaque named shape somehow corresponds to > 2 fields, it is also
 1044 -- not laid out by value
 1045 layoutArgShapeByVal Rust sh@(PExpr_NamedShape _ _ nmsh _)
 1046   | not (namedShapeCanUnfold nmsh)
 1047   , Just len <- llvmShapeLength sh
 1048   , bvLt (bvInt 16) len =
 1049     mzero
 1050 
 1051 -- Opaque named shapes that could potentially be laid out by value are an error,
 1052 -- because we do not know their representation
 1053 layoutArgShapeByVal Rust (PExpr_NamedShape _ _ nmsh _)
 1054   | not (namedShapeCanUnfold nmsh) =
 1055     lift $ fail $ renderDoc
 1056     (pretty "layoutArgShapeByVal: Cannot lay out opaque named shape by value:"
 1057      <+> pretty (namedShapeName nmsh))
 1058 
 1059 -- The ptr shape --> a single pointer value, if we know its length
 1060 layoutArgShapeByVal Rust (PExpr_PtrShape maybe_rw maybe_l sh)
 1061   | Just bp <- llvmBlockAdjustModalities maybe_rw maybe_l <$> shapeToBlock sh =
 1062     return $ argLayout1 $ ValPerm_LLVMBlock bp

 1065 layoutArgShapeByVal Rust (PExpr_PtrShape _ _ sh) =
 1066   lift rsPPInfo >>= \ppInfo ->
 1067   lift $ fail $ renderDoc $ fillSep
 1068   [pretty "layoutArgShapeByVal: Shape with unknown length:",
 1069    permPretty ppInfo sh]
 1070 
 1071 -- A field shape --> the contents of the field
 1072 layoutArgShapeByVal Rust (PExpr_FieldShape (LLVMFieldShape p)) =
 1073   return $ argLayout1 p
 1074 
 1075 -- Array shapes have unknown length, and so are never passed by value
 1076 layoutArgShapeByVal Rust (PExpr_ArrayShape _ _ _) = mzero
 1077 
 1078 -- Sequence shapes are only laid out as values in the Rust ABI if the result has
 1079 -- at most two fields
 1080 layoutArgShapeByVal Rust (PExpr_SeqShape sh1 sh2) =
 1081   do layout1 <- layoutArgShapeByVal Rust sh1
 1082      layout2 <- layoutArgShapeByVal Rust sh2
 1083      let layout = appendArgLayout layout1 layout2
 1084      if argLayoutNumFields layout <= 2 then return layout else mzero
 1085 
 1086 -- Disjunctive shapes are only laid out as values in the Rust ABI if both sides
 1087 -- can be laid out as values that we can coerce to have the same number of type
 1088 -- of fields.
 1089 --
 1090 -- FIXME: The check for whether we can do this coercion is currently done by
 1091 -- disjoinArgLayouts, but it is probably ABI-specific, so should be performed by
 1092 -- a function that knows how to join two lists of field types depending on the
 1093 -- ABI
 1094 layoutArgShapeByVal Rust (PExpr_OrShape sh1 sh2) =
 1095   do layout1 <- layoutArgShapeByVal Rust sh1
 1096      layout2 <- layoutArgShapeByVal Rust sh2
 1097      case disjoinArgLayouts layout1 layout2 of
 1098        Just layout -> return layout
 1099        Nothing -> mzero

 1133 -- register arguments of a function. If the argument is laid out as a value,
 1134 -- then it has no output permissions, but if it is laid out as a pointer, the
 1135 -- memory occupied by that pointer is returned with an empty shape.

 1145 -- | Compute the layout for the inputs and outputs of a function with the given

 1422   abstractVarsForLifetimes (Just l) p@(ValPerm_Conj ps)
 1423     | any (/= l) (mapMaybe atomicPermLifetime ps)
 1424     , tp:_ <- mapMaybe atomicPermType ps =
 1425       bindSomeMbWithPerms (abstractVarsForLifetimes Nothing p) $ \p' ->
 1426       ValPerm_Eq <$> PExpr_Var <$> someMbWithPermsVar1 tp p'
 1427   abstractVarsForLifetimes l (ValPerm_Conj ps) =
 1428     ValPerm_Conj <$> traverse (abstractVarsForLifetimes l) ps
 1429   abstractVarsForLifetimes l (ValPerm_Exists mb_p) =
 1430     -- Any existentials also become abstracted variables, so they can be bound
 1431     -- as ghosts or gouts (depending on whether they occur in the input or
 1432     -- output permissions)
 1433     mbSomeMbWithPerms knownRepr $ fmap (abstractVarsForLifetimes l) mb_p
 1434   abstractVarsForLifetimes _ p = pure p

 1436 -- NOTE: for AtomicPerms, we don't ever replace the permission itself, since we
 1437 -- don't want to replace each individual permission pi in a conjunction p1*..*pn
 1438 -- with an equality perm, but instead want to replace the entire conjunction all
 1439 -- at once. This is handled in the above case for ValPerm_Conj.
 1440 instance AbstractVarsForLifetimes (AtomicPerm a) where
 1441   abstractVarsForLifetimes _ (Perm_LLVMField fp) =
 1442     (\p -> Perm_LLVMField $ fp { llvmFieldContents = p }) <$>
 1443     abstractVarsForLifetimes (Just $ llvmFieldLifetime fp) (llvmFieldContents fp)
 1444   -- FIXME: we can't yet abstract array permissions, because shapes in arrays
 1445   -- could be repeated multiple times and thus we would have to somehow abstract
 1446   -- over multiple copies of the same variable for that to work...
 1447   abstractVarsForLifetimes _ (Perm_LLVMBlock bp) =
 1448     (\sh -> Perm_LLVMBlock $ bp { llvmBlockShape = sh }) <$>
 1449     abstractVarsForLifetimesSh (llvmBlockRW bp) (llvmBlockLifetime bp)
 1450     (llvmBlockShape bp)
 1451   abstractVarsForLifetimes _ (Perm_Struct ps) =
 1452     -- NOTE: for struct perms we want to abstract any permission with any
 1453     -- non-always lifetime, so we set l to always
 1454     Perm_Struct <$>
 1455     traverseRAssign (abstractVarsForLifetimes (Just PExpr_Always)) ps
 1456   abstractVarsForLifetimes _ p = pure p

 1458 -- | Like 'abstractVarsForLifetimes' but for an LLVM shape inside a @memblock@
 1459 -- with the given modalities
 1460 abstractVarsForLifetimesSh :: (1 <= w, KnownNat w) => PermExpr RWModalityType ->
 1461                               PermExpr LifetimeType ->
 1462                               PermExpr (LLVMShapeType w) ->
 1463                               SomeMbWithPerms (PermExpr (LLVMShapeType w))
 1464 abstractVarsForLifetimesSh _ l (PExpr_FieldShape (LLVMFieldShape p)) =
 1465     PExpr_FieldShape <$> LLVMFieldShape <$> abstractVarsForLifetimes (Just l) p
 1466 abstractVarsForLifetimesSh rw l (PExpr_PtrShape maybe_rw (Just l') sh)
 1467   | l /= l'
 1468   , rw' <- maybe rw id maybe_rw
 1469   , Just len <- llvmShapeLength sh =
 1470     -- NOTE: abstracting a shape should return one with the same length
 1471     bindSomeMbWithPerms (abstractVarsForLifetimesSh rw' l' sh) $ \sh' ->
 1472     PExpr_FieldShape <$> LLVMFieldShape <$> ValPerm_Eq <$> PExpr_Var <$>
 1473     someMbWithPermsVar1 knownRepr (ValPerm_LLVMBlock $
 1474                                    LLVMBlockPerm rw' l' (bvInt 0) len sh')
 1475 abstractVarsForLifetimesSh rw l (PExpr_PtrShape maybe_rw maybe_l sh) =
 1476   let rw' = maybe rw id maybe_rw in
 1477   PExpr_PtrShape maybe_rw maybe_l <$> abstractVarsForLifetimesSh rw' l sh
 1478 abstractVarsForLifetimesSh rw l (PExpr_SeqShape sh1 sh2) =
 1479     PExpr_SeqShape <$> abstractVarsForLifetimesSh rw l sh1 <*>
 1480     abstractVarsForLifetimesSh rw l sh2
 1481 abstractVarsForLifetimesSh rw l (PExpr_ExShape mb_sh) =
 1482   mbSomeMbWithPerms knownRepr $ fmap (abstractVarsForLifetimesSh rw l) mb_sh
 1483 abstractVarsForLifetimesSh _ _ sh = pure sh

 1488                        Mb ctx' (Mb ctx a) ->
 1489                        MbSomeMbWithPerms ctx a

 1491 mbAbstractVarsForLifetimes :: Mb ctx (ValuePerms ps) ->
 1492                               MbSomeMbWithPerms ctx (ValuePerms ps)
 1493 mbAbstractVarsForLifetimes mb_ps
 1494   | [nuMP| SomeMbWithPerms ctx' mb_ctx_ps' mb_ps' |] <-
 1495       mbMatch (fmap (abstractVarsForLifetimes Nothing) mb_ps)
 1496   , ctx'_prxs <- cruCtxProxies $ mbLift ctx' =
 1497     MbSomeMbWithPerms (mbLift ctx') (mbSwap ctx'_prxs mb_ctx_ps')
 1498     (mbSwap ctx'_prxs mb_ps')

 1500 -- | For both the input and output permissions of a function permission, find
 1501 -- all permissions @p@ in with a lifetime that are contained inside a struct
 1502 -- permission or a field or block permission with a different lifetime, and
 1503 -- replace each such permission with an @eq(z)@ permission for a fresh ghost
 1504 -- variable @z@ that is itself assigned permissions @p@.
 1505 abstractFunVarsForLifetimes :: Some3FunPerm -> Some3FunPerm
 1506 abstractFunVarsForLifetimes (Some3FunPerm
 1507                          (FunPerm ghosts args gouts ret ps_in ps_out))
 1508   | MbSomeMbWithPerms ghosts' mb_gps' mb_ps_in' <-
 1509       mbAbstractVarsForLifetimes ps_in
 1510   , MbSomeMbWithPerms gouts' mb_gops' mb_ps_out' <-
 1511       mbAbstractVarsForLifetimes ps_out
 1512   , ghosts_prxs <- cruCtxProxies ghosts
 1513   , args_prxs <- cruCtxProxies args =
 1514     Some3FunPerm $ mbGhostsFunPerm ghosts'
 1515     (mbCombineAssoc ghosts' ghosts_prxs args_prxs mb_gps') $
 1516     flip fmap mb_ps_in' $ \ps_in' ->
 1517     mbGoutsFunPerm ghosts args gouts ret ps_in' gouts' mb_gops' mb_ps_out'

 1558 -- | Just like `parseFunPermFromRust`, but returns a `Some3FunPerm`
 1559 parseSome3FunPermFromRust :: (Fail.MonadFail m, 1 <= w, KnownNat w) =>
 1560                         PermEnv -> prx w ->
 1561                         String -> m Some3FunPerm
 1562 parseSome3FunPermFromRust env w str
 1563   | Just i <- findIndex (== '>') str
 1564   , (gen_str, fn_str) <- splitAt (i+1) str
 1565   , Right (Generics rust_ls1 rust_tvars wc span) <-
 1566       parse (inputStreamFromString gen_str)
 1567   , Right (BareFn _ abi rust_ls2 fn_tp _) <-
 1568       parse (inputStreamFromString fn_str) =
 1569     runLiftRustConvM (mkRustConvInfo env) $
 1570     rsConvertFun w abi (Generics (rust_ls1 ++ rust_ls2) rust_tvars wc span) fn_tp
 1571 
 1572   | Just i <- findIndex (== '>') str
 1573   , (gen_str, _) <- splitAt (i+1) str
 1574   , Left err <- parse @(Generics Span) (inputStreamFromString gen_str) =
 1575     fail ("Error parsing generics: " ++ show err)
 1576 
 1577   | Just i <- findIndex (== '>') str
 1578   , (_, fn_str) <- splitAt (i+1) str
 1579   , Left err <- parse @(Ty Span) (inputStreamFromString fn_str) =
 1580     fail ("Error parsing generics: " ++ show err)
 1581 parseSome3FunPermFromRust _ _ str =
 1582     fail ("Malformed Rust type: " ++ str)
