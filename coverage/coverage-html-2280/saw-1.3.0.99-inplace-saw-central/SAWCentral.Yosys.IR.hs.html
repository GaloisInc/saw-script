   39 -- | The direction of a module port.
   40 data Direction
   41   = DirectionInput
   42   | DirectionOutput
   43   | DirectionInout
   44   deriving (Show, Eq, Ord)
   45 instance Aeson.FromJSON Direction where
   46   parseJSON (Aeson.String "input") = pure DirectionInput
   47   parseJSON (Aeson.String "output") = pure DirectionOutput
   48   parseJSON (Aeson.String "inout") = pure DirectionInout
   49   parseJSON v = fail $ "Failed to parse direction: " <> show v
   50 
   51 -- | The value of a connection
   52 data Bitrep
   53   = BitrepZero -- ^ Constant zero bit
   54   | BitrepOne -- ^ Constant one bit
   55   | BitrepX -- ^ Undefined bit X
   56   | BitrepZ -- ^ Undefined bit Z
   57   | Bitrep Integer -- ^ The signal bit with the given index
   58   deriving (Show, Eq, Ord)
   59 instance Aeson.FromJSON Bitrep where
   60   parseJSON (Aeson.String "0") = pure BitrepZero
   61   parseJSON (Aeson.String "1") = pure BitrepOne
   62   parseJSON (Aeson.String "x") = pure BitrepX
   63   parseJSON (Aeson.String "z") = pure BitrepZ
   64   parseJSON vn@Aeson.Number{} = Bitrep <$> Aeson.parseJSON vn
   65   parseJSON v = fail $ "Failed to parse bits: " <> show v
   66 
   67 -- ^ A module input/output port.
   68 data Port = Port
   69   { _portDirection :: Direction
   70   , _portBits :: [Bitrep] -- ^ Which bit indices within the module are associated with the port
   71   , _portOffset :: Integer -- currently unused
   72   , _portUpto :: Bool -- currently unused
   73   } deriving (Show, Eq, Ord)
   74 makeLenses ''Port
   75 instance Aeson.FromJSON Port where
   76   parseJSON = Aeson.withObject "port" $ \o -> do
   77     _portDirection <- o Aeson..: "direction"
   78     _portBits <- o Aeson..: "bits"
   79     _portOffset <- o Aeson..:? "offset" >>= \case
   80       Just off -> pure off
   81       Nothing -> pure 0
   82     _portUpto <- o Aeson..:? "upto" >>= \case
   83       Just (Aeson.Number 1) -> pure True
   84       _ -> pure False
   85     pure Port{..}

   87 -- | Return 'True' iff a given cell type is a primitive type
   88 cellTypeIsPrimitive :: Text -> Bool
   89 cellTypeIsPrimitive cellType =
   90   case Text.uncons cellType of
   91     Just ('$', _) -> True
   92     _ -> False

  136 -- | Mapping from primitive cell types to textual representation
  137 primitiveCellTypeToText :: Map.Map CellType Text
  138 primitiveCellTypeToText =
  139   Map.fromList [(y, x) | (x, y) <- Map.toList textToPrimitiveCellType]

  141 -- | All supported cell types. All types are primitives except for
  142 -- 'CellTypeUserType' which represents user-defined submodules
  143 data CellType
  144   = CellTypeNot
  145   | CellTypePos
  146   | CellTypeNeg
  147   | CellTypeAnd
  148   | CellTypeOr
  149   | CellTypeXor
  150   | CellTypeXnor
  151   | CellTypeReduceAnd
  152   | CellTypeReduceOr
  153   | CellTypeReduceXor
  154   | CellTypeReduceXnor
  155   | CellTypeReduceBool
  156   | CellTypeShl
  157   | CellTypeShr
  158   | CellTypeSshl
  159   | CellTypeSshr
  160   | CellTypeShiftx
  161   | CellTypeLt
  162   | CellTypeLe
  163   | CellTypeGt
  164   | CellTypeGe
  165   | CellTypeEq
  166   | CellTypeNe
  167   | CellTypeEqx
  168   | CellTypeNex
  169   | CellTypeAdd
  170   | CellTypeSub
  171   | CellTypeMul
  172   | CellTypeDiv
  173   | CellTypeMod
  174   | CellTypeLogicNot
  175   | CellTypeLogicAnd
  176   | CellTypeLogicOr
  177   | CellTypeMux
  178   | CellTypePmux
  179   | CellTypeDff
  180   | CellTypeFf
  181   | CellTypeUnsupportedPrimitive Text
  182   | CellTypeUserType Text
  183   deriving (Eq, Ord)
  184 instance Aeson.FromJSON CellType where
  185   parseJSON (Aeson.String s) =
  186     case s of
  187       "$adff"        -> throw $ YosysErrorUnsupportedFF "$adff"
  188       "$sdff"        -> throw $ YosysErrorUnsupportedFF "$sdff"
  189       "$aldff"       -> throw $ YosysErrorUnsupportedFF "$aldff"
  190       "$dffsr"       -> throw $ YosysErrorUnsupportedFF "$dffsr"
  191       "$dffe"        -> throw $ YosysErrorUnsupportedFF "$dffe"
  192       "$adffe"       -> throw $ YosysErrorUnsupportedFF "$adffe"
  193       "$sdffe"       -> throw $ YosysErrorUnsupportedFF "$sdffe"
  194       "$sdffce"      -> throw $ YosysErrorUnsupportedFF "$sdffce"
  195       "$aldffe"      -> throw $ YosysErrorUnsupportedFF "$aldffe"
  196       "$dffsre"      -> throw $ YosysErrorUnsupportedFF "$dffsre"
  197       _ | cellTypeIsPrimitive s ->
  198           case Map.lookup s textToPrimitiveCellType of
  199             Just cellType -> pure cellType
  200             -- XXX We should probably log a warning when generating CellTypeUnsupportedPrimitive, 
  201             -- we can't do that here however.    
  202             Nothing -> pure $ CellTypeUnsupportedPrimitive s
  203         | otherwise -> pure $ CellTypeUserType s
  204   parseJSON v = fail $ "Failed to parse cell type: " <> show v
  205 instance Show CellType where
  206   show ct = Text.unpack $
  207     case ct of
  208       CellTypeUserType ut -> ut
  209       CellTypeUnsupportedPrimitive t -> t
  210       _ | Just t <- Map.lookup ct primitiveCellTypeToText -> t
  211         | otherwise -> panic "Show CellType" ["Unknown primitive cell type"]
  212 
  213 -- | Extract the name from a user-defined submodule 'CellType'
  214 asUserType :: CellType -> Text
  215 asUserType cellType =
  216   case cellType of
  217     CellTypeUserType t -> t
  218     CellTypeUnsupportedPrimitive t -> t
  219     _ ->
  220       panic "asUserType"
  221             [  "Expected a user defined type, but got a primitive type: "
  222             ++ show cellType ]

  224 -- | A cell within an HDL module.
  225 data Cell bs = Cell
  226   { _cellHideName :: Bool -- ^ Whether the cell's name is human-readable
  227   , _cellType :: CellType -- ^ The cell type
  228   , _cellParameters :: Map Text Text -- ^ Metadata parameters
  229   , _cellAttributes :: Aeson.Value -- currently unused
  230   , _cellPortDirections :: Map Text Direction -- ^ Direction for each cell connection
  231   , _cellConnections :: Map Text bs -- ^ Bitrep for each cell connection
  232   } deriving (Show, Eq, Ord, Functor)
  233 makeLenses ''Cell
  234 instance Aeson.FromJSON (Cell [Bitrep]) where
  235   parseJSON = Aeson.withObject "cell" $ \o -> do
  236     _cellHideName <- o Aeson..:? "hide_name" >>= \case
  237       Just (Aeson.Number 1) -> pure True
  238       _ -> pure False
  239     _cellType <- o Aeson..: "type"
  240     _cellParameters <- o Aeson..: "parameters"
  241     _cellAttributes <- o Aeson..: "attributes"
  242     _cellPortDirections <- o Aeson..: "port_directions"
  243     _cellConnections <- o Aeson..: "connections"
  244     pure Cell{..}

  260 -- | A collection of multiple HDL modules (possibly with dependencies on each other).
  261 data YosysIR = YosysIR
  262   { _yosysCreator :: Text
  263   , _yosysModules :: Map Text Module
  264   } deriving (Show, Eq, Ord)
  265 makeLenses ''YosysIR
  266 instance Aeson.FromJSON YosysIR where
  267   parseJSON = Aeson.withObject "yosys" $ \o -> do
  268     _yosysCreator <- o Aeson..: "creator"
  269     _yosysModules <- o Aeson..: "modules"
  270     pure YosysIR{..}

  272 -- | Read a collection of HDL modules from a file produced by Yosys' write_json command.
  273 loadYosysIR :: MonadIO m => FilePath -> m YosysIR
  274 loadYosysIR p = liftIO $ Aeson.eitherDecodeFileStrict p >>= \case
  275   Left err -> throw . YosysError $ Text.pack err
  276   Right ir -> pure ir

  278 -- | Return the patterns for all of the input ports of a module
  279 moduleInputPorts :: Module -> Map Text [Bitrep]
  280 moduleInputPorts m =
  281   Map.fromList
  282   . Maybe.mapMaybe
  283   ( \(nm, ip) ->
  284       if ip ^. portDirection == DirectionInput || ip ^. portDirection == DirectionInout
  285       then Just (nm, ip ^. portBits)
  286       else Nothing
  287   )
  288   . Map.assocs
  289   $ m ^. modulePorts
  290 
  291 -- | Return the patterns for all of the output ports of a module
  292 moduleOutputPorts :: Module -> Map Text [Bitrep]
  293 moduleOutputPorts m =
  294   Map.fromList
  295   . Maybe.mapMaybe
  296   ( \(nm, ip) ->
  297       if ip ^. portDirection == DirectionOutput || ip ^. portDirection == DirectionInout
  298       then Just (nm, ip ^. portBits)
  299       else Nothing
  300   )
  301   . Map.assocs
  302   $ m ^. modulePorts
  303 
  304 -- | Return the patterns for all of the input connections of a cell
  305 cellInputConnections :: Cell [b] -> Map Text [b]
  306 cellInputConnections c = Map.intersection (c ^. cellConnections) inp
  307   where
  308     inp = Map.filter (\d -> d == DirectionInput || d == DirectionInout) $ c ^. cellPortDirections
  309 
  310 -- | Return the patterns for all of the output connections of a cell
  311 cellOutputConnections :: Ord b => Cell [b] -> Map Text [b]
  312 cellOutputConnections c = Map.intersection (c ^. cellConnections) out
  313   where
  314     out = Map.filter (\d -> d == DirectionOutput || d == DirectionInout) $ c ^. cellPortDirections
