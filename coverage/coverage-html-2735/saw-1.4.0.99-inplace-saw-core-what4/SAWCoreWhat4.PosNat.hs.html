   50 -- We include the KnownNat constraint here so that we can avoid using
   51 -- withKnownNat outside this module.  The downside is that we have two
   52 -- different runtime representations of these positive nats --- the
   53 -- one stored in the KnownNat, and the one stored in the NatRepr.
   54 
   55 -- | Runtime representation of a non-zero natural number
   56 data PosNat (n :: Nat) where
   57   PosNat :: (1 <= n, KnownNat n) => NatRepr n -> PosNat n
   58 
   59 -- | Check whether an integer is a positive nat
   60 somePosNat :: Integral a => a -> Maybe (Some PosNat)
   61 somePosNat n
   62   | Just (Some nr) <- someNat n,
   63     Just LeqProof  <- testLeq (knownNat @1) nr
   64   = withKnownNat nr $ Just (Some (PosNat nr))
   65   | otherwise
   66   = Nothing
   67

   74 -- | Add two numbers together and return a proof that their
   75 -- result is positive.
   76 addPosNat :: forall w1 w2.
   77   (1 <= w1, 1 <= w2, KnownNat w1, KnownNat w2) => PosNat (w1 + w2)
   78 addPosNat =
   79   let w1  = knownNat @w1
   80       w2  = knownNat @w2
   81       sm  = addNat w1 w2 in
   82   case leqAddPos w1 w2 of
   83     LeqProof -> withKnownNat sm $ PosNat sm
   84
