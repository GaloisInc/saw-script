   33 data Namespace =
   34   NamespaceCore | NamespaceCryptol | NamespaceFresh | NamespaceYosys | NamespaceLLVM
   35   deriving (Eq, Ord, Enum, Bounded)
   36 
   37 instance Hashable Namespace where
   38   hashWithSalt s ns = hashWithSalt s (fromEnum ns)
   39 
   40 renderNamespace :: Namespace -> Text
   41 renderNamespace = \case
   42   NamespaceCore -> "core"
   43   NamespaceCryptol -> "cryptol"
   44   NamespaceFresh -> "fresh"
   45   NamespaceYosys -> "yosys"
   46   NamespaceLLVM -> "llvm"
   47 
   48 instance Show Namespace where
   49   show ns = Text.unpack $ renderNamespace ns
   50 
   51 namespaceMap :: Map Text Namespace
   52 namespaceMap = Map.fromList $ map (\ns -> (renderNamespace ns, ns)) [minBound..maxBound]
   53 
   54 readNamespace :: MonadFail m => Text -> m Namespace
   55 readNamespace txt = case Map.lookup txt namespaceMap of
   56   Just ns -> pure ns
   57   Nothing -> fail $ "readNamespace: namespace not found: " ++ Text.unpack txt
   58 
   59 parsePath :: MonadFail m => Text -> m ([Text], Text)
   60 parsePath txt
   61   | not (Text.null txt)
   62   , ps <- map percentDecode $ Text.splitOn "/" txt
   63   = case ps of
   64       -- allow for a leading '/'
   65       (p:ps') | Text.null p -> mkPath ps'
   66       _ -> mkPath ps
   67 parsePath txt =
   68   fail $ "parsePath: invalid path: " ++ (Text.unpack txt)
   69 
   70 data URI = URI 
   71   { uriNamespace :: Namespace
   72   , uriBaseName :: Text
   73   , uriPath :: [Text]
   74   , uriIndex :: Maybe Int
   75   }
   76   deriving (Eq, Ord)
   77 
   78 instance Hashable URI where
   79   hashWithSalt s (URI a b c d) = s
   80     `hashWithSalt` a 
   81     `hashWithSalt` b
   82     `hashWithSalt` c
   83     `hashWithSalt` d
   84 
   85 percentEncode :: Text -> Text
   86 percentEncode txt = Text.replace "/" "%2F" $ Text.replace "%" "%25" txt
   87 
   88 percentDecode :: Text -> Text
   89 percentDecode txt = Text.replace "%25" "%" $ Text.replace "%2F" "/" txt
   90 
   91 mkPath :: (MonadFail m) => [Text] -> m ([Text], Text)
   92 mkPath ps = case (List.any Text.null ps,ps) of
   93   (False, _:_) -> return (List.init ps,List.last ps)
   94   _ -> fail $ "mkPath: invalid path: " ++ show ps
   95 
   96 mkURI :: (MonadFail m, Foldable t) => Namespace -> t Text -> Maybe Int -> m URI
   97 mkURI ns ps idx = do
   98   (ps', nm) <- mkPath (Foldable.toList ps)
   99   return $ URI ns nm ps' idx
  100 
  101 nonEmpty :: MonadFail m => String -> Text -> m ()
  102 nonEmpty msg txt = case Text.null txt of
  103   True -> fail msg
  104   False -> return ()
  105 
  106 splitM :: MonadFail m => Bool -> Char -> Text -> m (Text,Text)
  107 splitM fwd c txt = do
  108   nonEmpty err txt
  109   let txt' = if fwd then txt else Text.reverse txt
  110   let (lhs,rhs) = Text.break (==c) txt'
  111   nonEmpty err lhs
  112   case Text.uncons rhs of
  113     Nothing -> fail err
  114     Just(_,rhs') -> case fwd of
  115       True -> return (lhs, rhs')
  116       False -> return (Text.reverse rhs', Text.reverse lhs)
  117   where
  118     err = "splitM: failed to split on char '" ++ [c] ++ "' in '"
  119       ++ Text.unpack txt
  120 
  121 parseURI :: MonadFail m => Text -> m URI
  122 parseURI txt0 = do
  123   (ns, txt1) <- splitM True ':' txt0
  124   ns' <- readNamespace ns
  125   let (mi, txt3) = case splitM False '#' txt1 of
  126         Just (txt2,si)
  127           | Right (i,s) <- Text.decimal si
  128           , Text.null s
  129          -> (Just i, txt2)
  130         _ -> (Nothing, txt1)
  131   (ps,nm) <- parsePath txt3
  132   return $ URI ns' nm ps mi
  133 
  134 renderURI :: URI -> Text
  135 renderURI uri =
  136      renderNamespace (uriNamespace uri)
  137   <> ":" <> pathBody <> suffix
  138   where
  139     suffix :: Text
  140     suffix = case uriIndex uri of
  141       Nothing -> Text.empty
  142       Just i -> "#" <> Text.pack (show i)
  143 
  144     pathBody :: Text
  145     pathBody = case uriPath uri of
  146       [] -> percentEncode $ uriBaseName uri
  147       ps -> "/" <> Text.intercalate "/" (map percentEncode (ps ++ [uriBaseName uri]))
  148 
  149 instance Show URI where
  150   show uri = Text.unpack (renderURI uri)
