   57 -- Laziness is important here, as we will often create and partially
   58 -- traverse patterns for very large terms.
   59 data Pat = Atom Text | Var | App Pat Pat
   60     deriving Eq
   61 
   62 instance Show Pat where
   63   showsPrec _ (Atom s) = shows s
   64   showsPrec _ Var = showString "_"
   65   showsPrec p (App x y) =
   66       showParen (p > 5) (showsPrec 5 x . showString " " . showsPrec 6 y)
   67 
   68 isVarApp :: Pat -> Bool
   69 isVarApp t = case t of
   70   Atom _   -> False
   71   Var      -> True
   72   App t' _ -> isVarApp t'
   73 
   74 -- Start
   75 
   76 data Key = CombK | VarK | AtomK Text
   77 
   78 {-Keys are preorder lists of symbols -- Combinations, Vars, Atoms.
   79   Any term whose head is a Var is regarded entirely as a Var.
   80   Abstractions are also regarded as Vars;  this covers eta-conversion
   81     and "near" eta-conversions such as %x.?P(?f(x)).
   82 -}
   83 
   84 add_key_of_terms :: Pat -> [Key] -> [Key]
   85 add_key_of_terms t cs
   86   | isVarApp t = VarK : cs
   87   | otherwise  = add_key_of_terms' t cs
   88 
   89 -- | Precondition: not (isVarApp t).
   90 add_key_of_terms' :: Pat -> [Key] -> [Key]
   91 add_key_of_terms' (App f t) cs = CombK : add_key_of_terms' f (add_key_of_terms t cs)
   92 add_key_of_terms' (Atom c)  cs = AtomK c : cs
   93 add_key_of_terms' Var       _  = error "impossible"
   94 
   95 {-convert a term to a list of keys-}
   96 key_of_term :: Pat -> [Key]
   97 key_of_term t = add_key_of_terms t []
   98 {- ^ Required property: @depth (key_of_term t) = 1@
   99 depth :: [Key] -> Int
  100 depth [] = 0
  101 depth (CombK   : keys) = depth keys - 1
  102 depth (VarK    : keys) = depth keys + 1
  103 depth (AtomK _ : keys) = depth keys + 1
  104 -}
  105 
  106 {-Trees indexed by key lists: each arc is labelled by a key.
  107   Each node contains a list of items, and arcs to children.
  108   The empty key addresses the entire net.
  109   Lookup functions preserve order in items stored at same level.
  110 -}
  111 
  112 data Net a
  113   = Leaf [a]
  114   | Net { comb :: Net a, var :: Net a, atoms :: Map Text (Net a) }
  115   deriving Show
  116 
  117 {-
  118 Invariant: A well-formed term net should satisfy @valid 1@.
  119 Every sub-net should satisfy @valid n@ for some non-negative @n@.
  120 
  121 valid :: Int -> Net a -> Bool
  122 valid n (Leaf xs) = null xs || n == 0
  123 valid n (Net {comb, var, atoms}) =
  124   n > 0 && valid (n+1) comb && valid (n-1) var && all (valid (n-1)) (elems atoms)
  125 -}
  126 
  127 empty :: Net a
  128 empty = Leaf []
  129 
  130 is_empty :: Net a -> Bool
  131 is_empty (Leaf []) = True
  132 is_empty _ = False
  133 
  134 emptynet :: Net a
  135 emptynet = Net { comb = empty, var = empty, atoms = Map.empty }
  136 
  137 {-** Insertion into a discrimination net **-}
  138 
  139 {-Adds item x to the list at the node addressed by the keys.
  140   Creates node if not already present.
  141   The empty list of keys generates a Leaf node, others a Net node.
  142 -}
  143 insert :: forall a. (Eq a) => ([Key], a) -> Net a -> Net a
  144 insert (keys0, x) net = ins1 keys0 net
  145   where
  146     ins1 :: [Key] -> Net a -> Net a
  147     ins1 [] (Leaf xs)
  148       | x `elem` xs = Leaf xs
  149       | otherwise   = Leaf (x : xs)
  150     ins1 keys (Leaf []) = ins1 keys emptynet
  151     ins1 (CombK : keys) (Net {comb, var, atoms}) =
  152       Net {comb = ins1 keys comb, var = var, atoms = atoms}
  153     ins1 (VarK : keys) (Net {comb, var, atoms}) =
  154       Net {comb = comb, var = ins1 keys var, atoms = atoms}
  155     ins1 (AtomK a : keys) (Net {comb, var, atoms}) =
  156       let atoms' = Map.alter (Just . ins1 keys . fromMaybe empty) a atoms
  157       in Net {comb = comb, var = var, atoms = atoms'}
  158     ins1 [] (Net {}) = error "impossible"
  159     ins1 (_ : _) (Leaf (_ : _)) = error "impossible"
  160 
  161 insert_term :: (Eq a) => (Pat, a) -> Net a -> Net a
  162 insert_term (pat, x) = insert (key_of_term pat, x)
  163 
  164 {-** Deletion from a discrimination net **-}
  165 
  166 {-Create a new Net node if it would be nonempty-}
  167 newnet :: Net a -> Net a -> Map Text (Net a) -> Net a
  168 newnet comb var atoms =
  169   if is_empty comb && is_empty var && Map.null atoms
  170   then empty else Net { comb = comb, var = var, atoms = atoms }
  171 
  172 {-Deletes item x from the list at the node addressed by the keys.
  173   Returns Nothing if absent.  Collapses the net if possible. -}
  174 delete :: (Eq a) => ([Key], a) -> Net a -> Net a
  175 delete (keys0, x) net0 = del1 keys0 net0
  176   where
  177     -- | Invariant: @del1 keys net@ requires @valid (depth keys) net@.
  178     del1 [] (Leaf xs) = Leaf (List.delete x xs)
  179     del1 _ (Leaf []) = Leaf []
  180     del1 (CombK : keys) (Net {comb, var, atoms}) =
  181       newnet (del1 keys comb) var atoms
  182     del1 (VarK : keys) (Net {comb, var, atoms}) =
  183       newnet comb (del1 keys var) atoms
  184     del1 (AtomK a : keys) (Net {comb, var, atoms}) =
  185       let nonempty (Leaf []) = Nothing
  186           nonempty net = Just net
  187           atoms' = Map.update (nonempty . del1 keys) a atoms
  188       in newnet comb var atoms'
  189     del1 [] (Net {}) = error "impossible"
  190     del1 (_ : _) (Leaf (_ : _)) = error "impossible"
  191 
  192 delete_term :: (Eq a) => (Pat, a) -> Net a -> Net a
  193 delete_term (pat, x) = delete (key_of_term pat, x)

  195 {-** Retrieval functions for discrimination nets **-}
  196 
  197 {-Return the list of items at the given node, [] if no such node-}
  198 -- | Invariant: @lookup net keys@ requires @valid (depth keys) net@.
  199 lookup :: Net a -> [Key] -> [a]
  200 lookup (Leaf xs) [] = xs
  201 lookup (Leaf _) (_ : _) = []  {-non-empty keys and empty net-}
  202 lookup (Net {comb}) (CombK : keys) = lookup comb keys
  203 lookup (Net {var}) (VarK : keys) = lookup var keys
  204 lookup (Net {atoms}) (AtomK a : keys) =
  205   case Map.lookup a atoms of
  206     Just net -> lookup net keys
  207     Nothing -> []
  208 lookup (Net {}) [] = error "impossible"
  209 
  210 {-Skipping a term in a net.  Recursively skip 2 levels if a combination-}
  211 net_skip :: Net a -> [Net a] -> [Net a]
  212 net_skip (Leaf _) nets = nets
  213 net_skip (Net {comb, var, atoms}) nets =
  214   foldr net_skip (Map.foldr (:) (var : nets) atoms) (net_skip comb [])
  215 
  216 {-* Matching and Unification *-}
  217 
  218 {-conses the linked net, if present, to nets-}
  219 look1 :: (Map Text (Net a), Text) -> [Net a] -> [Net a]
  220 look1 (atoms, a) nets =
  221   case Map.lookup a atoms of
  222     Just net -> net : nets
  223     Nothing -> nets
  224 
  225 {-Return the nodes accessible from the term (cons them before nets)
  226   "unif" signifies retrieval for unification rather than matching.
  227   Var in net matches any term.
  228   Abs or Var in object: if "unif", regarded as wildcard,
  229                                    else matches only a variable in net.
  230 -}
  231 matching :: Bool -> Pat -> Net a -> [Net a] -> [Net a]
  232 matching unif = match
  233   where
  234     match :: Pat -> Net a -> [Net a] -> [Net a]
  235     match t net nets =
  236       case net of
  237         Leaf _ -> nets
  238         Net {var} ->
  239           case t of
  240             Var -> if unif then net_skip net nets else var : nets {-only matches Var in net-}
  241             _   -> rands t net (var : nets)  {-var could match also-}
  242     rands :: Pat -> Net a -> [Net a] -> [Net a]
  243     rands _ (Leaf _) nets = nets
  244     rands t (Net {comb, atoms}) nets =
  245       case t of
  246         Atom c    -> look1 (atoms, c) nets
  247         Var       -> nets
  248         App t1 t2 -> foldr (match t2) nets (rands t1 comb [])
  249 
  250 {- Invariant: Each list entry must be a Leaf. -}
  251 extract_leaves :: [Net a] -> [a]
  252 extract_leaves = concatMap $ \case
  253   Leaf xs -> xs
  254   Net{}   -> error "extract_leaves: Unexpected Net node"
  255 
  256 {-return items whose key could match pat, WHICH MUST BE BETA-ETA NORMAL-}
  257 match_term :: Net a -> Pat -> [a]
  258 match_term net pat = extract_leaves (matching False pat net [])
  259 
  260 {-return items whose key could unify with pat-}
  261 unify_term :: Net a -> Pat -> [a]
  262 unify_term net pat = extract_leaves (matching True pat net [])

  266 {-* operations on nets *-}
  267 
  268 {-subtraction: collect entries of second net that are NOT present in first net-}
  269 fun subtract eq net1 net2 =
  270   let
  271     fun subtr (Net _) (Leaf ys) = append ys
  272       | subtr (Leaf xs) (Leaf ys) =
  273           fold_rev (fn y => if member eq xs y then I else cons y) ys
  274       | subtr (Leaf _) (net as Net _) = subtr emptynet net
  275       | subtr (Net {comb = comb1, var = var1, atoms = atoms1})
  276             (Net {comb = comb2, var = var2, atoms = atoms2}) =
  277           subtr comb1 comb2
  278           #> subtr var1 var2
  279           #> Symtab.fold (fn (a, net) =>
  280             subtr (the_default emptynet (Symtab.lookup atoms1 a)) net) atoms2
  281   in subtr net1 net2 [] end;
  282 
  283 fun entries net = subtract (K false) empty net;

  287 {- merge -}

  289 cons_fst :: a -> ([a], b) -> ([a], b)
  290 cons_fst x (xs, y) = (x : xs, y)
  291 
  292 dest :: Net a -> [([Key], a)]
  293 dest (Leaf xs) = map ((,) []) xs
  294 dest (Net {comb, var, atoms}) =
  295   map (cons_fst CombK) (dest comb) ++
  296   map (cons_fst VarK) (dest var) ++
  297   concatMap (\(a, net) -> map (cons_fst (AtomK a)) (dest net)) (Map.assocs atoms)
  298 
  299 merge :: Eq a => Net a -> Net a -> Net a
  300 merge net1 net2 = foldl (flip insert) net1 (dest net2)
  301 
  302 content :: Net a -> [a]
  303 content (Leaf xs) = xs
  304 content (Net {comb, var, atoms}) =
  305   content comb ++
  306   content var ++
  307   concatMap content (Map.elems atoms)
