   81 -- | Recognizer for 'TNatLit' constructor.
   82 -- Returns constructor argument if matched.
   83 tokenNat :: Token -> Maybe Natural
   84 tokenNat (TNatLit n) = Just n
   85 tokenNat _           = Nothing

   87 -- | Recognizer for 'TIdent' constructor.
