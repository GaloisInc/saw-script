   31 -- | Test if a 'Term' is an application of a specific datatype with the
   32 -- supplied context of parameters and indices
   33 ctxAsDataTypeApp :: Name -> [param] ->
   34                     [index] -> Term ->
   35                     Maybe ([Term], [Term])
   36 ctxAsDataTypeApp d params ixs t =
   37   do let (f, args) = asApplyAll t
   38      d' <- asConstant f
   39      guard (d == d')
   40      guard (length args == length params + length ixs)
   41      let (params', ixs') = splitAt (length params) args
   42      pure (params', ixs')
   43 
   44 -- | Test whether a specific datatype occurs in a term.
   45 usesDataType :: Name -> Term -> Bool
   46 usesDataType d t =
   47   case unwrapTermF t of
   48     Constant d'
   49       | d' == d -> True
   50     FTermF (RecursorType d' _ _ _)
   51       | d' == d -> True
   52     FTermF (Recursor (recursorDataType -> d'))
   53       | d' == d -> True
   54     tf -> any (usesDataType d) tf
   55 
   56 -- | Check that a type is a valid application of datatype @d@ for use in
   57 -- specific ways in the type of a constructor for @d@. This requires that this
   58 -- application of @d@ be of the form
   59 --
   60 -- > d p1 .. pn x1 .. xm
   61 --
   62 -- where the @pi@ are the distinct bound variables bound in the @params@
   63 -- context, given as argument, and that the @xj@ have no occurrences of @d@. If
   64 -- the given type is of this form, return the @xj@.
   65 asCtorDTApp :: Name -> [ExtCns Term] ->
   66                [index] ->
   67                Term ->
   68                Maybe [Term]
   69 asCtorDTApp d params dt_ixs (ctxAsDataTypeApp d params dt_ixs ->
   70                                        Just (param_vars, ixs))
   71   | isVarList params param_vars && not (any (usesDataType d) ixs)
   72   = Just ixs
   73   where
   74     -- Check that the given list of terms is a list of named
   75     -- variables, one for each parameter
   76     isVarList :: [ExtCns Term] -> [Term] -> Bool
   77     isVarList _ [] = True
   78     isVarList (p : ps) ((asExtCns -> Just ec) : ts) =
   79       ec == p && isVarList ps ts
   80     isVarList _ _ = False
   81 asCtorDTApp _ _ _ _ = Nothing

   84 -- | Check that an argument for a constructor has one of the allowed forms
   85 asCtorArg ::
   86   SharedContext ->
   87   Name ->
   88   [ExtCns Term] ->
   89   [index] ->
   90   Term ->
   91   IO (Maybe CtorArg)
   92 asCtorArg _ d _ _ tp
   93   | not (usesDataType d tp)
   94   = pure $ Just (ConstArg tp)
   95 asCtorArg sc d params dt_ixs tp =
   96   do (zs, ret) <- scAsPiList sc tp
   97      case asCtorDTApp d params dt_ixs ret of
   98        Just ixs
   99          | not (any (usesDataType d . ecType) zs) ->
  100            pure $ Just (RecursiveArg zs ixs)
  101        _ ->
  102          pure Nothing
  103 
  104 -- | Check that a constructor type is a pi-abstraction that takes as input an
  105 -- argument of one of the allowed forms described by 'CtorArg'
  106 asPiCtorArg ::
  107   SharedContext ->
  108   Name ->
  109   [ExtCns Term] ->
  110   [index] ->
  111   Term ->
  112   IO (Maybe (Name, CtorArg, Term))
  113 asPiCtorArg sc d params dt_ixs t =
  114   scAsPi sc t >>= \case
  115     Nothing -> pure Nothing
  116     Just (ec, rest) ->
  117       asCtorArg sc d params dt_ixs (ecType ec) >>= \case
  118         Nothing -> pure Nothing
  119         Just arg -> pure $ Just (Name (ecVarIndex ec) (ecName ec), arg, rest)
  120 
  121 -- | Helper function for 'mkCtorArgStruct'
  122 mkCtorArgsIxs ::
  123   SharedContext ->
  124   Name ->
  125   [ExtCns Term] ->
  126   [index] ->
  127   Term ->
  128   IO (Maybe ([(Name, CtorArg)], [Term]))
  129 mkCtorArgsIxs _sc d params dt_ixs (asCtorDTApp d params dt_ixs -> Just ixs) =
  130   pure $ Just ([], ixs)
  131 mkCtorArgsIxs sc d params dt_ixs ty =
  132   asPiCtorArg sc d params dt_ixs ty >>= \case
  133     Nothing -> pure Nothing
  134     Just (x, arg, rest) ->
  135       mkCtorArgsIxs sc d params dt_ixs rest >>= \case
  136         Nothing -> pure Nothing
  137         Just (args, ixs) -> pure $ Just ((x, arg) : args, ixs)
  138 
  139 -- | Take in a datatype and bindings lists for its parameters and indices, and
  140 -- also a prospective type of a constructor for that datatype, where the
  141 -- constructor type is allowed to have the parameters but not the indices free.
  142 -- Test that the constructor type is an allowed type for a constructor of this
  143 -- datatype, and, if so, build a 'CtorArgStruct' for it.
  144 mkCtorArgStruct ::
  145   SharedContext ->
  146   Name ->
  147   [ExtCns Term] ->
  148   [ExtCns Term] ->
  149   Term ->
  150   IO (Maybe CtorArgStruct)
  151 mkCtorArgStruct sc d params dt_ixs ctor_tp =
  152   mkCtorArgsIxs sc d params dt_ixs ctor_tp >>= \case
  153     Nothing -> pure Nothing
  154     Just (args, ctor_ixs) ->
  155       pure $ Just (CtorArgStruct params args ctor_ixs dt_ixs)
