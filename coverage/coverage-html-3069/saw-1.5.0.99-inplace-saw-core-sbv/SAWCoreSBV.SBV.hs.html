  252 -- | Flatten an SValue to a sequence of components, each of which is
  253 -- either a symbolic word or a symbolic boolean. If the SValue
  254 -- contains any values built from data constructors, then return them
  255 -- encoded as a String.
  256 flattenSValue :: String -> SValue -> IO ([SVal], String)
  257 flattenSValue nm v = do
  258   mw <- toMaybeWord v
  259   case mw of
  260     Just w -> return ([w], "")
  261     Nothing ->
  262       case v of
  263         VUnit                     -> return ([], "")
  264         VPair x y                 -> do (xs, sx) <- flattenSValue nm =<< force x
  265                                         (ys, sy) <- flattenSValue nm =<< force y
  266                                         return (xs ++ ys, sx ++ sy)
  267         VEmptyRecord              -> pure ([], "")
  268         VRecordValue _ x y        -> do (xs, sx) <- flattenSValue nm =<< force x
  269                                         (ys, sy) <- flattenSValue nm y
  270                                         pure (xs ++ ys, sx ++ sy)
  271         VVector (V.toList -> ts)  -> do (xss, ss) <- unzip <$> traverse (force >=> flattenSValue nm) ts
  272                                         return (concat xss, concat ss)
  273         VBool sb                  -> return ([sb], "")
  274         VInt si                   -> return ([si], "")
  275         VIntMod 0 si              -> return ([si], "")
  276         VIntMod n si              -> return ([svRem si (svInteger KUnbounded (toInteger n))], "")
  277         VWord sw                  -> return (if intSizeOf sw > 0 then [sw] else [], "")
  278         VCtorApp i _ ps ts        -> do (xss, ss) <- unzip <$> traverse (force >=> flattenSValue nm) (ps++ts)
  279                                         return (concat xss, "_" ++ (Text.unpack (toShortName (nameInfo i))) ++ concat ss)
  280         VNat n                    -> return ([], "_" ++ show n)
  281         TValue (suffixTValue -> Just s)
  282                                   -> return ([], s)
  283         VFun {} -> fail $ "Cannot create uninterpreted higher-order function " ++ show nm
  284         _ -> fail $ "Cannot create uninterpreted function " ++ show nm ++ " with argument " ++ show v

  304 -- | Lifts a strict mux operation to a lazy mux
  305 lazyMux :: (SBool -> a -> a -> IO a) -> (SBool -> IO a -> IO a -> IO a)
  306 lazyMux muxFn c tm fm =
  307   case svAsBool c of
  308     Just True  -> tm
  309     Just False -> fm
  310     Nothing    -> do
  311       t <- tm
  312       f <- fm
  313       muxFn c t f

  315 -- @selectV merger maxValue valueFn vx@ treats @vx@ as an index, represented
  316 -- as a big-endian list of bits. It does a binary lookup, using @merger@ as an
  317 -- if-then-else operator. If the index is greater than @maxValue@, then it
  318 -- returns @valueFn maxValue@.
  319 selectV :: (SBool -> b -> b -> b) -> Natural -> (Natural -> b) -> SWord -> b
  320 selectV merger maxValue valueFn vx =
  321   case svAsInteger vx of
  322     Just i
  323       | i >= 0    -> valueFn (fromInteger i)
  324       | otherwise -> panic "selectV" ["Expected nonnegative integer; found " <> Text.pack (show i)]
  325     Nothing -> impl (intSizeOf vx) 0
  326   where
  327     impl _ x | x > maxValue || x < 0 = valueFn maxValue
  328     impl 0 y = valueFn y
  329     impl i y =
  330       -- NB: `i` counts down in each iteration, so we use svTestBit (a
  331       -- little-endian indexing function) to ensure that the bits are processed
  332       -- in big-endian order. Alternatively, we could have `i` count up and use
  333       -- svAt (a big-endian indexing function), but we use svTestBit as it is
  334       -- slightly cheaper to compute.
  335       merger (svTestBit vx j) (impl j (y `setBit` j)) (impl j y) where j = i - 1
  336 
  337 -- Big-endian version of svTestBit
  338 svAt :: SWord -> Int -> SBool
  339 svAt x i = svTestBit x (intSizeOf x - 1 - i)
  340 
  341 svUnpack :: SWord -> IO (Vector SBool)
  342 svUnpack x = return (V.generate (intSizeOf x) (svAt x))

  344 asWordList :: [SValue] -> Maybe [SWord]

  358 -- | op : (n : Nat) -> Vec n Bool -> Nat -> Vec n Bool
  359 bvShiftOp :: (SWord -> SWord -> SWord) -> (SWord -> Int -> SWord) -> SPrim
  360 bvShiftOp bvOp natOp =
  361   Prims.constFun $
  362   wordFun $ \x ->
  363   Prims.strictFun $ \y ->
  364   Prims.Prim $
  365     case y of
  366       VNat i | j < toInteger (maxBound :: Int) -> return (vWord (natOp x (fromInteger j)))
  367         where j = toInteger i `min` toInteger (intSizeOf x)
  368       VBVToNat _ v -> fmap (vWord . bvOp x) (toWord v)
  369       _        -> error $ unwords ["SAWCoreSBV.SBV.bvShiftOp", show y]
  370 
  371 -- bvShl : (w : Nat) -> Vec w Bool -> Nat -> Vec w Bool;
  372 bvShLOp :: SPrim
  373 bvShLOp = bvShiftOp svShiftLeft svShl
  374 
  375 -- bvShR : (w : Nat) -> Vec w Bool -> Nat -> Vec w Bool;
  376 bvShROp :: SPrim
  377 bvShROp = bvShiftOp svShiftRight svShr
  378 
  379 -- bvSShR : (w : Nat) -> Vec w Bool -> Nat -> Vec w Bool;
  380 bvSShROp :: SPrim
  381 bvSShROp = bvShiftOp bvOp natOp
  382   where
  383     bvOp w x = svUnsign (svShiftRight (svSign w) x)
  384     natOp w i = svUnsign (svShr (svSign w) i)

  389 -- primitive intToNat : Integer -> Nat;
  390 -- intToNat x == max 0 x
  391 intToNatOp :: SPrim
  392 intToNatOp =
  393   Prims.intFun $ \i ->
  394   Prims.PrimValue $
  395     case svAsInteger i of
  396       Just i'
  397         | 0 <= i'   -> VNat (fromInteger i')
  398         | otherwise -> VNat 0
  399       Nothing ->
  400         let z  = svInteger KUnbounded 0
  401             i' = svIte (svLessThan i z) z i
  402          in VIntToNat (VInt i')
  403 
  404 -- primitive natToInt :: Nat -> Integer;
  405 natToIntOp :: SPrim
  406 natToIntOp =
  407   Prims.natFun $ \n ->
  408   Prims.PrimValue $

  441 ------------------------------------------------------------
  442 -- Rotations and shifts
  443 
  444 svRol' :: SWord -> Integer -> SWord
  445 svRol' x i = svRol x (fromInteger (i `mod` toInteger (intSizeOf x)))
  446 
  447 svRor' :: SWord -> Integer -> SWord
  448 svRor' x i = svRor x (fromInteger (i `mod` toInteger (intSizeOf x)))
  449 
  450 svShl' :: SBool -> SWord -> Integer -> SWord
  451 svShl' b x i = svIte b (svNot (svShl (svNot x) j)) (svShl x j)
  452   where j = fromInteger (i `min` toInteger (intSizeOf x))
  453 
  454 svShr' :: SBool -> SWord -> Integer -> SWord
  455 svShr' b x i = svIte b (svNot (svShr (svNot x) j)) (svShr x j)
  456   where j = fromInteger (i `min` toInteger (intSizeOf x))
  457 
  458 svShiftL :: SBool -> SWord -> SWord -> SWord
  459 svShiftL b x i = svIte b (svNot (svShiftLeft (svNot x) i)) (svShiftLeft x i)
  460 
  461 svShiftR :: SBool -> SWord -> SWord -> SWord
  462 svShiftR b x i = svIte b (svNot (svShiftRight (svNot x) i)) (svShiftRight x i)

  492   Prims.natFun $ \n ->

  494   Prims.intModFun $ \y ->

  498 intModUnOp :: (SInteger -> SInteger) -> SPrim

  500   Prims.natFun $ \n ->
  501   Prims.intModFun $ \x ->
  502   Prims.PrimValue $
  503     VIntMod n (normalizeIntMod n (f x))

  505 normalizeIntMod :: Natural -> SInteger -> SInteger
  506 normalizeIntMod n x =
  507   case svAsInteger x of
  508     Nothing -> x
  509     Just i -> literalSInteger (i `mod` toInteger n)

  522 -- streamGet :: (a :: sort 0) -> Stream a -> Nat -> a;
  523 streamGetOp :: SPrim
  524 streamGetOp =
  525   Prims.tvalFun   $ \_tp ->
  526   Prims.strictFun $ \xs ->
  527   Prims.strictFun $ \ix ->
  528   Prims.Prim $ streamGet xs ix

  530 streamGet :: SValue -> SValue -> IO SValue
  531 streamGet xs ix =
  532   case ix of
  533     VNat n -> lookupSStream xs n
  534     VBVToNat _ w ->
  535       do ilv <- toWord w
  536          selectV (lazyMux muxBVal) ((2 ^ intSizeOf ilv) - 1) (lookupSStream xs) ilv
  537     v -> panic "streamGetOp" ["Expected Nat value; got " <> Text.pack (show v)]
  538 
  539 lookupSStream :: SValue -> Natural -> IO SValue
  540 lookupSStream (VExtra s) n = lookupSbvExtra s n
  541 lookupSStream _ _ = fail "expected Stream"
  542 
  543 lookupSbvExtra :: SbvExtra -> Natural -> IO SValue
  544 lookupSbvExtra (SStream f r) n =
  545   do m <- readIORef r
  546      case Map.lookup n m of
  547        Just v  -> return v
  548        Nothing -> do v <- f n
  549                      writeIORef r (Map.insert n v m)
  550                      return v

  552 -- Stream#rec :
  553 --   (a : sort 0) -> (p : Stream a -> sort 0) ->
  554 --   ((f : Nat -> a) -> p (MkStream a f)) -> (str : Stream a) -> p str
  555 streamRecOp :: SValue
  556 streamRecOp =
  557   VFun $ \_a -> pure $
  558   VFun $ \_p -> pure $
  559   vStrictFun $ \f1 -> pure $
  560   vStrictFun $ \xs ->
  561   do let f = vStrictFun $ \ix -> streamGet xs ix
  562      apply f1 (ready f)

  567 svPopcount :: SWord -> SWord
  568 svPopcount xs = if w == 0 then zero else foldr1 svPlus [ svIte b one zero | b <- bits ]
  569  where
  570  bits = svBlastLE xs
  571  w    = length bits
  572  one  = literalSWord w 1
  573  zero = literalSWord w 0
  574 
  575 svCountLeadingZeros :: SWord -> SWord
  576 svCountLeadingZeros xs = go 0 bits
  577  where
  578  bits = svBlastBE xs
  579  w    = length bits
  580  go !i []     = literalSWord w i
  581  go !i (b:bs) = svIte b (literalSWord w i) (go (i+1) bs)
  582 
  583 svCountTrailingZeros :: SWord -> SWord
  584 svCountTrailingZeros xs = go 0 bits
  585  where
  586  bits = svBlastLE xs
  587  w    = length bits
  588  go !i []     = literalSWord w i
  589  go !i (b:bs) = svIte b (literalSWord w i) (go (i+1) bs)
  590 
  591 -- | Ceiling (log_2 x)
  592 sLg2 :: SWord -> SWord
  593 sLg2 x = go 0
  594   where
  595     lit n = literalSWord (intSizeOf x) n
  596     go i | i < intSizeOf x = svIte (svLessEq x (lit (2^i))) (lit (toInteger i)) (go (i + 1))
  597          | otherwise       = lit (toInteger i)
  598 
  599 -- NB: SBV's division operation provides SMT-LIB's Euclidean division, which
  600 -- doesn't match the round-to-neg-infinity semantics of Cryptol, so we have to
  601 -- do some work to get the desired semantics.
  602 svIntDiv :: SVal -> SVal -> SVal
  603 svIntDiv a b = svSymbolicMerge KUnbounded True p (svQuot a b) (svQuot (svUNeg a) (svUNeg b))
  604   where
  605     z = svInteger KUnbounded 0
  606     p = svLessThan z b
  607 
  608 -- NB: SBV's division operation provides SMT-LIB's Euclidean division, which
  609 -- doesn't match the round-to-neg-infinity semantics of Cryptol, so we have to
  610 -- do some work to get the desired semantics.
  611 svIntMod :: SVal -> SVal -> SVal
  612 svIntMod a b = svSymbolicMerge KUnbounded True p (svRem a b) (svUNeg (svRem (svUNeg a) (svUNeg b)))
  613   where
  614     z = svInteger KUnbounded 0
  615     p = svLessThan z b

  634 -- | Abstract constants with names in the list 'unints' are kept as
  635 -- uninterpreted constants; all others are unfolded.
  636 sbvSolveBasic :: SharedContext -> Map Ident SPrim -> Set VarIndex -> Term -> IO SValue
  637 sbvSolveBasic sc addlPrims unintSet t = do
  638   m <- scGetModuleMap sc
  639 
  640   let variable (VarName ix nm) ty = parseUninterpreted [] (Text.unpack nm ++ "#" ++ show ix) ty
  641   let uninterpreted nm ty
  642         | Set.member (nameIndex nm) unintSet =
  643           let vn = VarName (nameIndex nm) (toShortName (nameInfo nm)) in Just (variable vn ty)
  644         | otherwise                          = Nothing
  645   let primHandler = Sim.defaultPrimHandler
  646   let mux = Prims.lazyMuxValue prims
  647   cfg <- Sim.evalGlobal m (Map.union constMap addlPrims) variable uninterpreted recursor primHandler mux
  648   Sim.evalSharedTerm cfg t

  650 parseUninterpreted :: [SVal] -> String -> TValue SBV -> IO SValue
  651 parseUninterpreted cws nm ty =
  652   case ty of
  653     (VPiType _ body)
  654       -> return $
  655          VFun $ \x ->
  656            do x' <- force x
  657               (cws', suffix) <- flattenSValue nm x'
  658               t2 <- applyPiBody body (ready x')
  659               parseUninterpreted (cws ++ cws') (nm ++ suffix) t2
  660 
  661     VBoolType
  662       -> return $ vBool $ mkUninterpreted KBool cws nm
  663 
  664     VIntType
  665       -> return $ vInteger $ mkUninterpreted KUnbounded cws nm
  666 
  667     VIntModType n
  668       -> return $ VIntMod n $ mkUninterpreted KUnbounded cws nm
  669 
  670     (VVecType n VBoolType)
  671       -> return $ vWord $ mkUninterpreted (KBounded False (fromIntegral n)) cws nm
  672 
  673     (VVecType n ety)
  674       -> do xs <- sequence $
  675                   [ parseUninterpreted cws (nm ++ "@" ++ show i) ety
  676                   | i <- [0 .. n-1] ]
  677             return (VVector (V.fromList (map ready xs)))
  678 
  679     VUnitType
  680       -> return VUnit
  681 
  682     (VPairType ty1 ty2)
  683       -> do x1 <- parseUninterpreted cws (nm ++ ".L") ty1
  684             x2 <- parseUninterpreted cws (nm ++ ".R") ty2
  685             return (VPair (ready x1) (ready x2))
  686 
  687     VDataType (nameInfo -> ModuleIdentifier "Prelude.EmptyType") [] []
  688       -> pure VEmptyRecord
  689     VDataType (nameInfo -> ModuleIdentifier "Prelude.RecordType")
  690       [VString fname, TValue ty1, TValue ty2] []
  691       -> do x1 <- parseUninterpreted cws (nm ++ "." ++ Text.unpack fname) ty1
  692             x2 <- parseUninterpreted cws nm ty2
  693             pure (VRecordValue fname (ready x1) x2)
  694 
  695     _ -> fail $ "could not create uninterpreted type for " ++ show ty
  696 
  697 mkUninterpreted :: Kind -> [SVal] -> String -> SVal
  698 mkUninterpreted k args nm =
  699   svUninterpreted k nm'
  700 #if MIN_VERSION_sbv(10,3,0)
  701                   (UINone True)
  702 #elif MIN_VERSION_sbv(10,0,0)
  703                   UINone
  704 #else
  705                   Nothing
  706 #endif
  707                   args
  708   where nm' = "|" ++ nm ++ "|" -- enclose name to allow primes and other non-alphanum chars

  710 sbvSATQuery :: SharedContext -> Map Ident SPrim -> SATQuery -> IO ([Labeler], [VarName], Symbolic SBool)
  711 sbvSATQuery sc addlPrims query =
  712   do t <- liftIO (satQueryAsTerm sc query)
  713      let qvars = Map.toList (satVariables query)
  714      let unintSet = satUninterp query
  715      let mkVars (vn, fot) = newVars (Text.unpack (vnName vn)) fot
  716 
  717      (labels, vars) <-
  718        flip evalStateT 0 $ unzip <$>
  719        mapM mkVars qvars
  720 
  721      m <- liftIO (scGetModuleMap sc)
  722 
  723      return (labels, map fst qvars,
  724        do vars' <- sequence vars
  725           let varMap = Map.fromList (zip (map (vnIndex . fst) qvars) vars')
  726 
  727           let mkUninterp (VarName ix nm) ty =
  728                 parseUninterpreted [] (Text.unpack nm ++ "#" ++ show ix) ty
  729           let variable vn tp
  730                 | Just v <- Map.lookup (vnIndex vn) varMap = pure v
  731                 | otherwise = mkUninterp vn tp
  732           let uninterpreted nm ty
  733                 | Set.member (nameIndex nm) unintSet =
  734                   let vn = VarName (nameIndex nm) (toShortName (nameInfo nm))
  735                   in Just (mkUninterp vn ty)
  736                 | otherwise                          = Nothing
  737           let primHandler = Sim.defaultPrimHandler
  738           let mux = Prims.lazyMuxValue prims
  739 
  740           cfg  <- liftIO (Sim.evalGlobal m (Map.union constMap addlPrims) variable uninterpreted recursor primHandler mux)
  741           bval <- liftIO (Sim.evalSharedTerm cfg t)
  742 
  743           case bval of
  744             VBool b -> return b
  745             _ -> fail $ "sbvSATQuery: non-boolean result type. " ++ show bval
  746       )

  748 data Labeler
  749    = BoolLabel String
  750    | IntegerLabel String
  751    | WordLabel String
  752    | ZeroWidthWordLabel
  753    | VecLabel
  754        FirstOrderType
  755       -- ^ The element type. It is necessary to store this in case the Vec is
  756       -- empty, in which case we cannot retrieve the type from the element
  757       -- values.
  758        (Vector Labeler)
  759    | TupleLabel (Vector Labeler)
  760    | RecLabel (Map FieldName Labeler)
  761      deriving (Show)

  763 nextId :: StateT Int IO String
  764 nextId = ST.get >>= (\s -> modify (+1) >> return ("x" ++ show s))
  765 
  766 nextId' :: String -> StateT Int IO String
  767 nextId' nm = nextId <&> \s -> s ++ "_" ++ nm
  768 
  769 unzipMap :: Map k (a, b) -> (Map k a, Map k b)
  770 unzipMap m = (fmap fst m, fmap snd m)
  771 
  772 newVars :: String -> FirstOrderType -> StateT Int IO (Labeler, Symbolic SValue)
  773 newVars nm fot =
  774   case fot of
  775     FOTBit ->
  776       nextId' nm <&> \s -> (BoolLabel s, vBool <$> existsSBool s)
  777     FOTInt ->
  778       nextId' nm <&> \s -> (IntegerLabel s, vInteger <$> existsSInteger s)
  779     FOTIntMod n ->
  780       nextId' nm <&> \s -> (IntegerLabel s, VIntMod n <$> existsSInteger s)
  781     FOTVec 0 FOTBit ->
  782       pure (ZeroWidthWordLabel, pure (vWord (literalSWord 0 0)))
  783     FOTVec n FOTBit ->
  784       nextId' nm <&> \s -> (WordLabel s, vWord <$> existsSWord s (fromIntegral n))
  785     FOTVec n tp ->
  786       do let f i = newVars (nm ++ "." ++ show i) tp
  787          (labels, vals) <- V.unzip <$> V.generateM (fromIntegral n) f
  788          pure (VecLabel tp labels, VVector <$> traverse (fmap ready) vals)
  789     FOTArray{} ->
  790       fail "FOTArray unimplemented for backend"
  791     FOTTuple ts ->
  792       do let f i t = newVars (nm ++ "." ++ show i) t
  793          (labels, vals) <- V.unzip <$> V.imapM f (V.fromList ts)
  794          pure (TupleLabel labels, vTuple <$> traverse (fmap ready) (V.toList vals))
  795     FOTRec tm ->
  796       do let f k t = newVars (nm ++ "." ++ Text.unpack k) t
  797          (labels, vals) <- unzipMap <$> (Map.traverseWithKey f tm)
  798          pure (RecLabel labels, vRecord <$> traverse (fmap ready) vals)

  801 getLabels ::
  802   [Labeler] ->
  803   Map String CV ->
  804   [VarName] -> Maybe [(VarName, FirstOrderValue)]
  805 
  806 getLabels ls d args
  807   | length args == length xs = Just (zip args xs)
  808   | otherwise = error $ unwords
  809                 [ "SBV SAT results do not match expected arguments "
  810                 , show (map vnName args), show xs]
  811 
  812   where
  813   xs = fmap getLabel ls
  814 
  815   getLabel (BoolLabel s)    = FOVBit (cvToBool (d Map.! s))
  816   getLabel (IntegerLabel s) = FOVInt (cvToInteger (d Map.! s))
  817 
  818   getLabel (WordLabel s)    = FOVWord (cvKind cv) (cvToInteger cv)
  819     where cv = d Map.! s
  820 
  821   getLabel ZeroWidthWordLabel = FOVWord 0 0
  822 
  823   getLabel (VecLabel t ns) = fovVec t (map getLabel (V.toList ns))
  824 
  825   getLabel (TupleLabel ns) = FOVTuple $ map getLabel (V.toList ns)
  826   getLabel (RecLabel ns) = FOVRec $ fmap getLabel ns
  827 
  828   cvKind cv =
  829     case kindOf cv of
  830       KBounded _ k -> fromIntegral k
  831       _                -> error "cvKind"
  832 
  833   cvToInteger cv =
  834     case cvVal cv of
  835       CInteger i -> i
  836       _               -> error "cvToInteger"

  842 newCodeGenVars :: (Natural -> Bool) -> FirstOrderType -> StateT Int IO (SBVCodeGen SValue)
  843 newCodeGenVars _checkSz FOTBit = nextId <&> \s -> (vBool <$> svCgInput KBool s)
  844 newCodeGenVars _checkSz FOTInt = nextId <&> \s -> (vInteger <$> svCgInput KUnbounded s)
  845 newCodeGenVars _checkSz (FOTIntMod _) = nextId <&> \s -> (vInteger <$> svCgInput KUnbounded s)
  846 newCodeGenVars checkSz (FOTVec n FOTBit)
  847   | n == 0    = nextId <&> \_ -> return (vWord (literalSWord 0 0))
  848   | checkSz n = nextId <&> \s -> vWord <$> cgInputSWord s (fromIntegral n)
  849   | otherwise = nextId <&> \s -> fail $ "Invalid codegen bit width for input variable \'" ++ s ++ "\': " ++ show n
  850 newCodeGenVars checkSz (FOTVec n (FOTVec m FOTBit))
  851   | m == 0    = nextId <&> \_ -> return (VVector $ V.fromList $ replicate (fromIntegral n) (ready $ vWord (literalSWord 0 0)))
  852   | checkSz m = do
  853       let k = KBounded False (fromIntegral m)
  854       vals <- nextId <&> \s -> svCgInputArr k (fromIntegral n) s
  855       return (VVector . V.fromList . fmap (ready . vWord) <$> vals)
  856   | otherwise = nextId <&> \s -> fail $ "Invalid codegen bit width for input variable array \'" ++ s ++ "\': " ++ show n
  857 newCodeGenVars checkSz (FOTVec n tp) = do
  858   vals <- V.replicateM (fromIntegral n) (newCodeGenVars checkSz tp)
  859   return (VVector <$> traverse (fmap ready) vals)
  860 newCodeGenVars _ (FOTArray{}) = fail "FOTArray unimplemented for backend"
  861 newCodeGenVars checkSz (FOTTuple ts) = do
  862   vals <- traverse (newCodeGenVars checkSz) ts
  863   return (vTuple <$> traverse (fmap ready) vals)
  864 newCodeGenVars checkSz (FOTRec tm) = do
  865   vals <- traverse (newCodeGenVars checkSz) tm
  866   return (vRecord <$> traverse (fmap ready) vals)
  867 
  868 cgInputSWord :: String -> Int -> SBVCodeGen SWord
  869 cgInputSWord s n = svCgInput (KBounded False n) s

  871 argTypes :: SharedContext -> Term -> IO ([Term], Term)
  872 argTypes sc t = do
  873   t' <- scWhnf sc t
  874   case t' of
  875     (R.asPi -> Just (_, t1, t2)) -> do
  876        (ts,res) <- argTypes sc t2
  877        return (t1:ts, res)
  878     _ -> return ([], t')

  880 sbvCodeGen_definition
  881   :: SharedContext
  882   -> Map Ident SPrim
  883   -> Set VarIndex
  884   -> Term
  885   -> (Natural -> Bool) -- ^ Allowed word sizes
  886   -> IO (SBVCodeGen (), [FirstOrderType], FirstOrderType)
  887 sbvCodeGen_definition sc addlPrims unintSet t checkSz = do
  888   ty <- scTypeOf sc t
  889   (argTs,resTy) <- argTypes sc ty
  890   shapes <- traverse (asFirstOrderType sc) argTs
  891   resultShape <- asFirstOrderType sc resTy
  892   bval <- sbvSolveBasic sc addlPrims unintSet t
  893   vars <- evalStateT (traverse (newCodeGenVars checkSz) shapes) 0
  894   let codegen = do
  895         args <- traverse (fmap ready) vars
  896         bval' <- liftIO (applyAll bval args)
  897         sbvSetResult checkSz resultShape bval'
  898   return (codegen, shapes, resultShape)

  901 sbvSetResult :: (Natural -> Bool)
  902              -> FirstOrderType
  903              -> SValue
  904              -> SBVCodeGen ()
  905 sbvSetResult _checkSz FOTBit (VBool b) = do
  906    svCgReturn b
  907 sbvSetResult checkSz (FOTVec n FOTBit) v
  908    | n == 0    = return ()
  909    | checkSz n = do
  910       w <- liftIO $ toWord v
  911       svCgReturn w
  912    | otherwise =
  913       fail $ "Invalid word size in result: " ++ show n
  914 sbvSetResult checkSz ft v = do
  915    void $ sbvSetOutput checkSz ft v 0
  916

  918 sbvSetOutput :: (Natural -> Bool)
  919              -> FirstOrderType
  920              -> SValue
  921              -> Int
  922              -> SBVCodeGen Int
  923 sbvSetOutput _checkSz FOTBit (VBool b) i = do
  924    svCgOutput ("out_"++show i) b
  925    return $! i+1
  926 sbvSetOutput checkSz (FOTVec n FOTBit) v i
  927    | n == 0    = return i
  928    | checkSz n = do
  929        w <- liftIO $ toWord v
  930        svCgOutput ("out_"++show i) w
  931        return $! i+1
  932    | otherwise =
  933        fail $ "Invalid word size in output " ++ show i ++ ": " ++ show n
  934 
  935 sbvSetOutput checkSz (FOTVec n t) (VVector xv) i = do
  936    xs <- liftIO $ traverse force $ V.toList xv
  937    unless (toInteger n == toInteger (length xs)) $
  938      fail "sbvCodeGen: vector length mismatch when setting output values"
  939    case asWordList xs of
  940      Just ws -> do svCgOutputArr ("out_"++show i) ws
  941                    return $! i+1
  942      Nothing -> foldM (\i' x -> sbvSetOutput checkSz t x i') i xs
  943 sbvSetOutput _checkSz (FOTTuple []) VUnit i =
  944    return i
  945 sbvSetOutput checkSz (FOTTuple [t]) v i = sbvSetOutput checkSz t v i
  946 sbvSetOutput checkSz (FOTTuple (t:ts)) (VPair l r) i = do
  947    l' <- liftIO $ force l
  948    r' <- liftIO $ force r
  949    sbvSetOutput checkSz t l' i >>= sbvSetOutput checkSz (FOTTuple ts) r'
  950 
  951 sbvSetOutput _checkSz (FOTRec fs) VUnit i | Map.null fs = do
  952    return i
  953 
  954 sbvSetOutput _checkSz (FOTRec fs) VEmptyRecord i | Map.null fs = return i
  955 
  956 sbvSetOutput checkSz (FOTRec fs) (VRecordValue fn x rest) i = do
  957    x' <- liftIO $ force x
  958    case Map.lookup fn fs of
  959      Just t -> do
  960        let fs' = Map.delete fn fs
  961        sbvSetOutput checkSz t x' i >>=
  962          sbvSetOutput checkSz (FOTRec fs') rest
  963      Nothing -> fail "sbvCodeGen: type mismatch when setting record output value"
  964 
  965 sbvSetOutput _checkSz _ft _v _i = do
  966    fail "sbvCode gen: type mismatch when setting output values"

  970            -> Map Ident SPrim
  971            -> Set VarIndex
  972            -> Maybe FilePath
  973            -> String
  974            -> Term
  975            -> IO ()
  976 sbvCodeGen sc addlPrims unintSet path fname t = do
  977   -- The SBV C code generator expects only these word sizes
  978   let checkSz n = n `elem` [8,16,32,64]
