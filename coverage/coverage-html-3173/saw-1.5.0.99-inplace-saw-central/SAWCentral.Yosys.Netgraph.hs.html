   53 --------------------------------------------------------------------------------
   54 -- ** Building a network graph from a Yosys module
   55 
   56 -- | A 'Netgraph' represents the data dependencies between 'Cell's in
   57 -- a module. The graph has 'CellInstName' keys which are the cell instance
   58 -- names.
   59 newtype Netgraph = Netgraph [(Cell, CellInstName, [CellInstName])]
   60 
   61 moduleNetgraph :: Map CellTypeName ConvertedModule -> Module -> Netgraph
   62 moduleNetgraph env m =
   63   let
   64     sources :: Map Bitrep CellInstName
   65     sources =
   66       Map.fromList $
   67       [ (b, cname)
   68       | (cname, c) <- Map.assocs (m ^. moduleCells)
   69       , b <- concat (Map.elems (cellOutputConnections c)) ]
   70 
   71     isMooreMachine :: CellTypeName -> Bool
   72     isMooreMachine t =
   73       case Map.lookup t env of
   74         Nothing -> False
   75         Just cm ->
   76           case _convertedModuleState cm of
   77             Nothing -> False
   78             Just (mt, _) -> mt == Moore
   79 
   80     asyncInputs :: CellTypeRegister -> Set PortName
   81     asyncInputs ctr =
   82       case ctr of
   83         CellTypeAdff _ -> Set.fromList ["ARST"]
   84         CellTypeAldff _ -> Set.fromList ["ALOAD", "AD"]
   85         CellTypeDff _ -> Set.empty
   86         CellTypeDffsr _ -> Set.fromList ["SET", "CLR"]
   87         CellTypeFf -> Set.empty
   88         CellTypeSdff _ -> Set.empty
   89         CellTypeSdffe -> Set.empty
   90 
   91     cellDeps :: Cell -> [CellInstName]
   92     cellDeps c =
   93       case c ^. cellType of
   94         CellTypeRegister ctr ->
   95           -- Register cells only have direct data dependencies on
   96           -- their asynchronous inputs.
   97           Set.toAscList $ Set.fromList $
   98           Maybe.mapMaybe (flip Map.lookup sources) $
   99           concat $ Map.elems $
  100           flip Map.restrictKeys (asyncInputs ctr) $
  101           cellInputConnections c
  102         CellTypeUserType t
  103           | isMooreMachine t -> []
  104         _ ->
  105           -- For other cells, we record direct data dependencies for
  106           -- all input ports.
  107           Set.toAscList $ Set.fromList $
  108           Maybe.mapMaybe (flip Map.lookup sources) $
  109           concat $ Map.elems $ cellInputConnections c
  110 
  111     nodes :: [(Cell, CellInstName, [CellInstName])]
  112     nodes = [ (c, cname, cellDeps c) | (cname, c) <- Map.assocs (m ^. moduleCells) ]
  113   in
  114     Netgraph nodes

  117 -- ** Building a SAWCore term from a network graph

  119 -- | A Mealy machine allows same-cycle data flow from inputs to
  120 -- outputs, while a Moore machine only allows the outputs to depend on
  121 -- the internal state.
  122 data MachineType = Mealy | Moore
  123   deriving Eq

  125 -- | A SAWCore translation of a module imported from the Yosys JSON format.
  126 data ConvertedModule = ConvertedModule
  127   { _convertedModuleTerm :: SC.Term
  128     -- ^ A SAWCore term representing the module semantics, of type
  129     -- @Input -> Output@ for combinational modules, @{step : Input ->
  130     -- State -> State, out : Input -> State -> Output}@ for Mealy
  131     -- machines, or @{step : Input -> State -> State, out : State ->
  132     -- Output}@ for Moore machines.
  133   , _convertedModuleState :: Maybe (MachineType, SC.Term)
  134     -- ^ The module's state type represented in both SAWCore and
  135     -- Cryptol, or 'Nothing' if the module is purely combinational.
  136   }
  137 makeLenses ''ConvertedModule

  139 -- | A mapping from wires to their corresponding terms.
  140 type WireEnv = Map [Bitrep] Preterm
  141 
  142 lookupPatternTerm ::
  143   SC.SharedContext ->
  144   YosysBitvecConsumer ->
  145   [Bitrep] ->
  146   WireEnv ->
  147   IO SC.Term
  148 lookupPatternTerm sc loc pat ts =
  149   case Map.lookup pat ts of
  150     -- if we can find that pattern exactly, great! use that term
  151     Just t -> scPreterm sc t
  152     -- otherwise, find each individual bit and append the terms
  153     Nothing ->
  154       do bits <-
  155            forM pat $ \b ->
  156            case Map.lookup [b] ts of
  157              Just t -> pure t
  158              Nothing -> yosysError $ YosysErrorNoSuchOutputBitvec (Text.pack $ show b) loc
  159          -- Yosys lists bits in little-endian order, while scVector expects big-endian, so reverse
  160          let ps = fusePreterms (reverse bits)
  161          scPreterms sc ps

  163 -- | Given a netgraph and an initial map from bit patterns to terms,
  164 -- populate that map with terms generated from the rest of the
  165 -- netgraph.
  166 netgraphToTerms ::
  167   SC.SharedContext ->
  168   Map CellTypeName ConvertedModule ->
  169   CellTypeName {- ^ Module type being translated, for error reporting -} ->
  170   Netgraph ->
  171   WireEnv ->
  172   Map CellInstName SC.Term {- ^ state inputs -} ->
  173   IO WireEnv
  174 netgraphToTerms sc env mname (Netgraph nodes) inputs states =
  175   foldM doVertex inputs (Graph.stronglyConnCompR nodes)
  176   where
  177     doVertex :: WireEnv -> Graph.SCC (Cell, CellInstName, [CellInstName]) -> IO WireEnv
  178     doVertex _ (Graph.CyclicSCC vs) =
  179       yosysError $ YosysErrorCyclicDependency mname [ cnm | (_, cnm, _) <- vs ]
  180     doVertex acc (Graph.AcyclicSCC (c, cnm, _deps)) =
  181       do let outputFields = Map.filter isOutput (c ^. cellPortDirections)
  182          let lookupConn portname =
  183                case Map.lookup portname (c ^. cellConnections) of
  184                  Nothing ->
  185                    yosysError $
  186                    YosysError $
  187                    "Malformed Yosys file: Missing port " <> portname <> " for cell " <> cnm
  188                  Just bs ->
  189                    pure bs
  190          let input portname =
  191                do bs <- lookupConn portname
  192                   lookupPatternTerm sc (YosysBitvecConsumerCell cnm portname) bs acc
  193          let inputBool portname =
  194                do t <- input portname
  195                   one <- SC.scNat sc 1
  196                   SC.scBvNonzero sc one t
  197          let lookupBoolParam pname =
  198                Maybe.fromMaybe True $ parseBool =<< Map.lookup pname (c ^. cellParameters)
  199          let lookupNatParam pname =
  200                Maybe.fromMaybe 0 $ parseNat =<< Map.lookup pname (c ^. cellParameters)
  201          case c ^. cellType of
  202            CellTypeCombinational ctc ->
  203              -- NOTE: All Yosys primitive combinational cell types
  204              -- have a single output port named "Y".
  205              do let doInput inm i =
  206                       do t <- lookupPatternTerm sc (YosysBitvecConsumerCell cnm inm) i acc
  207                          let w = fromIntegral (length i)
  208                          let s =
  209                                case Map.lookup (inm <> "_SIGNED") (c ^. cellParameters) of
  210                                  Just (Aeson.Number n) -> n > 0
  211                                  Just (Aeson.String x) -> textBinNat x > 0
  212                                  Just _ -> False
  213                                  Nothing -> False
  214                          pure (CellTerm t w s)
  215                 args <- Map.traverseWithKey doInput (cellInputConnections c)
  216                 bs <- lookupConn "Y"
  217                 let ywidth = fromIntegral (length bs)
  218                 t <- combCellToTerm sc ctc args ywidth
  219                 ts <- deriveTermsByIndices sc bs t
  220                 pure $ Map.union ts acc
  221            CellTypeRegister ctr ->
  222              -- NOTE: All Yosys primitive register cell types have a
  223              -- single output port named "Q".
  224              do bs <- lookupConn "Q"
  225                 let width = fromIntegral (length bs)
  226                 r <-
  227                   case Map.lookup cnm states of
  228                     Nothing ->
  229                       panic "netgraphToTerms" ["missing state for cell " <> cnm]
  230                     Just r ->
  231                       case ctr of
  232                         CellTypeAdff _ ->
  233                           do let arst_value = lookupNatParam "ARST_VALUE"
  234                              let arst_polarity = lookupBoolParam "ARST_POLARITY"
  235                              arst_value' <- SC.scBvConst sc width (fromIntegral arst_value)
  236                              arst <- inputBool "ARST"
  237                              -- complement reset signal if ARST_POLARITY=0
  238                              pos_arst <- if arst_polarity then pure arst else SC.scNot sc arst
  239                              -- Set output to reset value on ARST; else output state value
  240                              ty <- SC.scBitvector sc width
  241                              SC.scIte sc ty pos_arst arst_value' r
  242                         CellTypeAldff _ ->
  243                           do let aload_polarity = lookupBoolParam "ALOAD_POLARITY"
  244                              ad <- input "AD"
  245                              aload <- inputBool "ALOAD"
  246                              -- complement reset signal if ALOAD_POLARITY=0
  247                              pos_aload <- if aload_polarity then pure aload else SC.scNot sc aload
  248                              -- Set output to AD on ALOAD; else output state value
  249                              ty <- SC.scBitvector sc width
  250                              SC.scIte sc ty pos_aload ad r
  251                         CellTypeDffsr _ ->
  252                           do let set_polarity = lookupBoolParam "SET_POLARITY"
  253                              let clr_polarity = lookupBoolParam "CLR_POLARITY"
  254                              set <- input "SET"
  255                              clr <- input "CLR"
  256                              w <- SC.scNat sc width
  257                              pos_set <- if set_polarity then pure set else SC.scBvNot sc w set
  258                              neg_clr <- if clr_polarity then SC.scBvNot sc w clr else pure clr
  259                              -- CLR takes priority over SET
  260                              SC.scBvAnd sc w neg_clr =<< SC.scBvOr sc w pos_set r
  261                         -- For all register cell types without
  262                         -- asynchronous set/reset, the output is
  263                         -- always identical to the stored value @r@
  264                         -- from the state record.
  265                         CellTypeDff _ -> pure r
  266                         CellTypeFf -> pure r
  267                         CellTypeSdff _ -> pure r
  268                         CellTypeSdffe -> pure r
  269 
  270                 ts <- deriveTermsByIndices sc bs r
  271                 pure $ Map.union ts acc
  272            CellTypeUnsupportedPrimitive nm
  273              | Map.null outputFields ->
  274                  -- Cells with no output ports are debugging cells, which we can simply ignore.
  275                  pure acc
  276              | otherwise ->
  277                  yosysError $ YosysErrorUnsupportedFF nm
  278            CellTypeUserType submoduleName ->
  279              case Map.lookup submoduleName env of
  280                Nothing ->
  281                  yosysError $ YosysErrorNoSuchSubmodule submoduleName cnm
  282                Just cm ->
  283                  do inputArgs <-
  284                       case cm ^. convertedModuleState of
  285                         Just (Moore, _) -> pure []
  286                         _ ->
  287                           do let doInput inm i = lookupPatternTerm sc (YosysBitvecConsumerCell cnm inm) i acc
  288                              args <- Map.traverseWithKey doInput (cellInputConnections c)
  289                              rin <- cryptolRecord sc args
  290                              pure [rin]
  291                     let stateArgs = Maybe.maybeToList (Map.lookup cnm states)
  292                     f <-
  293                       case cm ^. convertedModuleState of
  294                         Nothing -> pure (cm ^. convertedModuleTerm)
  295                         Just _ -> SC.scRecordSelect sc (cm ^. convertedModuleTerm) "out"
  296                     r <- SC.scApplyAll sc f (inputArgs ++ stateArgs)
  297 
  298                     -- once we've built a term, insert it along with each of its bits
  299                     ts <-
  300                       forM (Map.assocs $ cellOutputConnections c) $ \(o, out) ->
  301                       do t <- cryptolRecordSelect sc outputFields r o
  302                          deriveTermsByIndices sc out t
  303                     pure $ Map.union (Map.unions ts) acc

  305 -- | Compute the new state value for a stateful cell type.
  306 -- This function should be called with a complete 'WireEnv' with
  307 -- values for every wire, including the output wires for the given
  308 -- cell.
  309 cellNewState ::
  310   SC.SharedContext ->
  311   Map CellTypeName ConvertedModule ->
  312   WireEnv ->
  313   CellInstName ->
  314   (Cell, SC.Term) ->
  315   IO SC.Term
  316 cellNewState sc env terms cnm (c, prevState) =
  317   case c ^. cellType of
  318     CellTypeRegister ctr ->
  319       case ctr of
  320         -- $adff, $adffe
  321         CellTypeAdff e ->
  322           do clk <- clockInput e
  323              CellTerm d width _ <- input "D" -- new value
  324              CellTerm q _ _ <- input "Q" -- old state value
  325              pos_arst <- inputBoolWithPolarity "ARST"
  326              let arst_value = Maybe.fromMaybe 0 (lookupNatParam "ARST_VALUE")
  327              arst_value' <- SC.scBvConst sc width (fromIntegral arst_value)
  328              ty <- SC.scBitvector sc width
  329              -- Set state to reset value on ARST; else if CLK then D; otherwise hold
  330              SC.scIte sc ty pos_arst arst_value' =<< SC.scIte sc ty clk d q
  331         -- $aldff, $aldffe
  332         CellTypeAldff e ->
  333           do clk <- clockInput e
  334              CellTerm ad _ _ <- input "AD" -- async load value
  335              CellTerm d width _ <- input "D" -- new value
  336              CellTerm q _ _ <- input "Q" -- old state value
  337              pos_aload <- inputBoolWithPolarity "ALOAD"
  338              ty <- SC.scBitvector sc width
  339              -- Set state to AD on ALOAD; else if CLK then D; otherwise hold
  340              SC.scIte sc ty pos_aload ad =<< SC.scIte sc ty clk d q
  341         -- $dff, $dffe
  342         CellTypeDff e ->
  343           do clk <- clockInput e
  344              CellTerm d width _ <- input "D" -- new value
  345              CellTerm q _ _ <- input "Q" -- old state value
  346              ty <- SC.scBitvector sc width
  347              -- update state to D on CLK; otherwise hold
  348              SC.scIte sc ty clk d q
  349         CellTypeFf ->
  350           -- $ff cell has no CLK input; it uses the global clock, so
  351           -- it transitions every time step
  352           cellTermTerm <$> input "D"
  353         -- $dffsr, $dffsre
  354         CellTypeDffsr e ->
  355           do clk <- clockInput e
  356              CellTerm d width _ <- input "D" -- new value
  357              CellTerm q _ _ <- input "Q" -- old state value
  358              CellTerm set _ _ <- input "SET"
  359              CellTerm clr _ _ <- input "CLR"
  360              let set_polarity = Maybe.fromMaybe True (lookupBoolParam "SET_POLARITY")
  361              let clr_polarity = Maybe.fromMaybe True (lookupBoolParam "CLR_POLARITY")
  362              w <- SC.scNat sc width
  363              ty <- SC.scBitvector sc width
  364              pos_set <- if set_polarity then pure set else SC.scBvNot sc w set
  365              neg_clr <- if clr_polarity then SC.scBvNot sc w clr else pure clr
  366              -- Set each bit to 0 on CLR; else 1 on SET; else D on CLK; otherwise hold
  367              SC.scBvAnd sc w neg_clr =<< SC.scBvOr sc w pos_set =<< SC.scIte sc ty clk d q
  368         -- $sdff, $sdffce
  369         CellTypeSdff e ->
  370           do clk <- clockInput e
  371              CellTerm d width _ <- input "D" -- new value
  372              CellTerm q _ _ <- input "Q" -- old state value
  373              pos_srst <- inputBoolWithPolarity "SRST"
  374              let srst_value = Maybe.fromMaybe 0 (lookupNatParam "SRST_VALUE")
  375              srst_value' <- SC.scBvConst sc width (fromIntegral srst_value)
  376              ty <- SC.scBitvector sc width
  377              -- Set state to reset value on CLK & SRST; else if CLK then D; otherwise hold
  378              d' <- SC.scIte sc ty pos_srst srst_value' d
  379              SC.scIte sc ty clk d' q
  380         CellTypeSdffe ->
  381           do CellTerm d width _ <- input "D" -- new value
  382              CellTerm q _ _ <- input "Q" -- old state value
  383              clk <- clockInput WithoutClockEnable -- ungated clock signal
  384              pos_srst <- inputBoolWithPolarity "SRST"
  385              pos_en <- inputBoolWithPolarity "EN"
  386              let srst_value = Maybe.fromMaybe 0 (lookupNatParam "SRST_VALUE")
  387              srst_value' <- SC.scBvConst sc width (fromIntegral srst_value)
  388              ty <- SC.scBitvector sc width
  389              -- Set state to reset value on CLK & SRST; else if CLK & EN then D; otherwise hold
  390              rst <- SC.scAnd sc clk pos_srst
  391              trigger <- SC.scAnd sc clk pos_en
  392              SC.scIte sc ty rst srst_value' =<< SC.scIte sc ty trigger d q
  393     CellTypeCombinational _ ->
  394       panic "cellNewState" ["unexpected combinational cell"]
  395     CellTypeUnsupportedPrimitive _ ->
  396       panic "cellNewState" ["unexpected unsupported cell"]
  397     CellTypeUserType submoduleName ->
  398       case Map.lookup submoduleName env of
  399         Nothing ->
  400           yosysError $ YosysErrorNoSuchSubmodule submoduleName cnm
  401         Just cm ->
  402           do -- Both Mealy and Moore machines have a "step" field of type `Input -> State -> State`
  403              f <- SC.scRecordSelect sc (cm ^. convertedModuleTerm) "step"
  404              let doInput inm i = lookupPatternTerm sc (YosysBitvecConsumerCell cnm inm) i terms
  405              args <- Map.traverseWithKey doInput (cellInputConnections c)
  406              rin <- cryptolRecord sc args
  407              SC.scApplyAll sc f [rin, prevState]
  408   where
  409     input :: PortName -> IO CellTerm
  410     input portname =
  411       do bs <- lookupConn portname
  412          t <- lookupPatternTerm sc (YosysBitvecConsumerCell cnm portname) bs terms
  413          let signed = False -- treat as unsigned
  414          pure (CellTerm t (fromIntegral (length bs)) signed)
  415     -- | Retrieve an input of type Bool.
  416     inputBool :: PortName -> IO SC.Term
  417     inputBool portname =
  418       do CellTerm t _ _ <- input portname
  419          one <- SC.scNat sc 1
  420          SC.scBvNonzero sc one t
  421     inputBoolWithPolarity :: PortName -> IO SC.Term
  422     inputBoolWithPolarity portname =
  423       do t <- inputBool portname
  424          let polarity = Maybe.fromMaybe True (lookupBoolParam (portname <> "_POLARITY"))
  425          -- complement signal if <portname>_POLARITY=0
  426          if polarity then pure t else SC.scNot sc t
  427     lookupConn portname =
  428       case Map.lookup portname (c ^. cellConnections) of
  429         Nothing ->
  430           yosysError $
  431           YosysError $
  432           "Malformed Yosys file: Missing port " <> portname <> " for cell " <> cnm
  433         Just bs ->
  434           pure bs
  435     lookupNatParam :: Text.Text -> Maybe Natural
  436     lookupNatParam pname = parseNat =<< Map.lookup pname (c ^. cellParameters)
  437     lookupBoolParam :: Text.Text -> Maybe Bool
  438     lookupBoolParam pname = parseBool =<< Map.lookup pname (c ^. cellParameters)
  439     -- | @clockInput WithoutClockEnable@ reads @CLK@ and enforces @CLK_POLARITY=1@.
  440     -- @clockInput WithClockEnable@ additionally reads input @EN@, inverts it if
  441     -- @EN_POLARITY=0@, then ANDs it with the clock.
  442     clockInput :: ClockEnable -> IO SC.Term
  443     clockInput e =
  444       do clk <- inputBool "CLK"
  445          let clk_polarity = Maybe.fromMaybe True (lookupBoolParam "CLK_POLARITY")
  446          let ty = ppCellType (c ^. cellType)
  447          -- We only support CLK_POLARITY=1, i.e. posedge CLK
  448          unless clk_polarity $
  449            yosysError $ YosysError $ "Unsupported " <> ty <> " with CLK_POLARITY=0"
  450          case e of
  451            WithClockEnable -> SC.scAnd sc clk =<< inputBoolWithPolarity "EN"
  452            WithoutClockEnable -> pure clk
  453 
  454 -- | Parse an Aeson value as a 'Bool', if possible.
  455 -- Note that Yosys encodes boolean parameters as either numbers like 0
  456 -- or 1, or strings like "0" and "1" (possibly with extra leading 0s).
  457 parseBool :: Aeson.Value -> Maybe Bool
  458 parseBool (Aeson.Number n) = Just (n > 0)
  459 parseBool (Aeson.String x) = Just (textBinNat x > 0)
  460 parseBool _ = Nothing
  461 
  462 -- | Parse an Aeson value as a 'Natural', if possible.
  463 parseNat :: Aeson.Value -> Maybe Natural
  464 parseNat (Aeson.String x) = Just (textBinNat x)
  465 parseNat _ = Nothing

  467 convertModule ::
  468   SC.SharedContext ->
  469   Map CellTypeName ConvertedModule ->
  470   CellTypeName ->
  471   Module ->
  472   IO ConvertedModule
  473 convertModule sc env mname m0 =
  474   do let m = renameDffInstances m0
  475      let ng = moduleNetgraph env m
  476 
  477      let inputPorts = moduleInputPorts m
  478      let outputPorts = moduleOutputPorts m
  479 
  480      let
  481        f :: [Bitrep] -> IO SC.Term
  482        f = SC.scBitvector sc . fromIntegral . length
  483 
  484      inputFields <- traverse f inputPorts
  485      inputRecordType <- cryptolRecordType sc inputFields
  486      inputVarName <- SC.scFreshVarName sc "input"
  487      inputRecord <- SC.scVariable sc inputVarName inputRecordType
  488 
  489      derivedInputs <-
  490        forM (Map.assocs inputPorts) $ \(nm, inp) ->
  491        do t <- cryptolRecordSelect sc inputFields inputRecord nm
  492           deriveTermsByIndices sc inp t
  493 
  494      -- Collect state types from all register cells in this module
  495      let
  496        registerCells :: Map CellInstName Cell
  497        registerCells = Map.filter cellIsRegister (m ^. moduleCells)
  498        registerPorts :: Map CellInstName [Bitrep]
  499        registerPorts = Map.mapMaybe (\c -> Map.lookup "Q" (c ^. cellConnections)) registerCells
  500      (stateFields1 :: Map CellInstName SC.Term) <- traverse f registerPorts
  501 
  502      -- Collect state types from all submodules
  503      let
  504        getSubmodule :: Cell -> Maybe ConvertedModule
  505        getSubmodule c =
  506          case c ^. cellType of
  507            CellTypeUserType t -> Map.lookup t env
  508            _ -> Nothing
  509        submodules :: Map CellInstName ConvertedModule
  510        submodules = Map.mapMaybe getSubmodule (m ^. moduleCells)
  511        stateSubmodules :: Map CellInstName (MachineType, SC.Term)
  512        stateSubmodules = Map.mapMaybe (\cm -> cm ^. convertedModuleState) submodules
  513        stateFields2 :: Map CellInstName SC.Term
  514        stateFields2 = fmap snd stateSubmodules
  515        stateFields :: Map CellInstName SC.Term
  516        stateFields = stateFields1 <> stateFields2
  517 
  518      stateRecordType <- cryptolRecordType sc stateFields
  519      stateVarName <- SC.scFreshVarName sc "state"
  520      stateRecord <- SC.scVariable sc stateVarName stateRecordType
  521      oldstates <-
  522        Map.traverseWithKey (\nm _inp -> cryptolRecordSelect sc stateFields stateRecord nm) stateFields
  523 
  524      oneBitType <- SC.scBitvector sc 1
  525      xMsg <- SC.scString sc "Attempted to read X bit"
  526      xTerm <- SC.scGlobalApply sc (SC.mkIdent SC.preludeName "error") [oneBitType, xMsg]
  527      zMsg <- SC.scString sc "Attempted to read Z bit"
  528      zTerm <- SC.scGlobalApply sc (SC.mkIdent SC.preludeName "error") [oneBitType, zMsg]
  529      let inputs = Map.unions $ mconcat
  530            [ [ Map.fromList
  531                [ ( [BitrepZero], PretermBvNat 1 0 )
  532                , ( [BitrepOne], PretermBvNat 1 1 )
  533                , ( [BitrepX], PretermSlice 0 1 0 xTerm )
  534                , ( [BitrepZ], PretermSlice 0 1 0 zTerm )
  535                ]
  536              ]
  537            , derivedInputs
  538            ]
  539 
  540      -- translate outputs of all cells in dependency order
  541      terms <- netgraphToTerms sc env mname ng inputs oldstates
  542      -- assemble the final output
  543      outputRecord <- cryptolRecord sc =<< mapForWithKeyM outputPorts
  544        (\onm out -> lookupPatternTerm sc (YosysBitvecConsumerOutputPort onm) out terms)
  545 
  546      -- translate the new state components of all stateful cells (if any)
  547      newstates <-
  548        Map.traverseWithKey (cellNewState sc env terms) $
  549        Map.intersectionWith (,) (m ^. moduleCells) oldstates
  550      newstateRecord <- cryptolRecord sc newstates
  551 
  552      case Map.null newstates of
  553        True ->
  554          -- No state; term for combinational module is function `Input -> Output`
  555          do t <- SC.scAbstractTerms sc [inputRecord] outputRecord
  556             pure $
  557               ConvertedModule
  558               { _convertedModuleTerm = t
  559               , _convertedModuleState = Nothing
  560               }
  561        False ->
  562          do step <- SC.scAbstractTerms sc [inputRecord, stateRecord] newstateRecord
  563             case IntSet.member (SC.vnIndex inputVarName) (SC.freeVars outputRecord) of
  564               True ->
  565                 -- Output depends on input, so we have a Mealy machine
  566                 do out <- SC.scAbstractTerms sc [inputRecord, stateRecord] outputRecord
  567                    t <- SC.scRecordValue sc [("out", out), ("step", step)]
  568                    pure $
  569                      ConvertedModule
  570                      { _convertedModuleTerm = t
  571                      , _convertedModuleState = Just (Mealy, stateRecordType)
  572                      }
  573               False ->
  574                 -- Output does not depend on input, so we have a Moore machine
  575                 do out <- SC.scAbstractTerms sc [stateRecord] outputRecord
  576                    t <- SC.scRecordValue sc [("out", out), ("step", step)]
  577                    pure $
  578                      ConvertedModule
  579                      { _convertedModuleTerm = t
  580                      , _convertedModuleState = Just (Moore, stateRecordType)
  581                      }
