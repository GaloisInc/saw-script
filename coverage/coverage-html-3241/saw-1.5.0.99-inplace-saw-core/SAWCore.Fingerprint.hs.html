   48 type FP = ReaderT FPCfg (State FPSt)

   50 -- | Read-only bits of state.
   51 data FPCfg = FPCfg
   52   { fpModuleMap :: !ModuleMap
   53     -- ^ The module map for looking up definitions.
   54   , fpUninterp  :: !(Set VarIndex)
   55     -- ^ The set of uninterpreted constants. When these are encountered, only
   56     --   the name and type are included in the serialization and not a definition, if
   57     --   it exists. See 'lookupName' and 'fpDef'.
   58   }
   59 
   60 -- | Mutable state tracked while serializing. Sequence numbers are used to
   61 --   preserve structural sharing. The first time we encounter a term or definition,
   62 --   we create a new sequence number and emit it along with a 'tagDef' followed by
   63 --   the definition. Subsequent encounters just emit the sequence number along with
   64 --   'tagRef'.
   65 data FPSt = FPSt
   66   { fpTermMemo    :: !(IntMap BS.Builder)
   67     -- ^ TermIndex -> 'tagRef' + sequence number (assigned on first serialization).
   68   , fpDefMemo     :: !(IntMap BS.Builder)
   69     -- ^ VarIndex of global Name -> 'tagRef' + sequence number.
   70   , fpNextRef     :: !Int
   71     -- ^ Next available sequence number.
   72   }
   73 
   74 -- | Markers used to preserve structural sharing without hashing.
   75 tagRef, tagDef :: Word8
   76 tagRef   = 0xF1  -- Reference to a previously serialized term or definition.
   77 tagDef   = 0xF2  -- Introduces a new sequence number followed by its body.
   78 
   79 ref :: Int -> BS.Builder
   80 ref n = byte tagRef <> bytes n
   81 
   82 def :: Int -> BS.Builder
   83 def n = byte tagDef <> bytes n
   84 
   85 freshRef :: MonadState FPSt m => m Int
   86 freshRef = state $ \s ->
   87   let n = fpNextRef s in (n, s { fpNextRef = n + 1 })
   88 
   89 -- | Memoize a term, associating a `TermIndex` with a new sequence number.
   90 --   Returns that sequence number prepended with 'tagDef'.
   91 rememberTerm :: MonadState FPSt m => Term -> m BS.Builder
   92 rememberTerm t = do
   93   n <- freshRef
   94   modify' $
   95     \s -> s { fpTermMemo = IntMap.insert (termIndex t) (ref n) $ fpTermMemo s }
   96   pure $ def n
   97 
   98 lookupTerm :: MonadState FPSt m => Term -> m (Maybe BS.Builder)
   99 lookupTerm t = IntMap.lookup (termIndex t) <$> gets fpTermMemo
  100 
  101 -- | Memoize a name, associating a `VarIndex` with a new sequence number.
  102 --   Returns that sequence number prepended with 'tagDef'.
  103 rememberDef :: MonadState FPSt m => Name -> m BS.Builder
  104 rememberDef nm = do
  105   n <- freshRef
  106   modify' $
  107     \s -> s { fpDefMemo = IntMap.insert (nameIndex nm) (ref n) $ fpDefMemo s }
  108   pure $ def n
  109 
  110 lookupDef :: MonadState FPSt m => Name -> m (Maybe BS.Builder)
  111 lookupDef nm = IntMap.lookup (nameIndex nm) <$> gets fpDefMemo
  112 
  113 -- | Lookup a name in the 'fpUninterp' set (the `Bool` result) and
  114 --   'fpModuleMap'.
  115 lookupName :: MonadReader FPCfg m => Name -> m (Bool, Maybe ResolvedName)
  116 lookupName nm = do
  117   cfg <- ask
  118   pure ( Set.member (nameIndex nm) $ fpUninterp cfg
  119        , lookupVarIndexInMap (nameIndex nm) $ fpModuleMap cfg
  120        )
  121 
  122 byte :: Word8 -> BS.Builder
  123 byte = BS.word8
  124 
  125 -- | Basic serialization of small morsels.
  126 class Bytes a where
  127   bytes :: a -> BS.Builder
  128 instance Bytes Int where
  129   bytes = BS.int64BE . fromIntegral
  130 instance Bytes Natural where
  131   bytes = BS.lazyByteString . encode
  132 instance Bytes Text where
  133   bytes txt = bytes (BS.length bs) <> BS.byteString bs
  134     where
  135       bs = encodeUtf8 txt
  136 instance Bytes NameInfo where
  137   bytes = \case
  138     ModuleIdentifier i -> byte 0x1 <> bytes (QN.ppQualName $ moduleIdentToQualName i)
  139     ImportedName qn _  -> byte 0x2 <> bytes (QN.ppQualName qn)
  140 instance Bytes Sort where
  141   bytes = \case
  142     PropSort   -> byte 0x1
  143     TypeSort n -> byte 0x2 <> bytes n
  144 instance Bytes FirstOrderType where
  145   bytes = \case
  146     FOTBit       -> byte 0x1
  147     FOTInt       -> byte 0x2
  148     FOTRational  -> byte 0x3
  149     FOTIntMod n  -> byte 0x4 <> bytes n
  150     FOTVec n t   -> byte 0x5 <> bytes n <> bytes t
  151     FOTArray k v -> byte 0x6 <> bytes k <> bytes v
  152     FOTTuple ts  -> byte 0x7 <> bytes (length ts)
  153                              <> foldMap bytes ts
  154     FOTRec m     -> byte 0x8 <> bytes (Map.size m)
  155                              <> foldMap (\(k, v) -> bytes k <> byte 0 <> bytes v)
  156                                         (Map.toAscList m)
  157 
  158 fpTerms :: VarCtx -> [Term] -> FP BS.Builder
  159 fpTerms ctx ts = do
  160   bs <- mapM (fpTerm ctx) ts
  161   pure $ bytes (length ts) <> mconcat bs
  162 
  163 fpTerm :: VarCtx -> Term -> FP BS.Builder
  164 fpTerm ctx t = lookupTerm t >>= \case
  165   Just n  -> pure n
  166   Nothing -> do
  167     tag <- rememberTerm t
  168     body <- fpTermF ctx (unwrapTermF t)
  169     pure $ tag <> body
  170 
  171 fpTermF :: VarCtx -> TermF Term -> FP BS.Builder
  172 fpTermF ctx = \case
  173 
  174   App e1 e2 -> do
  175     b1 <- fpTerm ctx e1
  176     b2 <- fpTerm ctx e2
  177     pure $ byte 0x1 <> b1 <> b2
  178 
  179   Lambda x ty body -> do
  180     bty   <- fpTerm ctx ty
  181     bbody <- fpTerm (consVarCtx x ctx) body
  182     pure $ byte 0x2 <> bty <> bbody
  183 
  184   Pi x ty body -> do
  185     bty   <- fpTerm ctx ty
  186     bbody <- fpTerm (consVarCtx x ctx) body
  187     pure $ byte 0x3 <> bty <> bbody
  188 
  189   Constant nm -> do
  190     bdef <- fpDef nm
  191     pure $ byte 0x4 <> bdef
  192 
  193   Variable vn ty
  194     | Just dbi <- lookupVarCtx vn ctx -> -- Bound var: use index.
  195       pure $ byte 0x5 <> bytes dbi
  196     | otherwise -> do                    -- Free var: use name + type.
  197       bty <- fpTerm ctx ty
  198       pure $ byte 0x6 <> bytes (vnName vn) <> bty
  199 
  200   FTermF (Sort s _) -> -- ignoring SortFlags
  201     pure $ byte 0x7 <> bytes s
  202 
  203   FTermF (StringLit txt) ->
  204     pure $ byte 0x8 <> bytes txt
  205 
  206   FTermF (ArrayValue elemTy elems) -> do
  207     bty <- fpTerm ctx elemTy
  208     bs  <- fpTerms ctx $ V.toList elems
  209     pure $ byte 0x9 <> bty <> bs
  210 
  211   FTermF (Recursor cr) -> do
  212     dtB <- fpDef $ recursorDataType cr
  213     pure $ byte 0xa
  214         <> bytes (recursorSort cr)
  215         <> dtB
  216 
  217 fpDef :: Name -> FP BS.Builder
  218 fpDef nm = do
  219   lookupDef nm >>= \case
  220     Just v  -> pure v
  221     Nothing -> do
  222       tag  <- rememberDef nm
  223       body <- lookupName nm >>= \case
  224         (True, Just (ResolvedDef d)) -> do -- Uninterpreted.
  225           bty <- fpTerm emptyVarCtx $ defType d
  226           pure $ byte 0x11 <> ni <> bty
  227         (_, Just (ResolvedDef d))
  228           | Just b <- defBody d -> do
  229             bty   <- fpTerm emptyVarCtx $ defType d
  230             bbody <- fpTerm emptyVarCtx b
  231             pure $ byte 0x12 <> ni <> bty <> bbody
  232         (_, Just (ResolvedDef d)) -> do
  233           bty <- fpTerm emptyVarCtx $ defType d
  234           pure $ byte 0x13 <> ni <> bty
  235         (_, Just (ResolvedDataType dt)) -> do
  236           bty    <- fpTerm emptyVarCtx $ dtType dt
  237           bctors <- fpTerms emptyVarCtx $ ctorType <$> dtCtors dt
  238           pure $ byte 0x14 <> ni <> bty <> bctors
  239         (_, Just (ResolvedCtor c)) -> do
  240           bty <- fpTerm emptyVarCtx $ ctorType c
  241           pure $ byte 0x15 <> ni <> bty
  242         (_, Nothing) -> pure $ byte 0x16 <> ni
  243       pure $ tag <> body
  244   where
  245     ni :: BS.Builder
  246     ni = bytes $ nameInfo nm
  247 
  248 fpAssert :: VarCtx -> SATAssert -> FP BS.Builder
  249 fpAssert ctx = \case
  250   BoolAssert tm -> do
  251     b <- fpTerm ctx tm
  252     pure $ byte 0x21 <> b
  253   UniversalAssert vars hyps concl -> do
  254     let varTys = foldMap (bytes . snd) vars
  255         ctx'   = foldl' (\c (v, _) -> consVarCtx v c) ctx vars
  256     bhyps  <- fpTerms ctx' hyps
  257     bconcl <- fpTerm ctx' concl
  258     pure $ byte 0x22
  259         <> bytes (length vars)
  260         <> varTys
  261         <> bhyps
  262         <> bconcl

  264 -- | Produce a SHA256 fingerprint of a SATQuery. This fingerprint should be:
  265 --   * Stable across runs (has no VarIndex/TermIndex dependence).
  266 --   * Alpha-equivalence-respecting (uses de Bruijn indices for bound variables).
  267 --   * Definition-sensitive (transitively serializes constant bodies).
  268 --   * Relatively fast.
  269 fingerprintSATQuery :: ByteString -> ModuleMap -> SATQuery -> BS.ByteString
  270 fingerprintSATQuery prefix mm satq =
  271     SHA256.hashlazy $ prefix
  272                    <> singleton 0
  273                    <> BS.toLazyByteString (evalState (runReaderT satq' cfg) st0)
  274   where
  275     cfg :: FPCfg
  276     cfg = FPCfg mm $ satUninterp satq
  277 
  278     st0 :: FPSt
  279     st0 = FPSt mempty mempty 0
  280 
  281     vars :: [(VarName, FirstOrderType)]
  282     vars = Map.toAscList $ satVariables satq
  283 
  284     varTys :: BS.Builder
  285     varTys = foldMap (bytes . snd) vars
  286 
  287     queryCtx :: VarCtx
  288     queryCtx = foldl' (\c (v, _) -> consVarCtx v c) emptyVarCtx vars
  289 
  290     -- Note: asserts are not sorted here, so re-ordering the same
  291     -- asserts will produce a different fingerprint.
  292     satq' :: FP BS.Builder
  293     satq' = do
  294       asserts <- mapM (fpAssert queryCtx) $ satAsserts satq
  295       pure $ bytes (length vars) <> varTys <> mconcat asserts
