   51 -- Functor typeclass extensions {{{1
   52 
   53 class EqFoldable f where
   54   -- | Compares equality of arguments.
   55   fequal :: Eq x => f x -> f x -> Bool
   56 
   57 class EqFoldable f => OrdFoldable f where
   58   -- | Compares arguments.
   59   fcompare :: Ord x => f x -> f x -> Ordering
   60 
   61 class ShowFoldable f where
   62   fshow :: Show x => f x -> String
   63   fshow x = fshows x ""
   64   fshows :: Show x => f x -> ShowS
   65   fshows x = (fshow x ++)

   67 -- Term {{{1
   68 
   69 -- | A term with the given operator.
   70 newtype Term f = Term { unTerm :: f (Term f) }
   71 -- ^ FIXME: Name clash with saw-core Term type. Rename to something else.
   72 
   73 instance EqFoldable f => Eq (Term f) where
   74   Term x == Term y = x `fequal` y
   75 
   76 instance OrdFoldable f => Ord (Term f) where
   77   (Term x) `compare` (Term y) = x `fcompare` y
   78 
   79 instance ShowFoldable app => Show (Term app) where
   80   showsPrec _ (Term x) s = "Term " ++ fshows x s
   81 
   82 -- | Evaluate a term using the given monadic computation.
   83 evalTermM :: (Monad m, Traversable f) => (f v -> m v) -> Term f -> m v
   84 evalTermM fn (Term t) = fn =<< mapM (evalTermM fn) t
   85 
   86 foldTerm :: Functor f => (f r -> r) -> Term f -> r
   87 foldTerm f = go
   88   where go (Term x) = f (fmap go x)

   90 -- Class {{{1
   91 
   92 newtype Class = Class Integer
   93   deriving (Eq, Ord)

   95 -- ClassApp {{{1
   96 
   97 newtype ClassApp f = ClassApp (f Class)
   98 
   99 instance EqFoldable f => Eq (ClassApp f) where
  100   ClassApp x == ClassApp y = x `fequal` y
  101 
  102 instance OrdFoldable f => Ord (ClassApp f) where
  103   (ClassApp x) `compare` (ClassApp y) = x `fcompare` y

  105 -- CCSet {{{1
  106 
  107 data CCSet f = CC {
  108          -- | Number of classes added so far.
  109          ccCnt :: !Integer
  110          -- | Maps representative to equivalent terms that were added to set.
  111        , ccElementMap :: !(Map Class (Set (Term f)))
  112          -- | Maps classes to the representative.
  113        , ccRepMap :: !(Map Class Class)
  114          -- | Maps applications to the representative.
  115        , ccAppMap :: !(Map (ClassApp f) Class)
  116        }
  117 
  118 instance OrdFoldable f => Eq (CCSet f) where
  119   x == y = toSet x == toSet y
  120 
  121 instance OrdFoldable f => Ord (CCSet f) where
  122   x `compare` y = toSet x `compare` toSet y
  123 
  124 instance (OrdFoldable f, ShowFoldable f) => Show (CCSet f) where
  125   show cc = "fromList " ++ show (toList cc)
  126 
  127 -- | Return empty set.
  128 empty :: CCSet f
  129 empty = CC { ccCnt = 0
  130            , ccElementMap = Map.empty
  131            , ccRepMap = Map.empty
  132            , ccAppMap = Map.empty
  133            }
  134 
  135 -- | Returns all terms equivalent to term in set.
  136 -- N.B. The list may be empty if the term is not in the set.
  137 findEquivalent :: (OrdFoldable f, Traversable f)
  138                => Term f -> CCSet f -> Set (Term f)
  139 findEquivalent t cc =
  140   case getExistingClass cc t of
  141     Just cl -> Map.findWithDefault Set.empty cl (ccElementMap cc)
  142     Nothing -> Set.empty
  143 
  144 -- | Returns true if the two terms are considered equivalent.
  145 areEquivalent :: (OrdFoldable f, Traversable f)
  146               => Term f -> Term f -> CCSet f -> Bool
  147 areEquivalent x y cc =
  148   flip MTL.evalState cc $ do
  149     xCl <- getClass x
  150     yCl <- getClass y
  151     return (xCl == yCl)
  152 
  153 -- | Checks that all terms in the list are equivalent, and returns
  154 -- a pair of inequivalent terms if a counterexample is found.
  155 checkEquivalence :: (OrdFoldable f, Traversable f)
  156                  => [Term f] -> CCSet f -> Maybe (Term f, Term f)
  157 checkEquivalence (x:y:r) cc
  158   | areEquivalent x y cc = checkEquivalence (y:r) cc
  159   | otherwise = Just (x,y)
  160 checkEquivalence _ _ = Nothing
  161 
  162 -- | Insert term into set.
  163 insertTerm :: (OrdFoldable f, Traversable f) => Term f -> CCSet f -> CCSet f
  164 insertTerm = MTL.execState . getClass
  165 
  166 -- | Assert that two terms are equal.
  167 insertEquation :: (OrdFoldable f, Traversable f)
  168                => Term f -> Term f -> CCSet f -> CCSet f
  169 insertEquation x y cc =
  170   flip MTL.execState cc $ do
  171     xCl <- getClass x
  172     yCl <- getClass y
  173     MTL.modify $ processEquivalences [(xCl, yCl)]
  174 
  175 -- | Assert all elements in list are equal.
  176 insertEquivalenceClass :: (OrdFoldable f, Traversable f)
  177                        => [Term f] -> CCSet f -> CCSet f
  178 insertEquivalenceClass [] s = s
  179 insertEquivalenceClass [x] s = insertTerm x s
  180 insertEquivalenceClass (x:y:r) s =
  181   insertEquivalenceClass (y:r) $! insertEquation x y s
  182 
  183 -- | Take union of two equivalence classes.
  184 union :: (OrdFoldable f, Traversable f) => CCSet f -> CCSet f -> CCSet f
  185 union x y =
  186   foldl' (\s cl -> insertEquivalenceClass (Set.toList cl) s)
  187          x
  188          (Map.elems (ccElementMap y))
  189 
  190 fromSet :: (OrdFoldable f, Traversable f) => Set (Set (Term f)) -> CCSet f
  191 fromSet = Set.fold (insertEquivalenceClass . Set.toList) empty
  192 
  193 toSet :: OrdFoldable f => CCSet f -> Set (Set (Term f))
  194 toSet = Set.fromList . Map.elems . ccElementMap
  195 
  196 -- | Create a congruence closure set from the double list of terms,
  197 -- where each list of terms denotes an equivalence class.
  198 fromList :: (OrdFoldable f, Traversable f) => [[Term f]] -> CCSet f
  199 fromList = foldl' (flip insertEquivalenceClass) empty
  200 
  201 -- | Convert a congruence closure set into a double list of terms.
  202 toList :: OrdFoldable f => CCSet f -> [[Term f]]
  203 toList = map Set.toList . Set.toList . toSet
  204 
  205 -- CCSet implementation {{{1
  206 
  207 -- | Returns a class for term if one exists.
  208 getExistingClass :: (OrdFoldable f, Traversable f) => CCSet f -> Term f -> Maybe Class
  209 getExistingClass cc = evalTermM (\app -> Map.lookup (ClassApp app) (ccAppMap cc))
  210 
  211 type Pair t = (t,t)
  212 
  213 processEquivalences :: (Functor f, OrdFoldable f)
  214                     => [Pair Class] -> CCSet f -> CCSet f
  215 processEquivalences [] cc = cc
  216 processEquivalences ((x,y):rest) cc
  217   | xCl == yCl = processEquivalences rest cc
  218   | otherwise =
  219     processEquivalences
  220       ([ (l,r) | l NE.:| rl <- Map.elems appClassListMap, r <- rl ] ++ rest)
  221       (cc { ccElementMap = Map.fromListWith Set.union
  222                          $ map (\(cl,tl) -> (mapFn cl, tl))
  223                          $ Map.toList (ccElementMap cc)
  224           , ccRepMap = Map.insert xCl yCl $ Map.map mapFn repMap
  225           , ccAppMap = Map.map NE.head appClassListMap
  226           })
  227  where repMap = ccRepMap cc
  228        xCl = Map.findWithDefault x x repMap
  229        yCl = Map.findWithDefault y y repMap
  230        mapFn z = if z == xCl then yCl else z
  231        appClassListMap = Map.fromListWith (<>)
  232          [ (ClassApp (mapFn <$> app), mapFn cl NE.:| [])
  233          | (ClassApp app,cl) <- Map.toList (ccAppMap cc) ]
  234 
  235 type Comp f = MTL.State (CCSet f)
  236 
  237 -- | Add term to equivalence class.
  238 addTerm :: OrdFoldable f => Class -> Term f -> Comp f ()
  239 addTerm cl t = do
  240   MTL.modify $ \cc ->
  241     cc { ccElementMap
  242           = Map.insertWith Set.union cl (Set.singleton t)
  243           $ ccElementMap cc }
  244 
  245 getClass :: (OrdFoldable f, Traversable f) => Term f -> Comp f Class
  246 getClass (Term t) = do
  247   app <- ClassApp <$> mapM getClass t
  248   appMap <- MTL.gets ccAppMap
  249   case Map.lookup app appMap of
  250     Just cl -> cl <$ addTerm cl (Term t)
  251     Nothing -> do
  252       cc <- MTL.get
  253       let cnt = ccCnt cc
  254       let cl = Class cnt
  255       MTL.put cc { ccCnt = cnt + 1
  256                  , ccAppMap = Map.insert app cl (ccAppMap cc)
  257                  }
  258       cl <$ addTerm cl (Term t)
  259

  262 -- |
  263 splitClass :: (OrdFoldable f, Traversable f)
  264            => [Term f] -> Set (CCSet f) -> Set (CCSet f)
  265 splitClass [] sets = sets
  266 splitClass (h:l) sets = splitClass l (sets `Set.union` newSets)
  267  where insertEqualToH t = Set.map (insertEquation h t) sets
  268        newSets = Set.unions (map insertEqualToH l)
  269 
  270 -- | Returns all congruence-closed sets generated by the may alias
  271 -- configurations.
  272 mayAliases :: (OrdFoldable f, Traversable f)
  273            => [[Term f]] -> CCSet f -> Set (CCSet f)
  274 mayAliases l s = List.foldr splitClass (Set.singleton s) l
