   64 -- | A named field with the given What4 base type.
   65 data SequentialField tp = SequentialField
   66   { _sequentialFieldName :: Text
   67   , _sequentialFieldTypeRepr :: W4.BaseTypeRepr tp
   68   }
   69 makeLenses ''SequentialField
   70 
   71 -- | A typed record associating field names with What4 base types.
   72 data SequentialFields ctx = SequentialFields
   73   { _sequentialFields :: Ctx.Assignment SequentialField ctx
   74   , _sequentialFieldsIndex :: Map Text (Some (Ctx.Index ctx))
   75   }
   76 makeLenses ''SequentialFields
   77 
   78 -- | Convert a mapping from names to widths into a typed mapping from those names to What4 bitvectors of those widths.
   79 sequentialReprs ::
   80   forall m.
   81   MonadIO m =>
   82   Map Text Natural ->
   83   m (Some SequentialFields)
   84 sequentialReprs fs = do
   85   let assocs = Map.assocs fs
   86   Some fields <- go assocs
   87   idxs <- Ctx.traverseAndCollect (\idx _ -> pure [Some idx]) fields
   88   let index = zipWith (\(nm, _) idx -> (nm, idx)) assocs $ reverse idxs
   89   pure $ Some SequentialFields
   90     { _sequentialFields = fields
   91     , _sequentialFieldsIndex = Map.fromList index
   92     }
   93   where
   94     go :: [(Text, Natural)] -> m (Some (Ctx.Assignment SequentialField))
   95     go [] = pure $ Some Ctx.empty
   96     go ((nm, n):ns) = case someNat n of
   97       Just (Some nr) | Just LeqProof <- testLeq (knownNat @1) nr -> do
   98         let field = SequentialField
   99               { _sequentialFieldName = nm
  100               , _sequentialFieldTypeRepr = W4.BaseBVRepr nr
  101               }
  102         Some rest <- go ns
  103         pure $ Some $ Ctx.extend rest field
  104       _ -> throw $ YosysErrorInvalidStateFieldWidth nm

  106 -- | Given information about field names and types alongside an appropriately-typed What4 struct value,
  107 -- explode that struct into a mapping from field names to fresh typed SAWCore constants and SAWCore What4 simulator values.
  108 -- (This is used to unpack a What4 struct into a representation that is more convenient to manipulate in SAWCore.)
  109 ecBindingsOfFields ::
  110   MonadIO m =>
  111   W4.B.ExprBuilder n st fs ->
  112   SC.SharedContext ->
  113   Text {- ^ Prefix to prepend to all field names -} ->
  114   Map Text SC.Term {- ^ Mapping from field names to SAWCore types -} ->
  115   SequentialFields ctx {- ^ Mapping from field names to What4 base types -} ->
  116   W4.SymStruct (W4.B.ExprBuilder n st fs) ctx {- ^ What4 record to deconstruct -} ->
  117   m (Map Text (SC.ExtCns SC.Term, SimW4.SValue (W4.B.ExprBuilder n st fs)))
  118 ecBindingsOfFields sym sc pfx fs s inp = fmap Map.fromList . forM (Map.assocs fs) $ \(baseName, ty) -> do
  119   let nm = pfx <> baseName
  120   ec <- liftIO $ SC.scFreshEC sc nm ty
  121   val <- case s ^. sequentialFieldsIndex . at nm of
  122     Just (Some idx)
  123       | sf <- s ^. sequentialFields . ixF' idx
  124       , W4.BaseBVRepr _nr <- sf ^. sequentialFieldTypeRepr
  125         -> do
  126           inpExpr <- liftIO $ W4.structField sym inp idx
  127           pure . Sim.VWord $ W4.DBV inpExpr
  128     _ -> throw $ YosysErrorTransitionSystemMissingField nm
  129   pure (baseName, (ec, val))

  131 -- | Given a sequential circuit and a query, construct and write to disk a Sally transition system encoding that query.
  132 queryModelChecker ::
  133   MonadIO m =>
  134   W4.B.ExprBuilder n st fs ->
  135   SC.SharedContext ->
  136   YosysSequential ->
  137   FilePath {- ^ Path to write the resulting Sally input -} ->
  138   SC.TypedTerm {- ^ A boolean function of three parameters: an 8-bit cycle counter, a record of "fixed" inputs, and a record of circuit outputs -} ->
  139   Set.Set Text {- ^ Names of circuit inputs that are fixed -}->
  140   m ()
  141 queryModelChecker sym sc sequential path query fixedInputs = do
  142   -- there are 5 classes of field:
  143   --  - fixed inputs (inputs from the circuit named in the fixed set, assumed to be constant across cycles
  144   --  - variable inputs (all other inputs from the circuit)
  145   --  - outputs (all outputs from the circuit)
  146   --  - state (all circuit flip-flop states)
  147   --  - internal (right now, just a cycle counter)
  148   let (fixedInputWidths, variableInputWidths) = Map.partitionWithKey (\nm _ -> Set.member nm fixedInputs) $ sequential ^. yosysSequentialInputWidths
  149   let (fixedInputFields, variableInputFields) = Map.partitionWithKey (\nm _ -> Set.member nm fixedInputs) $ sequential ^. yosysSequentialInputFields
  150   let internalWidths = Map.singleton "cycle" 8
  151   internalFields <- forM internalWidths $ \w -> liftIO $ SC.scBitvector sc w
  152 
  153   -- the "inputs" for our transition system are exclusively the circuit's variable inputs
  154   Some inputFields <- sequentialReprs variableInputWidths
  155   let inputReprs = TraversableFC.fmapFC (view sequentialFieldTypeRepr) $ inputFields ^. sequentialFields
  156   let inputNames = TraversableFC.fmapFC (Const . W4.safeSymbol . Text.unpack . view sequentialFieldName) $ inputFields ^. sequentialFields
  157 
  158   -- while the transition system "states" are everything else: flip-flop states, fixed inputs, all outputs, and the cycle counter
  159   let combinedWidths = Map.unions
  160         [ sequential ^. yosysSequentialStateWidths
  161         , Map.mapKeys ("stateinput_"<>) fixedInputWidths
  162         , Map.mapKeys ("stateoutput_"<>) $ sequential ^. yosysSequentialOutputWidths
  163         , Map.mapKeys ("internal_"<>) internalWidths
  164         ]
  165   Some stateFields <- sequentialReprs combinedWidths
  166   let stateReprs = TraversableFC.fmapFC (view sequentialFieldTypeRepr) $ stateFields ^. sequentialFields
  167   let stateNames = TraversableFC.fmapFC (Const . W4.safeSymbol . Text.unpack . view sequentialFieldName) $ stateFields ^. sequentialFields
  168   let ts = W4.TransitionSystem
  169         { W4.inputReprs = inputReprs
  170         , W4.inputNames = inputNames
  171         , W4.stateReprs = stateReprs
  172         , W4.stateNames = stateNames
  173         , W4.initialStatePredicate = \cur -> do
  174             -- initially , we assert that cycle = 0
  175             curInternalBindings <- ecBindingsOfFields sym sc "internal_" internalFields stateFields cur
  176             cycleVal <- case Map.lookup "cycle" curInternalBindings of
  177               Just (ec, _) -> SC.scExtCns sc ec
  178               Nothing -> throw $ YosysErrorTransitionSystemMissingField "cycle"
  179             zero <- SC.scBvConst sc 8 0
  180             wnat <- SC.scNat sc 8
  181             cyclePred <- SC.scBvEq sc wnat cycleVal zero
  182             ref <- IORef.newIORef Map.empty
  183             let args = Map.unions $ fmap (Map.fromList . fmap (\(ec, x) -> (SC.ecVarIndex ec, x)) . Map.elems)
  184                   [ curInternalBindings
  185                   ]
  186             sval <- SimW4.w4SolveBasic sym sc Map.empty args ref Set.empty cyclePred
  187             case sval of
  188               Sim.VBool b -> pure b
  189               _ -> throw YosysErrorTransitionSystemBadType
  190         , W4.stateTransitions = \input cur next -> do
  191             -- there is exactly one state transition, defined by the SAWCore function f representing the circuit
  192             -- here, we assert that:
  193             --  - the new value of each state field is equal to the same field in f(<previous state + all inputs>)
  194             --  - the new value of each output is equal to the same output in f(<previous state + all inputs>)
  195             --  - the new value of each fixed input is equal to the old value of that fixed input
  196             --  - the new cycle counter is equal to the old cycle counter plus one
  197             inputBindings <- ecBindingsOfFields sym sc "" (fst <$> variableInputFields) inputFields input
  198             curBindings <- ecBindingsOfFields sym sc "" (fst <$> (sequential ^. yosysSequentialStateFields)) stateFields cur
  199             curFixedInputBindings <- ecBindingsOfFields sym sc "stateinput_" (fst <$> fixedInputFields) stateFields cur
  200             curInternalBindings <- ecBindingsOfFields sym sc "internal_" internalFields stateFields cur
  201             nextBindings <- ecBindingsOfFields sym sc "" (fst <$> (sequential ^. yosysSequentialStateFields)) stateFields next
  202             nextFixedInputBindings <- ecBindingsOfFields sym sc "stateinput_" (fst <$> fixedInputFields) stateFields next
  203             nextOutputBindings <- ecBindingsOfFields sym sc "stateoutput_" (fst <$> (sequential ^. yosysSequentialOutputFields)) stateFields next
  204             nextInternalBindings <- ecBindingsOfFields sym sc "internal_" internalFields stateFields next
  205             inps <- fmap Map.fromList . forM (Map.assocs $ sequential ^. yosysSequentialInputWidths) $ \(nm, _) ->
  206               let bindings = if Set.member nm fixedInputs then curFixedInputBindings else inputBindings
  207               in case Map.lookup nm bindings of
  208                 Just (ec, _) -> (nm,) <$> SC.scExtCns sc ec
  209                 Nothing -> throw $ YosysErrorTransitionSystemMissingField nm
  210             states <- forM curBindings $ \(ec, _) -> SC.scExtCns sc ec
  211             inpst <- cryptolRecord sc states
  212             domainRec <- cryptolRecord sc $ Map.insert "__state__" inpst inps
  213             codomainRec <- liftIO $ SC.scApply sc (sequential ^. yosysSequentialTerm . SC.ttTermLens) domainRec
  214             codomainFields <- insertStateField sc (sequential ^. yosysSequentialStateFields) $ sequential ^. yosysSequentialOutputFields
  215             outst <- cryptolRecordSelect sc codomainFields codomainRec "__state__"
  216             stPreds <- forM (Map.assocs $ sequential ^. yosysSequentialStateWidths) $ \(nm, w) -> do
  217               val <- cryptolRecordSelect sc (sequential ^. yosysSequentialStateFields) outst nm
  218               wnat <- SC.scNat sc w
  219               new <- case Map.lookup nm nextBindings of
  220                 Just (ec, _) -> SC.scExtCns sc ec
  221                 Nothing -> throw $ YosysErrorTransitionSystemMissingField nm
  222               liftIO $ SC.scBvEq sc wnat new val
  223             outputPreds <- forM (Map.assocs $ sequential ^. yosysSequentialOutputWidths) $ \(nm, w) -> do
  224               val <- cryptolRecordSelect sc codomainFields codomainRec nm
  225               wnat <- SC.scNat sc w
  226               new <- case Map.lookup nm nextOutputBindings of
  227                 Just (ec, _) -> SC.scExtCns sc ec
  228                 Nothing -> throw $ YosysErrorTransitionSystemMissingField nm
  229               liftIO $ SC.scBvEq sc wnat new val
  230             fixedInputPreds <- forM (Map.assocs fixedInputWidths) $ \(nm, w) -> do
  231               wnat <- SC.scNat sc w
  232               val <- case Map.lookup nm curFixedInputBindings of
  233                 Just (ec, _) -> SC.scExtCns sc ec
  234                 Nothing -> throw $ YosysErrorTransitionSystemMissingField nm
  235               new <- case Map.lookup nm nextFixedInputBindings of
  236                 Just (ec, _) -> SC.scExtCns sc ec
  237                 Nothing -> throw $ YosysErrorTransitionSystemMissingField nm
  238               liftIO $ SC.scBvEq sc wnat new val
  239             cycleIncrement <- do
  240               wnat <- SC.scNat sc 8
  241               val <- case Map.lookup "cycle" curInternalBindings of
  242                 Just (ec, _) -> SC.scExtCns sc ec
  243                 Nothing -> throw $ YosysErrorTransitionSystemMissingField "cycle"
  244               one <- SC.scBvConst sc 8 1
  245               incremented <- SC.scBvAdd sc wnat val one
  246               new <- case Map.lookup "cycle" nextInternalBindings of
  247                 Just (ec, _) -> SC.scExtCns sc ec
  248                 Nothing -> throw $ YosysErrorTransitionSystemMissingField "cycle"
  249               liftIO $ SC.scBvEq sc wnat new incremented
  250             identity <- SC.scBool sc True
  251             conj <- foldM (SC.scAnd sc) identity $ stPreds <> outputPreds <> fixedInputPreds <> [cycleIncrement]
  252             ref <- IORef.newIORef Map.empty
  253             let args = Map.unions $ fmap (Map.fromList . fmap (\(ec, x) -> (SC.ecVarIndex ec, x)) . Map.elems)
  254                   [ inputBindings
  255                   , curBindings
  256                   , curFixedInputBindings
  257                   , curInternalBindings
  258                   , nextBindings
  259                   , nextOutputBindings
  260                   , nextFixedInputBindings
  261                   , nextInternalBindings
  262                   ]
  263             sval <- SimW4.w4SolveBasic sym sc Map.empty args ref Set.empty conj
  264             w4Conj <- case sval of
  265               Sim.VBool b -> pure b
  266               _ -> throw YosysErrorTransitionSystemBadType
  267             pure
  268               [ (W4.systemSymbol "default!", w4Conj)
  269               ]
  270         , W4.queries = \cur -> do
  271             -- here we generate a single query, corresponding to the provided query term q
  272             -- this is q applied to:
  273             --  - the cycle counter (an 8-bit bitvector)
  274             --  - a record of the fixed inputs (as usual really a SAWCore tuple, ordered per the Cryptol record type)
  275             --  - a record of the outputs
  276             curFixedInputBindings <- ecBindingsOfFields sym sc "stateinput_" (fst <$> fixedInputFields) stateFields cur
  277             curOutputBindings <- ecBindingsOfFields sym sc "stateoutput_" (fst <$> (sequential ^. yosysSequentialOutputFields)) stateFields cur
  278             curInternalBindings <- ecBindingsOfFields sym sc "internal_" internalFields stateFields cur
  279             fixedInps <- fmap Map.fromList . forM (Map.assocs fixedInputWidths) $ \(nm, _) ->
  280               case Map.lookup nm curFixedInputBindings of
  281                 Nothing -> throw $ YosysErrorTransitionSystemMissingField nm
  282                 Just (ec, _) -> (nm,) <$> SC.scExtCns sc ec
  283             outputs <- fmap Map.fromList . forM (Map.assocs $ sequential ^. yosysSequentialOutputWidths) $ \(nm, _) ->
  284               case Map.lookup nm curOutputBindings of
  285                 Nothing -> throw $ YosysErrorTransitionSystemMissingField nm
  286                 Just (ec, _) -> (nm,) <$> SC.scExtCns sc ec
  287             cycleVal <- case Map.lookup "cycle" curInternalBindings of
  288               Just (ec, _) -> SC.scExtCns sc ec
  289               Nothing -> throw $ YosysErrorTransitionSystemMissingField "cycle"
  290             fixedInputRec <- cryptolRecord sc fixedInps
  291             outputRec <- cryptolRecord sc outputs
  292             result <- liftIO $ SC.scApplyAll sc (query ^. SC.ttTermLens) [cycleVal, fixedInputRec, outputRec]
  293             ref <- IORef.newIORef Map.empty
  294             let args = Map.unions $ fmap (Map.fromList . fmap (\(ec, x) -> (SC.ecVarIndex ec, x)) . Map.elems)
  295                   [ curOutputBindings
  296                   , curFixedInputBindings
  297                   , curInternalBindings
  298                   ]
  299             sval <- SimW4.w4SolveBasic sym sc Map.empty args ref Set.empty result
  300             w4Pred <- case sval of
  301               Sim.VBool b -> pure b
  302               _ -> throw . YosysError $ "Invalid type when converting predicate to What4: " <> Text.pack (show sval)
  303             pure [w4Pred]
  304         }
  305   sts <- liftIO $ Sally.exportTransitionSystem sym Sally.mySallyNames ts
  306   sexp <- liftIO $ Sally.sexpOfSally sym sts
  307   liftIO . BS.writeFile path . encodeUtf8 . Text.pack . show $ Sally.sexpToDoc sexp
  308   pure ()
