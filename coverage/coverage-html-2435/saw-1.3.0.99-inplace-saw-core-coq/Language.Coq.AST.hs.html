   10 module Language.Coq.AST where
   11 
   12 import Data.String (IsString(..))
   13 
   14 newtype Ident = Ident String
   15   deriving (Eq, Ord)
   16 
   17 instance Show Ident where
   18   show (Ident s) = show s
   19 
   20 instance IsString Ident where
   21   fromString s = Ident s
   22 
   23 newtype ModuleName = ModuleName String
   24   deriving (Eq, Ord)
   25 
   26 instance Show ModuleName where
   27   show (ModuleName s) = show s
   28 
   29 data Sort
   30   = Prop
   31   | Set
   32   | Type
   33   deriving (Show)
   34 
   35 data Term
   36   = Lambda [Binder] Term
   37   | Fix Ident [Binder] Term Term
   38   | Pi [PiBinder] Term
   39   | Let Ident [Binder] (Maybe Type) Term Term
   40   | If Term Term Term
   41   | App Term [Term]
   42   | Sort Sort
   43   | Var Ident
   44     -- | A variable that needs to be printed with a leading at sign in order to
   45     -- make all arguments explicit
   46   | ExplVar Ident
   47     -- | A ascription @tm : tp@ of a type to a term
   48   | Ascription Term Term
   49   | NatLit Integer
   50   | ZLit Integer
   51   | List [Term]
   52   | StringLit String
   53   | Scope Term String
   54   | Ltac String
   55   deriving (Show)
   56 
   57 -- | Type synonym useful for indicating when a term is used as a type.
   58 type Type = Term
   59 
   60 -- | An 'Ident' with an optional 'Type', which may be explicit or implicit.
   61 -- For use representing the bound variables in 'Lambda's, 'Let's, etc.
   62 data Binder
   63   = Binder Ident (Maybe Type)
   64   | ImplicitBinder Ident (Maybe Type)
   65     deriving (Show)
   66 
   67 -- | An 'Type' with an optional 'Ident', which may be explicit or implicit.
   68 -- For use representing arguments in 'Pi' types.
   69 data PiBinder
   70   = PiBinder (Maybe Ident) Type
   71   | PiImplicitBinder (Maybe Ident) Type
   72     deriving (Show)
   73 
   74 -- Because saw-core does not give very helpful access to the parameters and
   75 -- indices, we just follow their style and define the constructor by its fully
   76 -- applied return type.
   77 data Constructor = Constructor
   78   { constructorName    :: Ident
   79   , constructorType    :: Term
   80   }
   81   deriving (Show)
   82 
   83 data Inductive = Inductive
   84   { inductiveName         :: Ident
   85   , inductiveParameters   :: [Binder]
   86   , inductiveIndices      :: [PiBinder]
   87   , inductiveSort         :: Sort
   88   , inductiveConstructors :: [Constructor]
   89   }
   90   deriving (Show)
   91 
   92 data Decl
   93   = Axiom Ident Type
   94   | Comment String
   95   | Definition Ident [Binder] (Maybe Type) Term
   96   | Parameter Ident Type
   97   | Variable Ident Type
   98   | InductiveDecl Inductive
   99   | Section Ident [Decl]
  100   | Snippet String
  101   deriving (Show)
