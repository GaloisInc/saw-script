   10 module Language.Coq.AST where
   11 
   12 import Data.String (IsString(..))
   13 
   14 newtype Ident = Ident String
   15   deriving (Eq, Ord)
   16 
   17 instance Show Ident where
   18   show (Ident s) = show s
   19 
   20 instance IsString Ident where
   21   fromString s = Ident s
   22 
   23 data Sort
   24   = Prop
   25   | Set
   26   | Type
   27   deriving (Show)
   28 
   29 data Term
   30   = Lambda [Binder] Term
   31   | Fix Ident [Binder] Term Term
   32   | Pi [PiBinder] Term
   33   | Let Ident [Binder] (Maybe Type) Term Term
   34   | If Term Term Term
   35   | App Term [Term]
   36   | Sort Sort
   37   | Var Ident
   38     -- | A variable that needs to be printed with a leading at sign in order to
   39     -- make all arguments explicit
   40   | ExplVar Ident
   41     -- | A ascription @tm : tp@ of a type to a term
   42   | Ascription Term Term
   43   | NatLit Integer
   44   | ZLit Integer
   45   | List [Term]
   46   | StringLit String
   47   | Scope Term String
   48   | Ltac String
   49   deriving (Show)
   50 
   51 -- | Type synonym useful for indicating when a term is used as a type.
   52 type Type = Term
   53 
   54 -- | An 'Ident' with an optional 'Type', which may be explicit or implicit.
   55 -- For use representing the bound variables in 'Lambda's, 'Let's, etc.
   56 data Binder
   57   = Binder Ident (Maybe Type)
   58   | ImplicitBinder Ident (Maybe Type)
   59     deriving (Show)
   60 
   61 -- | An 'Type' with an optional 'Ident', which may be explicit or implicit.
   62 -- For use representing arguments in 'Pi' types.
   63 data PiBinder
   64   = PiBinder (Maybe Ident) Type
   65   | PiImplicitBinder (Maybe Ident) Type
   66     deriving (Show)
   67 
   68 -- Because saw-core does not give very helpful access to the parameters and
   69 -- indices, we just follow their style and define the constructor by its fully
   70 -- applied return type.
   71 data Constructor = Constructor
   72   { constructorName    :: Ident
   73   , constructorType    :: Term
   74   }
   75   deriving (Show)
   76 
   77 data Inductive = Inductive
   78   { inductiveName         :: Ident
   79   , inductiveParameters   :: [Binder]
   80   , inductiveIndices      :: [PiBinder]
   81   , inductiveSort         :: Sort
   82   , inductiveConstructors :: [Constructor]
   83   }
   84   deriving (Show)
   85 
   86 data Decl
   87   = Axiom Ident Type
   88   | Comment String
   89   | Definition Ident [Binder] (Maybe Type) Term
   90   | Parameter Ident Type
   91   | Variable Ident Type
   92   | InductiveDecl Inductive
   93   | Section Ident [Decl]
   94   | Snippet String
   95   deriving (Show)
