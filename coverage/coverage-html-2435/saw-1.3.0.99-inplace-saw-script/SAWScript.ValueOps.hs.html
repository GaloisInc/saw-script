   79 isVUnit :: Value -> Bool
   80 isVUnit (VTuple []) = True
   81 isVUnit _ = False

   83 indexValue :: SS.Pos -> Value -> Value -> Value
   84 indexValue pos (VArray vs) (VInteger x)
   85     | i < length vs = vs !! i
   86     | otherwise = error $ show pos ++ ": Array index out of bounds"
   87     where i = fromInteger x
   88 indexValue pos v1 v2 =
   89     panic "indexValue" [
   90         "Type error that escaped the typechecker",
   91         "Source position: " <> Text.pack (show pos),
   92         "Array value: " <> Text.pack (show v1),
   93         "Index value: " <> Text.pack (show v2)
   94     ]
   95 
   96 lookupValue :: SS.Pos -> Value -> Text -> Value
   97 lookupValue pos (VRecord vm) name =
   98     case M.lookup name vm of
   99       Nothing -> error $ show pos ++ ": No such record field: " ++ Text.unpack name
  100       Just x -> x
  101 lookupValue pos v1 v2 =
  102     panic "lookupValue" [
  103         "Type error that escaped the typechecker",
  104         "Source position: " <> Text.pack (show pos),
  105         "Array value: " <> Text.pack (show v1),
  106         "Index value: " <> Text.pack (show v2)
  107     ]
  108 
  109 tupleLookupValue :: SS.Pos -> Value -> Integer -> Value
  110 tupleLookupValue pos (VTuple vs) i
  111   | 0 <= i && fromIntegral i < length vs = vs !! fromIntegral i
  112   | otherwise = error $ show pos ++ ": No such tuple index: " ++ show i
  113 tupleLookupValue pos v1 v2 =
  114     panic "tupleLookupValue" [
  115         "Type error that escaped the typechecker",
  116         "Source position: " <> Text.pack (show pos),
  117         "Array value: " <> Text.pack (show v1),
  118         "Index value: " <> Text.pack (show v2)
  119     ]

  127 -- | A version of 'Control.Exception.bracket' specialized to 'TopLevel'. We
  128 -- can't use 'Control.Monad.Catch.bracket' because it requires 'TopLevel' to
  129 -- implement 'Control.Monad.Catch.MonadMask', which it can't do.
  130 bracketTopLevel :: TopLevel a -> (a -> TopLevel b) -> (a -> TopLevel c) -> TopLevel c
  131 bracketTopLevel acquire release action =
  132   do  resource <- acquire
  133       try (action resource) >>= \case
  134         Left (bad :: X.SomeException) -> release resource >> throwM bad
  135         Right good -> release resource >> pure good

  137 combineRW :: TopLevelCheckpoint -> TopLevelRW -> IO TopLevelRW
  138 combineRW (TopLevelCheckpoint chk scc) rw =
  139   do cenv' <- CEnv.combineCryptolEnv (rwCryptol chk) (rwCryptol rw)
  140      sc' <- restoreSharedContext scc (rwSharedContext rw)
  141      return chk{ rwCryptol = cenv'
  142                , rwSharedContext = sc'
  143                }

  145 -- | Represents the mutable state of the TopLevel monad
  146 --   that can later be restored.
  147 data TopLevelCheckpoint =
  148   TopLevelCheckpoint
  149     TopLevelRW
  150     SharedContextCheckpoint
  151 
  152 makeCheckpoint :: TopLevelRW -> IO TopLevelCheckpoint
  153 makeCheckpoint rw =
  154   do scc <- checkpointSharedContext (rwSharedContext rw)
  155      return (TopLevelCheckpoint rw scc)
  156 
  157 restoreCheckpoint :: TopLevelCheckpoint -> TopLevel ()
  158 restoreCheckpoint chk =
  159   do rw <- getTopLevelRW
  160      rw' <- io (combineRW chk rw)
  161      putTopLevelRW rw'
  162 
  163 -- | Capture the current state of the TopLevel monad
  164 --   and return an action that, if invoked, resets
  165 --   the state back to that point.
  166 checkpoint :: TopLevel (() -> TopLevel ())
  167 checkpoint = TopLevel_ $
  168   do chk <- liftIO . makeCheckpoint =<< get
  169      return $ \_ ->
  170        do printOutLnTop Info "Restoring state from checkpoint"
  171           restoreCheckpoint chk
  172 
  173 -- | Capture the current proof state and return an
  174 --   action that, if invoked, resets the state back to that point.
  175 proof_checkpoint :: ProofScript (() -> ProofScript ())
  176 proof_checkpoint =
  177   do ps <- get
  178      return $ \_ ->
  179        do scriptTopLevel (printOutLnTop Info "Restoring proof state from checkpoint")
  180           put ps

  182 withLocalEnv :: LocalEnv -> TopLevel a -> TopLevel a
  183 withLocalEnv env m = do
  184   prevEnv <- gets rwLocalEnv
  185   modify (\rw -> rw { rwLocalEnv = env })
  186   result <- m
  187   modify (\rw -> rw { rwLocalEnv = prevEnv })
  188   return result
