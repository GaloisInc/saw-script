   24 -- | Parse an LLVM module into a list of declarations
   25 --   Yields an Interaction so that we can talk to the user about what went wrong
   26 getDeclsLLVM ::
   27   (AIG.IsAIG l g) =>
   28   AIG.Proxy l g ->
   29   SharedContext ->
   30   LLVMModule arch ->
   31   IO (Interaction (Maybe [Decl]))
   32 getDeclsLLVM _proxy _sc lm =
   33     return $ do
   34       let (untranslateable, translations) =
   35             partitionEithers . for (modDefines (modAST lm)) $ \def ->
   36                maybe (Left (symStr (defName def))) Right $
   37                   symDefineToDecl def
   38 
   39       when (not . null $ untranslateable) $ do
   40          separator ThinSep
   41          liftF . flip Warning () $ "No translation for the following signatures in " ++ modFilePath lm ++ ":"
   42          bulleted $ map (\u -> Text.unpack ("'" <> u <> "'")) untranslateable
   43 
   44       return $ Just translations
   45 
   46    where
   47       symStr (Symbol s) = Text.pack s
   48 
   49       symDefineToDecl symDefine =
   50          let name = symStr $ defName symDefine
   51              tidName (Typed _ (Ident n)) = Text.pack n
   52              args = mapM (\tid -> Arg (tidName tid) <$> memTypeToStdType (typedType tid)) $ defArgs symDefine
   53              retType = memTypeToStdType (defRetType symDefine)
   54          in Decl name <$> retType <*> args
   55 
   56       memTypeToStdType t = case t of
   57          PrimType (Integer 8)  -> Just $ bitSeqType 8
   58          PrimType (Integer 16) -> Just $ bitSeqType 16
   59          PrimType (Integer 32) -> Just $ bitSeqType 32
   60          PrimType (Integer 64) -> Just $ bitSeqType 64
   61          _ -> Nothing
