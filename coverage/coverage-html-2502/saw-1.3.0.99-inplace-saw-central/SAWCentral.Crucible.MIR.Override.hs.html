  129 -- | When a specification is used as a composition override, this function
  130 -- checks that the postconditions of the specification fully specify (via
  131 -- @mir_points_to@ statements) the values of all local mutable allocations
  132 -- (which are declared in the preconditions via @mir_alloc_mut@) and all
  133 -- mutable static items. If not, this function will raise an appropriate error
  134 -- message. See @Note [MIR compositional verification and mutable allocations]@.
  135 checkMutableAllocPostconds ::
  136   Options ->
  137   SharedContext ->
  138   MIRCrucibleContext ->
  139   CrucibleMethodSpecIR ->
  140   OverrideMatcher MIR md ()
  141 checkMutableAllocPostconds opts sc cc cs = do
  142   sub <- use setupValueSub
  143 
  144   -- Gather all of the references used in `mir_points_to` statements in the
  145   -- postconditions. This corresponds to step (1) of the plan in
  146   -- Note [MIR compositional verification and mutable allocations]
  147   postRefs <- Set.fromList <$>
  148     traverse
  149       (\(MirPointsTo _cond ref _val) -> do
  150         MIRVal refShp refVal <-
  151           resolveSetupValueMIR opts cc sc cs ref
  152         case testRefShape refShp of
  153           Just IsRefShape{} ->
  154             pure $ MirReferenceMuxConcrete refVal
  155           Nothing ->
  156             panic "checkMutableAllocPostconds" [
  157                 "Unexpected non-reference type:",
  158                 "   " <> Text.pack (show $ PP.pretty $ shapeMirTy refShp)
  159             ])
  160       (cs ^. MS.csPostState ^. MS.csPointsTos)
  161 
  162   -- Check if a `mir_alloc_mut`-computed allocation isn't used in a
  163   -- `mir_points_to` statement in the postconditions, and if so, error.
  164   -- This corresponds to step (2) of the plan in
  165   -- Note [MIR compositional verification and mutable allocations]
  166   let mutAllocSpecs =
  167         Map.filter (\(Some mas) -> view isMut mas) $
  168         cs ^. MS.csPreState . MS.csAllocs
  169   let mutAllocRefs =
  170         map
  171           (\(Some mp, Some spec) ->
  172             ( MirReferenceMuxConcrete $ mp ^. mpRef
  173             , spec ^. maConditionMetadata
  174             ))
  175           (Map.elems (Map.intersectionWith (,) sub mutAllocSpecs))
  176   F.for_ mutAllocRefs $ \(mutAllocRef, cond) ->
  177     unless (Set.member mutAllocRef postRefs) $
  178       fail $ underspecified_mut_alloc_err cond
  179 
  180   -- Check if a mutable static isn't used in a `mir_points_to` statement in the
  181   -- postconditions, and if so, error.
  182   -- This corresponds to step (3) of the plan in
  183   -- Note [MIR compositional verification and mutable allocations]
  184   let mutStatics =
  185         Map.filter (view Mir.sMutable) $
  186         col ^. Mir.statics
  187   let mutStaticRefs =
  188         map
  189         (\(mutStaticDid, (_, Mir.StaticVar gv)) ->
  190           ( mutStaticDid
  191           , MirReferenceMuxConcrete $ staticRefMux sym gv
  192           ))
  193         (Map.toList
  194           (Map.intersectionWith
  195             (,) mutStatics (colState ^. Mir.staticMap)))
  196   F.for_ mutStaticRefs $ \(mutStaticDid, mutStaticRef) ->
  197     unless (Set.member mutStaticRef postRefs) $
  198       fail $ underspecified_mut_static_err mutStaticDid
  199   where
  200     sym      = cc ^. mccSym
  201     colState = cc ^. mccRustModule . Mir.rmCS
  202     col      = colState ^. Mir.collection

  237 -- value has exactly one leaf with a 'W4.Pred' that concretely evaluates to
  238 -- 'W4.truePred'. If this is not the case, this function will panic.

  408 {-
  409 Note [MIR compositional verification and mutable allocations]
  410 ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
  411 When using compositional verification in the SAW MIR backend, we want to ensure
  412 that any specifications that are used as compositional overrides properly
  413 specify the values of local mutable allocations (i.e., things allocated with
  414 mir_alloc_mut) and mutable static items (i.e., things declared with `static
  415 mut`) in the postconditions of the specs. (See the "Compositional Verification"
  416 section of the SAW manual for more discussion on why this is important.) This
  417 is something that the SAW LLVM backend also must do, but the MIR backend
  418 differs from how it accomplishes this in a number of key ways.
  419 
  420 The main entrypoint to this check is the `checkMutableAllocPostconds` function,
  421 which is roughly the MIR counterpart to the LLVM backend's
  422 `SAWCentral.Crucible.LLVM.Override.invalidateMutableAllocs` function. The LLVM
  423 function takes the approach of invalidating the underlying memory of
  424 underspecified mutable allocations or global variables such that if they are
  425 read from later, then a simulation-time error is thrown. The MIR memory model,
  426 on the other hand, does not have a notion of memory invalidation that we can
  427 readily use (see https://github.com/GaloisInc/crucible/issues/1109), so the SAW
  428 MIR backend requires a slightly different approach.
  429 
  430 The SAW MIR backend instead implements a stricter check: any specification used
  431 as a compositional override *must* specify the values of all local mutable
  432 allocations and the values of all mutable static items in the postconditions.
  433 There are no exceptions to this rule: if you do not specify one, then SAW will
  434 throw an error before simulation begins. This differs from the LLVM backend in
  435 that you can get away with not specifying the value of a mutable allocation or
  436 global variable in an overrides postconditions, provided that that value is
  437 never read from during simulation.
  438 
  439 In order to implement the necessary checks in the `checkMutableAllocPostconds`
  440 function, we employ the following plan:
  441 
  442 1. Compute all of the MirReferenceMux values corresponding to each
  443    mir_points_to statement in the postconditions of the compositional override
  444    and put them into a Set.
  445 
  446 2. Gather all of the MirReferenceMux values corresponding to local mutable
  447    references that were allocated in the preconditions. If one of these
  448    values is not contained in the Set of things produced in step (1),
  449    then throw an error.
  450 
  451 3. Gather all of the MirReferenceMux values corresponding to mutable static
  452    items that might be used in the program. If one of these values is not
  453    contained in the Set of things produced in step (1), then throw an error.
  454 
  455 Using a Set turns this from an O(n) operation to an O(log n) one, which can be
  456 important for specifications that have lots of mutable allocations or mutable
  457 static items.
  458 
  459 There is one wrinkle not mentioned in the plan above: how exactly do you put
  460 MirReferenceMux values into a Set? Doing so would require MirReferenceMux to
  461 have an Ord instance in order to use Set operations, and MirReferenceMux has no
  462 such instance. Indeed, it's not clear how to define such an instance:
  463 MirReferenceMux values can contain symbolic information in the general case,
  464 which makes it tricky to return a definite True-or-False answer regarding
  465 whether two values are equal.
  466 
  467 Thankfully, it happens to be the case that all MirReferenceMux values that we
  468 check in `checkMutableAllocPostconds` are concrete. Therefore, we can compare
  469 MirReferenceMux values by concretizing the symbolic information whenever
  470 necessary. The cmp*Concretely family of functions implements these concrete
  471 comparisons. More precisely:
  472 
  473 * The `cmpRefMuxConcretely` function checks that the FancyMuxTree in a
  474   MirReferenceMux consists of exactly one leaf with a Pred that concretely
  475   evaluates to truePred. (This must be the case if the FancyMuxTree is
  476   concrete.) The function then takes the MirReference values from the leaf
  477   of each FancyMuxTree and compares them using `cmpRefConcretely`.
  478 
  479 * The `cmpRefConcretely` function is very close to the OrdSkel instance for
  480   MirReference, except that it concretizes SymBV values as necessary using
  481   the `cmpSymBVConcretely` function.
  482 
  483 * We create a MirReferenceMuxConcrete newtype around MirReferenceMux, and
  484   we give MirReferenceMuxConcrete an Ord instance defined in terms of
  485   `cmpRefMuxConcretely`. We then put MirReferenceMuxConcrete values into
  486   the Set in step (1) of the plan above.
  487 
  488 If there is symbolic information at any point in this process, then a panic is
  489 thrown. If we ever add the ability to create symbolic MirReferenceMux values in
  490 a specification (e.g., via a MIR equivalent of the llvm_fresh_pointer command),
  491 then we will need to rethink this plan.
  492 -}

  494 computeReturnValue ::
  495   Options                     {- ^ saw script debug and print options     -} ->
  496   MIRCrucibleContext          {- ^ context of the crucible simulation     -} ->

  524 -- | Generate assertions that all of the memory allocations matched by
  525 -- an override's precondition are disjoint.
  526 enforceDisjointness ::
  527   MIRCrucibleContext -> W4.ProgramLoc -> StateSpec -> OverrideMatcher MIR w ()
  528 enforceDisjointness cc loc ss =
  529   do let sym = cc^.mccSym
  530      sub <- OM (use setupValueSub)
  531      let mems = Map.elems $ Map.intersectionWith (,) (view MS.csAllocs ss) sub
  532 
  533      let md = MS.ConditionMetadata
  534               { MS.conditionLoc = loc
  535               , MS.conditionTags = mempty
  536               , MS.conditionType = "memory region disjointness"
  537               , MS.conditionContext = ""
  538               }
  539      -- Ensure that all regions are disjoint from each other.
  540      sequence_
  541         [ do c <- liftIO $ W4.notPred sym =<< equalRefsPred cc p q
  542              addAssert c md a
  543 
  544         | let a = Crucible.SimError loc $
  545                     Crucible.AssertFailureSimError "Memory regions not disjoint" ""
  546         , (_, Some p) : ps <- tails mems
  547         , (_, Some q)      <- ps
  548         ]

  644 handleSingleOverrideBranch :: forall rtp args ret.
  645   Options            {- ^ output/verbosity options                      -} ->
  646   SharedContext      {- ^ context for constructing SAW terms            -} ->
  647   MIRCrucibleContext {- ^ context for interacting with Crucible         -} ->
  648   W4.ProgramLoc      {- ^ Location of the call site for error reporting -} ->
  649   IORef MetadataMap ->
  650   Crucible.FnHandle args ret {- ^ the handle for this function -} ->
  651   OverrideWithPreconditions MIR ->
  652   Crucible.OverrideSim (SAWCruciblePersonality Sym) Sym MIR rtp args ret
  653      (Crucible.RegValue Sym ret)
  654 handleSingleOverrideBranch opts sc cc call_loc mdMap h (OverrideWithPreconditions preconds cs st) =
  655   mccWithBackend cc $ \bak -> do
  656   let sym = backendGetSym bak
  657   let fnName = cs ^. MS.csMethod
  658   let retTy = Crucible.handleReturnType h
  659 
  660   liftIO $ printOutLn opts Info $ unwords
  661     [ "Found a single potential override for"
  662     , show fnName
  663     ]
  664 
  665   -- First assert the override preconditions
  666   liftIO $ forM_ preconds $ \(md,W4.LabeledPred p r) ->
  667     do (ann,p') <- W4.annotateTerm sym p
  668        let caller = unwords ["Override called from:", show (W4.plSourceLoc call_loc)]
  669        let md' = md{ MS.conditionContext = MS.conditionContext md ++ caller }
  670        modifyIORef mdMap (Map.insert ann md')
  671        Crucible.addAssertion bak (Crucible.LabeledPred p' r)
  672 
  673   g <- Crucible.readGlobals
  674   res <- liftIO $ runOverrideMatcher sym g
  675      (st^.setupValueSub)
  676      (st^.termSub)
  677      (st^.osFree)
  678      (st^.osLocation)
  679      (methodSpecHandler_poststate opts sc cc retTy cs)
  680   case res of
  681     Left (OF loc rsn)  ->
  682       -- TODO, better pretty printing for reasons
  683       liftIO
  684         $ Crucible.abortExecBecause
  685         $ Crucible.AssertionFailure
  686         $ Crucible.SimError loc
  687         $ Crucible.AssertFailureSimError "assumed false" (show rsn)
  688     Right (ret,st') ->
  689       do liftIO $ forM_ (st'^.osAssumes) $ \(_md,asum) ->
  690            Crucible.addAssumption bak
  691             $ Crucible.GenericAssumption (st^.osLocation) "override postcondition" asum
  692          Crucible.writeGlobals (st'^.overrideGlobals)
  693          Crucible.overrideReturn' (Crucible.RegEntry retTy ret)

  695 handleOverrideBranches :: forall rtp args ret.
  696   Options            {- ^ output/verbosity options                      -} ->
  697   SharedContext      {- ^ context for constructing SAW terms            -} ->
  698   MIRCrucibleContext {- ^ context for interacting with Crucible         -} ->
  699   W4.ProgramLoc      {- ^ Location of the call site for error reporting -} ->
  700   NE.NonEmpty (MS.CrucibleMethodSpecIR MIR)
  701     {- ^ specification for current function override -} ->
  702   Crucible.FnHandle args ret {- ^ the handle for this function -} ->
  703   [OverrideWithPreconditions MIR] ->
  704   ([OverrideWithPreconditions MIR],[OverrideWithPreconditions MIR],[OverrideWithPreconditions MIR]) ->
  705   Crucible.OverrideSim (SAWCruciblePersonality Sym) Sym MIR rtp args ret
  706      (Crucible.RegValue Sym ret)
  707 
  708 handleOverrideBranches opts sc cc call_loc css h branches (true, false, unknown) =
  709   mccWithBackend cc $ \bak -> do
  710   let sym = backendGetSym bak
  711   let fnName = show $ NE.head css ^. MS.csMethod
  712   Crucible.RegMap args <- Crucible.getOverrideArgs
  713 
  714   -- Collapse the preconditions to a single predicate
  715   branches' <- liftIO $ forM (true ++ unknown) $
  716     \(OverrideWithPreconditions preconds cs st) ->
  717       W4.andAllOf sym (folded . _2 . W4.labeledPred) preconds <&>
  718         \precond -> (precond, cs, st)
  719 
  720   -- Now use crucible's symbolic branching machinery to select between the branches.
  721   -- Essentially, we are doing an n-way if statement on the precondition predicates
  722   -- for each override, and selecting the first one whose preconditions hold.
  723   --
  724   -- Then, in the body of the branch, we run the poststate handler to update the
  725   -- memory state, compute return values and compute postcondition predicates.
  726   --
  727   -- For each override branch that doesn't fail outright, we assume the relevant
  728   -- postconditions, update the crucible global variable state, and return the
  729   -- computed return value.
  730   --
  731   -- We add a final default branch that simply fails unless some previous override
  732   -- branch has already succeeded.
  733   liftIO $ printOutLn opts Info $ unwords
  734     [ "Branching on"
  735     , show (length branches')
  736     , "override variants of"
  737     , fnName
  738     , "..."
  739     ]
  740   let retTy = Crucible.handleReturnType h
  741   res <- Crucible.regValue <$> Crucible.callOverride h
  742      (Crucible.mkOverride' "overrideBranches" retTy
  743        (Crucible.symbolicBranches Crucible.emptyRegMap $
  744          [ ( precond
  745            , do g <- Crucible.readGlobals
  746                 res <- liftIO $ runOverrideMatcher sym g
  747                    (st^.setupValueSub)
  748                    (st^.termSub)
  749                    (st^.osFree)
  750                    (st^.osLocation)
  751                    (methodSpecHandler_poststate opts sc cc retTy cs)
  752                 case res of
  753                   Left (OF loc rsn)  ->
  754                     -- TODO, better pretty printing for reasons
  755                     liftIO
  756                       $ Crucible.abortExecBecause
  757                       $ Crucible.AssertionFailure
  758                       $ Crucible.SimError loc
  759                       $ Crucible.AssertFailureSimError "assumed false" (show rsn)
  760                   Right (ret,st') ->
  761                     do liftIO $ forM_ (st'^.osAssumes) $ \(_md,asum) ->
  762                          Crucible.addAssumption bak
  763                           $ Crucible.GenericAssumption (st^.osLocation) "override postcondition" asum
  764                        Crucible.writeGlobals (st'^.overrideGlobals)
  765                        Crucible.overrideReturn' (Crucible.RegEntry retTy ret)
  766            , Just (W4.plSourceLoc (cs ^. MS.csLoc))
  767            )
  768          | (precond, cs, st) <- branches'
  769          ] ++
  770          [ let e prettyArgs symFalse unsat = show $ PP.vcat $ concat
  771                  [ [ PP.pretty $
  772                      "No override specification applies for " ++ fnName ++ "."
  773                    ]
  774                  , [ "Arguments:"
  775                    , bullets '-' prettyArgs
  776                    ]
  777                  , if | not (null false) ->
  778                         [ PP.vcat
  779                           [ PP.pretty (unwords
  780                               [ "The following overrides had some preconditions"
  781                               , "that failed concretely:"
  782                               ])
  783                           , bullets '-' (map ppConcreteFailure false)
  784                           ]
  785                         ]
  786                       -- See comment on ppSymbolicFailure: this needs more
  787                       -- examination to see if it's useful.
  788                       -- - | not (null symFalse) ->
  789                       --   [ PP.text (unwords
  790                       --       [ "The following overrides had some preconditions "
  791                       --       , "that failed symbolically:"
  792                       --       ]) PP.<$$> bullets '-' (map ppSymbolicFailure symFalse)
  793                       --   ]
  794 
  795                       -- Note that we only print these in case no override had
  796                       -- individually (concretely or symbolically) false
  797                       -- preconditions.
  798                       | not (null unsat) && null false && null symFalse ->
  799                         [ PP.vcat
  800                           [ PP.pretty (unwords
  801                             [ "The conjunction of these overrides' preconditions"
  802                             , "was unsatisfiable, meaning your override can never"
  803                             , "apply. You probably have unintentionally specified"
  804                             , "mutually exclusive/inconsistent preconditions."
  805                             ])
  806                           , bullets '-' (unsat ^.. each . owpMethodSpec . to MS.ppMethodSpec)
  807                           ]
  808                         ]
  809                       | null false && null symFalse ->
  810                         [ PP.pretty (unwords
  811                             [ "No overrides had any single concretely or"
  812                             , "symbolically failing preconditions."
  813                             ])
  814                         ]
  815                       | otherwise -> []
  816                  , if | simVerbose opts < 3 ->
  817                         [ PP.pretty $ unwords
  818                           [ "Run SAW with --sim-verbose=3 to see a description"
  819                           , "of each override."
  820                           ]
  821                         ]
  822                       | otherwise ->
  823                         [ PP.vcat
  824                           [ "Here are the descriptions of each override:"
  825                           , bullets '-'
  826                             (branches ^.. each . owpMethodSpec . to MS.ppMethodSpec)
  827                           ]
  828                         ]
  829                  ]
  830            in ( W4.truePred sym
  831               , liftIO $ do
  832                   -- Now that we're failing, do the additional work of figuring out
  833                   -- if any overrides had symbolically false preconditions
  834                   symFalse <- catMaybes <$> (forM unknown $ \owp ->
  835                     findFalsePreconditions bak owp <&>
  836                       \case
  837                         [] -> Nothing
  838                         ps -> Just (owp, ps))
  839 
  840                   prettyArgs <-
  841                     ppArgs sym cc (NE.head css) (Crucible.RegMap args)
  842 
  843                   unsat <-
  844                     filterM
  845                       (unsatPreconditions bak (owpPreconditions . each . _2 . W4.labeledPred))
  846                       branches
  847 
  848                   Crucible.addFailedAssertion bak
  849                     (Crucible.GenericSimError (e prettyArgs symFalse unsat))
  850               , Just (W4.plSourceLoc call_loc)
  851               )
  852          ]))
  853      (Crucible.RegMap args)
  854   liftIO $ printOutLn opts Info $ unwords ["Applied override!", fnName]
  855   return res

  877 -- | Map the given substitution over all 'SetupTerm' constructors in
  878 -- the given 'SetupValue'.
  879 instantiateSetupValue ::
  880   SharedContext     ->
  881   IntMap Term       ->
  882   SetupValue        ->
  883   IO SetupValue
  884 instantiateSetupValue sc s v =
  885   case v of
  886     MS.SetupVar _                     -> return v
  887     MS.SetupTerm tt                   -> MS.SetupTerm <$> doTerm tt
  888     MS.SetupArray elemTy vs           -> MS.SetupArray elemTy <$> mapM (instantiateSetupValue sc s) vs
  889     MS.SetupStruct did vs             -> MS.SetupStruct did <$> mapM (instantiateSetupValue sc s) vs
  890     MS.SetupEnum enum_                -> MS.SetupEnum <$> instantiateSetupEnum enum_
  891     MS.SetupTuple x vs                -> MS.SetupTuple x <$> mapM (instantiateSetupValue sc s) vs
  892     MS.SetupSlice slice               -> MS.SetupSlice <$> instantiateSetupSlice slice
  893     MS.SetupNull empty                -> absurd empty
  894     MS.SetupGlobal _ _                -> return v
  895     MS.SetupElem m a i                -> MS.SetupElem m <$> instantiateSetupValue sc s a <*> pure i
  896     MS.SetupField _ _ _               -> return v
  897     MS.SetupCast _ _                  -> return v
  898     MS.SetupUnion empty _ _           -> absurd empty
  899     MS.SetupGlobalInitializer _ _     -> return v
  900     MS.SetupMux x c t f               -> MS.SetupMux x
  901                                            <$> doTerm c
  902                                            <*> instantiateSetupValue sc s t
  903                                            <*> instantiateSetupValue sc s f
  904   where
  905     doTerm (TypedTerm schema t) = TypedTerm schema <$> scInstantiateExt sc s t
  906 
  907     instantiateSetupEnum :: MirSetupEnum -> IO MirSetupEnum
  908     instantiateSetupEnum (MirSetupEnumVariant adt variant variantIdx vs) =
  909       MirSetupEnumVariant adt variant variantIdx <$>
  910       mapM (instantiateSetupValue sc s) vs
  911     instantiateSetupEnum (MirSetupEnumSymbolic adt discr variants) =
  912       MirSetupEnumSymbolic adt discr <$>
  913       mapM (mapM (instantiateSetupValue sc s)) variants
  914 
  915     instantiateSetupSlice :: MirSetupSlice -> IO MirSetupSlice
  916     instantiateSetupSlice (MirSetupSliceRaw ref len) =
  917       MirSetupSliceRaw
  918         <$> instantiateSetupValue sc s ref
  919         <*> instantiateSetupValue sc s len

  921       MirSetupSlice sliceInfo <$> instantiateSetupValue sc s arr

  923       arr' <- instantiateSetupValue sc s arr

  963 -- | Process a "points_to" statement from the precondition section of
  964 -- the CrucibleSetup block. First, load the value from the address
  965 -- indicated by 'ptr', and then match it against the pattern 'val'.
  966 learnPointsTo ::
  967   Options                    ->
  968   SharedContext              ->
  969   MIRCrucibleContext         ->
  970   CrucibleMethodSpecIR       ->
  971   MS.PrePost                 ->
  972   MirPointsTo                ->
  973   OverrideMatcher MIR w ()
  974 learnPointsTo opts sc cc spec prepost (MirPointsTo md reference referents) =
  975   mccWithBackend cc $ \bak ->
  976   do let col = cc ^. mccRustModule . Mir.rmCS ^. Mir.collection
  977      globals <- OM (use overrideGlobals)
  978      MIRVal referenceShp referenceVal <-
  979        resolveSetupValueMIR opts cc sc spec reference
  980      -- By the time we reach here, we have already checked (in mir_points_to)
  981      -- that we are in fact dealing with a reference value, so the call to
  982      -- `testRefShape` below should always succeed.
  983      IsRefShape _ referenceInnerMirTy _ referenceInnerTpr <-
  984        case testRefShape referenceShp of
  985          Just irs -> pure irs
  986          Nothing ->
  987            panic "learnPointsTo" [
  988                "Unexpected non-reference type:",
  989                "   " <> Text.pack (show $ PP.pretty $ shapeMirTy referenceShp)
  990            ]
  991      let innerShp = tyToShapeEq col referenceInnerMirTy referenceInnerTpr
  992      referentVal <- firstPointsToReferent referents
  993      v <- liftIO $ Mir.readMirRefIO bak globals iTypes
  994        referenceInnerTpr referenceVal
  995      matchArg opts sc cc spec prepost md (MIRVal innerShp v) referentVal
  996   where
  997     iTypes = cc ^. mccIntrinsicTypes

 1032 newtype MatchProj

 1036 -- | Match the value of a function argument with a symbolic 'SetupValue'.
 1037 matchArg ::
 1038   forall w.
 1039   Options              {- ^ saw script print out opts -} ->
 1040   SharedContext        {- ^ context for constructing SAW terms -} ->
 1041   MIRCrucibleContext   {- ^ context for interacting with Crucible -} ->
 1042   CrucibleMethodSpecIR {- ^ specification for current function override -} ->
 1043   MS.PrePost ->
 1044   MS.ConditionMetadata ->
 1045   MIRVal               {- ^ concrete simulation value -} ->
 1046   SetupValue           {- ^ expected specification value -} ->
 1047   OverrideMatcher MIR w ()
 1048 matchArg opts sc cc cs prepost md = go False []
 1049   where
 1050 
 1051   go ::
 1052     -- The internal Bool argument is to keep track of whether the given
 1053     -- SetupValue is inside a SetupCast. The SetupVar case uses different
 1054     -- types depending on whether it is in a SetupCast or not, but the
 1055     -- SetupCast case can't just do a recursive call with different types,
 1056     -- since we don't know what the types should be until we get to the
 1057     -- SetupVar. So instead we just pass along the fact that we are recursing
 1058     -- inside a SetupCast.
 1059     Bool ->
 1060     -- Due to projection SetupValue constructors like SetupElem, which may be
 1061     -- arbitrarily nested, we keep a stack of MatchProjs to track which part
 1062     -- of the given SetupValue we are trying to match against. For instance,
 1063     -- [MatchIndex 3, MatchIndex 7] means we are trying to match the MIRVal
 1064     -- against index 7 of index 3 of the SetupValue.
 1065     [MatchProj] ->
 1066     MIRVal ->
 1067     SetupValue ->
 1068     OverrideMatcher MIR w ()
 1069 
 1070   go inCast projStack actual expected
 1071     | MS.SetupTerm expectedTT <- expected
 1072     , TypedTermSchema (Cryptol.Forall [] [] tyexpr) <- ttType expectedTT
 1073     , Right tval <- Cryptol.evalType mempty tyexpr
 1074     = do sym <- Ov.getSymInterface
 1075          (tval', expectedTerm') <-
 1076            applyProjToTerm sym fail_ projStack tval (ttTerm expectedTT)
 1077          realTerm <- valueToSC sym fail_ tval' actual
 1078          matchTerm sc md prepost realTerm expectedTerm'

 1416 {-
 1417 Note [Matching slices in overrides]
 1418 ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
 1419 Matching slice references in overrides is surprisingly tricky. Suppose we have
 1420 expected and actual slice reference values, both of type &[T]. At a high level,
 1421 we need to check two things:
 1422 
 1423 1. The lengths of the expected and actual slices are the same.
 1424 2. The underlying references (of type `*const T`) are the same.
 1425 
 1426 (1) is fairly straightforward, but (2) is easy to mess up. It's tempting to
 1427 just call `matchArg` on the underlying references, but don't do this! These
 1428 reference values are derived from array references, which are of type &[T; N],
 1429 but calling matchArg on something of type `*const T` will associate the
 1430 reference's AllocIndex to something that points to a value of type T, not a
 1431 value of type [T; N]. This leads to disaster later when checking mir_points_to
 1432 statements for that reference value in the precondition of the override, as it
 1433 will incorrectly require the right-hand side to be of type T, not [T; N]. (See
 1434 #2045 for an example of this actually happening.)
 1435 
 1436 Instead, we want to call `matchArg` on the *array reference value* associated
 1437 with a slice, not the raw reference value itself. To do this, we take the raw
 1438 reference value and use mirRef_peelIndexIO, a crucible-mir memory model
 1439 operation which "peels back" the indexing operation that raw slice references
 1440 use, thereby turning a `*const T` value into a `&[T; N]` value. It's a bit
 1441 indirect, but it avoids needing to plumb around the original array reference
 1442 value alongside the slice's raw reference value.
 1443 
 1444 We do something similar for &str slices, as crucible-mir backs them with an
 1445 array reference value of type &[u8; N].
 1446 
 1447 This assumes that all slice reference values passed to an override were derived
 1448 from crucible-mir's indexing operations, as this is crucial for
 1449 mirRef_peelIndexIO to work. This is currently the case on every example we have
 1450 tried, but if we encounter an example that breaks this assumption, then we will
 1451 need to rethink this approach.
 1452 -}

 1539 --   of method specifications.  The main work done in this function to manage
 1540 --   the process of selecting between several possible different override
 1541 --   specifications that could apply.  We want a proof to succeed if _any_
 1542 --   choice of method spec allows the proof to go through, which is a slightly
 1543 --   awkward thing to fit into the symbolic simulation framework.
 1544 --
 1545 --   The main work of determining the preconditions, postconditions, memory
 1546 --   updates and return value for a single specification is done by
 1547 --   the @methodSpecHandler_prestate@ and @methodSpecHandler_poststate@ functions.
 1548 --
 1549 --   In a first phase, we attempt to apply the precondition portion of each of
 1550 --   the given method specifications.  Each of them that might apply generate
 1551 --   a substitution for the setup variables and a collection of preconditions
 1552 --   that guard the specification.  We use these preconditions to compute
 1553 --   a multiway symbolic branch, one for each override which might apply.
 1554 --
 1555 --   In the body of each of the individual branches, we compute the postcondition
 1556 --   actions of the corresponding method specification.  This will update memory
 1557 --   and compute function return values, in addition to assuming postcondition

 1604           pure $
 1605             PP.vcat
 1606             [ MS.ppMethodSpec methodSpec
 1607             , "Arguments:"
 1608             , bullets '-' prettyArgs
 1609             , ppOverrideFailure failureReason
 1610             ]
 1611     in
 1612       case partitionEithers (F.toList prestates) of
 1613           (errs, []) -> do
 1614             msgs <-
 1615               mapM (\(cs, err) ->
 1616                       ("*" PP.<>) . PP.indent 2 <$> prettyError cs err)
 1617                    (zip (F.toList css) errs)
 1618             fail $ show $
 1619               PP.vcat ["All overrides failed during structural matching:", PP.vcat msgs]
 1620           (_, ss) -> liftIO $
 1621             forM ss $ \(cs,st) ->
 1622               return (OverrideWithPreconditions (st^.osAsserts) cs st)
 1623 
 1624   -- Now we do a second phase of simple compatibility checking: we check to see
 1625   -- if any of the preconditions of the various overrides are concretely false.
 1626   -- If so, there's no use in branching on them with @symbolicBranches@.
 1627   (true, false, unknown) <- liftIO $ partitionOWPsConcrete sym branches
 1628 
 1629   -- Check if there is only a single override branch that might apply at this
 1630   -- point.  If so, commit to it and handle that case specially. If there is
 1631   -- more than one (or zero) branches that might apply, go to the general case.
 1632   case true ++ unknown of
 1633     [singleBranch] | ?singleOverrideSpecialCase ->
 1634          handleSingleOverrideBranch opts sc cc call_loc mdMap h singleBranch
 1635     _ -> handleOverrideBranches opts sc cc call_loc css h branches (true, false, unknown)
 1636

 1791 valueToSC ::
 1792   forall w.
 1793   Sym ->
 1794   (forall a. OverrideMatcher MIR w a) {- ^ what to do on failure -} ->
 1795   Cryptol.TValue ->
 1796   MIRVal ->
 1797   OverrideMatcher MIR w Term
 1798 valueToSC sym fail_ tval (MIRVal shp val) =
 1799   case (tval, shp) of
 1800     (Cryptol.TVBit, PrimShape _ W4.BaseBoolRepr) ->
 1801       liftIO (toSC sym st val)
 1802     (Cryptol.TVSeq n Cryptol.TVBit, PrimShape _ (W4.BaseBVRepr w))
 1803       |  n == 8, Just _ <- W4.testEquality w (W4.knownNat @8)
 1804       -> liftIO (toSC sym st val)
 1805       |  n == 16, Just _ <- W4.testEquality w (W4.knownNat @16)
 1806       -> liftIO (toSC sym st val)
 1807       |  n == 32, Just _ <- W4.testEquality w (W4.knownNat @32)
 1808       -> liftIO (toSC sym st val)
 1809       |  n == 64, Just _ <- W4.testEquality w (W4.knownNat @64)
 1810       -> liftIO (toSC sym st val)
 1811       |  n == 128, Just _ <- W4.testEquality w (W4.knownNat @128)
 1812       -> liftIO (toSC sym st val)
 1813     (Cryptol.TVTuple [], UnitShape _) ->
 1814       liftIO (scUnitValue sc)
 1815     (Cryptol.TVTuple tys, TupleShape _ _ flds)
 1816       |  length tys == Ctx.sizeInt (Ctx.size flds)
 1817       -> do terms <-
 1818               traverse
 1819                 fieldToSC
 1820                 (zip tys (FC.toListFC Some (Ctx.zipWith Functor.Pair flds val)))
 1821             liftIO (scTupleReduced sc terms)
 1822     (Cryptol.TVSeq n cryty, ArrayShape _ _ arrShp)
 1823       |  Mir.MirVector_Vector vals <- val
 1824       ,  toInteger (V.length vals) == n
 1825       -> do terms <- V.toList <$>
 1826               traverse (\v -> valueToSC sym fail_ cryty (MIRVal arrShp v)) vals
 1827             t <- shapeToTerm sc arrShp
 1828             liftIO (scVectorReduced sc t terms)
 1829       |  Mir.MirVector_PartialVector vals <- val
 1830       ,  toInteger (V.length vals) == n
 1831       -> do let vals' = V.toList $
 1832                         V.map (readMaybeType sym "vector element" (shapeType arrShp)) vals
 1833             terms <-
 1834               traverse (\v -> valueToSC sym fail_ cryty (MIRVal arrShp v)) vals'
 1835             t <- shapeToTerm sc arrShp
 1836             liftIO (scVectorReduced sc t terms)
 1837       |  Mir.MirVector_Array{} <- val
 1838       -> fail "valueToSC: Symbolic arrays not supported"
 1839     _ ->
 1840       fail_
 1841   where
 1842     st = sym ^. W4.userState
 1843     sc = saw_ctx st
 1844 
 1845     fieldToSC ::
 1846          (Cryptol.TValue, Some (Functor.Product FieldShape (Crucible.RegValue' Sym)))
 1847       -> OverrideMatcher MIR w Term
 1848     fieldToSC (ty, Some (Functor.Pair fldShp (Crucible.RV tm))) = do
 1849       mirVal <-
 1850         case fldShp of
 1851           ReqField shp' ->
 1852             pure $ MIRVal shp' tm
 1853           OptField shp' ->
 1854             pure $ MIRVal shp'
 1855                  $ readMaybeType sym "field" (shapeType shp') tm
 1856       valueToSC sym fail_ ty mirVal
 1857 
 1858 -- | Apply a stack of projections to a 'Term'.
 1859 applyProjToTerm ::
 1860   Sym ->
 1861   (forall a. OverrideMatcher MIR w a) {- ^ what to do on failure -} ->
 1862   [MatchProj] {- ^ stack of projections -} ->
 1863   Cryptol.TValue {- ^ Cryptol type of the term argument -} ->
 1864   Term ->
 1865   OverrideMatcher MIR w (Cryptol.TValue, Term)
 1866     -- ^ result term and its Cryptol type
 1867 applyProjToTerm sym fail_ = go
 1868   where
 1869     go [] tp term = pure (tp, term)
 1870     go (MatchIndex i : projStack) (Cryptol.TVSeq sz tp) term
 1871       | i >= 0 && fromIntegral i < sz = do
 1872         doIndex <- liftIO $ indexSeqTerm sym (sz, tp) term
 1873         go projStack tp =<< liftIO (doIndex i)
 1874     go _ _ _ = fail_
 1875

 1877 applyProjToMIRVal ::

 1879   Sym ->
 1880   [MatchProj] {- ^ stack of projections -} ->
 1881   MIRVal ->
 1882   MaybeT m MIRVal
 1883 applyProjToMIRVal _ [] mv = pure mv

 1885                   (MatchIndex i : projStack)
