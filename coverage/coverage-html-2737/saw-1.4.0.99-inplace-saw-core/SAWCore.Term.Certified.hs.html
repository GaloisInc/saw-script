   74 --------------------------------------------------------------------------------
   75 -- * Certified typed terms
   76 
   77 -- | An abstract datatype pairing a well-formed 'Raw.Term' with its type.
   78 -- A 'Term' represents a typing judgment of the form @Ctx |- e : t@, where
   79 -- @Ctx@ is the typing context for variables, @e@ is the raw term, and
   80 -- @t@ is its type.
   81 data Term =
   82   Term
   83   (IntMap Raw.Term) -- ^ typing context
   84   Raw.Term -- ^ value
   85   Raw.Term -- ^ type
   86 
   87 -- | The raw term of a 'Term'.
   88 rawTerm :: Term -> Raw.Term
   89 rawTerm (Term _ trm _) = trm
   90 
   91 -- | The type of a 'Term' as a raw term.
   92 rawType :: Term -> Raw.Term
   93 rawType (Term _ _ typ) = typ
   94 
   95 -- | The typing context of a 'Term', keyed by the 'VarIndex' of each
   96 -- 'VarName' in the term.
   97 rawCtx :: Term -> IntMap Raw.Term
   98 rawCtx (Term ctx _ _) = ctx
   99

  102 -- | Reduce the given 'Raw.Term' to WHNF, using all reductions allowed by
  103 -- the SAWCore type system.
  104 scTypeCheckWHNF :: SharedContext -> Raw.Term -> IO Raw.Term
  105 scTypeCheckWHNF sc t =
  106   do (_, t') <- rewriteSharedTerm sc (addConvs natConversions emptySimpset :: Simpset ()) t
  107      Raw.scWhnf sc t'
  108 
  109 -- | Check if two terms are "convertible for type-checking", meaning that they
  110 -- are convertible up to 'natConversions'.
  111 scTypeConvertible :: SharedContext -> Raw.Term -> Raw.Term -> IO Bool
  112 scTypeConvertible sc t1 t2 = Raw.scConvertibleEval sc scTypeCheckWHNF True t1 t2
  113 
  114 -- | Check whether one type is a subtype of another: Either they are
  115 -- convertible, or they are both Pi types with convertible argument
  116 -- types and result sorts @s1@ and @s2@ with @s1 <= s2@.
  117 scSubtype :: SharedContext -> Raw.Term -> Raw.Term -> IO Bool
  118 scSubtype sc t1 t2
  119   | alphaEquiv t1 t2 = pure True
  120   | otherwise =
  121     do t1' <- Raw.scWhnf sc t1
  122        t2' <- Raw.scWhnf sc t2
  123        case (t1', t2') of
  124          (asSort -> Just s1, asSort -> Just s2) ->
  125            pure (s1 <= s2)
  126          (unwrapTermF -> Pi x1 a1 b1, unwrapTermF -> Pi x2 a2 b2)
  127            | x1 == x2 ->
  128              (&&) <$> scTypeConvertible sc a1 a2 <*> scSubtype sc b1 b2
  129            | otherwise ->
  130              do conv1 <- scTypeConvertible sc a1 a2
  131                 var1 <- Raw.scVariable sc x1 a1
  132                 b2' <- Raw.scInstantiateExt sc (IntMap.singleton (vnIndex x2) var1) b2
  133                 conv2 <- scSubtype sc b1 b2'
  134                 pure (conv1 && conv2)
  135          _ ->
  136            scTypeConvertible sc t1' t2'

  141 -- | Compute the type of a 'Term'.
  142 scTypeOf :: SharedContext -> Term -> IO Term
  143 scTypeOf sc (Term ctx _tm tp) =
  144   do tp_tp <- Raw.scTypeOf' sc ctx tp
  145      -- Shrink typing context if possible
  146      let ctx' = pruneContext (freeVars tp_tp) ctx
  147      pure (Term ctx' tp tp_tp)
  148 
  149 -- | Reduce a 'Term' to WHNF using all reductions and conversions
  150 -- allowed by the SAWCore type system (see also 'scTypeCheckWHNF').
  151 scWHNF :: SharedContext -> Term -> IO Term
  152 scWHNF sc (Term ctx tm tp) =
  153   do tm' <- scTypeCheckWHNF sc tm
  154      pure (Term ctx tm' tp)
  155 
  156 -- | Reduce a 'Term' to WHNF using beta reductions, tuple and record
  157 -- reductions, recursor reductions, and unfolding definitions.
  158 scWhnf :: SharedContext -> Term -> IO Term
  159 scWhnf sc (Term ctx tm tp) =
  160   do tm' <- Raw.scWhnf sc tm
  161      pure (Term ctx tm' tp)

  163 scGlobal :: SharedContext -> Ident -> IO Term

  166      tp <- Raw.scTypeOfIdent sc ident
  167      pure (Term IntMap.empty tm tp)

  169 -- | @scAscribe sc arg t@ ascribes the type @t@ to term @arg@.
  170 -- The original type of @arg@ must be a subtype of type @t@.
  171 scAscribe :: SharedContext -> Term -> Term -> IO Term
  172 scAscribe sc arg t =
  173   do ok <- scSubtype sc (rawType arg) (rawTerm t)
  174      unless ok $ fail $ unlines $
  175        ["Not a subtype",
  176         "expected: " ++ showTerm (rawTerm t),
  177         "got: " ++ showTerm (rawType arg)]
  178      let tm = rawTerm arg
  179      let tp = rawTerm t
  180      ctx <- unifyContexts "scTypedApply" (rawCtx arg) (rawCtx t)
  181      pure (Term ctx tm tp)

  195 scOpaqueConstant :: SharedContext -> NameInfo -> Term -> IO Term
  196 scOpaqueConstant sc nmi ty =
  197   do _s <- ensureSort sc (rawType ty)
  198      unless (IntMap.null (rawCtx ty)) $
  199        fail "scOpaqueConstant"
  200      let tp = rawTerm ty
  201      tm <- Raw.scOpaqueConstant sc nmi tp
  202      pure (Term IntMap.empty tm tp)

  204 scInstantiate :: SharedContext -> IntMap Term -> Term -> IO Term
  205 scInstantiate sc sub t0 =
  206   do _ <- unifyContexts "scInstantiate" (fmap rawType sub) (rawCtx t0) -- ensure sub is well-typed
  207      let tm0 = rawTerm t0
  208      tm <- Raw.scInstantiateExt sc (fmap rawTerm sub) tm0
  209      let tp = rawType t0
  210      let ctx0 = IntMap.difference (rawCtx t0) sub
  211      let contexts = ctx0 : map rawCtx (IntMap.elems (IntMap.restrictKeys sub (freeVars tm0)))
  212      ctx <- unifyContextList "scInstantiate" contexts
  213      pure (Term ctx tm tp)

  218 -- possible errors: not a pi type, bad argument type, context mismatch
  219 scApply :: SharedContext -> Term -> Term -> IO Term
  220 scApply sc f arg =
  221   do tm <- Raw.scApply sc (rawTerm f) (rawTerm arg)
  222      (vnIndex -> i, t1, t2) <- ensurePi sc (rawType f)
  223      ok <- scSubtype sc (rawType arg) t1
  224      unless ok $ fail $ unlines $
  225        ["Not a subtype", "expected: " ++ showTerm t1, "got: " ++ showTerm (rawType arg)]
  226      tp <- Raw.scInstantiateExt sc (IntMap.singleton i (rawTerm arg)) t2
  227      ctx <- unifyContexts "scApply" (rawCtx f) (rawCtx arg)
  228      pure (Term ctx tm tp)

  243 -- possible errors: not a type, context mismatch, variable free in context
  244 scLambda :: SharedContext -> VarName -> Term -> Term -> IO Term
  245 scLambda sc x t body =
  246   do _s <- ensureSort sc (rawType t)
  247      tm <- Raw.scLambda sc x (rawTerm t) (rawTerm body)
  248      ensureNotFreeInContext x body
  249      _ <- unifyContexts "scLambda" (IntMap.singleton (vnIndex x) (rawTerm t)) (rawCtx body)
  250      ctx <- unifyContexts "scLambda" (rawCtx t) (IntMap.delete (vnIndex x) (rawCtx body))
  251      tp <- Raw.scPi sc x (rawTerm t) (rawType body)
  252      pure (Term ctx tm tp)
  253 
  254 -- possible errors: not a variable, context mismatch, variable free in context
  255 scAbstract :: SharedContext -> Term -> Term -> IO Term
  256 scAbstract sc var body =
  257   case asVariable (rawTerm var) of
  258     Nothing -> fail "scAbstract: Not a variable"
  259     Just (x, _) ->
  260       do ensureNotFreeInContext x body
  261          tm <- Raw.scLambda sc x (rawType var) (rawTerm body)
  262          tp <- Raw.scPi sc x (rawType var) (rawType body)
  263          ctx0 <- unifyContexts "scAbstract" (rawCtx var) (rawCtx body)
  264          let ctx = IntMap.delete (vnIndex x) ctx0
  265          pure (Term ctx tm tp)
  266 
  267 -- possible errors: not a type, context mismatch, variable free in context
  268 scPi :: SharedContext -> VarName -> Term -> Term -> IO Term
  269 scPi sc x t body =
  270   do tm <- Raw.scPi sc x (rawTerm t) (rawTerm body)
  271      ensureNotFreeInContext x body
  272      _ <- unifyContexts "scPi" (IntMap.singleton (vnIndex x) (rawTerm t)) (rawCtx body)
  273      ctx <- unifyContexts "scPi" (rawCtx t) (IntMap.delete (vnIndex x) (rawCtx body))
  274      s1 <- ensureSort sc (rawType t)
  275      s2 <- ensureSort sc (rawType body)
  276      tp <- Raw.scSort sc (piSort s1 s2)
  277      pure (Term ctx tm tp)
  278 
  279 scGeneralize :: SharedContext -> Term -> Term -> IO Term
  280 scGeneralize sc var body =
  281   case asVariable (rawTerm var) of
  282     Nothing -> fail "scGeneralize: Not a variable"
  283     Just (x, _) ->
  284       do tp <- scTypeOf sc var
  285          scPi sc x tp body

  361 -- possible errors: not a datatype, bad elimination sort
  362 scRecursor :: SharedContext -> Name -> Sort -> IO Term
  363 scRecursor sc nm s =
  364   do mm <- Raw.scGetModuleMap sc
  365      case lookupVarIndexInMap (nameIndex nm) mm of
  366        Just (ResolvedDataType dt) ->
  367          do unless (Raw.allowedElimSort dt s) $ fail "Disallowed propositional elimination"
  368             let d = dtName dt
  369             let nparams = length (dtParams dt)
  370             let nixs = length (dtIndices dt)
  371             let ctorOrder = map ctorName (dtCtors dt)
  372             let crec = CompiledRecursor d s nparams nixs ctorOrder
  373             tm <- Raw.scFlatTermF sc (Recursor crec)
  374             tp <- Raw.scRecursorType sc dt s
  375             pure (Term IntMap.empty tm tp)
  376        _ ->
  377          fail "datatype not found"

  379 -- possible errors: field not a type, context mismatch
  380 scRecordType :: SharedContext -> [(FieldName, Term)] -> IO Term
  381 scRecordType sc fields =
  382   do tm <- Raw.scRecordType sc (map (fmap rawTerm) fields)
  383      sorts <- traverse (ensureSort sc . rawType . snd) fields
  384      tp <- Raw.scSort sc (foldl max (mkSort 0) sorts)
  385      ctx <- unifyContextList "scRecordType" (map (rawCtx . snd) fields)
  386      pure (Term ctx tm tp)

  445 -- | Prune a typing context by dropping indices unreachable from the
  446 -- given 'IntSet'.
  447 pruneContext :: IntSet -> IntMap Raw.Term -> IntMap Raw.Term
  448 pruneContext vs0 ctx = IntMap.restrictKeys ctx (reachable mempty vs0)
  449   where
  450     fvs = fmap freeVars ctx
  451     reachable old new
  452       | IntSet.null new = old
  453       | otherwise = reachable old' new'
  454           where old' = old <> new
  455                 new' = IntSet.difference (Fold.fold (IntMap.restrictKeys fvs new)) old'
  456 
  457 -- | Two typing contexts are unifiable if they agree perfectly on all
  458 -- entries where they overlap.
  459 unifyContexts :: String -> IntMap Raw.Term -> IntMap Raw.Term -> IO (IntMap Raw.Term)
  460 unifyContexts msg ctx1 ctx2 =
  461   do let check t1 t2 =
  462            unless (t1 == t2) $
  463            fail $ unlines ["unifyContexts: context mismatch", msg,
  464                            "t1: " ++ showTerm t1,
  465                            "t2: " ++ showTerm t2]
  466      sequence_ (IntMap.intersectionWith check ctx1 ctx2)
  467      pure (IntMap.union ctx1 ctx2)
  468 
  469 unifyContextList :: String -> [IntMap Raw.Term] -> IO (IntMap Raw.Term)
  470 unifyContextList msg = foldM (unifyContexts msg) IntMap.empty
  471 
  472 ensureRecognizer :: String -> SharedContext -> (Raw.Term -> Maybe a) -> Raw.Term -> IO a
  473 ensureRecognizer s sc f trm =
  474   case f trm of
  475     Just a -> pure a
  476     Nothing ->
  477       do trm' <- scTypeCheckWHNF sc trm
  478          case f trm' of
  479            Just a -> pure a
  480            Nothing ->
  481              fail $ "ensureRecognizer: Expected " ++ s ++ ", found: " ++ showTerm trm'
  482 
  483 ensureSort :: SharedContext -> Raw.Term -> IO Sort
  484 ensureSort sc tp = ensureRecognizer "Sort" sc asSort tp

  486 ensurePi :: SharedContext -> Raw.Term -> IO (VarName, Raw.Term, Raw.Term)

  489 ensurePairType :: SharedContext -> Raw.Term -> IO (Raw.Term, Raw.Term)
  490 ensurePairType sc tp = ensureRecognizer "PairType" sc asPairType tp

  492 ensureRecordType :: SharedContext -> Raw.Term -> IO (Map FieldName Raw.Term)

  495 piSort :: Sort -> Sort -> Sort
  496 piSort s1 s2 = if s2 == propSort then propSort else max s1 s2

  498 -- | Check whether the given 'VarName' occurs free in the type of
  499 -- another variable in the context of the given 'Term', and fail if it
  500 -- does.
  501 ensureNotFreeInContext :: VarName -> Term -> IO ()
  502 ensureNotFreeInContext x body =
  503   when (any (IntSet.member (vnIndex x) . freeVars) (rawCtx body)) $
  504     fail $ "Variable occurs free in context: " ++ show (vnName x)
