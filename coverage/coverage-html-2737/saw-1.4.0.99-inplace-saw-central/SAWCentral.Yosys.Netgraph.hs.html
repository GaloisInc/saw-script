   46 --------------------------------------------------------------------------------
   47 -- ** Building a network graph from a Yosys module
   48 
   49 -- | A 'Netgraph' represents the data dependencies between 'Cell's in
   50 -- a module. The graph has 'Text' keys which are the cell instance
   51 -- names.
   52 data Netgraph = Netgraph
   53   { _netgraphGraph :: Graph.Graph
   54   , _netgraphNodeFromVertex :: Graph.Vertex -> (Cell [Bitrep], Text, [Text])
   55   -- , _netgraphVertexFromKey :: Text -> Maybe Graph.Vertex
   56   }
   57 makeLenses ''Netgraph
   58 
   59 moduleNetgraph :: Module -> Netgraph
   60 moduleNetgraph m =
   61   let
   62     sources :: Map Bitrep Text
   63     sources =
   64       Map.fromList $
   65       [ (b, cname)
   66       | (cname, c) <- Map.assocs (m ^. moduleCells)
   67       , b <- concat (Map.elems (cellOutputConnections c)) ]
   68 
   69     cellDeps :: Cell [Bitrep] -> [Text]
   70     cellDeps c
   71       | cellIsRegister c = []
   72       | otherwise =
   73         Set.toAscList $ Set.fromList $
   74         Maybe.mapMaybe (flip Map.lookup sources) $
   75         concat $ Map.elems $ cellInputConnections c
   76 
   77     nodes :: [(Cell [Bitrep], Text, [Text])]
   78     nodes = [ (c, cname, cellDeps c) | (cname, c) <- Map.assocs (m ^. moduleCells) ]
   79 
   80     (_netgraphGraph, _netgraphNodeFromVertex, _netgraphVertexFromKey) =
   81       Graph.graphFromEdges nodes
   82   in
   83     Netgraph{..}

   86 -- ** Building a SAWCore term from a network graph
   87 
   88 data ConvertedModule = ConvertedModule
   89   { _convertedModuleTerm :: SC.Term
   90   , _convertedModuleType :: SC.Term
   91   , _convertedModuleCryptolType :: C.Type
   92   }
   93 makeLenses ''ConvertedModule
   94 
   95 lookupPatternTerm ::
   96   (MonadIO m, Ord b, Show b) =>
   97   SC.SharedContext ->
   98   YosysBitvecConsumer ->
   99   [b] ->
  100   Map [b] Preterm ->
  101   m SC.Term
  102 lookupPatternTerm sc loc pat ts =
  103   case Map.lookup pat ts of
  104     Just t -> liftIO $ scPreterm sc t
  105     Nothing -> do
  106       bits <- forM pat $ \b -> do
  107         case Map.lookup [b] ts of
  108           Just t -> pure t
  109           Nothing -> throw $ YosysErrorNoSuchOutputBitvec (Text.pack $ show b) loc
  110       -- Yosys lists bits in little-endian order, while scVector expects big-endian, so reverse
  111       let ps = fusePreterms (reverse bits)
  112       liftIO $ scPreterms sc ps
  113 
  114 -- | Given a netgraph and an initial map from bit patterns to terms, populate that map with terms
  115 -- generated from the rest of the netgraph.
  116 netgraphToTerms ::
  117   (MonadIO m) =>
  118   SC.SharedContext ->
  119   Map Text ConvertedModule ->
  120   Netgraph ->
  121   Map [Bitrep] Preterm ->
  122   m (Map [Bitrep] Preterm)
  123 netgraphToTerms sc env ng inputs
  124   | length (Graph.scc $ ng ^. netgraphGraph) /= length (ng ^. netgraphGraph)
  125   = do
  126       throw $ YosysError "Network graph contains a cycle after splitting on DFFs; SAW does not currently support analysis of this circuit"
  127   | otherwise = do
  128       let sorted = reverseTopSort $ ng ^. netgraphGraph
  129       foldM
  130         ( \acc v -> do
  131             let (c, cnm, _deps) = ng ^. netgraphNodeFromVertex $ v
  132             let outputFields = Map.filter (\d -> d == DirectionOutput || d == DirectionInout) $ c ^. cellPortDirections
  133             if
  134               -- special handling for $dff/$ff nodes - we read their /output/ from the inputs map, and later detect and write their /input/ to the state
  135               | c ^. cellType == CellTypeDff
  136               , Just dffout <- Map.lookup "Q" $ c ^. cellConnections -> do
  137                   r <- lookupPatternTerm sc (YosysBitvecConsumerCell cnm "Q") dffout acc
  138                   ts <- deriveTermsByIndices sc dffout r
  139                   pure $ Map.union ts acc
  140               | c ^. cellType == CellTypeFf
  141               , Just ffout <- Map.lookup "Q" $ c ^. cellConnections -> do
  142                   r <- lookupPatternTerm sc (YosysBitvecConsumerCell cnm "Q") ffout acc
  143                   ts <- deriveTermsByIndices sc ffout r
  144                   pure $ Map.union ts acc
  145               | otherwise -> do
  146                   args <- fmap Map.fromList . forM (Map.assocs $ cellInputConnections c) $ \(inm, i) -> do -- for each input bit pattern
  147                     -- if we can find that pattern exactly, great! use that term
  148                     -- otherwise, find each individual bit and append the terms
  149                     (inm,) <$> lookupPatternTerm sc (YosysBitvecConsumerCell cnm inm) i acc
  150 
  151                   r <- primCellToTerm sc c args >>= \case
  152                     Just r -> pure r
  153                     Nothing ->
  154                       let submoduleName = asUserType $ c ^. cellType in
  155                       case Map.lookup submoduleName env of
  156                         Just cm -> do
  157                           r <- cryptolRecord sc args
  158                           liftIO $ SC.scApply sc (cm ^. convertedModuleTerm) r
  159                         Nothing ->
  160                             throw $ YosysErrorNoSuchSubmodule  submoduleName cnm
  161 
  162                   -- once we've built a term, insert it along with each of its bits
  163                   ts <- forM (Map.assocs $ cellOutputConnections c) $ \(o, out) -> do
  164                     t <- cryptolRecordSelect sc outputFields r o
  165                     deriveTermsByIndices sc out t
  166                   pure $ Map.union (Map.unions ts) acc
  167         )
  168         inputs
  169         sorted

  171 convertModule ::
  172   MonadIO m =>
  173   SC.SharedContext ->
  174   Map Text ConvertedModule ->
  175   Module ->
  176   m ConvertedModule
  177 convertModule sc env m = do
  178   let ng = moduleNetgraph m
  179 
  180   let inputPorts = moduleInputPorts m
  181   let outputPorts = moduleOutputPorts m
  182 
  183   inputFields <- forM inputPorts
  184     (\inp -> do
  185         liftIO . SC.scBitvector sc . fromIntegral $ length inp
  186     )
  187   outputFields <- forM outputPorts
  188     (\out -> do
  189         liftIO . SC.scBitvector sc . fromIntegral $ length out
  190     )
  191   inputRecordType <- cryptolRecordType sc inputFields
  192   outputRecordType <- cryptolRecordType sc outputFields
  193   inputRecord <- liftIO $ SC.scFreshVariable sc "input" inputRecordType
  194 
  195   derivedInputs <- forM (Map.assocs inputPorts) $ \(nm, inp) -> do
  196     t <- liftIO $ cryptolRecordSelect sc inputFields inputRecord nm
  197     deriveTermsByIndices sc inp t
  198 
  199   oneBitType <- liftIO $ SC.scBitvector sc 1
  200   xMsg <- liftIO $ SC.scString sc "Attempted to read X bit"
  201   xTerm <- liftIO $ SC.scGlobalApply sc (SC.mkIdent SC.preludeName "error") [oneBitType, xMsg]
  202   zMsg <- liftIO $ SC.scString sc "Attempted to read Z bit"
  203   zTerm <- liftIO $ SC.scGlobalApply sc (SC.mkIdent SC.preludeName "error") [oneBitType, zMsg]
  204   let inputs = Map.unions $ mconcat
  205         [ [ Map.fromList
  206             [ ( [BitrepZero], PretermBvNat 1 0 )
  207             , ( [BitrepOne], PretermBvNat 1 1 )
  208             , ( [BitrepX], PretermSlice 0 1 0 xTerm )
  209             , ( [BitrepZ], PretermSlice 0 1 0 zTerm )
  210             ]
  211           ]
  212         , derivedInputs
  213         ]
  214 
  215   terms <- netgraphToTerms sc env ng inputs
  216   outputRecord <- cryptolRecord sc =<< mapForWithKeyM outputPorts
  217     (\onm out -> lookupPatternTerm sc (YosysBitvecConsumerOutputPort onm) out terms)
  218 
  219   t <- liftIO $ SC.scAbstractTerms sc [inputRecord] outputRecord
  220   ty <- liftIO $ SC.scFun sc inputRecordType outputRecordType
  221 
  222   let toCryptol (nm, rep) = (C.mkIdent nm, C.tWord . C.tNum $ length rep)
  223   let cty = C.tFun
  224         (C.tRec . C.recordFromFields $ toCryptol <$> Map.assocs inputPorts)
  225         (C.tRec . C.recordFromFields $ toCryptol <$> Map.assocs outputPorts)
  226   pure ConvertedModule
  227     { _convertedModuleTerm = t
  228     , _convertedModuleType = ty
  229     , _convertedModuleCryptolType = cty
  230     }
