  104 -- *** PointsTo
  105 
  106 overlapPointsTo :: JVMPointsTo -> JVMPointsTo -> Bool
  107 overlapPointsTo =
  108   \case
  109     JVMPointsToField _ p1 f1 _ ->
  110       \case
  111         JVMPointsToField _ p2 f2 _ -> p1 == p2 && f1 == f2
  112         _                          -> False
  113     JVMPointsToStatic _ f1 _ ->
  114       \case
  115         JVMPointsToStatic _ f2 _   -> f1 == f2
  116         _                          -> False
  117     JVMPointsToElem _ p1 i1 _ ->
  118       \case
  119         JVMPointsToElem _ p2 i2 _  -> p1 == p2 && i1 == i2
  120         JVMPointsToArray _ p2 _    -> p1 == p2
  121         _                          -> False
  122     JVMPointsToArray _ p1 _ ->
  123       \case
  124         JVMPointsToElem _ p2 _ _   -> p1 == p2
  125         JVMPointsToArray _ p2 _    -> p1 == p2
  126         _                          -> False
  127 
  128 ppPointsTo :: JVMPointsTo -> PPL.Doc ann
  129 ppPointsTo =
  130   \case
  131     JVMPointsToField _loc ptr fid val ->
  132       MS.ppAllocIndex ptr <> PPL.pretty "." <> PPL.pretty (J.fieldIdName fid)
  133       PPL.<+> PPL.pretty "points to"
  134       PPL.<+> opt MS.ppSetupValue val
  135     JVMPointsToStatic _loc fid val ->
  136       PPL.pretty (J.unClassName (J.fieldIdClass fid)) <> PPL.pretty "." <> PPL.pretty (J.fieldIdName fid)
  137       PPL.<+> PPL.pretty "points to"
  138       PPL.<+> opt MS.ppSetupValue val
  139     JVMPointsToElem _loc ptr idx val ->
  140       MS.ppAllocIndex ptr <> PPL.pretty "[" <> PPL.pretty idx <> PPL.pretty "]"
  141       PPL.<+> PPL.pretty "points to"
  142       PPL.<+> opt MS.ppSetupValue val
  143     JVMPointsToArray _loc ptr val ->
  144       MS.ppAllocIndex ptr
  145       PPL.<+> PPL.pretty "points to"
  146       PPL.<+> opt MS.ppTypedTerm val
  147   where
  148     opt = maybe (PPL.pretty "<unspecified>")
  149 
  150 instance PPL.Pretty JVMPointsTo where
  151   pretty = ppPointsTo

  168 initialDefCrucibleMethodSpecIR ::
  169   CB.Codebase ->
  170   J.ClassName ->
  171   J.Method ->
  172   ProgramLoc ->
  173   MS.CrucibleMethodSpecIR CJ.JVM
  174 initialDefCrucibleMethodSpecIR cb cname method loc =
  175   let methId = JVMMethodId (J.methodKey method) cname
  176       retTy = J.methodReturnType method
  177       argTys = thisType ++ J.methodParameterTypes method
  178   in MS.makeCrucibleMethodSpecIR methId argTys retTy loc cb
  179   where thisType = if J.methodIsStatic method then [] else [J.ClassType cname]
  180 
  181 initialCrucibleSetupState ::
  182   JVMCrucibleContext ->
  183   (J.Class, J.Method) ->
  184   ProgramLoc ->
  185   Setup.CrucibleSetupState CJ.JVM
  186 initialCrucibleSetupState cc (cls, method) loc =
  187   Setup.makeCrucibleSetupState () cc $
  188     initialDefCrucibleMethodSpecIR
  189       (cc ^. jccCodebase)
  190       (J.className cls)
  191       method
  192       loc

  206 {-
  207 -- | Represent `CrucibleMethodSpecIR` as a function term in SAW-Core.
  208 methodSpecToTerm :: SharedContext -> CrucibleMethodSpecIR -> MaybeT IO Term
  209 methodSpecToTerm sc spec =
  210       -- 1. fill in the post-state user variable holes with final
  211       -- symbolic state
  212   let _ppts = _csPointsTos $ _csPostState $ instantiateUserVars spec
  213       -- 2. get the free variables in post points to's (note: these
  214       -- should be contained in variables bound by pre-points-tos)
  215 
  216       -- 3. abstract the free variables in each post-points-to
  217 
  218       -- 4. put every abstracted post-points-to in a tuple
  219 
  220       -- 5. Create struct type with fields being names of free variables
  221 
  222       -- 6. Create a lambda term bound to a struct-typed variable that returns the tuple
  223   in lift $ scLambda sc undefined undefined undefined
  224 
  225 -- | Rewrite the `csPostPointsTos` to substitute the elements of the
  226 -- final symbolic state for the fresh variables created by the user in
  227 -- the post-state.
  228 instantiateUserVars :: CrucibleMethodSpecIR -> CrucibleMethodSpecIR
  229 instantiateUserVars _spec = undefined
