   28 -- A stack trace is a list of call sites: positions with the name of
   29 -- (and other information about) the function being called.
   30 --
   31 -- As it's being collected, the trace has that form. For printing, we
   32 -- shift the positions by one so we report each call site as a
   33 -- position in the function that called it. This produces stack traces
   34 -- consistent with user expectations. See `prepareTrace` below.
   35 --
   36 -- The plumbing in the interpreter necessary to collect this
   37 -- information is not entirely trivial, thanks to the need to handle
   38 -- monadic actions, and particularly monadic actions passed as
   39 -- callbacks to builtins, some of which may not actually involve any
   40 -- direct evaluation in the interpreter, and is described elsewhere.
   41 -- (See Value.hs.)
   42 --
   43 -- The interaction of Haskell lazy evaluation with the interpreter's
   44 -- execution semantics is also complicated and difficult to reason
   45 -- about, which does not help.
   46 --
   47 -- In order to avoid a cyclic dependency with Value.hs we handle only
   48 -- Text in the stack trace itself; the printing happens when trace
   49 -- frames are created. Lazy evaluation should make this not ruinously
   50 -- expensive.
   51 
   52 -- | Call information for a stack trace.
   53 --
   54 --   This bundles a function name and arguments and whatever else we
   55 --   might want to record for examination (e.g. values of locals) in a
   56 --   single type so the rest of the trace logic doesn't have to think
   57 --   directly about what's in it.
   58 --
   59 --   Note that to avoid horrible cycles in the module dependency graph,
   60 --   we can't have Values here, so we need to print arguments, values
   61 --   of locals, etc. to text at the point we generate the frame. Lazy
   62 --   evaluation should prevent that from being ruinously expensive.
   63 --
   64 --   (FUTURE: it might make sense for each frame to be a prettyprinter
   65 --   document rather than Text.)
   66 newtype TraceFunc = TraceFunc Text
   67 
   68 -- | Entry in a stack trace.
   69 --
   70 -- In order to have a correct and coherent (and useful) stack trace,
   71 -- we want the following information for each call that is currently
   72 -- in progress:
   73 --    - the call site as a source position
   74 --    - the name and all other info (a TraceFunc) describing the
   75 --      function/action/builtin/other object being called
   76 --
   77 -- This should take effect at the point when the call is actually
   78 -- executed, that is, we begin executing in the function, and should
   79 -- be cleared off again when that's done.
   80 --
   81 -- We don't need the source position of the called object, because the
   82 -- next position in the stack should be in that function. (Tail call
   83 -- optimization would break that expectation, but we don't do that.)
   84 data TraceFrame = TraceFrame Pos TraceFunc
   85 
   86 -- A stack trace is a list of frames, with the most recent frame at
   87 -- the head of the list.
   88 data Trace = Trace [TraceFrame]
   89 
   90 -- | The starting trace.
   91 empty :: Trace
   92 empty = Trace []
   93 
   94 
   95 -- | Push a fresh frame on a trace.
   96 push :: Pos -> Text -> Trace -> Trace
   97 push callpos func (Trace frames) =
   98   let func' = TraceFunc func
   99       frame = TraceFrame callpos func'
  100   in
  101   Trace (frame : frames)
  102 
  103 -- | Pop a frame off a trace.
  104 pop :: Trace -> Trace
  105 pop (Trace frames) = case frames of
  106   [] -> panic "Trace.pop" ["Popping empty stack"]
  107   _ : frames' -> Trace frames'
  108

  110 -- | Print a TraceFunc. Simple for now, but we're going to want more
  111 --   later.
  112 ppTraceFunc :: TraceFunc -> Text
  113 ppTraceFunc (TraceFunc name) = name
  114 
  115 -- | Prepare a trace for printing.
  116 --
  117 --   The trace is a list of call sites, and that's not the way we want
  118 --   to print it.
  119 --
  120 --   Suppose we had this code:
  121 --
  122 --      foo.saw:1: let f1 z = ...;
  123 --      foo.saw:2: let f2 y = f1 (y + 1);
  124 --      foo.saw:3: let f3 x = f2 (x + 1);
  125 --      foo.saw.4: return (f3 0);
  126 --
  127 --   and we crash and produce a trace inside f1. (Assume none of this
  128 --   is monadic because that just introduces additional complications
  129 --   that aren't relevant here.)
  130 --
  131 --   The current position will be foo.saw:1, and the trace will contain:
  132 --      foo.saw:2 called f1
  133 --      foo.saw:3 called f2
  134 --      foo.saw:4 called f3
  135 --
  136 --   but what we want to print is
  137 --      foo.saw:1 in f1 (z = 2)
  138 --      foo.saw:2 in f2 (y = 1)
  139 --      foo.saw:3 in f3 (x = 0)
  140 --      foo.saw:4 at top level
  141 --
  142 --   that is, we want to insert the current position and shift the
  143 --   rest of the positions up by one. (Failure to do this coherently
  144 --   was one of the root causes of #2108.)
  145 --
  146 -- Do this shift, and inject Nothing for "at top level".
  147 prepareTrace :: Pos -> [TraceFrame] -> [(Pos, Maybe TraceFunc)]
  148 prepareTrace curpos tfs = case tfs of
  149     [] -> [(curpos, Nothing)]
  150     (TraceFrame nextpos func) : tfs' ->
  151         (curpos, Just func) : prepareTrace nextpos tfs'
  152 
  153 -- | Pretty-print a trace. This also takes the current position
  154 --   to print at the bottom.
  155 --
  156 --   We don't reverse the frame, so the most recent call comes out at
  157 --   the top, like gdb and most debuggers. Shift the current position
  158 --   into the trace as described above before printing.
  159 --
  160 --   Takes the current position as well as the trace.  Note that the
  161 --   "current position" should be read as "what's currently
  162 --   executing", which in most cases will be PosInsideBuiltin. Don't
  163 --   use the interpreter's last execution position; that will result
  164 --   in a confusingly messed-up trace.
  165 --
  166 ppTrace :: Trace -> Pos -> Text
  167 ppTrace (Trace frames) curpos =
  168   let frames' =
  169         let ppEntry (pos, mfunc) =
  170               let pos' = Text.pack (show pos)
  171                   mfunc' = case mfunc of
  172                     Nothing -> " (at top level)"
  173                     Just func -> " in " <> ppTraceFunc func
  174               in
  175               "   " <> pos' <> mfunc'
  176         in
  177         map ppEntry $ prepareTrace curpos frames
  178   in
  179   Text.unlines frames'
