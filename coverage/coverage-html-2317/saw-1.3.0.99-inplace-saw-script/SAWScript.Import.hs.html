   27 loadFile :: Options -> FilePath -> IO [Stmt]
   28 loadFile opts fname = do
   29   printOutLn opts Info $ "Loading file " ++ show fname
   30   ftext <- TextIO.readFile fname
   31   let (tokens, optmsg) = lexSAW fname ftext
   32   case optmsg of
   33       Nothing -> return ()
   34       Just (vrb, pos, txt) -> do
   35           -- XXX: the print functions should take care of printing the position
   36           -- (clean this up when we clean out the printing infrastructure)
   37           let txt' = show pos ++ ": " ++ Text.unpack txt
   38           -- XXX: the print functions should also take care of exiting on an error
   39           -- (immediately or later). For now, throw errors and print anything else.
   40           case vrb of
   41               Error -> throwIO $ userError txt'
   42               _ -> printOutLn opts vrb txt'
   43   either throwIO return (parseFile tokens)
   44 
   45 parseFile :: [Token Pos] -> Either ParseError [Stmt]
   46 parseFile tokens = do
   47   case parseModule tokens of
   48     Left err -> Left err
   49     Right stmts -> Right stmts
   50 
   51 findAndLoadFile :: Options -> FilePath -> IO [Stmt]
   52 findAndLoadFile opts fp = do
   53   let paths = importPath opts
   54   mfname <- findFile paths fp
   55   case mfname of
   56     Nothing -> fail $ unlines $
   57         [ "Couldn't find file: " ++ show fp
   58         , "  Searched in directories:"
   59         ] ++ map ("    " ++) paths
   60     Just fname -> loadFile opts fname
   61 
   62 #if __GLASGOW_HASKELL__ < 706
   63 findFile :: [FilePath] -> String -> IO (Maybe FilePath)
   64 findFile paths fileName = search paths
   65   where
   66     search :: [FilePath] -> IO (Maybe FilePath)
   67     search [] = return Nothing
   68     search (d:ds) = do
   69         let path = d </> fileName
   70         b <- doesFileExist path
   71         if b then return (Just path)
   72              else search ds
   73 #endif
