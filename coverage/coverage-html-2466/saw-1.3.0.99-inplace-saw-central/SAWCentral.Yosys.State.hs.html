   45 -- | A SAWCore translation of an HDL module alongside some type
   46 -- information that is useful to keep around.
   47 data YosysSequential = YosysSequential
   48   { _yosysSequentialTerm :: SC.TypedTerm
   49     -- ^ The "raw" SAWCore term derived from the module, which
   50     -- includes a __state__ field in the input and output.
   51   , _yosysSequentialStateFields :: Map Text (SC.Term, C.Type)
   52     -- ^ A mapping from each state field name to a SAWCore and Cryptol type.
   53   , _yosysSequentialInputFields :: Map Text (SC.Term, C.Type)
   54     -- ^ A mapping from each input to a SAWCore and Cryptol type.
   55   , _yosysSequentialOutputFields :: Map Text (SC.Term, C.Type)
   56     -- ^ A mapping from each output to a SAWCore and Cryptol type.
   57   , _yosysSequentialInputWidths :: Map Text Natural
   58     -- ^ A mapping from each input to a width.
   59   , _yosysSequentialOutputWidths :: Map Text Natural
   60     -- ^ A mapping from each output to a width.
   61   , _yosysSequentialStateWidths :: Map Text Natural
   62     -- ^ A mapping from each state field to a width.
   63   }
   64 makeLenses ''YosysSequential

   66 -- | Add a record-typed field named __states__ to the given mapping of
   67 -- field names to types.
   68 insertStateField ::
   69   MonadIO m =>
   70   SC.SharedContext ->
   71   Map Text (SC.Term, C.Type) {- ^ The field types of "__states__" -} ->
   72   Map Text (SC.Term, C.Type) {- ^ The mapping to update -} ->
   73   m (Map Text (SC.Term, C.Type))
   74 insertStateField sc stateFields fields = do
   75   stateRecordType <- fieldsToType sc stateFields
   76   stateRecordCryptolType <- fieldsToCryptolType stateFields
   77   pure $ Map.insert "__state__" (stateRecordType, stateRecordCryptolType) fields

   79 -- | Translate a stateful HDL module into SAWCore
   80 convertModuleInline ::
   81   MonadIO m =>
   82   SC.SharedContext ->
   83   Module ->
   84   m YosysSequential
   85 convertModuleInline sc m = do
   86   let ng = moduleNetgraph m
   87 
   88   let netnames =
   89         Map.fromList
   90         [ (n ^. netnameBits, t)
   91         | (t, n) <- Map.assocs (m ^. moduleNetnames), not (n ^. netnameHideName) ]
   92 
   93   let bestName t c =
   94         fromMaybe (cellIdentifier t) $
   95         do bs <- Map.lookup "Q" (c ^. cellConnections)
   96            Map.lookup bs netnames
   97 
   98   -- construct SAWCore and Cryptol types
   99   let dffs =
  100         Map.fromList
  101         [ (bestName t c, c) | (t, c) <- Map.assocs (m ^. moduleCells), cellIsRegister c ]
  102 
  103   stateWidths <- forM dffs $ \c ->
  104     case Map.lookup "Q" $ c ^. cellConnections of
  105       Nothing -> panic "convertModuleInline" ["Missing expected output name for $dff cell"]
  106       Just b -> pure . fromIntegral $ length b
  107 
  108   stateFields <- forM stateWidths $ \w -> do
  109     t <- liftIO $ SC.scBitvector sc w
  110     let cty = C.tWord $ C.tNum w
  111     pure (t, cty)
  112 
  113   let inputPorts = moduleInputPorts m
  114   let outputPorts = moduleOutputPorts m
  115   inputFields <- forM inputPorts $ \inp -> do
  116     ty <- liftIO . SC.scBitvector sc . fromIntegral $ length inp
  117     let cty = C.tWord . C.tNum $ length inp
  118     pure (ty, cty)
  119   outputFields <- forM outputPorts $ \out -> do
  120     ty <- liftIO . SC.scBitvector sc . fromIntegral $ length out
  121     let cty = C.tWord . C.tNum $ length out
  122     pure (ty, cty)
  123 
  124   domainFields <- insertStateField sc stateFields inputFields
  125   codomainFields <- insertStateField sc stateFields outputFields
  126 
  127   domainRecordType <- fieldsToType sc domainFields
  128   domainCryptolRecordType <- fieldsToCryptolType domainFields
  129   -- codomainRecordType <- fieldsToType sc codomainFields
  130   codomainCryptolRecordType <- fieldsToCryptolType codomainFields
  131 
  132   -- convert module into term
  133   domainRecordEC <- liftIO $ SC.scFreshEC sc "input" domainRecordType
  134   domainRecord <- liftIO $ SC.scExtCns sc domainRecordEC
  135 
  136   derivedInputs <- forM (Map.assocs inputPorts) $ \(nm, inp) -> do
  137     t <- liftIO $ cryptolRecordSelect sc domainFields domainRecord nm
  138     deriveTermsByIndices sc inp t
  139 
  140   preStateRecord <- liftIO $ cryptolRecordSelect sc domainFields domainRecord "__state__"
  141   derivedPreState <- forM (Map.assocs dffs) $ \(cnm, c) ->
  142     case Map.lookup "Q" $ c ^. cellConnections of
  143       Nothing -> panic "convertModuleInline" ["Missing expected output name for $dff cell"]
  144       Just b -> do
  145         t <- liftIO $ cryptolRecordSelect sc stateFields preStateRecord cnm
  146         deriveTermsByIndices sc b t
  147 
  148   oneBitType <- liftIO $ SC.scBitvector sc 1
  149   xMsg <- liftIO $ SC.scString sc "Attempted to read X bit"
  150   xTerm <- liftIO $ SC.scGlobalApply sc (SC.mkIdent SC.preludeName "error") [oneBitType, xMsg]
  151   zMsg <- liftIO $ SC.scString sc "Attempted to read Z bit"
  152   zTerm <- liftIO $ SC.scGlobalApply sc (SC.mkIdent SC.preludeName "error") [oneBitType, zMsg]
  153   let inputs = Map.unions $ mconcat
  154         [ [ Map.fromList
  155             [ ( [BitrepZero], PretermBvNat 1 0 )
  156             , ( [BitrepOne], PretermBvNat 1 1 )
  157             , ( [BitrepX], PretermSlice 0 1 0 xTerm )
  158             , ( [BitrepZ], PretermSlice 0 1 0 zTerm )
  159             ]
  160           ]
  161         , derivedInputs
  162         , derivedPreState
  163         ]
  164 
  165   terms <- netgraphToTerms sc Map.empty ng inputs
  166 
  167   postStateFields <- mapForWithKeyM dffs $ \cnm c ->
  168     case Map.lookup "D" $ c ^. cellConnections of
  169       Nothing -> panic "convertModuleInline" ["Missing expected input name for $dff cell"]
  170       Just b -> do
  171         t <- lookupPatternTerm sc (YosysBitvecConsumerCell cnm "D") b terms
  172         pure t
  173   postStateRecord <- cryptolRecord sc postStateFields
  174 
  175   outputRecord <- cryptolRecord sc . Map.insert "__state__" postStateRecord =<< mapForWithKeyM outputPorts
  176     (\onm out -> lookupPatternTerm sc (YosysBitvecConsumerOutputPort onm) out terms)
  177 
  178   -- construct result
  179   t <- liftIO $ SC.scAbstractExts sc [domainRecordEC] outputRecord
  180   -- ty <- liftIO $ SC.scFun sc domainRecordType codomainRecordType
  181   _ <- validateTerm sc "translating a sequential circuit" t
  182   let cty = C.tFun domainCryptolRecordType codomainCryptolRecordType
  183   pure YosysSequential
  184     { _yosysSequentialTerm = SC.TypedTerm (SC.TypedTermSchema $ C.tMono cty) t
  185     , _yosysSequentialStateFields = stateFields
  186     , _yosysSequentialInputFields = inputFields
  187     , _yosysSequentialOutputFields = outputFields
  188     , _yosysSequentialInputWidths = fromIntegral . length <$> inputPorts
  189     , _yosysSequentialOutputWidths = fromIntegral . length <$> outputPorts
  190     , _yosysSequentialStateWidths = stateWidths
  191     }

  193 -- | Given a SAWCore term with an explicit state, iterate the term the
  194 -- given number of times. The resulting term has a parameter for the
  195 -- initial state, the resulting Cryptol types does not.
  196 composeYosysSequentialHelper ::
  197   forall m.
  198   MonadIO m =>
  199   SC.SharedContext ->
  200   YosysSequential ->
  201   Integer ->
  202   m (SC.Term, C.Type)
  203 composeYosysSequentialHelper sc s n = do
  204   let t = SC.ttTerm $ s ^. yosysSequentialTerm
  205 
  206   width <- liftIO . SC.scNat sc $ fromIntegral n
  207   extendedInputFields <- forM (s ^. yosysSequentialInputFields) $ \(ty, cty) -> do
  208     exty <- liftIO $ SC.scVecType sc width ty
  209     let excty = C.tSeq (C.tNum n) cty
  210     pure (exty, excty)
  211   extendedOutputFields <- forM (s ^. yosysSequentialOutputFields) $ \(ty, cty) -> do
  212     exty <- liftIO $ SC.scVecType sc width ty
  213     let excty = C.tSeq (C.tNum n) cty
  214     pure (exty, excty)
  215   extendedInputType <- fieldsToType sc extendedInputFields
  216   extendedInputCryptolType <- fieldsToCryptolType extendedInputFields
  217   extendedInputRecordEC <- liftIO $ SC.scFreshEC sc "input" extendedInputType
  218   extendedInputRecord <- liftIO $ SC.scExtCns sc extendedInputRecordEC
  219   extendedOutputCryptolType <- fieldsToCryptolType extendedOutputFields
  220 
  221   allInputs <- fmap Map.fromList . forM (Map.keys extendedInputFields) $ \nm -> do
  222     inp <- liftIO $ cryptolRecordSelect sc extendedInputFields extendedInputRecord nm
  223     pure (nm, inp)
  224 
  225   codomainFields <- insertStateField sc (s ^. yosysSequentialStateFields) $ s ^. yosysSequentialOutputFields
  226 
  227   let
  228     buildIntermediateInput :: Integer -> SC.Term -> m SC.Term
  229     buildIntermediateInput i st = do
  230       inps <- fmap Map.fromList . forM (Map.assocs allInputs) $ \(nm, inp) -> do
  231         case Map.lookup nm $ s ^. yosysSequentialInputFields of
  232           Nothing -> throw . YosysError $ "Invalid input: " <> nm
  233           Just (elemty, _) -> do
  234             idx <- liftIO . SC.scNat sc $ fromIntegral i
  235             idxed <- liftIO $ SC.scAt sc width elemty inp idx
  236             pure (nm, idxed)
  237       let inpsWithSt = Map.insert "__state__" st inps
  238       cryptolRecord sc inpsWithSt
  239 
  240     summarizeOutput :: SC.Term -> m (SC.Term, Map Text SC.Term)
  241     summarizeOutput outrec = do
  242       outstate <- liftIO $ cryptolRecordSelect sc codomainFields outrec "__state__"
  243       outputs <- fmap Map.fromList . forM (Map.assocs $ s ^. yosysSequentialOutputFields) $ \(nm, (ty, _)) -> do
  244         out <- liftIO $ cryptolRecordSelect sc codomainFields outrec nm
  245         wrapped <- liftIO $ SC.scSingle sc ty out
  246         pure (nm, wrapped)
  247       pure (outstate, outputs)
  248 
  249     compose1 :: Integer -> (SC.Term, Map Text SC.Term) -> m (SC.Term, Map Text SC.Term)
  250     compose1 i (st, outs) = do
  251       inprec <- buildIntermediateInput i st
  252       outrec <- liftIO $ SC.scApply sc t inprec
  253       (st', outs') <- summarizeOutput outrec
  254       mergedOuts <- fmap Map.fromList . forM (Map.assocs outs') $ \(nm, arr) -> do
  255         case (Map.lookup nm $ s ^. yosysSequentialOutputFields, Map.lookup nm outs) of
  256           (Just (ty, _), Just rest) -> do
  257             restlen <- liftIO . SC.scNat sc $ fromIntegral i
  258             arrlen <- liftIO $ SC.scNat sc 1
  259             appended <- liftIO $ SC.scAppend sc restlen arrlen ty rest arr
  260             pure (nm, appended)
  261           _ -> pure (nm, arr)
  262       pure (st', mergedOuts)
  263 
  264   stateType <- fieldsToType sc $ s ^. yosysSequentialStateFields
  265   initialStateEC <- liftIO $ SC.scFreshEC sc "initial_state" stateType
  266   initialState <- liftIO $ SC.scExtCns sc initialStateEC
  267   (_, outputs) <- foldM (\acc i -> compose1 i acc) (initialState, Map.empty) [0..n]
  268 
  269   outputRecord <- cryptolRecord sc outputs
  270   res <- liftIO $ SC.scAbstractExts sc [initialStateEC, extendedInputRecordEC] outputRecord
  271   let cty = C.tFun extendedInputCryptolType extendedOutputCryptolType
  272 
  273   pure (res, cty)

  275 -- | Given a SAWCore term with an explicit state, iterate the term the
  276 -- given number of times. Accessing the initial state produces an
  277 -- error.
  278 composeYosysSequential ::
  279   forall m.
  280   MonadIO m =>
  281   SC.SharedContext ->
  282   YosysSequential ->
  283   Integer ->
  284   m SC.TypedTerm
  285 composeYosysSequential sc s n = do
  286   (t, cty) <- composeYosysSequentialHelper sc s n
  287   stateType <- fieldsToType sc $ s ^. yosysSequentialStateFields
  288   initialStateMsg <- liftIO $ SC.scString sc "Attempted to read initial state of sequential circuit"
  289   initialState <- liftIO $ SC.scGlobalApply sc (SC.mkIdent SC.preludeName "error") [stateType, initialStateMsg]
  290   res <- liftIO $ SC.scApply sc t initialState
  291   pure $ SC.TypedTerm (SC.TypedTermSchema $ C.tMono cty) res

  293 -- | Given a SAWCore term with an explicit state, iterate the term the
  294 -- given number of times. The resulting term has a parameter for the
  295 -- initial state.
  296 composeYosysSequentialWithState ::
  297   forall m.
  298   MonadIO m =>
  299   SC.SharedContext ->
  300   YosysSequential ->
  301   Integer ->
  302   m SC.TypedTerm
  303 composeYosysSequentialWithState sc s n = do
  304   (t, cty) <- composeYosysSequentialHelper sc s n
  305   scty <- fieldsToCryptolType $ s ^. yosysSequentialStateFields
  306   pure $ SC.TypedTerm (SC.TypedTermSchema . C.tMono $ C.tFun scty cty) t
