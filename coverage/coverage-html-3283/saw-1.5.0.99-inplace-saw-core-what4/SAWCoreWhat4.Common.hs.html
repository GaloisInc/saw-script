  107 termOfTValue :: SharedContext -> TValue (What4 sym) -> IO Term
  108 termOfTValue sc val =
  109   case val of
  110     VBoolType -> scBoolType sc
  111     VIntType -> scIntegerType sc
  112     VRationalType -> scRationalType sc
  113     VVecType n a ->
  114       do n' <- scNat sc n
  115          a' <- termOfTValue sc a
  116          scVecType sc n' a'
  117     VDataType (ModuleIdentifier "Prelude.UnitType") [] []
  118       -> scUnitType sc
  119     VDataType (ModuleIdentifier "Prelude.PairType") [TValue a, TValue b] []
  120       -> do a' <- termOfTValue sc a
  121             b' <- termOfTValue sc b
  122             scPairType sc a' b'
  123     VDataType (ModuleIdentifier "Prelude.EmptyType") [] []
  124       -> scRecordType sc []
  125     VDataType (ModuleIdentifier "Prelude.RecordType")
  126       [VString fname, TValue a, TValue b] []
  127       -> do fname' <- scString sc fname
  128             a' <- termOfTValue sc a
  129             b' <- termOfTValue sc b
  130             scGlobalApply sc "Prelude.RecordType" [fname', a', b']
  131     _ -> fail $ "termOfTValue: " ++ show val
  132 
  133 termOfSValue :: SharedContext -> SValue sym -> IO Term
  134 termOfSValue sc val =
  135   case val of
  136     VCtorApp 0 _ []
  137       -> scUnitValue sc
  138     VNat n
  139       -> scNat sc n
  140     TValue tv -> termOfTValue sc tv
  141     _ -> fail $ "termOfSValue: " ++ show val
