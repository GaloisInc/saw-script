   33 -- | Combines LLVM Modules into a single, composite Module.  This is akin to
   34 -- linking, but just from the perspective of what is needed for SAW/Cryptol
   35 -- analysis.
   36 --
   37 -- Note that there is a Semigroup instance for 'Module', which is necessary for
   38 -- the @llvm_pretty@ implementation of the @LLVM@ Monad containing a @WriterT
   39 -- Module@, but it should *only* be used for building a 'Module' from elements
   40 -- that do *not* have unresolved cross-Module symbols or other considerations.
   41 -- The implementation here provides proper combination of independent Modules,
   42 -- allowing symbol references in one module to be resolved by definitions in the
   43 -- other, etc.
   44 llvmModuleCombine :: Module -> Module -> Module
   45 llvmModuleCombine a addModule =
   46   let defs = a ^. modDefinesLens
   47       decls = a ^. modDeclaresLens
   48       newDefs = b ^. modDefinesLens
   49       newDecls = b ^. modDeclaresLens
   50       rmvDefined = flip (foldr removeDefined)
   51       newDeclsLessOldDefs = rmvDefined defs newDecls
   52       oldDeclsLessNewDefs = rmvDefined newDefs decls
   53       joinedName n = Just $ fromMaybe "..." n <> "+" <> fromMaybe "..." (modSourceName b)
   54       newUmdBase = let umIdxs = umIndex <$> modUnnamedMd a
   55                    in bool (UnnamedMdIdx 0) (succ $ maximum umIdxs) $ null umIdxs
   56       -- unnamed metadata is referenced almost everywhere, so update that globally
   57       -- first:
   58       b = updateUmd newUmdBase addModule
   59   in a
   60      & modSourceNameLens %~ joinedName
   61      & modDeclaresLens .~ (oldDeclsLessNewDefs <> newDeclsLessOldDefs)
   62      & modDefinesLens %~ deConflict (b ^. modDefinesLens)
   63      & modTypesLens <>~ b ^. modTypesLens
   64      & modNamedMdLens <>~ b ^. modNamedMdLens
   65      & modComdatLens <>~ b ^. modComdatLens
   66      & modGlobalsLens <>~ b ^. modGlobalsLens
   67      & modInlineAsmLens <>~ b ^. modInlineAsmLens
   68      & modAliasesLens <>~ b ^. modAliasesLens
   69   -- TODO Globals any should override Linkage external for the same name
   70   -- TODO extend modUnnamedMd and propagate those changes
   71   -- TODO verify modTriple and modDataLayout are the same?
   72

   74 -- | A Define takes precedence over a Declare.  When combining modules, module A
   75 -- may Declare a function that is handled by a Define in module B, so get rid of
   76 -- the Declare when putting A and B together.
   77 
   78 removeDefined :: Define -> [Declare] -> [Declare]
   79 removeDefined def = filter ((def ^. defNameLens /=) . view decNameLens)
   80 
   81 
   82 -- | Module A and Module B may have a Define with the same name (Symbol).  This
   83 -- is normal when linking multiple modules together, and is resolved by linkers
   84 -- as guided by the Linkage information for the two Definitions, usually by
   85 -- either renaming or merging.
   86 
   87 deConflict :: [Define] -> [Define] -> [Define]
   88 deConflict new curr = uncurry (<>) $ foldl deConflictDef (curr, new) new
   89   where
   90     deConflictDef (ads, bds) bd =
   91       case find (((==) `on` defName) bd) ads of
   92         Nothing -> (ads, bds)
   93         Just ad -> handle ads bds ad bd
   94     handle ads bds ad bd =
   95       case bd ^. defLinkageLens of
   96         Just Private -> (ads, renameDef bd (view defNameLens <$> ads) bds)
   97         Just LinkerPrivate -> (ads, renameDef bd (view defNameLens <$> ads) bds)
   98         Just LinkerPrivateWeak ->
   99           (ads, renameDef bd (view defNameLens <$> ads) bds) -- ??
  100         Just LinkerPrivateWeakDefAuto ->
  101           (ads, renameDef bd (view defNameLens <$> ads) bds) -- ??
  102         Just Internal -> (ads, renameDef bd (view defNameLens <$> ads) bds)
  103         Just AvailableExternally ->
  104           -- Never happen: not allowed on defines.  Ignore
  105           (ads, bds)
  106         Just Linkonce -> (mergeDef ad bd ads, removeDef bd bds)
  107         Just Weak -> (mergeDef ad bd ads, removeDef bd bds)
  108         Just Common -> (mergeDef ad bd ads, removeDef bd bds)
  109         Just ExternWeak -> (mergeDef ad bd ads, removeDef bd bds)
  110         Just LinkonceODR -> (mergeDef ad bd ads, removeDef bd bds)
  111         Just WeakODR -> (mergeDef ad bd ads, removeDef bd bds)
  112         Just Appending -> (appendDef ad bd ads, removeDef bd bds)
  113         Just External ->
  114           -- This should never happen: it is truly a symbol conflict.  A
  115           -- linker would reject this, but here we will just preserve the
  116           -- original.
  117           (ads, removeDef bd bds)
  118         Just DLLImport -> (ads, removeDef bd bds) -- ??
  119         Just DLLExport -> (ads, removeDef bd bds) -- ??
  120         Nothing ->
  121           -- No linkage specified.  The default is 'External', with associated
  122           -- considerations as documented for that case above.
  123           (ads, removeDef bd bds)
  124 
  125 -- Note: Used for Linkonce, Weak, Common, ExternWeak, LinkonceODR, WeakODR. LLVM
  126 -- docs say "merged", but also indicates that maybe there is a replacement
  127 -- instead?  For now, treat "merged" as appending.
  128 mergeDef :: Define -> Define -> [Define] -> [Define]
  129 mergeDef = appendDef
  130 
  131 appendDef :: Define -> Define -> [Define] -> [Define]
  132 appendDef d1 d2 =
  133   let appenD = d1 & defBodyLens <>~ d2 ^. defBodyLens
  134   in (appenD :) . filter (((/=) `on` defName) d1)
  135 
  136 removeDef :: Define -> [Define] -> [Define]
  137 removeDef d = filter (((/=) `on` defName) d)

  140 -- Renames the Defined symbol to a new name using a discriminator to avoid a
  141 -- conflict.  Only valid for renaming Private/Internal Defines such that changing
  142 -- any reference to the original Symbol to the new Symbol in the provided set of
  143 -- Defines is sufficient to change all references.  Note therefore this excludes:
  144 -- renaming of global variables, changing a GlobalAlias.
  145 
  146 renameDef :: Define -> [Symbol] -> [Define] -> [Define]
  147 renameDef toRename known inDefs =
  148   -- KWQ TODO: needs to change GlobalAlias aliasName?
  149   --
  150   let getNewName nm n =
  151         -- Note: adds a "discriminator" to the name in a way that is valid for
  152         -- both C functions and C++ mangled names (see
  153         -- https://itanium-cxx-abi.github.io/cxx-abi/abi.html#mangling-scope).
  154         let nn = if n < 10
  155                  then nm <> "_" <> show n
  156                  else nm <> "__" <> show n <> "_"
  157         in case find ((fromString nn ==) . defName) inDefs of
  158              Just _ -> getNewName nm $ succ n
  159              Nothing ->
  160                if fromString nn `elem` known
  161                then getNewName nm $ succ n
  162                else nn
  163       (Symbol oldname) = defName toRename
  164       newName = Symbol $ getNewName oldname (1 :: Integer)
  165   in changeSym (defName toRename) newName inDefs
  166 
  167 
  168 changeSym :: Symbol -> Symbol -> [Define] -> [Define]
  169 changeSym old new = biplate %~ chngSym
  170   where
  171     chngSym s = bool s new $ old == s

  173 -- | Adjusts all unnamed metadata indices in the Module to begin at the specified
  174 -- newBase, which allows this module to be combined without conflict with a
  175 -- module whose metadata indices are all below the newBase.
  176 updateUmd :: UnnamedMdIdx -> Module -> Module
  177 updateUmd newBase = biplate +~ newBase
