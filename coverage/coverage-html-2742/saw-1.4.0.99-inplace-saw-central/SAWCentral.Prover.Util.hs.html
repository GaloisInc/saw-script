   11 -- | Is this a bool, or something that returns bool.
   12 checkBooleanType :: C.Type -> IO ()
   13 checkBooleanType ty
   14   | C.tIsBit ty                 = return ()
   15   | Just (_,ty') <- C.tIsFun ty = checkBooleanType ty'
   16   | otherwise = fail ("Invalid non-boolean type: " ++ pretty ty)
   17 
   18 -- | Make sure that this schema is monomorphic, and either boolean,
   19 -- or something that returns a boolean.
   20 checkBooleanSchema :: TypedTermType -> IO ()
   21 checkBooleanSchema (TypedTermSchema (C.Forall [] [] t)) = checkBooleanType t
   22 checkBooleanSchema (TypedTermSchema s) = fail ("Invalid polymorphic type: " ++ pretty s)
   23 checkBooleanSchema tp =
   24   fail ("Expected boolean type, but got " ++ show (ppTypedTermType tp))

   29 liftCexBB :: [FiniteType] -> [Bool] -> Either String [FiniteValue]
   30 liftCexBB tys bs =
   31   case readFiniteValues tys bs of
   32     Nothing -> Left "Failed to lift counterexample"
   33     Just fvs -> Right fvs
   34 
   35 -- | Lift a counterexample containing little-endian words
   36 liftLECexBB :: [FiniteType] -> [Bool] -> Either String [FiniteValue]
   37 liftLECexBB tys bs =
   38   case readFiniteValuesLE tys bs of
   39     Nothing -> Left "Failed to lift counterexample"
