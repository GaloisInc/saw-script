   57 revMap :: (Ord k, Ord u) => (v -> Maybe u) -> Map k v -> Map u k
   58 revMap f m = 
   59   let 
   60     go (cnm,t) = case f t of
   61       Just snm -> Just (snm,cnm)
   62       Nothing -> Nothing
   63   in Map.fromList $ mapMaybe go (Map.toList m)

   66 extraPrims :: C.PrimMap -> [(SAW.Ident, C.Name)]
   67 extraPrims pm = map go
   68   [ ("Prelude.Integer", "Integer")
   69   , ("Prelude.Bool", "Bit")
   70   , ("Cryptol.PIntegral", "Integral")
   71   , ("Cryptol.PRing", "Ring")
   72   , ("Cryptol.PLiteral", "Literal")
   73   , ("Cryptol.PEq", "Eq")
   74   , ("Cryptol.PCmp", "Cmp")
   75   ]
   76   where
   77     go (x,txt) = (x, C.lookupPrimType (C.prelPrim txt) pm)

   79 initTTEnv :: SharedContext -> CryptolEnv -> IO TTEnv
   80 initTTEnv sc env = case lookupModule C.preludeName (CrySAW.eModuleEnv env) of
   81   Just prelude ->
   82     let 
   83       pmap = ifacePrimMap $ lmInterface prelude
   84       exprPrims = fmap (\x -> C.lookupPrimDecl x pmap) $ revMap asConstant (CrySAW.ePrims env)
   85       typePrims = fmap (\x -> C.lookupPrimType x pmap) $ revMap asConstant (CrySAW.ePrimTypes env)
   86     in return $
   87       TTEnv 
   88         { ttEnvVars = Bimap.empty
   89         , ttConstMap = Map.unions [revMap asConstant (CrySAW.eTermEnv env), exprPrims, typePrims]
   90         , ttExtras = Map.fromList (extraPrims pmap)
   91         , ttCryEnv = env
   92         , ttSc = sc
   93         }
   94   Nothing -> fail "initTTEnv: missing Cryptol prelude"

   96 checkConvertible :: Term -> Term -> TT ()
   97 checkConvertible t1 t2 = do
   98   sc <- asks ttSc
   99   (liftIO $ scConvertible sc t1 t2) >>= \case
  100     True -> return ()
  101     False -> do
  102       let ppts opts = do
  103             t1' <- liftIO $ prettyTerm sc opts t1
  104             t2' <- liftIO $ prettyTerm sc opts t2
  105             return $ PP.vcat [t1', PP.indent 2 "vs.",t2']
  106       withContext (CallContext  "checkConvertible" ppts) $ 
  107         fail "Terms are not convertible"

  112 revTopProofs :: C.Expr -> C.Expr
  113 revTopProofs = go []
  114   where
  115     unwind prfs e = case prfs of
  116       x:xs -> C.EProofAbs x (unwind xs e)
  117       [] -> e
  118     go prfs = \case
  119       C.ETAbs tp e -> C.ETAbs tp (go prfs e)
  120       C.ELocated loc e -> C.ELocated loc (go prfs e)
  121       C.EProofAbs prf e -> go (prf:prfs) e
  122       e -> unwind prfs e

  124 -- | SAW sometimes reverses the guards when importing expressions, in which case we need to reverse the
  125 --   result after type inference in order to recover the original type/term
  126 revGuards :: (C.Schema, C.Expr) -> (C.Schema, C.Expr)
  127 revGuards (s,e) = (s { C.sProps = reverse (C.sProps s)}, revTopProofs e)
  128 
  129 validateImport :: Term -> (C.Schema, C.Expr) -> TT (C.Schema, C.Expr)
  130 validateImport t (s, e) = do
  131   cenv <- asks ttCryEnv
  132   sc <- asks ttSc
  133   cryenv <- liftIO $ let ?fileReader = BS.readFile in CrySAW.mkCryEnv cenv
  134   s' <- liftIO $ CrySAW.importSchema sc cryenv s
  135   e' <- liftIO $ CrySAW.importExpr sc cryenv e
  136   tT <- liftIO $ scTypeOf sc t
  137   checkConvertible tT s'
  138   checkConvertible e' t
  139   return (s,e)

  141 inferSchemaExpr :: Term -> TT (C.Schema, C.Expr)
  142 inferSchemaExpr t = let ?fileReader = BS.readFile in do
  143   pe <- translateAsExpr t
  144   cenv <- asks ttCryEnv
  145   (r,cenv') <- liftIO $ CrySAW.inferExpr cenv pe
  146   -- the order of the guards is somewhat inconsistent, so we try
  147   -- either the original or reverse orderings and return the one that validates
  148   local (\env -> env { ttCryEnv = cenv' }) $
  149     validateImport t r <|> validateImport t (revGuards r)

  151 -- | Attempt to convert a SAWCore term into an equivalent Cryptol expression and corresponding
  152 --   schema. Only expected to work for terms that are the immediate result of importing a Cryptol
  153 --   expression, or a simple transformation of one.
  154 termToSchemaExpr :: 
  155  SharedContext -> CryptolEnv -> Term -> IO (Either TTError (C.Schema, C.Expr))
  156 termToSchemaExpr sc cenv t = do
  157   env <- initTTEnv sc cenv
  158   runTT env $ inferSchemaExpr t

  160 lookupPrim :: Text -> TT Term
  161 lookupPrim nm = do
  162   cenv <- asks ttCryEnv
  163   prelude <- mreturn $ lookupModule C.preludeName (CrySAW.eModuleEnv cenv)
  164   let pmap = ifacePrimMap $ lmInterface prelude
  165   let pnm = C.prelPrim nm
  166   cnm <- mreturn $ Map.lookup pnm (C.primDecls pmap)
  167   mreturn $ Map.lookup cnm (eTermEnv cenv)

  169 propToLambda :: Term -> TT Term
  170 propToLambda t = withTermContext "propToLambda" t $ go t
  171   where
  172     go :: Term -> TT Term
  173     go e = case asPi e of
  174       Just (vn, tp, body) -> do
  175         sc <- asks ttSc
  176         case asEqTrue tp of
  177           Just tp' -> do
  178             body' <- go body
  179             imp <- lookupPrim "==>"
  180             liftIO $ scApplyAll sc imp [tp',body']
  181           Nothing -> do
  182             body' <- go body
  183             liftIO $ scLambda sc vn tp body'
  184       Nothing -> case asEqTrue e of
  185         Just e' -> return e'
  186         Nothing -> fail "Unexpected term shape"

  188 -- | Attempt to convert a SAWCore proposition into an equivalent Cryptol predicate and corresponding
  189 --   schema. Only expected to work for terms that are the immediate result of importing a Cryptol
  190 --   expression, or a simple transformation of one.
  191 propToSchemaExpr :: SharedContext -> CryptolEnv -> Term -> IO (Either TTError (C.Schema, C.Expr))
  192 propToSchemaExpr sc cenv t = do
  193   env <- initTTEnv sc cenv
  194   runTT env $ propToLambda t >>= inferSchemaExpr

  220 data TTError = TTError { _ttErrMsg :: String, ttErrContext :: [CallContext], ttErrCommitted :: Bool }

  222 debug :: Bool

  225 prettyTTError :: PPS.Opts -> TTError -> IO PPS.Doc
  226 prettyTTError opts (TTError msg ts _) | debug = do
  227   docs <- mapM (prettyContext opts) ts
  228   return $ PP.vcat $ [ "Translation to Cryptol failed: ", PP.pretty msg ] ++ docs
  229 prettyTTError opts (TTError msg ts _) | Just ts' <- NE.nonEmpty ts = do
  230   prettyFirst <- prettyContext opts (NE.head ts')
  231   case length ts > 1 of
  232     True -> do
  233       prettyLast <- prettyContext opts (NE.last ts')
  234       return $ PP.vcat
  235         [ "Translation to Cryptol failed:"
  236         , PP.pretty msg PP.<> ":"
  237         , PP.indent 2 prettyFirst
  238         , "in subterm:"
  239         , PP.indent 2 prettyLast
  240         ]
  241     False -> do
  242       return $ PP.vcat $
  243         [ "Translation to Cryptol failed:"
  244         , PP.pretty msg PP.<> ":"
  245         , PP.indent 2 prettyFirst
  246         ]
  247 prettyTTError _ (TTError msg _ _) = return $ PP.vcat
  248   [ "Translation to Cryptol failed: ", PP.pretty msg  ]

  250 bindPName :: VarIndex -> Text -> TTEnv -> TTEnv
  251 bindPName idx pname env = 
  252   env { ttEnvVars = Bimap.insert idx pname (ttEnvVars env) }

  254 lookupVarName :: VarIndex -> TT P.PName
  255 lookupVarName idx = do
  256   m <- asks ttEnvVars
  257   case Bimap.lookup idx m of
  258     Just txt -> return $ P.UnQual' (C.mkIdent txt) C.SystemName
  259     Nothing -> fail $ "lookupVarName: could not find variable: " ++ show idx

  261 constToName :: SAW.Name -> TT C.Name
  262 constToName nm = do
  263   m <- asks ttConstMap
  264   mT <- asks ttExtras
  265   msum 
  266     [ mreturn $ Map.lookup nm m
  267     , do SAW.ModuleIdentifier ident <- return $ SAW.nameInfo nm
  268          mreturn $ Map.lookup ident mT
  269     , fail $ "No corresponding Cryptol name for SAW constant: " ++ 
  270         Text.unpack (SAW.toAbsoluteName $ SAW.nameInfo nm)
  271     ]
  272 
  273 bindVarName :: SAW.VarName  -> TTEnv -> TTEnv
  274 bindVarName vn env = go 0
  275   where 
  276     go :: Integer -> TTEnv
  277     go i =
  278       let pname = if i == 0 then SAW.vnName vn else 
  279             SAW.vnName vn <> Text.pack (show i)
  280       in case Bimap.lookupR pname (ttEnvVars env) of
  281         Just idx -> case idx == SAW.vnIndex vn of
  282           True -> env
  283           False -> go (i+1)
  284         Nothing -> bindPName (SAW.vnIndex vn) pname env

  286 mreturn :: MonadPlus m => Maybe a -> m a
  287 mreturn (Just a) = return a

  290 newtype TT a = TT { unTT :: ExceptT TTError (ReaderT TTEnv IO) a }
  291   deriving (Functor, Applicative, Monad, MonadReader TTEnv, MonadError TTError)
  292 
  293 instance MonadIO TT where
  294   liftIO f = do
  295     mres <- TT $ liftIO (try f)
  296     case mres of
  297       Left (e :: IOException) -> throwError $ TTError (show e) [] True
  298       Right a -> return a
  299 
  300 withError :: MonadError e m => (e -> e) -> m a -> m a
  301 withError f g = catchError g (\e -> throwError (f e))
  302 
  303 -- | Commit to an alternative by considering any uncaught errors thrown
  304 --   by the sub-computation to be unrecoverable.
  305 commit :: TT a -> TT a
  306 commit = withError (\e -> (e {ttErrCommitted = True}))
  307 
  308 withContext :: CallContext -> TT a -> TT a
  309 withContext ctx f = withError (\e -> (e {ttErrContext = ctx  : ttErrContext e})) f
  310 
  311 withTermContext :: String -> Term -> TT a -> TT a
  312 withTermContext msg t f = do
  313   sc <- asks ttSc
  314   withContext (CallContext msg (\opts -> prettyTerm sc opts t)) f

  316 withNameContext :: String -> SAW.Name -> TT a -> TT a
  317 withNameContext msg nm = withContext $ CallContext msg (\_ -> return $ PP.pretty $ prettySawName nm)

  319 -- Alternative branches are implicit try-catches as long as the
  320 -- thrown error is not committed (i.e. uncaught during a 'commit' action).
  321 instance Alternative TT where
  322   empty = fail ""
  323   f <|> g =
  324     catchError f $ \e -> case ttErrCommitted e of
  325       -- re-throw any committed errors from 'f', as they are considered non-recoverable
  326       True -> throwError e
  327       -- otherwise, attempt the alternative 'g'
  328       False -> catchError g $ \e2 -> case e2 of
  329         -- if the second error is from "empty" (i.e. no more alternatives in an msum), then
  330         -- re-throw the original error, as it is likely to be more informative
  331         TTError "" [] False -> throwError e
  332         _ -> throwError e2
  333 
  334 instance MonadPlus TT

  336 instance MonadFail TT where
  337   fail msg = throwError $ TTError msg [] False

  351 translateAsType :: Term -> TT Type
  352 translateAsType t = alts "translateAsType" t
  353   [ do [n,a] <- mreturn $ asGlobalApply "Cryptol.seq" t
  354        commit $ do
  355          n' <- translateAsType n
  356          a' <- translateAsType a
  357          return $ P.TSeq n' a'
  358   , do [n] <- mreturn $ asGlobalApply "Cryptol.TCNum" t
  359        n' <- mreturn $ asNat n
  360        return $ P.TNum (fromIntegral n')
  361        -- NOTE: this will intentionally fail if 'b' depends on 'a'
  362   , do (_, a, b) <- mreturn $ asPi t
  363        b' <- translateAsType b
  364        commit $ do
  365         a' <- translateAsType a
  366         return $ P.TFun a' b'
  367   , do (vn, tT) <- mreturn $ asVariable t
  368        _ <- translateAsKind tT
  369        commit $ do
  370         nm <- lookupVarName (SAW.vnIndex vn)
  371         return $ userT nm []
  372   , do (tc, nm, ts) <- translateAsTCon t
  373        case tc of
  374          C.TC _ -> return ()
  375          C.TF _ -> return ()
  376          _ -> fail $ "Unexpected type constructor: " ++ (show $ pp tc)
  377        commit $ do
  378          ts' <- mapM translateAsType ts
  379          return $ userT nm ts'
  380   ]

  382 nameToTCon :: SAW.Name -> TT (C.TCon, Name)
  383 nameToTCon nm = withNameContext "nameToTCon" nm $ msum
  384   [ do nm' <- constToName nm
  385        tc <- mreturn $ C.builtInType nm'
  386        return $ (tc, C.nameToDefPName nm')
  387   , fail $ "Could not find Cryptol type for: " ++ prettySawName nm
  388   ]

  390 translateAsTCon :: Term -> TT (C.TCon, Name, [Term])
  391 translateAsTCon t = do
  392   let (f, ts) = asApplyAll t
  393   nm <- mreturn $ asConstant f
  394   (tc, pnm) <- nameToTCon nm
  395   return $ (tc, pnm, ts)

  397 translateAsConstraint :: Term -> TT Prop
  398 translateAsConstraint t = withTermContext "translateAsConstraint" t $ do
  399   (C.PC _ , nm, ts) <- translateAsTCon t
  400   commit $ do
  401     ts' <- mapM translateAsType ts
  402     return $ P.CType $ userT nm ts'

  404 translateAsKind :: Term -> TT P.Kind
  405 translateAsKind t = alts "translateAsKind" t
  406   [ mreturn $ isGlobalDef "Cryptol.Num" t >> return P.KNum
  407   , do (TypeSort 0) <- mreturn $ asSort t
  408        return P.KType

  411 termType :: Term -> TT Term
  412 termType t = case termSortOrType t of
  413   Left s -> fail $ "termType: unexpected sort: " ++ show s
  414   Right tT -> return tT

  416 translateApp :: [Term] -> TT ([Type],[Expr])
  417 translateApp [] = return ([],[])
  418 translateApp (arg:args) = do
  419   (argTs,argEs) <- translateApp args
  420   alts "translateApp" arg
  421     [ do argT <- termType arg
  422          _ <- translateAsConstraint argT
  423          return (argTs, argEs)
  424     , do arg' <- translateAsExpr arg
  425          return (argTs,arg':argEs)
  426     , do arg' <- translateAsType arg
  427          return (arg':argTs,argEs)
  428     ]
  429 
  430 eApps :: P.Expr n -> [P.Expr n] -> P.Expr n
  431 eApps e [] = e
  432 eApps e (arg:args) = eApps (P.EApp e arg) args

  434 isValueName :: C.Name -> Bool
  435 isValueName nm = case C.nameNamespace nm of
  436   C.NSValue -> True
  437   C.NSConstructor -> True
  438   _ -> False

  440 translateLambda :: [(SAW.VarName, Term)] -> Term -> TT Expr
  441 translateLambda vars fn = withVars vars $ \vs -> do
  442   let asParam = \case
  443         CryParam nm tp -> 
  444           Just $ P.PTyped (P.PVar (noLoc nm)) tp
  445         _ -> Nothing
  446   let vars' = mapMaybe asParam vs
  447   fn' <- translateAsExpr fn
  448   case vars' of
  449     [] -> return fn'
  450     _ -> return $ P.EFun P.emptyFunDesc vars' fn'
  451 
  452 translateAsExpr :: Term -> TT Expr
  453 translateAsExpr t = alts "translateAsExpr" t
  454   [ do (fn, args@(_:_)) <- return $ asApplyAll t
  455        fn' <- translateAsExpr fn
  456        commit $ do
  457          (argTs,argEs) <- translateApp args
  458          return $ eApps (P.EAppT fn' (map P.PosInst argTs)) argEs
  459   , do (vars@(_:_), fn) <- return $ asLambdaList t
  460        translateLambda vars fn
  461   , do (vn,tT) <- mreturn $ asVariable t
  462        tT' <- translateAsType tT
  463        commit $ do
  464          nm <- lookupVarName (SAW.vnIndex vn)
  465          return $ P.ETyped (P.EVar nm) tT'
  466   , do nm <- mreturn $ asConstant t
  467        nm' <- constToName nm
  468        True <- return $ isValueName nm'
  469        return $ P.EVar (C.nameToDefPName nm')
  470   ]
  471 
  472 withVars :: [(SAW.VarName, Term)] -> ([CryptolVar] -> TT a) -> TT a
  473 withVars [] f = f []
  474 withVars ((vn,t):vs) f = withVar vn t $ \p -> withVars vs $ \ps ->
  475   f (p:ps)
  476 
  477 withVar :: SAW.VarName -> Term -> (CryptolVar -> TT a) -> TT a
  478 withVar vn t f = alts "withVar"  t
  479   [ do t' <- translateAsType t
  480        commit $ local (bindVarName vn) $ do
  481           nm <- lookupVarName (SAW.vnIndex vn)
  482           f $ (CryParam nm t')
  483   , do c <- translateAsConstraint t
  484        commit $ f (CryConstraint c)
  485   , do tk <- translateAsKind t
  486        commit $ local (bindVarName vn) $ do
  487           nm <- lookupVarName (SAW.vnIndex vn)
  488           let tp = P.TParam nm (Just tk) Nothing
  489           f $ (CryTParam tp)
  490   ]
