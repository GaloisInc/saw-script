   46 data TopTTEnv = 
   47     TopTTEnv { ttSc :: SAW.SharedContext, ttCryEnv :: SAW.CryptolEnv, ttPPOpts :: PPS.Opts }
   48 
   49 newtype TopTT a = TopTT { unTopTT :: ExceptT String (ReaderT TopTTEnv IO) a }
   50   deriving (Applicative, Functor, Monad, MonadError String, MonadReader TopTTEnv, MonadIO)
   51 
   52 instance MonadFail TopTT where
   53   fail msg = throwError msg
   54 
   55 runTopTT :: TopTTEnv -> TopTT a -> IO (Either String a)
   56 runTopTT env f = runReaderT (runExceptT (unTopTT f)) env
   57 
   58 execTopTT :: TopTTEnv -> TopTT () -> IO (Maybe String)
   59 execTopTT env f = runTopTT env f >>= \case
   60   Left msg -> return $ Just msg
   61   Right () -> return Nothing

   63 prettyTerm :: Term -> TopTT PPS.Doc
   64 prettyTerm t = do
   65   sc <- asks ttSc
   66   opts <- asks ttPPOpts
   67   liftIO $ SAW.prettyTerm sc opts t
   68 
   69 -- | Lift any free variables into a bound Pi. Has no effect on closed terms.
   70 liftFrees :: Term -> TopTT Term
   71 liftFrees t = do
   72   sc <- asks ttSc
   73   let varTypes = SAW.varTypes t
   74   varNames <- forM (Set.toAscList $ termVarNames t) $ \vn -> 
   75     case IntMap.lookup (SAW.vnIndex vn) varTypes of
   76       Just tT -> return (vn, tT)
   77       Nothing -> do
   78         doc <- prettyTerm t
   79         fail $ "Invalid term shape: " ++ show doc
   80   liftIO $ SAW.scPiList sc varNames t

   82 writeTerm ::
   83   Text ->
   84   FilePath ->
   85   Term ->
   86   TopTT ()
   87 writeTerm tnm dest t = do
   88   sc <- asks ttSc
   89   cenv <- asks ttCryEnv
   90   opts <- asks ttPPOpts
   91   t' <- liftFrees t
   92   let mkterm = case SAW.termSortOrType t' of
   93         Left SAW.PropSort -> SAW.propToSchemaExpr
   94         _ -> SAW.termToSchemaExpr
   95   (liftIO $ mkterm sc cenv t') >>= \case
   96     Left err -> do
   97       msg <- liftIO $ SAW.prettyTTError opts err 
   98       fail (PPS.render opts msg)
   99     Right (s,e) -> do
  100       let 
  101         thynm = takeBaseName dest
  102         thynm' = Isabelle.TheoryName thynm False
  103         tnm' = Isabelle.Name thynm' (Text.unpack tnm) Isabelle.NoSyn Isabelle.Term
  104         sel = TargetExpr tnm' s e
  105       writeTarget (takeDirectory dest) sel

  107 withCryptolModule :: Either SAW.ExtCryptolModule FilePath -> (ModName -> TopTT a) -> TopTT a
  108 withCryptolModule mm f = case mm of
  109   Left (SAW.ECM_LoadedModule m _) -> f (thing m)
  110   Left _ -> fail $ "Cannot translate SAW internal cryptol module"
  111   Right fp -> do
  112     modEnv <- asks (SAW.eModuleEnv . ttCryEnv)
  113     let ?fileReader = BS.readFile
  114     (nm, modEnv') <- liftIO $ SAW.liftModuleM modEnv $
  115       tcTopEntitytName <$> MB.loadModuleByPath True fp
  116     local (\env -> env { ttCryEnv = (ttCryEnv env){SAW.eModuleEnv = modEnv'}}) $ 
  117       f nm
  118 
  119 writeCryptolModules ::
  120   [SAW.ExtCryptolModule] ->
  121   [FilePath] ->
  122   FilePath ->
  123   TopTT ()
  124 writeCryptolModules extmods sources dest = go [] $ map Left extmods ++ map Right sources
  125   where
  126     go nms = \case
  127       (m:ms) -> withCryptolModule m $ \nm -> go (nm:nms) ms
  128       [] -> writeTarget dest (ModuleNames nms)
  129 
  130 writeTarget :: FilePath -> TargetSelect -> TopTT ()
  131 writeTarget dest sel = do
  132   cenv <- asks ttCryEnv
  133   let
  134     me = SAW.eModuleEnv cenv
  135     mods = lmLoadedModules $ meLoadedModules me
  136 
  137     opts = emptyOpts 
  138         { isaDestDir_ = dest
  139         , isaImportPrefix_ = CryptolImage
  140         , verbosity_ = 1
  141         , targetSelect_ = sel
  142         , loggerMsg_ = putStrLn
  143         , loggerErr_ = putStrLn
  144         }
  145   liftIO $ processModules opts mods [] [] >>= \case
  146     True -> return ()
  147     False -> fail "Translation to Isabelle failed or was incomplete."
