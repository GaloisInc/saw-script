   16 data TranslationError =
   17     UnsupportedDecl Cry.Decl
   18   | UnsupportedExpr Cry.Expr
   19   | UnsupportedType Cry.Type
   20   | UnsupportedTypeDecl Cry.Name Cry.TySyn
   21   | UnsupportedEntity String
   22   | UnsupportedRecursion [Cry.Decl]
   23   | UnexpectedSignature Cry.Schema Cry.Expr
   24   | UnknownVariableIdent Int
   25   | UnsupportedMatchShape [[Cry.Match]]
   26   | TypeMismatch Cry.Type Cry.Type
   27   | UnexpectedPolymorphism Cry.Schema
   28   | TypeCheckFailure Cry.Expr
   29   | StubbedFunction Cry.Decl
   30   | MissingCryName Cry.Name
   31   | CryptolTypeOfError Cry.Expr String
   32   | ModuleTranslationFailure Cry.ModName
   33   | LocatedError Position.Range TranslationError
   34   | InnerErrors TranslationError [TranslationError]
   35   | MonadFailure String
   36 
   37 stripLocation :: Position.Range -> TranslationError -> TranslationError
   38 stripLocation rng = \case
   39   LocatedError rng' er | Position.rangeWithin rng rng' -> er
   40   InnerErrors er errs -> InnerErrors (stripLocation rng er) (map (stripLocation rng) errs)
   41   er -> er
   42 
   43 joinRange :: Position.Range -> Position.Range -> Position.Range
   44 joinRange rng1 rng2 = case (Position.source rng1, Position.source rng2) of
   45   ("", _) -> rng2
   46   (_,"") -> rng1
   47   _ -> if Position.rangeWithin rng1 rng2 then rng1 else rng2
   48 
   49 addLocation :: Position.Range -> TranslationError -> TranslationError
   50 addLocation rng er | rng == Position.emptyRange = stripLocation rng er
   51 addLocation rng er = case er of
   52   LocatedError rng' er' -> LocatedError (joinRange rng rng') er'
   53   InnerErrors (LocatedError rng' er') ers -> 
   54     addLocation rng $ LocatedError rng' (InnerErrors er' ers)
   55   _ -> LocatedError rng (stripLocation rng er)
   56 
   57 nontrivial :: TranslationError -> Bool
   58 nontrivial = \case
   59   LocatedError _ e -> nontrivial e
   60   InnerErrors e es -> nontrivial e || any nontrivial es
   61   MonadFailure "" -> False
   62   _ -> True
   63 
   64 innerErrors :: TranslationError -> [TranslationError] -> TranslationError
   65 innerErrors er ers = case filter nontrivial ers of
   66   [] -> er
   67   ers' -> InnerErrors er ers'
   68 
   69 data OuterTranslationError where
   70   OuterTranslationError :: Options.HasOptions => TranslationError -> OuterTranslationError
   71 
   72 pp :: (Options.HasOptions, Show a, PP a) => a -> String
   73 pp a = if Options.verbosity >= 3 then pretty a ++ "\n (" ++ show a ++ ")" else pretty a
   74 
   75 ppList :: (Options.HasOptions, Show a, PP a) => [a] -> String
   76 ppList a = intercalate "\n" $ map pp a
   77 
   78 ppNames :: [Name.Name] -> String
   79 ppNames nms = intercalate " " $ map Name.cleanName nms
   80 
   81 flatErrors :: TranslationError -> [TranslationError]
   82 flatErrors = \case
   83   InnerErrors e es -> flatErrors e ++ (concat $ map flatErrors es)
   84   LocatedError rng e -> fmap (addLocation rng) $ flatErrors e
   85   e | nontrivial e -> [e]
   86   _ -> []
   87 
   88 simpleError :: TranslationError -> TranslationError
   89 simpleError e = case NE.nonEmpty (flatErrors e) of
   90   Just es | NE.length es > 1 -> InnerErrors (NE.head es) [NE.last es]
   91   _ -> e
   92 
   93 instance Show OuterTranslationError where
   94   show (OuterTranslationError er) = case er of
   95     UnsupportedDecl d -> "Unsupported cryptol declaration: " ++ pp d
   96     UnsupportedExpr e -> "Unsupported cryptol expression: " ++ pp e
   97     UnsupportedType t -> "Unsupported cryptol type: " ++ pp t
   98     UnsupportedTypeDecl nm t -> "Unsupported type declaration: " ++ pp nm ++ " = " ++ pp t
   99     UnsupportedEntity nm -> "Unsupported translation entity: " ++ show nm
  100     UnexpectedSignature s e -> "Unexpected signature: " ++ pp s ++ " for cryptol declaration: " ++ pp e
  101     UnsupportedRecursion ds -> "Unsupported recursive declarations:\n" ++ ppList ds
  102     UnknownVariableIdent idx -> "Unknown variable identifier: " ++ show idx
  103     UnsupportedMatchShape matches ->
  104       "Unsupported match shape: " ++ show (map length matches)
  105     TypeMismatch t1 t2 -> "Type mismatch: " ++ pp t1 ++ " vs. " ++ pp t2
  106     UnexpectedPolymorphism s -> "Unexpected polymorphic type: " ++ pp s
  107     TypeCheckFailure e -> "Failed to type check cryptol expression: " ++ pp e
  108     StubbedFunction d -> "Unexpected function stub: " ++ pp d
  109     MissingCryName nm -> "No entry in cryptol environment for name: " ++ pp nm
  110     CryptolTypeOfError e msg -> "Could not determine type of Cryptol expression: \n" 
  111       ++ pp e ++ "\n" ++ msg
  112     ModuleTranslationFailure mnm -> "Failed to translate Cryptol module: " ++ pp mnm
  113     LocatedError rng e -> pp rng ++ "\n" ++ showErr e
  114     InnerErrors e [] -> showErr e
  115     InnerErrors e es -> showErr e ++ "\n" ++ (intercalate "\n" $ map showErr es)
  116     MonadFailure msg -> msg
  117 
  118 showErr :: Options.HasOptions => TranslationError -> String
  119 showErr er = case Options.verbosity < 2 of
  120   True -> show (OuterTranslationError (simpleError er))
  121   False -> show (OuterTranslationError er)
