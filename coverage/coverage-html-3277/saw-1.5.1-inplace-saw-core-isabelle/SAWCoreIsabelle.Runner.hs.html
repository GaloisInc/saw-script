   71 unused_thys :: [String]
   72 unused_thys = ["Unsupported.thy", "Unconstrain_Typedef.thy", "Word0.thy"]

   74 bsToText :: BS.ByteString -> T.Text
   75 bsToText bs = T.decodeUtf8 (BS.toStrict (toLazyByteString (byteStringHex bs)))

   78 -- | Hash of all (used) Cryptol theories. Should correspond to
   79 -- 'cryptol_theory_hash' in Cryptol.thy. Notably the individual file hashes
   80 -- are converted into UTF-8 strings, combined, and then hashed together.
   81 -- This is to simplify the Isabelle side of the hash computation, which
   82 -- doesn't provide a convenient mechanism for natively combining hashes.
   83 cryptolTheoryHash :: String
   84 cryptolTheoryHash = 
   85   let 
   86     used_thys = filter (\(thy,_) -> not (List.elem thy unused_thys)) isaThys
   87     txt = bsToText $ BS.concat (map snd used_thys)
   88   in T.unpack $ bsToText $ SHA1.hash (T.encodeUtf8 txt)
   89 
   90 isaThys :: [(String, BS.ByteString)]
   91 isaThys = $(do
   92   let dir = "saw-core-isabelle" </> "isabelle" </> "theories"
   93   files <- runIO $ listDirectory dir
   94   bs <- runIO $ forM files $ \f -> do
   95     content <- BS.readFile (dir </> f)
   96     return (takeFileName f, SHA1.hash content)
   97   lift bs)

   99 isBuiltinThy :: String -> Bool
  100 isBuiltinThy str = List.any (\(nm,_) -> nm == str <> ".thy") isaThys

  102 renameImports ::
  103   HasOptions => Theory.Theory -> IO Theory.Theory
  104 renameImports thy = do
  105   decls <- Decl.traverseImports (Theory.thyDecls thy) $ \nm -> 
  106     case isBuiltinThy (Name.thyNm nm) of
  107       True -> case Options.isaImportPrefix of
  108         Options.NoPrefix -> return nm
  109         Options.CryptolImage ->
  110           return $ nm { Name.thyNm = "\"Cryptol." ++ Name.thyNm nm ++ "\"" }
  111         Options.CustomPrefix s ->
  112           return $ nm { Name.thyNm = "\"" ++ s ++ Name.thyNm nm ++ "\"" }
  113       False -> return nm
  114   return $ thy { Theory.thyDecls = decls }

  116 extractRecordDecls :: Theory.Theory -> (Set.Set Decl.Decl, Theory.Theory)
  117 extractRecordDecls thy =
  118   let (recDs, ds) = Decl.strip (\case Decl.RecordDecl{} -> True; _ -> False) (Theory.thyDecls thy)
  119   in (Set.fromList recDs, thy { Theory.thyDecls = ds })

  122 recordDeclThy :: Decl.Decl -> Theory.Theory
  123 recordDeclThy d@(Decl.RecordDecl flds) = 
  124   let thynm = Name.nmThy $ Decl.recordName flds
  125   in Theory.Theory thynm (Decl.singleton d) 
  126 recordDeclThy _ = error "recordDeclThy: unexpected Decl kind"

  128 allDepsOf :: Deps.CryptolDeps -> [Cry.ModName] -> Set.Set Cry.ModName
  129 allDepsOf deps ms = 
  130   let acc_deps mnm = Set.union (fromMaybe Set.empty $ fmap snd $ Map.lookup mnm (Deps.moduleDeps deps))
  131   in foldr acc_deps Set.empty ms

  134 asModule :: Set.Set Cry.ModName -> Cry.LoadedModule -> Maybe (Cry.Module, [Cry.ModName])
  135 asModule selected lm
  136   | Cry.lmName lm /= Cry.preludeName
  137   , Set.member (Cry.lmName lm) selected
  138   = Just (Cry.lmModule lm, (Set.toList $ Cry.fiImportDeps (Cry.lmFileInfo lm)))
  139 asModule _ _ = Nothing

  141 data TranslateState = TranslateState
  142   { trErrs :: [Error.TranslationError]
  143   , trThys :: Theory.Theories
  144   }
  145 
  146 resultState :: [Error.TranslationError] -> Theory.Theory -> TranslateState
  147 resultState ers res = TranslateState ers (Theory.singleton res)
  148 
  149 instance Monoid TranslateState where
  150   mempty = TranslateState mempty mempty
  151 
  152 instance Semigroup TranslateState where
  153   TranslateState errs1 thys1 <> TranslateState errs2 thys2 =
  154     TranslateState (errs1 <> errs2) (thys1 <> thys2)
  155 
  156 printErrs :: (HasOptions, IO.MonadIO m) => TranslateState -> m ()
  157 printErrs st = forM_ (trErrs st) $ \terr -> err $ Error.showErr terr
  158 
  159 writeResult :: HasOptions => [Error.TranslationError] -> Theory.Theory -> WriterT TranslateState IO ()
  160 writeResult ers res = do
  161   let st = resultState ers res
  162   case ers of
  163     [] -> do
  164       log 0 $ "Successfully translated theory " ++ (show $ Theory.thyNm res)
  165       tell st
  166     _ -> case Options.keepGoing of
  167       True -> do
  168         err $ "Errors raised while translating theory " ++ (show $ Theory.thyNm res)
  169         tell st
  170       False -> do
  171         let errMsg = "Translating was unsuccessful for theory " ++ (show $ Theory.thyNm res)
  172         err errMsg
  173         printErrs st
  174         log (-1) $ "Set 'keep-going' flag to attempt an incomplete translation."
  175         IO.liftIO $ throw (RunnerError errMsg)
  176 
  177 processModules ::
  178   Options.Options ->
  179   [Cry.LoadedModule] ->
  180   [Cry.DeclGroup] -> 
  181   [Cry.TySyn]  ->
  182   IO Bool
  183 processModules opts loadedModules extraDecls extraTys = Options.withOptions opts $ do
  184   let
  185     outDir = Options.isaDestDir
  186     cryDeps = Deps.mkCryptolDeps loadedModules extraDecls extraTys
  187     allMods =  map Cry.lmName loadedModules
  188     
  189   trresult <- execWriterT $ do 
  190     imports <- case Options.targetSelect of
  191       Options.AllModules -> return allMods
  192       Options.NamedModules nms -> 
  193         return $ filter (\m -> elem (Text.unpack $ Cry.modNameToText m) nms) allMods
  194       Options.ModuleNames nms ->
  195         return $ filter (\m -> elem m nms) allMods
  196       Options.TargetExpr nm sch e -> do
  197         (errs,res) <- IO.liftIO $ Translate.translateSingleExprIO cryDeps nm sch e
  198         writeResult errs res
  199         let deps = Theory.thyImports res
  200         return $ filter (\m -> elem (Translate.modToTheoryPure m) deps) allMods
  201 
  202     let allDeps = allDepsOf cryDeps imports
  203     let modules = mapMaybe (asModule allDeps) loadedModules
  204     forM_ modules $ \(crymod,cryimports) -> do
  205       log 1 $ pretty crymod
  206       (errs,res) <- IO.liftIO $ Translate.translateModuleIO cryDeps cryimports crymod
  207       writeResult errs res
  208     log 1 $ "\n"
  209   case (trErrs trresult, Theory.null (trThys trresult))  of
  210     ([], False) -> log 0 $ "Translation successful. Generating theory files..."
  211     (_, True) -> log (-1) $ "Could not find any modules to translate"
  212     _ -> do
  213       err $ "Errors raised during translation:"
  214       printErrs trresult
  215       log (-1) $ "Resulting theories will be incomplete. Generating theory files..."
  216   log 1 $ "\n"
  217   templates <- Template.loadAll
  218   
  219   (recordThys,thys) <- Theory.forTheories (trThys trresult) (mempty,[]) $ \thy (recordThys,thys) -> do
  220     let (recDecls', thy') = extractRecordDecls thy
  221     let recordThys' = Set.map recordDeclThy recDecls'
  222     let thy'' = foldr (\rthy -> Theory.addDecl (Decl.Import (Theory.thyNm rthy))) thy' recordThys'
  223     return (Set.union recordThys recordThys', thy'':thys)
  224   
  225   let recordThys' = map (Theory.addDecl (Decl.Import cryThy)) (Set.toList recordThys)
  226   let allThys = recordThys' ++ thys
  227 
  228   forM_ allThys $ \thy -> do
  229     let theoryFile = Name.thyNm (Theory.thyNm thy) <> ".thy"
  230     log 0 $ "Generating " ++ theoryFile
  231     thy' <- renameImports thy
  232     let content = Output.withTemplates templates $ Output.out thy'
  233     let outputFile = outDir </> theoryFile
  234     result <- tryIO $ T.writeFile outputFile (T.pack content)
  235     case result of
  236       Left e -> fatalErr $ "Failed to write output file '" ++ outputFile ++ "': " ++ show e
  237       Right () -> do
  238         log (-1) $ "Successfully wrote '" ++ outputFile ++ "'"
  239   return $ (null $ trErrs trresult) && (not $ null allThys)
  240 
  241 tryIO :: IO a -> IO (Either IOException a)
  242 tryIO f = catch (Right <$> f) (\e -> return $ Left (e :: IOException))
