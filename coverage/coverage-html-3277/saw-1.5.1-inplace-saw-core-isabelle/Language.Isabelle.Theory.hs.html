   37 data Theory = Theory { thyNm :: Name.TheoryName, thyDecls :: Decl.Decls }
   38   deriving (Eq, Ord)
   39 
   40 mergeThys :: Theory -> Theory -> Theory
   41 mergeThys thy1 thy2 = case thyNm thy1 == thyNm thy2 of
   42   True -> Theory (thyNm thy1) $ (thyDecls thy1) <> (thyDecls thy2)
   43   False -> Panic.panic "mergeThys: Theory name clash" [show (thyNm thy1), show (thyNm thy2)]
   44 
   45 data Theories = Theories { thyMap :: Map.Map Name.TheoryName Theory }
   46 
   47 null :: Theories -> Bool
   48 null thys = Map.null (thyMap thys)
   49 
   50 instance Semigroup Theories where
   51   (Theories thys1) <> (Theories thys2) = Theories
   52     (Map.merge Map.preserveMissing Map.preserveMissing (Map.zipWithMatched (\_ -> mergeThys)) thys1 thys2)
   53 
   54 
   55 instance Monoid Theories where
   56   mempty = Theories Map.empty
   57 
   58 singleton :: Theory -> Theories
   59 singleton thy = Theories (Map.singleton (thyNm thy) thy)
   60 
   61 -- | Create a theory with a variant of the given name that
   62 --   does not clash with any existing theories.
   63 freshTheory :: Theories -> Name.TheoryName -> Theory
   64 freshTheory (Theories thys) nm =
   65   let
   66       nms = map Name.thyNm $ Map.keys thys
   67       freshNm = Name.freshIdent nms (Name.thyNm nm)
   68   in  mkTheory (nm { Name.thyNm = freshNm })
   69 
   70 mkTheory :: Name.TheoryName -> Theory
   71 mkTheory nm = Theory nm mempty
   72 
   73 adjustDecls :: (Decl.Decls -> Decl.Decls) -> Theory -> Theory
   74 adjustDecls f thy = thy { thyDecls = f (thyDecls thy) }
   75 
   76 addDecl :: Decl.Decl -> Theory -> Theory
   77 addDecl d = adjustDecls (\ds -> Decl.singleton d <> ds)
   78 
   79 addDecls :: Name.TheoryName -> Decl.Decls -> Theories -> Theories
   80 addDecls nm ds thys = thys {thyMap =  Map.insertWith mergeThys nm (Theory nm ds) (thyMap thys)}
   81 
   82 lookup :: Name.TheoryName -> Theories -> Theory
   83 lookup nm thys = fromMaybe (mkTheory nm) $ Map.lookup nm (thyMap thys)
   84 
   85 adjust :: Name.TheoryName -> (Decl.Decls -> Decl.Decls) -> Theories -> Theories
   86 adjust nm f thys = thys { thyMap = Map.adjust (\thy -> thy { thyDecls = f (thyDecls thy) }) nm (thyMap thys) }
   87 
   88 update :: Theory -> Theories -> Theories
   89 update thy = addDecls (thyNm thy) (thyDecls thy)
   90 
   91 -- | Drop the theory with the given name and return it, along with the
   92 --   modified 'Theories'.
   93 --   Resulting 'Theory' is empty if it is not in the map.
   94 drop :: Name.TheoryName -> Theories -> (Theory, Theories)
   95 drop nm thys =
   96   let
   97     upd k _thy1 _thy2 = mkTheory k
   98     (res, thys') = Map.insertLookupWithKey upd nm (mkTheory nm) (thyMap thys)
   99   in (fromMaybe (mkTheory nm) res, Theories thys')
  100 
  101 forTheories :: Monad m => Theories -> a -> (Theory -> a -> m a) -> m a
  102 forTheories thys a f = foldrM f a (thyMap thys)

  105 defGraph :: [Decl.Decl] ->
  106    (Graph.Graph
  107   , Graph.Vertex -> (Decl.Decl, Name.Name, [Name.Name])
  108   , Name.Name -> Maybe Graph.Vertex)
  109 defGraph decls =
  110   let
  111     go d = case d of
  112       Decl.Definition _ b _args _ _ -> [(d, Binding.bindName b, deps)]
  113       Decl.ConstDecl _ b -> [(d, Binding.bindName b, [])]
  114       Decl.Commented _ d' ->
  115         map (\(_,nm,deps') -> (d,nm,deps')) (go d')
  116       Decl.TypeSyn _ nm _ -> [(d, nm, deps)]
  117       Decl.FixpointLocale nms ->
  118         let locale_name = Name.SimpleName (Decl.localeName nms)
  119         in [(d,locale_name, deps)] ++ map (\nm -> (Decl.NoDecl,nm, [locale_name])) nms
  120       Decl.DatatypeDecl _ nm _ -> [(d, nm, deps)]
  121       {- Decl.TypeArgSyntax b _ _ _ -> Just (d, syntaxName (Binding.bindName b), [Binding.bindName b]) -}
  122       _ -> []
  123       where
  124         deps = Decl.dependencies d
  125     raw_graph = concat $ map go decls
  126   in Graph.graphFromEdges raw_graph
  127 
  128 orderDecls :: Theory -> (Decl.Decl -> Bool) -> [Decl.Decl]
  129 orderDecls thy f =
  130   let (graph, nodeFromVertex, _) = defGraph (nub $ Decl.filter f (thyDecls thy))
  131   in map (\v -> let (d, _, _) = nodeFromVertex v in d) (Graph.reverseTopSort graph)
  132 
  133 isDef :: Decl.Decl -> Bool
  134 isDef = \case
  135   Decl.Definition{} -> True
  136   Decl.ConstDecl{} -> True
  137   Decl.FixpointLocale{} -> True
  138   _ -> False
  139

  141 thyImports :: Theory -> [Name.TheoryName]
  142 thyImports thy = 
  143   nub $ Decl.mapMaybe (\case Decl.Import nm -> Just nm; _ -> Nothing) (thyDecls thy)
  144 
  145 instance Output.Output Theory where
  146   out thy =
  147     let
  148       is = thyImports thy
  149       defs = orderDecls thy isDef
  150       recordDecls = nub $ Decl.filter (\case Decl.RecordDecl{} -> True; _ -> False) (thyDecls thy)
  151       tdecls = nub $ Decl.filter (\case Decl.TypeDecl{} -> True;  _ -> False) (thyDecls thy)
  152       hdecls = nub $ Decl.filter (\case Decl.HashDecl{} -> True;  _ -> False) (thyDecls thy)
  153 
  154 
  155       tsyns = orderDecls thy (\case Decl.TypeSyn{} -> True; Decl.DatatypeDecl{} -> True;_ -> False)
  156       declLine d = case d of
  157         Decl.NoDecl{} -> Nothing
  158         _ -> Just (Output.out d <> "\n")
  159       declLines ds = mapMaybe declLine ds
  160 
  161       noDecls = List.null tdecls && List.null tsyns && List.null defs && List.null comments
  162       -- Un-associated comments
  163       comments = Decl.filter (\case Decl.NoDecl{} -> True; _ -> False) (thyDecls thy)
  164     in
  165       unlines $
  166         [ "theory " ++ Output.quotes (Output.out (Name.thyNm (thyNm thy)))
  167         , "imports " ++ intercalate " " (sort (map (Output.out . Decl.Import) is))
  168         , "begin"
  169         , ""
  170         ]
  171         ++ declLines recordDecls
  172         ++ (if noDecls then ["end"] else
  173           ["context includes cryptol_translation_syntax begin"]
  174           ++ declLines tdecls
  175           ++ []
  176           ++ declLines tsyns
  177           ++ []
  178           ++ declLines defs
  179           ++ []
  180           ++ declLines hdecls
  181           ++ []
  182           ++ declLines comments
  183           ++ []
  184           ++ [ "end", "end" ])
