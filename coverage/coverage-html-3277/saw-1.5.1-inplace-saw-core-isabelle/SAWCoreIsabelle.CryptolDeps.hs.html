   33 instance Hashable CryEntity where
   34   hashWithSalt s = \case
   35     CryDecl d -> hashWithSalt s (show $ pp $ stripProofApps d)
   36     CryTySyn t -> hashWithSalt s (show $ pp t)
   37     CryNominalType nt -> hashWithSalt s (show $ pp nt)

   39 data CryEntity = 
   40     CryDecl Cry.Decl
   41   | CryTySyn Cry.TySyn
   42   | CryNominalType Cry.NominalType

   44 instance TraverseNames CryEntity where
   45   traverseNamesIP = \case
   46     CryDecl d -> CryDecl <$> traverseNamesIP d
   47     CryTySyn t -> CryTySyn <$> traverseNamesIP t
   48     CryNominalType nt -> CryNominalType <$> traverseNamesIP nt

   50 cryEntityName :: CryEntity -> Cry.Name
   51 cryEntityName = \case
   52   CryDecl d -> Cry.dName d
   53   CryTySyn t -> Cry.tsName t
   54   CryNominalType nt -> Cry.ntName nt
   55 
   56 instance Eq CryEntity where
   57   ce1 == ce2 = (cryEntityName ce1 == cryEntityName ce2)
   58 
   59 stripProofApps :: TraverseExprs t => t -> t
   60 stripProofApps = mapSubExprs go
   61   where
   62     go = \case
   63       Cry.EProofAbs _ e -> e
   64       Cry.EProofApp e -> e
   65       e -> e
   66 
   67 foldMapNames :: (TraverseNames t, Monoid m) => (Cry.Name -> m) -> t -> m
   68 foldMapNames f t = execWriter $ traverseNames (\e -> tell (f e) >> return e) t
   69 
   70 containsName :: TraverseNames t => Cry.Name -> t -> Bool
   71 containsName nm t = getAny $ foldMapNames (\nm_ -> Any (nm == nm_)) t

   73 modulesToCryEnv :: [Cry.Module] -> [CryEntity]
   74 modulesToCryEnv ms = 
   75   let 
   76     decls = concat $ concatMap (map Cry.groupDecls . Cry.mDecls) ms
   77     tys = concatMap (Map.elems . Cry.mTySyns) ms
   78     tnms = concatMap (Map.elems . Cry.mNominalTypes) ms
   79   in map CryDecl decls ++ map CryTySyn tys ++ map CryNominalType tnms

   81 data CryptolDeps = CryptolDeps 
   82   { entityDeps :: Map.Map Cry.Name (CryEntity, Set.Set Cry.Name)
   83   , moduleDeps :: Map.Map Cry.ModName (Cry.LoadedModule, Set.Set Cry.ModName)
   84   }
   85 
   86 nameToModName :: Cry.Name -> Maybe Cry.ModName
   87 nameToModName nm = case Cry.nameInfo nm of
   88   Cry.GlobalName _ og | Cry.TopModule mnm <- Cry.ogModule og -> Just mnm
   89   _ -> Nothing
   90 
   91 collapseMaybeSet :: Ord a => Set.Set (Maybe a) -> Set.Set a
   92 collapseMaybeSet = Set.map fromJust . Set.delete Nothing
   93 
   94 moduleDepsOf :: TraverseNames t => CryptolDeps -> t -> Set.Set Cry.ModName
   95 moduleDepsOf d = foldMapNames $ \nm -> 
   96   let dep_nms = fromMaybe Set.empty $ fmap snd $ Map.lookup nm (entityDeps d)
   97       dep_mods = collapseMaybeSet $ Set.map nameToModName dep_nms
   98       acc_deps mnm = Set.union (fromMaybe Set.empty $ fmap snd $ Map.lookup mnm (moduleDeps d))
   99   in Set.fold acc_deps Set.empty dep_mods

  101 stripGuard :: Cry.Schema -> Cry.Schema
  102 stripGuard (Cry.Forall vs _ t) = Cry.Forall vs [] t
  103 
  104 mkTyEnv :: CryptolDeps -> Map.Map Cry.Name Cry.Schema
  105 mkTyEnv ces = Map.mapMaybe go (entityDeps ces)
  106   where
  107     go = \case
  108       (CryDecl d,_) -> Just (stripGuard $ Cry.dSignature d)
  109       _ -> Nothing

  111 mkCryptolDeps :: [Cry.LoadedModule] -> [Cry.DeclGroup] -> [Cry.TySyn] -> CryptolDeps
  112 mkCryptolDeps lms extraDecls extraTys = 
  113   let edeps = cryEntityDeps $ modulesToCryEnv (map Cry.lmModule lms) 
  114        ++ map CryDecl (concat $ map Cry.groupDecls extraDecls)
  115        ++ map CryTySyn extraTys
  116       mdeps = cryModuleDeps lms
  117   in CryptolDeps edeps mdeps

  119 collectNames :: TraverseNames t => t -> Set.Set Cry.Name
  120 collectNames = foldMapNames $ \nm -> 
  121   case Cry.nameInfo nm of 
  122     Cry.LocalName{} -> Set.empty
  123     _ -> Set.singleton nm
  124 
  125 cryEntityDeps :: 
  126   [CryEntity] ->
  127   Map.Map Cry.Name (CryEntity, Set.Set Cry.Name)
  128 cryEntityDeps nes = 
  129   let
  130     raw_graph = map (\ne -> (ne, cryEntityName ne, Set.toList $ collectNames ne)) nes
  131     (gr,vert_to_node,nm_to_vert) = Graph.graphFromEdges raw_graph
  132     get_deps ne = case nm_to_vert (cryEntityName ne) of
  133       Just v -> foldr (\v' acc -> 
  134             let (_,_,deps) = vert_to_node v' in foldr Set.insert acc deps)
  135             Set.empty
  136             (Graph.reachable gr v) 
  137       Nothing -> Set.empty
  138   in Map.fromList $ map (\ne -> (cryEntityName ne, (ne, get_deps ne))) nes
  139 
  140 cryModuleDeps :: 
  141   [Cry.LoadedModule] ->
  142   Map.Map Cry.ModName (Cry.LoadedModule, Set.Set Cry.ModName)
  143 cryModuleDeps nes = 
  144   let
  145     raw_graph = map (\ne -> (ne, Cry.lmName ne, Set.toList $ Cry.fiImportDeps (Cry.lmFileInfo ne))) nes
  146     (gr,vert_to_node,nm_to_vert) = Graph.graphFromEdges raw_graph
  147     get_deps ne = case nm_to_vert (Cry.lmName ne) of
  148       Just v -> foldr (\v' acc -> 
  149             let (_,_,deps) = vert_to_node v' in foldr Set.insert acc deps)
  150             (Set.singleton (Cry.lmName ne))
  151             (Graph.reachable gr v) 
  152       Nothing -> Set.singleton (Cry.lmName ne)
  153   in Map.fromList $ map (\ne -> (Cry.lmName ne, (ne, get_deps ne))) nes
