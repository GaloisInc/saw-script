   28 import           Prelude hiding (log)

   30 import qualified Control.Monad.IO.Class as IO
   31 import qualified Cryptol.Parser.AST as Cry (ModName)
   32 import qualified Cryptol.TypeCheck.AST as Cry
   33 
   34 import qualified Language.Isabelle.Name as Isabelle
   35 
   36 data ImportPrefix = NoPrefix | CryptolImage | CustomPrefix String
   37 
   38 data TargetSelect = 
   39       AllModules 
   40     | NamedModules [String] 
   41     | ModuleNames [Cry.ModName]
   42     | TargetExpr Isabelle.Name Cry.Schema Cry.Expr
   43 
   44 data Options = Options {
   45   crySources_ :: [FilePath]
   46 , isaDestDir_ :: FilePath
   47 , isaImportPrefix_ :: ImportPrefix
   48 , verbosity_ :: Integer
   49 , keepGoing_ :: Bool
   50 , cryptolDirs_ :: [FilePath]
   51 , targetSelect_ :: TargetSelect
   52 , functionStubs_ :: [String]
   53 , loggerMsg_ :: String -> IO ()
   54 , loggerErr_ :: String -> IO ()
   55 }
   56 
   57 emptyOpts :: Options 
   58 emptyOpts = Options [] "" NoPrefix 0 False [] (NamedModules []) [] (\_ -> return ()) (\_ -> return ())
   59 
   60 type HasOptions = (?options :: Options)
   61 
   62 crySources :: HasOptions => [FilePath]
   63 crySources = crySources_ ?options
   64 
   65 isaDestDir :: HasOptions => FilePath
   66 isaDestDir = isaDestDir_ ?options
   67 
   68 isaImportPrefix :: HasOptions => ImportPrefix
   69 isaImportPrefix = isaImportPrefix_ ?options
   70 
   71 verbosity :: HasOptions => Integer
   72 verbosity = verbosity_ ?options
   73 
   74 allOptions :: HasOptions => Options
   75 allOptions = ?options
   76 
   77 keepGoing :: HasOptions => Bool
   78 keepGoing = keepGoing_ ?options
   79 
   80 cryptolDirs :: HasOptions => [FilePath]
   81 cryptolDirs = cryptolDirs_ ?options
   82 
   83 targetSelect :: HasOptions => TargetSelect
   84 targetSelect = targetSelect_ ?options
   85 
   86 functionStubs :: HasOptions => [String]
   87 functionStubs = functionStubs_ ?options
   88 
   89 log :: (HasOptions, IO.MonadIO m) => Integer -> String -> m ()
   90 log v msg = IO.liftIO $ if v < verbosity then loggerMsg_ ?options msg else return ()
   91 
   92 logErr :: (HasOptions, IO.MonadIO m) => Integer -> String -> m ()
   93 logErr v msg = IO.liftIO $ if v < verbosity then loggerErr_ ?options msg else return ()
   94 
   95 addLogger :: (String -> IO ()) -> Options -> Options
   96 addLogger f opts = opts { loggerMsg_ = (\msg -> loggerMsg_ opts msg >> f msg)}
   97 
   98 addErrLogger :: (String -> IO ()) -> Options -> Options
   99 addErrLogger f opts = opts { loggerErr_ = (\msg -> loggerErr_ opts msg >> f msg)}
  100 
  101 withOptions :: Options -> (HasOptions => a) -> a
  102 withOptions opts f = let ?options = opts in f
