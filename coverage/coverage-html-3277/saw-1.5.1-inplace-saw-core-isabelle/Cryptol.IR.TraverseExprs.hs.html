   24 class TraverseExprs t where
   25   -- | Traverse over constituent expressions, without recursing into sub-expressions.
   26   traverseExprs :: Applicative f => (Expr -> f Expr) -> t -> f t
   27 
   28 -- | Traverse over constituent expressions, first recursing into sub-expressions.
   29 traverseSubExprs ::
   30   forall t m. (TraverseExprs t, Monad m) => (Expr -> m Expr) -> (t -> m t)
   31 traverseSubExprs f = traverseExprs go
   32   where
   33     go :: Expr -> m Expr
   34     go e = traverseImmSubExprs go e >>= f
   35 
   36 mapSubExprs :: TraverseExprs t => (Expr -> Expr) -> t -> t
   37 mapSubExprs f x = runIdentity (traverseSubExprs (Identity . f) x)
   38 
   39 mapExprs :: TraverseExprs t => (Expr -> Expr) -> t -> t
   40 mapExprs f x = runIdentity (traverseExprs (Identity . f) x)
   41 
   42 foldMapSubExprs :: (TraverseExprs t, Monoid m) => (Expr -> m) -> t -> m
   43 foldMapSubExprs f t = execWriter $ traverseSubExprs (\e -> tell (f e) >> return e) t
   44 
   45 foldMapExprs :: (TraverseExprs t, Monoid m) => (Expr -> m) -> t -> m
   46 foldMapExprs f t = execWriter $ traverseExprs (\e -> tell (f e) >> return e) t
   47 
   48 -- | Traverse over immediate sub-expressions.
   49 traverseImmSubExprs :: forall f. Applicative f => (Expr -> f Expr) -> Expr -> f Expr
   50 traverseImmSubExprs f se = case se of
   51   EList es t -> EList <$> go es <*> pure t
   52   ETuple es -> ETuple <$> go es
   53   ERec mp -> ERec <$> traverse go mp
   54   ESel e sel -> ESel <$> go e <*> pure sel
   55   ESet t e1 sel e2 -> ESet <$> pure t <*> go e1 <*> pure sel <*> go e2
   56   EIf eP eT eF -> EIf <$> go eP <*> go eT <*> go eF
   57   ECase e alts malt -> ECase
   58     <$> go e <*> traverse go alts <*> go malt
   59   EComp t1 t2 e matches -> EComp
   60     <$> pure t1 <*> pure t2 <*> go e <*> go matches
   61   EVar n -> EVar <$> pure n
   62   ETAbs tps e -> ETAbs <$> pure tps <*> go e
   63   ETApp e t -> ETApp <$> go e <*> pure t
   64   EApp e1 e2 -> EApp <$> go e1 <*> go e2
   65   EAbs n t e -> EAbs <$> pure n <*> pure t <*> go e
   66   ELocated r e -> ELocated <$> pure r <*> go e
   67   EProofAbs p e -> EProofAbs <$> pure p <*> go e
   68   EProofApp e -> EProofApp <$> go e
   69   EWhere e decls -> EWhere <$> go e <*> traverse go decls
   70   EPropGuards gs t  -> EPropGuards <$> traverse doG gs <*> pure t
   71     where doG (xs, e) = (,) <$> pure xs <*> go e
   72   where
   73     go :: forall t. TraverseExprs t => t -> f t
   74     go = traverseExprs f
   75 
   76 instance TraverseExprs a => TraverseExprs [a] where
   77   traverseExprs f = traverse (traverseExprs f)
   78 
   79 instance TraverseExprs a => TraverseExprs (Maybe a) where
   80   traverseExprs f = traverse (traverseExprs f)
   81 
   82 instance (Ord a, TraverseExprs a) => TraverseExprs (Set a) where
   83   traverseExprs f = fmap Set.fromList . traverseExprs f . Set.toList
   84 
   85 instance TraverseExprs a => TraverseExprs (Located a) where
   86   traverseExprs f (Located r a) = Located r <$> traverseExprs f a
   87 
   88 instance TraverseExprs Expr where
   89   traverseExprs f = f
   90 
   91 instance TraverseExprs CaseAlt where
   92   traverseExprs f (CaseAlt xs e) =
   93     CaseAlt <$> pure xs <*> traverseExprs f e
   94 
   95 instance TraverseExprs Match where
   96   traverseExprs f = \case
   97     From x t1 t2 e -> From <$> pure x <*> pure t1 <*> pure t2 <*> traverseExprs f e
   98     Let d -> Let <$> traverseExprs f d
   99 
  100 instance TraverseExprs DeclDef where
  101   traverseExprs f = \case
  102     DPrim -> pure DPrim
  103     DForeign ffi me -> DForeign <$> pure ffi <*> traverseExprs f me
  104     DExpr e -> DExpr <$> traverseExprs f e
  105 
  106 instance TraverseExprs DeclGroup where
  107   traverseExprs f = \case
  108     Recursive ds -> Recursive <$> traverseExprs f ds
  109     NonRecursive d -> NonRecursive <$> traverseExprs f d
  110 
  111 instance TraverseExprs Decl where
  112   traverseExprs f Decl{..} = Decl
  113     <$> pure dName <*> pure dSignature <*> traverseExprs f dDefinition <*> pure dPragmas 
  114     <*> pure dInfix <*> pure dFixity <*> pure dDoc
