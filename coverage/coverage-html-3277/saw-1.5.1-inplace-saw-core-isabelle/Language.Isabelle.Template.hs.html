   24 data Template = Template { raw_content :: String, name :: String }
   25   deriving (Eq, Ord, Show)

   27 newtype Applied = Applied String
   28   deriving (Eq, Ord)
   29 
   30 instance Show Applied where
   31   show (Applied s) = s

   33 embeddedTemplates :: [(FilePath, BS.ByteString)]
   34 embeddedTemplates = $(FE.embedDir ("saw-core-isabelle" </> "isabelle" </> "templates"))
   35 
   36 
   37 loadAll :: IO (Map String Template)
   38 loadAll = do
   39   loaded <- forM embeddedTemplates $ \(file, bs) -> 
   40     case stripExtension "thy_template" (takeFileName file) of
   41       Just nm -> do
   42         let str = T.unpack (BS.decode BS.localeEncoding bs)
   43         return $ Just (nm, Template str nm)
   44       Nothing -> return Nothing
   45   return $ Map.fromList $ catMaybes loaded
   46 
   47 apply :: Template -> [String] -> Applied
   48 apply t args = Applied $ foldr go (raw_content t) (zip [1..] args)
   49   where
   50     go :: (Int, String) -> String -> String
   51     go (idx, arg) s = subRegex (mkRegex ("\\$" ++ show idx)) s arg
