   60 doTranslation ::
   61   Options.HasOptions =>
   62   Deps.CryptolDeps ->
   63   [Cry.ModName] ->
   64   Name.TheoryName ->
   65   IsaM () ->
   66   IO ([Error.TranslationError], Theory.Theory)
   67 doTranslation cryEnv imports thyNm f = do
   68   env <- initIsaEnv cryEnv Nothing thyNm
   69   (st, iout) <- execIsaM env $ do
   70     addDecl (Decl.Import cryThy)
   71     mapM_ importModName imports
   72     f
   73   return $ (outWarns iout, stThy st)

   75 translateModuleIO ::
   76   Options.HasOptions =>
   77   Deps.CryptolDeps ->
   78   [Cry.ModName] ->
   79   Cry.Module ->
   80   IO ([Error.TranslationError], Theory.Theory)
   81 translateModuleIO cryEnv imports crymod  = do
   82   let thyNm = modToTheoryPure (Cry.mName crymod)
   83   doTranslation cryEnv imports thyNm $ 
   84     translateModule crymod

   87   Options.HasOptions =>

   97 importModuleDeps :: TraverseNames t => t -> IsaM ()
   98 importModuleDeps = fmap void . traverseNames $ \nm -> case Deps.nameToModName nm of
   99   Just mnm -> importModName mnm >> return nm
  100   Nothing -> return nm

  102 importModName :: ModName -> IsaM ()

  104   thynm <- modToTheory mnm

  115 translateNominalType :: Cry.NominalType -> IsaM ()
  116 translateNominalType nt = withTParams (Cry.ntParams nt) $ 
  117   case Cry.ntDef nt of
  118     Cry.Enum ctrs -> do
  119       ctrs' <- forM ctrs $ \ctr -> do
  120         flds <- mapM translateType (Cry.ecFields ctr)
  121         nm <- translateName (Cry.ecName ctr)
  122         return (nm, flds)
  123       nm' <- translateName (Cry.ntName nt)
  124       tps <- mapM translateTParam (Cry.ntParams nt)
  125       addDecl $ Decl.DatatypeDecl tps nm' ctrs'
  126     Cry.Struct sc -> do
  127       let tysyn = Cry.TySyn 
  128             { Cry.tsName = Cry.ntName nt 
  129             , Cry.tsParams = Cry.ntParams nt
  130             , Cry.tsConstraints = Cry.ntConstraints nt
  131             , Cry.tsDef = Cry.TRec (Cry.ntFields sc)
  132             , Cry.tsDoc = Cry.ntDoc nt
  133             }
  134       translateTySynDecl tysyn True
  135     Cry.Abstract -> do
  136       tps <- mapM translateTParam (Cry.ntParams nt)
  137       nm <- translateName (Cry.ntName nt)
  138       addDecl $ Decl.TypeDecl tps nm

  140 warnMaybe :: IsaM_ () -> IsaM ()
  141 warnMaybe f = catchMaybe f $ \e -> do
  142   warn e
  143   addDecl (Decl.Commented ("Translation failure: " ++ Error.showErr e) Decl.NoDecl)

  145 translateModule :: Cry.Module -> IsaM ()
  146 translateModule m = do
  147   deps <- asks envCryDeps
  148   let mpos = case Map.lookup (Cry.mName m) (Deps.moduleDeps deps) of
  149         Just (lm, _) | Cry.InFile fp <- Cry.lmFilePath lm -> 
  150           Just (Position.emptyRange { Position.source = fp })
  151         _ -> Nothing
  152   maybeWith withPosition mpos $ 
  153     rethrow' (Error.ModuleTranslationFailure (Cry.mName m)) $ do
  154       forM_ (Cry.mTySyns m) $ \tysyn -> 
  155         warnMaybe (translateTySynDecl tysyn False)
  156       forM_ (Cry.mNominalTypes m) $ \nt -> 
  157         warnMaybe (translateNominalType nt)
  158       forM_ (Cry.mDecls m) $ \d ->
  159         warnMaybe (translateDeclGroup (Deps.stripProofApps d))

  161 translateSingleExpr :: Name.Name -> Cry.Schema -> Cry.Expr -> IsaM ()
  162 translateSingleExpr nm schem e = 
  163   withSchemaExpr schem (Deps.stripProofApps e) $ \schem' body -> do
  164     importModuleDeps schem
  165     importModuleDeps e
  166     (args, body') <- translateAbs body
  167     addDecl $ Decl.Definition False (Binding.Binding nm schem') args body' []

  169 translateRecordType :: Cry.RecordMap Cry.Ident Cry.Type -> IsaM Type
  170 translateRecordType rm = do
  171   args <- forM (Cry.canonicalFields rm) $ \(fieldNm, fieldTp) -> do
  172     fieldTp' <- translateType fieldTp
  173     return $ (fieldNm, fieldTp')
  174   addDecl $ Decl.RecordDecl (map (Cry.unpackIdent . fst) args)
  175   let flds = map fst (Cry.canonicalFields rm)
  176   args' <- forM args $ \(nm,tp) -> do
  177     return (recordField flds nm, tp)
  178   return $ tRecord args'

  181 translateTySynDecl :: Cry.TySyn -> Bool -> IsaM ()
  182 translateTySynDecl tysyn fromNominal = rethrow' (UnsupportedTypeDecl nm tysyn) $ do
  183   let tparams = Cry.tsParams tysyn
  184   {- NOTE: we don't need to include the constraints in the 
  185      translated output, because
  186      Cryptol will enforce that they are present in any signatures
  187      using this type synonym.
  188   -}
  189   withTParams tparams $ do
  190     t' <- translateType (Cry.tsDef tysyn)
  191     nm' <- translateName nm
  192     targs <- mapM translateTParam tparams
  193     addDecl $ Decl.TypeSyn targs nm' t'
  194     when fromNominal $ do
  195       {- we need to create a dummy constructor if this synonym was
  196          originally a nominal type -}
  197       guards <- mapM translateType (Cry.tsConstraints tysyn)
  198       let ct = tAbs targs (tGuard guards (tFun t' t'))
  199       let b = Binding.Binding nm' ct
  200       let x = Name.SimpleName "x"
  201       addDecl $ Decl.Definition False b [Binding.Binding x t'] (Expr.ConstrainT (Expr.Var x) t') []
  202   where
  203     nm = Cry.tsName tysyn

  205 translateSchema :: Cry.Schema -> IsaM Type
  206 translateSchema s = do
  207   tyArgs <- mapM translateTParam (Cry.sVars s)
  208   guards <- mapM translateType (Cry.sProps s)
  209   body <- translateType (Cry.sType s)
  210   return $ tAbs tyArgs (tGuard guards body)

  215 translateDeclGroup :: Cry.DeclGroup -> IsaM ()
  216 translateDeclGroup = \case
  217   Cry.Recursive ds -> do
  218     bodies <- fmap (zip (map Cry.dName ds)) $ mapM translateDecl ds
  219     let toRecMap (_, (b,mbody)) = case mbody of
  220           Right _ -> Just (Binding.bindName b, Binding.bindName (Decl.asRecImpl b))
  221           _ -> Nothing
  222     let recMap = Map.fromList $ mapMaybe toRecMap bodies
  223     let fixRecs = Expr.mapNames (\nm -> fromMaybe nm (Map.lookup nm recMap))  
  224 
  225     bs <- fmap catMaybes $ forM bodies $ \(crynm, (b,mbody)) -> do
  226       case mbody of
  227         Right (args, body)  -> do
  228           addDecl (Decl.Commented "Recursive call stub (to be overridden)" $ (Decl.ConstDecl False (Decl.asRecImpl b)))
  229           addDecl (Decl.Commented "Recursive call schema" $ Decl.Definition False (Decl.asRecSpec b) args (fixRecs body) [])
  230           return (Just (crynm, Binding.bindName b))
  231         Left (StubbedFunction{}) | isStubbedFnName b -> do
  232           addDecl (Decl.Commented ("Stubbed Function") (Decl.ConstDecl False b))
  233           return Nothing
  234         Left er | Options.keepGoing -> do
  235           addDecl (Decl.Commented ("Incomplete translation: " ++ Error.showErr er) (Decl.ConstDecl False b))
  236           return Nothing
  237         Left er -> throwError er
  238     addDecl (Decl.FixpointLocale (map snd bs))
  239     mapM_ (\(crynm,isanm) -> addHashDecl crynm isanm) bs
  240 
  241   Cry.NonRecursive d -> do
  242     (b,mbody) <- translateDecl d
  243     case mbody of
  244       Left (StubbedFunction{}) -> addDecl (Decl.ConstDecl False b)
  245       Left er -> case Options.keepGoing of
  246         True -> addDecl (Decl.Commented ("Incomplete translation: " ++ Error.showErr er) (Decl.ConstDecl False b))
  247         False -> throwError er
  248       Right (args, body) -> do
  249         addDecl (Decl.Definition False b args body [])
  250         addHashDecl (Cry.dName d) (Binding.bindName b)

  259 setNameKind :: Cry.Kind -> Name.Name -> Name.Name

  265 withTParams :: [Cry.TParam] -> IsaM a -> IsaM a
  266 withTParams tps f = do
  267   tps' <- mapM go tps
  268   withNames tps' f
  269   where
  270     go :: Cry.TParam -> IsaM (Int, Name.Name)
  271     go tp = withPosition (Cry.tvarSource $ Cry.tpInfo tp) $ do
  272       nm' <- case Cry.tvarDesc (Cry.tpInfo tp) of
  273         Cry.TVFromSignature nm -> translateName nm
  274         _ -> simpleNameType "'"
  275       return (Cry.tpUnique tp, setNameKind (Cry.tpKind tp) nm')

  277 addGuards :: [Cry.Prop] -> Cry.Schema -> Cry.Schema
  278 addGuards gs s = s { Cry.sProps = nub $ Cry.sProps s ++ gs}
  279 
  280 extraGuard :: Cry.Ctxt -> Cry.TParam -> IsaM (Maybe (Cry.Prop))
  281 extraGuard ctx tv = case Cry.tpKind tv of
  282   Cry.KNum -> case Cry.cryIsFinType ctx tp of
  283     Cry.SolvedIf [] -> return Nothing
  284     _ -> do
  285       let rng = Cry.tvarSource $ Cry.tpInfo tv
  286       Options.log (-1) $  "[warning] at " ++ pp rng ++ "\n Type parameter is not constrained to fin: " ++ pp tp
  287       return $ Just $ Cry.TCon (Cry.PC Cry.PFin) [tp]
  288   Cry.KType -> return $ Just $ Cry.TCon (Cry.PC Cry.PEq) [tp]
  289   _ -> return Nothing
  290   where
  291     tp = Cry.TVar (Cry.TVBound tv)

  293 withDeclBinding :: Cry.Decl -> (Binding.Binding -> Cry.Expr -> IsaM a) -> IsaM a
  294 withDeclBinding d f = rethrow' (UnsupportedDecl d) $ case Cry.dDefinition d of
  295   Cry.DPrim -> throwError $ UnsupportedEntity $ "DeclDef.DPrim"
  296   Cry.DForeign{} -> throwError $ UnsupportedEntity $ "DeclDef.Foreign"
  297   Cry.DExpr e -> withSchemaExpr (Cry.dSignature d) e $ \t body -> do
  298     nm <- translateName (Cry.dName d)
  299     f (Binding.Binding nm t) body
  300 
  301 withSchemaExpr :: Cry.Schema -> Cry.Expr -> (Type -> Cry.Expr -> IsaM a) -> IsaM a
  302 withSchemaExpr s e f = do
  303   let 
  304     (tparams, body) = stripETAbs e
  305     props = Cry.sProps s
  306     ctx = Cry.buildSolverCtxt props
  307   guards <- catMaybes <$> mapM (extraGuard ctx) (Cry.sVars s)
  308   let s' = addGuards guards s
  309 
  310   case tparams == (Cry.sVars s') of
  311     True -> withTParams (Cry.sVars s') $ do
  312       t <- translateSchema s'
  313       f t body
  314     _ -> throwError $ UnexpectedSignature s' e

  316 tryError :: MonadError e m => m a -> m (Either e a)
  317 tryError action = (Right <$> action) `catchError` (pure . Left)

  319 translateDecl :: Cry.Decl -> IsaM (Binding.Binding, Either TranslationError ([Binding.Binding], Expr))
  320 translateDecl d = withDeclBinding d $ \b body -> do
  321   case isStubbedFnName b of
  322     True -> return (b, Left $ StubbedFunction d)
  323     False -> do
  324       res <- (tryError $ translateAbs body)
  325       return $ (b, res)

  327 -- | Returns Undefined if translation fails
  328 translateDecl' :: Cry.Decl -> IsaM (Binding.Binding, Expr)
  329 translateDecl' d = withDeclBinding d $ \b body -> (tryError $ translateExpr body) >>= \case
  330   Left err -> return $ (b, Expr.Undefined (Binding.bindType b) (Error.showErr err))
  331   Right body' -> return $ (b, body')

  341 pathToTheory :: Cry.ModPath -> IsaM Name.TheoryName
  342 pathToTheory = \case
  343   Cry.TopModule mnm -> do
  344     thynm <- modToTheory mnm
  345     cthy <- curTheory
  346     if (thynm == cthy || isBuiltinThy thynm) then
  347       return thynm
  348     else 
  349       return $ thynm { Name.thyQualified = True}
  350   Cry.Nested{} -> throwError $ UnsupportedEntity $ "ModPath.Nested"

  352 isBinOp :: Cry.Name -> Bool
  353 isBinOp nm = isJust (Cry.nameFixity nm)

  355 translateName :: Cry.Name -> IsaM Name.Name
  356 translateName nm = do
  357   let ident = Cry.nameIdent nm
  358   k <- case Cry.nameNamespace nm of
  359     Cry.NSType -> return $ Name.Typ
  360     Cry.NSValue -> return $ Name.Term
  361     Cry.NSConstructor -> return $ Name.Term
  362     Cry.NSModule -> throwError $ UnsupportedEntity $ "Namespace.NSModule"
  363 
  364   thynm <- case Cry.nameInfo nm of
  365     Cry.GlobalName _ on -> pathToTheory (Cry.ogModule on)
  366     Cry.LocalName{} -> curTheory
  367   let syn = case Cry.nameFixity nm of
  368         Just{} -> Syntax.InfixSyn (Cry.unpackIdent ident)
  369         Nothing -> Syntax.NoSyn
  370   let isaIdent = case (Name.isTypeK k, Cry.nameInfo nm) of
  371         (True, Cry.LocalName{}) -> "'" ++ Cry.unpackIdent ident
  372         _ -> Cry.unpackIdent ident
  373   return $ Name.Name thynm isaIdent syn k

  380 translateTCon :: Cry.TCon -> [Cry.Type] -> IsaM Type
  381 translateTCon tc es = case (tc, es) of
  382   (Cry.PC Cry.PEqual, [e1, e2]) -> tEquals <$> translateType e1 <*> translateType e2
  383   (Cry.PC Cry.PLiteral, [e1, e2]) -> tLiteral <$> translateType e1 <*> translateType e2
  384   (Cry.PC Cry.PFin, [e1]) -> tFinite <$> translateType e1
  385   (Cry.PC Cry.PGeq, [e1,e2]) -> tGeq <$> translateType e1 <*> translateType e2
  386   (Cry.PC Cry.PSignedCmp, [t]) -> tSignedCmp <$> translateType t
  387   (Cry.PC Cry.PRing, [t]) -> tRing <$> translateType t
  388   (Cry.PC Cry.PIntegral, [t]) -> tIntegral <$> translateType t
  389   (Cry.PC Cry.PEq, [t]) -> tEq <$> translateType t
  390   (Cry.PC Cry.PLogic, [t]) -> tLogic <$> translateType t
  391   (Cry.PC Cry.PCmp, [t]) -> tCmp <$> translateType t
  392   (Cry.PC Cry.PAnd, [t1,t2]) -> tAnd <$> translateType t1 <*> translateType t2
  393   (Cry.PC Cry.PZero, [t]) -> tZero <$> translateType t
  394 
  395   (Cry.TC (Cry.TCNum i), []) -> return $ Expr.IntLit i
  396   (Cry.TC Cry.TCBit, []) -> return $ tBit
  397   _ | Just e1 <- asWordType (Cry.TCon tc es) -> tWord <$> translateType e1
  398   (Cry.TC Cry.TCSeq, [e1, e2]) -> tSeq <$> translateType e1 <*> translateType e2
  399   (Cry.TC Cry.TCInteger, []) -> return $ tInt
  400   (Cry.TC Cry.TCInf, []) | Options.keepGoing -> do
  401     warn (UnsupportedType (Cry.TCon tc es))
  402     mkUnsupportedT "Inf"
  403 
  404   (Cry.TC Cry.TCFun, [e1,e2]) -> tFun <$> translateType e1 <*> translateType e2
  405   (Cry.TC (Cry.TCTuple _), _) -> tTuple <$> mapM translateType es
  406   (Cry.TC Cry.TCIntMod, [t]) -> tZ <$> translateType t
  407   
  408   (Cry.TF Cry.TCWidth, [e1]) -> tWidth <$> translateType e1
  409   (Cry.TF Cry.TCMul, [e1, e2]) -> tMul <$> translateType e1 <*> translateType e2
  410   (Cry.TF Cry.TCMod, [e1, e2]) -> tMod <$> translateType e1 <*> translateType e2
  411   (Cry.TF Cry.TCSub, [e1, e2]) -> tMinus <$> translateType e1 <*> translateType e2
  412   (Cry.TF Cry.TCDiv, [e1, e2]) -> tDiv <$> translateType e1 <*> translateType e2
  413   (Cry.TF Cry.TCAdd, [t1, t2]) -> tAdd <$> translateType t1 <*> translateType t2
  414   (Cry.TF Cry.TCExp, [t1, t2]) -> tExp <$> translateType t1 <*> translateType t2
  415   (Cry.TF Cry.TCLenFromThenTo, [x,y,z]) -> 
  416     tFromThenTo <$> translateType x <*> translateType y <*> translateType z
  417   (Cry.TF Cry.TCMin, [t1,t2]) -> tMin <$> translateType t1 <*> translateType t2
  418   (Cry.TF Cry.TCMax, [t1,t2]) -> tMax <$> translateType t1 <*> translateType t2
  419   _ -> throwError $ UnsupportedType $ Cry.TCon tc es

  421 translateType :: Cry.Type -> IsaM Type
  422 translateType = rethrow UnsupportedType $ \case
  423   Cry.TCon tc args -> translateTCon tc args
  424   Cry.TVar (Cry.TVFree _ _ _ tinfo) -> case Cry.tvarDesc tinfo of
  425     Cry.TVFromSignature nm -> nameToVar =<< translateName nm
  426     ts -> throwError $ UnsupportedEntity $ "! TypeSource.TVFromSignature: " ++ (show ts)
  427   Cry.TVar (Cry.TVBound tp) -> translateTParam tp
  428   Cry.TUser userT args _ -> do
  429     userT' <- translateName userT
  430     args' <- mapM translateType args
  431     return $ Expr.Ctr args' userT'
  432   Cry.TRec r -> translateRecordType r
  433   Cry.TNominal nt args -> do
  434     userT' <- translateName (Cry.ntName nt)
  435     args' <- mapM translateType args
  436     return $ Expr.Ctr args' userT'

  445   (Cry.EVar nm, [t1, t2]) 
  446     | Cry.GlobalName{} <- Cry.nameInfo nm, "number" <- Cry.nameIdent nm

  458 mkUnsupportedT :: String -> IsaM Type
  459 mkUnsupportedT nm = do
  460   base <- simpleNameType (nm ++ "'")
  461   addDecl (Decl.TypeDecl [] base)
  462   lhs <- simpleNameType nm
  463   let rhs = tUnsupported (Expr.Var base)
  464   addDecl (Decl.TypeSyn [] lhs rhs)
  465   addDecl (Decl.Import unsupportedThy)
  466   return (Expr.Var lhs)

  473 listAbs :: [(Cry.Name,Cry.Type)] -> Cry.Expr -> Cry.Expr
  474 listAbs ((nm,t):xs) e = Cry.EAbs nm t (listAbs xs e)
  475 listAbs [] e = e
  476 
  477 stripAbs :: Cry.Expr -> ([(Cry.Name,Cry.Type)], Cry.Expr)
  478 stripAbs = \case
  479   Cry.EAbs nm t e ->
  480     let (hdr, e') = stripAbs e
  481     in ((nm,t):hdr, e')
  482   -- lift abstractions out of where-bindings if there are no
  483   -- name clashes. It's unclear if clashes of this kind 
  484   -- are actually possible in
  485   -- practice, so this is mostly for safety.
  486   Cry.EWhere e decls -> 
  487     let 
  488       (hdr, e') = stripAbs e
  489       go ((nm,t):args) = if Deps.containsName nm decls then
  490         ([], listAbs ((nm,t):args) e')
  491         else 
  492           let (hdr_, e_) = go args
  493           in ((nm,t):hdr_, e_)
  494       go [] = ([], e')
  495       (hdr', body) = go hdr
  496     in (hdr', Cry.EWhere body decls)
  497 
  498   e -> ([],e)

  500 stripApp :: Cry.Expr -> (Cry.Expr, [Cry.Expr])
  501 stripApp = \case
  502   Cry.EApp e arg ->
  503     let (e', args) = stripApp e
  504     in (e', args ++ [arg])
  505   e -> (e, [])

  507 typeOf :: Cry.Expr -> IsaM (Cry.Type)
  508 typeOf e = do
  509   tyEnv <- asks envTypeEnv
  510   (IO.liftIO $ (Right <$> go tyEnv e) 
  511     `catch` (\(er :: SomeException) -> return $ Left $ Error.CryptolTypeOfError e (show er))) >>= \case
  512     Right s -> return $ Cry.sType s
  513     Left er -> throwError er
  514   where
  515     go :: Map.Map Cry.Name Cry.Schema -> Cry.Expr -> IO Cry.Schema
  516     go tyEnv e_top = evaluate $!! (Cry.fastSchemaOf tyEnv e_top)

  518 translateAbs :: Cry.Expr -> IsaM ([Binding.Binding], Expr)
  519 translateAbs e = do
  520   let (hdr, body) = stripAbs e
  521   hdr' <- forM hdr $ \(nm,t) -> do
  522     nm' <- translateName nm
  523     t' <- translateType t
  524     return $ Binding.Binding nm' t'
  525   let hdr_schem = map (\(nm, t) -> (nm, Cry.tMono t)) hdr
  526   body' <- withBindings hdr_schem $ translateExpr body
  527   return $ (hdr', body')
  528 
  529 absExpr :: [Binding.Binding] -> Expr -> Expr
  530 absExpr binds e = eAbs (map (\b -> bind (Binding.bindName b) (Binding.bindType b)) binds) e

  532 translateMatches ::
  533   Type {-^ target seq elems -} ->
  534   Cry.Expr ->
  535   [Cry.Match] ->
  536   IsaM (Maybe Type, Expr)
  537 translateMatches _ e_final [] = do
  538   e_final' <- translateExpr e_final
  539   return (Nothing, e_final')
  540 translateMatches outer_elemT e_final (Cry.Let d:ms) = do
  541   withDeclBinding d $ \b body -> do
  542     body' <- translateExpr body
  543     (len, result) <- translateMatches outer_elemT e_final ms
  544     return $ (len, letExpr [(b,body')] result)
  545 translateMatches outer_elemT e ((Cry.From nm lenT elemT body):ms) = do
  546   elemT' <- translateType elemT
  547   nm' <- translateName nm
  548   body' <- translateExpr body
  549   lenT' <- translateType lenT
  550   withBindings [(nm, Cry.tMono elemT)] $ do
  551     (m_inner_len, match_body) <- translateMatches outer_elemT e ms
  552     case m_inner_len of
  553       Just inner_len -> do
  554         let target_len = (tMul lenT' inner_len)
  555         let s = seqConcatMap inner_len elemT' lenT' outer_elemT (nm', match_body) body'
  556         return (Just target_len, s)
  557       Nothing -> do
  558         return $ (Just lenT', seqCompr1 lenT' elemT' outer_elemT (nm', match_body) body')

  561 tupleType tys = Cry.TCon (Cry.TC (Cry.TCTuple (length tys))) tys

  564 matchToTuple :: [Cry.Match] -> [(Cry.Schema, Cry.Name)]
  565 matchToTuple ms = map go ms
  566   where
  567     go :: Cry.Match -> (Cry.Schema, Cry.Name)
  568     go = \case
  569       Cry.From nm _ elemT _ -> (Cry.tMono elemT, nm)
  570       Cry.Let d -> (Cry.dSignature d, Cry.dName d)

  572 toMono :: Cry.Schema -> IsaM Cry.Type
  573 toMono s = case Cry.isMono s of
  574   Just t -> return t
  575   Nothing -> throwError $ UnexpectedPolymorphism s

  577 asWordType :: Cry.Type -> Maybe Cry.Type
  578 asWordType (Cry.TCon (Cry.TC Cry.TCSeq) [e1, (Cry.TCon (Cry.TC Cry.TCBit) [])]) = Just e1
  579 asWordType _ = Nothing

  582 isHexType :: Cry.Type -> Bool
  583 isHexType t = isJust (asWordType t)

  585 recordField :: [Cry.Ident] -> Cry.Ident -> Name.Name
  586 recordField flds field = 
  587   let rnm = Decl.recordName (map Cry.unpackIdent flds)
  588   in Name.Name (Name.nmThy rnm) (Cry.unpackIdent field) Syntax.NoSyn Name.Term

  590 isRec :: Cry.Type -> Maybe (Cry.RecordMap Cry.Ident Cry.Type)
  591 isRec t = Cry.tIsRec t <|> do
  592   (nt,_) <- Cry.tIsNominal t
  593   Cry.Struct sc <- return $ Cry.ntDef nt
  594   return $ Cry.ntFields sc

  596 lookupEnumCon :: [Cry.EnumCon] -> Cry.Ident -> Maybe Cry.EnumCon
  597 lookupEnumCon [] _ = Nothing
  598 lookupEnumCon (ec:ecs) i = case Cry.nameIdent (Cry.ecName ec) == i of
  599   True -> Just ec
  600   False -> lookupEnumCon ecs i

  602 translateExpr :: Cry.Expr -> IsaM (Expr)
  603 translateExpr = rethrow UnsupportedExpr $ \case
  604   Cry.EList es (Cry.TCon (Cry.TC Cry.TCBit) []) -> do
  605     es' <- mapM translateExpr es
  606     return $ constrain (wordFromBits es') (tWord (Expr.IntLit (fromIntegral (length es))))
  607 
  608   Cry.EList es t -> do
  609     es' <- mapM translateExpr es
  610     t' <- translateType t
  611     let es'' = map (\e_ -> coerceToExpr e_ t') es'
  612     return $ constrain (seqExpr es'') (tSeq (Expr.IntLit (fromIntegral (length es))) t')
  613 
  614   Cry.ETuple es -> tupleExpr <$> mapM translateExpr es
  615   Cry.ERec rm -> do
  616     let flds = map fst (Cry.canonicalFields rm)
  617     args <- forM (Cry.canonicalFields rm) $ \(fieldNm, fieldExpr) -> do
  618       fieldTp' <- translateExpr fieldExpr
  619       return $ (recordField flds fieldNm, fieldTp')
  620     return $ recordCtr args
  621 
  622   Cry.ESel e sel -> do
  623     case sel of
  624       Cry.ListSel i _ -> (Cry.tIsSeq <$> typeOf e) >>= \case
  625         Just (lenT, elemT) -> do
  626           lenT' <- translateType lenT
  627           elemT' <- translateType elemT
  628           e' <- translateExpr e
  629           return $ seqSel lenT' elemT' tInt  e' (constrain (Expr.IntLit (fromIntegral i)) tInt)
  630         Nothing -> throwError $ UnsupportedEntity $ "Expr.ESel." ++ show sel
  631       Cry.TupleSel i _ -> (Cry.tIsTuple <$> typeOf e) >>= \case 
  632         Just ts -> do
  633           let len = length ts
  634           x <- simpleNameExpr "x"
  635           e' <- translateExpr e
  636           t' <- translateType (ts !! i)
  637           return $ constrain (tupleSelect x e' i len) t'
  638         Nothing -> throwError $ UnsupportedEntity $ "Expr.ESel." ++ show sel
  639       Cry.RecordSel field (Just flds) -> do
  640         e' <- translateExpr e
  641         -- Qualify record accessors to handle name clashes
  642         return $ appExpr (Expr.Var (recordField flds field)) [e']
  643       Cry.RecordSel field Nothing -> (isRec <$> typeOf e) >>= \case 
  644         Just rm -> translateExpr (Cry.ESel e (Cry.RecordSel field (Just (map fst (Cry.canonicalFields rm)))))
  645         Nothing -> do
  646           throwError $ UnsupportedEntity $ "Cry.RecordSel: unexpected source record type"
  647 
  648   Cry.ESet t e sel v -> case sel of
  649     Cry.RecordSel field (Just flds) -> do
  650       e' <- translateExpr e
  651       v' <- translateExpr v
  652       return $ recordUpdate e' (recordField flds field) v'
  653     Cry.RecordSel field Nothing -> (isRec <$> typeOf e) >>= \case 
  654         Just rm -> translateExpr (Cry.ESet t e (Cry.RecordSel field (Just (map fst (Cry.canonicalFields rm)))) v)
  655         Nothing -> do
  656           throwError $ UnsupportedEntity $ "Cry.RecordSel: unexpected source record type"
  657     _ -> throwError $ UnsupportedEntity $ "Cry.ESet: only record types are supported"
  658 
  659   Cry.EIf p eT eF -> do
  660     p' <- translateExpr p
  661     eT' <- translateExpr eT
  662     eF' <- translateExpr eF
  663     return $ ifExpr p' eT' eF'
  664 
  665   Cry.ECase e cs dflt -> do
  666     e' <- translateExpr e
  667     eT <- typeOf e
  668     case Cry.tNoUser eT of
  669       Cry.TNominal nt _ -> case Cry.ntDef nt of
  670         Cry.Enum ctrs -> do
  671           cs' <- forM (Map.toList cs) $ \(nm,Cry.CaseAlt vars body) -> case lookupEnumCon ctrs nm of
  672             Just ctr -> do
  673               ctr' <- translateName (Cry.ecName ctr) >>= nameToVar
  674               withBindings (map (\(n,t) -> (n, Cry.tMono t)) vars) $ do
  675                 vars' <- mapM (\(n,_) -> translateName n >>= nameToVar) vars
  676                 body' <- translateExpr body
  677                 return $ (appExpr ctr' vars', body')
  678             Nothing -> fail $ "Missing constructor for name: " ++ show nm
  679           case dflt of
  680             Nothing -> return $ Expr.Case e' cs'
  681             Just (Cry.CaseAlt vars body) -> case vars of
  682               [(v,_)] | Just{} <- T.stripPrefix "__" $ Cry.identText $ Cry.nameIdent v -> do
  683                 body' <- translateExpr body
  684                 return $ Expr.Case e' (cs' ++ [(dummy, body')])
  685               _ -> fail $ "Unexpected default case" ++ show (length vars)
  686         _ -> fail "Unexpected case expression"
  687       _ -> fail "Unexpected type for case expression"
  688 
  689   (Cry.EComp _lenT elemT e [inner_matches]) -> do
  690     elemT' <- translateType elemT
  691     (_, result) <- translateMatches elemT' e inner_matches
  692     return result
  693   (Cry.EComp _lenT elemT e matches) -> do
  694     elemT' <- translateType elemT
  695     let 
  696       go :: [Cry.Match] -> IsaM ([(Cry.Name, Cry.Schema)], ((Type,Type), Expr))
  697       go inner_matches = do
  698         let (matchSchems, matchNms) = unzip $ matchToTuple inner_matches
  699         matchTs <- mapM toMono matchSchems
  700         matchTs' <- mapM translateType matchTs
  701         (mslen, result) <- translateMatches (tTuple matchTs') (Cry.ETuple (map Cry.EVar matchNms)) inner_matches
  702         case mslen of
  703           Nothing -> throwError $ UnsupportedMatchShape matches
  704           Just slen -> return (zip matchNms matchSchems, ((slen, (tTuple matchTs')), result))
  705     (match_names, match_bodies) <- unzip <$> (mapM go matches)
  706     let ((len_final, elem_final), body) = seqZipMany match_bodies
  707     withBindings (concat match_names) $ do
  708       e' <- translateExpr e
  709       nms' <- forM match_names $ \nms -> forM nms $ \(nm,schem) -> do
  710         nm' <- translateName nm
  711         nmT <- toMono schem >>= translateType
  712         return (nm',nmT)
  713       return $ seqCompr1' len_final elem_final elemT' (tupleAbs nms' e') body
  714 
  715   Cry.EVar nm  -> nameToVar =<< translateName nm
  716   Cry.ETAbs{} -> throwError $ UnsupportedEntity $ "Cry.ETAbs"
  717 
  718   et@Cry.ETApp{} -> do
  719     let (e,ts) = collectTApps et
  720     case unNumber e ts of
  721       Just (i,t) -> do
  722         t' <- translateType t
  723         return $ constrain (Expr.IntLitCtr i (isHexType t)) t'
  724       _ -> do
  725         ts' <- mapM translateType ts
  726         e' <- translateExpr e
  727         let ret = tApp e' ts'
  728         case e of 
  729           Cry.EVar nm | Just{} <- Cry.nameFixity nm ->
  730             return $ unOp ret
  731           _ -> return ret
  732 
  733   Cry.EApp (Cry.EApp e1 e2) e3 |
  734     (Cry.EVar nm,ts) <- collectTApps e1,
  735     isBinOp nm
  736     -> do
  737     nm' <- translateName nm
  738     ts' <- mapM translateType ts
  739     let e1' = tApp (Expr.Var nm') ts'
  740 
  741     e2' <- translateExpr e2
  742     e3' <- translateExpr e3
  743 
  744     return $ Expr.BinOp e2' e1' e3'
  745 
  746   e@(Cry.EApp{}) -> do
  747     let (e1, args) = stripApp e
  748     e1' <- translateExpr e1
  749     args' <- mapM translateExpr args
  750     case e1' of
  751       {- adds an empty type argument for user-made definitions with
  752          arguments in order to invoke the implicit argument coercion
  753          parser -}
  754       Expr.Var nm 
  755         | not (isBuiltinName nm)
  756         , not (null args) -> 
  757           return $ appExpr (Expr.TApp e1' []) args'
  758       _ -> return $ appExpr e1' args'
  759 
  760   e@(Cry.EAbs{}) -> do
  761     (binds,body) <- translateAbs e
  762     return $ absExpr binds body
  763 
  764   Cry.ELocated r e -> withPosition r $ translateExpr e
  765   -- do we need proofs?
  766   Cry.EProofAbs _ e -> translateExpr e
  767   Cry.EProofApp e -> translateExpr e
  768   Cry.EWhere body grps -> withDeclGroups grps $ do
  769     let f grp = case grp of
  770           -- a let-binding inside a recursive block is deemed recursive
  771           -- regardless if it actually has mutually recursive bindings
  772           -- in general this can be detected by examining the bindings themselves,
  773           -- but this handles the common case where only one binding exists
  774           Cry.Recursive [d] -> (:[]) <$> translateDecl' d
  775           Cry.Recursive ds | Options.keepGoing -> do
  776             undef <- mkUndefined (UnsupportedRecursion ds)
  777             forM ds $ \d -> withDeclBinding d $ \b _ ->
  778               return (b, undef (Binding.bindType b))
  779           Cry.Recursive ds -> do
  780             throwError $ UnsupportedRecursion ds
  781           Cry.NonRecursive d -> (:[]) <$> translateDecl' d
  782     binds <- concat <$> mapM f grps
  783     body' <- translateExpr body
  784     return $ letExpr binds body'
  785   Cry.EPropGuards [g] _ -> translateGuards Nothing [g]
  786   -- if we have multiple branches, we need to coerce each into
  787   -- the target type, in case the branches disagree on the Isabelle type
  788   Cry.EPropGuards gs t -> do
  789     t' <- translateType t
  790     translateGuards (Just t') gs
  791 
  792 maybeCoerce :: Maybe Type -> Expr -> Expr
  793 maybeCoerce Nothing e = e
  794 maybeCoerce (Just t) e = coerceToExpr e t
  795 
  796 translateGuards :: Maybe Type -> [([Cry.Type], Cry.Expr)] -> IsaM (Expr)
  797 translateGuards outerT [(ts,e)] = do
  798   e' <- translateExpr e
  799   ts' <- mapM translateType ts
  800   return $ eAssuming ts' (maybeCoerce outerT e')
  801 translateGuards outerT ((ts,e):gs) = do
  802   gs' <- translateGuards outerT gs
  803   e' <- translateExpr e
  804   ts' <- mapM translateType ts
  805   return $ ifExpr (eHolds ts') (maybeCoerce outerT e') gs'
  806 translateGuards _ [] = throwError $ UnsupportedEntity $ "Expr.EPropGuards: empty"
