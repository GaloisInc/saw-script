   36 data Decl = 
   37     Definition 
   38       Bool {- ^ qualified -}
   39       Binding.Binding {- ^ const name and type -}
   40       [Binding.Binding] {- ^ argument name/types -}
   41       Expr.Expr {- ^ body -}
   42       [Name.Name] {- ^ extra dependencies -}
   43   | ConstDecl Bool Binding.Binding
   44   | TypeSyn [Expr.Type] Name.Name Expr.Type
   45   | TypeDecl [Expr.Type] Name.Name
   46   | Import Name.TheoryName
   47   | RecordDecl [String]
   48   | DatatypeDecl [Expr.Type] Name.Name [(Name.Name,[Expr.Type])]
   49   | Commented String Decl
   50   | FixpointLocale [Name.Name]
   51   | HashDecl 
   52       Name.Name {- ^ const name -}
   53       Int {- ^ deep cryptol hash -}
   54   | NoDecl
   55   deriving (Eq, Ord, Show)
   56 
   57 newtype Decls = Decls { toList :: [Decl] }
   58   deriving (Semigroup, Monoid, Eq, Ord)
   59 
   60 traverseDecls:: Monad m =>  Decls -> (Decl -> m Decl) -> m Decls
   61 traverseDecls (Decls ds) f = do
   62   ds' <- mapM f (List.nub ds)
   63   return $ Decls (List.nub ds')
   64 
   65 traverseImports :: Monad m =>  Decls -> (Name.TheoryName -> m Name.TheoryName) -> m Decls
   66 traverseImports ds f = do
   67   let (imports, Decls ds') = strip (\case Import{} -> True; _ -> False) ds
   68   imports' <- mapM (\case Import nm -> Import <$> f nm; _ -> undefined) (List.nub imports)
   69   return $ Decls (imports' ++ ds')
   70 
   71 liftDeclPred :: (Decl -> Bool) -> Decl -> Bool
   72 liftDeclPred f = \case
   73   Commented _ d -> liftDeclPred f d
   74   d -> f d
   75 
   76 strip :: (Decl -> Bool) -> Decls -> ([Decl], Decls)
   77 strip f (Decls ds) = let (ds_match, ds') = List.partition (liftDeclPred f) ds in (ds_match, Decls ds')
   78 
   79 filter :: (Decl -> Bool) -> Decls -> [Decl]
   80 filter f (Decls ds ) = List.filter (liftDeclPred f) ds
   81 
   82 mapMaybe :: (Decl -> Maybe a) -> Decls -> [a]
   83 mapMaybe f (Decls ds) = Maybe.mapMaybe f ds
   84 
   85 singleton :: Decl -> Decls
   86 singleton d = Decls [d]

   88 maybeQualifyCmd :: Bool -> String -> String

   92 definitionCmd :: Output.HasOutput => Bool -> Binding.Binding -> Maybe ([Binding.Binding], Expr.Expr) -> String
   93 definitionCmd qualify b (Just (args,e)) = maybeQualifyCmd qualify $
   94   "cryptol_definition " ++ 
   95         Binding.prettyBindOuter b <> " where\n" ++
   96         (Output.quotes $ (Output.out (Binding.bindName b)) <> " "
   97           <> (Output.spaces (map (Output.out . Binding.bindName) args)) <> " \\<equiv> " <> 
   98                 (Output.indent 1 $ (Output.out e)))
   99 definitionCmd qualify b Nothing = maybeQualifyCmd qualify $
  100   "cryptol_definition " <> Binding.prettyBindOuter b

  102 freshTs :: Int -> ([String], [String])
  103 freshTs i = splitAt i (take (i*2) (map (\s -> "'" ++ s) (Cry.nameList [])))

  105 recordName :: [String] -> Name.Name
  106 recordName args =
  107   let 
  108     rawnm = List.intercalate "_" args ++ "_recordT"
  109     thynm = Name.TheoryName (Name.cleanName $ Name.SimpleName rawnm) True
  110   in Name.Name thynm rawnm Syntax.NoSyn Name.Typ

  113 recordCmd :: Output.HasOutput => [String] -> Template.Template -> String
  114 recordCmd args template =
  115   let
  116     nm = Name.cleanName $ recordName (map Output.out args)
  117     commas = List.intercalate ","
  118     sorts s = commas $ List.replicate (length args + 1) s
  119     (ts, ts') = freshTs (length args + 1)
  120     coercion = mapRecord nm args (\(_,x) -> coerceExpr x)
  121     fields = map (\(anm,t) -> Output.out anm ++ " :: " ++ t) (zip args ts)
  122     catom s = "(" ++ s ++ "::coercible_atom" ++ ")"
  123   in Output.out $ Template.apply template $ 
  124     [ nm
  125     , commas (map catom ts)
  126     , commas (map catom ts')
  127     , commas (take (length args) ts)
  128     , sorts "coercible_atom"
  129     , sorts "strip_type"
  130     , Output.indent 1 $ Output.lines $ fields
  131     , Output.out coercion
  132     , Output.out $ Output.spaces $ Output.addSep "," fields
  133     , sorts "zero"
  134     , Output.out $ recordCtr $ (map (\s -> (Name.SimpleName s, Expr.IntLit 0)) args) ++ [(dots, Expr.IntLit 0)]
  135     , sorts "_"
  136     , commas ts
  137     ]

  139 collectDeps :: Expr.Expr -> [Name.Name]
  140 collectDeps e = Expr.foldSubExprs (\case Expr.Ctr _ nm -> addNm nm; Expr.Var nm -> addNm nm; _ -> id) e []
  141   where
  142     addNm :: Name.Name -> [Name.Name] -> [Name.Name]
  143     addNm nm nms = (nm:nms)
  144 
  145 dependencies :: Decl -> [Name.Name]
  146 dependencies = \case
  147   Definition _ _ _ e extra -> extra ++ collectDeps e
  148   ConstDecl{} -> []
  149   Commented _ d -> dependencies d
  150   TypeDecl{} -> []
  151   TypeSyn _ _ rhs -> collectDeps rhs
  152   RecordDecl{} -> []
  153   DatatypeDecl targs _ flds -> concat $ map collectDeps $ ((concat $ map (\(_,ts) -> ts) flds) List.\\ targs)
  154   Import{} -> []
  155   FixpointLocale nms -> (map asRecImpl nms ++ map asRecSpec nms)
  156   HashDecl nm _ -> [nm]
  157   NoDecl -> []
  158 
  159 tyHeader :: Output.HasOutput => [Expr.Expr] -> Name.Name -> String
  160 tyHeader [] nm = Output.out nm
  161 tyHeader es nm = Output.brackets (List.intercalate "," (map Output.out es)) ++ " " ++ Output.out nm
  162 
  163 lcp2 :: Eq a => [a] -> [a] -> [a]
  164 lcp2 xs ys = map fst (takeWhile (\(x,y) -> x == y) (zip xs ys))
  165 
  166 lcp :: Eq a => [[a]]-> [a]
  167 lcp [] = []
  168 lcp (x:xs) = foldl lcp2 x xs
  169 
  170 asRecImpl :: Name.HasIdent t => t -> t
  171 asRecImpl = Name.mapIdent (\ident -> ident <> "_impl")
  172 
  173 asRecSpec :: Name.HasIdent t => t -> t
  174 asRecSpec = Name.mapIdent (\ident -> ident <> "_spec")

  176 fixPointLocale :: 
  177  Output.HasOutput => 
  178  [Name.Name] -> 
  179  Template.Template ->
  180  String
  181 fixPointLocale baseNames template = 
  182   let 
  183     impls = map asRecImpl baseNames
  184     specs = map asRecSpec baseNames
  185     asms = map (\(lhs,rhs) -> eqExpr (Expr.Var lhs) (Expr.Var rhs)) (zip specs impls)
  186     defs = map (\(lhs,rhs) -> eqExpr (Expr.Var lhs) (Expr.Var rhs)) (zip baseNames impls)
  187     def_template = (Output.getTemplate "Fixpoint_Locale_Def")
  188 
  189     def_cmds = map (\(nm,e) -> 
  190          Output.out $ Template.apply def_template [Output.out nm, Output.out e])
  191          (zip impls defs)
  192 
  193     locale_name = localeName baseNames
  194     aliases = 
  195       map (\nm -> "alias " ++ Output.out nm ++ " = " 
  196                  ++ locale_name ++ "." ++ Output.out nm) baseNames
  197     hides = 
  198       map (\nm -> "hide_fact " ++ locale_name ++ "." ++ Output.out nm ++ "_def") baseNames
  199   in Output.out $ Template.apply template $ 
  200      [ locale_name
  201      , Output.spaces (Output.addSep "and" (map (\e -> Output.quotes (Output.out e)) asms))
  202      , Output.lines def_cmds
  203      , Output.lines aliases
  204      , Output.lines hides
  205      ]

  207 localeName :: [Name.Name] -> String
  208 localeName bs = lcp (map Name.identOf bs) ++ "_rec_spec"

  210 isTNum :: Expr.Type -> Bool
  211 isTNum t = case t of
  212   Expr.Var tnm | Name.TNum <- Name.nmKind tnm -> True
  213   _ -> False
  214 
  215 nameIdx :: Name.Name -> Int -> Name.Name
  216 nameIdx nm 0 = nm
  217 nameIdx nm i = Name.mapIdent (\x -> x ++ show i) nm
  218 
  219 freshVars :: [Expr.Type] -> [Expr.Type]
  220 freshVars = go [] 0
  221   where
  222     go acc i = \case
  223       ts@(Expr.Var nm:ts') -> 
  224         let t' = Expr.Var (nameIdx nm i)
  225         in case elem t' (ts ++ acc) of
  226           True -> go acc (i+1) (t':ts')
  227           False -> go (t':acc) 0 ts'
  228       (t':ts) -> go (t':acc) 0 ts
  229       [] -> reverse acc
  230 
  231 mkSameType :: Output.HasOutput => (Expr.Type, Expr.Type) -> String
  232 mkSameType (t1, t2) = Output.brackets $ case isTNum t1 of
  233   True -> "LEN" ++ Output.brackets (Output.out t1) ++ " = " ++ "LEN" ++ Output.brackets (Output.out t2)
  234   False -> "same_type TYPE" ++ Output.brackets (Output.out t1) ++ " TYPE" ++ Output.brackets (Output.out t2)

  236 datatypeDecl :: 
  237  Output.HasOutput => 
  238  [Expr.Type] -> 
  239  Name.Name ->
  240  [(Name.Name,[Expr.Type])] ->
  241  Template.Template ->
  242  String
  243 datatypeDecl targs nm flds template =
  244   let targs' = freshVars targs
  245       defs = map (\(fnm,fts) -> Name.cleanName fnm ++ " " ++ List.intercalate " " (map (Output.quotes . Output.out) fts)) flds
  246       sorts s = List.intercalate "," $ map (\t -> if isTNum t then "_" else s) targs
  247       commas xs = List.intercalate "," (map Output.out xs)
  248       shownat n = Output.brackets $ show n ++ "::nat"
  249       natcase n = case n of
  250         0 -> "0"
  251         _ -> "Suc (" ++ natcase (n-1) ++ ")"
  252       catom s = case isTNum s of
  253         True -> Output.out s
  254         False -> "(" ++ Output.out s ++ "::coercible_atom" ++ ")"
  255       mk_comp f n = Output.brackets $ List.intercalate " o " (replicate n f)
  256       mk_strip (n,(_,fts)) = Output.brackets $ case length fts of
  257         0 -> "strip (" ++ shownat n ++ ",())"
  258         1 -> "curry strip " ++ shownat n
  259         _ -> mk_comp "curry" (length fts - 1) ++ (Output.brackets $ "curry strip " ++ shownat n)
  260       mk_unstrip_rhs (fnm,fts) = case length fts of
  261         0 -> "\\<lambda> _. " ++ Output.out fnm
  262         1 -> Output.out fnm ++ " o unstrip"
  263         _ -> mk_comp "uncurry" (length fts - 1) ++ " "
  264                 ++ Output.out fnm ++ " o unstrip"
  265       mk_unstrip (n,r) = natcase n ++ " \\<Rightarrow> " ++ mk_unstrip_rhs r
  266       mk_strip_type tp = case tp of
  267         Expr.Var tnm | Name.TNum <- Name.nmKind tnm -> 
  268           "Tn LEN(" ++ Output.out tnm ++ ")"
  269         _ -> "strip_type TYPE(" ++ Output.out tp ++ ")"
  270       mk_coerce (fnm,fts) = Output.brackets $ case length fts of
  271         0 -> Output.out fnm
  272         1 -> Output.out fnm ++ " o coerce"
  273         _ -> let s = mk_comp "uncurry" (length fts - 1) ++ " " ++ Output.out fnm ++ " o coerce"
  274           in  mk_comp "curry" (length fts - 1) ++ " " ++ Output.brackets s
  275 
  276   in Output.out $ Template.apply template $ 
  277     [ Name.cleanName nm
  278     , commas targs
  279     , List.intercalate " | " defs
  280     , commas $ map mk_strip_type targs
  281     , commas targs'
  282     , sorts "coercible_atom"
  283     , List.intercalate " " (map mk_strip (zip [0::Integer ..] flds))
  284     , List.intercalate " | " (map mk_unstrip (zip [0::Integer ..] flds))
  285     , sorts "_"
  286     , sorts "strip_type"
  287     , commas $ map catom targs
  288     , commas $ map catom targs'
  289     , List.intercalate " \\<and> " $ map mkSameType (zip targs targs')
  290     , ""
  291     , List.intercalate " " $ map mk_coerce flds
  292     ]

  294 instance Output.Output Decl where
  295   out = \case
  296     Definition q b args e _ -> definitionCmd q b (Just (args,e))
  297     ConstDecl q b -> definitionCmd q b Nothing
  298     Import i -> Output.out (Name.thyNm i)
  299     TypeSyn targs nm rhs -> 
  300         "type_synonym " ++ tyHeader targs nm ++ " = " ++ Output.quotes (Output.out rhs)
  301     TypeDecl targs nm -> "typedecl " ++ tyHeader targs nm
  302     RecordDecl args -> recordCmd args (Output.getTemplate "Coercible_Record")
  303     DatatypeDecl targs nm flds -> datatypeDecl targs nm flds (Output.getTemplate "Coercible_Datatype")
  304     FixpointLocale args -> fixPointLocale args (Output.getTemplate "Fixpoint_Locale")
  305     Commented msg d -> "(* " ++ msg ++ "*)\n" ++ Output.out d
  306     HashDecl nm h -> "declare [[set_const_hash " ++ Output.out nm ++ " 0x" ++ showHex (fromIntegral h :: Word) "" ++ "]]"
  307     NoDecl -> ""
