   31 data OutputData =
   32   OutputData { curIndent :: Int, prevIndent :: Int, templates :: Map String Template.Template}
   33 
   34 noOutputData :: OutputData
   35 noOutputData = OutputData 0 0 Map.empty
   36 
   37 type HasOutput = (?outputData :: OutputData)
   38 
   39 class Output t where
   40   out  :: HasOutput => t -> String
   41 
   42 instance Output String where
   43   out = id
   44 
   45 indent :: HasOutput => Int -> (HasOutput => t) -> t
   46 indent i f =
   47     let
   48       old_data = ?outputData
   49       old_indent = curIndent old_data
   50     in let
   51       ?outputData = old_data { curIndent = (i + old_indent), prevIndent = old_indent }
   52     in f
   53 
   54 withTemplates :: Map String Template.Template -> (HasOutput => t) -> t
   55 withTemplates m f =
   56   let ?outputData = noOutputData { templates = m} in f
   57 
   58 getTemplate :: HasOutput => String -> Template.Template
   59 getTemplate nm = case Map.lookup nm (templates ?outputData) of
   60   Just template -> template
   61   Nothing -> Panic.panic "Unexpected missing template" [nm]
   62 
   63 brackets :: String -> String
   64 brackets s = "(" ++ s ++ ")"
   65 
   66 quotes :: String -> String
   67 quotes s = "\"" ++ s ++ "\""
   68 
   69 cartouche :: String -> String
   70 cartouche s = "\\<open>" ++ s ++ "\\<close>"
   71 
   72 tab :: Int
   73 tab = 2
   74 
   75 getIndent :: HasOutput => Int
   76 getIndent = curIndent ?outputData
   77 
   78 line :: HasOutput => String
   79 line = "\n" ++ concat (replicate (getIndent * tab) " ")
   80 
   81 
   82 prefix_padding :: HasOutput => String
   83 prefix_padding =
   84   let i = getIndent - (prevIndent ?outputData)
   85   in if i > 0 then
   86     concat (replicate (i * tab) " ")
   87   else ""
   88 
   89 lines :: HasOutput => [String] -> String
   90 lines [] = ""
   91 lines [l] = l
   92 lines (l:ls) = l ++ line ++ (indent 0 $ go ls)
   93   where
   94     go :: HasOutput => [String] -> String
   95     go [] = ""
   96     go [l'] = l'
   97     go (l':ls') = prefix_padding ++ l' ++ line ++ (indent 0 $ go ls')
   98 
   99 addSep :: String -> [String] -> [String]
  100 addSep _ [] = []
  101 addSep _ [l] = [l]
  102 addSep sep (l:ls) = (l ++ sep):(addSep sep ls)
  103 
  104 addSepPrefix :: String -> [String] -> [String]
  105 addSepPrefix _ [] = []
  106 addSepPrefix sep (l:ls) = (replicate (length sep) ' ' ++ l):(go ls)
  107   where
  108     go = \case
  109       (l':ls') -> (sep ++ l'):(go ls')
  110       [] -> []
  111 
  112 spaces :: [String] -> String
  113 spaces [s] = s
  114 spaces (s1:s2) = s1 ++ " " ++ (spaces s2)
  115 spaces [] = ""
  116 
  117 instance Output Template.Applied where
  118   out t = show t
