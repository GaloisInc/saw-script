   32 replace :: String -> String -> String -> String
   33 replace _ _ [] = []
   34 replace inp outp src@(hd:tl) =
   35   let (hdr,rest) = splitAt (length inp) src
   36   in case hdr == inp of
   37     True -> outp ++ replace inp outp rest
   38     False -> hd : replace inp outp tl

   40 cleanTheoryName :: Ident -> Ident
   41 cleanTheoryName i = replace "::" "_" i

   43 data TheoryName = TheoryNameCtr { thyNm :: Ident, thyQualified :: Bool }
   44   deriving (Eq, Ord)

   46 instance Show TheoryName where
   47   show nm = thyNm nm

   49 pattern TheoryName :: Ident -> Bool -> TheoryName
   50 pattern TheoryName nm b <- TheoryNameCtr nm b where
   51   TheoryName nm b = TheoryNameCtr (cleanTheoryName nm) b

   54 class HasIdent t where
   55   identOf :: t -> Ident
   56   default identOf :: (HasName t) => t -> Ident
   57   identOf t = cleanName (nameOf t)

   59   mapIdent :: (Ident -> Ident) -> t -> t
   60   default mapIdent :: (HasName t) => (Ident -> Ident) -> t -> t
   61   mapIdent f = mapName (\nm -> nm { nmStr = f (nmStr nm) })

   88 thyQual :: (HasOutput, Output t) => TheoryName -> t -> String
   89 thyQual thy t = case thyQualified thy of
   90   True -> thyNm thy ++ "." ++ out t

   97 qualifiedIdent :: HasName t => t -> String
   98 qualifiedIdent t = case nameOf t of
   99   nm@Name{} -> thyNm (nmThy nm) ++ "." ++ cleanName nm
  100   nm@SimpleName{} -> cleanName nm

  102 keywords :: [String]
  103 keywords = ["mod", "div", "xor", "undefined"]
  104 
  105 stripSuffix :: Eq a => [a] -> [a] -> Maybe [a]
  106 stripSuffix suf l = case stripPrefix suf (reverse l) of
  107   Just l' -> Just (reverse l')
  108   Nothing -> Nothing
  109 
  110 guardKeywords :: String -> String
  111 guardKeywords s = if elem s keywords then
  112   s ++ "'"
  113   else case stripSuffix "'" s of
  114     Just s' -> guardKeywords s' ++ "'"
  115     Nothing -> s
  116 
  117 symbolOverrides :: String -> String
  118 symbolOverrides s = case s of
  119   "==>" -> "\\<longrightarrow>"
  120   "/\\\\" -> "\\<and>"
  121   "\\\\/" -> "\\<or>"
  122   _ -> s
  123 
  124 cleanPrefix :: String -> String -> Ident -> Ident
  125 cleanPrefix pref repl s = case stripPrefix pref s of
  126   Just s' -> repl ++ s'
  127   -- avoid name clashes with things already prefixed with the replacement
  128   Nothing -> case stripPrefix repl s of
  129     Just{} -> repl ++ s
  130     Nothing -> s
  131 
  132 cleanSuffix :: String -> String -> Ident -> Ident
  133 cleanSuffix pref repl s = case stripSuffix pref s of
  134   Just s' -> s' ++ repl
  135   -- avoid name clashes with things already suffixed with the replacement
  136   Nothing -> case stripSuffix repl s of
  137     Just{} -> s ++ repl
  138     Nothing -> s
  139 
  140 cleanSymbol :: Char -> String
  141 cleanSymbol c = case c of
  142   '(' -> "lb"
  143   ')' -> "rb"
  144   ' ' -> "_"
  145   '<' -> "lt"
  146   '>' -> "gt"
  147   '=' -> "eq"
  148   _ -> [c]
  149 
  150 cleanNameSymbols :: Ident -> Ident
  151 cleanNameSymbols s = concat $ map cleanSymbol s
  152 
  153 
  154 cleanName :: Name -> Ident
  155 cleanName nm =
  156      symbolOverrides
  157    $ guardBackslash
  158    $ cleanPrefix "_" "i_"
  159    $ cleanPrefix "$" "s_"
  160    $ cleanSuffix "_" "_U"
  161    $ cleanNameSymbols
  162    $ guardKeywords (nmStr nm)
  163 
  164 guardUnderscores :: Ident -> Ident
  165 guardUnderscores i = concat $ map go i
  166   where
  167     go :: Char -> String
  168     go c = case [c] of
  169       "_" -> "'_"
  170       s -> s
  171 
  172 guardBackslash :: Ident -> Ident
  173 guardBackslash i = concat $ map go i
  174   where
  175     go :: Char -> String
  176     go c = case [c] of
  177       "\\" -> "\\\\"
  178       s -> s

  180 -- | Infinite list of name variants.
  181 variants :: Name -> [Name]
  182 variants nm = map (\nm_ -> nm { nmStr = pref ++ nm_ }) (Cry.nameList baseNms)
  183   where
  184     baseNms = case nmStr nm of
  185       "" -> []
  186       "'" -> []
  187       _ -> [nmStr nm]
  188     pref = case nmStr nm of
  189       "'" -> "'"
  190       _ -> ""
  191 
  192 -- | Infinite list of identifier variants
  193 variantIdents :: Ident -> [Ident]
  194 variantIdents nm = (map (\suf -> nm ++ suf) (Cry.nameList []))
  195 
  196 -- | Returns a variant of a name that does not clash with the given list
  197 freshIdent ::
  198   [Ident] {- ^ existing identifiers -} ->
  199   Ident {- ^ prospective name -} ->
  200   Ident
  201 freshIdent exNms tryNm = go (tryNm:variantIdents tryNm)
  202   where
  203     go :: [String] -> String
  204     go (nm:nms) | nm `elem` exNms = go nms
  205     go (nm:_) = nm
  206     go [] = error "freshIdent: impossible empty list"
  207 
  208 instance Output Name where
  209   out t = case t of
  210     Name{} -> case nmSyn t of
  211       InfixSyn s -> symbolOverrides $ guardBackslash $ s
  212       _ -> thyQual (nmThy t) (cleanName t)
  213     SimpleName{} -> cleanName t
