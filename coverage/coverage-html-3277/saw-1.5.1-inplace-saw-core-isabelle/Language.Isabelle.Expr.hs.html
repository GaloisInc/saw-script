   35 data Expr =
   36     Ctr [Expr] Name
   37   | Binder String [Expr] Expr
   38   | BinOp Expr Expr Expr
   39   | App Expr [Expr]
   40   | TApp Expr [Type]
   41   | Var Name
   42   | ConstrainT Expr Type
   43   | IntLitCtr
   44       Integer
   45       Bool -- true if this should be printed as hex
   46   | Undefined Type String
   47   | Let [Expr] Expr
   48   | Case Expr [(Expr,Expr)]
   49   | EmptyExpr
   50   deriving (Eq, Ord, Show)
   51 
   52 pattern IntLit :: Integer -> Expr
   53 pattern IntLit i <- IntLitCtr i _ where
   54   IntLit i = IntLitCtr i False
   55 
   56 traverseSubExprs :: forall m. Monad m => (Expr -> m Expr) -> Expr -> m Expr
   57 traverseSubExprs f = go
   58   where
   59     go :: Expr -> m Expr
   60     go e = f =<< case e of
   61       Ctr es nm -> Ctr <$> mapM go es <*> pure nm
   62       Binder str nms e1 -> Binder <$> pure str <*> pure nms <*> go e1
   63       BinOp e1 e2 e3 -> BinOp <$> go e1 <*> go e2 <*> go e3
   64       App e1 es -> App <$> go e1 <*> mapM go es
   65       TApp e1 ts -> TApp <$> go e1 <*> mapM go ts
   66       Var nm -> Var <$> pure nm
   67       ConstrainT e1 t -> ConstrainT <$> go e1 <*> pure t
   68       IntLitCtr i b -> IntLitCtr <$> pure i <*> pure b
   69       Undefined tp msg -> Undefined <$> go tp <*> pure msg
   70       Let bs e1 -> Let <$> mapM go bs <*> go e1
   71       Case e1 es -> Case <$> go e1 <*> mapM (\(e2,e3) -> (,) <$> go e2 <*> go e3) es
   72       EmptyExpr -> pure EmptyExpr
   73 
   74 liftIdentity :: (forall m. Monad m => (x -> m x) -> y -> m y) -> (x -> x) -> y -> y
   75 liftIdentity f g e = runIdentity (f (\e' -> Identity (g e')) e)
   76 
   77 mapSubExprs :: (Expr -> Expr) -> Expr -> Expr
   78 mapSubExprs = liftIdentity traverseSubExprs
   79 
   80 foldSubExprs :: (Expr -> a -> a) -> Expr -> a -> a
   81 foldSubExprs f e st =
   82   execState (traverseSubExprs (\e' -> modify (\s -> f e' s) >> return e) e) st
   83 
   84 traverseNames :: forall m. Monad m => (Name.Name -> m Name.Name) -> Expr -> m Expr
   85 traverseNames f = traverseSubExprs $ \case
   86   Ctr args nm -> Ctr <$> pure args <*> f nm
   87   Var nm -> Var <$> f nm
   88   e -> pure e
   89 
   90 mapNames :: (Name.Name -> Name.Name) -> Expr -> Expr
   91 mapNames = liftIdentity traverseNames
   92 
   93 
   94 foldNames :: (Name.Name -> a -> a) -> Expr -> a -> a
   95 foldNames f e st =
   96   execState (traverseNames (\nm -> modify (\s -> f nm s) >> return nm) e) st

   98 mkTAppSyn :: Type -> [Type] -> Expr
   99 mkTAppSyn body ts = Ctr (body:ts) $
  100   Name.Name (Name.TheoryName "Cryptol" False)
  101     "TApp" (Syntax.ListSyn Syntax.Prefix "," "`{" "}" ) Name.Term

  103 exprToString :: Output.HasOutput => Expr -> String
  104 exprToString = go
  105   where
  106     braks :: Expr -> String
  107     braks te = case te of
  108       Var{} -> go te
  109       Ctr [] _ -> go te
  110       IntLit{} -> go te
  111       Binder{} -> go te -- adds brackets itself
  112       _ -> Output.brackets $ go te
  113 
  114     braks_postfix :: Expr -> String
  115     braks_postfix te = case te of
  116       Ctr _ nm | Syntax.InfixSyn{} <- Name.nmSyn nm -> braks te
  117       Ctr _ nm | Syntax.ListSyn{} <- Name.nmSyn nm -> braks te
  118       BinOp{} -> braks te
  119       _ -> go te
  120 
  121     abs_arg :: Expr -> String
  122     abs_arg te = case te of
  123       ConstrainT{} -> braks te
  124       _ -> go te
  125 
  126     go :: Expr -> String
  127     go te = case te of
  128       Ctr [] nm |
  129         Syntax.SymbolSyn s <- Name.nmSyn nm -> s
  130       Ctr [] nm |
  131         Syntax.InfixSyn s <- Name.nmSyn nm -> s
  132       Ctr [] nm |
  133         Syntax.NoSyn <- Name.nmSyn nm -> out nm
  134       Ctr es nm -> case (Name.nmSyn nm, es) of
  135         (Syntax.InfixSyn s, [e1,e2]) ->  Output.spaces $
  136           [ braks e1, s, braks e2 ]
  137         (Syntax.Symbol Syntax.Prefix s, _) -> Output.spaces $
  138           (s:(map braks es))
  139         (Syntax.MixFix1 s1 s2, [e]) -> s1 ++ go e ++ s2
  140         (Syntax.MixFix2 s1 s2, [e1, e2]) -> s1 ++ go e1 ++ s2 ++ braks_postfix e2
  141         (Syntax.MixFix3 s1 s2 s3, [e1, e2, e3]) -> Output.spaces $
  142           [ s1, go e1, s2, braks e2, s3, braks e3 ]
  143         (Syntax.ListSyn Syntax.Nofix sep start end, _) ->
  144           start ++ intercalate sep (map go es) ++ end
  145         (Syntax.ListSyn f sep start end, (body:es')) -> case f of
  146           Syntax.Prefix -> braks body ++ start ++ intercalate sep (map go es') ++ end
  147           Syntax.Postfix -> start ++ intercalate sep (map go es') ++ end ++ braks body
  148           _ -> bad
  149         (Syntax.NoSyn, _) -> case Name.isTypeK (Name.nmKind nm) of
  150           True -> (Output.brackets $ intercalate "," (map go es)) ++ " " ++ out nm
  151           False -> out nm ++ " " ++ intercalate " " (map braks es)
  152         _ -> bad
  153       Binder bstr nms e ->
  154         Output.brackets $ bstr
  155           ++ (Output.spaces $ map abs_arg nms)
  156           ++ ". " ++ (Output.indent 1 $ braks e)
  157       BinOp e1 op e2 -> Output.spaces
  158         ([braks e1, go op, braks e2])
  159       App e' args->
  160         go e' ++ " " ++ (Output.spaces (map braks args))
  161       TApp e ts -> go $ mkTAppSyn e ts
  162       Var nm -> out nm
  163       ConstrainT e1 typ -> out e1 ++ " :: " ++ out typ
  164       IntLitCtr s False -> show s
  165       IntLitCtr s True -> "0x" ++ (showHex s "")
  166       Undefined tp msg -> Output.brackets $ "undefined ''" ++ msg ++ "'' :: " ++ out tp
  167       Let bs e1 ->
  168         Output.line ++ "let" ++ (Output.indent 1 $ Output.line ++
  169             Output.lines (Output.addSep ";" (map out bs))) ++ Output.line ++ "in " ++ (Output.indent 1 $ (braks e1))
  170       Case e [(pat,body)] ->
  171         Output.spaces $ ["case", Output.out e, "of", Output.out pat, "\\<Rightarrow>", Output.out body]
  172       Case e es -> Output.lines $
  173         [ Output.spaces ["case", Output.out e, "of"] ]
  174         ++
  175         (Output.indent 1 $ 
  176           Output.addSepPrefix "| " $ (map (\(pat,body) -> 
  177             Output.spaces [ Output.out pat, "\\<Rightarrow>", Output.brackets $ Output.out body ]) es))
  178       EmptyExpr -> ""
  179       where
  180         bad = Panic.panic "Output Expr: Unexpected signature" [show te]
  181 
  182 instance Output Expr where
  183   out = exprToString
