   43 data NameEnv = NameEnv
   44   { nameToIdent :: Map.Map Name.Name Int, identToName :: Map.Map Int Name.Name, minIdent :: Int }

   46 emptyNameEnv :: NameEnv
   47 emptyNameEnv = NameEnv Map.empty Map.empty (-1)

   49 -- | Insert an identifier into the name environment, with a suggested name.
   50 --   Does nothing if the identifier already exists, creates a variant name
   51 --   if the name is already in the environment (with a different identifier).
   52 insertName :: Maybe Int -> Name.Name -> NameEnv -> (Name.Name, NameEnv)
   53 insertName midx nm env = case midx of
   54   Just idx | Just nm' <- Map.lookup idx (identToName env) -> (nm', env)
   55   _ ->
   56     let
   57       used = Set.map Name.nmStr $ Map.keysSet (nameToIdent env) 
   58       nm' = fromJust $ find (\nm_ -> not $ Set.member (Name.nmStr nm_) used ) (Name.variants nm)
   59       idx = case midx of
   60         Just idx' -> idx'
   61         Nothing -> minIdent env - 1
   62     in (nm', NameEnv { nameToIdent = Map.insert nm' idx (nameToIdent env), identToName = Map.insert idx nm' (identToName env), minIdent = min idx (minIdent env) })

   85 newtype IsaM_ a = IsaM { unIsaM :: (RWS.RWST IsaEnv IsaOut IsaState (CME.ExceptT Error.TranslationError IO)) a}
   86   deriving (Applicative, Functor, Monad, IO.MonadIO

   88            , RWS.MonadState IsaState
   89            , RWS.MonadWriter IsaOut)

   91 instance CME.MonadError Error.TranslationError IsaM_ where
   92   catchError f hdl = IsaM (CME.catchError (unIsaM f) (\e -> unIsaM (hdl e)))
   93   throwError e = IsaM $ do
   94     rng <- asks envSourceRange
   95     CME.throwError $ Error.addLocation rng e

   97 instance Alternative IsaM_ where
   98   empty = fail ""
   99   f <|> g = CME.catchError f (\_ -> g)

  101 instance MonadPlus IsaM_

  103 mreturn :: Alternative m => Maybe a -> m a
  104 mreturn (Just a) = pure a
  105 mreturn Nothing = empty

  109 initIsaState :: IsaEnv -> IsaState
  110 initIsaState env = IsaState (Theory.mkTheory $ envCurTheory env)

  112 initIsaEnv :: Options.HasOptions 
  113            => Deps.CryptolDeps
  114            -> Maybe SAWEnv
  115            -> Name.TheoryName
  116            -> IO IsaEnv
  117 initIsaEnv cryEnv sawEnv thyNm = do
  118   let tyenv = Deps.mkTyEnv cryEnv
  119   hashCacheRef <- IO.newMVar Map.empty
  120   return $ IsaEnv thyNm
  121     Position.emptyRange tyenv emptyNameEnv cryEnv 
  122     hashCacheRef Options.allOptions sawEnv

  124 getSAWEnv :: IsaM SAWEnv

  129 execIsaM :: IsaEnv -> IsaM () -> IO (IsaState, IsaOut)
  130 execIsaM env f = 
  131   let st = initIsaState env
  132   in runIsaM env st f >>= \case
  133       Left er -> return (st, IsaOut { outWarns = [er] })
  134       Right ((), st', w) -> return (st', w)
  135 
  136 runIsaM :: IsaEnv -> IsaState -> IsaM a -> IO (Either Error.TranslationError (a, IsaState, IsaOut))
  137 runIsaM env st f = Options.withOptions (envOptions env) $ do
  138   CME.runExceptT $  RWS.runRWST (unIsaM f) env st

  184 withFreshName :: Name.IsaKind -> (Name.Name -> IsaM a) -> IsaM a
  185 withFreshName k f = do
  186   nmEnv <- asks envNameEnv
  187   nm <- simpleName k ""
  188   let (nm', nmEnv') = insertName Nothing nm nmEnv
  189   local (\env -> env { envNameEnv = nmEnv' }) $ f nm'

  192 withNames :: [(Int, Name.Name)] -> IsaM a -> IsaM a
  193 withNames [] f = f
  194 withNames ((idx,nm):nms) f = local (\env -> env { envNameEnv = snd (insertName (Just idx) nm (envNameEnv env)) })
  195   $ withNames nms f

  215 rethrow' :: Error.TranslationError -> IsaM a -> IsaM a
  216 rethrow' er f = CME.catchError (wrapWarns er f) $ \er_inner -> 
  217   CME.throwError (Error.innerErrors er [er_inner])
  218 
  219 wrapWarns :: Error.TranslationError -> IsaM a -> IsaM a
  220 wrapWarns er f = do
  221   rng <- asks envSourceRange
  222   RWS.censor (\w -> IsaOut{outWarns = case outWarns w of
  223       [] -> []
  224       _ -> [Error.addLocation rng $ Error.innerErrors er (outWarns w)]}) f
  225 
  226 rethrow :: (t -> Error.TranslationError) -> (t -> IsaM a) -> t -> IsaM a
  227 rethrow mkerr f t = rethrow' (mkerr t) (f t)
  228 
  229 warn :: Error.TranslationError -> IsaM ()
  230 warn err = do
  231   rng <- asks envSourceRange
  232   RWS.tell (IsaOut $ [Error.addLocation rng err])

  238 shallowHashOf :: Cry.Name -> IsaM Int
  239 shallowHashOf crynm = do
  240   ref <- asks envHashCache
  241   cache <- IO.liftIO $ IO.readMVar ref
  242   case Map.lookup crynm cache of
  243     Just h -> return h
  244     Nothing -> getCryEntity crynm >>= \case
  245       Just (ce,_) -> do
  246         let h = hash ce
  247         IO.liftIO $ IO.modifyMVar_ ref (\cache_ -> return $ Map.insert crynm h cache_)
  248         return h
  249       Nothing -> return $ hash (show $ pp crynm)
  250 
  251 deepHashOf :: Cry.Name -> IsaM Int
  252 deepHashOf crynm = getCryEntity crynm >>= \case
  253   Just (ce,deps) -> do
  254     ce_deps <- mapM shallowHashOf (Set.toList deps)
  255     return $ (hash (ce, ce_deps))
  256   Nothing -> CME.throwError $ Error.MissingCryName crynm
  257 
  258 getCryEntity :: Cry.Name -> IsaM  (Maybe (Deps.CryEntity, Set.Set Cry.Name))
  259 getCryEntity crynm = do
  260   m <- asks envCryDeps
  261   return $ Map.lookup crynm (Deps.entityDeps m)
  262 
  263 deepHashFromEnv :: Deps.CryptolDeps -> Cry.Name -> IO (Maybe Int)
  264 deepHashFromEnv cryEnv cryNm = Options.withOptions Options.emptyOpts $ do
  265   env <- initIsaEnv cryEnv Nothing (Name.TheoryName "<interactive>" False)
  266   runIsaM env (initIsaState env) (deepHashOf cryNm) >>= \case
  267     Right (a,_,_) -> return $ Just a
  268     Left{} -> return Nothing
