   84 ifExpr :: Expr -> Expr -> Expr -> Expr
   85 ifExpr eP eT eF = case (eT, eF) of
   86   (Expr.ConstrainT eT' t, Expr.ConstrainT eF' t') | t == t' -> 
   87     constrain (Expr.Ctr [eP, eT', eF'] $
   88       Name.Name holThy "If" (Syntax.MixFix3 "if" "then" "else") Name.Term) t
   89   _ -> Expr.Ctr [eP, eT, eF] $
   90     Name.Name holThy "If" (Syntax.MixFix3 "if" "then" "else coerce") Name.Term

   92 letExpr' :: [Expr] -> Expr -> Expr
   93 letExpr' binds (Expr.Let binds' ebody) = letExpr' (binds ++ binds') ebody
   94 letExpr' binds ebody = Expr.Let binds ebody
   95 
   96 letExpr :: [(Binding.Binding, Expr)] -> Expr -> Expr
   97 letExpr binds eBody = letExpr' (map bindEq binds) eBody
   98   where
   99     bindEq (b,e) = eqExpr (Expr.Var (Binding.bindName b)) (coerceToExpr e (Binding.bindType b))

  216 tupleSelect :: Name.Name -> Expr -> Int -> Int -> Expr
  217 tupleSelect nm e sel len = case splitAt sel (replicate len dummy) of
  218   (hdr,_:ftr) ->
  219    let binds = hdr ++ (Expr.Var nm:ftr)
  220     in appExpr (eAbs [(tupleExpr binds)] (Expr.Var nm)) [e]
  221   _ -> Panic.panic "Unexpected tuple selection" [show e, show sel, show len]

  223 tupleAbs :: [[(Name.Name,Type)]] -> Expr -> Expr
  224 tupleAbs bs e = eAbs [tupleExpr (map go bs)] e
  225   where
  226     go :: [(Name.Name, Type)] -> Expr
  227     go binds = tupleExpr (map (\(nm,t) -> Expr.ConstrainT (Expr.Var nm) t) binds)

  307 seqZip :: Type -> Type -> Type -> Type -> Expr -> Expr -> Expr
  308 seqZip lenT1 lenT2 t1 t2 e1 e2 = snd (seqZip' lenT1 lenT2 t1 t2 e1 e2)
  309 
  310 seqZip' :: Type -> Type -> Type -> Type -> Expr -> Expr -> ((Type, Type), Expr)
  311 seqZip' n a m b e1 e2 =
  312   let f = if n == m then
  313         tApp (Expr.Ctr [] $ (Name.Name seqThy "zip_seq" (Syntax.Symbol Syntax.Prefix "zip") Name.Term)) [n, a, b]
  314       else
  315         tApp (Expr.Ctr [] $ (Name.Name seqThy "zip_seq_mismatched" (Syntax.Symbol Syntax.Prefix "zip_mismatch")  Name.Term)) [n, a, m, b]
  316   in (((tMin n m),(tTuple [a, b])), appExpr f [e1, e2])

  318 seqZipMany :: [((Type,Type), Expr)] -> ((Type, Type), Expr)
  319 seqZipMany [] = Panic.panic "seqZipMany: empty" []
  320 seqZipMany [x] = x
  321 seqZipMany (((lenT,eT),e):tes) = let ((lenT',eT'),e') = seqZipMany tes in seqZip' lenT eT lenT' eT' e e'

  363 mapRecord :: String -> [String] -> ((Name.Name, Expr.Expr) -> Expr.Expr) -> Expr.Expr
  364 mapRecord recNm nms f =
  365   eAbs [x] $ Expr.Case x [(recordCtr nmargs, recordCtr (map (\(nm,arg) -> (nm, f (nm,arg))) nmargs))]
  366   where
  367     x = Expr.Var $ Name.SimpleName "x"
  368     nmargs = zip (map Name.SimpleName nms ++ [dots]) args
  369     args = map (\n -> Expr.Var (Name.SimpleName (n ++ "__" ++ recNm))) (Cry.nameList [])
