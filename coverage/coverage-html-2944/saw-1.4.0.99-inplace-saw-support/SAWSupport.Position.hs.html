   12 module SAWSupport.Position (

   15   where
   16 
   17 import Data.Text (Text)
   18 
   19 import qualified Prettyprinter as PP
   20 
   21 import qualified SAWSupport.Pretty as PPS
   22 
   23 class IsPosition t where
   24     ppPosition :: t -> Text
   25     prettyPosition :: t -> PPS.Doc
   26 
   27     -- Either function is sufficient to define the other.
   28     --
   29     -- There's an implicit assumption here that the printing options
   30     -- don't affect the way positions print. If that changes, the
   31     -- interface should change accordingly.
   32     ppPosition pos = PPS.renderText PPS.defaultOpts $ prettyPosition pos
   33     prettyPosition pos = PP.pretty $ ppPosition pos
