   59 -- | Insert multiple keys that map to the same value in a map.
   60 mapInsertKeys :: Ord k => [k] -> a -> Map k a -> Map k a
   61 mapInsertKeys keys val m = foldr (\i -> Map.insert i val) m keys

   64 -- Nothing if none of the keys are in the map.
   65 mapLookupAny :: Ord k => [k] -> Map k a -> Maybe a
   66 mapLookupAny keys m = listToMaybe $ catMaybes $ map (\k -> Map.lookup k m) keys

   88 -- | Return a string representation of the elapsed time since start
   89 timeIt :: (NFData a, MonadIO m) => m a -> m (a, String)
   90 timeIt action = do
   91         start <- liftIO $ getClockTime
   92         r <- action
   93         end <- rnf r `seq` liftIO getClockTime
   94         let itd = diffClockTimes end start
   95             td = normalizeTimeDiff itd
   96             vals = dropWhile (\(v, _) -> v == 0) (zip [tdYear td, tdMonth td, tdDay td, tdHour td, tdMin td] "YMDhm")
   97             sec = ' ' : show (tdSec td) ++ dropWhile (/= '.') pico
   98             pico = showFFloat (Just 3) (((10**(-12))::Double) * fromIntegral (tdPicosec td)) "s"
   99         return $ (r, dropWhile isSpace $ concatMap (\(v, c) -> ' ':show v ++ [c]) vals ++ sec)

  114 -- | Atempt to find class with given name, or throw ExecException if no class
  115 -- with that name exists. Class name should be in slash-separated form.
  116 lookupClass :: JSS.Codebase -> Pos -> JSS.ClassName -> IO JSS.Class
  117 lookupClass cb site nm = do
  118   maybeCl <- JSS.tryLookupClass cb nm
  119   case maybeCl of
  120     Nothing -> do
  121      let msg = ftext ("The Java class " ++ JSS.slashesToDots (JSS.unClassName nm) ++ " could not be found.")
  122          res = unwords [ "Please check that the path to Java is set correctly"
  123                        , "(either through the --java-bin-dirs option or your PATH)"
  124                        , "and you are using Java 8 or earlier"
  125                        , "(SAW does not support 9+ currently)."
  126                        ]
  127       in throwIOExecException site msg res
  128     Just cl -> return cl
  129 
  130 -- | Returns method with given name in this class or one of its subclasses.
  131 -- Throws an ExecException if method could not be found or is ambiguous.
  132 findMethod :: JSS.Codebase -> Pos -> String -> JSS.Class -> IO (JSS.Class, JSS.Method)
  133 findMethod cb site nm initClass = impl [] initClass
  134   where javaClassName = JSS.slashesToDots (JSS.unClassName (JSS.className initClass))
  135         methodType m = (JSS.methodParameterTypes m, JSS.methodReturnType m)
  136         baseName m = JSS.methodName m
  137         typedName m = JSS.methodName m ++ ":" ++ unparseMethodDescriptor (methodType m)
  138         methodMatches m = nm `elem` [baseName m, typedName m]
  139         impl names cl =
  140           let candidates = filter (not . JSS.methodIsAbstract) (JSS.classMethods cl) in
  141           case filter methodMatches candidates of
  142             [] -> do
  143               case JSS.superClass cl of
  144                 Nothing ->
  145                   let msg = ftext $ "Could not find method " ++ nm
  146                               ++ " in class " ++ javaClassName ++ ".\n"
  147                               ++ "Available methods: "
  148                               ++ unlines names
  149                       res = "Please check that the class and method are correct."
  150                    in throwIOExecException site msg res
  151                 Just superName ->
  152                   do super <- lookupClass cb site superName
  153                      impl (names ++ map baseName candidates) super
  154             [method] -> return (cl,method)
  155             l -> let msg = "The method " ++ show nm ++ " in class " ++ javaClassName ++ " is ambiguous. " ++
  156                            "Methods can be disambiguated by appending a type descriptor: " ++
  157                            unlines [ show (typedName m) | m <- l ]
  158                      res = "Please disambiguate method name."
  159                   in throwIOExecException site (ftext msg) res

  161 throwFieldNotFound :: JSS.Type -> String -> [String] -> ExceptT String IO a
  162 throwFieldNotFound tp fieldName names = throwE msg
  163   where
  164     msg = "Values with type \'" ++ show tp ++
  165           "\' do not contain field named " ++
  166           fieldName ++ "."
  167           ++ "\nAvailable fields:\n" ++ unlines names
  168 
  169 findField :: JSS.Codebase -> Pos -> JSS.Type -> String -> ExceptT String IO JSS.FieldId
  170 findField _  _ tp@(JSS.ArrayType _) nm = throwFieldNotFound tp nm []
  171 findField cb site tp@(JSS.ClassType clName) nm = impl [] =<< lift (lookupClass cb site clName)
  172   where
  173     impl nms cl =
  174       case filter (\f -> nm `elem` names f) $ JSS.classFields cl of
  175         [] -> do
  176           case JSS.superClass cl of
  177             Nothing -> throwFieldNotFound tp nm nms
  178             Just superName ->
  179               do super <- lift $ lookupClass cb site superName
  180                  impl (nms ++ map JSS.fieldName (JSS.classFields cl)) super
  181         [f] -> return $ JSS.FieldId (JSS.className cl) (JSS.fieldName f) (JSS.fieldType f)
  182         _ -> throwE $
  183              "internal: Found multiple fields with the same name: " ++ nm
  184       where
  185         names f =
  186           do prefix <- ["", JSS.unClassName (JSS.className cl) ++ "."]
  187              suffix <- ["", ":" ++ unparseTypeDescriptor (JSS.fieldType f)]
  188              pure (prefix ++ JSS.fieldName f ++ suffix)
  189 
  190 findField _ _ _ _ =
  191   throwE "Primitive types cannot be dereferenced."

  194 -- | Convert a non-negative integer to to an ordinal string.
  195 --
  196 -- Note: @0 -> "0th"@, so do @'ordinal' (n + 1)@ if you want one-based
  197 -- results.
  198 ordinal :: (Integral a, Show a) => a -> String
  199 -- Not sure what to do with negative integers so bail.
  200 ordinal n | n < 0 = error "Only non-negative cardinals are supported."
  201           | otherwise = show n ++ suffix
  202   where
  203     suffix =
  204       if inTens then "th"
  205       else case n `mod` 10 of
  206              1 -> "st"
  207              2 -> "nd"
  208              3 -> "rd"
  209              _ -> "th"
  210     inTens = (n `mod` 100) `div` 10 == 1

  212 handleException :: Options -> CE.SomeException -> IO a
  213 handleException opts e
  214     | Just (_ :: ExitCode) <- CE.fromException e = CE.throw e
  215     | Just ioe <- CE.fromException e =
  216          printOutLn opts Error (displayIOE ioe) >> exitProofUnknown
  217     | otherwise =
  218          printOutLn opts Error (CE.displayException e) >> exitProofUnknown
  219 
  220  where
  221  displayIOE ioe
  222    | IOE.isUserError ioe = IOE.ioeGetErrorString ioe
  223    | otherwise = CE.displayException ioe
  224 
  225 exitProofFalse,exitProofUnknown,exitProofSuccess :: IO a
  226 exitProofFalse = exitWith (ExitFailure 1)
  227 exitProofUnknown = exitWith (ExitFailure 2)
  228 exitProofSuccess = exitSuccess

  232 unparseTypeDescriptor :: JSS.Type -> String
  233 unparseTypeDescriptor =
  234   \case
  235     JSS.ArrayType ty -> "[" ++ unparseTypeDescriptor ty
  236     JSS.BooleanType  -> "Z"
  237     JSS.ByteType     -> "B"
  238     JSS.CharType     -> "C"
  239     JSS.ClassType cn -> "L" ++ JSS.unClassName cn ++ ";"
  240     JSS.DoubleType   -> "D"
  241     JSS.FloatType    -> "F"
  242     JSS.IntType      -> "I"
  243     JSS.LongType     -> "J"
  244     JSS.ShortType    -> "S"
  245 
  246 unparseMethodDescriptor :: ([JSS.Type], Maybe JSS.Type) -> String
  247 unparseMethodDescriptor (args, ret) =
  248   "(" ++ concatMap unparseTypeDescriptor args ++ ")" ++
  249   maybe "V" unparseTypeDescriptor ret

  252 neGroupOn ::
  253   Ord b =>
  254   (a -> b) {- ^ equivalence class projection -} ->
  255   [a] -> [NE.NonEmpty a]
  256 neGroupOn f = NE.groupBy ((==) `on` f) . sortBy (compare `on` f)

  258 neNubOrd ::
  259   Ord a =>
  260   NE.NonEmpty a ->
  261   NE.NonEmpty a
  262 neNubOrd (hd NE.:| tl) = hd NE.:| loop (Set.singleton hd) tl
  263   where
  264     loop _prev [] = []
  265     loop prev (x:xs)
  266       | Set.member x prev = loop prev xs
  267       | otherwise         = x : loop (Set.insert x prev) xs
