   56 -- | Infer the type of an untyped term and complete it to a 'Term'.
   57 inferCompleteTerm ::
   58   SharedContext -> Maybe ModuleName -> Un.UTerm -> IO (Either PPS.Doc Term)
   59 inferCompleteTerm sc mnm t =
   60   do res <- runTCM (typeInferCompleteUTerm t) sc mnm
   61      case res of
   62        Right t' -> pure (Right t')
   63        Left err ->
   64          do ne <- scGetNamingEnv sc
   65             let opts = PPS.defaultOpts
   66             pure (Left (prettyTCError opts ne err))
   67 
   68 -- | Pretty-print a type-checking error
   69 ppTCError :: TCError -> String
   70 ppTCError e =
   71   PPS.render PPS.defaultOpts $
   72   prettyTCError PPS.defaultOpts emptyDisplayNameEnv e
   73 
   74 -- | Pretty-print a type-checking error
   75 prettyTCError :: PPS.Opts -> DisplayNameEnv -> TCError -> PPS.Doc
   76 prettyTCError opts ne e = helper Nothing e where
   77 
   78   ppWithPos :: Maybe Pos -> [PPS.Doc] -> PPS.Doc
   79   ppWithPos maybe_p strs =
   80     case maybe_p of
   81       Just p -> vcat (pretty (ppPos p) : strs)
   82       Nothing -> vcat strs
   83 
   84   helper :: Maybe Pos -> TCError -> PPS.Doc
   85   helper mp err =
   86     case err of
   87       UnboundName str ->
   88         ppWithPos mp [ "Unbound name:" <+> pretty str ]
   89       EmptyVectorLit ->
   90         ppWithPos mp [ "Empty vector literal"]
   91       NoSuchDataType d ->
   92         ppWithPos mp [ "No such data type:" <+> pretty (show d) ]
   93       DeclError nm reason ->
   94         ppWithPos mp [ "Malformed declaration for" <+> pretty (show nm), pretty reason ]
   95       ErrorPos p err' ->
   96         helper (Just p) err'
   97       ErrorUTerm t err' ->
   98         ppWithPos mp [ "While typechecking term:"
   99                      , indent 2 $ Un.prettyUTerm t
  100                      , helper mp err'
  101                      ]
  102       TermError err' ->
  103         ppWithPos mp [ prettyTermError opts ne err' ]

  108 -- | The 'ReaderT' environment for a computation to typecheck a
  109 -- SAWCore parser AST.
  110 data TCEnv =
  111   TCEnv
  112   { tcSharedContext :: SharedContext -- ^ the SAW context
  113   , tcModName :: Maybe ModuleName -- ^ the current module name
  114   , tcLocals :: Map LocalName (VarName, Term) -- ^ the mapping of display names to variables
  115   }
  116 
  117 -- | Errors that can occur during type-checking
  118 data TCError
  119   = UnboundName Text
  120   | EmptyVectorLit
  121   | NoSuchDataType NameInfo
  122   | DeclError Text String
  123   | ErrorPos Pos TCError
  124   | ErrorUTerm Un.UTerm TCError
  125   | TermError SC.TermError
  126 
  127 -- | The monad for computations to typecheck a SAWCore parser AST.
  128 newtype TCM a = TCM (ReaderT TCEnv (ExceptT TCError IO) a)
  129   deriving (Functor, Applicative, Monad, MonadIO)
  130 
  131 runTCM ::
  132   TCM a -> SharedContext -> Maybe ModuleName -> IO (Either TCError a)
  133 runTCM (TCM m) sc mnm =
  134   runExceptT (runReaderT m (TCEnv sc mnm Map.empty))
  135 
  136 -- | Read the current module name
  137 askSharedContext :: TCM SharedContext
  138 askSharedContext = TCM (asks tcSharedContext)
  139 
  140 -- | Read the current module name
  141 askModName :: TCM (Maybe ModuleName)
  142 askModName = TCM (asks tcModName)
  143 
  144 -- | Read the current context of named variables
  145 askLocals :: TCM (Map LocalName (VarName, Term))
  146 askLocals = TCM (asks tcLocals)
  147 
  148 -- | Look up the current module name, raising an error if it is not set
  149 getModuleName :: TCM ModuleName
  150 getModuleName =
  151   do maybe_modname <- askModName
  152      case maybe_modname of
  153        Just mnm -> return mnm
  154        Nothing ->
  155          panic "getModuleName" ["Current module name not set during typechecking"]
  156 
  157 -- | Throw a type-checking error.
  158 throwTCError :: TCError -> TCM a
  159 throwTCError e = TCM (throwError e)
  160 
  161 -- | Augment and rethrow any 'TCError' thrown by the given computation.
  162 rethrowTCError :: (TCError -> TCError) -> TCM a -> TCM a
  163 rethrowTCError f (TCM m) = TCM (catchError m (throwError . f))
  164 
  165 -- | Run a type-checking computation @m@ and tag any error it throws with the
  166 -- 'ErrorUTerm' constructor.
  167 withErrorUTerm :: Un.UTerm -> TCM a -> TCM a
  168 withErrorUTerm t = rethrowTCError (ErrorUTerm t)
  169 
  170 -- | Run a type-checking computation @m@ and tag any error it throws with the
  171 -- given position, using the 'ErrorPos' constructor, unless that error is
  172 -- already tagged with a position.
  173 atPos :: Pos -> TCM a -> TCM a
  174 atPos p = rethrowTCError (ErrorPos p)
  175 
  176 -- | Lift a computation from the SAWCore term-building monad 'SC.SCM'
  177 -- into 'TCM'.
  178 liftSCM :: SC.SCM a -> TCM a
  179 liftSCM m =
  180   do sc <- askSharedContext
  181      result <- liftIO $ SC.runSCM sc m
  182      case result of
  183        Left e -> throwTCError (TermError e)
  184        Right x -> pure x

  188 -- | Resolve a name.
  189 inferResolveName :: Text -> TCM Term
  190 inferResolveName n =
  191   do nctx <- askLocals
  192      mnm <- getModuleName
  193      sc <- askSharedContext
  194      mm <- liftIO $ scGetModuleMap sc
  195      let ident = mkIdent mnm n
  196      case (Map.lookup n nctx, resolveNameInMap mm ident) of
  197        (Just (vn, tp), _) ->
  198          liftSCM $ SC.scmVariable vn tp
  199        (_, Just rn) ->
  200          do let c = resolvedNameName rn
  201             liftSCM $ SC.scmConst c
  202        (Nothing, Nothing) ->
  203          throwTCError $ UnboundName n
  204 
  205 -- | The debugging level
  206 debugLevel :: Int
  207 debugLevel = 0

  209 -- | Print debugging output if 'debugLevel' is greater than 0
  210 typeInferDebug :: String -> TCM ()
  211 typeInferDebug str | debugLevel > 0 = liftIO $ traceIO str
  212 typeInferDebug _ = return ()

  214 -- | Completely typecheck an untyped SAWCore AST.
  215 typeInferCompleteUTerm :: Un.UTerm -> TCM Term
  216 typeInferCompleteUTerm t =
  217   do typeInferDebug ("typechecking term: " ++ show t)
  218      res <- atPos (pos t) $ withErrorUTerm t $ typeInferCompleteTerm t
  219      ty <- liftSCM $ SC.scmTypeOf res
  220      typeInferDebug ("completed typechecking term: " ++ show t ++ "\n"
  221                      ++ "type = " ++ show ty)
  222      return res
  223 
  224 -- | Main workhorse function for type inference on untyped terms
  225 typeInferCompleteTerm :: Un.UTerm -> TCM Term
  226 typeInferCompleteTerm uterm =
  227   case uterm of
  228     Un.Name (PosPair _ n) ->
  229       inferResolveName n
  230 
  231     Un.Sort _ srt h ->
  232       liftSCM $ SC.scmSortWithFlags srt h
  233 
  234     Un.Recursor (PosPair _ str) s ->
  235       do mnm <- getModuleName
  236          sc <- askSharedContext
  237          mm <- liftIO $ scGetModuleMap sc
  238          let dt_ident = mkIdent mnm str
  239          dt <- case findDataTypeInMap dt_ident mm of
  240            Just d -> return d
  241            Nothing -> throwTCError $ NoSuchDataType (ModuleIdentifier dt_ident)
  242          liftSCM $ SC.scmRecursor (dtName dt) s
  243 
  244     Un.App f arg ->
  245       -- Only call typeInferCompleteUTerm on the function arguments, to
  246       -- avoid adding too many 'atPos' and 'withErrorUTerm' annotations.
  247       do f' <- typeInferCompleteTerm f
  248          arg' <- typeInferCompleteUTerm arg
  249          liftSCM $ SC.scmApply f' arg'
  250 
  251     Un.Lambda _ [] t ->
  252       typeInferCompleteUTerm t
  253     Un.Lambda p ((Un.termVarLocalName -> x, tp) : ctx) t ->
  254       do tp_trm <- typeInferCompleteUTerm tp
  255          _ <- liftSCM $ SC.scmEnsureSortType tp_trm
  256          vn <- liftSCM $ SC.scmFreshVarName x
  257          -- Don't call typeInferCompleteUTerm with a suffix of the
  258          -- original variable context, to avoid adding too many
  259          -- 'withErrorUTerm' annotations.
  260          body <- withVar x vn tp_trm $
  261            typeInferCompleteTerm $ Un.Lambda p ctx t
  262          liftSCM $ SC.scmLambda vn tp_trm body
  263 
  264     Un.Pi _ [] t ->
  265       typeInferCompleteUTerm t
  266     Un.Pi p ((Un.termVarLocalName -> x, tp) : ctx) t ->
  267       do tp' <- typeInferCompleteUTerm tp
  268          vn <- liftSCM $ SC.scmFreshVarName x
  269          -- Don't call typeInferCompleteUTerm with a suffix of the
  270          -- original variable context, to avoid adding too many
  271          -- 'withErrorUTerm' annotations.
  272          body <- withVar x vn tp' $
  273            typeInferCompleteTerm $ Un.Pi p ctx t
  274          liftSCM $ SC.scmPi vn tp' body
  275 
  276     Un.RecordValue _ elems ->
  277       do typed_elems <-
  278            mapM (\(PosPair _ fld, t) -> (fld,) <$> typeInferCompleteUTerm t) elems
  279          liftSCM $ SC.scmRecordValue typed_elems
  280     Un.RecordType _ elems ->
  281       do typed_elems <-
  282            mapM (\(PosPair _ fld, t) -> (fld,) <$> typeInferCompleteUTerm t) elems
  283          liftSCM $ SC.scmRecordType typed_elems
  284     Un.RecordProj t prj ->
  285       do t' <- typeInferCompleteUTerm t
  286          liftSCM $ SC.scmRecordSelect t' prj
  287 
  288     Un.UnitValue _ ->
  289       liftSCM $ SC.scmUnitValue
  290     Un.UnitType _ ->
  291       liftSCM $ SC.scmUnitType
  292 
  293     Un.PairValue _ t1 t2 ->
  294       do t1' <- typeInferCompleteUTerm t1
  295          t2' <- typeInferCompleteUTerm t2
  296          liftSCM $ SC.scmPairValue t1' t2'
  297     Un.PairType _ t1 t2 ->
  298       do t1' <- typeInferCompleteUTerm t1
  299          t2' <- typeInferCompleteUTerm t2
  300          liftSCM $ SC.scmPairType t1' t2'
  301     Un.PairLeft t ->
  302       do t' <- typeInferCompleteUTerm t
  303          liftSCM $ SC.scmPairLeft t'
  304     Un.PairRight t ->
  305       do t' <- typeInferCompleteUTerm t
  306          liftSCM $ SC.scmPairRight t'
  307 
  308     Un.TypeConstraint t _ tp ->
  309       do t' <- typeInferCompleteUTerm t
  310          tp' <- typeInferCompleteUTerm tp
  311          liftSCM $ SC.scmAscribe t' tp'
  312 
  313     Un.NatLit _ i ->
  314       liftSCM $ SC.scmNat i
  315     Un.StringLit _ str ->
  316       liftSCM $ SC.scmString str
  317     Un.VecLit _ ts ->
  318       do typed_ts <- mapM typeInferCompleteUTerm ts
  319          typed_tp <-
  320            case typed_ts of
  321              (t1:_) -> liftSCM $ SC.scmTypeOf t1
  322              [] -> throwTCError $ EmptyVectorLit
  323          liftSCM $ SC.scmVector typed_tp typed_ts
  324     Un.BVLit _ bits ->
  325       do tp <- liftSCM $ SC.scmGlobalDef "Prelude.Bool"
  326          let bit b = SC.scmGlobalDef (if b then "Prelude.True" else "Prelude.False")
  327          bit_tms <- liftSCM $ traverse bit bits
  328          liftSCM $ SC.scmVector tp bit_tms
  329 
  330     Un.BadTerm _ ->
  331       -- Should be unreachable, since BadTerms represent parse errors, that should
  332       -- already have been signaled before type inference
  333       panic "typeInferCompleteTerm" ["Type inference encountered a BadTerm"]

  340 -- | Type-check a list of declarations and insert them into the current module
  341 processDecls :: [Un.Decl] -> TCM ()
  342 processDecls [] = return ()
  343 
  344 processDecls (Un.InjectCodeDecl ns txt : rest) =
  345   do mnm <- getModuleName
  346      sc <- askSharedContext
  347      liftIO $ scInjectCode sc mnm ns txt
  348      processDecls rest
  349 
  350 processDecls (Un.TypedDef nm params rty body : rest) =
  351   processDecls (Un.TypeDecl NoQualifier nm (Un.Pi (pos nm) params rty) :
  352                 Un.TermDef nm (map fst params) body : rest)
  353 
  354 processDecls (Un.TypeDecl NoQualifier (PosPair p nm) tp :
  355               Un.TermDef (PosPair _ ((== nm) -> True)) vars body : rest) =
  356   -- Type-checking for definitions
  357   (atPos p $
  358    do
  359      sc <- askSharedContext
  360      -- Step 1: type-check the type annotation, and make sure it is a type
  361      typed_tp <- typeInferCompleteUTerm tp
  362      void $ liftSCM $ SC.scmEnsureSortType typed_tp
  363      let def_tp = typed_tp
  364      def_tp_whnf <- liftIO $ scWhnf sc def_tp
  365 
  366      -- Step 2: assign types to the bound variables of the definition, by
  367      -- peeling off the pi-abstraction variables in the type annotation. Any
  368      -- remaining body of the pi-type is the required type for the def body.
  369      matchResult <-
  370        runMaybeT $ matchPiWithNames (map Un.termVarLocalName vars) def_tp_whnf
  371      (ctx, req_body_tp) <-
  372        case matchResult of
  373          Just x -> return x
  374          Nothing ->
  375              throwTCError $
  376              DeclError nm ("More variables " ++ show (map Un.termVarLocalName vars) ++
  377                            " than length of function type:\n" ++
  378                            ppTermPureDefaults typed_tp)
  379 
  380      -- Step 3: type-check the body of the definition in the context of its
  381      -- variables, and build a function that takes in those variables
  382      def_tm <-
  383        withCtx ctx $
  384        do body' <- typeInferCompleteUTerm body
  385           body_ty <- liftSCM $ SC.scmTypeOf body'
  386           ok <- liftIO $ scSubtype sc body_ty req_body_tp
  387           unless ok $ throwTCError (TermError (SC.ApplyNotSubtype req_body_tp body')) -- FIXME: Wrong error
  388           let vts = map (\(_, v, t) -> (v, t)) ctx
  389           result <- liftIO $ scLambdaList sc vts body'
  390           liftSCM $ SC.scmAscribe result def_tp
  391 
  392      -- Step 4: add the definition to the current module
  393      mnm <- getModuleName
  394      let nmi = ModuleIdentifier (mkIdent mnm nm)
  395      void $ liftIO $ scDefineConstant sc nmi def_tm) >>
  396   processDecls rest
  397 
  398 processDecls (Un.TypeDecl NoQualifier (PosPair p nm) _ : _) =
  399   atPos p $ throwTCError $ DeclError nm "Definition without defining equation"
  400 
  401 processDecls (Un.TypeDecl _ (PosPair p nm) _ :
  402               Un.TermDef (PosPair _ ((== nm) -> True)) _ _ : _) =
  403   atPos p $ throwTCError $ DeclError nm "Primitive or axiom with definition"
  404 
  405 processDecls (Un.TypeDecl q (PosPair p nm) tp : rest) =
  406   (atPos p $
  407    do typed_tp <- typeInferCompleteUTerm tp
  408       void $ liftSCM $ SC.scmEnsureSortType typed_tp
  409       mnm <- getModuleName
  410       let ident = mkIdent mnm nm
  411       let def_tp = typed_tp
  412       sc <- askSharedContext
  413       liftIO $ scDeclarePrim sc ident q def_tp) >>
  414   processDecls rest
  415 
  416 processDecls (Un.TermDef (PosPair p nm) _ _ : _) =
  417   atPos p $ throwTCError $ DeclError nm "Dangling definition without a type"
  418 
  419 processDecls (Un.DataDecl nm param_ctx dt_tp c_decls : rest) =
  420   do processDataDecl nm param_ctx dt_tp c_decls
  421      processDecls rest
  422 
  423 -- | Type-check a data type declaration and insert it into the current module.
  424 processDataDecl :: PosPair Text -> Un.UTermCtx -> Un.UTerm -> [Un.CtorDecl] -> TCM ()
  425 processDataDecl (PosPair p nm) param_ctx dt_tp c_decls =
  426   let param_ctx' = map (\(x, t) -> (Un.termVarLocalName x, t)) param_ctx in
  427   atPos p $
  428   -- Step 1: type-check the parameters
  429   typeInferCompleteInCtx param_ctx' $ \params -> do
  430   let dtParams = map (\(_, v, t, _) -> (v, t)) params
  431   let err :: String -> TCM a
  432       err msg = throwTCError $ DeclError nm msg
  433 
  434   -- Step 2: type-check the type given for d, and make sure it is of the form
  435   -- (i1:ix1) -> ... -> (in:ixn) -> Type s for some sort s. Then form the full
  436   -- type of d as (p1:param1) -> ... -> (i1:ix1) -> ... -> Type s
  437   (dt_ixs, dtSort) <-
  438     case Un.asPiList dt_tp of
  439       (ixs, Un.Sort _ s h) | h == noFlags ->
  440         return (ixs, s) -- NB, don't allow `isort`, etc.
  441       _ -> err "Wrong form for type of datatype"
  442   let dt_ixs' = map (\(x, t) -> (Un.termVarLocalName x, t)) dt_ixs
  443   dt_ixs_typed <- typeInferCompleteCtx dt_ixs'
  444   let dtIndices = map (\(_, v, t, _) -> (v, t)) dt_ixs_typed
  445 
  446   -- Step 3: Declare d as a free variable, so we can typecheck the constructors
  447   dtType <- liftSCM $ SC.scmPiList (dtParams ++ dtIndices) =<< SC.scmSort dtSort
  448   dtVarName <- liftSCM $ SC.scmFreshVarName nm
  449 
  450   -- Step 4: typecheck the constructors, and build Ctors for them
  451   typed_ctors <-
  452     withVar nm dtVarName dtType $
  453     mapM (\(Un.Ctor (PosPair p' c) ctx body) ->
  454            (c,) <$> typeInferCompleteUTerm (Un.Pi p' ctx body)) c_decls
  455   mnm <- getModuleName
  456   ctors <-
  457     withVar nm dtVarName dtType $
  458     forM typed_ctors $ \(c, tp) ->
  459     do let nmi = ModuleIdentifier (mkIdent mnm c)
  460        let result = mkCtorSpec nmi dtVarName dtParams dtIndices tp
  461        case result of
  462          Just spec -> pure spec
  463          Nothing -> err ("Malformed type form constructor: " ++ show c)
  464 
  465   -- Step 5: Declare the datatype with the given ctors
  466   motiveName <- liftSCM $ SC.scmFreshVarName "p"
  467   argName <- liftSCM $ SC.scmFreshVarName "arg"
  468   let dts =
  469         SC.DataTypeSpec
  470         { SC.dtsNameInfo = ModuleIdentifier (mkIdent mnm nm)
  471         , SC.dtsParams = dtParams
  472         , SC.dtsIndices = dtIndices
  473         , SC.dtsSort = dtSort
  474         , SC.dtsCtors = ctors
  475         , SC.dtsMotiveName = motiveName
  476         , SC.dtsArgName = argName
  477         }
  478   void $ liftSCM $ SC.scmDefineDataType dts
  479

  481 -- | Typecheck a module and, on success, insert it into the current context
  482 tcInsertModule :: SharedContext -> Un.Module -> IO ()
  483 tcInsertModule sc (Un.Module (PosPair _ mnm) imports decls) = do
  484   -- First, insert an empty module for mnm
  485   scLoadModule sc $ emptyModule mnm
  486   -- Next, process all the imports
  487   forM_ imports $ \imp ->
  488     do let imn = val $ Un.importModName imp
  489        i_exists <- scModuleIsLoaded sc imn
  490        unless i_exists $ fail $ "Imported module not found: " ++ show imn
  491        scImportModule sc (Un.nameSatsConstraint (Un.importConstraints imp) . Text.unpack) imn mnm
  492   -- Finally, process all the decls
  493   decls_res <- runTCM (processDecls decls) sc (Just mnm)
  494   case decls_res of
  495     Left err -> fail $ ppTCError err
  496     Right _ -> return ()

  503 -- | Pattern match a nested pi-abstraction, like 'asPiList', but only match as
  504 -- far as the supplied list of variables, and use them as the new names
  505 matchPiWithNames ::
  506   [LocalName] -> Term -> MaybeT TCM ([(LocalName, VarName, Term)], Term)
  507 matchPiWithNames [] tp = pure ([], tp)
  508 matchPiWithNames (var : vars) tp =
  509   case asPi tp of
  510     Nothing -> mzero
  511     Just (x, arg_tp, body_tp) ->
  512       do vn <-
  513            if IntSet.member (vnIndex x) (freeVars body_tp)
  514            then -- dependent function type: use name from Pi type
  515              pure x
  516            else -- non-dependent function: use 'var' as base name
  517              lift $ liftSCM $ SC.scmFreshVarName var
  518          (ctx, body) <- matchPiWithNames vars body_tp
  519          pure ((var, vn, arg_tp) : ctx, body)

  521 -- | Run a type-checking computation in a typing context extended with a new
  522 -- variable with the given name and type.
  523 withVar :: LocalName -> VarName -> Term -> TCM a -> TCM a
  524 withVar x vn tp (TCM m) =
  525   TCM $
  526   local (\env -> env { tcLocals = Map.insert x (vn, tp) (tcLocals env) }) m
  527 
  528 -- | Run a type-checking computation in a typing context extended by a list of
  529 -- variables and their types. See 'withVar'.
  530 withCtx :: [(LocalName, VarName, Term)] -> TCM a -> TCM a
  531 withCtx = flip (foldr (\(x, vn, tp) -> withVar x vn tp))
  532 
  533 -- | Perform type inference on a context, i.e., a list of variable names and
  534 -- their associated types. This will give us 'Term's for each type, as
  535 -- well as their 'Sort's, since the type of any type is a 'Sort'.
  536 typeInferCompleteCtx ::
  537   [(LocalName, Un.UTerm)] -> TCM [(LocalName, VarName, Term, Sort)]
  538 typeInferCompleteCtx [] = pure []
  539 typeInferCompleteCtx ((x, tp) : ctx) =
  540   do tp' <- typeInferCompleteUTerm tp
  541      s <- liftSCM $ SC.scmEnsureSortType tp'
  542      vn <- liftSCM $ SC.scmFreshVarName x
  543      ctx' <- withVar x vn tp' (typeInferCompleteCtx ctx)
  544      pure ((x, vn, tp', s) : ctx')
  545 
  546 -- | Perform type inference on a context via 'typeInferCompleteCtx', and then
  547 -- run a computation in that context via 'withCtx', also passing in that context
  548 -- to the computation
  549 typeInferCompleteInCtx ::
  550   [(LocalName, Un.UTerm)] ->
  551   ([(LocalName, VarName, Term, Sort)] -> TCM a) -> TCM a
  552 typeInferCompleteInCtx ctx f =
  553   do typed_ctx <- typeInferCompleteCtx ctx
  554      withCtx (map (\(x, v, t, _) -> (x, v, t)) typed_ctx) (f typed_ctx)

  560 -- | Test if a 'Term' is an application of a specific datatype with the
  561 -- supplied context of parameters and indices
  562 ctxAsDataTypeApp ::
  563   VarName -> [param] -> [index] -> Term -> Maybe ([Term], [Term])
  564 ctxAsDataTypeApp d params ixs t =
  565   do let (f, args) = asApplyAll t
  566      (d', _) <- asVariable f
  567      guard (d == d')
  568      guard (length args == length params + length ixs)
  569      let (params', ixs') = splitAt (length params) args
  570      pure (params', ixs')
  571 
  572 -- | Test whether a specific datatype occurs in a term.
  573 usesDataType :: VarName -> Term -> Bool
  574 usesDataType d t = IntMap.member (vnIndex d) (varTypes t)
  575 
  576 -- | Check that a type is a valid application of datatype @d@ for use in
  577 -- specific ways in the type of a constructor for @d@. This requires that this
  578 -- application of @d@ be of the form
  579 --
  580 -- > d p1 .. pn x1 .. xm
  581 --
  582 -- where the @pi@ are the distinct bound variables bound in the @params@
  583 -- context, given as argument, and that the @xj@ have no occurrences of @d@. If
  584 -- the given type is of this form, return the @xj@.
  585 asCtorDTApp :: VarName -> [(VarName, Term)] -> [index] -> Term -> Maybe [Term]
  586 asCtorDTApp d params dt_ixs (ctxAsDataTypeApp d params dt_ixs ->
  587                                        Just (param_vars, ixs))
  588   | isVarList (map fst params) param_vars && not (any (usesDataType d) ixs)
  589   = Just ixs
  590   where
  591     -- Check that the given list of terms is a list of named
  592     -- variables, one for each parameter
  593     isVarList :: [VarName] -> [Term] -> Bool
  594     isVarList _ [] = True
  595     isVarList (p : ps) ((asVariable -> Just (x, _)) : ts) =
  596       x == p && isVarList ps ts
  597     isVarList _ _ = False
  598 asCtorDTApp _ _ _ _ = Nothing
  599 
  600 
  601 -- | Check that an argument for a constructor has one of the allowed forms
  602 asCtorArg ::
  603   VarName ->
  604   [(VarName, Term)] ->
  605   [index] ->
  606   Term ->
  607   Maybe CtorArg
  608 asCtorArg d _ _ tp
  609   | not (usesDataType d tp)
  610   = Just (ConstArg tp)
  611 asCtorArg d params dt_ixs tp =
  612   do let (zs, ret) = asPiList tp
  613      case asCtorDTApp d params dt_ixs ret of
  614        Just ixs
  615          | not (any (usesDataType d . snd) zs) ->
  616            Just (RecursiveArg zs ixs)
  617        _ ->
  618          Nothing
  619 
  620 -- | Check that a constructor type is a pi-abstraction that takes as input an
  621 -- argument of one of the allowed forms described by 'CtorArg'
  622 asPiCtorArg ::
  623   VarName ->
  624   [(VarName, Term)] ->
  625   [index] ->
  626   Term ->
  627   Maybe (VarName, CtorArg, Term)
  628 asPiCtorArg d params dt_ixs t =
  629   case asPi t of
  630     Nothing -> Nothing
  631     Just (nm, tp, rest) ->
  632       case asCtorArg d params dt_ixs tp of
  633         Nothing -> Nothing
  634         Just arg -> Just (nm, arg, rest)
  635 
  636 -- | Helper function for 'mkCtorArgStruct'
  637 mkCtorArgsIxs ::
  638   VarName ->
  639   [(VarName, Term)] ->
  640   [index] ->
  641   Term ->
  642   Maybe ([(VarName, CtorArg)], [Term])
  643 mkCtorArgsIxs d params dt_ixs (asCtorDTApp d params dt_ixs -> Just ixs) =
  644   Just ([], ixs)
  645 mkCtorArgsIxs d params dt_ixs ty =
  646   case asPiCtorArg d params dt_ixs ty of
  647     Nothing -> Nothing
  648     Just (x, arg, rest) ->
  649       case mkCtorArgsIxs d params dt_ixs rest of
  650         Nothing -> Nothing
  651         Just (args, ixs) -> Just ((x, arg) : args, ixs)
  652 
  653 -- | Take in a datatype and bindings lists for its parameters and indices, and
  654 -- also a prospective type of a constructor for that datatype, where the
  655 -- constructor type is allowed to have the parameters but not the indices free.
  656 -- Test that the constructor type is an allowed type for a constructor of this
  657 -- datatype, and, if so, build a 'CtorSpec' for it.
  658 mkCtorSpec ::
  659   NameInfo ->
  660   VarName ->
  661   [(VarName, Term)] ->
  662   [(VarName, Term)] ->
  663   Term ->
  664   Maybe CtorSpec
  665 mkCtorSpec nmi d params dt_ixs ctor_tp =
  666   case mkCtorArgsIxs d params dt_ixs ctor_tp of
  667     Nothing -> Nothing
  668     Just (args, ctor_ixs) ->
  669       Just (CtorSpec nmi args ctor_ixs)
