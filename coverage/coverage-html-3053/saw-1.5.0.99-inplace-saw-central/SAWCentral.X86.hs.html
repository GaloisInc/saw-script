  202 -- | Run a top-level proof.
  203 -- Should be used when making a standalone proof script.
  204 proof ::
  205   (FilePath -> IO ByteString) ->
  206   PathSatSolver ->
  207   ArchitectureInfo X86_64 ->
  208   FilePath {- ^ ELF binary -} ->
  209   Maybe FilePath {- ^ Cryptol spec, if any -} ->
  210   [(ByteString,Integer,Unit)] ->
  211   Fun ->
  212   IO (SharedContext,Integer,[Goal])
  213 proof fileReader pss archi file mbCry globs fun =
  214   do sc  <- mkSharedContext
  215      halloc  <- newHandleAllocator
  216      scLoadPreludeModule sc
  217      scLoadCryptolModule sc
  218      sym <- newSAWCoreExprBuilder sc False
  219      SomeOnlineBackend bak <- newSAWCoreBackend pss sym
  220      let ?fileReader = fileReader
  221      cenv <- loadCry sym mbCry
  222      mvar <- mkMemVar "saw_x86:llvm_memory" halloc
  223      proofWithOptions Options
  224        { fileName = file
  225        , function = fun
  226        , archInfo = archi
  227        , backend = SomeBackend bak
  228        , allocator = halloc
  229        , memvar = mvar
  230        , cryEnv = cenv
  231        , extraGlobals = globs
  232        }

  234 -- | Run a proof using the given backend.
  235 -- Useful for integrating with other tool.
  236 proofWithOptions :: Options -> IO (SharedContext,Integer,[Goal])
  237 proofWithOptions opts =
  238   do let path = fileName opts
  239      elf <- getRelevant path =<< getElf path
  240      translate opts elf (function opts)

  242 -- | Add interpretations for the symbolic functions, by looking
  243 -- them up in the Cryptol environment.  There should be definitions
  244 -- for "aesenc", "aesenclast", and "clmul".
  245 registerSymFuns :: Opts -> IO (SymFuns Sym)
  246 registerSymFuns opts =
  247   do let sym = optsSym opts
  248      st  <- sawCoreState sym
  249      sfs <- newSymFuns sym
  250 
  251      sawRegisterSymFunInterp st (fnAesEnc     sfs) (mk2 "aesenc")
  252      sawRegisterSymFunInterp st (fnAesEncLast sfs) (mk2 "aesenclast")
  253      sawRegisterSymFunInterp st (fnClMul      sfs) (mk2 "clmul")
  254 
  255      return sfs
  256 
  257   where
  258   err nm xs =
  259       "Type error in call to " <> nm <>
  260       ": expected 2 arguments, found " <> Text.pack (show $ length xs)
  261 
  262   mk2 nm _sc xs = case xs of
  263                     [_,_] -> cryTerm opts nm xs
  264                     _     -> fail $ Text.unpack $ err nm xs

  276 -- | Parse an elf file.
  277 getElf :: FilePath -> IO (Elf.ElfHeaderInfo 64)
  278 getElf path =
  279   do bs <- BS.readFile path
  280      case Elf.decodeElfHeaderInfo bs of
  281        Left (off, msg) ->
  282            malformed path $ "Invalid ELF header at offset " <> Text.pack (show off) <>
  283                             ": " <> Text.pack msg
  284        Right (Elf.SomeElf hdr) ->
  285            let elfmachine = Elf.headerMachine (Elf.header hdr)
  286                elfclass = Elf.headerClass (Elf.header hdr)
  287            in case (elfmachine, elfclass) of
  288                (Elf.EM_X86_64, Elf.ELFCLASS64) ->
  289                    pure hdr
  290                (Elf.EM_X86_64, _) ->
  291                    -- Note that 32-bit x86 is a different machine; if
  292                    -- we do see a 32-bit x86_64 bin though it might be
  293                    -- one of the several 32-on-64 ABIs (akin to mips
  294                    -- N32) that haven't caught on, so call it
  295                    -- unsupported rather than malformed.
  296                    unsupported path $ "Unexpected ELF class " ++ show elfclass
  297                (_, _) ->
  298                    unsupported path $ "Unexpected ELF machine " ++ show elfmachine

  301 -- | Extract a Macaw "memory" from an ELF file and resolve symbols.
  302 getRelevant :: FilePath -> Elf.ElfHeaderInfo 64 -> IO RelevantElf
  303 getRelevant path elf =
  304   case (memoryForElf opts elf, memoryForElfAllSymbols opts elf) of
  305     (Left err, _) -> malformed path $ Text.pack err
  306     (_, Left err) -> malformed path $ Text.pack err
  307     (Right (mem, faddrs, _warnings, _errs), Right (_, addrs, _, _)) ->
  308       do let toEntry msym = (memSymbolStart msym, memSymbolName msym)
  309          return RelevantElf { memory = mem
  310                             , funSymMap = Map.fromList (map toEntry faddrs)
  311                             , symMap = Map.fromList (map toEntry addrs)
  312                             }
  313 
  314   where
  315   -- XXX: What options do we want?
  316   opts = LoadOptions { loadOffset = Just 0
  317                      }

  322 -- | Find the address(es) of a symbol by name.
  323 findSymbols :: AddrSymMap 64 -> ByteString -> [ MemSegmentOff 64 ]
  324 findSymbols addrs nm = Map.findWithDefault [] nm invertedMap
  325   where
  326   invertedMap = Map.fromListWith (++) [ (y,[x]) | (x,y) <- Map.toList addrs ]
  327 
  328 -- | Find the single address of a symbol, or fail.
  329 findSymbol :: FilePath -> AddrSymMap 64 -> ByteString -> IO (MemSegmentOff 64)
  330 findSymbol path addrs nm =
  331   case findSymbols addrs nm of
  332     [addr] -> return $! addr
  333     []     -> malformed path ("Could not find function " <> Text.pack (show nm))
  334     _      -> malformed path ("Multiple definitions for " <> Text.pack (show nm))

  337 loadGlobal ::
  338   RelevantElf ->
  339   (ByteString, Integer, Unit) ->
  340   IO [(String, Integer, Unit, [Integer])]
  341 loadGlobal elf (nm,n,u) =
  342   case findSymbols (symMap elf) nm of
  343     [] -> do print $ symMap elf
  344              err "Global not found"
  345     _  -> mapM loadLoc (findSymbols (symMap elf) nm)
  346   where
  347   mem   = memory elf
  348   sname = BSC.unpack nm
  349 
  350   readOne a = case u of
  351                 Bytes  -> check (readWord8    mem a)
  352                 Words  -> check (readWord16le mem a)
  353                 DWords -> check (readWord32le mem a)
  354                 QWords -> check (readWord64le mem a)
  355                 _      -> err ("unsuported global size: " ++ show u)
  356 
  357   nextAddr = incAddr (bytesToInteger (1 *. u))
  358 
  359   addrsFor o = take (fromIntegral n) (iterate nextAddr o)
  360 
  361   check :: (Show b, Integral a) => Either b a -> IO Integer
  362   check res = case res of
  363                 Left e  -> err (show e)
  364                 Right a -> return (fromIntegral a)
  365 
  366 
  367   loadLoc off = do let start = segoffAddr off
  368                        a  = memWordToUnsigned (addrOffset start)
  369                    is <- mapM readOne (addrsFor start)
  370                    return (sname, a, u, is)
  371 
  372   err :: [Char] -> IO a
  373   err xs = fail $ unlines
  374                     [ "Failed to load global."
  375                     , "*** Global: " ++ show nm
  376                     , "*** Error: " ++ xs
  377                     ]

  401 callHandler :: Overrides -> CallHandler
  402 callHandler callMap sym = Macaw.LookupFunctionHandle $ \st mem regs -> do
  403   case lookupX86Reg X86_IP regs of
  404     Just (RV ptr) | LLVMPointer base off <- ptr ->
  405       case (asNat base, BV.asUnsigned <$> asBV off) of
  406         (Just b, Just o) ->
  407            case Map.lookup (b,o) callMap of
  408              Just h  -> case h sym of
  409                           Macaw.LookupFunctionHandle f -> f st mem regs
  410              Nothing ->
  411                fail ("No over-ride for function: " ++ show (ppPtr ptr))
  412 
  413         _ -> fail ("Non-static call: " ++ show (ppPtr ptr))
  414 
  415     _ -> fail "[Bug?] Failed to obtain the value of the IP register."

  418 -- | Verify the given function.  The function matches it sepcification,
  419 -- as long as the returned goals can be discharged.
  420 -- Returns the shared context and the goals (from the Sym)
  421 -- and the integer is the (aboslute) address of the function.
  422 translate ::
  423   Options -> RelevantElf -> Fun -> IO (SharedContext, Integer, [Goal])
  424 translate opts elf fun =
  425   do let name = funName fun
  426      sayLn ("Translating function: " ++ BSC.unpack name)
  427 
  428      -- TODO? do we need to pass in the mdMap into more places in this mode?
  429      mdMap <- newIORef mempty
  430 
  431      let ?memOpts = Crucible.defaultMemOptions
  432      let ?recordLLVMAnnotation = \_ _ _ -> return ()
  433 
  434      let bak   = backend opts
  435          sym   = case bak of SomeBackend b -> backendGetSym b
  436          sopts = Opts { optsBackend = bak, optsCry = cryEnv opts, optsMvar = memvar opts }
  437 
  438      sfs <- registerSymFuns sopts
  439 
  440      (globs,st,checkPost) <-
  441         case funSpec fun of
  442           NewStyle mkSpec debug ->
  443             do gss <- mapM (loadGlobal elf) (extraGlobals opts)
  444                spec0 <- mkSpec (cryEnv opts)
  445                let spec = spec0 {specGlobsRO = concat (specGlobsRO spec0:gss)}
  446                (gs,st,po) <- verifyMode spec sopts
  447                debug st
  448                return (gs,st,\st1 -> debug st1 >> po st1)
  449 
  450      addr <- doSim opts elf sfs name globs st checkPost
  451 
  452      gs <- getGoals bak mdMap
  453      sc <- saw_ctx <$> sawCoreState sym
  454      return (sc, addr, gs)

  465 doSim ::
  466   (?memOpts::Crucible.MemOptions, Crucible.HasLLVMAnn Sym) =>
  467   Options ->
  468   RelevantElf ->
  469   SymFuns Sym ->
  470   ByteString ->
  471   (GlobalMap Sym Crucible.Mem 64, Overrides) ->
  472   State ->
  473   (State -> IO ()) ->
  474   IO Integer
  475 doSim opts elf sfs name (globs,overs) st checkPost
  476   | SomeBackend bak <- backend opts =
  477   do say "  Looking for address... "
  478      let path = fileName opts
  479      addr <- findSymbol path (symMap elf) name
  480      -- addr :: MemSegmentOff 64
  481      let addrInt =
  482            let seg :: MemSegment 64
  483                seg = segoffSegment addr
  484            in if segmentBase seg == 0
  485                  then toInteger (segmentOffset seg + segoffOffset addr)
  486                  else error "  Not an absolute address"
  487 
  488      sayLn (show addr)
  489 
  490      SomeCFG cfg <- statusBlock "  Constructing CFG... "
  491                     $ makeCFG opts elf name addr
  492 
  493      -- writeFile "XXX.hs" (show cfg)
  494 
  495      let sym  = backendGetSym bak
  496          mvar = memvar opts
  497 
  498      setSimulatorVerbosity 0 sym
  499 
  500      execResult <- statusBlock "  Simulating... " $ do
  501        let crucRegTypes = crucArchRegTypes x86
  502        let macawStructRepr = StructRepr crucRegTypes
  503        -- The global pointer validity predicate is required if your memory
  504        -- representation has gaps that are not supposed to be mapped and you
  505        -- want to verify that no memory accesses touch unmapped regions.
  506        --
  507        -- The memory setup for this verifier does not have that problem, and
  508        -- thus does not need any additional validity predicates.
  509        let noExtraValidityPred _ _ _ _ = return Nothing
  510        let archEvalFns = x86_64MacawEvalFn sfs defaultMacawArchStmtExtensionOverride
  511        let lookupSyscall = unsupportedSyscalls "saw-script"
  512        let mmConf = MemModelConfig
  513                       { globalMemMap = globs
  514                       , lookupFunctionHandle = callHandler overs sym
  515                       , lookupSyscallHandle = lookupSyscall
  516                       , mkGlobalPointerValidityAssertion = noExtraValidityPred
  517                       , resolvePointer = pure
  518                       , concreteImmutableGlobalRead = \_ _ -> pure Nothing
  519                       , lazilyPopulateGlobalMem = \_ _ -> pure
  520                       }
  521        let ctx :: SimContext (MacawSimulatorState Sym) Sym (MacawExt X86_64)
  522            ctx = initSimContext
  523                    bak
  524                    llvmIntrinsicTypes
  525                    (allocator opts)
  526                    stdout
  527                    (FnBindings $
  528                      insertHandleMap
  529                        (cfgHandle cfg)
  530                        (UseCFG cfg (postdomInfo cfg))
  531                        emptyHandleMap)
  532                    (macawExtensions archEvalFns mvar mmConf)
  533                    MacawSimulatorState
  534        let initGlobals = insertGlobal mvar (stateMem st) emptyGlobals
  535 
  536        executeCrucible []
  537          $ InitialState ctx initGlobals defaultAbortHandler macawStructRepr
  538          $ runOverrideSim macawStructRepr
  539          $ do let args :: RegMap Sym (MacawFunctionArgs X86_64)
  540                   args = RegMap (singleton (RegEntry macawStructRepr
  541                                                       (stateRegs st)))
  542               crucGenArchConstraints x86 $
  543                   do r <- callCFG cfg args
  544                      mem <- readGlobal mvar
  545                      let regs = regValue r
  546                      let sta = State { stateMem = mem, stateRegs = regs }
  547                      liftIO (checkPost sta)
  548                      pure regs
  549 
  550      case execResult of
  551        FinishedResult {} -> pure ()
  552        AbortedResult {}  -> sayLn "[Warning] Function never returns"
  553        TimeoutResult {}  -> timeout path
  554 
  555      return addrInt
  556

  563 -- | Generate a CFG for the function at the given address.
  564 makeCFG ::
  565   Options ->
  566   RelevantElf ->
  567   ByteString ->
  568   MemSegmentOff 64 ->
  569   IO TheCFG
  570 makeCFG opts elf name addr =
  571   do (_,Some funInfo) <- return $ analyzeFunction addr UserRequest empty
  572      -- writeFile "MACAW.cfg" (show (pretty funInfo))
  573      mkFunCFG x86 (allocator opts) cruxName posFn funInfo

  576   cruxName  = functionNameFromText txtName
  577   empty = emptyDiscoveryState (memory elf) (funSymMap elf) (archInfo opts)

  582 -- Goals
  583 
  584 data Goal = Goal
  585   { gAssumes :: [ Term ]              -- ^ Assuming these
  586   , gShows   :: Term                  -- ^ We need to show this
  587   , gMd      :: ConditionMetadata     -- ^ Metadata about the goal
  588   , gMessage :: SimErrorReason        -- ^ We should say this if the proof fails
  589   }
  590 
  591 gLoc :: Goal -> ProgramLoc
  592 gLoc = conditionLoc . gMd
  593 
  594 -- | The proposition that needs proving (i.e., assumptions imply conclusion)
  595 gGoal :: SharedContext -> Goal -> IO Prop
  596 gGoal sc g0 = boolToProp sc [] =<< go (gAssumes g)
  597   where
  598   g = g0 { gAssumes = mapMaybe skip (gAssumes g0) }
  599 
  600   _shG = do putStrLn "Assuming:"
  601             mapM_ _shT (gAssumes g)
  602             putStrLn "Shows:"
  603             _shT (gShows g)
  604 
  605 
  606   _shT t = do
  607       t' <- ppTerm sc PPS.defaultOpts t
  608       putStrLn ("  " ++ t')
  609 
  610   skip a = case asBool a of
  611              Just True -> Nothing
  612              _         -> Just a
  613 
  614   go xs = case xs of
  615             []     -> return (gShows g)
  616             a : as -> scImplies sc a =<< go as
  617 
  618 getGoals :: SomeBackend Sym -> IORef MetadataMap -> IO [Goal]
  619 getGoals (SomeBackend bak) mdMap =
  620   do finalMdMap <- readIORef mdMap
  621      obls <- maybe [] goalsToList <$> getProofObligations bak
  622      st <- sawCoreState sym
  623      mapM (toGoal st finalMdMap) obls
  624   where
  625   sym = backendGetSym bak
  626 
  627   toGoal st finalMdMap (ProofGoal asmps g) =
  628     do a1 <- toSC sym st =<< assumptionsPred sym asmps
  629        p  <- toSC sym st (g ^. labeledPred)
  630        let simErr = g^.labeledPredMsg
  631        let loc = simErrorLoc simErr
  632        let msg = simErrorReason simErr
  633        let defaultMd = ConditionMetadata
  634                        { conditionLoc = loc
  635                        , conditionTags = mempty
  636                        , conditionType = "safety assertion"
  637                        , conditionContext = ""
  638                        }
  639        let md = fromMaybe defaultMd $
  640                   do ann <- W4.getAnnotation sym (g^.labeledPred)
  641                      Map.lookup ann finalMdMap
  642        return Goal { gAssumes = [a1]
  643                    , gShows   = p
  644                    , gMd      = md
  645                    , gMessage = msg
  646                    }
  647 
  648 instance Show Goal where
  649   showsPrec _ g = showString "Goal { gAssumes = "
  650                 . showList (map (show . ppTermPureDefaults) (gAssumes g))
  651                 . showString ", gShows = " . shows (ppTermPureDefaults (gShows g))
  652                 . showString ", gLoc = " . shows (gLoc g)
  653                 . showString ", gMessage = " . shows (show (gMessage g))
  654                 . showString " }"

  667 -- Calling Convention
  668 -- see: http://refspecs.linuxfoundation.org/elf/x86_64-abi-0.99.pdf
  669 -- Need to preserve: %rbp, %rbx, %r12--%r15
  670 -- Preserve control bits in MXCSR
  671 -- Preserve x87 control word.
  672 -- On entry:
  673 --   CPU is in x87 mode
  674 --   DF in $rFLAGS is clear one entry and return.
  675 -- "Red zone" 128 bytes past the end of the stack %rsp.
  676 --    * not modified by interrupts

  680 -- Errors
  681 
  682 -- | Exception for hitting an unsupported object or feature. The arguments
  683 --   are the filename we were looking at, and a message.
  684 data X86Unsupported = X86Unsupported FilePath String deriving Show
  685 
  686 -- | Exception for miscellaneous errors during verification. The arguments
  687 --   are the filename we were looking at, also optionally a function/symbol
  688 --   name, and a message.
  689 data X86Error       = X86Error FilePath (Maybe Text) Text deriving Show
  690 
  691 instance Exception X86Unsupported
  692 instance Exception X86Error
  693 
  694 unsupported :: FilePath -> String -> IO a
  695 unsupported path x = throwIO (X86Unsupported path x)
  696 
  697 malformed :: FilePath -> Text -> IO a
  698 malformed path x = throwIO (X86Error path Nothing x)
  699 
  700 timeout :: FilePath -> IO a
  701 timeout path = throwIO (X86Error path Nothing "Execution timed out")
