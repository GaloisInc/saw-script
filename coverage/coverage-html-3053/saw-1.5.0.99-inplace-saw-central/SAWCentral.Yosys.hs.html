   62 -- ** Building the module graph from Yosys IR
   63 
   64 data Modgraph = Modgraph
   65   { _modgraphGraph :: Graph.Graph
   66   , _modgraphNodeFromVertex :: Graph.Vertex -> (Module, CellTypeName, [CellTypeName])
   67   -- , _modgraphVertexFromKey :: Text -> Maybe Graph.Vertex
   68   }
   69 makeLenses ''Modgraph
   70 
   71 -- | Given a Yosys IR, construct a graph of intermodule dependencies.
   72 yosysIRModgraph :: YosysIR -> Modgraph
   73 yosysIRModgraph ir =
   74   let
   75     moduleToNode :: (CellTypeName, Module) -> (Module, CellTypeName, [CellTypeName])
   76     moduleToNode (nm, m) = (m, nm, deps)
   77       where
   78         deps = Text.pack . show . view cellType <$> Map.elems (m ^. moduleCells)
   79     nodes = moduleToNode <$> Map.assocs (ir ^. yosysModules)
   80     (_modgraphGraph, _modgraphNodeFromVertex, _modgraphVertexFromKey)
   81       = Graph.graphFromEdges nodes
   82   in Modgraph{..}

   84 -- | Given a Yosys IR, construct a map from module names to SAWCore terms alongside SAWCore and Cryptol types
   85 convertYosysIR ::
   86   SC.SharedContext ->
   87   YosysIR ->
   88   IO (Map CellTypeName ConvertedModule)
   89 convertYosysIR sc ir =
   90   do let mg = yosysIRModgraph ir
   91      let sorted = reverseTopSort $ mg ^. modgraphGraph
   92      foldM
   93        (\env v ->
   94           do let (m, nm, _) = mg ^. modgraphNodeFromVertex $ v
   95              cm <- convertModule sc env m
   96              n <- Nonce.freshNonce Nonce.globalNonceGenerator
   97              let qn = QN.fromNameIndex
   98                    QN.NamespaceYosys
   99                    nm
  100                    (fromIntegral $ Nonce.indexValue n)
  101              let ni = SC.mkImportedName qn
  102              let body = cm ^. convertedModuleTerm
  103              tc <- SC.scDefineConstant sc ni body
  104              let cm' = cm { _convertedModuleTerm = tc }
  105              pure $ Map.insert nm cm' env
  106        )
  107        Map.empty
  108        sorted

  110 -- | Given a Yosys IR, construct a map from module names to TypedTerms
  111 yosysIRToTypedTerms ::
  112   SC.SharedContext ->
  113   YosysIR ->
  114   IO (Map Text SC.TypedTerm)
  115 yosysIRToTypedTerms sc ir =
  116   do env <- convertYosysIR sc ir
  117      traverse (\cm -> SC.mkTypedTerm sc (cm ^. convertedModuleTerm)) env
  118 
  119 -- | Given a Yosys IR, construct a SAWCore record containing terms for each module
  120 yosysIRToRecordTerm ::
  121   SC.SharedContext ->
  122   YosysIR ->
  123   IO SC.TypedTerm
  124 yosysIRToRecordTerm sc ir =
  125   do env <- convertYosysIR sc ir
  126      record <- cryptolRecord sc $ view convertedModuleTerm <$> env
  127      SC.mkTypedTerm sc record

  129 -- | Given a Yosys IR, construct a value representing a specific module with all submodules inlined
  130 yosysIRToSequential ::
  131   SC.SharedContext ->
  132   YosysIR ->
  133   CellTypeName ->
  134   IO YosysSequential
  135 yosysIRToSequential sc ir nm =
  136   case Map.lookup nm $ ir ^. yosysModules of
  137     Nothing ->
  138       yosysError $ YosysError $ mconcat
  139       [ "Could not find module: "
  140       , nm
  141       ]
  142     Just m -> convertModuleInline sc m

  149 -- | Produces a Term given the path to a JSON file produced by the
  150 -- Yosys write_json command. The resulting term is a Cryptol record,
  151 -- where each field corresponds to one HDL module exported by Yosys.
  152 -- Each HDL module is in turn represented by a function from a record
  153 -- of input port values to a record of output port values.
  154 yosys_import :: FilePath -> TopLevel SC.TypedTerm
  155 yosys_import path =
  156   do sc <- getSharedContext
  157      ir <- liftIO $ loadYosysIR path
  158      liftIO $ yosysIRToRecordTerm sc ir

  160 -- | Proves equality between a combinational HDL module and a
  161 -- specification. Note that terms derived from HDL modules are first
  162 -- class, and are not restricted to yosys_verify: they may also be
  163 -- used with SAW's typical Term infrastructure like sat, prove_print,
  164 -- term rewriting, etc. yosys_verify simply provides a convenient and
  165 -- familiar interface, similar to llvm_verify or jvm_verify.
  166 yosys_verify ::
  167   SC.TypedTerm {- ^ Term corresponding to the HDL module -} ->
  168   [SC.TypedTerm] {- ^ Preconditions for the equality -} ->
  169   SC.TypedTerm {- ^ Specification term of the same type as the HDL module -} ->
  170   [YosysTheorem] {- ^ Overrides to apply -} ->
  171   ProofScript () ->
  172   TopLevel YosysTheorem
  173 yosys_verify ymod preconds other specs tactic =
  174   do sc <- getSharedContext
  175      newmod <- liftIO $ foldM (flip $ applyOverride sc)
  176        (SC.ttTerm ymod)
  177        specs
  178      mpc <-
  179        case preconds of
  180          [] -> pure Nothing
  181          (pc:pcs) ->
  182            do t <- foldM (\a b -> liftIO $ SC.scAnd sc a b) (SC.ttTerm pc) (SC.ttTerm <$> pcs)
  183               pure . Just $ SC.TypedTerm (SC.ttType pc) t
  184      thm <- liftIO $ buildTheorem sc ymod newmod mpc other
  185      prop <- liftIO $ theoremProp sc thm
  186      _ <- Builtins.provePrintPrim tactic prop
  187      pure thm

  189 -- | Import a single sequential HDL module. N.B. SAW expects the
  190 -- sequential module to exist entirely within a single Yosys module.
  191 yosys_import_sequential ::
  192   Text {- ^ Name of the HDL module -} ->
  193   FilePath {- ^ Path to the Yosys JSON file -} ->
  194   TopLevel YosysSequential
  195 yosys_import_sequential nm path =
  196   do sc <- getSharedContext
  197      ir <- liftIO $ loadYosysIR path
  198      liftIO $ yosysIRToSequential sc ir nm

  200 -- | Extracts a term from the given sequential module with the state
  201 -- eliminated by iterating the term over the given concrete number of
  202 -- cycles. The resulting term has no state field in the inputs or
  203 -- outputs. Each input and output field is replaced with an array of
  204 -- that field's type (array length being the number of cycles
  205 -- specified).
  206 yosys_extract_sequential ::
  207   YosysSequential ->
  208   Integer {- ^ Number of cycles to iterate term -} ->
  209   TopLevel SC.TypedTerm
  210 yosys_extract_sequential s n =
  211   do sc <- getSharedContext
  212      liftIO $ composeYosysSequential sc s n
  213 
  214 -- | Like `yosys_extract_sequential`, but the resulting term has an
  215 -- additional parameter to specify the initial state.
  216 yosys_extract_sequential_with_state ::
  217   YosysSequential ->
  218   Integer {- ^ Number of cycles to iterate term -} ->
  219   TopLevel SC.TypedTerm
  220 yosys_extract_sequential_with_state s n =
  221   do sc <- getSharedContext
  222      liftIO $ composeYosysSequentialWithState sc s n

  224 -- | Extracts a term from the given sequential module. This term has
  225 -- explicit fields for the state of the circuit in the input and
  226 -- output record types.
  227 yosys_extract_sequential_raw :: YosysSequential -> TopLevel SC.TypedTerm
  228 yosys_extract_sequential_raw s = pure $ s ^. yosysSequentialTerm

  230 -- | Export a query over the given sequential module to an input file for the Sally model checker.
  231 yosys_verify_sequential_sally ::
  232   YosysSequential ->
  233   FilePath {- ^ Path to write the resulting Sally input -} ->
  234   SC.TypedTerm {- ^ A boolean function of three parameters: an 8-bit cycle counter, a record of "fixed" inputs, and a record of circuit outputs -} ->
  235   [Text] {- ^ Names of circuit inputs that are fixed -} ->
  236   TopLevel ()
  237 yosys_verify_sequential_sally s path q fixed =
  238   do sc <- getSharedContext
  239      sym <- liftIO $ Common.newSAWCoreExprBuilder sc False
  240      liftIO $ queryModelChecker sym sc s path q $ Set.fromList fixed
