   55 data Namespace =
   56   NamespaceCore | NamespaceCryptol | NamespaceFresh | NamespaceYosys | NamespaceLLVM | NamespaceFree
   57   deriving (Eq, Ord, Enum, Bounded, Lift)
   58 
   59 instance Hashable Namespace where
   60   hashWithSalt s ns = hashWithSalt s (fromEnum ns)
   61 
   62 renderNamespace :: Namespace -> Text
   63 renderNamespace = \case
   64   NamespaceCore -> "core"
   65   NamespaceCryptol -> "cryptol"
   66   NamespaceFresh -> "fresh"
   67   NamespaceYosys -> "yosys"
   68   NamespaceLLVM -> "llvm"
   69   -- distinguished namespace for ad-hoc free variables
   70   NamespaceFree -> "free"
   71 
   72 instance Show Namespace where
   73   show ns = Text.unpack $ renderNamespace ns
   74 
   75 namespaceMap :: Map Text Namespace
   76 namespaceMap = Map.fromList $ map (\ns -> (renderNamespace ns, ns)) [minBound..maxBound]
   77 
   78 readNamespace :: Text -> Maybe Namespace
   79 readNamespace txt = Map.lookup txt namespaceMap

   81 -- | A name with optional additional qualification
   82 data QualName = QualName
   83   { path :: [Text]
   84   , subPath :: [Text]
   85   , baseName :: Text
   86   , index :: Maybe Int
   87   , namespace :: Maybe Namespace
   88   }
   89   deriving (Eq, Ord, Lift)
   90 
   91 fullPath :: QualName -> [Text]
   92 fullPath qn = NE.toList (fullPathNE qn)
   93 
   94 fullPathNE :: QualName -> NE.NonEmpty Text
   95 fullPathNE qn =  NE.prependList (path qn) (NE.prependList (subPath qn) (NE.singleton $ baseName qn))
   96 
   97 instance Hashable QualName where
   98   hashWithSalt s (QualName a b c d e) = s
   99     `hashWithSalt` a
  100     `hashWithSalt` b
  101     `hashWithSalt` c
  102     `hashWithSalt` d
  103     `hashWithSalt` e
  104 
  105 simpleName :: Text -> QualName
  106 simpleName nm = QualName [] [] nm Nothing Nothing
  107 
  108 fromPath :: Namespace -> NE.NonEmpty Text -> QualName
  109 fromPath ns ps = QualName (NE.init ps) [] (NE.last ps) Nothing (Just ns)
  110 
  111 fromNameIndex :: Namespace -> Text -> Int -> QualName
  112 fromNameIndex ns nm i = QualName [] [] nm (Just i) (Just ns)
  113 
  114 -- | Append a base name to a given 'QualName', pushing the existing
  115 --   base name into the path (or subpath, if it is nonempty).
  116 qualify :: QualName -> Text -> QualName
  117 qualify qn txt = do
  118   let prevBaseName = baseName qn
  119   case subPath qn of
  120     [] -> qn { path = path qn ++ [prevBaseName], baseName = txt }
  121     sps -> qn { subPath = sps ++ [prevBaseName], baseName = txt }
  122 
  123 -- | Split a qualified name into a qualifier, base name and index.
  124 split :: QualName -> Maybe (QualName, Text, Maybe Int)
  125 split qn = do
  126   path' <-
  127     do sps <- NE.nonEmpty $ subPath qn
  128        return $ qn { subPath = NE.init sps, baseName = NE.last sps, index = Nothing}
  129     <|>
  130     do ps <- NE.nonEmpty $ path qn
  131        return $ qn { path = NE.init ps, baseName = NE.last ps, index = Nothing}
  132   return $ (path', baseName qn, index qn)
  133 
  134 -- | Generate and render a fresh name from an existing name using the
  135 --   given print function. Avoids clashes with any of the given "used" names.
  136 freshVariant :: Set Text -> (QualName -> Text) -> QualName -> Text
  137 freshVariant used f base_qn = go Set.empty base_qn
  138   where
  139     go attempts qn =
  140       let text = f qn
  141       in case Set.member text attempts  of
  142         -- this is just to avoid infinite loops if the printing function misbehaves
  143         True -> panic "freshen" ["Rendering function did not create a fresh name: " <> text]
  144         False -> case Set.member text used  of
  145           False -> text
  146           True ->
  147             let
  148               next_ix = case index qn of
  149                 Nothing -> Just 1
  150                 Just i -> Just (i + 1)
  151               next_qn = qn { index = next_ix }
  152             in go (Set.insert text attempts) next_qn
  153 
  154 -- | Simple rendering of a qualified name that may discard fields in order
  155 --   to be syntactically valid for use in a binding.
  156 asBoundVar :: QualName -> Text
  157 asBoundVar qn =
  158   let base = case qn of
  159         QualName [] [] nm _ Nothing | validPathElem nm -> nm
  160         _ -> "x"
  161   in case index qn of
  162         Nothing -> base
  163         Just idx -> base <> Text.pack (show idx)
  164 
  165 -- | True if the given path element may be printed directly. If not, it
  166 -- must be prefixed with '!?', quoted and escaped.
  167 validPathElem :: Text -> Bool
  168 validPathElem txt = case Text.uncons txt of
  169   Just (c, txt') ->
  170     (isAlpha c || c == '_') &&
  171     Text.all (\c_ -> isAlphaNum c_ || c_ == '\'' || c_ == '_') txt'
  172   Nothing -> False
  173 
  174 data PrintMode =
  175     AlwaysPrint -- ^ field is printed if it is available
  176   | NeverPrint  -- ^ field is never printed
  177   | MaybePrint  -- ^ prints one or two aliases:
  178                 --   one with the field present (if available)
  179                 --   and one with it absent
  180 
  181 -- | Print options for controlling which aliases to render for a 'QualName'.
  182 data POpts = POpts
  183   { pPath :: PrintMode
  184   , pSubPath :: PrintMode
  185   , pNamespace :: PrintMode
  186   , pIndex :: PrintMode
  187   }
  188 
  189 -- | Print all possible aliases
  190 allAliasesPOpts :: POpts
  191 allAliasesPOpts = POpts MaybePrint MaybePrint MaybePrint MaybePrint
  192 
  193 -- | Print the most qualified alias
  194 fullyQualifiedPOpts :: POpts
  195 fullyQualifiedPOpts = POpts AlwaysPrint AlwaysPrint AlwaysPrint AlwaysPrint
  196 
  197 -- | Valid aliases for a qualified name, from least to most precise,
  198 --   constrained by the provided 'POpts'.
  199 aliasesOpts :: POpts -> QualName -> [Text]
  200 aliasesOpts opts qn = do
  201   ns <- namespaceSuffix
  202   ix <- indexSuffix
  203   ps <- pathText
  204   sps <- subPathText
  205   let pathSuffix = if Text.null sps && Text.null ps then Text.empty else ":"
  206   let bn = pathElem $ baseName qn
  207   return $ ps <> sps <> pathSuffix <> bn <> ix <> ns
  208   where
  209     opt :: PrintMode -> Maybe Text -> [Text]
  210     opt mode mtxt = case (mode, mtxt) of
  211       (NeverPrint,_) -> [Text.empty]
  212       (_, Nothing) -> [Text.empty]
  213       (AlwaysPrint, Just txt) -> [txt]
  214       (MaybePrint, Just txt) -> [Text.empty, txt]
  215 
  216     indexSuffix :: [Text]
  217     indexSuffix = opt (pIndex opts) $
  218       fmap (\i -> "`" <> Text.pack (show i)) (index qn)
  219 
  220     namespaceSuffix :: [Text]
  221     namespaceSuffix = opt (pNamespace opts) $
  222       (fmap (\ns -> "@" <> renderNamespace ns) (namespace qn))
  223 
  224     pathElem :: Text -> Text
  225     pathElem txt = case validPathElem txt of
  226       True -> txt
  227       False -> "!?" <> ppStringLiteral txt
  228 
  229     pathText :: [Text]
  230     pathText = opt (pPath opts) $
  231       case path qn of
  232         [] -> Nothing
  233         ps -> Just $ Text.intercalate "::" (map pathElem ps) <> ":"
  234 
  235     subPathText :: [Text]
  236     subPathText = opt (pSubPath opts) $
  237       case subPath qn of
  238         [] -> Nothing
  239         sp -> Just $ "|:" <> Text.intercalate "::" (map pathElem sp) <> ":"
  240 
  241 -- | Aliases for a qualified name, from least to most precise.
  242 --   The list is nonempty, as it will always include the rendered base
  243 --   name as the first element.
  244 aliases :: QualName -> [Text]
  245 aliases qn = aliasesOpts allAliasesPOpts qn
  246 
  247 -- | Fully-qualified rendering of a qualified name. Equivalent to
  248 --   the last element of 'aliases'.
  249 ppQualName :: QualName -> Text
  250 ppQualName qn = case aliasesOpts fullyQualifiedPOpts qn of
  251   [nm] -> nm
  252   nms -> panic "ppQualName" $
  253     ["unexpected number of fully-qualified aliases: "] ++ nms
  254 
  255 instance PP.Pretty QualName where
  256   pretty qn = PP.pretty $ ppQualName qn
  257 
  258 instance Show QualName where
  259   show qn = Text.unpack (ppQualName qn)
