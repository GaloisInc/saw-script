   82 -- Module Names ----------------------------------------------------------------
   83 
   84 newtype ModuleName = ModuleName QN.QualName
   85   deriving (Eq, Ord, Hashable, TH.Lift)
   86 
   87 instance Show ModuleName where
   88   show mn = Text.unpack (moduleNameText mn)
   89 
   90 moduleNameText :: ModuleName -> Text
   91 moduleNameText (ModuleName qn) = QN.ppQualName (qn{ QN.namespace = Nothing })
   92 
   93 moduleNamePieces :: ModuleName -> [Text]
   94 moduleNamePieces (ModuleName qn) = QN.fullPath qn

   96 -- | Create a module name given a list of strings with the top-most
   97 -- module name given first.
   98 mkModuleName :: [Text] -> ModuleName
   99 mkModuleName [] = panic "mkModuleName" ["empty list"]
  100 mkModuleName (nm:nms) = ModuleName $ QN.fromPath QN.NamespaceCore (nm :| nms)
  101 
  102 preludeName :: ModuleName
  103 preludeName = mkModuleName ["Prelude"]

  108 newtype Ident = Ident QN.QualName
  109   deriving (Eq, Ord, Hashable)
  110 
  111 identModule :: Ident -> ModuleName
  112 identModule (Ident qn) = case QN.split qn of
  113   Just (q, _, _) -> ModuleName q
  114   Nothing -> panic "identModule" ["invalid Ident"]
  115 
  116 identBaseName :: Ident -> Text
  117 identBaseName (Ident qn) = QN.baseName qn
  118 
  119 instance Show Ident where
  120   show i = Text.unpack (identText i)
  121 
  122 identText :: Ident -> Text
  123 identText (Ident qn) = QN.ppQualName (qn{ QN.namespace = Nothing })
  124 
  125 identPieces :: Ident -> NonEmpty Text
  126 identPieces (Ident qn) = QN.fullPathNE qn
  127 
  128 identName :: Ident -> String
  129 identName = Text.unpack . identBaseName

  131 mkIdent :: ModuleName -> Text -> Ident
  132 mkIdent (ModuleName m) s = Ident $ QN.qualify m s
  133 
  134 -- | Make a \"rocq-safe\" identifier from a string that might contain
  135 -- non-identifier characters, where we use the SAW core notion of identifier
  136 -- characters as letters, digits, underscore and primes. Any disallowed
  137 -- character is mapped to the string @__xNN@, where @NN@ is the hexadecimal code
  138 -- for that character. Additionally, a SAW core identifier is not allowed to
  139 -- start with a prime, so a leading underscore is added in such a case.
  140 mkSafeIdent :: ModuleName -> String -> Ident
  141 mkSafeIdent _ [] = fromString "_"
  142 mkSafeIdent mnm nm =
  143   let is_safe_char c = isAlphaNum c || c == '_' || c == '\'' in
  144   mkIdent mnm $ Text.pack $
  145   (if nm!!0 == '\'' then ('_' :) else id) $
  146   concatMap
  147   (\c -> if is_safe_char c then [c] else
  148            "__x" ++ showHex (ord c) "")
  149   nm
  150 
  151 -- | Parse a fully-qualified identifier. Supports either '.' or '::' as path separator.
  152 parseIdent :: String -> Ident
  153 parseIdent s0 = Ident $ case (Text.splitOn sep t0) of
  154   (t1:t2) -> QN.fromPath QN.NamespaceCore (t1:|t2)
  155   _ -> panic "parseIdent" ["invalid identifier: " <> t0]
  156   where
  157     sep = case Text.any (\c -> c=='.') t0 of
  158       True -> "."
  159       False -> "::"
  160     t0 = Text.pack s0
  161 
  162 instance IsString Ident where
  163   fromString = parseIdent

  169 -- | Descriptions of the origins of names that may be in scope
  170 data NameInfo =
  171   NameInfo { nameInfoQualName :: QN.QualName, nameInfoImported :: Bool}
  172   deriving (Eq,Ord,Show)
  173 
  174 -- | This name arises from an exported declaration from a module
  175 pattern ModuleIdentifier :: Ident -> NameInfo
  176 pattern ModuleIdentifier i <- NameInfo (Ident -> i) False  where
  177   ModuleIdentifier i = NameInfo (moduleIdentToQualName i) False
  178 
  179 -- | This name was imported from some other programming language/scope
  180 pattern ImportedName ::
  181   QN.QualName -> -- ^ An absolutely-qualified name, which is required to be unique
  182   [Text] ->  -- ^ A collection of aliases for this name.  Shorter or "less-qualified"
  183              --   aliases should be nearer the front of the list
  184   NameInfo
  185 pattern ImportedName qn as <- NameInfo (qualNameWithAliases -> (qn,as)) True
  186 
  187 mkImportedName :: QN.QualName -> NameInfo
  188 mkImportedName qn = NameInfo qn True
  189 
  190 {-# COMPLETE ModuleIdentifier,ImportedName #-}
  191 
  192 qualNameWithAliases:: QN.QualName -> (QN.QualName, [Text])
  193 qualNameWithAliases qn = (qn,QN.aliases qn)

  195 instance Hashable NameInfo where
  196   hashWithSalt x (NameInfo a b) = x `hashWithSalt` a `hashWithSalt` b
  197 
  198 toQualName :: NameInfo -> QN.QualName
  199 toQualName ni = nameInfoQualName ni
  200 
  201 nameAliases :: NameInfo -> [Text]
  202 nameAliases ni = QN.aliases $ nameInfoQualName ni
  203 
  204 toShortName :: NameInfo -> Text
  205 toShortName ni = QN.baseName $ nameInfoQualName ni
  206 
  207 toAbsoluteName :: NameInfo -> Text
  208 toAbsoluteName ni = QN.ppQualName $ nameInfoQualName ni

  210 moduleIdentToQualName :: Ident -> QN.QualName
  211 moduleIdentToQualName (Ident qn) = qn

  213 scFreshQualName :: Text -> VarIndex -> QN.QualName
  214 scFreshQualName nm i = QN.fromNameIndex QN.NamespaceFresh (if Text.null nm then "_" else nm) i

  216 -- Global Names ----------------------------------------------------------------
  217 
  218 type VarIndex = Int
  219 
  220 -- | A global name with a unique ID. We maintain a global invariant
  221 -- that the 'VarIndex' and the 'NameInfo' must be in a strict
  222 -- one-to-one correspondence: Each 'VarIndex' is paired with a unique
  223 -- 'NameInfo', and each 'NameInfo' is paired with a unique 'VarIndex'.
  224 data Name =
  225   Name
  226   { nameIndex :: !VarIndex
  227   , nameInfo :: !NameInfo
  228   }
  229   deriving (Show)
  230 
  231 -- | Because of the global uniqueness invariant, comparing the
  232 -- 'VarIndex' is sufficient to ensure equality of names.
  233 instance Eq Name where
  234   x == y = nameIndex x == nameIndex y
  235 
  236 instance Ord Name where
  237   compare x y = compare (nameIndex x) (nameIndex y)
  238 
  239 -- | For hashing, we consider only the 'NameInfo' and not the
  240 -- 'VarIndex'; this gives a stable hash value for a particular name,
  241 -- even if the unique IDs are assigned differently from run to run.
  242 instance Hashable Name where
  243   hashWithSalt x nm = hashWithSalt x (nameInfo nm)
  244

  246 -- Variable Names --------------------------------------------------------------
  247 
  248 -- | A variable name with a unique ID.
  249 -- We assume a global uniqueness invariant: The 'VarIndex' determines the name.
  250 data VarName =
  251   VarName
  252   { vnIndex :: !VarIndex
  253   , vnName :: !Text
  254   }
  255   deriving (Show)
  256 
  257 -- | Because of the global uniqueness invariant, comparing the
  258 -- 'VarIndex' is sufficient to ensure equality of variable names.
  259 instance Eq VarName where
  260   x == y = vnIndex x == vnIndex y
  261 
  262 instance Ord VarName where
  263   compare x y = compare (vnIndex x) (vnIndex y)
  264 
  265 -- | For hashing, we consider only the base name and not the
  266 -- 'VarIndex'; this gives a stable hash value for a particular name,
  267 -- even if the unique IDs are assigned differently from run to run.
  268 instance Hashable VarName where
  269   hashWithSalt x vn = hashWithSalt x (vnName vn)
  270 
  271 -- | A wildcard variable name consisting of an underscore "_".
  272 wildcardVarName :: VarName
  273 wildcardVarName = VarName 0 "_"
  274 
  275 -- | A data type representing a context of bound 'VarName's.
  276 -- It maps each 'VarName' to a de Bruijn index.
  277 data VarCtx =
  278   VarCtx
  279   !Int -- ^ Context size
  280   !(IntMap Int) -- ^ Mapping from VarIndex to (size - de Bruijn index)
  281 
  282 -- | The empty 'VarName' context.
  283 emptyVarCtx :: VarCtx
  284 emptyVarCtx = VarCtx 0 IntMap.empty
  285 
  286 -- | Extend a 'VarCtx' with a new bound 'VarName'.
  287 -- The new name has de Bruijn index 0; the index of each previous name
  288 -- is incremented by one.
  289 consVarCtx :: VarName -> VarCtx -> VarCtx
  290 consVarCtx x (VarCtx size m) =
  291   let size' = size + 1
  292   in VarCtx size' (IntMap.insert (vnIndex x) size' m)
  293 
  294 -- | Look up the de Bruijn index of the given 'VarName' in a 'VarCtx'.
  295 -- Return 'Nothing' if the name is not present in the context.
  296 lookupVarCtx :: VarName -> VarCtx -> Maybe Int
  297 lookupVarCtx x (VarCtx size m) =
  298   case IntMap.lookup (vnIndex x) m of
  299     Just i -> Just (size - i)
  300     Nothing -> Nothing

  303 -- Display Name Environments --------------------------------------------------------
  304 
  305 -- | A 'DisplayNameEnv' encodes the mappings between display names (type
  306 -- 'Text') and internal names (type 'VarIndex'). Multiple display
  307 -- names may be associated with each internal name; these are stored
  308 -- in priority order, with preferred display names first.
  309 data DisplayNameEnv =
  310   DisplayNameEnv
  311   { displayNames :: !(IntMap [Text]) -- Keyed by VarIndex; preferred names come first.
  312   , displayIndexes :: !(Map Text IntSet)
  313   }
  314 -- Invariants: The 'displayNames' and 'displayIndexes' maps should be
  315 -- inverses of each other. That is, 'displayNames' maps @i@ to a list
  316 -- containing @x@ if and only if 'displayIndexes' maps @x@ to a set
  317 -- containing @i@.
  318 
  319 emptyDisplayNameEnv :: DisplayNameEnv
  320 emptyDisplayNameEnv = DisplayNameEnv mempty mempty
  321 
  322 -- | Extend a 'DisplayNameEnv' by providing new 'Text' display names
  323 -- for a given 'VarIndex'. Display names should be provided in
  324 -- priority order, with preferred names first. If the 'VarIndex'
  325 -- already exists in the map, then add the new display names with
  326 -- higher priority while keeping the old ones.
  327 extendDisplayNameEnv :: VarIndex -> [Text] -> DisplayNameEnv -> DisplayNameEnv
  328 extendDisplayNameEnv i aliases env =
  329   DisplayNameEnv
  330   { displayNames = IntMap.insertWith List.union i aliases (displayNames env)
  331   , displayIndexes = foldr insertAlias (displayIndexes env) aliases
  332   }
  333   where
  334     insertAlias :: Text -> Map Text IntSet -> Map Text IntSet
  335     insertAlias x m = Map.insertWith IntSet.union x (IntSet.singleton i) m
  336 
  337 -- | Filter display names in a 'DisplayNameEnv' that satisfy a predicate.
  338 filterDisplayNameEnv :: (Text -> Bool) -> DisplayNameEnv -> DisplayNameEnv
  339 filterDisplayNameEnv p env =
  340   DisplayNameEnv
  341   { displayNames = IntMap.filter (not . null) $ fmap (filter p) $ displayNames env
  342   , displayIndexes = Map.filterWithKey (\k _ -> p k) $ displayIndexes env
  343   }
  344 
  345 -- | Delete a 'VarIndex' from the display environment, returning
  346 --   any aliases to it. Has no effect if the index is not present.
  347 deleteDisplayNameEnv :: VarIndex -> DisplayNameEnv -> ([Text], DisplayNameEnv)
  348 deleteDisplayNameEnv i env =
  349   case IntMap.updateLookupWithKey (\_ _ -> Nothing) i (displayNames env) of
  350   (Nothing, _) -> ([], env)
  351   (Just names, displayNames') ->
  352     (names, DisplayNameEnv
  353       { displayNames = displayNames'
  354       , displayIndexes = foldr (Map.update filter_idxs) (displayIndexes env) names
  355       })
  356   where
  357     filter_idxs is =
  358       let is' = IntSet.delete i is
  359       in case IntSet.null is' of
  360         True -> Nothing
  361         False -> Just is'
  362 
  363 -- | Merge two 'DisplayNameEnv's, giving higher priority to display
  364 -- names from the first argument.
  365 mergeDisplayNameEnv :: DisplayNameEnv -> DisplayNameEnv -> DisplayNameEnv
  366 mergeDisplayNameEnv env1 env2 =
  367   DisplayNameEnv
  368   { displayNames = IntMap.unionWith List.union (displayNames env1) (displayNames env2)
  369   , displayIndexes = Map.unionWith IntSet.union (displayIndexes env1) (displayIndexes env2)
  370   }
  371 
  372 -- | Look up a 'Text' display name in an environment, returning the
  373 -- list of indexes it could resolve to.
  374 resolveDisplayName :: DisplayNameEnv -> Text -> [VarIndex]
  375 resolveDisplayName env x =
  376   case Map.lookup x (displayIndexes env) of
  377     Nothing -> []
  378     Just s -> IntSet.elems s
  379 
  380 -- | Return the first display name for the given 'VarIndex' that is
  381 -- unambiguous in the naming environment. If there is no unambiguous
  382 -- alias, then return 'Nothing'.
  383 bestDisplayName :: DisplayNameEnv -> VarIndex -> Maybe Text
  384 bestDisplayName env i =
  385   do aliases <- IntMap.lookup i (displayNames env)
  386      go aliases
  387   where
  388     go [] = Nothing
  389     go (x : xs) =
  390       case Map.lookup x (displayIndexes env) of
  391         Nothing -> panic "bestDisplayName" ["Invariant violated: Missing key " <> x]
  392         Just vs
  393           | IntSet.size vs == 1 -> Just x
  394           | otherwise -> go xs
  395
