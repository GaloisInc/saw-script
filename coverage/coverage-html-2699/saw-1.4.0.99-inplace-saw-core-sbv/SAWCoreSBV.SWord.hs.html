   30 type SBool = SVal
   31 type SWord = SVal
   32 type SInteger = SVal

   34 fromBitsLE :: [SBool] -> SWord
   35 fromBitsLE bs = foldl' f (literalSWord 0 0) bs
   36   where f w b = svJoin (svToWord1 b) w
   37 
   38 literalSWord :: Int -> Integer -> SWord
   39 literalSWord w i = svInteger (KBounded False w) i
   40 
   41 literalSInteger :: Integer -> SWord
   42 literalSInteger i = svInteger KUnbounded i

   44 forallSWord :: String -> Int -> Symbolic SWord
   45 forallSWord nm w = symbolicEnv >>= liftIO . svMkSymVar (NonQueryVar (Just ALL)) (KBounded False w) (Just nm)
   46 
   47 forallSWord_ :: Int -> Symbolic SWord
