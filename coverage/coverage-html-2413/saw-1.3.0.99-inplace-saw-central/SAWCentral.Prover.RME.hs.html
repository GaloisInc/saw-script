   17 -- | Bit-blast a proposition and check its validity using RME.
   18 proveRME ::
   19   SATQuery {- ^ The query to be proved -} ->
   20   TopLevel (Maybe CEX, String)
   21 proveRME satq = getSharedContext >>= \sc -> liftIO $
   22   RME.withBitBlastedSATQuery sc Map.empty satq $ \lit shapes ->
   23     case RME.sat lit of
   24       Nothing -> return (Nothing, "RME")
   25       Just cex -> do
   26         let m = Map.fromList cex
   27         let n = sum (map (sizeFiniteType . snd) shapes)
   28         let bs = map (maybe False id . flip Map.lookup m) $ take n [0..]
   29         let r = liftCexBB (map snd shapes) bs
   30         case r of
   31           Left err -> fail $ "Can't parse counterexample: " ++ err
   32           Right vs
   33             | length shapes == length vs -> do
   34               let model = zip (map fst shapes) (map toFirstOrderValue vs)
   35               return (Just model, "RME")
   36             | otherwise -> fail $ unwords ["RME SAT results do not match expected arguments", show shapes, show vs]
