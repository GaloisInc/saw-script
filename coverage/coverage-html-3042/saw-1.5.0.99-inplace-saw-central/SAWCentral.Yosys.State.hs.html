   44 -- | A SAWCore translation of an HDL module alongside some type
   45 -- information that is useful to keep around.
   46 data YosysSequential = YosysSequential
   47   { _yosysSequentialTerm :: SC.TypedTerm
   48     -- ^ The "raw" SAWCore term derived from the module, which
   49     -- includes a __state__ field in the input and output.
   50   , _yosysSequentialStateFields :: Map Text (SC.Term, C.Type)
   51     -- ^ A mapping from each state field name to a SAWCore and Cryptol type.
   52   , _yosysSequentialInputFields :: Map Text (SC.Term, C.Type)
   53     -- ^ A mapping from each input to a SAWCore and Cryptol type.
   54   , _yosysSequentialOutputFields :: Map Text (SC.Term, C.Type)
   55     -- ^ A mapping from each output to a SAWCore and Cryptol type.
   56   , _yosysSequentialInputWidths :: Map Text Natural
   57     -- ^ A mapping from each input to a width.
   58   , _yosysSequentialOutputWidths :: Map Text Natural
   59     -- ^ A mapping from each output to a width.
   60   , _yosysSequentialStateWidths :: Map Text Natural
   61     -- ^ A mapping from each state field to a width.
   62   }
   63 makeLenses ''YosysSequential

   65 -- | Add a record-typed field named __states__ to the given mapping of
   66 -- field names to types.
   67 insertStateField ::
   68   SC.SharedContext ->
   69   Map Text (SC.Term, C.Type) {- ^ The field types of "__states__" -} ->
   70   Map Text (SC.Term, C.Type) {- ^ The mapping to update -} ->
   71   IO (Map Text (SC.Term, C.Type))
   72 insertStateField sc stateFields fields =
   73   do stateRecordType <- fieldsToType sc stateFields
   74      stateRecordCryptolType <- fieldsToCryptolType stateFields
   75      pure $ Map.insert "__state__" (stateRecordType, stateRecordCryptolType) fields

   77 -- | Translate a stateful HDL module into SAWCore
   78 convertModuleInline ::
   79   SC.SharedContext ->
   80   Module ->
   81   IO YosysSequential
   82 convertModuleInline sc m =
   83   do let ng = moduleNetgraph m
   84 
   85      let netnames =
   86            Map.fromList
   87            [ (n ^. netnameBits, t)
   88            | (t, n) <- Map.assocs (m ^. moduleNetnames), not (n ^. netnameHideName) ]
   89 
   90      let bestName t c =
   91            fromMaybe (cellIdentifier t) $
   92            do bs <- Map.lookup "Q" (c ^. cellConnections)
   93               Map.lookup bs netnames
   94 
   95      -- construct SAWCore and Cryptol types
   96      let dffs =
   97            Map.fromList
   98            [ (bestName t c, c) | (t, c) <- Map.assocs (m ^. moduleCells), cellIsRegister c ]
   99 
  100      stateWidths <- forM dffs $ \c ->
  101        case Map.lookup "Q" $ c ^. cellConnections of
  102          Nothing -> panic "convertModuleInline" ["Missing expected output name for $dff cell"]
  103          Just b -> pure . fromIntegral $ length b
  104 
  105      stateFields <-
  106        forM stateWidths $ \w ->
  107        do t <- SC.scBitvector sc w
  108           let cty = C.tWord $ C.tNum w
  109           pure (t, cty)
  110 
  111      let inputPorts = moduleInputPorts m
  112      let outputPorts = moduleOutputPorts m
  113      inputFields <-
  114        forM inputPorts $ \inp ->
  115        do ty <- SC.scBitvector sc . fromIntegral $ length inp
  116           let cty = C.tWord . C.tNum $ length inp
  117           pure (ty, cty)
  118      outputFields <-
  119        forM outputPorts $ \out ->
  120        do ty <- SC.scBitvector sc . fromIntegral $ length out
  121           let cty = C.tWord . C.tNum $ length out
  122           pure (ty, cty)
  123 
  124      domainFields <- insertStateField sc stateFields inputFields
  125      codomainFields <- insertStateField sc stateFields outputFields
  126 
  127      domainRecordType <- fieldsToType sc domainFields
  128      domainCryptolRecordType <- fieldsToCryptolType domainFields
  129      -- codomainRecordType <- fieldsToType sc codomainFields
  130      codomainCryptolRecordType <- fieldsToCryptolType codomainFields
  131 
  132      -- convert module into term
  133      domainRecord <- SC.scFreshVariable sc "input" domainRecordType
  134 
  135      derivedInputs <-
  136        forM (Map.assocs inputPorts) $ \(nm, inp) ->
  137        do t <- cryptolRecordSelect sc domainFields domainRecord nm
  138           deriveTermsByIndices sc inp t
  139 
  140      preStateRecord <- cryptolRecordSelect sc domainFields domainRecord "__state__"
  141      derivedPreState <-
  142        forM (Map.assocs dffs) $ \(cnm, c) ->
  143        case Map.lookup "Q" $ c ^. cellConnections of
  144          Nothing -> panic "convertModuleInline" ["Missing expected output name for $dff cell"]
  145          Just b ->
  146            do t <- cryptolRecordSelect sc stateFields preStateRecord cnm
  147               deriveTermsByIndices sc b t
  148 
  149      oneBitType <- SC.scBitvector sc 1
  150      xMsg <- SC.scString sc "Attempted to read X bit"
  151      xTerm <- SC.scGlobalApply sc (SC.mkIdent SC.preludeName "error") [oneBitType, xMsg]
  152      zMsg <- SC.scString sc "Attempted to read Z bit"
  153      zTerm <- SC.scGlobalApply sc (SC.mkIdent SC.preludeName "error") [oneBitType, zMsg]
  154      let inputs = Map.unions $ mconcat
  155            [ [ Map.fromList
  156                [ ( [BitrepZero], PretermBvNat 1 0 )
  157                , ( [BitrepOne], PretermBvNat 1 1 )
  158                , ( [BitrepX], PretermSlice 0 1 0 xTerm )
  159                , ( [BitrepZ], PretermSlice 0 1 0 zTerm )
  160                ]
  161              ]
  162            , derivedInputs
  163            , derivedPreState
  164            ]
  165 
  166      terms <- netgraphToTerms sc Map.empty ng inputs
  167 
  168      postStateFields <-
  169        mapForWithKeyM dffs $ \cnm c ->
  170        case Map.lookup "D" $ c ^. cellConnections of
  171          Nothing -> panic "convertModuleInline" ["Missing expected input name for $dff cell"]
  172          Just b -> lookupPatternTerm sc (YosysBitvecConsumerCell cnm "D") b terms
  173      postStateRecord <- cryptolRecord sc postStateFields
  174 
  175      outputRecord <- cryptolRecord sc . Map.insert "__state__" postStateRecord =<< mapForWithKeyM outputPorts
  176        (\onm out -> lookupPatternTerm sc (YosysBitvecConsumerOutputPort onm) out terms)
  177 
  178      -- construct result
  179      t <- SC.scAbstractTerms sc [domainRecord] outputRecord
  180      -- ty <- SC.scFun sc domainRecordType codomainRecordType
  181      let cty = C.tFun domainCryptolRecordType codomainCryptolRecordType
  182      pure YosysSequential
  183        { _yosysSequentialTerm = SC.TypedTerm (SC.TypedTermSchema $ C.tMono cty) t
  184        , _yosysSequentialStateFields = stateFields
  185        , _yosysSequentialInputFields = inputFields
  186        , _yosysSequentialOutputFields = outputFields
  187        , _yosysSequentialInputWidths = fromIntegral . length <$> inputPorts
  188        , _yosysSequentialOutputWidths = fromIntegral . length <$> outputPorts
  189        , _yosysSequentialStateWidths = stateWidths
  190        }

  192 -- | Given a SAWCore term with an explicit state, iterate the term the
  193 -- given number of times. The resulting term has a parameter for the
  194 -- initial state, the resulting Cryptol types does not.
  195 composeYosysSequentialHelper ::
  196   SC.SharedContext ->
  197   YosysSequential ->
  198   Integer ->
  199   IO (SC.Term, C.Type)
  200 composeYosysSequentialHelper sc s n =
  201   do let t = SC.ttTerm $ s ^. yosysSequentialTerm
  202 
  203      width <- SC.scNat sc $ fromIntegral n
  204      extendedInputFields <-
  205        forM (s ^. yosysSequentialInputFields) $ \(ty, cty) ->
  206        do exty <- SC.scVecType sc width ty
  207           let excty = C.tSeq (C.tNum n) cty
  208           pure (exty, excty)
  209      extendedOutputFields <-
  210        forM (s ^. yosysSequentialOutputFields) $ \(ty, cty) ->
  211        do exty <- SC.scVecType sc width ty
  212           let excty = C.tSeq (C.tNum n) cty
  213           pure (exty, excty)
  214      extendedInputType <- fieldsToType sc extendedInputFields
  215      extendedInputCryptolType <- fieldsToCryptolType extendedInputFields
  216      extendedInputRecord <- SC.scFreshVariable sc "input" extendedInputType
  217      extendedOutputCryptolType <- fieldsToCryptolType extendedOutputFields
  218 
  219      allInputs <-
  220        fmap Map.fromList . forM (Map.keys extendedInputFields) $ \nm ->
  221        do inp <- cryptolRecordSelect sc extendedInputFields extendedInputRecord nm
  222           pure (nm, inp)
  223 
  224      codomainFields <- insertStateField sc (s ^. yosysSequentialStateFields) $ s ^. yosysSequentialOutputFields
  225 
  226      let
  227        buildIntermediateInput :: Integer -> SC.Term -> IO SC.Term
  228        buildIntermediateInput i st =
  229          do inps <-
  230               fmap Map.fromList . forM (Map.assocs allInputs) $ \(nm, inp) ->
  231               case Map.lookup nm $ s ^. yosysSequentialInputFields of
  232                 Nothing -> throw . YosysError $ "Invalid input: " <> nm
  233                 Just (elemty, _) ->
  234                   do idx <- SC.scNat sc $ fromIntegral i
  235                      idxed <- SC.scAt sc width elemty inp idx
  236                      pure (nm, idxed)
  237             let inpsWithSt = Map.insert "__state__" st inps
  238             cryptolRecord sc inpsWithSt
  239 
  240        summarizeOutput :: SC.Term -> IO (SC.Term, Map Text SC.Term)
  241        summarizeOutput outrec =
  242          do outstate <- cryptolRecordSelect sc codomainFields outrec "__state__"
  243             outputs <-
  244               fmap Map.fromList . forM (Map.assocs $ s ^. yosysSequentialOutputFields) $ \(nm, (ty, _)) ->
  245               do out <- cryptolRecordSelect sc codomainFields outrec nm
  246                  wrapped <- SC.scSingle sc ty out
  247                  pure (nm, wrapped)
  248             pure (outstate, outputs)
  249 
  250        compose1 :: Integer -> (SC.Term, Map Text SC.Term) -> IO (SC.Term, Map Text SC.Term)
  251        compose1 i (st, outs) =
  252          do inprec <- buildIntermediateInput i st
  253             outrec <- SC.scApply sc t inprec
  254             (st', outs') <- summarizeOutput outrec
  255             mergedOuts <-
  256               fmap Map.fromList . forM (Map.assocs outs') $ \(nm, arr) ->
  257               case (Map.lookup nm $ s ^. yosysSequentialOutputFields, Map.lookup nm outs) of
  258                 (Just (ty, _), Just rest) ->
  259                   do restlen <- SC.scNat sc $ fromIntegral i
  260                      arrlen <- SC.scNat sc 1
  261                      appended <- SC.scAppend sc restlen arrlen ty rest arr
  262                      pure (nm, appended)
  263                 _ -> pure (nm, arr)
  264             pure (st', mergedOuts)
  265 
  266      stateType <- fieldsToType sc $ s ^. yosysSequentialStateFields
  267      initialState <- SC.scFreshVariable sc "initial_state" stateType
  268      (_, outputs) <- foldM (\acc i -> compose1 i acc) (initialState, Map.empty) [0..n-1]
  269 
  270      outputRecord <- cryptolRecord sc outputs
  271      res <- SC.scAbstractTerms sc [initialState, extendedInputRecord] outputRecord
  272      let cty = C.tFun extendedInputCryptolType extendedOutputCryptolType
  273 
  274      pure (res, cty)

  276 -- | Given a SAWCore term with an explicit state, iterate the term the
  277 -- given number of times. Accessing the initial state produces an
  278 -- error.
  279 composeYosysSequential ::
  280   SC.SharedContext ->
  281   YosysSequential ->
  282   Integer ->
  283   IO SC.TypedTerm
  284 composeYosysSequential sc s n =
  285   do (t, cty) <- composeYosysSequentialHelper sc s n
  286      stateType <- fieldsToType sc $ s ^. yosysSequentialStateFields
  287      initialStateMsg <- SC.scString sc "Attempted to read initial state of sequential circuit"
  288      initialState <- SC.scGlobalApply sc (SC.mkIdent SC.preludeName "error") [stateType, initialStateMsg]
  289      res <- SC.scApply sc t initialState
  290      pure $ SC.TypedTerm (SC.TypedTermSchema $ C.tMono cty) res
  291 
  292 -- | Given a SAWCore term with an explicit state, iterate the term the
  293 -- given number of times. The resulting term has a parameter for the
  294 -- initial state.
  295 composeYosysSequentialWithState ::
  296   SC.SharedContext ->
  297   YosysSequential ->
  298   Integer ->
  299   IO SC.TypedTerm
  300 composeYosysSequentialWithState sc s n =
  301   do (t, cty) <- composeYosysSequentialHelper sc s n
  302      scty <- fieldsToCryptolType $ s ^. yosysSequentialStateFields
  303      pure $ SC.TypedTerm (SC.TypedTermSchema . C.tMono $ C.tFun scty cty) t
