   40 -- | Within SAWScript, we represent an object language term as a
   41 -- SAWCore shared term paired with a Cryptol type schema. The Cryptol
   42 -- type is used for type inference/checking of inline Cryptol
   43 -- expressions.
   44 
   45 data TypedTerm =
   46   TypedTerm
   47   { ttType :: TypedTermType
   48   , ttTerm :: Term
   49   }
   50   deriving Show
   51 
   52 
   53 -- | The different notion of Cryptol types that
   54 --   a SAWCore term might have.
   55 data TypedTermType
   56   = TypedTermSchema C.Schema
   57   | TypedTermKind   C.Kind
   58   | TypedTermOther  Term
   59  deriving Show
   60 
   61 
   62 ppTypedTerm :: TypedTerm -> PP.Doc ann
   63 ppTypedTerm (TypedTerm tp tm) =
   64   PP.unAnnotate (ppTerm PPS.defaultOpts tm)
   65   PP.<+> PP.pretty ":" PP.<+>
   66   ppTypedTermType tp
   67 
   68 ppTypedTermType :: TypedTermType -> PP.Doc ann
   69 ppTypedTermType (TypedTermSchema sch) =
   70   PP.viaShow (C.ppPrec 0 sch)
   71 ppTypedTermType (TypedTermKind k) =
   72   PP.viaShow (C.ppPrec 0 k)
   73 ppTypedTermType (TypedTermOther tp) =
   74   PP.unAnnotate (ppTerm PPS.defaultOpts tp)
   75 
   76 ppTypedVariable :: TypedVariable -> PP.Doc ann
   77 ppTypedVariable (TypedVariable ctp vn _tp) =
   78   PP.unAnnotate (PP.pretty (vnName vn))
   79   PP.<+> PP.pretty ":" PP.<+>
   80   PP.viaShow (C.ppPrec 0 ctp)

   83 -- | Convert the 'ttType' field of a 'TypedTerm' to a SAW core term
   84 ttTypeAsTerm :: SharedContext -> Env -> TypedTerm -> IO Term
   85 ttTypeAsTerm sc env (TypedTerm (TypedTermSchema schema) _) =
   86   importSchema sc env schema
   87 ttTypeAsTerm sc _ (TypedTerm (TypedTermKind k) _) = importKind sc k
   88 ttTypeAsTerm _ _ (TypedTerm (TypedTermOther tp) _) = return tp
   89 
   90 ttTermLens :: Functor f => (Term -> f Term) -> TypedTerm -> f TypedTerm
   91 ttTermLens f tt = tt `seq` fmap (\x -> tt{ttTerm = x}) (f (ttTerm tt))
   92 
   93 ttIsMono :: TypedTermType -> Maybe C.Type
   94 ttIsMono ttp =
   95   case ttp of
   96     TypedTermSchema sch -> C.isMono sch
   97     _ -> Nothing
   98 
   99 mkTypedTerm :: SharedContext -> Term -> IO TypedTerm
  100 mkTypedTerm sc trm = do
  101   ty <- scTypeOf sc trm
  102   ct <- scCryptolType sc ty
  103   let ttt = case ct of
  104         Nothing        -> TypedTermOther ty
  105         Just (Left k)  -> TypedTermKind k
  106         Just (Right t) -> TypedTermSchema (C.tMono t)
  107   return (TypedTerm ttt trm)
  108 
  109 -- | Apply a function-typed 'TypedTerm' to an argument.
  110 --   This operation fails if the type of the argument does
  111 --   not match the function.
  112 applyTypedTerm :: SharedContext -> TypedTerm -> TypedTerm -> IO TypedTerm
  113 applyTypedTerm sc x y = applyTypedTerms sc x [y]
  114 
  115 -- | Apply a 'TypedTerm' to a list of arguments. This operation fails
  116 -- if the first 'TypedTerm' does not have a function type of
  117 -- sufficient arity, or if the types of the arguments do not match
  118 -- the type of the function.
  119 applyTypedTerms :: SharedContext -> TypedTerm -> [TypedTerm] -> IO TypedTerm
  120 applyTypedTerms sc (TypedTerm _ fn) args =
  121   do trm <- foldM (scApply sc) fn (map ttTerm args)
  122      ty <- scTypeCheckError sc trm
  123      -- NB, scCryptolType can behave in strange ways due to the non-injectivity
  124      -- of the mapping from Cryptol to SAWCore types. Perhaps we would be better
  125      -- to combine the incoming type information instead of applying and then
  126      -- reconstructing here.
  127      ct <- scCryptolType sc ty
  128      let ttt = case ct of
  129            Nothing        -> TypedTermOther ty
  130            Just (Left k)  -> TypedTermKind k
  131            Just (Right t) -> TypedTermSchema (C.tMono t)
  132      return (TypedTerm ttt trm)

  135 -- | Create an abstract defined constant with the specified name and body.
  136 defineTypedTerm :: SharedContext -> Text -> TypedTerm -> IO TypedTerm
  137 defineTypedTerm sc name (TypedTerm schema t) =
  138   do ty <- scTypeOf sc t
  139      TypedTerm schema <$> scFreshConstant sc name t ty

  141 -- | Make a tuple value from a list of 'TypedTerm's. This operation
  142 -- fails if any 'TypedTerm' in the list has a polymorphic type.
  143 tupleTypedTerm :: SharedContext -> [TypedTerm] -> IO TypedTerm
  144 tupleTypedTerm sc tts =
  145   case traverse (ttIsMono . ttType) tts of
  146     Nothing -> fail "tupleTypedTerm: invalid polymorphic term"
  147     Just ctys ->
  148       TypedTerm (TypedTermSchema (C.tMono (C.tTuple ctys))) <$>
  149         scTuple sc (map ttTerm tts)
  150 
  151 -- | Given a 'TypedTerm' with a tuple type, return a list of its
  152 -- projected components. This operation fails if the 'TypedTerm' does
  153 -- not have a tuple type.
  154 destTupleTypedTerm :: SharedContext -> TypedTerm -> IO [TypedTerm]
  155 destTupleTypedTerm sc (TypedTerm tp t) =
  156   case C.tIsTuple =<< ttIsMono tp of
  157     Nothing -> fail "asTupleTypedTerm: not a tuple type"
  158     Just ctys ->
  159       do let len = length ctys
  160          let idxs = take len [1 ..]
  161          ts <- traverse (\i -> scTupleSelector sc t i len) idxs
  162          pure $ zipWith TypedTerm (map (TypedTermSchema . C.tMono) ctys) ts

  164 -- First order types and values ------------------------------------------------
  165 
  166 cryptolTypeOfFirstOrderType :: FirstOrderType -> C.Type
  167 cryptolTypeOfFirstOrderType fot =
  168   case fot of
  169     FOTBit -> C.tBit
  170     FOTInt -> C.tInteger
  171     FOTIntMod n -> C.tIntMod (C.tNum n)
  172     FOTVec n t -> C.tSeq (C.tNum n) (cryptolTypeOfFirstOrderType t)
  173     -- NB, special case, don't produce 1-tuples
  174     FOTTuple [x] -> cryptolTypeOfFirstOrderType x
  175     FOTTuple ts -> C.tTuple (map cryptolTypeOfFirstOrderType ts)
  176     FOTArray a b ->
  177       C.tArray
  178       (cryptolTypeOfFirstOrderType a)
  179       (cryptolTypeOfFirstOrderType b)
  180     FOTRec m ->
  181       C.tRec $
  182       C.recordFromFields $
  183       [ (C.mkIdent l, cryptolTypeOfFirstOrderType t)
  184       | (l, t) <- Map.assocs m ]
  185 
  186 typedTermOfFirstOrderValue :: SharedContext -> FirstOrderValue -> IO TypedTerm
  187 typedTermOfFirstOrderValue sc fov =
  188   do let fot = firstOrderTypeOf fov
  189      let cty = cryptolTypeOfFirstOrderType fot
  190      t <- scFirstOrderValue sc fov
  191      pure $ TypedTerm (TypedTermSchema (C.tMono cty)) t

  195 data TypedVariable =
  196   TypedVariable
  197   { tvCType :: C.Type
  198   , tvName :: VarName
  199   , tvType :: Term
  200   }
  201   deriving Show
  202 
  203 -- | Recognize 'TypedTerm's that are variables.
  204 asTypedVariable :: TypedTerm -> Maybe TypedVariable
  205 asTypedVariable (TypedTerm tp t) =
  206   do cty <- ttIsMono tp
  207      ec <- asVariable t
  208      pure $ TypedVariable cty (ecName ec) t
  209 
  210 -- | Make a 'TypedTerm' from a 'TypedVariable'.
  211 typedTermOfVariable :: SharedContext -> TypedVariable -> IO TypedTerm
  212 typedTermOfVariable sc (TypedVariable cty vn ty) =
  213   TypedTerm (TypedTermSchema (C.tMono cty)) <$> scVariable sc (EC vn ty)
  214 
  215 abstractTypedVars :: SharedContext -> [TypedVariable] -> TypedTerm -> IO TypedTerm
  216 abstractTypedVars sc tvars (TypedTerm (TypedTermSchema (C.Forall params props ty)) trm) =
  217   do let tys = map tvCType tvars
  218      let vars = map (\(TypedVariable _ vn tp) -> (vn, tp)) tvars
  219      let ty' = foldr C.tFun ty tys
  220      trm' <- scLambdaList sc vars trm
  221      pure $ TypedTerm (TypedTermSchema (C.Forall params props ty')) trm'
  222 abstractTypedVars sc tvars (TypedTerm (TypedTermKind k) trm) =
  223   do let vars = map (\(TypedVariable _ vn tp) -> (vn, tp)) tvars
  224      trm' <- scLambdaList sc vars trm
  225      pure $ TypedTerm (TypedTermKind k) trm'
  226 abstractTypedVars sc tvars (TypedTerm (TypedTermOther _tp) trm) =
  227   do let vars = map (\(TypedVariable _ vn tp) -> (vn, tp)) tvars
  228      trm' <- scLambdaList sc vars trm
  229      tp'  <- scTypeOf sc trm'
  230      pure $ TypedTerm (TypedTermOther tp') trm'

  234 -- | In SAWScript, we can refer to a Cryptol module as a first class
  235 -- value. These are represented simply as maps from names to typed
  236 -- terms.
  237 
  238 data CryptolModule =
  239   CryptolModule
  240     (Map C.Name C.TySyn)    -- type synonyms
  241     (Map C.Name TypedTerm)  -- symbols (mapping to SawCore things).
  242 
  243 showCryptolModule :: CryptolModule -> String
  244 showCryptolModule (CryptolModule sm tm) =
  245   unlines $
  246     (if Map.null sm then [] else
  247        "Type Synonyms" : "=============" : map showTSyn (Map.elems sm) ++ [""]) ++
  248     "Symbols" : "=======" : concatMap showBinding (Map.assocs tm)
  249   where
  250     showTSyn (C.TySyn name params _props rhs _doc) =
  251       "    " ++ unwords (C.pretty (nameIdent name) : map C.pretty params) ++ " = " ++ C.pretty rhs
  252 
  253     showBinding (name, TypedTerm (TypedTermSchema schema) _) =
  254       ["    " ++ C.pretty (nameIdent name) ++ " : " ++ C.pretty schema]
  255     showBinding _ =
  256       []
