   60 extractUninterp ::
   61   (?recordEC :: BoundECRecorder) =>
   62   SharedContext ->
   63   ModuleMap ->
   64   Map Ident TmPrim {- ^ additional primitives -} ->
   65   Map VarIndex TmValue {- ^ ExtCns values -} ->
   66   Set VarIndex {- ^ 'unints' Constants in this set are kept uninterpreted -} ->
   67   Set VarIndex {- ^ 'opaque' Constants in this set are not evaluated -} ->
   68   Term ->
   69   IO (Term, ReplaceUninterpMap)
   70 extractUninterp sc m addlPrims ecVals unintSet opaqueSet t =
   71   do mapref <- newIORef mempty
   72      cfg <- mfix (\cfg -> Sim.evalGlobal' m (Map.union addlPrims (constMap sc cfg))
   73                              (extcns cfg mapref) (uninterpreted cfg mapref) (neutral cfg) (primHandler cfg))
   74      v <- Sim.evalSharedTerm cfg t
   75      tv <- evalType cfg =<< scTypeOf sc t
   76      t' <- readBackValue sc cfg tv v
   77      replMap <- readIORef mapref
   78      return (t', replMap)
   79 
   80  where
   81     uninterpreted cfg mapref tf ec@(EC ix _nm ty)
   82       | Set.member ix opaqueSet = Just $
   83           do tm <- scTermF sc tf
   84              reflectTerm sc cfg ty tm
   85       | Set.member ix unintSet = Just (replace sc cfg mapref ec)
   86       | otherwise = Nothing
   87 
   88     extcns cfg mapref tf ec@(EC ix _nm ty)
   89       | Set.member ix unintSet = replace sc cfg mapref ec
   90       | otherwise =
   91           case Map.lookup ix ecVals of
   92             Just v  -> return v
   93             Nothing ->
   94               do tm <- scTermF sc tf
   95                  reflectTerm sc cfg ty tm
   96 
   97     neutral cfg env nt =
   98       do env' <- traverse (\(x,ty) -> readBackValue sc cfg ty =<< force x) env
   99          tm   <- instantiateVarList sc 0 env' =<< neutralToSharedTerm sc nt
  100          tyv  <- evalType cfg =<< scTypeOf sc tm
  101          reflectTerm sc cfg tyv tm
  102 
  103     primHandler cfg pn _msg env tp =
  104       do pn'  <- traverse (readBackTValue sc cfg) pn
  105          args <- reverse <$> traverse (\(x,ty) -> readBackValue sc cfg ty =<< force x) env
  106          prim <- scFlatTermF sc (Primitive pn')
  107          f    <- foldM (scApply sc) prim args
  108          reflectTerm sc cfg tp f
  109 
  110 replace ::
  111   (?recordEC :: BoundECRecorder) =>
  112   SharedContext ->
  113   Sim.SimulatorConfig TermModel ->
  114   IORef ReplaceUninterpMap ->
  115   ExtCns (TValue TermModel) ->
  116   IO (Value TermModel)
  117 replace sc cfg mapref ec = loop [] (ecType ec)
  118   where
  119     loop :: [Term] -> TValue TermModel -> IO (Value TermModel)
  120     loop env (VPiType nm xty body) =
  121         return $ VFun nm $ \x ->
  122           do ty <- applyPiBody body x
  123              xtm <- readBackValue sc cfg xty =<< force x
  124              loop (xtm:env) ty
  125 
  126     loop env ty =
  127       do let args = reverse env
  128          ty' <- readBackTValue sc cfg ty
  129          newec <- scFreshEC sc (toShortName (ecName ec)) ty'
  130          modifyIORef mapref (Map.alter (Just . ((newec,args):) . fromMaybe []) (ecVarIndex ec))
  131          reflectTerm sc cfg ty =<< scFlatTermF sc (ExtCns newec)

  133 normalizeSharedTerm ::
  134   SharedContext ->
  135   ModuleMap ->
  136   Map Ident TmPrim {- ^ additional primitives -} ->
  137   Map VarIndex TmValue {- ^ ExtCns values -} ->
  138   Set VarIndex {- ^ opaque constants -} ->
  139   Term ->
  140   IO Term
  141 normalizeSharedTerm sc m addlPrims =
  142   normalizeSharedTerm' sc m (const $ Map.union addlPrims)
  143 
  144 normalizeSharedTerm' ::
  145   SharedContext ->
  146   ModuleMap ->
  147   (Sim.SimulatorConfig TermModel -> Map Ident TmPrim -> Map Ident TmPrim)
  148     {- ^ function which adds additional primitives -} ->
  149   Map VarIndex TmValue {- ^ ExtCns values -} ->
  150   Set VarIndex {- ^ opaque constants -} ->
  151   Term ->
  152   IO Term
  153 normalizeSharedTerm' sc m primsFn ecVals opaqueSet t =
  154   do let ?recordEC = \_ec -> return ()
  155      cfg <- mfix (\cfg -> Sim.evalGlobal' m (primsFn cfg (constMap sc cfg))
  156                               (extcns cfg) (constants cfg) (neutral cfg) (primHandler cfg))
  157      v <- Sim.evalSharedTerm cfg t
  158      tv <- evalType cfg =<< scTypeOf sc t
  159      readBackValue sc cfg tv v
  160 
  161   where
  162     constants cfg tf ec
  163       | Set.member (ecVarIndex ec) opaqueSet = Just $
  164           do let ?recordEC = \_ec -> return ()
  165              tm <- scTermF sc tf
  166              reflectTerm sc cfg (ecType ec) tm
  167 
  168       | otherwise = Nothing
  169 
  170     extcns cfg tf ec =
  171       let ?recordEC = \_ec -> return () in
  172       case Map.lookup (ecVarIndex ec) ecVals of
  173         Just v  -> return v
  174         Nothing ->
  175           do tm <- scTermF sc tf
  176              reflectTerm sc cfg (ecType ec) tm
  177 
  178     neutral cfg env nt =
  179       do let ?recordEC = \_ec -> return ()
  180          env' <- traverse (\(x,ty) -> readBackValue sc cfg ty =<< force x) env
  181          tm   <- instantiateVarList sc 0 env' =<< neutralToSharedTerm sc nt
  182          tyv  <- evalType cfg =<< scTypeOf sc tm
  183          reflectTerm sc cfg tyv tm
  184 
  185     primHandler cfg pn _msg env tp =
  186       do let ?recordEC = \_ec -> return ()
  187          pn'  <- traverse (readBackTValue sc cfg) pn
  188          args <- reverse <$> traverse (\(x,ty) -> readBackValue sc cfg ty =<< force x) env
  189          prim <- scFlatTermF sc (Primitive pn')
  190          f    <- foldM (scApply sc) prim args
  191          reflectTerm sc cfg tp f

  209 data VExtra
  210   = VExtraTerm
  211        !(TValue TermModel) -- type of the term
  212        !Term               -- term value (closed term!)
  213   | VExtraStream
  214        !(TValue TermModel) -- type of the stream elements
  215        !(Thunk TermModel -> MValue TermModel) -- function to compute stream values
  216        !(IORef (Map Natural (Value TermModel))) -- cache of concrete values
  217        !(Lazy IO Term) -- stream value as a term (closed term!)
  218 
  219 instance Show VExtra where
  220   show (VExtraTerm ty tm) = "<extra> " ++ showTerm tm ++ " : " ++ show ty
  221   show (VExtraStream ty _ _ _) = "<stream of " ++ show ty ++ ">"

  232 readBackTValue :: (?recordEC :: BoundECRecorder) =>
  233   SharedContext ->
  234   Sim.SimulatorConfig TermModel ->
  235   TValue TermModel ->
  236   IO Term
  237 readBackTValue sc cfg = loop
  238   where
  239   loop tv =
  240     case tv of
  241       VUnitType -> scUnitType sc
  242       VBoolType -> scBoolType sc
  243       VStringType -> scStringType sc
  244       VIntType -> scIntegerType sc
  245       VSort s  -> scSort sc s
  246       VIntModType m ->
  247         do m' <- scNat sc m
  248            scIntModType sc m'
  249       VArrayType t1 t2 ->
  250         do t1' <- loop t1
  251            t2' <- loop t2
  252            scArrayType sc t1' t2'
  253       VVecType n t ->
  254         do n' <- scNat sc n
  255            t' <- loop t
  256            scVecType sc n' t'
  257       VPairType t1 t2 ->
  258         do t1' <- loop t1
  259            t2' <- loop t2
  260            scPairType sc t1' t2'
  261       VRecordType fs ->
  262         do fs' <- traverse (traverse loop) fs
  263            scRecordType sc fs'
  264       VDataType nm ps vs ->
  265         do nm' <- traverse loop nm
  266            (ps',vs') <- splitAt (length ps) <$> readBackDataTypeParams (primType nm) (ps++vs)
  267            scDataTypeAppParams sc nm' ps' vs'
  268       VPiType{} ->
  269         do (ecs, tm) <- readBackPis tv
  270            scGeneralizeExts sc ecs tm
  271       VRecursorType d ps m mty ->
  272         do d'   <- traverse loop d
  273            ps'  <- readBackDataTypeParams (primType d) ps
  274            m'   <- readBackValue sc cfg mty m
  275            mty' <- loop mty
  276            scFlatTermF sc (RecursorType d' ps' m' mty')
  277       VTyTerm _s tm ->
  278         pure tm
  279 
  280   readBackDataTypeParams (VPiType _nm t body) (v:vs) =
  281     do v' <- readBackValue sc cfg t v
  282        t' <- applyPiBody body (ready v)
  283        vs' <- readBackDataTypeParams t' vs
  284        return (v':vs')
  285 
  286   readBackDataTypeParams _ [] = return []
  287   readBackDataTypeParams ty _ =
  288       panic "readBackTValue / readBackDataTypeParams" [
  289           "Arity mismatch in data type in readback",
  290           "Remaining type: " <> Text.pack (show ty)
  291       ]
  292 
  293   readBackPis (VPiType nm t pibody) =
  294     do t' <- loop t
  295        ec <- scFreshEC sc nm t'
  296        ?recordEC ec
  297        ecTm <- scExtCns sc ec
  298        ecVal <- delay (reflectTerm sc cfg t ecTm)
  299        body <- applyPiBody pibody ecVal
  300        (ecs,body') <- readBackPis body
  301        return (ec:ecs, body')
  302 
  303   readBackPis t =
  304     do tm <- loop t
  305        return ([], tm)

  307 evalType :: Sim.SimulatorConfig TermModel -> Term -> IO (TValue TermModel)
  308 evalType cfg tm =
  309   Sim.evalSharedTerm cfg tm >>= \case
  310     TValue tv -> pure tv
  311     _ -> panic "evalType" ["Expected type value"]

  313 reflectTerm ::
  314   (?recordEC :: BoundECRecorder) =>
  315   SharedContext ->
  316   Sim.SimulatorConfig TermModel ->
  317   TValue TermModel ->
  318   Term {- ^ closed term to reflect -} ->
  319   IO (Value TermModel)
  320 reflectTerm sc cfg = loop
  321   where
  322   loop tv tm = case tv of
  323     VUnitType -> pure VUnit
  324     VBoolType -> return (VBool (Left tm))
  325     VIntType  -> return (VInt (Left tm))
  326     VIntModType m -> return (VIntMod m (Left tm))
  327     VVecType n VBoolType -> return (VWord (Left (n,tm)))
  328     VArrayType a b ->
  329       do a' <- readBackTValue sc cfg a
  330          b' <- readBackTValue sc cfg b
  331          return (VArray (TMArray a a' b b' tm))
  332 
  333     VSort s -> pure (TValue (VTyTerm s tm))
  334 
  335     VPiType nm t pibody ->
  336       return (VFun nm (\x ->
  337         do v <- force x
  338            tbody <- applyPiBody pibody x
  339            xtm <- readBackValue sc cfg t v
  340            body <- scApply sc tm xtm
  341            reflectTerm sc cfg tbody body
  342         ))
  343 
  344     -- TODO: eta-expanding vectors like this kinda sucks.  It would be
  345     -- better if we could treat them abstractly, as we are doing below
  346     -- with records, pairs and data types.  However, that would be
  347     -- a pretty invasive change in the primitives.
  348     VVecType n t ->  -- NB: t /= Bool
  349       do a  <- readBackTValue sc cfg t
  350          n' <- scNat sc n
  351          vs <- V.generateM (fromIntegral n) $ \i -> -- TODO fromIntegral...
  352                    do i' <- scNat sc (fromIntegral i)
  353                       tm' <- scAt sc n' a tm i'
  354                       ready <$> reflectTerm sc cfg t tm'
  355          pure (VVector vs)
  356 
  357     VStringType{}   -> return (VExtra (VExtraTerm tv tm))
  358     VRecordType{}   -> return (VExtra (VExtraTerm tv tm))
  359     VPairType{}     -> return (VExtra (VExtraTerm tv tm))
  360     VDataType{}     -> return (VExtra (VExtraTerm tv tm))
  361     VRecursorType{} -> return (VExtra (VExtraTerm tv tm))
  362     VTyTerm{}       -> return (VExtra (VExtraTerm tv tm))
  363 
  364 -- | Given a value, which must have the given type,
  365 --   reconstruct a closed term representing the value.
  366 readBackValue ::
  367   (?recordEC :: BoundECRecorder) =>
  368   SharedContext ->
  369   Sim.SimulatorConfig TermModel ->
  370   TValue TermModel ->
  371   Value TermModel ->
  372   IO Term
  373 readBackValue sc cfg = loop
  374   where
  375     loop _ VUnit = scUnitValue sc
  376 
  377     loop _ (VNat n) = scNat sc n
  378 
  379     loop _ (VBVToNat w n) =
  380       do tm <- loop (VVecType (fromIntegral w) VBoolType) n
  381          scBvToNat sc (fromIntegral w) tm
  382 
  383     loop _ (VIntToNat i) =
  384       do tm <- loop VIntType i
  385          scIntToNat sc tm
  386 
  387     loop _ (VBool (Left tm)) = return tm
  388     loop _ (VBool (Right b)) = scBool sc b
  389 
  390     loop _ (VInt (Left tm)) = return tm
  391     loop _ (VInt (Right i)) = scIntegerConst sc i
  392 
  393     loop _ (VIntMod _ (Left tm)) = return tm
  394     loop _ (VIntMod m (Right i)) =
  395       do m' <- scNat sc m
  396          i' <- scIntegerConst sc i
  397          scToIntMod sc m' i'
  398 
  399     loop _ (VWord (Left (_,tm))) = return tm
  400     loop _ (VWord (Right bv)) = scBvConst sc (fromIntegral (width bv)) (unsigned bv)
  401 
  402     loop _ (VArray (TMArray _ _ _ _ tm)) = return tm
  403 
  404     loop _ (VString s) = scString sc s
  405 
  406     loop _ (TValue tv) = readBackTValue sc cfg tv
  407 
  408     loop _ (VExtra (VExtraTerm _tp tm)) = return tm
  409     loop _ (VExtra (VExtraStream _tp _fn _ref tm)) = liftIO (force tm)
  410 
  411     loop tv@VPiType{} v@VFun{} =
  412       do (ecs, tm) <- readBackFuns tv v
  413          scAbstractExtsEtaCollapse sc ecs tm
  414 
  415     loop (VPairType t1 t2) (VPair v1 v2) =
  416       do tm1 <- loop t1 =<< force v1
  417          tm2 <- loop t2 =<< force v2
  418          scPairValueReduced sc tm1 tm2
  419 
  420     loop (VVecType _n tp) (VVector vs) =
  421       do tp' <- readBackTValue sc cfg tp
  422          vs' <- traverse (loop tp <=< force) (V.toList vs)
  423          scVectorReduced sc tp' vs'
  424 
  425     loop (VDataType _nm _ps _ixs) (VCtorApp cnm ps vs) =
  426       do cnm' <- traverse (readBackTValue sc cfg) cnm
  427          (ps',vs') <- splitAt (length ps) <$> readBackCtorArgs cnm (primType cnm) (ps++vs)
  428          scCtorAppParams sc cnm' ps' vs'
  429 
  430     loop (VRecordType fs) (VRecordValue vs) =
  431       do let fm = Map.fromList fs
  432          let build (k,v) =
  433                   case Map.lookup k fm of
  434                     Nothing ->
  435                        panic "readBackValue / loop" $ [
  436                            "Field mismatch in record value:",
  437                            "Fields in type:"
  438                          ] ++ fs' ++ [
  439                            "Fields in value:"
  440                          ] ++ vs'
  441                        where showField (f, ty) =
  442                                  "   " <> f <> ": " <> Text.pack (show ty)
  443                              showValue (f, _v) =
  444                                  -- v is a thunk, can't print it
  445                                  -- XXX: we're going to force them all
  446                                  -- anyway, maybe do that as a first
  447                                  -- pass on vs?
  448                                  "   " <> f
  449                              fs' = map showField fs
  450                              vs' = map showValue vs
  451                     Just t ->
  452                        do tm <- loop t =<< force v
  453                           return (k,tm)
  454          vs' <- Map.fromList <$> traverse build vs
  455          scRecord sc vs'
  456 
  457     loop tv v =
  458         panic "readBackValue / loop" [
  459             "Type mismatch",
  460             "Type we have: " <> Text.pack (show tv),
  461             "For value: " <> Text.pack (show v)
  462         ]
  463 
  464     readBackCtorArgs cnm (VPiType _nm tv body) (v:vs) =
  465       do v' <- force v
  466          t  <- loop tv v'
  467          ty <- applyPiBody body (ready v')
  468          ts <- readBackCtorArgs cnm ty vs
  469          pure (t:ts)
  470     readBackCtorArgs _ (VDataType _ _ _) [] = pure []
  471     readBackCtorArgs cnm _ _ =
  472       panic "readBackValue / readBackCtorArgs" [
  473           "Constructor type mismatch: " <> Text.pack (show cnm)
  474       ]
  475 
  476 
  477     readBackFuns (VPiType _ tv pibody) (VFun nm f) =
  478       do t' <- readBackTValue sc cfg tv
  479          ec <- scFreshEC sc nm t'
  480          ?recordEC ec
  481          ecTm <- scExtCns sc ec
  482          ecVal <- delay (reflectTerm sc cfg tv ecTm)
  483          tbody <- applyPiBody pibody ecVal
  484          body  <- f ecVal
  485          (ecs,tm) <- readBackFuns tbody body
  486          return (ec:ecs, tm)
  487     readBackFuns tv v =
  488       do tm <- loop tv v
  489          return ([], tm)

  494 boolTerm :: SharedContext -> VBool TermModel -> IO Term
  495 boolTerm _ (Left tm) = pure tm
  496 boolTerm sc (Right b) = scBool sc b
  497 
  498 wordWidth :: VWord TermModel -> Natural
  499 wordWidth (Left (n,_)) = n
  500 wordWidth (Right bv)   = fromIntegral (Prim.width bv)
  501 
  502 wordTerm :: SharedContext -> VWord TermModel -> IO Term
  503 wordTerm _  (Left (_,tm)) = pure tm
  504 wordTerm sc (Right bv) =
  505   scBvConst sc (fromIntegral (Prim.width bv)) (Prim.unsigned bv)
  506 
  507 wordValue :: SharedContext -> BitVector -> IO (Natural, Term)
  508 wordValue sc bv =
  509   do let n = fromIntegral (Prim.width bv)
  510      tm <- scBvConst sc n (Prim.unsigned bv)
  511      pure (n, tm)

  513 intTerm :: SharedContext -> VInt TermModel -> IO Term
  514 intTerm _ (Left tm) = pure tm
  515 intTerm sc (Right i) = scIntegerConst sc i

  517 unOp ::
  518   SharedContext ->
  519   (SharedContext -> t -> IO t') ->
  520   (a -> b) ->
  521   Either t a -> IO (Either t' b)
  522 unOp sc termOp _valOp (Left t) = Left <$> termOp sc t
  523 unOp _sc _termOp valOp (Right x) = pure . Right $! valOp x
  524 
  525 binOp ::
  526   SharedContext ->
  527   (SharedContext -> a -> IO t) ->
  528   (SharedContext -> t -> t -> IO t') ->
  529   (a -> a -> b) ->
  530   Either t a -> Either t a -> IO (Either t' b)
  531 binOp sc toTerm termOp valOp x y =
  532   case (x, y) of
  533     (Right xv, Right yv) -> pure . Right $! valOp xv yv
  534     (Left xtm, Right yv) ->
  535       do ytm <- toTerm sc yv
  536          Left <$> termOp sc xtm ytm
  537     (Right xv, Left ytm) ->
  538       do xtm <- toTerm sc xv
  539          Left <$> termOp sc xtm ytm
  540     (Left xtm, Left ytm) ->
  541       Left <$> termOp sc xtm ytm

  544 bvUnOp ::
  545   SharedContext ->
  546   (SharedContext -> Term -> Term -> IO Term) ->
  547   (BitVector -> BitVector) ->
  548   VWord TermModel -> IO (VWord TermModel)
  549 bvUnOp sc termOp valOp = \case
  550   Right bv -> pure . Right $! valOp bv
  551   Left (n,tm) ->
  552     do n' <- scNat sc n
  553        tm' <- termOp sc n' tm
  554        pure (Left (n,tm'))
  555

  557 bvBinOp ::
  558   SharedContext ->
  559   (SharedContext -> Term -> Term -> Term -> IO Term) ->
  560   (BitVector -> BitVector -> BitVector) ->
  561   VWord TermModel -> VWord TermModel -> IO (VWord TermModel)
  562 bvBinOp sc termOp valOp = binOp sc wordValue termOp' valOp
  563   where
  564     termOp' _ (n,x) (_,y) =
  565       do n' <- scNat sc n
  566          tm <- termOp sc n' x y
  567          pure (n, tm)

  570 intDivOp ::
  571   SharedContext ->
  572   (SharedContext -> Term -> Term -> IO Term) ->
  573   (Integer -> Integer -> Integer) ->
  574   VInt TermModel -> VInt TermModel -> IO (VInt TermModel)
  575 
  576 -- Special case for concrete divide-by-0:
  577 -- return the term instead of producing an error
  578 intDivOp sc termOp _intOp x y@(Right 0) =
  579   do x' <- intTerm sc x
  580      y' <- intTerm sc y
  581      Left <$> termOp sc x' y'
  582 intDivOp sc termOp intOp x y =
  583   binOp sc scIntegerConst termOp intOp x y

  586 bvDivOp ::
  587   SharedContext ->
  588   (SharedContext -> Term -> Term -> Term -> IO Term) ->
  589   (BitVector -> BitVector -> Maybe BitVector) ->
  590   VWord TermModel -> VWord TermModel -> IO (VWord TermModel)
  591 bvDivOp sc termOp valOp x y = fixup =<< binOp sc wordValue termOp' valOp x y
  592   where
  593     -- Special case for concrete divide-by-0:
  594     -- return the term instead of producing an error
  595     fixup (Left l) = pure (Left l)
  596     fixup (Right (Just bv)) = pure (Right bv)
  597     fixup (Right Nothing) =
  598       do let n = wordWidth x
  599          n' <- scNat sc n
  600          xtm <- wordTerm sc x
  601          ytm <- wordTerm sc y
  602          tm <- termOp sc n' xtm ytm
  603          pure (Left (n,tm))
  604 
  605     termOp' _ (n,xtm) (_,ytm) =
  606       do n' <- scNat sc n
  607          tm <- termOp sc n' xtm ytm
  608          pure (n, tm)

  610 bvCmpOp ::
  611   SharedContext ->
  612   (SharedContext -> Term -> Term -> Term -> IO Term) ->
  613   (BitVector -> BitVector -> Bool) ->
  614   VWord TermModel -> VWord TermModel -> IO (VBool TermModel)
  615 bvCmpOp sc termOp valOp = binOp sc wordValue termOp' valOp
  616   where
  617     termOp' _ (n,x) (_,y) =
  618       do n' <- scNat sc n
  619          termOp sc n' x y

  632        Right bv -> pure (fmap Right (Prim.unpackBitVector bv))

  634        Left (n,tm) ->

  636                  a  <- scBoolType sc

  670                   tm  <- scAppend sc xn' yn' a xtm ytm

 1078 -- | (n : Nat) -> Vec n Bool -> Nat -> Vec n Bool
 1079 bvShiftOp ::
 1080   (?recordEC :: BoundECRecorder) =>
 1081   SharedContext ->
 1082   Sim.SimulatorConfig TermModel ->
 1083   (Natural -> Natural) ->
 1084   (SharedContext -> Term -> Term -> Term -> IO Term) ->
 1085   (BitVector -> Int -> BitVector) ->
 1086   TmPrim
 1087 bvShiftOp sc cfg szf tmOp bvOp =
 1088   Prims.natFun $ \n0 ->
 1089   Prims.strictFun $ \w ->
 1090   Prims.strictFun $ \amt ->
 1091   Prims.Prim $
 1092     case (w, amt) of
 1093       (VWord (Right w'), VNat amt') ->
 1094         let amt'' = fromInteger (min (toInteger (Prim.width w')) (toInteger amt'))
 1095          in pure . VWord . Right $! bvOp w' amt''
 1096       _ ->
 1097         do let n = szf n0
 1098            n0'  <- scNat sc n0
