   71 -- | The direction of a module port.
   72 data Direction
   73   = DirectionInput
   74   | DirectionOutput
   75   | DirectionInout
   76   deriving (Show, Eq, Ord)
   77 
   78 isInput :: Direction -> Bool
   79 isInput d =
   80   case d of
   81     DirectionInput -> True
   82     DirectionOutput -> False
   83     DirectionInout -> True
   84 
   85 isOutput :: Direction -> Bool
   86 isOutput d =
   87   case d of
   88     DirectionInput -> False
   89     DirectionOutput -> True
   90     DirectionInout -> True
   91 
   92 instance Aeson.FromJSON Direction where
   93   parseJSON (Aeson.String "input") = pure DirectionInput
   94   parseJSON (Aeson.String "output") = pure DirectionOutput
   95   parseJSON (Aeson.String "inout") = pure DirectionInout
   96   parseJSON v = fail $ "Failed to parse direction: " <> show v

   98 -- | The value of a connection
   99 data Bitrep
  100   = BitrepZero -- ^ Constant zero bit
  101   | BitrepOne -- ^ Constant one bit
  102   | BitrepX -- ^ Undefined bit X
  103   | BitrepZ -- ^ Undefined bit Z
  104   | Bitrep Integer -- ^ The signal bit with the given index
  105   deriving (Show, Eq, Ord)
  106 
  107 instance Aeson.FromJSON Bitrep where
  108   parseJSON (Aeson.String "0") = pure BitrepZero
  109   parseJSON (Aeson.String "1") = pure BitrepOne
  110   parseJSON (Aeson.String "x") = pure BitrepX
  111   parseJSON (Aeson.String "z") = pure BitrepZ
  112   parseJSON vn@Aeson.Number{} = Bitrep <$> Aeson.parseJSON vn
  113   parseJSON v = fail $ "Failed to parse bits: " <> show v

  218 -- | Indicates whether a primitive register cell is a variant with a
  219 -- clock-enable input port.
  220 data ClockEnable
  221   = WithoutClockEnable
  222   | WithClockEnable
  223   deriving (Eq, Ord)
  224 
  225 -- | All supported primitive register cell types.
  226 -- All register cell types have a numeric parameter WIDTH and a single
  227 -- output port Q of size WIDTH.
  228 -- See the Yosys documentation for the cell definitions:
  229 -- <https://yosyshq.readthedocs.io/projects/yosys/en/latest/cell/word_reg.html>
  230 data CellTypeRegister
  231   = CellTypeAdff ClockEnable
  232     -- ^ D-type flip-flop with asynchronous reset (@$adff@).
  233     -- Also a variant with clock-enable (@$adffe@).
  234   | CellTypeAldff ClockEnable
  235     -- ^ D-type flip-flop with asynchronous load (@$aldff@).
  236     -- Also a variant with clock-enable (@$aldffe@).
  237   | CellTypeDff ClockEnable
  238     -- ^ D-type flip-flop (@$dff@).
  239     -- Also a variant with clock-enable (@$dffe@).
  240   | CellTypeDffsr ClockEnable -- ^ 'True' for @$dffsre@, 'False' for  @$dffsr@
  241     -- ^ D-type flip-flop with asynchronous per-bit set and reset (@$dffsr@).
  242     -- Reset takes precedence when both set and reset are active.
  243     -- Also a variant with clock-enable (@$dffsre@).
  244   | CellTypeFf
  245     -- ^ Flip-flop with implicit global clock (@$ff@).
  246     -- <https://yosyshq.readthedocs.io/projects/yosys/en/latest/cell/word_formal.html#formal.$ff>
  247   | CellTypeSdff ClockEnable
  248     -- ^ D-type flip-flop with synchronous reset (@$sdff@).
  249     -- Also a variant with clock-enable (@$sdffce@).
  250     -- NOTE: Unlike @$sdffe@, @$sdffce@ requires an active
  251     -- clock-enable signal for the synchronous reset signal to have
  252     -- any effect.
  253   | CellTypeSdffe
  254     -- ^ D-type flip-flop with synchronous reset and clock-enable
  255     -- (@$sdffe@).
  256     -- NOTE: Unlike @$sdffce$, @$sdffe@ allows the synchronous reset
  257     -- signal to reset the register even on a clock edge when the
  258     -- clock-enable is inactive.
  259   | CellTypeDlatch
  260     -- ^ D-type latch (@$dlatch@), transparent while EN input is active.
  261   | CellTypeAdlatch
  262     -- ^ D-type latch with asynchronous reset (@$adlatch@).
  263   | CellTypeDlatchsr
  264     -- ^ D-type latch with asynchronous per-bit set and reset (@$dlatchsr@).
  265     -- Reset takes precedence when both set and reset are active.
  266   | CellTypeSr
  267     -- ^ SR-type latch with asynchronous per-bit set and reset (@$sr@).
  268     -- Reset takes precedence when both set and reset are active.
  269   deriving (Eq, Ord)
  270 
  271 -- | All supported cell types.
  272 -- All types are primitives except for 'CellTypeUserType' which
  273 -- represents user-defined submodules.
  274 -- Invariants: 'CellTypeUnsupportedPrimitive' should only be applied
  275 -- to names satisfying 'cellTypeIsPrimitive', and 'CellTypeUserType'
  276 -- should only be applied to names *not* satisfying
  277 -- 'cellTypeIsPrimitive'.
  278 data CellType
  279   = CellTypeCombinational CellTypeCombinational
  280   | CellTypeRegister CellTypeRegister
  281   | CellTypeUnsupportedPrimitive CellTypeName
  282   | CellTypeUserType CellTypeName
  283   deriving (Eq, Ord)
  284 
  285 instance Aeson.FromJSON CellType where
  286   parseJSON (Aeson.String s) =
  287     case s of
  288       _ | cellTypeIsPrimitive s ->
  289           case Map.lookup s textToPrimitiveCellType of
  290             Just cellType -> pure cellType
  291             -- XXX We should probably log a warning when generating CellTypeUnsupportedPrimitive,
  292             -- we can't do that here however.
  293             Nothing -> pure $ CellTypeUnsupportedPrimitive s
  294         | otherwise -> pure $ CellTypeUserType s
  295   parseJSON v = fail $ "Failed to parse cell type: " <> show v

  297 ppCellTypeCombinational :: CellTypeCombinational -> Text
  298 ppCellTypeCombinational ctc =
  299   case ctc of
  300     CellTypeNot        -> "$not"
  301     CellTypePos        -> "$pos"
  302     CellTypeNeg        -> "$neg"
  303     CellTypeAnd        -> "$and"
  304     CellTypeOr         -> "$or"
  305     CellTypeXor        -> "$xor"
  306     CellTypeXnor       -> "$xnor"
  307     CellTypeReduceAnd  -> "$reduce_and"
  308     CellTypeReduceOr   -> "$reduce_or"
  309     CellTypeReduceXor  -> "$reduce_xor"
  310     CellTypeReduceXnor -> "$reduce_xnor"
  311     CellTypeReduceBool -> "$reduce_bool"
  312     CellTypeShl        -> "$shl"
  313     CellTypeShr        -> "$shr"
  314     CellTypeSshl       -> "$sshl"
  315     CellTypeSshr       -> "$sshr"
  316     CellTypeShift      -> "$shift"
  317     CellTypeShiftx     -> "$shiftx"
  318     CellTypeLt         -> "$lt"
  319     CellTypeLe         -> "$le"
  320     CellTypeGt         -> "$gt"
  321     CellTypeGe         -> "$ge"
  322     CellTypeEq         -> "$eq"
  323     CellTypeNe         -> "$ne"
  324     CellTypeEqx        -> "$eqx"
  325     CellTypeNex        -> "$nex"
  326     CellTypeAdd        -> "$add"
  327     CellTypeSub        -> "$sub"
  328     CellTypeMul        -> "$mul"
  329     CellTypeDiv        -> "$div"
  330     CellTypeMod        -> "$mod"
  331     CellTypeLogicNot   -> "$logic_not"
  332     CellTypeLogicAnd   -> "$logic_and"
  333     CellTypeLogicOr    -> "$logic_or"
  334     CellTypeMux        -> "$mux"
  335     CellTypePmux       -> "$pmux"
  336     CellTypeBmux       -> "$bmux"
  337     CellTypeBUF        -> "$_BUF_"
  338 
  339 instance Show CellTypeCombinational where
  340   show ctc = Text.unpack (ppCellTypeCombinational ctc)
  341 
  342 ppCellTypeRegister :: CellTypeRegister -> Text
  343 ppCellTypeRegister ctr =
  344   case ctr of
  345     CellTypeAdff e   -> ceCases e "$adff"  "$adffe"
  346     CellTypeAldff e  -> ceCases e "$aldff" "$aldffe"
  347     CellTypeDff e    -> ceCases e "$dff"   "$dffe"
  348     CellTypeDffsr e  -> ceCases e "$dffsr" "$dffsre"
  349     CellTypeFf       -> "$ff"
  350     CellTypeSdff e   -> ceCases e "$sdff"  "$sdffce"
  351     CellTypeSdffe    -> "$sdffe"
  352     CellTypeDlatch   -> "$dlatch"
  353     CellTypeAdlatch  -> "$adlatch"
  354     CellTypeDlatchsr -> "$dlatchsr"
  355     CellTypeSr       -> "$sr"
  356   where
  357     ceCases :: ClockEnable -> Text -> Text -> Text
  358     ceCases WithoutClockEnable x _ = x
  359     ceCases WithClockEnable _ y = y
  360 
  361 ppCellType :: CellType -> Text
  362 ppCellType ct =
  363   case ct of
  364     CellTypeCombinational ctc -> ppCellTypeCombinational ctc
  365     CellTypeRegister ctr -> ppCellTypeRegister ctr
  366     CellTypeUnsupportedPrimitive t -> t
  367     CellTypeUserType t -> t
  368 
  369 instance Show CellType where
  370   show ct = Text.unpack (ppCellType ct)

  372 -- | A cell within an HDL module.
  373 data Cell =
  374   Cell
  375   { _cellHideName :: Bool -- ^ Whether the cell's name is human-readable (default: False)
  376   , _cellType :: CellType -- ^ The cell type
  377     -- NB: Yosys's documentation for write_json doesn't impose any restrictions
  378     -- on what type parameter values may take on, so we opt to be as permissive
  379     -- as possible by using Aeson Values. Most of the time, these Values will
  380     -- be strings, but they can also be numbers on occasion (e.g., if you call
  381     -- write_json using the -compat-int flag).
  382   , _cellParameters :: Map Text Aeson.Value -- ^ Metadata parameters
  383   , _cellAttributes :: Maybe Aeson.Value -- currently unused
  384   , _cellPortDirections :: Map PortName Direction -- ^ Direction for each cell connection
  385   , _cellConnections :: Map PortName [Bitrep] -- ^ Bitrep for each cell connection
  386   }
  387   deriving (Show, Eq, Ord)
  388 
  389 makeLenses ''Cell
  390 
  391 instance Aeson.FromJSON Cell where
  392   parseJSON = Aeson.withObject "cell" $ \o -> do
  393     _cellHideName <- Maybe.maybe False (/= (0::Int)) <$> o Aeson..:? "hide_name"
  394     _cellType <- o Aeson..: "type"
  395     _cellParameters <- o Aeson..: "parameters"
  396     _cellAttributes <- o Aeson..:? "attributes"
  397     _cellPortDirections <- o Aeson..: "port_directions"
  398     _cellConnections <- o Aeson..: "connections"
  399     pure Cell{..}

  454 -- | Read a collection of HDL modules from a file produced by Yosys' write_json command.
  455 loadYosysIR :: FilePath -> IO YosysIR
  456 loadYosysIR p = Aeson.eitherDecodeFileStrict p >>= \case
  457   Left err -> yosysError $ YosysError $ Text.pack err
  458   Right ir -> pure ir

  460 -- | Return the patterns for all of the input ports of a module
  461 moduleInputPorts :: Module -> Map PortName [Bitrep]
  462 moduleInputPorts m =
  463   Map.mapMaybe
  464   ( \ip ->
  465       if ip ^. portDirection == DirectionInput || ip ^. portDirection == DirectionInout
  466       then Just (ip ^. portBits)
  467       else Nothing
  468   )
  469   $ m ^. modulePorts
  470 
  471 -- | Return the patterns for all of the output ports of a module
  472 moduleOutputPorts :: Module -> Map PortName [Bitrep]
  473 moduleOutputPorts m =
  474   Map.mapMaybe
  475   ( \ip ->
  476       if isOutput (ip ^. portDirection)
  477       then Just (ip ^. portBits)
  478       else Nothing
  479   )
  480   $ m ^. modulePorts
  481 
  482 -- | Return the patterns for all of the input connections of a cell
  483 cellInputConnections :: Cell -> Map PortName [Bitrep]
  484 cellInputConnections c = Map.intersection (c ^. cellConnections) inp
  485   where
  486     inp = Map.filter isInput (c ^. cellPortDirections)
  487 
  488 -- | Return the patterns for all of the output connections of a cell
  489 cellOutputConnections :: Cell -> Map PortName [Bitrep]
  490 cellOutputConnections c = Map.intersection (c ^. cellConnections) out
  491   where
  492     out = Map.filter isOutput (c ^. cellPortDirections)

  494 -- | Test whether a 'Cell' is a stateful register, such as a DFF.
  495 cellIsRegister :: Cell -> Bool
  496 cellIsRegister c =
  497   case c ^. cellType of
  498     CellTypeRegister _ -> True
  499     _ -> False

  501 -- | Swap out machine-generated names of register cells for
  502 -- user-provided names from the netnames section of the module,
  503 -- wherever possible.
  504 -- If no suitable name exists in the netnames table, then use function
  505 -- 'cellIdentifier' to produce a lexically-valid field name.
  506 renameRegisterInstances :: Module -> Module
  507 renameRegisterInstances m = set moduleCells cells' m
  508   where
  509     cells' :: Map CellInstName Cell
  510     cells' =
  511       Map.fromList $
  512       map (\(t, c) -> (bestName t c, c)) $
  513       Map.toList (m ^. moduleCells)
  514 
  515     netnames :: Map [Bitrep] CellInstName
  516     netnames =
  517       Map.fromList
  518       [ (n ^. netnameBits, t)
  519       | (t, n) <- Map.assocs (m ^. moduleNetnames), not (n ^. netnameHideName) ]
  520 
  521     bestName :: CellInstName -> Cell -> CellInstName
  522     bestName t c
  523       | cellIsRegister c =
  524           Maybe.fromMaybe (cellIdentifier t) $
  525           do bs <- Map.lookup "Q" (c ^. cellConnections)
  526              Map.lookup bs netnames
  527       | otherwise =
  528           t

  530 -- | If a 'Module' has an @attributes@ section with an @hdlname@
  531 -- entry, then use it to produce a suitably nice 'CellTypeName'.
  532 betterModuleName :: Module -> Maybe CellTypeName
  533 betterModuleName m =
  534   do basename <- m ^. moduleHdlName
  535      let params = Map.elems (m ^. moduleParameters)
  536      pure (Text.intercalate "_" (basename : map (Text.pack . show) params))
  537 
  538 -- | Replace machine-generated names of modules with simpler names
  539 -- based on user-provided names and parameters (if any).
  540 -- If no suitable name exists, then use function 'cellIdentifier' to
  541 -- produce a lexically-valid module name.
  542 renameCellTypeNames :: YosysIR -> YosysIR
  543 renameCellTypeNames ir = renameYosysIR renaming ir
  544   where
  545     newName :: CellTypeName -> Module -> CellTypeName
  546     newName name m =
  547       case betterModuleName m of
  548         Just name' -> name'
  549         Nothing -> cellIdentifier name
  550 
  551     renaming :: Map CellTypeName CellTypeName
  552     renaming = Map.mapWithKey newName (ir ^. yosysModules)
  553 
  554 -- | Apply a substitution to the 'CellTypeName's in the given
  555 -- 'YosysIR'.
  556 renameYosysIR :: Map CellTypeName CellTypeName -> YosysIR -> YosysIR
  557 renameYosysIR sub ir = set yosysModules newModules ir
  558   where
  559     newModules =
  560       Map.fromList
  561       [ (Maybe.fromMaybe name (Map.lookup name sub), renameModule m)
  562       | (name, m) <- Map.toList (ir ^. yosysModules) ]
  563 
  564     renameModule :: Module -> Module
  565     renameModule m = over moduleCells (fmap renameCell) m
  566 
  567     renameCell :: Cell -> Cell
  568     renameCell c = over cellType renameCellType c
  569 
  570     renameCellType :: CellType -> CellType
  571     renameCellType ct =
  572       case ct of
  573         CellTypeCombinational _ -> ct
  574         CellTypeRegister _ -> ct
  575         CellTypeUserType name ->
  576           maybe ct CellTypeUserType (Map.lookup name sub)
  577         CellTypeUnsupportedPrimitive name ->
  578           maybe ct CellTypeUserType (Map.lookup name sub)
