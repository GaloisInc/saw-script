  102 --------------------------------------------------------------------------------
  103 -- ** SetupValue
  104 
  105 -- The following type families are extension fields, which describe what
  106 -- SetupValues are legal for which language backends. Instances for these type
  107 -- families adhere to the following conventions:
  108 --
  109 -- - If a SetupValue constructor is used in a particular backend and there is
  110 --   some data that is /only/ used by that backend, then the corresponding
  111 --   extension field should have a @type instance@ that evaluates to the type of
  112 --   that data. For instance, the @XSetupCast@ instance for @LLVM@ might
  113 --   evaluate to an LLVM @Type@ to represent the type to cast to.
  114 --
  115 -- - If a SetupValue constructor is used in a particular backend but there is
  116 --   no backend-specific information, then the corresponding extension field
  117 --   should have a @type instance@ that evaluates to @()@.
  118 --
  119 -- - If a SetupValue constructor is not used in a particular backend, its
  120 --   corresponding extension field should have a @type instance@ that evaluates
  121 --   to Void. Any code that pattern matches on the constructor should then
  122 --   dispatch on the Void value using the 'absurd' function.
  123 
  124 type family XSetupNull ext
  125 type family XSetupStruct ext
  126 type family XSetupEnum ext
  127 type family XSetupTuple ext
  128 type family XSetupSlice ext
  129 type family XSetupArray ext
  130 type family XSetupElem ext
  131 type family XSetupField ext
  132 type family XSetupGlobal ext
  133 type family XSetupCast ext
  134 type family XSetupUnion ext
  135 type family XSetupGlobalInitializer ext
  136 type family XSetupMux ext
  137 
  138 -- | From the manual: \"The SetupValue type corresponds to values that can occur
  139 -- during symbolic execution, which includes both 'Term' values, pointers, and
  140 -- composite types consisting of either of these (both structures and arrays).\"
  141 data SetupValue ext where
  142   SetupVar    :: AllocIndex -> SetupValue ext
  143   SetupTerm   :: TypedTerm -> SetupValue ext
  144   SetupNull   :: XSetupNull ext -> SetupValue ext
  145   SetupStruct :: XSetupStruct ext -> [SetupValue ext] -> SetupValue ext
  146   SetupArray  :: XSetupArray ext -> [SetupValue ext] -> SetupValue ext
  147   SetupElem   :: XSetupElem ext -> SetupValue ext -> Int -> SetupValue ext
  148   SetupField  :: XSetupField ext -> SetupValue ext -> String -> SetupValue ext
  149   SetupCast   :: XSetupCast ext -> SetupValue ext -> SetupValue ext
  150   SetupUnion  :: XSetupUnion ext -> SetupValue ext -> String -> SetupValue ext
  151 
  152   -- | A tuple value. At the moment, this is only ever used for MIR
  153   -- verification.
  154   SetupTuple :: XSetupTuple ext -> [SetupValue ext] -> SetupValue ext
  155   -- | A slice value. At the moment, this is only ever used for MIR
  156   -- verification.
  157   SetupSlice :: XSetupSlice ext -> SetupValue ext
  158   -- | An enumeration value. At the moment, this is only ever used for MIR
  159   -- verification.
  160   SetupEnum :: XSetupEnum ext -> SetupValue ext
  161 
  162   -- | A pointer to a global variable
  163   SetupGlobal :: XSetupGlobal ext -> String -> SetupValue ext
  164   -- | This represents the value of a global's initializer.
  165   SetupGlobalInitializer ::
  166     XSetupGlobalInitializer ext -> String -> SetupValue ext
  167   -- | A mux of two 'SetupValue's based on a boolean-typed 'TypedTerm'.
  168   SetupMux ::
  169     XSetupMux ext ->
  170     TypedTerm
  171       {- ^ The conditional expression. Must be of Cryptol type @Bit@. -} ->
  172     SetupValue ext
  173       {- ^ The value to use if the conditional expression is true. -} ->
  174     SetupValue ext
  175       {- ^ The value to use if the conditional expression is false. -} ->
  176     SetupValue ext
  177 
  178 -- | This constraint can be solved for any ext so long as '()', 'Void', and any
  179 --   other types used in the right-hand sides of extension field
  180 --   @type instance@s have the constraint. Unfortunately, GHC can't (yet?)
  181 --   reason over the equations in our closed type family and realize that, so we
  182 --   must explicitly abstract this reasoning out into a type synonym.
  183 type SetupValueHas (c :: Type -> Constraint) ext =
  184   ( c (XSetupNull ext)
  185   , c (XSetupStruct ext)
  186   , c (XSetupEnum ext)
  187   , c (XSetupTuple ext)
  188   , c (XSetupSlice ext)
  189   , c (XSetupArray ext)
  190   , c (XSetupElem ext)
  191   , c (XSetupField ext)
  192   , c (XSetupCast ext)
  193   , c (XSetupUnion ext)
  194   , c (XSetupGlobal ext)
  195   , c (XSetupGlobalInitializer ext)
  196   , c (XSetupMux ext)
  197   )
  198 
  199 deriving instance (SetupValueHas Show ext) => Show (SetupValue ext)
  200 
  201 -- TypedTerm is neither Eq nor Ord
  202 -- deriving instance (SetupValueHas Eq ext) => Eq (SetupValue ext)
  203 -- deriving instance (SetupValueHas Ord ext) => Ord (SetupValue ext)
