   21 basic_ss :: SharedContext -> IO (Simpset a)
   22 basic_ss sc =
   23   do rs1 <- concat <$> traverse (defRewrites sc) defs
   24      rs2 <- scEqsRewriteRules sc eqs
   25      return $ addConvs procs (addRules (rs1 ++ rs2) emptySimpset)
   26      where
   27        eqs = map (mkIdent preludeName)
   28          [ "unsafeCoerce_same"
   29          , "not_not"
   30          , "true_implies"
   31          , "implies__eq"
   32          , "and_True1"
   33          , "and_False1"
   34          , "and_True2"
   35          , "and_False2"
   36          , "and_idem"
   37          , "or_True1"
   38          , "or_False1"
   39          , "or_True2"
   40          , "or_False2"
   41          , "or_idem"
   42          , "not_True"
   43          , "not_False"
   44          , "not_or"
   45          , "not_and"
   46          , "ite_true"
   47          , "ite_false"
   48          , "ite_not"
   49          , "ite_nest1"
   50          , "ite_nest2"
   51          , "ite_fold_not"
   52          , "ite_eq"
   53          , "ite_true"
   54          , "ite_false"
   55          , "or_triv1"
   56          , "and_triv1"
   57          , "or_triv2"
   58          , "and_triv2"
   59          , "bvAddZeroL"
   60          , "bvAddZeroR"
   61          , "bveq_sameL"
   62          , "bveq_sameR"
   63          , "bveq_same2"
   64          , "bvNat_bvToNat"
   65          ]
   66        defs = map (mkIdent (mkModuleName ["Cryptol"])) ["seq", "ecEq", "ecNotEq"]
   67        procs = [tupleConversion, recordConversion] ++
   68                bvConversions ++ natConversions ++ vecConversions

   72 defRewrites :: SharedContext -> Ident -> IO [RewriteRule a]
   73 defRewrites sc ident =
   74   scFindDef sc ident >>= \maybe_def ->
   75   case maybe_def of
   76     Nothing -> return []
   77     Just def -> scDefRewriteRules sc def
