  193 prettyPointsTo :: LLVMPointsTo arch -> PPL.Doc ann
  194 prettyPointsTo (LLVMPointsTo _md cond ptr val) =
  195   MS.prettySetupValue ptr
  196   PPL.<+> PPL.pretty "points to"
  197   PPL.<+> PPL.pretty val
  198   PPL.<+> maybe PPL.emptyDoc (\tt -> PPL.pretty "if" PPL.<+> prettyTypedTermPure tt) cond
  199 prettyPointsTo (LLVMPointsToBitfield _md ptr fieldName val) =
  200   MS.prettySetupValue ptr <> PPL.pretty ("." ++ fieldName)
  201   PPL.<+> PPL.pretty "points to (bitfield)"
  202   PPL.<+> MS.prettySetupValue val
  203 
  204 instance PPL.Pretty (LLVMPointsTo arch) where
  205   pretty = prettyPointsTo

  228 data SetupError
  229   = InvalidReturnType L.Type
  230   | InvalidArgTypes [L.Type]
  231 
  232 prettySetupError :: SetupError -> PPL.Doc ann
  233 prettySetupError (InvalidReturnType t) =
  234   PPL.pretty "Can't lift return type" PPL.<+>
  235   PPL.viaShow (Crucible.LLVM.ppType t) PPL.<+>
  236   PPL.pretty "to a Crucible type."
  237 prettySetupError (InvalidArgTypes ts) =
  238   PPL.pretty "Can't lift argument types " PPL.<+>
  239   PPL.encloseSep PPL.lparen PPL.rparen PPL.comma (map (PPL.viaShow . Crucible.LLVM.ppType) ts) PPL.<+>
  240   PPL.pretty "to Crucible types."

  242 resolveArgs ::
  243   (?lc :: CL.TypeContext) =>
  244   [L.Type] ->
  245   Either SetupError [CL.MemType]
  246 resolveArgs args = do
  247   -- TODO: make sure we resolve aliases
  248   let mtys = traverse CL.liftMemType args
  249   -- TODO: should the error message be propagated?
  250   either (\_ -> Left (InvalidArgTypes args)) Right mtys

  252 resolveRetTy ::
  253   (?lc :: CL.TypeContext) =>
  254   L.Type ->
  255   Either SetupError (Maybe CL.MemType)
  256 resolveRetTy ty = do
  257   -- TODO: make sure we resolve aliases
  258   let ret = CL.liftRetType ty
  259   -- TODO: should the error message be propagated?
  260   either (\_ -> Left (InvalidReturnType ty)) Right ret

  262 initialDefCrucibleMethodSpecIR ::
  263   (?lc :: CL.TypeContext) =>
  264   LLVMModule arch ->
  265   L.Define ->
  266   ProgramLoc ->
  267   Maybe Text ->
  268   Either SetupError (MS.CrucibleMethodSpecIR (LLVM arch))
  269 initialDefCrucibleMethodSpecIR llvmModule def loc parent = do
  270   args <- resolveArgs (L.typedType <$> L.defArgs def)
  271   ret <- resolveRetTy (L.defRetType def)
  272   let L.Symbol nm = L.defName def
  273   let methId = LLVMMethodId (Text.pack nm) parent
  274   return $ MS.makeCrucibleMethodSpecIR methId args ret loc llvmModule
  275 
  276 initialDeclCrucibleMethodSpecIR ::
  277   (?lc :: CL.TypeContext) =>
  278   LLVMModule arch ->
  279   L.Declare ->
  280   ProgramLoc ->
  281   Maybe Text ->
  282   Either SetupError (MS.CrucibleMethodSpecIR (LLVM arch))
  283 initialDeclCrucibleMethodSpecIR llvmModule dec loc parent = do
  284   args <- resolveArgs (L.decArgs dec)
  285   ret <- resolveRetTy (L.decRetType dec)
  286   let L.Symbol nm = L.decName dec
  287   let methId = LLVMMethodId (Text.pack nm) parent
  288   return $ MS.makeCrucibleMethodSpecIR methId args ret loc llvmModule

  316 -- *** AllLLVM
  317 
  318 -- | Universal/polymorphic quantification over an 'LLVMArch'
  319 --
  320 -- The following type synonym and associated constructor/destructor are
  321 -- equivalent to this definition:
  322 -- @
  323 -- data AllLLVM t =
  324 --   MkAllLLVM { getAllLLVM :: forall arch. t (LLVM arch) }
  325 -- @
  326 -- But they preserve the instances from 'All' and 'Compose'.
  327 type AllLLVM t = All (Compose t LLVM)
  328 
  329 -- This doesn't work :(
  330 --
  331 -- pattern AllLLVM :: (forall arch. t (LLVM arch)) -> AllLLVM t
  332 -- pattern AllLLVM x = All (Compose x)
  333 
  334 mkAllLLVM :: forall t. (forall arch. t (LLVM arch)) -> AllLLVM t
  335 mkAllLLVM x = All (Compose x)
  336 
  337 getAllLLVM :: forall t. AllLLVM t -> (forall arch. t (LLVM arch))
  338 getAllLLVM (All (Compose x)) = x
  339 
  340 -- Constructors for 'SetupValue' which are architecture-polymorphic
  341 
  342 anySetupTerm :: TypedTerm -> AllLLVM MS.SetupValue
  343 anySetupTerm typedTerm = mkAllLLVM (MS.SetupTerm typedTerm)
  344 
  345 anySetupArray :: [AllLLVM MS.SetupValue] -> AllLLVM MS.SetupValue
  346 anySetupArray vals = mkAllLLVM (MS.SetupArray () $ map (\a -> getAllLLVM a) vals)
  347 
  348 anySetupStruct :: Bool -> [AllLLVM MS.SetupValue] -> AllLLVM MS.SetupValue
  349 anySetupStruct b vals = mkAllLLVM (MS.SetupStruct b $ map (\a -> getAllLLVM a) vals)
  350 
  351 anySetupElem :: AllLLVM MS.SetupValue -> Int -> AllLLVM MS.SetupValue
  352 anySetupElem val idx = mkAllLLVM (MS.SetupElem () (getAllLLVM val) idx)
  353 
  354 anySetupCast :: AllLLVM MS.SetupValue -> L.Type -> AllLLVM MS.SetupValue
  355 anySetupCast val ty = mkAllLLVM (MS.SetupCast ty (getAllLLVM val))
  356 
  357 anySetupField :: AllLLVM MS.SetupValue -> Text -> AllLLVM MS.SetupValue
  358 anySetupField val field = mkAllLLVM (MS.SetupField () (getAllLLVM val) field)
  359 
  360 anySetupUnion :: AllLLVM MS.SetupValue -> Text -> AllLLVM MS.SetupValue
  361 anySetupUnion val uname = mkAllLLVM (MS.SetupUnion () (getAllLLVM val) uname)
  362 
  363 anySetupNull :: AllLLVM MS.SetupValue
  364 anySetupNull = mkAllLLVM (MS.SetupNull ())
  365 
  366 anySetupGlobal :: Text -> AllLLVM MS.SetupValue
  367 anySetupGlobal globalName = mkAllLLVM (MS.SetupGlobal () globalName)
  368 
  369 anySetupGlobalInitializer :: Text -> AllLLVM MS.SetupValue
  370 anySetupGlobalInitializer globalName =
  371   mkAllLLVM (MS.SetupGlobalInitializer () globalName)

  373 --------------------------------------------------------------------------------
  374 -- *** SomeLLVM
  375 
  376 -- | Existential quantification over an 'LLVMArch'
  377 --
  378 -- The following type synonym and associated constructor/destructor are
  379 -- equivalent to this definition:
  380 -- @
  381 -- data SomeLLVM t = forall arch. MkSomeLLVM (t (LLVM arch))
  382 -- @
  383 -- But they preserve the instances from 'Some' and 'Compose'.
  384 type SomeLLVM t = Some (Compose t LLVM)
  385 
  386 pattern SomeLLVM :: t (LLVM arch) -> SomeLLVM t
  387 pattern SomeLLVM x = Some (Compose x)
  388 {-# COMPLETE SomeLLVM #-}
  389 
  390 mkSomeLLVM :: t (LLVM arch) -> SomeLLVM t
  391 mkSomeLLVM x = Some (Compose x)
  392 
  393 getSomeLLVM :: forall t. (forall arch. t (LLVM arch)) -> AllLLVM t
  394 getSomeLLVM x = All (Compose x)

  397 -- *** ResolvedState
  398 
  399 -- | Record the initialization of the pointer represented by the given
  400 -- SetupValue.
  401 markResolved ::
  402   MS.SetupValue (LLVM arch) ->
  403   ResolvedPath {-^ path within this object (if any) -} ->
  404   LLVMResolvedState ->
  405   LLVMResolvedState
  406 markResolved val0 path0 rs = go path0 val0
  407   where
  408     go path val =
  409       case val of
  410         MS.SetupVar n         -> rs & rsAllocs %~ Map.alter (ins path) n
  411         MS.SetupGlobal _ name -> rs & rsGlobals %~ Map.alter (ins path) name
  412         MS.SetupElem _ v idx  -> go (ResolvedElem idx : path) v
  413         MS.SetupField _ v fld -> go (ResolvedField fld : path) v
  414         MS.SetupCast tp v     -> go (ResolvedCast tp : path) v
  415         _                     -> rs
  416 
  417     ins path Nothing = Just [path]
  418     ins path (Just paths) = Just (path : paths)
  419 
  420 -- | Test whether the pointer represented by the given SetupValue has
  421 -- been initialized already.
  422 testResolved ::
  423   MS.SetupValue (LLVM arch) ->
  424   ResolvedPath {-^ path within this object (if any) -} ->
  425   LLVMResolvedState ->
  426   Bool
  427 testResolved val0 path0 rs = go path0 val0
  428   where
  429     go path val =
  430       case val of
  431         MS.SetupVar n         -> test path (Map.lookup n (_rsAllocs rs))
  432         MS.SetupGlobal _ c    -> test path (Map.lookup c (_rsGlobals rs))
  433         MS.SetupElem _ v idx  -> go (ResolvedElem idx : path) v
  434         MS.SetupField _ v fld -> go (ResolvedField fld : path) v
  435         MS.SetupCast tp v     -> go (ResolvedCast tp : path) v
  436         _                     -> False
  437 
  438     test _ Nothing = False
  439     test path (Just paths) = any (overlap path) paths
  440 
  441     overlap (x : xs) (y : ys) = x == y && overlap xs ys
  442     overlap [] _ = True
  443     overlap _ [] = True
