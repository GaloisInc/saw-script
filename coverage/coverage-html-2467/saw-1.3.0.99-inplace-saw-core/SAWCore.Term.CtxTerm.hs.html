   33 -- | Test if a 'Term' is an application of a specific datatype with the
   34 -- supplied context of parameters and indices
   35 ctxAsDataTypeApp :: Name -> [(LocalName, tp1)] ->
   36                     [(LocalName, tp2)] -> Term ->
   37                     Maybe ([Term], [Term])
   38 ctxAsDataTypeApp d params ixs t =
   39   do let (f, args) = asApplyAll t
   40      d' <- asConstant f
   41      guard (d == d')
   42      guard (length args == length params + length ixs)
   43      let (params', ixs') = splitAt (length params) args
   44      pure (params', ixs')

   50 -- | A specification of the type of an argument for a constructor of datatype
   51 -- @d@, that has a specified list @ixs@ of indices, inside a context @ctx@ of
   52 -- parameters and earlier arguments
   53 data CtorArg where
   54   -- | A fixed, constant type
   55   ConstArg :: Term -> CtorArg
   56   -- | The construct @'RecursiveArg [(z1,tp1),..,(zn,tpn)] [e1,..,ek]'@
   57   -- specifies a recursive argument type of the form
   58   --
   59   -- > (z1::tp1) -> .. -> (zn::tpn) -> d p1 .. pm e1 .. ek
   60   --
   61   -- where @d@ is the datatype, the @zi::tpi@ are the elements of the Pi
   62   -- context (the first argument to 'RecursiveArgType'), the @pi@ are the
   63   -- parameters of @d@ (not given here), and the @ei@ are the type indices of
   64   -- @d@.
   65   RecursiveArg ::
   66     [(LocalName, Term)] ->
   67     [Term] ->
   68     CtorArg
   69 
   70 -- | A structure that defines the parameters, arguments, and return type indices
   71 -- of a constructor, using 'Term' and friends to get the bindings right
   72 data CtorArgStruct =
   73   CtorArgStruct
   74   {
   75     ctorParams :: [(LocalName, Term)],
   76     ctorArgs :: [(LocalName, CtorArg)],
   77     ctorIndices :: [Term],
   78     dataTypeIndices :: [(LocalName, Term)]
   79   }

   86 -- | Generic method for testing whether a datatype occurs in an object
   87 class UsesDataType a where
   88   usesDataType :: Name -> a -> Bool
   89 
   90 instance UsesDataType (TermF Term) where
   91   usesDataType d (Constant d')
   92     | d' == d = True
   93 --  usesDataType d (FTermF (DataTypeApp d' _ _))
   94 --    | d' == d = True
   95   usesDataType d (FTermF (RecursorType d' _ _ _))
   96     | d' == d = True
   97   usesDataType d (FTermF (Recursor rec))
   98     | recursorDataType rec == d = True
   99   usesDataType d tf = any (usesDataType d) tf
  100 
  101 instance UsesDataType Term where
  102   usesDataType d = usesDataType d . unwrapTermF
  103 
  104 instance UsesDataType [(LocalName, Term)] where
  105   usesDataType _ [] = False
  106   usesDataType d ((_, tp) : tps) = usesDataType d tp || usesDataType d tps

  109 -- | Check that a type is a valid application of datatype @d@ for use in
  110 -- specific ways in the type of a constructor for @d@. This requires that this
  111 -- application of @d@ be of the form
  112 --
  113 -- > d p1 .. pn x1 .. xm
  114 --
  115 -- where the @pi@ are the distinct bound variables bound in the @params@
  116 -- context, given as argument, and that the @xj@ have no occurrences of @d@. If
  117 -- the given type is of this form, return the @xj@.
  118 asCtorDTApp :: Name -> [(LocalName, Term)] ->
  119                [(LocalName, Term)] ->
  120                [(LocalName, tp1)] ->
  121                [(LocalName, tp2)] ->
  122                Term ->
  123                Maybe [Term]
  124 asCtorDTApp d params dt_ixs ctx1 ctx2 (ctxAsDataTypeApp d params dt_ixs ->
  125                                        Just (param_vars, ixs))
  126   | isVarList params ctx1 ctx2 param_vars &&
  127     not (any (usesDataType d) ixs)
  128   = Just ixs
  129   where
  130     -- Check that the given list of terms is a list of bound variables, one for
  131     -- each parameter, in the context extended by the given arguments
  132     isVarList :: [(LocalName, tp1)] ->
  133                  [(LocalName, tp2)] ->
  134                  [(LocalName, tp3)] ->
  135                  [Term] ->
  136                  Bool
  137     isVarList _ _ _ [] = True
  138     isVarList (_ : ps) c1 c2 ((unwrapTermF -> LocalVar i) : ts) =
  139       i == length ps + length c1 + length c2 &&
  140       isVarList ps c1 c2 ts
  141     isVarList _ _ _ _ = False
  142 asCtorDTApp _ _ _ _ _ _ = Nothing

  145 -- | Check that an argument for a constructor has one of the allowed forms
  146 asCtorArg :: Name -> [(LocalName, Term)] ->
  147              [(LocalName, Term)] ->
  148              [(LocalName, tp)] ->
  149              Term ->
  150              Maybe CtorArg
  151 asCtorArg d params dt_ixs prevs (asPiList ->
  152                                  (zs,
  153                                   asCtorDTApp d params dt_ixs prevs zs ->
  154                                   Just ixs))
  155   | not (usesDataType d zs)
  156   = Just (RecursiveArg zs ixs)
  157 asCtorArg d _ _ _ tp
  158   | not (usesDataType d tp)
  159   = Just (ConstArg tp)
  160 asCtorArg _ _ _ _ _ = Nothing
  161 
  162 -- | Check that a constructor type is a pi-abstraction that takes as input an
  163 -- argument of one of the allowed forms described by 'CtorArg'
  164 asPiCtorArg :: Name -> [(LocalName, Term)] ->
  165                [(LocalName, Term)] ->
  166                [(LocalName, tp)] ->
  167                Term ->
  168                Maybe (LocalName, CtorArg, Term)
  169 asPiCtorArg d params dt_ixs prevs t =
  170   case asPi t of
  171     Just (x, asCtorArg d params dt_ixs prevs -> Just arg, rest) ->
  172       Just (x, arg, rest)
  173     _ ->
  174       Nothing
  175 
  176 -- | Helper function for 'mkCtorArgStruct'
  177 mkCtorArgsIxs :: Name -> [(LocalName, Term)] ->
  178                  [(LocalName, Term)] ->
  179                  [(LocalName, CtorArg)] ->
  180                  Term ->
  181                  Maybe ([(LocalName, CtorArg)], [Term])
  182 mkCtorArgsIxs d params dt_ixs prevs (asPiCtorArg d params dt_ixs prevs ->
  183                                      Just (x, arg, rest)) =
  184   case mkCtorArgsIxs d params dt_ixs (prevs ++ [(x, arg)]) rest of
  185     Just (args, ixs) -> Just ((x, arg) : args, ixs)
  186     Nothing -> Nothing
  187 mkCtorArgsIxs d params dt_ixs prevs (asCtorDTApp d params dt_ixs prevs [] ->
  188                                      Just ixs) =
  189   Just ([], ixs)
  190 mkCtorArgsIxs _ _ _ _ _ = Nothing

  193 -- | Take in a datatype and bindings lists for its parameters and indices, and
  194 -- also a prospective type of a constructor for that datatype, where the
  195 -- constructor type is allowed to have the parameters but not the indices free.
  196 -- Test that the constructor type is an allowed type for a constructor of this
  197 -- datatype, and, if so, build a 'CtorArgStruct' for it.
  198 mkCtorArgStruct ::
  199   Name ->
  200   [(LocalName, Term)] ->
  201   [(LocalName, Term)] ->
  202   Term ->
  203   Maybe CtorArgStruct
  204 mkCtorArgStruct d params dt_ixs ctor_tp =
  205   case mkCtorArgsIxs d params dt_ixs [] ctor_tp of
  206     Just (args, ctor_ixs) ->
  207       Just (CtorArgStruct params args ctor_ixs dt_ixs)
  208     Nothing -> Nothing
