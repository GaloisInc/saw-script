  128 -- | 'ImportVisibility' - Should a given import (see 'importCryptolModule')
  129 -- result in all symbols being visible (as they are for focused
  130 -- modules in the Cryptol REPL) or only public symbols?  Making all
  131 -- symbols visible is useful for verification and code generation.
  132 --
  133 -- NOTE: this notion of public vs. private symbols is specific to
  134 -- SAWScript and distinct from Cryptol's notion of private
  135 -- definitions.
  136 --
  137 data ImportVisibility
  138   = OnlyPublic       -- ^ behaves like a normal Cryptol "import"
  139   | PublicAndPrivate -- ^ allows viewing of both "private" sections
  140                      --   and (arbitrarily nested) submodules.
  141   deriving (Eq, Show)

  166 -- | Lookup a name in a map containg Cryptol names.
  167 -- The string corresponds to the Cryptol name we are looking for.
  168 -- If it is unqualifed, then we return any entry associated with the given
  169 -- name.  If the string is qualified (i.e., has @::@), then we only consider
  170 -- entries from the module in the qualified.
  171 -- The result is either the corresponding value, or a list of the
  172 lookupIn :: Text -> Map T.Name b -> Either [T.Name] b
  173 lookupIn nm mp =
  174   case [ x | x <- Map.toList mp, matches (fst x) ] of
  175     [ (_,v) ] -> Right v
  176     opts      -> Left (map fst opts)
  177   where
  178   matches = nameMatcher nm
  179 
  180 
  181 -- | Parse a string into a function that will match names.
  182 -- If the string is unqualified (i.e., no `::`), then we match all
  183 -- names with the given identifier.  Otherwise, we only match the
  184 -- ones in the module specified by the qualifier.
  185 nameMatcher :: Text -> T.Name -> Bool
  186 nameMatcher xs =
  187     case C.modNameChunksText (textToModName xs) of
  188       []  -> const False
  189       [x] -> (x ==) . C.identText . MN.nameIdent
  190       cs  -> \n ->
  191                 case MN.nameInfo n of
  192                   MN.LocalName {} -> False
  193                   MN.GlobalName _ og ->
  194                     let (top,ns) = C.modPathSplit (C.ogModule og)
  195                     in last cs == identText (C.ogName og) &&
  196                        init cs == C.modNameChunksText top ++ map identText ns

  201 -- | initCryptolEnv - Create initial CryptolEnv, this involves loading
  202 --   the built-in modules (preludeName, arrayName,
  203 --   preludeReferenceName) and translating them into SAWCore, and
  204 --   putting them into scope.
  205 --
  206 --   NOTE: submodules in these built-in modules are supported in this code.
  207 --
  208 initCryptolEnv ::
  209   (?fileReader :: FilePath -> IO ByteString) =>
  210   SharedContext -> IO CryptolEnv
  211 initCryptolEnv sc = do
  212   modEnv0 <- M.initialModuleEnv
  213 
  214   -- Set the Cryptol include path (TODO: we may want to do this differently)
  215   (binDir, _) <- splitExecutablePath
  216   let instDir = normalise . joinPath . init . splitPath $ binDir
  217   mCryptolPath <- lookupEnv "CRYPTOLPATH"
  218   let cryptolPaths =
  219         case mCryptolPath of
  220           Nothing -> []
  221           Just path ->
  222 #if defined(mingw32_HOST_OS) || defined(__MINGW32__)
  223             -- Windows paths search from end to beginning
  224             reverse (splitSearchPath path)
  225 #else
  226             splitSearchPath path
  227 #endif
  228   let modEnv1 = modEnv0 { ME.meSearchPath = cryptolPaths ++
  229                            (instDir </> "lib") : ME.meSearchPath modEnv0 }
  230 
  231   -- Load Cryptol prelude and magic Array module
  232   (_, modEnv2) <-
  233     liftModuleM modEnv1 $
  234       do _ <- MB.loadModuleFrom False (MM.FromModule preludeName)
  235          _ <- MB.loadModuleFrom False (MM.FromModule arrayName)
  236          return ()
  237 
  238   -- Load Cryptol reference implementation
  239   ((_,refTop), modEnv3) <-
  240     liftModuleM modEnv2 $
  241       MB.loadModuleFrom False (MM.FromModule preludeReferenceName)
  242   let refMod = T.tcTopEntityToModule refTop
  243 
  244   -- Set up reference implementation redirections
  245   let refDecls = T.mDecls refMod
  246   let nms = Set.toList (MI.ifsPublic (TIface.genIfaceNames refMod))
  247 
  248   let refPrims = Map.fromList
  249                   [ (prelPrim (identText (MN.nameIdent nm)), T.EWhere (T.EVar nm) refDecls)
  250                   | nm <- nms ]
  251   let cryEnv0 = C.emptyEnv { C.envRefPrims = refPrims }
  252 
  253   -- Generate SAWCore translations for all values in scope
  254   termEnv <- genTermEnv sc modEnv3 cryEnv0
  255 
  256   -- The module names in P.Import are now Located, so give them an empty position.
  257   let preludeName'          = locatedUnknown preludeName
  258       preludeReferenceName' = locatedUnknown preludeReferenceName
  259       arrayName'            = locatedUnknown arrayName
  260 
  261   return CryptolEnv
  262     { eImports    =
  263         [ mkImport OnlyPublic preludeName'          Nothing Nothing
  264         , mkImport OnlyPublic preludeReferenceName' (Just preludeReferenceName) Nothing
  265         , mkImport OnlyPublic arrayName'            Nothing Nothing
  266         ]
  267     , eModuleEnv  = modEnv3
  268     , eExtraNames = mempty
  269     , eExtraTypes = Map.empty
  270     , eExtraTSyns = Map.empty
  271     , eTermEnv    = termEnv
  272     , ePrims      = Map.empty
  273     , ePrimTypes  = Map.empty
  274     , eFFITypes   = Map.empty
  275     }

  278 -- | Translate all declarations in all loaded modules to SAWCore terms
  279 --   NOTE: used only for initialization code.
  280 --
  281 genTermEnv :: SharedContext -> ME.ModuleEnv -> C.Env -> IO (Map T.Name Term)
  282 genTermEnv sc modEnv cryEnv0 = do
  283   let declGroups = concatMap T.mDecls
  284                  $ filter (not . T.isParametrizedModule)
  285                  $ ME.loadedModules modEnv
  286       nominals   = loadedNonParamNominalTypes modEnv
  287   cryEnv1 <- C.genCodeForNominalTypes sc nominals cryEnv0
  288   cryEnv2 <- C.importTopLevelDeclGroups sc C.defaultPrimitiveOptions cryEnv1 declGroups
  289   return (C.envE cryEnv2)

  303 ioParseGeneric ::
  304   (P.Config -> Text -> Either P.ParseError a) -> InputText -> IO a
  305 ioParseGeneric parse inp = ioParseResult (parse cfg str)
  306   where
  307   cfg = P.defaultConfig { P.cfgSource = inpFile inp }
  308   -- XXX this is kind of gross; maybe sometime we get a second parser
  309   -- entry point that takes a start position... (this is saw-script #2175)
  310   str = Text.concat [ Text.replicate (inpLine inp - 1) "\n"
  311                     , Text.replicate (inpCol inp - 1) " "
  312                     , inpText inp ]
  313 
  314 ioParseResult :: Either P.ParseError a -> IO a
  315 ioParseResult res = case res of
  316   Right a -> return a
  317   Left e  -> fail $ "Cryptol parse error:\n" ++ show (P.ppError e) -- X.throwIO (ParseError e)

  322 -- | @'getNamingEnv' env@ - get the full 'MR.NamingEnv' based on all
  323 -- the imports (@eImports env@).
  324 getNamingEnv :: CryptolEnv -> MR.NamingEnv
  325 getNamingEnv env =
  326   eExtraNames env
  327   `MR.shadowing`
  328   (mconcat $ map (getNamingEnvForImport (eModuleEnv env))
  329                  (eImports env)
  330   )
  331 
  332 -- | get Naming Env for one Import.
  333 getNamingEnvForImport :: ME.ModuleEnv
  334                       -> (ImportVisibility, T.Import)
  335                       -> MR.NamingEnv
  336 getNamingEnvForImport modEnv (vis, imprt) =
  337     MN.interpImportEnv imprt -- adjust for qualified imports
  338   $ computeNamingEnv lm vis
  339 
  340   where
  341   modName :: C.ModName
  342   modName = P.thing $ T.iModule imprt
  343 
  344   lm = case ME.lookupModule modName modEnv of
  345          Just lm' -> lm'
  346          Nothing  -> panic "getNamingEnvForImport"
  347                        ["cannot lookupModule: " <> Text.pack(show modName)]
  348 
  349 -- | compute the NamingEnv based on the ImportVisibility.
  350 computeNamingEnv :: ME.LoadedModule -> ImportVisibility -> MR.NamingEnv
  351 computeNamingEnv lm vis =
  352   case vis of
  353     PublicAndPrivate -> envPublicAndPrivate  -- all names defined, pub & pri
  354     OnlyPublic       -> envPublic            -- i.e., what's exported.
  355 
  356   where
  357   -- NamingEnvs: --
  358 
  359     -- | envTopLevels
  360     --    - Does not include privates in submodules (which makes for
  361     --      much of the complications of this function).
  362     --    - Includes everything in scope at the toplevel of 'lm' module
  363     envTopLevels :: MR.NamingEnv
  364     envTopLevels = ME.lmNamingEnv lm
  365 
  366     -- | envPublicAndPrivate - awkward as envTopLevels excludes privates
  367     envPublicAndPrivate :: MR.NamingEnv
  368     envPublicAndPrivate =
  369        -- nab all the names defined in module (from toplevel scope):
  370        MN.filterUNames (`Set.member` nmsDefined) envTopLevels
  371        <>
  372        -- we must create a new NamingEnv (since the privates are not
  373        -- in `envTopLevels`):
  374        MN.namingEnvFromNames' MN.nameToPNameWithQualifiers nmsPrivate
  375 
  376     envPublic :: MR.NamingEnv
  377     envPublic = MN.filterUNames
  378                   (`Set.member` nmsPublic)
  379                   envTopLevels
  380 
  381   -- Name Sets: --
  382 
  383     -- | names in scope at Top level of module
  384     nmsTopLevels :: Set MN.Name
  385     nmsTopLevels = MN.namingEnvNames envTopLevels
  386 
  387     -- | names defined in module and in submodules
  388     --   - this includes `PublicAndPrivate` names!
  389     --   - includes submodule names, type synonyms, and nominal types
  390     nmsDefined :: Set MN.Name
  391     nmsDefined =
  392         -- definitions from all submodules:
  393         ( Set.unions
  394         $ map (MI.ifsDefines . T.smIface)
  395         $ Map.elems
  396         $ T.mSubmodules
  397         $ ME.lmModule lm
  398         )
  399       `Set.union`
  400         -- definitions at the top module:
  401         (MI.ifsDefines $ MI.ifNames $ ME.lmInterface lm)
  402 
  403 
  404     nmsPublic :: Set MN.Name
  405     nmsPublic = MI.ifsPublic $ MI.ifNames $ ME.lmInterface lm
  406 
  407     nmsPrivate :: Set MN.Name
  408     nmsPrivate = nmsDefined Set.\\ nmsTopLevels

  417 -- | Like Cryptol's 'ME.loadedNominalTypes', except that it only returns
  418 -- nominal types from non-parameterized modules, which are currently the only

  430   case out of

  433       do MM.setNameSeeds seeds

  445 mkCryEnv ::
  446   (?fileReader :: FilePath -> IO ByteString) =>
  447   CryptolEnv -> IO C.Env
  448 mkCryEnv env =
  449   do let modEnv = eModuleEnv env
  450      let ifaceDecls = getAllIfaceDecls modEnv
  451      (types, _) <-
  452        liftModuleM modEnv $
  453        do prims <- MB.getPrimMap
  454           -- noIfaceParams because we don't support translating functors yet
  455           infInp <- MB.genInferInput P.emptyRange prims NoParams ifaceDecls
  456           let newtypeCons = Map.fromList
  457                               [ con
  458                               | nt <- Map.elems (TM.inpNominalTypes infInp)
  459                               , con <- T.nominalTypeConTypes nt
  460                               ]
  461           pure (newtypeCons `Map.union` TM.inpVars infInp)
  462      let types' = Map.union (eExtraTypes env) types
  463      let terms = eTermEnv env
  464      let cryEnv = C.emptyEnv
  465            { C.envE = terms
  466            , C.envC = types'
  467            , C.envPrims = ePrims env
  468            , C.envPrimTypes = ePrimTypes env
  469            }
  470      return cryEnv

  472 translateExpr ::
  473   (?fileReader :: FilePath -> IO ByteString) =>
  474   SharedContext -> CryptolEnv -> T.Expr -> IO Term
  475 translateExpr sc env expr =
  476   do cryEnv <- mkCryEnv env
  477      C.importExpr sc cryEnv expr

  479 translateDeclGroups ::
  480   (?fileReader :: FilePath -> IO ByteString) =>
  481   SharedContext -> CryptolEnv -> [T.DeclGroup] -> IO CryptolEnv
  482 translateDeclGroups sc env dgs =
  483   do cryEnv  <- mkCryEnv env
  484      cryEnv' <- C.importTopLevelDeclGroups sc C.defaultPrimitiveOptions cryEnv dgs
  485      let decls = concatMap T.groupDecls dgs
  486      let names = map T.dName decls
  487      let newTypes = Map.fromList [ (T.dName d, T.dSignature d) | d <- decls ]
  488      let addName name = MR.shadowing (MN.singletonNS C.NSValue (P.mkUnqual (MN.nameIdent name)) name)
  489      return env { eExtraNames = foldr addName (eExtraNames env) names
  490                 , eExtraTypes = Map.union (eExtraTypes env) newTypes
  491                 , eTermEnv    = C.envE cryEnv'
  492                 }

  496 -- | Restore a `CryptolEnv` from a checkpoint. The first argument
  497 --   `chkEnv` is the `CryptolEnv` saved by / copied into the
  498 --   checkpoint; the second argument `newEnv` is the current one
  499 --   we wish to overwrite by rolling back to the checkpoint.
  500 --
  501 --   We need to keep the newer name supply so as to not reuse names
  502 --   already issued, in case some of those are still floating around
  503 --   after the restore. (They should not... but bugs happen.)
  504 --
  505 --   We also ought to invalidate terms constructed since the checkpoint
  506 --   was taken, like SAWCore does. See #2859.
  507 --
  508 restoreCryptolEnv :: CryptolEnv -> CryptolEnv -> CryptolEnv
  509 restoreCryptolEnv chkEnv newEnv =
  510     let newMEnv = eModuleEnv newEnv
  511         chkMEnv = eModuleEnv chkEnv
  512         menv' = chkMEnv { ME.meNameSeeds = ME.meNameSeeds newMEnv }
  513     in
  514     chkEnv { eModuleEnv = menv' }

  520 -- | ExtCryptolModule - Extended CryptolModule; we keep track of
  521 --   whether this module came directly from a constructed
  522 --   `CryptolModule` or whether it came from parsing a Cryptol module
  523 --   from filesystem (in which case it is loaded).
  524 --
  525 --
  526 data ExtCryptolModule =
  527     -- | source is parsed/loaded
  528     ECM_LoadedModule
  529         (P.Located C.ModName)
  530         PPS.Doc      -- ^ how we show this on SAWScript CLI.
  531 
  532     -- | source is internal/constructed (e.g., via cryptol_prims)
  533   | ECM_CryptolModule  CryptolModule
  534 
  535 -- | create the string needed for display in the CLI.
  536 --
  537 --  - FIXME: This function, with the ECM_LoadedModule constructor, are
  538 --    a bit ad hoc!  Currently `ExtCryptolModule` is exposed to the
  539 --    CLI *and* requires a way to show this type to the user (as
  540 --    implemented here) to support the user interface.  As the state
  541 --    isn't available when we want to display this value, we compute
  542 --    the "display" String when we construct `ExtCryptolModule` values.
  543 --
  544 --    The best solution is to implement Issue #2680 (Add `:cbrowse`) in
  545 --    order to both improve the user interface and remove this awkward code.
  546 --    Implementing #2680 will also address Issue #2700.
  547 --
  548 --  - Update: this is still problematic but now it's at least a ppdoc
  549 --    and not a string. Note that it uses PPS.Doc (wired to the
  550 --    SAWCore annotations) rather than a generic `PP.Doc ann` because
  551 --    we're storing the doc in the `ExtCryptolModule` and making it
  552 --    polymorphic there creates pointless complications.
  553 --
  554 prettyExtCryptolModule :: ExtCryptolModule -> PPS.Doc
  555 prettyExtCryptolModule =
  556   \case
  557     ECM_LoadedModule name doc ->
  558       -- Cryptol's prettyprinter is not compatible with ours
  559       let name' = CryPP.pretty (P.thing name) in
  560       "Loaded module" <+> PP.squotes name' <> ":" <> PP.hardline <> doc
  561     ECM_CryptolModule cm  ->
  562       PP.vsep [
  563           "Internal module:",
  564           prettyCryptolModule cm
  565       ]

  567 -- | loadCryptolModule - load a cryptol module and return the
  568 -- `ExtCryptolModule`.  The contents of the module are not directly
  569 -- imported into the environment.
  570 --
  571 -- This is used to implement the "cryptol_load" primitive in which a
  572 -- handle to the module is returned and can be bound to a SAWScript
  573 -- variable.
  574 --
  575 -- NOTE: Bringing the module into {{-}} scope is not handled
  576 --       here; it is done rather in `bindExtCryptolModule`, ONLY if the
  577 --       user binds the `cryptolModule` returned here at the SAW
  578 --       command line.
  579 loadExtCryptolModule ::
  580   (?fileReader :: FilePath -> IO ByteString) =>
  581   SharedContext ->
  582   CryptolEnv ->
  583   FilePath ->
  584   IO (ExtCryptolModule, CryptolEnv)
  585 loadExtCryptolModule sc env path =
  586   do
  587   (m, env') <- loadAndTranslateModule sc env (Left path)
  588   let doc = PP.vsep [
  589               "Public interface",
  590               prettyCryptolModule (mkCryptolModule m env')
  591           ]
  592           -- How to show, need to compute this here, because the show function
  593           -- (of course) has no access to the state.
  594           --
  595           -- FIXME: Since the complete public and private interface is
  596           -- extractable, we should show the whole thing with public,
  597           -- private, typesyns, constructors, submodules.
  598           -- See Issue #2700
  599           --

  605 -- | loadCryptolModule
  606 --
  607 -- NOTE RE CALLS TO THIS:
  608 --  - There is only the path to this function from the command line,
  609 --    and it is only via the experimental command,
  610 --      "write_rocq_cryptol_module".
  611 --
  612 -- This function (note `mkCryptolModule`) returns the public types and values
  613 -- of the module in a `CryptolModule` structure.
  614 --
  615 loadCryptolModule ::
  616   (?fileReader :: FilePath -> IO ByteString) =>
  617   SharedContext ->
  618   CryptolEnv ->
  619   FilePath ->
  620   IO (CryptolModule, CryptolEnv)
  621 loadCryptolModule sc env path =
  622   do
  623   (mod', env') <- loadAndTranslateModule sc env (Left path)
  624   return (mkCryptolModule mod' env', env')

  627 -- | mkCryptolModule m env - translate a @m :: T.Module@ to a `CryptolModule`
  628 --
  629 -- This function returns the public types and values of the module `m`
  630 -- as a `CryptolModule` structure.
  631 mkCryptolModule :: T.Module -> CryptolEnv -> CryptolModule
  632 mkCryptolModule m env =
  633   let
  634       ifaceDecls = getAllIfaceDecls (eModuleEnv env)
  635       types = Map.map MI.ifDeclSig (MI.ifDecls ifaceDecls)
  636       -- we're keeping only the exports of `m`:
  637       vNameSet = MEx.exported C.NSValue (T.mExports m)
  638       tNameSet = MEx.exported C.NSType  (T.mExports m)
  639   in
  640       CryptolModule
  641         -- create Map of type synonyms:
  642         (Map.filterWithKey
  643            (\k _ -> Set.member k tNameSet)
  644            (T.mTySyns m)
  645         )
  646 
  647         -- create Map of the `TypedTerm` s:
  648         ( Map.filterWithKey (\k _ -> Set.member k vNameSet)
  649         $ Map.intersectionWith
  650              (\t x -> TypedTerm (TypedTermSchema t) x)
  651              types
  652              (eTermEnv env)
  653         )

  655 -- | bindExtCryptolModule - add extra bindings to the Cryptol
  656 --     environment {{-}}, this happens when an `ExtCryptolModule` is
  657 --     bound in the SAWScript code.  (This may be referred to as a
  658 --     "magic bind").
  659 --
  660 --   NOTE RE CALLS TO THIS: Three SAWScript variants get us here:
  661 --      > D <- cryptol_load "PATH"
  662 --
  663 --    which results in a call to `bindLoadedModule`.
  664 --    And each of these
  665 --
  666 --      > x <- return (cryptol_prims ())
  667 --      > let x = cryptol_prims ()
  668 --
  669 --    will result in calling `bindCryptolModule`.
  670 --
  671 -- NOTE:
  672 --  - The `ExtCryptolModule` datatype and these functions are a bit
  673 --    adhoc.
  674 --  - Ideally thse would go away with further improvements to the
  675 --    external interface and corresponding implementation changes.
  676 --    - See #2569.
  677 --    - Re `bindCryptolModule` below
  678 --      - It is more general than what is needed.
  679 --      - It is somewhat duplicating functionality that we already have with
  680 --        `importCryptolModule`, this could go away in the future.
  681 --
  682 --  - See also the discusion of `cryptol_load` in CHANGES.md.
  683 
  684 bindExtCryptolModule ::
  685   (P.ModName, ExtCryptolModule) -> CryptolEnv -> CryptolEnv
  686 bindExtCryptolModule (modName, ecm) =
  687   case ecm of
  688     ECM_CryptolModule cm   -> bindCryptolModule (modName, cm)
  689     ECM_LoadedModule  nm _ -> bindLoadedModule  (modName, nm)

  691 -- | bindLoadedModule - when we have a `cryptol_load` created object,
  692 -- add the module into the import list.
  693 bindLoadedModule ::
  694   (P.ModName, P.Located C.ModName) -> CryptolEnv -> CryptolEnv
  695 bindLoadedModule (asName, origName) env =
  696   env {eImports = mkImport PublicAndPrivate origName (Just asName) Nothing
  697                 : eImports env
  698       }

  700 -- | bindCryptolModule - when we have a `cryptol_prims ()` created
  701 --   object, add the `CryptolModule` to the relevant maps in the
  702 --   `CryptolEnv` See `bindExtCryptolModule` above.
  703 bindCryptolModule :: (P.ModName, CryptolModule) -> CryptolEnv -> CryptolEnv
  704 bindCryptolModule (modName, CryptolModule sm tm) env =
  705   env { eExtraNames = flip (foldr addName) (Map.keys tm') $
  706                       flip (foldr addTSyn) (Map.keys sm) $
  707                       eExtraNames env
  708       , eExtraTSyns = Map.union sm (eExtraTSyns env)
  709       , eExtraTypes = Map.union (fmap fst tm') (eExtraTypes env)
  710       , eTermEnv    = Map.union (fmap snd tm') (eTermEnv env)
  711       }
  712   where
  713     -- | `tm'` is the typed terms from `tm` that have Cryptol schemas
  714     tm' = Map.mapMaybe f tm
  715           where
  716           f (TypedTerm (TypedTermSchema s) x) = Just (s,x)
  717           f _                                 = Nothing
  718 
  719     addName name =
  720       MN.shadowing
  721        (MN.singletonNS C.NSValue (P.mkQual modName (MN.nameIdent name)) name)
  722 
  723     addTSyn name =
  724       MN.shadowing
  725         (MN.singletonNS C.NSType (P.mkQual modName (MN.nameIdent name)) name)

  770 loadAndTranslateModule ::
  771   (?fileReader :: FilePath -> IO ByteString) =>
  772   SharedContext             {- ^ Shared context for creating terms -} ->
  773   CryptolEnv                {- ^ Extend this environment -} ->
  774   Either FilePath P.ModName {- ^ Where to find the module -} ->
  775   IO (T.Module, CryptolEnv)
  776 loadAndTranslateModule sc env src =
  777   do let modEnv = eModuleEnv env
  778      (mtop, modEnv') <- liftModuleM modEnv $
  779        case src of
  780          Left path -> MB.loadModuleByPath True path
  781          Right mn  -> snd <$> MB.loadModuleFrom True (MM.FromModule mn)
  782      m <- case mtop of
  783             T.TCTopModule mod'  -> pure mod'
  784             T.TCTopSignature {} ->
  785               fail $
  786                 "Expected a module, but "
  787                 ++ (case src of
  788                       Left  path -> show path
  789                       Right mn   -> show mn
  790                    )
  791                 ++ " is an interface."
  792 
  793      checkNotParameterized m
  794 
  795      -- Regenerate SharedTerm environment:
  796      let oldModNames   = map ME.lmName
  797                        $ ME.lmLoadedModules
  798                        $ ME.meLoadedModules modEnv
  799          isNew m'      = T.mName m' `notElem` oldModNames
  800          newModules    = filter isNew
  801                        $ map ME.lmModule
  802                        $ ME.lmLoadedModules
  803                        $ ME.meLoadedModules modEnv'
  804          newDeclGroups = concatMap T.mDecls newModules
  805          newNominal    = Map.difference (loadedNonParamNominalTypes modEnv')
  806                                         (loadedNonParamNominalTypes modEnv)
  807 
  808      newTermEnv <-
  809        do oldCryEnv <- mkCryEnv env
  810           cEnv      <- C.genCodeForNominalTypes sc newNominal oldCryEnv
  811           newCryEnv <- C.importTopLevelDeclGroups
  812                         sc C.defaultPrimitiveOptions cEnv newDeclGroups
  813           return (C.envE newCryEnv)
  814 
  815      return ( m
  816             , env { eModuleEnv = modEnv'
  817                   , eTermEnv   = newTermEnv
  818                   , eFFITypes  = updateFFITypes m newTermEnv (eFFITypes env)
  819                   }
  820             )
  821 
  822 checkNotParameterized :: T.Module -> IO ()
  823 checkNotParameterized m =
  824   when (T.isParametrizedModule m) $
  825     fail $ unlines [ "Cannot load parameterized modules directly."
  826                    , "Either use a ` import, or make a module instantiation."
  827                    ]
  828 
  829 updateFFITypes :: T.Module -> Map MN.Name Term -> Map NameInfo T.FFI -> Map NameInfo T.FFI
  830 updateFFITypes m eTermEnv' eFFITypes' =
  831   foldr (\(nm, ty) -> Map.insert (getNameInfo nm) ty)
  832                        eFFITypes'
  833                        (T.findForeignDecls m)
  834   where
  835   getNameInfo nm =
  836     case Map.lookup nm eTermEnv' of
  837       Just tm ->
  838         case asConstant tm of
  839           Just n -> nameInfo n
  840           Nothing ->
  841             panic "updateFFITypes" [
  842                 "SAWCore term of Cryptol name is not Constant",
  843                 "Name: " <> Text.pack (show nm),
  844                 "Term: " <> Text.pack (ppTermPureDefaults tm)
  845             ]
  846       Nothing ->
  847         panic "updateFFITypes" [
  848             "Cannot find foreign function in term environment: " <> Text.pack (show nm)
  849         ]

  854 -- | @'importCryptolModule' sc env src as vis imps@ - extend the Cryptol
  855 --   environment with a module.  Closely mirrors the sawscript command "import".
  856 --
  857 -- NOTE:
  858 --  - the module can be qualified or not (per 'as' argument).
  859 --  - per 'vis' we can import public definitions or *all* (i.e., internal
  860 --    and public) definitions.
  861 
  862 importCryptolModule ::
  863   (?fileReader :: FilePath -> IO ByteString) =>
  864   SharedContext             {- ^ Shared context for creating terms -} ->
  865   CryptolEnv                {- ^ Extend this environment -} ->
  866   Either FilePath P.ModName {- ^ Where to find the module -} ->
  867   Maybe P.ModName           {- ^ Name qualifier -} ->
  868   ImportVisibility          {- ^ What visibility to give symbols from this module -} ->
  869   Maybe P.ImportSpec        {- ^ What to import -} ->
  870   IO CryptolEnv
  871 importCryptolModule sc env src as vis imps =
  872   do
  873   (mod', env') <- loadAndTranslateModule sc env src
  874   let import' = mkImport vis (locatedUnknown (T.mName mod')) as imps
  875   return $ env' {eImports = import' : eImports env }
  876 
  877 mkImport :: ImportVisibility
  878          -> P.Located C.ModName
  879          -> Maybe C.ModName
  880          -> Maybe T.ImportSpec
  881          -> (ImportVisibility, T.Import)
  882 mkImport vis nm as imps =
  883     let im = P.Import { T.iModule = nm
  884                       , T.iAs     = as
  885                       , T.iSpec   = imps
  886                       , T.iInst   = Nothing
  887                       , T.iDoc    = Nothing
  888                       }
  889     in
  890     (vis, im)

  951 resolveIdentifier ::
  952   (HasCallStack, ?fileReader :: FilePath -> IO ByteString) =>
  953   CryptolEnv -> Text -> IO (Maybe T.Name)
  954 resolveIdentifier env nm =
  955   case splitOn (pack "::") nm of
  956     []  -> pure Nothing
  957            -- FIXME: shouldn't this be error?
  958     [i] -> doResolve (P.mkUnqual (C.mkIdent i))
  959     xs  -> let (qs,i) = (init xs, last xs)
  960            in  doResolve (P.Qual (C.packModName qs) (C.mkIdent i))
  961     -- FIXME: Is there no function that parses Text into PName?
  962 
  963   where
  964   modEnv = eModuleEnv env
  965   nameEnv = getNamingEnv env
  966 
  967   doResolve pnm =
  968     SMT.withSolver (return ()) (meSolverConfig modEnv) $ \solver ->
  969     do let minp = MM.ModuleInput {
  970                MM.minpCallStacks = True,
  971                MM.minpSaveRenamed = False,
  972                MM.minpEvalOpts = pure defaultEvalOpts,
  973                MM.minpByteReader = ?fileReader,
  974                MM.minpModuleEnv = modEnv,
  975                MM.minpTCSolver = solver
  976            }
  977        (res, _ws) <- MM.runModuleM minp $
  978           MM.interactive (MB.rename interactiveName nameEnv (MR.renameVar MR.NameUse pnm))
  979        case res of
  980          Left _ -> pure Nothing
  981          Right (x,_) -> pure (Just x)

  983 parseTypedTerm ::
  984   (HasCallStack, ?fileReader :: FilePath -> IO ByteString) =>
  985   SharedContext -> CryptolEnv -> InputText -> IO TypedTerm
  986 parseTypedTerm sc env input = do
  987   -- Parse:
  988   pexpr <- ioParseExpr input
  989   -- Translate:
  990   pExprToTypedTerm sc env pexpr
  991 
  992 pExprToTypedTerm ::
  993   (?fileReader :: FilePath -> IO ByteString) =>
  994   SharedContext -> CryptolEnv -> P.Expr P.PName -> IO TypedTerm
  995 pExprToTypedTerm sc env pexpr = do
  996   let modEnv = eModuleEnv env
  997 
  998   ((expr, schema), modEnv') <- liftModuleM modEnv $ do
  999 
 1000     -- Eliminate patterns:
 1001     npe <- MM.interactive (MB.noPat pexpr)
 1002 
 1003     let nameEnv = getNamingEnv env
 1004     let npe' = MR.rename npe
 1005     re <- MM.interactive (MB.rename interactiveName nameEnv npe')
 1006       -- NOTE: if a name is not in scope, it is reported here.
 1007 
 1008     -- Infer types
 1009     let ifDecls = getAllIfaceDecls modEnv
 1010     let range = fromMaybe P.emptyRange (P.getLoc re)
 1011     prims <- MB.getPrimMap
 1012     -- noIfaceParams because we don't support functors yet
 1013     tcEnv <- MB.genInferInput range prims NoParams ifDecls
 1014     let tcEnv' = tcEnv { TM.inpVars = Map.union (eExtraTypes env) (TM.inpVars tcEnv)
 1015                        , TM.inpTSyns = Map.union (eExtraTSyns env) (TM.inpTSyns tcEnv)
 1016                        }
 1017 
 1018     out <- MM.io (T.tcExpr re tcEnv')
 1019     MM.interactive (runInferOutput out)
 1020 
 1021   let env' = env { eModuleEnv = modEnv' }
 1022 
 1023   -- Translate
 1024   trm <- translateExpr sc env' expr
 1025   return (TypedTerm (TypedTermSchema schema) trm)

 1027 parseDecls ::
 1028   (?fileReader :: FilePath -> IO ByteString) =>
 1029   SharedContext -> CryptolEnv -> InputText -> IO CryptolEnv
 1030 parseDecls sc env input = do
 1031   let modEnv = eModuleEnv env
 1032   let ifaceDecls = getAllIfaceDecls modEnv
 1033 
 1034   -- Parse
 1035   (decls :: [P.Decl P.PName]) <- ioParseDecls input
 1036 
 1037   (tmodule, modEnv') <- liftModuleM modEnv $ do
 1038 
 1039     -- Eliminate patterns
 1040     (npdecls :: [P.Decl P.PName]) <- MM.interactive (MB.noPat decls)
 1041 
 1042     -- Convert from 'Decl' to 'TopDecl' so that types will be generalized
 1043     let topdecls = [ P.Decl (P.TopLevel P.Public Nothing d) | d <- npdecls ]
 1044 
 1045     -- Resolve names
 1046     (_nenv, rdecls) <- MM.interactive (MB.rename interactiveName (getNamingEnv env) (MR.renameTopDecls interactiveName topdecls))
 1047 
 1048     -- Create a Module to contain the declarations
 1049     let rmodule = P.Module { P.mName = locatedUnknown interactiveName
 1050                            , P.mDef  = P.NormalModule rdecls
 1051                            , P.mInScope = mempty
 1052                            , P.mDocTop = Nothing
 1053                            }
 1054 
 1055     -- Infer types
 1056     let range = fromMaybe P.emptyRange (P.getLoc rdecls)
 1057     prims <- MB.getPrimMap
 1058     -- noIfaceParams because we don't support functors yet
 1059     tcEnv <- MB.genInferInput range prims NoParams ifaceDecls
 1060     let tcEnv' = tcEnv { TM.inpVars = Map.union (eExtraTypes env) (TM.inpVars tcEnv)
 1061                        , TM.inpTSyns = Map.union (eExtraTSyns env) (TM.inpTSyns tcEnv)
 1062                        }
 1063 
 1064     out <- MM.io (TM.runInferM tcEnv' (TI.inferTopModule rmodule))
 1065     tmodule <- MM.interactive (runInferOutput out)
 1066     m <- case tmodule of
 1067            T.TCTopModule m -> pure m
 1068            T.TCTopSignature {} ->
 1069               fail "Expected a module, but found an interface."
 1070     return m
 1071 
 1072   -- Add new type synonyms and their name bindings to the environment
 1073   let syns' = Map.union (eExtraTSyns env) (T.mTySyns tmodule)
 1074   let addName name = MR.shadowing (MN.singletonNS C.NSType (P.mkUnqual (MN.nameIdent name)) name)
 1075   let names' = foldr addName (eExtraNames env) (Map.keys (T.mTySyns tmodule))
 1076   let env' = env { eModuleEnv = modEnv', eExtraNames = names', eExtraTSyns = syns' }
 1077 
 1078   -- Translate
 1079   let dgs = T.mDecls tmodule
 1080   translateDeclGroups sc env' dgs

 1082 parseSchema ::
 1083   (?fileReader :: FilePath -> IO ByteString) =>
 1084   CryptolEnv -> InputText -> IO T.Schema
 1085 parseSchema env input = do
 1086   let modEnv = eModuleEnv env
 1087 
 1088   -- Parse
 1089   pschema <- ioParseSchema input
 1090 
 1091   fmap fst $ liftModuleM modEnv $ do
 1092 
 1093     -- Resolve names
 1094     let nameEnv = getNamingEnv env
 1095     rschema <- MM.interactive (MB.rename interactiveName nameEnv (MR.rename pschema))
 1096 
 1097     let ifDecls = getAllIfaceDecls modEnv
 1098     let range = fromMaybe P.emptyRange (P.getLoc rschema)
 1099     prims <- MB.getPrimMap
 1100     -- noIfaceParams because we don't support functors yet
 1101     tcEnv <- MB.genInferInput range prims NoParams ifDecls
 1102     let tcEnv' = tcEnv { TM.inpTSyns = Map.union (eExtraTSyns env) (TM.inpTSyns tcEnv) }
 1103     let infer =
 1104           case rschema of
 1105             P.Forall [] [] t _ -> do
 1106               let k = Nothing -- allow either kind KNum or KType
 1107               (t', goals) <- TM.collectGoals $ TK.checkType t k
 1108               return (T.Forall [] [] t', goals)
 1109             _ -> TK.checkSchema TM.AllowWildCards rschema
 1110     out <- MM.io (TM.runInferM tcEnv' infer)
 1111     (schema, _goals) <- MM.interactive (runInferOutput out)
 1112     --mapM_ (MM.io . print . TP.ppWithNames TP.emptyNameMap) goals
 1113     return (schemaNoUser schema)

 1129   case t of
 1130     T.TCon tc ts     -> T.TCon tc (map typeNoUser ts)

 1176 moduleCmdResult (res, ws) = do

 1179     Right (a, me) -> return (a, me)
 1180     Left err      -> errX' $ "Cryptol:" <+> CryPP.pretty err
 1181   where
 1182     -- If all warnings are about type defaults, pretend there are no warnings at

 1184     suppressDefaulting :: MM.ModuleWarning -> [MM.ModuleWarning]
 1185     suppressDefaulting w =
 1186       case w of
 1187         MM.RenamerWarnings xs -> [MM.RenamerWarnings xs]
 1188         MM.TypeCheckWarnings nm xs ->
 1189           case filter (notDefaulting . snd) xs of
 1190             [] -> []
 1191             xs' -> [MM.TypeCheckWarnings nm xs']
 1192 
 1193     notDefaulting :: TE.Warning -> Bool
 1194     notDefaulting (TE.DefaultingTo {}) = False
 1195     notDefaulting _ = True
