  217 -- | The workhorse for the @mir_alloc@ family of commands.
  218 mir_alloc_internal ::
  219   MirPointerKind ->
  220   Mir.Mutability ->
  221   Int ->
  222   Mir.Ty ->
  223   CrucibleSetup MIR SetupValue
  224 mir_alloc_internal pkind mut len mty = do
  225   st <- get
  226   let mcc = st ^. Setup.csCrucibleContext
  227       col = mcc ^. mccRustModule ^. Mir.rmCS ^. Mir.collection
  228 
  229   -- We disallow creating slice references (e.g., &[u8] or &str) using
  230   -- mir_alloc, as it is unclear what length to give the resulting slice
  231   -- value.
  232   case mty of
  233     Mir.TySlice _ ->
  234       fail $ unlines
  235         [ "mir_alloc and mir_alloc_mut cannot be used to create slice references."
  236         , "Use the mir_slice_value or mir_slice_range_value functions to take a"
  237         , "slice of an array reference instead."
  238         ]
  239     Mir.TyStr ->
  240       fail $ unlines
  241         [ "mir_alloc and mir_alloc_mut cannot be used to create str references."
  242         , "Use the mir_str_slice_value or mir_str_slice_range_value functions"
  243         , "to take a slice of a `u8` array reference instead."
  244         ]
  245     _ ->
  246       pure ()
  247 
  248   loc <- getW4Position "mir_alloc"
  249   Some tpr <- case Mir.tyToRepr col mty of
  250     Left err -> fail ("Unsupported type in mir_alloc: " ++ err)
  251     Right x -> return x
  252   n <- Setup.csVarCounter <<%= MS.nextAllocIndex
  253   tags <- view Setup.croTags
  254   let md = MS.ConditionMetadata
  255           { MS.conditionLoc = loc
  256           , MS.conditionTags = tags
  257           , MS.conditionType = "fresh allocation"
  258           , MS.conditionContext = ""
  259           }
  260   Setup.currentState . MS.csAllocs . at n ?=
  261     Some (MirAllocSpec { _maConditionMetadata = md
  262                       , _maType = tpr
  263                       , _maPtrKind = pkind
  264                       , _maMutbl = mut
  265                       , _maMirType = mty
  266                       , _maLen = len
  267                       })
  268   return (MS.SetupVar n)

  306 mir_execute_func :: [SetupValue] -> MIRSetupM ()
  307 mir_execute_func args =
  308   MIRSetupM $
  309   do st <- get
  310      let cc = st ^. Setup.csCrucibleContext
  311      let mspec = st ^. Setup.csMethodSpec
  312      let env = MS.csAllocations mspec
  313      let nameEnv = MS.csTypeNames mspec
  314      let argTys = mspec ^. MS.csArgs
  315      let
  316        checkArg i expectedTy val =
  317          do valTy <- typeOfSetupValue cc env nameEnv val
  318             unless (checkCompatibleTys expectedTy valTy) $
  319               X.throwM (MIRArgTypeMismatch i expectedTy valTy)
  320      let
  321        checkArgs _ [] [] = pure ()
  322        checkArgs i [] vals =
  323          X.throwM (MIRArgNumberWrong i (i + length vals))
  324        checkArgs i tys [] =
  325          X.throwM (MIRArgNumberWrong (i + length tys) i)
  326        checkArgs i (ty : tys) (val : vals) =
  327          do checkArg i ty val
  328             checkArgs (i + 1) tys vals
  329      checkArgs 0 argTys args
  330      Setup.crucible_execute_func args

  428 -- | See 'mir_fresh_expanded_val'.
  429 --
  430 -- This is the recursively-called worker function.
  431 constructExpandedSetupValue ::
  432   MIRCrucibleContext ->
  433   SharedContext ->
  434   Text ->
  435   TypeShape tp ->
  436   CrucibleSetup MIR SetupValue
  437 constructExpandedSetupValue cc sc = go
  438   where
  439     col = cc ^. mccRustModule . Mir.rmCS . Mir.collection
  440 
  441     go :: forall tp.
  442           Text ->
  443           TypeShape tp ->
  444           CrucibleSetup MIR SetupValue
  445     go pfx shp =
  446       case shp of
  447         PrimShape ty _ -> do
  448           fv <- freshPrimVariable pfx ty
  449           pure $ MS.SetupTerm fv
  450         TupleShape _ elems -> do
  451           flds <- mapM (goAgElem pfx) (zip [0..] elems)
  452           pure $ MS.SetupTuple () flds
  453         ArrayShape ty elemTy _ elemShp _ ->
  454           case ty of
  455             Mir.TyArray _ n -> do
  456               elems <-
  457                 traverse
  458                   (\i -> go (pfx <> "_" <> Text.pack (show i)) elemShp)
  459                   [0..n-1]
  460               pure $ MS.SetupArray elemTy elems
  461             _ ->
  462               panic "constructExpandedSetupValue" [
  463                   "ArrayShape with non-TyArray type: " ,
  464                   "   " <> Text.pack (show (PP.pretty ty))
  465               ]
  466         StructShape ty _ fldShps ->
  467           case ty of
  468             Mir.TyAdt adtName _ _ ->
  469               case col ^. Mir.adts . at adtName of
  470                 Just adt@(Mir.Adt _ kind _ _ _ _ _) ->
  471                   case kind of
  472                     Mir.Struct -> do
  473                       flds <- goFlds pfx fldShps
  474                       pure $ MS.SetupStruct adt flds
  475                     _ ->
  476                       panic "constructExpandedSetupValue" [
  477                           "Expected struct, encountered " <> Text.pack (show kind)
  478                       ]
  479                 Nothing ->
  480                   adt_not_found_panic "StructShape" adtName
  481             _ ->
  482               non_adt_type_panic "StructShape" ty
  483         EnumShape ty _ variantShps _ discrShp ->
  484           case ty of
  485             Mir.TyAdt adtName _ _ ->
  486               case col ^. Mir.adts . at adtName of
  487                 Just adt@(Mir.Adt _ kind _ _ _ _ _) ->
  488                   case kind of
  489                     Mir.Enum _ ->
  490                       MS.SetupEnum <$> goEnum pfx adt discrShp variantShps
  491                     Mir.Struct ->
  492                       panic "constructExpandedSetupValue" [
  493                           "Expected enum, encountered struct"
  494                       ]
  495                     Mir.Union ->
  496                       panic "constructExpandedSetupValue" [
  497                           "Expected enum, encountered union"
  498                       ]
  499                 Nothing ->
  500                   adt_not_found_panic "EnumShape" adtName
  501             _ ->
  502               non_adt_type_panic "EnumShape" ty
  503         TransparentShape ty shp' ->
  504           case ty of
  505             Mir.TyAdt adtName _ _ -> do
  506               case col ^. Mir.adts . at adtName of
  507                 Just adt@(Mir.Adt adtNm kind variants _ _ _ _) ->
  508                   case kind of
  509                     Mir.Struct -> do
  510                       val <- go pfx shp'
  511                       pure $ MS.SetupStruct adt [val]
  512                     Mir.Enum{}
  513                       -- `repr(transparent)` enum values use MirSetupEnumVariant
  514                       -- rather than MirSetupEnumSymbolic. See the Haddocks for
  515                       -- MirSetupEnumSymbolic for an explanation.
  516                       |  [variant] <- variants
  517                       -> do val <- go pfx shp'
  518                             pure $ MS.SetupEnum
  519                                  $ MirSetupEnumVariant adt variant 0 [val]
  520 
  521                       |  otherwise
  522                       -> panic "constructExpandedSetupValue" [
  523                              "`repr(transparent)` enum that doesn't have exactly one variant",
  524                              "Enum: " <> Text.pack (show adtNm),
  525                              "Number of variants: " <> Text.pack (show (length variants))
  526                          ]
  527                     Mir.Union ->
  528                       panic "constructExpandedSetupValue" [
  529                           "Unexpected `repr(transparent)` union: " <>
  530                               Text.pack (show adtName)
  531                       ]
  532                 Nothing ->
  533                   adt_not_found_panic "TransparentShape" adtName
  534             _ ->
  535               non_adt_type_panic "TransparentShape" ty
  536         RefShape ty _ _ _ ->
  537           X.throwM $ MIRFreshExpandedValueUnsupportedType ty
  538         SliceShape ty _ _ _ ->
  539           X.throwM $ MIRFreshExpandedValueUnsupportedType ty
  540         FnPtrShape ty _ _ ->
  541           X.throwM $ MIRFreshExpandedValueUnsupportedType ty

  543     -- Create a fresh symbolic enum value, as described in
  544     -- Note [Symbolic enums] in SAWCentral.Crucible.MIR.Setup.Value.
  545     goEnum ::
  546       forall discrShp variantCtx.
  547       Text ->
  548       Mir.Adt ->
  549       TypeShape discrShp ->
  550       Ctx.Assignment VariantShape variantCtx ->
  551       CrucibleSetup MIR MirSetupEnum
  552     goEnum pfx adt@(Mir.Adt _ _ variants _ _ _ _) discrShp variantShps =
  553       mccWithBackend cc $ \bak ->
  554       do -- First, create a symbolic discriminant value.
  555          IsBVShape discrTy discrW <- pure $ testDiscriminantIsBV discrShp
  556          let discrPfx = pfx <> "_discr"
  557          discrVar <- freshPrimVariable discrPfx discrTy
  558          let discrVal = MS.SetupTerm discrVar
  559 
  560          -- Next, add Crucible assumptions that constraint the discriminant
  561          -- to be equal to one of the possible variants' discriminant values.
  562          -- This assumption will be of the form:
  563          --
  564          --   (discr == 0) \/ (discr == 1) \/ ...
  565          --
  566          -- It's tempting to simplify this assumption to just
  567          -- `discr < num_variants`, but this will not work in the event that
  568          -- the enum uses explicit discriminants for some variants, e.g.,
  569          --
  570          --   enum E {
  571          --       E0 = 42,
  572          --       E1,
  573          --   }
  574          discrWNat <- liftIO $ scNat sc $ natValue discrW
  575          possibleDiscrTerms <- liftIO $
  576            traverse (\discr -> do
  577                       discrNat  <- scNat sc $ fromInteger discr
  578                       scBvNat sc discrWNat discrNat)
  579                     (map getEnumVariantDiscr variants)
  580          scFalse <- liftIO $ scBool sc False
  581          possibleDiscrPredTerm <- liftIO $
  582            foldM
  583              (\z possibleDiscrTerm -> do
  584                p <- scBvEq sc discrWNat (ttTerm discrVar) possibleDiscrTerm
  585                scOr sc p z)
  586              scFalse
  587              possibleDiscrTerms
  588          possibleDiscrPred <- liftIO $ resolveSAWPred cc possibleDiscrPredTerm
  589          loc <- SS.toW4Loc "mir_fresh_expanded_value" <$> lift (lift getPosition)
  590          liftIO $ Crucible.addAssumption bak $
  591            Crucible.GenericAssumption
  592              loc "Symbolic enum discriminant constraints" possibleDiscrPred
  593 
  594          -- Finally, create symbolic fields for each of the possible variants.
  595          let variantAssns :: [Some (Ctx.Assignment FieldShape)]
  596              variantAssns =
  597                FC.toListFC
  598                  (\(VariantShape fldShps) -> Some fldShps)
  599                  variantShps
  600          variantVals <-
  601            zipWithM
  602              (\variant (Some fldShps) ->
  603                let variantPfx = pfx <> "_" <> getEnumVariantShortName variant in
  604                goFlds variantPfx fldShps)
  605              variants
  606              variantAssns
  607 
  608          pure $ MirSetupEnumSymbolic adt discrVal variantVals

  610     goFlds :: forall ctx.
  611               Text ->
  612               Ctx.Assignment FieldShape ctx ->
  613               CrucibleSetup MIR [SetupValue]

  615       FCI.itoListFC
  616         (\idx fldShp ->
  617           let pfx' = pfx <> "_" <> Text.pack (show idx) in

  619             ReqField shp' -> go pfx' shp'

  807 -- | Perform a set of validity checks on the LHS reference or pointer value in a
  808 -- 'mir_points_to' or 'mir_points_to_multi' command. In particular:
  809 --
  810 -- * Check that the LHS is the expected type (reference or raw pointer for
  811 --   'mir_points_to', and only raw pointer for 'mir_points_to_multi').
  812 -- * Make sure that it does not contain any casts.
  813 --
  814 -- Returns the 'Mir.Ty' that the LHS points to.
  815 mir_points_to_check_lhs_validity ::
  816   SetupValue {- ^ LHS reference/pointer -} ->
  817   W4.ProgramLoc {- ^ The location in the program -} ->
  818   MirPointsToMode {- ^ Which kind of @mir_points_to@ -} ->
  819   Setup.CrucibleSetupT MIR TopLevel Mir.Ty
  820 mir_points_to_check_lhs_validity ref loc mode =
  821   do cc <- getMIRCrucibleContext
  822      st <- get
  823      let env = MS.csAllocations (st ^. Setup.csMethodSpec)
  824          nameEnv = MS.csTypeNames (st ^. Setup.csMethodSpec)
  825      refTy <- typeOfSetupValue cc env nameEnv ref
  826      when (containsCast ref) $
  827        throwCrucibleSetup loc $
  828          "The first argument of " ++ mirPointsToCommandName mode
  829          ++ " must not contain any casts"
  830      let throwWrongType acceptedType =
  831            throwCrucibleSetup loc $ "lhs not a " ++ acceptedType ++ ": "
  832                                  ++ show (PP.pretty refTy)
  833      case mode of
  834        PointsToSingle ->
  835          case refTy of
  836            Mir.TyRef referentTy _ -> pure referentTy
  837            Mir.TyRawPtr referentTy _ -> pure referentTy
  838            _ -> throwWrongType "reference or raw pointer type"
  839        PointsToMulti ->
  840          case refTy of
  841            Mir.TyRawPtr referentTy _ -> pure referentTy
  842            _ -> throwWrongType "raw pointer type"

  887 mir_verify ::
  888   Mir.RustModule ->
  889   Text {- ^ method name -} ->
  890   [Lemma] {- ^ overrides -} ->
  891   Bool {- ^ path sat checking -} ->
  892   MIRSetupM () ->
  893   ProofScript () ->
  894   TopLevel Lemma
  895 mir_verify rm nm lemmas checkSat setup tactic =
  896   do start <- io getCurrentTime
  897      opts <- getOptions
  898 
  899      -- set up the metadata map for tracking proof obligation metadata
  900      mdMap <- io $ newIORef mempty
  901 
  902      cc <- setupCrucibleContext rm
  903      SomeOnlineBackend bak <- pure (cc^.mccBackend)
  904      let sym = cc^.mccSym
  905      let globals0 = cc^.mccSymGlobalState
  906 
  907      sosp <- rwSingleOverrideSpecialCase <$> getTopLevelRW
  908      let ?singleOverrideSpecialCase = sosp
  909 
  910      pos <- getPosition
  911      let loc = SS.toW4Loc "_SAW_mir_verify" pos
  912 
  913      profFile <- rwProfilingFile <$> getTopLevelRW
  914      (writeFinalProfile, pfs) <- io $ setupProfiling sym "mir_verify" profFile
  915 
  916      fn <- findFn rm nm
  917      let st0 = initialCrucibleSetupState cc fn loc
  918 
  919      -- execute commands of the method spec
  920      io $ W4.setCurrentProgramLoc sym loc
  921      methodSpec <- execMIRSetup setup st0
  922 
  923      printOutLnTop Info $
  924        unwords ["Verifying", show (methodSpec ^. MS.csMethod), "..."]
  925 
  926      -- construct the initial state for verifications
  927      (args, assumes, env, globals1) <- io $ verifyPrestate cc methodSpec globals0
  928 
  929      -- save initial path conditions
  930      frameIdent <- io $ Crucible.pushAssumptionFrame bak
  931 
  932      -- check for contradictory preconditions
  933      when (detectVacuity opts) $
  934        Vacuity.checkAssumptionsForContradictions
  935          sym
  936          methodSpec
  937          tactic
  938          assumes
  939 
  940      -- run the symbolic execution
  941      printOutLnTop Info $
  942        unwords ["Simulating", show (methodSpec ^. MS.csMethod), "..."]
  943      top_loc <- SS.toW4Loc "mir_verify" <$> getPosition
  944      (ret, globals2) <-
  945        io $ verifySimulate opts cc pfs methodSpec args assumes top_loc lemmas globals1 checkSat mdMap
  946 
  947      -- collect the proof obligations
  948      asserts <- verifyPoststate cc
  949                     methodSpec env globals2 ret mdMap
  950 
  951      -- restore previous assumption state
  952      _ <- io $ Crucible.popAssumptionFrame bak frameIdent
  953 
  954      -- attempt to verify the proof obligations
  955      printOutLnTop Info $
  956        unwords ["Checking proof obligations", show (methodSpec ^. MS.csMethod), "..."]
  957      (stats,vcstats) <- verifyObligations cc methodSpec tactic assumes asserts
  958      io $ writeFinalProfile
  959 
  960      let lemmaSet = Set.fromList (map (view MS.psSpecIdent) lemmas)
  961      end <- io getCurrentTime
  962      let diff = diffUTCTime end start
  963      ps <- io (MS.mkProvedSpec MS.SpecProved methodSpec stats vcstats lemmaSet diff)
  964      returnMIRProof ps

 1054   Text ->
 1055   Mir.Ty ->
 1056   SetupValue ->

 1058 mir_vec_of prefix elemTy contents = do
 1059   st <- MIRSetupM get
 1060   let cc = st ^. Setup.csCrucibleContext
 1061       mspec = st ^. Setup.csMethodSpec
 1062       env = MS.csAllocations mspec
 1063       nameEnv = MS.csTypeNames mspec
 1064 
 1065   -- Check the type of contents
 1066   contentsTy <- MIRSetupM $ typeOfSetupValue cc env nameEnv contents
 1067   case contentsTy of
 1068     Mir.TyArray elemTy' len -> do
 1069 
 1070       -- Ensure array element type matches passed-in type
 1071       unless (checkCompatibleTys elemTy elemTy') $
 1072         MIRSetupM $ X.throwM $ MIRVecOfTypeMismatch elemTy elemTy'
 1073 
 1074       -- Find all the internal ADTs used in Vec
 1075       let rm = cc ^. mccRustModule
 1076           col = rm ^. Mir.rmCS . Mir.collection
 1077           find name tyArgs = mirTopLevel $ mir_find_adt rm name tyArgs
 1078       globalAllocAdt <-
 1079         find globalAllocId []
 1080       vecTAdt <-
 1081         find vecId [elemTy, mir_adt globalAllocAdt]
 1082       rawVecTAdt <-
 1083         find "alloc::raw_vec::RawVec" [elemTy, mir_adt globalAllocAdt]
 1084       typedAllocatorTAdt <-
 1085         find "crucible::alloc::TypedAllocator" [elemTy]

 1087         find "alloc::raw_vec::RawVecInner" [mir_adt typedAllocatorTAdt]

 1089         find "core::ptr::unique::Unique" [mir_u8]

 1091         find "core::ptr::non_null::NonNull" [mir_u8]
 1092       phantomDataU8Adt <-
 1093         find "core::marker::PhantomData" [mir_u8]

 1095         find "core::marker::PhantomData" [elemTy]
 1096       usizeNoHighBitAdt <-
 1097         find "core::num::niche_types::UsizeNoHighBit" []

 1100       sc <- mirTopLevel getSharedContext
 1101       let transCry cryEnv =
 1102             let ?fileReader = BSS.readFile
 1103             in  MIRSetupM . liftIO . CryEnv.pExprToTypedTerm sc cryEnv
 1104       cryEnv <- mirTopLevel getCryptolEnv
 1105       let sizeBits = knownNat @Mir.SizeBits
 1106 
 1107       -- Declare symbolic variables and assertions
 1108       ptr <- mir_alloc_raw_ptr_const_multi len elemTy
 1109       mir_points_to_multi ptr contents
 1110       cap <-
 1111         case Map.lookup elemTy (col ^. Mir.layouts) of
 1112           Just (Just Mir.Layout { Mir._laySize = elemSize })
 1113             | elemSize == 0 ->
 1114               -- The ZST case won't work yet at least until crucible issues

 1121             | otherwise -> do

 1124                   cryEnv' = CryEnv.bindTypedTerm (capIdent, cap) cryEnv

 1126               -- cap <= isize::MAX / sizeof::<elemTy>

 1132               pure cap
 1133           Just Nothing -> MIRSetupM $ X.throwM $ MIRVecOfElemTyNotSized elemTy

 1137

 1140             MS.SetupStruct vecTAdt
 1141                            [rawVec, MS.SetupTerm lenTerm]

 1150                 where
 1151                 uniqueU8Ptr =
 1152                   MS.SetupStruct uniqueU8Adt
 1153                                  [nonNullU8Ptr, phantomDataU8]
 1154                   where
 1155                   nonNullU8Ptr =
 1156                     MS.SetupStruct nonNullU8Adt
 1157                                    [u8Ptr]
 1158                     where
 1159                     u8Ptr = mir_cast_raw_ptr ptr mir_u8
 1160                 capNoHighBit =
 1161                   MS.SetupStruct usizeNoHighBitAdt
 1162                                  [MS.SetupTerm cap]
 1163                 typedAllocator =
 1164                   MS.SetupStruct typedAllocatorTAdt

 1512   MIRCrucibleContext                        {- ^ crucible context        -} ->
 1513   MethodSpec                                {- ^ specification           -} ->
 1514   Map MS.AllocIndex (Some (MirPointer Sym)) {- ^ allocation substitution -} ->
 1515   Crucible.SymGlobalState Sym               {- ^ global variables        -} ->
 1516   Maybe MIRVal                              {- ^ optional return value   -} ->
 1517   IORef MetadataMap                         {- ^ metadata map            -} ->
 1518   TopLevel [(String, MS.ConditionMetadata, Term)] {- ^ generated labels and verification conditions -}
 1519 verifyPoststate cc mspec env0 globals ret mdMap =
 1520   mccWithBackend cc $ \bak ->
 1521   do opts <- getOptions
 1522      sc <- getSharedContext
 1523      poststateLoc <- SS.toW4Loc "_SAW_MIR_verifyPoststate" <$> getPosition
 1524      io $ W4.setCurrentProgramLoc sym poststateLoc
 1525 
 1526      -- This discards all the obligations generated during
 1527      -- symbolic execution itself, i.e., which are not directly
 1528      -- generated from specification postconditions. This
 1529      -- is, in general, unsound.
 1530      skipSafetyProofs <- gets rwSkipSafetyProofs
 1531      when skipSafetyProofs (io (Crucible.clearProofObligations bak))
 1532 
 1533      let vars0 = IntMap.fromList
 1534            [ (vnIndex vn, (vn, tvType tt))
 1535            | tt <- mspec ^. MS.csPreState . MS.csFreshVars
 1536            , let vn = tvName tt ]
 1537      terms0 <- io $ scVariables sc vars0
 1538 
 1539      let initialFree = Set.fromList (map (vnIndex . tvName)
 1540                                     (view (MS.csPostState . MS.csFreshVars) mspec))
 1541      matchPost <- io $
 1542           runOverrideMatcher sym globals env0 terms0 initialFree poststateLoc $
 1543            do matchResult opts sc
 1544               learnCond opts sc cc mspec MS.PostState (mspec ^. MS.csPostState)
 1545 
 1546      st <- case matchPost of
 1547              Left err      -> fail (show err)
 1548              Right (_, st) -> return st
 1549      io $ forM_ (view osAsserts st) $ \(md, Crucible.LabeledPred p r) ->
 1550        do (ann,p') <- W4.annotateTerm sym p
 1551           modifyIORef mdMap (Map.insert ann md)
 1552           Crucible.addAssertion bak (Crucible.LabeledPred p' r)
 1553 
 1554      finalMdMap <- io $ readIORef mdMap
 1555      obligations <- io $ Crucible.getProofObligations bak
 1556      io $ Crucible.clearProofObligations bak
 1557      io $ mapM (verifyObligation sc finalMdMap) (maybe [] Crucible.goalsToList obligations)
 1558 
 1559   where
 1560     sym = cc^.mccSym
 1561 
 1562     verifyObligation sc finalMdMap
 1563       (Crucible.ProofGoal hyps (Crucible.LabeledPred concl simErr)) =
 1564       do st         <- sawCoreState sym
 1565          hypTerm <- toSC sym st =<< Crucible.assumptionsPred sym hyps
 1566          conclTerm  <- toSC sym st concl
 1567          obligation <- scImplies sc hypTerm conclTerm
 1568          let loc = Crucible.simErrorLoc simErr
 1569          let err = Crucible.simErrorReason simErr
 1570          let defaultMd = MS.ConditionMetadata
 1571                          { MS.conditionLoc = loc
 1572                          , MS.conditionTags = mempty
 1573                          , MS.conditionType = "safety assertion"
 1574                          , MS.conditionContext = ""
 1575                          }
 1576          let md = fromMaybe defaultMd $
 1577                     do ann <- W4.getAnnotation sym concl
 1578                        Map.lookup ann finalMdMap
 1579          return (Crucible.simErrorReasonMsg err, md, obligation)
 1580 
 1581     matchResult opts sc =
 1582       case (mspec ^. MS.csRet, ret, mspec ^. MS.csRetValue) of
 1583 
 1584         -- Non-unit function:
 1585         --
 1586         -- mir_return has *already* checked that the user-specified SetupValue

 1589         -- What mir_return does NOT check is whether the *actual* value produced

 1592         -- This branch enforces that semantic obligation: it compares the

 1601                    , MS.conditionContext = ""

 1814   Options ->
 1815   SharedContext ->
 1816   MIRCrucibleContext ->
 1817   -- | The CFG's MIR argument types.
 1818   [Mir.Ty] ->
 1819   -- | The CFG's MIR return type.

 1856 -- Find the ADT definition that is monomorphized from `origName` with `substs`.
 1857 -- This should only be used on types that are known to be present in the crate
 1858 -- after dead code elimination - for example, because the type appears in the
 1859 -- signature of a function that's being translated.
 1860 findAdt :: Mir.Collection -> Mir.DefId -> Mir.Substs -> TopLevel Mir.Adt

 1862     case List.find (\adt -> adt ^. Mir.adtOrigSubsts == substs) insts of
 1863         Just x -> return x
 1864         Nothing -> fail $ "Unknown ADT: " ++ show (origName, substs)

 1866     insts = col ^. Mir.adtsOrig . at origName . to (fromMaybe [])

 1868 findFnInstance :: MonadFail m => Mir.Collection -> Mir.DefId -> Mir.Substs -> m Text
 1869 findFnInstance col origName substs =
 1870   case found of
 1871     Just i -> pure (Mir.idText i)
 1872     Nothing
 1873       | hasInstances -> fail ("Could not find function instance: " ++ show (Mir.cleanVariantName origName))
 1874       | origName `Map.member` (col ^. Mir.functions) -> pure (Mir.idText origName)
 1875       | otherwise -> fail ("Could not find function " ++ show (Mir.cleanVariantName origName))
 1876   where
 1877   (hasInstances, found) = foldr check (False, Nothing) (col ^. Mir.intrinsics)
 1878   check i (hasIs, rest) =
 1879     let inst = i ^. Mir.intrInst
 1880     in case inst ^. Mir.inKind of
 1881           Mir.IkItem
 1882             | inst ^. Mir.inDefId == origName ->
 1883               (True, if inst ^. Mir.inSubsts == substs
 1884                        then Just (i ^. Mir.intrName)
 1885                        else rest)
 1886           _ -> (hasIs, rest)

 2332 {-
 2333 Note [Translating MIR statics in SAW]
 2334 ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
 2335 Translating top-level static values in the SAW MIR backend requires some care.
 2336 This is because we want to treat immutable and mutable static differently:
 2337 
 2338 * Immutable statics should be implicitly initialized in each specification
 2339   without any action required on the user's end.
 2340 
 2341 * Mutable statics should /not/ be initialized in a specification unless a user
 2342   explicitly declares it with `mir_points_to (mir_static ...) ...`. This is
 2343   because a mutable static's value can change over the course of a program, so
 2344   we require users to be precise about what the value should be before and after
 2345   invoking a function.
 2346 
 2347 This poses a challenge when translating static values in SAW. It is tempting to
 2348 only translate the immutable statics and not the mutable statics (similarly to
 2349 how the SAW LLVM backend handles translation), but this will not work in
 2350 general. Consider this program:
 2351 
 2352   static mut S1: u32 = 1;
 2353   static S2: u32 = unsafe { S1 };
 2354 
 2355   pub fn f() {
 2356       ... S2 ...
 2357   }
 2358 
 2359 The initial value of S2 (an immutable static) depends on having first translated
 2360 the initial value of S1 (a mutable static). If we do not translate mutable
 2361 statics in SAW, the translation of S1 will not succeed.
 2362 
 2363 We solve this problem by doing the following:
 2364 
 2365 1. Use Crucible to translate all top-level static values (both immutable and
 2366    mutable). This produces a SymGlobalState that contains every static's
 2367    initializer value.
 2368 
 2369 2. Consult the updated SymGlobalState to produce the following:
 2370 
 2371    (a) A subset of the SymGlobalState that only contains the initializer values
 2372        for the immutable statics. This will be used later to initialize the
 2373        Crucible state when translating the function being verified with
 2374        mir_verify.
 2375 
 2376    (b) A MirStaticInitializerMap mapping all static variables (both immutable
 2377        and mutable) to their initializer values. This will be used later to
 2378        power the mir_static_initializer function.
 2379 
 2380 This is a bit clunky, but it's unclear how to do better without significantly
 2381 changing how crucible-mir translates top-level static values.
 2382 -}
