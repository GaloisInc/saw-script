   94 -----------------------------------------------------------------------
   95 -- | Make sure the class is in the database and allocate handles for its
   96 -- methods and static fields
   97 --
   98 loadJavaClass :: String -> TopLevel J.Class
   99 loadJavaClass str = do
  100   cb <- getJavaCodebase
  101   c <- io $ findClass cb str
  102   prepareClassTopLevel str
  103   return c
  104 
  105 -----------------------------------------------------------------------
  106 -- | Allocate the method handles/global static variables for the given
  107 -- class and add them to the current translation context
  108 prepareClassTopLevel :: String -> TopLevel ()
  109 prepareClassTopLevel str = do
  110 
  111    cb <- getJavaCodebase
  112 
  113    -- get class from codebase
  114    c <- io $ findClass cb str
  115 
  116    -- get current ctx
  117    ctx0 <- getJVMTrans
  118 
  119    -- make sure that we haven't already processed this class
  120    unless (Map.member (J.className c) (CJ.classTable ctx0)) $ do
  121 
  122      -- add handles/global variables for this class
  123      halloc <- getHandleAlloc
  124      ctx <- io $ execStateT (CJ.extendJVMContext halloc c) ctx0
  125 
  126      -- update ctx
  127      putJVMTrans ctx
  128 
  129

  133 -- | Extract a JVM method to saw-core
  134 --
  135 jvm_extract :: J.Class -> String -> TopLevel TypedTerm
  136 jvm_extract c mname = do
  137   sc <- getSharedContext
  138   cb <- getJavaCodebase
  139   opts <- getOptions
  140   pathSatSolver <- gets rwPathSatSolver
  141   let verbosity = simVerbose opts
  142   let gen       = Nonce.globalNonceGenerator
  143 
  144 
  145   traceM $ "extracting " ++ mname
  146   (mcls, meth) <- io $ CJ.findMethod cb mname c
  147   when (not (J.methodIsStatic meth)) $ do
  148        fail $ unlines [ "Crucible can only extract static methods" ]
  149 
  150   let className = J.className c
  151 
  152   -- allocate all of the handles/static vars that are directly referenced by
  153   -- this class
  154   let refs = CJ.initClasses ++ Set.toList (CJ.classRefs c)
  155   mapM_ (prepareClassTopLevel . J.unClassName) refs
  156 
  157   halloc <- getHandleAlloc
  158   ctx <- getJVMTrans
  159 
  160   io $ do -- only the IO monad, nothing else
  161           sym <- newSAWCoreExprBuilder sc False
  162           SomeOnlineBackend bak <- newSAWCoreBackend pathSatSolver sym
  163           st  <- sawCoreState sym
  164           CJ.setSimulatorVerbosity verbosity sym
  165 
  166           (CJ.JVMHandleInfo _m2 h) <- CJ.findMethodHandle ctx mcls meth
  167 
  168           (ecs, args) <- setupArgs sc sym h
  169 
  170           res <- CJ.runMethodHandle bak SAWCruciblePersonality halloc
  171                      ctx verbosity className h args
  172 
  173           case res of
  174             Crucible.FinishedResult _ pr -> do
  175               gp <- getGlobalPair opts pr
  176               let regval = gp^.Crucible.gpValue
  177               let regty = Crucible.regType regval
  178               let failure :: forall a. IO a
  179                   failure = fail $ unwords ["Unexpected return type:", show regty]
  180               t <- Crucible.asSymExpr regval (toSC sym st) failure
  181               cty <-
  182                 case Crucible.asBaseType regty of
  183                   Crucible.NotBaseType -> failure
  184                   Crucible.AsBaseType bt ->
  185                     case baseCryptolType bt of
  186                       Nothing -> failure
  187                       Just cty -> return cty
  188               let tt = TypedTerm (TypedTermSchema (Cryptol.tMono cty)) t
  189               abstractTypedExts sc (toList ecs) tt
  190             Crucible.AbortedResult _ _ar -> do
  191               fail $ unlines [ "Symbolic execution failed." ]
  192             Crucible.TimeoutResult _cxt -> do
  193               fail $ unlines [ "Symbolic execution timed out." ]
