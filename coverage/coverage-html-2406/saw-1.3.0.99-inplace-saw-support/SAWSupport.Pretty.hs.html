   20 -- Prettyprinting infrastructure.
   21 --
   22 -- This module is meant to be used like this:
   23 --    import qualified SAWSupport.Pretty as PPS
   24 --
   25 -- or if you like, it's reasonable to import the print functions unqualified:
   26 --    import SAWSupport.Pretty (ppNat, ppTypeConstraint)
   27 --    import qualified SAWSupport.Pretty as PPS
   28 --
   29 -- This module is a haphazard collection of prettyprinting
   30 -- infrastructure that used to be scattered in random places around
   31 -- the tree. Most of it came from (a) saw-core's Term prettyprinter,
   32 -- (b) saw-script's Value prettyprinter, or (c) saw-script's AST
   33 -- prettyprinter, none of which shared any logic other than the
   34 -- upstream prettyprinter package.
   35 --
   36 -- XXX: there's more in MrSolver but it will require untangling.
   37 --
   38 -- XXX: Someone should try to systematize the naming. In particular
   39 -- we'd like to have some kind of convention like (perhaps)
   40 -- "prettyFoo" generates a Doc for a Foo and "ppFoo" generates a Doc
   41 -- and also renders it to Text. (Note that in the prettyprinter
   42 -- package, "pretty" generates Doc and there is no "pp", so this
   43 -- choice seems preferable to the other way around.) I've done some
   44 -- of this, but there's a lot more out in the rest of the tree that
   45 -- ought to be updated to match it.
   46 --
   47 -- XXX: should come up with a better name for pShowText, and remove
   48 -- pShow entirely. These shouldn't be called "show" because they
   49 -- aren't Show instances and don't have the properties of Show
   50 -- instances. And we don't want people writing or using Show instances
   51 -- except maybe for extremely simple types.
   52 --
   53 -- XXX: The ShowS material at the bottom should go away; that requires
   54 -- cleaning up the callers to use prettyprinting instead of Show
   55 -- instances.
   56 --
   57 -- XXX: Also we should crack down on callers using "show" instead of
   58 -- "render" to produce output text.
   59 --
   60 -- It would also be nice in the long run to be able to do
   61 --    import qualified Prettyprinter as PP
   62 --    import qualified SAWSupport.Pretty as PP
   63 -- without this causing problems.
   64 --
   65 -- It may also be that this should be split into two or more modules,
   66 -- one meant to be used qualified and one meant for selective import,
   67 -- or something. It isn't at all clear yet what a good interface is.
   68 --
   69 -- FUTURE: there's also been some talk of adopting Cryptol's technique
   70 -- of wrapping the upstream prettyprinter and exposing a monad, or
   71 -- some other similar way of allowing options to be applied to the
   72 -- document after it's assembled instead of needing to be threaded
   73 -- through everywhere.
   74 --
   75 -- XXX: in any event, it's problematic that seemingly more than half
   76 -- of the call sites for things that want options pass the default
   77 -- options. If we're actually going to support the things the options
   78 -- allows the user to set, we should make sure the user's settings
   79 -- make it to the places that do prints. This includes errors.
   80 --

  120 data Style
  121   = PrimitiveStyle
  122   | ConstantStyle
  123   | ExtCnsStyle
  124   | LocalVarStyle
  125   | DataTypeStyle
  126   | CtorAppStyle
  127   | RecursorStyle
  128   | FieldNameStyle
  129 
  130 -- TODO: assign colors for more styles
  131 -- FUTURE: is it possible to get non-angry-fruit-salad colors?
  132 colorStyle :: Style -> PP.AnsiStyle
  133 colorStyle =
  134   \case
  135     PrimitiveStyle -> mempty
  136     ConstantStyle -> PP.colorDull PP.Blue
  137     ExtCnsStyle -> PP.colorDull PP.Red
  138     LocalVarStyle -> PP.colorDull PP.Green
  139     DataTypeStyle -> mempty
  140     CtorAppStyle -> mempty
  141     RecursorStyle -> mempty
  142     FieldNameStyle -> mempty
  143 
  144 type Doc = PP.Doc Style

  148 -- Options
  149 
  150 -- | Global options for pretty-printing
  151 data Opts = Opts {
  152     -- | Passed to the "useAscii" setting of Cryptol's prettyprinter.
  153     --   Default is false.
  154     ppUseAscii :: Bool,
  155 
  156     -- | The base to print integers in; default is 10.
  157     ppBase :: Int,
  158 
  159     -- | Whether to print in color; default is false.
  160     ppColor :: Bool,
  161 
  162     -- | Whether to show the names of local variables. Default is
  163     --   true. If set to false, prints the deBruijn indexes instead.
  164     ppShowLocalNames :: Bool,
  165 
  166     -- | Maximum depth to recurse into terms. If not set, no limit.
  167     --   Default is unset.
  168     ppMaxDepth :: Maybe Int,
  169 
  170     -- | The numeric identifiers, as seen in the 'memoFresh' field of
  171     --   'MemoVar', of SAWCore variables that shouldn't be inlined.
  172     ppNoInlineMemoFresh :: [Int],
  173 
  174     -- | The way to display SAWCore memoization variables.
  175     --   Default is Incremental.
  176     ppMemoStyle :: MemoStyle,
  177 
  178     -- | Minimum sharing level required to memoize SAWCore subterms.
  179     --   Default is 2 (i.e., any sharing).
  180     ppMinSharing :: Int
  181  }
  182 
  183 -- | How should memoization variables be displayed?
  184 --
  185 -- Note: actual text stylization is the province of 'ppMemoVar', this just
  186 -- describes the semantic information 'ppMemoVar' should be prepared to display.
  187 data MemoStyle
  188   = Incremental
  189   -- ^ 'Incremental' says to display a term's memoization variable with the
  190   -- value of a counter that increments after a term is memoized. The first
  191   -- term to be memoized will be displayed with '1', the second with '2', etc.
  192   | Hash Int
  193   -- ^ 'Hash i' says to display a term's memoization variable with the first 'i'
  194   -- digits of the term's hash.
  195   | HashIncremental Int
  196   -- ^ 'HashIncremental i' says to display a term's memoization variable with
  197   -- _both_ the first 'i' digits of the term's hash _and_ the value of the
  198   -- counter described in 'Incremental'.
  199 
  200 -- | Default options for pretty-printing.
  201 --
  202 --   If the default 'ppMemoStyle' changes, be sure to update the help
  203 --   text in the interpreter functions that control the memoization
  204 --   style to reflect this change to users.
  205 defaultOpts :: Opts
  206 defaultOpts = Opts {
  207     ppUseAscii = False,
  208     ppBase = 10,
  209     ppColor = False,
  210     ppShowLocalNames = True,
  211     ppMaxDepth = Nothing,
  212     ppNoInlineMemoFresh = mempty,
  213     ppMemoStyle = Incremental,
  214     ppMinSharing = 2
  215  }
  216 
  217 limitMaxDepth :: Opts -> Int -> Opts
  218 limitMaxDepth opts limit =
  219   let adjust depth = case depth of
  220         Nothing -> Just limit
  221         Just d -> Just $ if d > limit then limit else d
  222   in
  223   opts { ppMaxDepth = adjust $ ppMaxDepth opts }

  234 -- Common prettyprint operations
  235 -- (for base types and common constructs not tied to any particular AST)
  236 
  237 -- | Pretty-print an integer in the correct base
  238 prettyNat :: Opts -> Integer -> Doc
  239 prettyNat (Opts{..}) i
  240   | ppBase > 36 = pretty i
  241   | otherwise = prefix <> pretty value
  242   where
  243     prefix = case ppBase of
  244       2  -> "0b"
  245       8  -> "0o"
  246       10 -> mempty
  247       16 -> "0x"
  248       _  -> "0" <> pretty '<' <> pretty ppBase <> pretty '>'
  249 
  250     value  = showIntAtBase (toInteger ppBase) (digits !!) i ""
  251     digits = "0123456789abcdefghijklmnopqrstuvwxyz"
  252 
  253 -- | Pretty-print a type constraint (also known as an ascription) @x : tp@
  254 --   This is the formatting used by SAWCore.
  255 prettyTypeConstraint :: Doc -> Doc -> Doc
  256 prettyTypeConstraint x tp =
  257   PP.hang 2 $ PP.group $ PP.vsep [PP.annotate LocalVarStyle x, ":" <+> tp]
  258 
  259 -- | Pretty-print a type signature.
  260 --   This is the formatting used by SAWScript.
  261 --   XXX: should probably unify with prettyTypeConstraint
  262 prettyTypeSig :: PP.Doc ann -> PP.Doc ann -> PP.Doc ann
  263 prettyTypeSig n t = n <+> PP.pretty ':' <+> t
  264 
  265 -- | Concatenate n copies of a doc.
  266 replicate :: Integer -> PP.Doc ann -> PP.Doc ann
  267 replicate n d
  268   | n < 1 = PP.emptyDoc
  269   | True  = d <> replicate (n-1) d
  270 
  271 -- | Add a comma to d1 and append d2.
  272 --   (Functionally internal to commaSepAll.)
  273 commaSepDoc :: PP.Doc ann -> PP.Doc ann -> PP.Doc ann
  274 commaSepDoc d1 d2 = d1 <> PP.comma <+> d2
  275 
  276 -- | Concatenate ds, appending a comma to each.
  277 commaSepAll :: [PP.Doc ann] -> PP.Doc ann
  278 commaSepAll ds = case ds of
  279   [] -> PP.emptyDoc
  280   _  -> foldl1 commaSepDoc ds

  283 ------------------------------------------------------------
  284 -- Render documents
  285 
  286 render :: Opts -> Doc -> String
  287 render opts doc =
  288   TextL.unpack (PP.renderLazy (style (PP.layoutPretty layoutOpts doc)))
  289   where
  290     -- ribbon width 64, with effectively unlimited right margin
  291     layoutOpts = PP.LayoutOptions (PP.AvailablePerLine 8000 0.008)
  292     style = if ppColor opts then PP.reAnnotateS colorStyle else PP.unAnnotateS
  293 
  294 pShow :: PrettyPrec a => a -> String
  295 pShow = show . prettyPrec 0
  296 
  297 pShowText :: PrettyPrec a => a -> Text
  298 pShowText = PPT.renderStrict . PP.layoutPretty PP.defaultLayoutOptions . prettyPrec 0
