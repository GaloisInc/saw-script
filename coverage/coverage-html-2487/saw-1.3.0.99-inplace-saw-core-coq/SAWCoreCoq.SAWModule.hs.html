   50   M.TranslationConfiguration ->
   51   Maybe ModuleName ->

   58 dropPi :: Coq.Term -> Coq.Term
   59 dropPi (Coq.Pi (_ : t) r) = Coq.Pi t r
   60 dropPi (Coq.Pi _       r) = dropPi r
   61 dropPi t                  = t

   63 translateCtor ::
   64   ModuleTranslationMonad m =>
   65   [Coq.Binder] -> -- list of parameters to drop from `ctorType`
   66   Ctor -> m Coq.Constructor
   67 translateCtor inductiveParameters (Ctor {..}) = do
   68   maybe_constructorName <-
   69     case nameInfo ctorName of
   70       ModuleIdentifier ident -> liftTermTranslationMonad $ TermTranslation.translateIdentToIdent ident
   71       ImportedName{} -> pure Nothing
   72   let constructorName = case maybe_constructorName of
   73         -- Drop qualifiers from constructor name
   74         Just (Coq.Ident n) -> Coq.Ident (reverse (takeWhile (/= '.') (reverse n)))
   75         Nothing -> error "translateCtor: unexpected translation for constructor"
   76   constructorType <-
   77     -- Unfortunately, `ctorType` qualifies the inductive type's name in the
   78     -- return type.
   79     -- dropModuleNameWithinCtor <$>
   80     -- Unfortunately, `ctorType` comes with the inductive parameters universally
   81     -- quantified.
   82     (\ t -> iterate dropPi t !! length inductiveParameters) <$>
   83     (liftTermTranslationMonad (TermTranslation.translateTerm ctorType))
   84   return $ Coq.Constructor
   85     { constructorName
   86     , constructorType
   87     }

   89 translateDataType :: ModuleTranslationMonad m => DataType -> m Coq.Decl
   90 -- translateDataType (DataType {..})
   91 --   | trace ("translateDataType: " ++ show dtName) False = undefined
   92 translateDataType (DataType {..}) =
   93   case nameInfo dtName of
   94     ModuleIdentifier dtIdent ->
   95       atDefSite <$> findSpecialTreatment dtIdent >>= \case
   96       DefPreserve            -> translateNamed $ Coq.Ident (identName dtIdent)
   97       DefRename   targetName -> translateNamed $ targetName
   98       DefReplace  str        -> return $ Coq.Snippet str
   99       DefSkip                -> return $ skipped dtIdent
  100     ImportedName{} ->
  101       translateNamed $ Coq.Ident (Text.unpack (toShortName (nameInfo dtName)))
  102   where
  103     translateNamed :: ModuleTranslationMonad m => Coq.Ident -> m Coq.Decl
  104     translateNamed name = do
  105       let inductiveName = name
  106       (inductiveParameters, inductiveIndices) <-
  107         liftTermTranslationMonad $
  108         TermTranslation.translateParamsEC dtParams $ \ps ->
  109         TermTranslation.translateParamsEC dtIndices $ \ixs ->
  110         -- Translating the indices of a data type should never yield
  111         -- Inhabited constraints, so the result of calling
  112         -- `translateParams dtIndices` above should only return Binders and not
  113         -- any ImplicitBinders. Moreover, `translateParams` always returns
  114         -- Binders where the second field is `Just t`, where `t` is the type.
  115         let errorBecause msg = error $ "translateDataType.translateNamed: " ++ msg in
  116         let bs = map (\case Coq.Binder s (Just t) ->
  117                               Coq.PiBinder (Just s) t
  118                             Coq.Binder _ Nothing ->
  119                               errorBecause "encountered a Binder without a Type"
  120                             Coq.ImplicitBinder{} ->
  121                               errorBecause "encountered an implicit binder")
  122                      ixs in
  123         return (ps, bs)
  124       let inductiveSort = TermTranslation.translateSort dtSort
  125       inductiveConstructors <- mapM (translateCtor inductiveParameters) dtCtors
  126       return $ Coq.InductiveDecl $ Coq.Inductive
  127         { inductiveName
  128         , inductiveParameters
  129         , inductiveIndices
  130         , inductiveSort
  131         , inductiveConstructors
  132         }

  151 translateDef :: ModuleTranslationMonad m => Def -> m Coq.Decl
  152 translateDef (Def {..}) = {- trace ("translateDef " ++ show defIdent) $ -} do
  153   specialTreatment <- findSpecialTreatment' (nameInfo defName)
  154   translateAccordingly (atDefSite specialTreatment)
  155 
  156   where
  157 
  158     translateAccordingly :: ModuleTranslationMonad m => DefSiteTreatment -> m Coq.Decl
  159     translateAccordingly  DefPreserve           = translateNamed $ Coq.Ident (Text.unpack (toShortName (nameInfo defName)))
  160     translateAccordingly (DefRename targetName) = translateNamed $ targetName
  161     translateAccordingly (DefReplace  str)      = return $ Coq.Snippet str
  162     translateAccordingly  DefSkip               = return $ skipped' (nameInfo defName)
  163 
  164     translateNamed :: ModuleTranslationMonad m => Coq.Ident -> m Coq.Decl
  165     translateNamed name = liftTermTranslationMonad (go defQualifier defBody)
  166 
  167       where
  168 
  169         go :: TermTranslation.TermTranslationMonad m => DefQualifier -> Maybe Term -> m Coq.Decl
  170         go NoQualifier    Nothing     = error "Terms should have a body (unless axiom/primitive)"
  171         go NoQualifier    (Just body) = Coq.Definition
  172                                         <$> pure name
  173                                         <*> pure []
  174                                         <*> (Just <$> TermTranslation.translateTerm defType)
  175                                         <*> TermTranslation.translateTerm body
  176         go AxiomQualifier _           = Coq.Axiom
  177                                         <$> pure name
  178                                         <*> TermTranslation.translateTerm defType
  179         go PrimQualifier  _           = Coq.Axiom
  180                                         <$> pure name
  181                                         <*> TermTranslation.translateTerm defType
  182 
  183 liftTermTranslationMonad ::
  184   (forall n. TermTranslation.TermTranslationMonad   n => n a) ->
  185   (forall m. ModuleTranslationMonad m => m a)
  186 liftTermTranslationMonad n = do
  187   configuration <- asks translationConfiguration
  188   (modname, mm) <- asks otherConfiguration
  189   let r = TermTranslation.runTermTranslationMonad configuration modname mm [] [] n
  190   case r of
  191     Left  e      -> Except.throwError e
  192     Right (a, _) -> do
  193       return a

  195 translateDecl ::
  196   M.TranslationConfiguration ->
  197   Maybe ModuleName ->
  198   ModuleMap ->
  199   ModuleDecl ->
  200   Doc ann
  201 translateDecl configuration modname mm decl =
  202   case decl of
  203   TypeDecl td -> do
  204     case runModuleTranslationMonad configuration modname mm (translateDataType td) of
  205       Left e -> error $ show e
  206       Right (tdecl, _) -> Coq.ppDecl tdecl
  207   DefDecl dd -> do
  208     case runModuleTranslationMonad configuration modname mm (translateDef dd) of
  209       Left e -> error $ show e
  210       Right (tdecl, _) -> Coq.ppDecl tdecl
  211   InjectCodeDecl ns txt
  212     | ns == "Coq" -> pretty txt
  213     | otherwise   -> mempty
