  112 -- | A Definition contains an identifier, the type of the definition, and an
  113 -- optional body (axioms and primitives do not have a body)

  115   Def

  118   , defType :: Term

  128 -- | A specification of a constructor
  129 data Ctor =
  130   forall d params ixs.
  131   Ctor
  132   { ctorName :: !Ident
  133     -- ^ The name of this constructor
  134   , ctorVarIndex :: !VarIndex
  135     -- ^ Unique var index for this constructor
  136   , ctorArgStruct :: CtorArgStruct d params ixs
  137     -- ^ Arguments to the constructor
  138   , ctorDataType :: !(PrimName Term)
  139     -- ^ The datatype this constructor belongs to
  140   , ctorType :: Term
  141     -- ^ Cached type of the constructor, which should always be equal to
  142     --
  143     -- > (p1::P1) -> .. -> (pn::Pn) -> (x1::arg1) -> .. -> (xm::argm) ->
  144     -- >  d p1 .. pn ix1 .. ixk
  145     --
  146     -- where the @pi@ are the 'ctorParams', the @argi@ are the types specified
  147     -- by the 'ctorArgs', and the @ixi@ are the 'ctorDataTypeIndices'. Note that
  148     -- this type should always be top-level, i.e., have no free variables.
  149   , ctorElimTypeFun :: [Term] -> Term -> IO Term
  150     -- ^ Cached function for generating the type of an eliminator for this
  151     -- constructor by passing it a list of parameters and a @p_ret@ function,
  152     -- also known as the "motive function", which itself must have type
  153     --
  154     -- > (x1::ix1) -> .. -> (xm::ixm) -> d p1 .. pn x1 .. xm -> Type i
  155     --
  156     -- where the @ps@ are the parameters and the @ix@s are the indices of
  157     -- datatype @d@
  158   , ctorIotaReduction ::
  159        Term {- ^ recursor term -} ->
  160        Map VarIndex Term {- ^ constructor eliminators -} ->
  161        [Term] {- ^ constructor arguments -} ->
  162        IO Term
  163     -- ^ Cached function for computing the result of one step of iota
  164     --   reduction of the term
  165     --
  166     -- > RecursorApp rec ixs (c params args)
  167     --
  168     --   The arguments to this function are the recusor value, the
  169     --   the map from the recursor that maps constructors to eliminator
  170     --   functions, and the arguments to the constructor.
  171 
  172   , ctorIotaTemplate :: Term
  173     -- ^ Cached term used for computing iota reductions.  It has free variables
  174     --   @rec@, @elim@ and @args@, in that order so that the last @arg@ is the
  175     --   most recently-bound variable with deBruijn index 0.  The @rec@ variable
  176     --   represents the recursor value, @elim@ represents the eliminator function
  177     --   for the constructor, and @args@ represent the arguments to this constructor.
  178   }
  179 
  180 -- | Return the number of parameters of a constructor
  181 ctorNumParams :: Ctor -> Int
  182 ctorNumParams (Ctor { ctorArgStruct = CtorArgStruct {..}}) =
  183   bindingsLength ctorParams
  184 
  185 -- | Return the number of non-parameter arguments of a constructor
  186 ctorNumArgs :: Ctor -> Int
  187 ctorNumArgs (Ctor { ctorArgStruct = CtorArgStruct {..}}) =
  188   bindingsLength ctorArgs
  189 
  190 -- | Compute the ExtCns that uniquely references a constructor
  191 ctorPrimName :: Ctor -> PrimName Term
  192 ctorPrimName ctor = PrimName (ctorVarIndex ctor) (ctorName ctor) (ctorType ctor)
  193 
  194 lift2 :: (a -> b) -> (b -> b -> c) -> a -> a -> c
  195 lift2 f h x y = h (f x) (f y)
  196 
  197 instance Eq Ctor where
  198   (==) = lift2 ctorName (==)
  199 
  200 instance Ord Ctor where
  201   compare = lift2 ctorName compare
  202 
  203 instance Show Ctor where
  204   show = show . ctorName

  209 -- | An inductively-defined datatype
  210 data DataType =
  211   DataType
  212   { dtName :: Ident
  213     -- ^ The name of this datatype
  214   , dtVarIndex :: !VarIndex
  215     -- ^ Unique var index for this data type
  216   , dtParams :: [(LocalName, Term)]
  217     -- ^ The context of parameters of this datatype
  218   , dtIndices :: [(LocalName, Term)]
  219     -- ^ The context of indices of this datatype
  220   , dtSort :: Sort
  221     -- ^ The universe of this datatype
  222   , dtCtors :: [Ctor]
  223     -- ^ The list of constructors of this datatype
  224   , dtType :: Term
  225     -- ^ The cached type of this datatype, which should always be equal to
  226     --
  227     -- > (p1::P1) -> .. -> (pn::Pn) -> (i1::I1) -> .. -> (im::Im) -> 'dtSort'
  228     --
  229     -- where the @pi@ are the 'dtParams' and the @ii@ are the 'dtIndices'. Note
  230     -- that this type should always be top-level, i.e., have no free variables.
  231   }
  232 
  233 -- | Return the number of parameters of a datatype
  234 dtNumParams :: DataType -> Int
  235 dtNumParams dt = length $ dtParams dt
  236 
  237 -- | Return the number of indices of a datatype
  238 dtNumIndices :: DataType -> Int
  239 dtNumIndices dt = length $ dtIndices dt
  240 
  241 -- | Compute the ExtCns that uniquely references a datatype
  242 dtPrimName :: DataType -> PrimName Term
  243 dtPrimName dt = PrimName (dtVarIndex dt) (dtName dt) (dtType dt)
  244 
  245 instance Eq DataType where
  246   (==) = lift2 dtName (==)
  247 
  248 instance Ord DataType where
  249   compare = lift2 dtName compare
  250 
  251 instance Show DataType where
  252   show = show . dtName

  257 -- | Declarations that can occur in a module
  258 data ModuleDecl = TypeDecl DataType
  259                 | DefDecl Def
  260                     -- | Code to inject directly in-stream when
  261                     --   doing translations.
  262                 | InjectCodeDecl
  263                     Text {- ^ Code namespace -}
  264                     Text {- ^ Code to inject -}

  266 -- | The different sorts of things that a 'Text' name can be resolved to
  267 data ResolvedName
  268   = ResolvedCtor Ctor
  269   | ResolvedDataType DataType
  270   | ResolvedDef Def
  271 
  272 -- | Get the 'Ident' for a 'ResolvedName'
  273 resolvedNameIdent :: ResolvedName -> Ident
  274 resolvedNameIdent (ResolvedCtor ctor) = ctorName ctor
  275 resolvedNameIdent (ResolvedDataType dt) = dtName dt
  276 resolvedNameIdent (ResolvedDef d) = defIdent d
  277 
  278 -- | Modules define namespaces of datatypes, constructors, and definitions,
  279 -- i.e., mappings from 'Text' names to these objects. A module is allowed to
  280 -- map a 'Text' name to an object defined in a different module. Modules also
  281 -- keep a record of the top-level declarations and the imports that were used to
  282 -- build them.
  283 data Module = Module {
  284           moduleName    :: !ModuleName
  285         , moduleImports :: !(Map ModuleName Module)
  286         , moduleResolveMap :: !(Map Text ResolvedName)
  287         , moduleRDecls   :: [ModuleDecl] -- ^ All declarations in reverse order they were added.
  288         }
  289 
  290 -- | Get the names of all modules imported by the given one
  291 moduleImportNames :: Module -> [ModuleName]
  292 moduleImportNames m = Map.keys (moduleImports m)
  293 
  294 emptyModule :: ModuleName -> Module
  295 emptyModule nm =
  296   Module { moduleName = nm
  297          , moduleImports = Map.empty
  298          , moduleResolveMap = Map.empty
  299          , moduleRDecls = []
  300          }
  301 
  302 
  303 -- | Resolve a 'Text' name in the namespace defined by a 'Module', to either a
  304 -- 'Ctor', 'DataType', or 'Def'
  305 resolveName :: Module -> Text -> Maybe ResolvedName
  306 resolveName m str = Map.lookup str (moduleResolveMap m)
  307 
  308 -- | Resolve a 'Text' name to a 'Ctor'
  309 findCtor :: Module -> Text -> Maybe Ctor
  310 findCtor m str =
  311   resolveName m str >>= \case { ResolvedCtor ctor -> Just ctor; _ -> Nothing }
  312 
  313 -- | Resolve a 'Text' name to a 'DataType'
  314 findDataType :: Module -> Text -> Maybe DataType
  315 findDataType m str =
  316   resolveName m str >>= \case { ResolvedDataType d -> Just d; _ -> Nothing }
  317 
  318 -- | Resolve a 'Text' name to a 'Def'
  319 findDef :: Module -> Text -> Maybe Def
  320 findDef m str =
  321   resolveName m str >>= \case { ResolvedDef d -> Just d; _ -> Nothing }
  322 
  323 
  324 -- | Insert a 'ResolvedName' into a 'Module', adding a mapping from the 'Text'
  325 -- name of that resolved name to it. Signal an error in the case of a name
  326 -- clash, i.e., an existing binding for the same 'Text' name.
  327 insResolvedName :: Module -> ResolvedName -> Module
  328 insResolvedName m nm =
  329   let str = identBaseName $ resolvedNameIdent nm in
  330   if Map.member str (moduleResolveMap m) then
  331     panic "insResolvedName" [
  332         "inserting duplicate name " <> str <> " into module " <>
  333             moduleNameText (moduleName m)
  334     ]
  335   else
  336     m { moduleResolveMap = Map.insert str nm (moduleResolveMap m) }
  337 
  338 -- | @insImport i m@ returns the module obtained by importing @i@ into @m@,
  339 -- using a predicate to specify which names are imported from @i@ into @m@. In
  340 -- the case of name clashes, an error is signaled.
  341 insImport :: (ResolvedName -> Bool) -> Module -> Module -> Module
  342 insImport name_p i m =
  343   (foldl' insResolvedName m $ Map.elems $
  344    Map.filter name_p (moduleResolveMap i))
  345   { moduleImports = Map.insert (moduleName i) i (moduleImports m) }
  346 
  347 -- | Insert a 'DataType' declaration, along with its 'Ctor's, into a module
  348 insDataType :: Module -> DataType -> Module
  349 insDataType m dt =
  350   foldl' insResolvedName (m { moduleRDecls = TypeDecl dt : moduleRDecls m}) $
  351   (ResolvedDataType dt : map ResolvedCtor (dtCtors dt))
  352 
  353 -- | Insert an "incomplete" datatype, used as part of building up a 'DataType'
  354 -- to typecheck its constructors. This incomplete datatype must have no
  355 -- constructors, and it will not be added to the 'moduleRDecls' list until it is
  356 -- completed by 'completeDataType'.
  357 beginDataType :: Module -> DataType -> Module
  358 beginDataType m dt =
  359    if null (dtCtors dt) then
  360      insResolvedName m (ResolvedDataType dt)
  361    else
  362      panic "insTempDataType" ["attempt to insert a non-empty temporary datatype"]
  363 
  364 -- | Complete a datatype, by adding its constructors
  365 completeDataType :: Module -> Ident -> [Ctor] -> Module
  366 completeDataType m (identBaseName -> str) ctors =
  367   case resolveName m str of
  368     Just (ResolvedDataType dt)
  369       | null (dtCtors dt) ->
  370         let dt' = dt {dtCtors = ctors} in
  371         flip (foldl' insResolvedName) (map ResolvedCtor ctors) $
  372         m { moduleResolveMap =
  373               Map.insert str (ResolvedDataType dt') (moduleResolveMap m),
  374               moduleRDecls = TypeDecl dt' : moduleRDecls m }
  375     Just (ResolvedDataType _) ->
  376       panic "completeDataType" ["datatype already completed: " <> str]
  377     Just _ ->
  378       panic "completeDataType" ["not a datatype: " <> str]
  379     Nothing ->
  380       panic "completeDataType" ["datatype not found: " <> str]
  381 
  382 
  383 -- | Insert an "injectCode" declaration
  384 insInjectCode :: Module -> Text -> Text -> Module
  385 insInjectCode m ns txt =
  386   m{ moduleRDecls = InjectCodeDecl ns txt : moduleRDecls m }

  388 -- | Insert a definition into a module
  389 insDef :: Module -> Def -> Module
  390 insDef m d =
  391   insResolvedName
  392   (m { moduleRDecls = DefDecl d : moduleRDecls m })
  393   (ResolvedDef d)

  395 -- | Get the resolved names that are local to a module
  396 localResolvedNames :: Module -> [ResolvedName]
  397 localResolvedNames m =
  398   filter ((== moduleName m) . identModule . resolvedNameIdent)
  399   (Map.elems $ moduleResolveMap m)
  400 
  401 -- | Get all data types defined in a module
  402 moduleDataTypes :: Module -> [DataType]
  403 moduleDataTypes =
  404   foldr' (\case { ResolvedDataType dt -> (dt :); _ -> id } ) [] .
  405   localResolvedNames
  406 
  407 -- | Get all constructors defined in a module
  408 moduleCtors :: Module -> [Ctor]
  409 moduleCtors =
  410   foldr' (\case { ResolvedCtor ctor -> (ctor :); _ -> id } ) [] .
  411   localResolvedNames
  412 
  413 -- | Get all definitions defined in a module
  414 moduleDefs :: Module -> [Def]
  415 moduleDefs =
  416   foldr' (\case { ResolvedDef d -> (d :); _ -> id } ) [] .
  417   localResolvedNames
  418 
  419 -- | Get all declarations that are local to a module, i.e., that defined names
  420 -- that were not imported from some other module
  421 moduleDecls :: Module -> [ModuleDecl]
  422 moduleDecls = reverse . moduleRDecls
  423 
  424 -- | Get all local declarations in a module that are marked as primitives
  425 modulePrimitives :: Module -> [Def]
  426 modulePrimitives m =
  427     [ def
  428     | DefDecl def <- moduleDecls m
  429     , defQualifier def == PrimQualifier
  430     ]
  431 
  432 -- | Get all local declarations in a module that are marked as axioms
  433 moduleAxioms :: Module -> [Def]
  434 moduleAxioms m =
  435     [ def
  436     | DefDecl def <- moduleDecls m
  437     , defQualifier def == AxiomQualifier
  438     ]
  439 
  440 -- | Get all local declarations in a module that are not marked as primitives or
  441 -- axioms
  442 moduleActualDefs :: Module -> [Def]
  443 moduleActualDefs m =
  444     [ def
  445     | DefDecl def <- moduleDecls m
  446     , defQualifier def == NoQualifier
  447     ]

  449 -- | The type of mappings from module names to modules
  450 type ModuleMap = HashMap ModuleName Module
  451 
  452 -- | Resolve an 'Ident' to a 'Ctor' in a 'ModuleMap'
  453 findCtorInMap :: ModuleMap -> Ident -> Maybe Ctor
  454 findCtorInMap m i =
  455   HMap.lookup (identModule i) m >>= flip findCtor (identBaseName i)
  456 
  457 -- | Resolve an 'Ident' to a 'DataType' in a 'ModuleMap'
  458 findDataTypeInMap :: ModuleMap -> Ident -> Maybe DataType
  459 findDataTypeInMap m i =
  460   HMap.lookup (identModule i) m >>= flip findDataType (identBaseName i)
  461 
  462 -- | Get all definitions defined in any module in an entire module map. Note
  463 -- that the returned list might have redundancies if a definition is visible /
  464 -- imported in multiple modules in the module map.
  465 allModuleDefs :: ModuleMap -> [Def]
  466 allModuleDefs modmap = concatMap moduleDefs (HMap.elems modmap)
  467 
  468 -- | Get all local declarations from all modules in an entire module map
  469 allModuleDecls :: ModuleMap -> [ModuleDecl]
  470 allModuleDecls modmap = concatMap moduleDecls (HMap.elems modmap)
  471 
  472 -- | Get all local declarations from all modules in an entire module map that
  473 -- are marked as primitives
  474 allModulePrimitives :: ModuleMap -> [Def]
  475 allModulePrimitives modmap =
  476     [ def
  477     | DefDecl def <- allModuleDecls modmap
  478     , defQualifier def == PrimQualifier
  479     ]
  480 
  481 -- | Get all local declarations from all modules in an entire module map that
  482 -- are marked as axioms
  483 allModuleAxioms :: ModuleMap -> [Def]
  484 allModuleAxioms modmap =
  485     [ def
  486     | DefDecl def <- allModuleDecls modmap
  487     , defQualifier def == AxiomQualifier
  488     ]
  489 
  490 -- | Get all local declarations from all modules in an entire module map that
  491 -- are marked as neither primitives nor axioms
  492 allModuleActualDefs :: ModuleMap -> [Def]
  493 allModuleActualDefs modmap =
  494     [ def
  495     | DefDecl def <- allModuleDecls modmap
  496     , defQualifier def == NoQualifier
  497     ]

  499 -- | Get all datatypes in all modules in a module map
  500 allModuleDataTypes :: ModuleMap -> [DataType]
  501 allModuleDataTypes modmap = concatMap moduleDataTypes (HMap.elems modmap)
  502 
  503 -- | Get all constructors in all modules in a module map

  508 -- | Pretty-print a 'Module'
  509 --
  510 --   This converts to the PPModule representation used by the actual module
  511 --   printer in Pretty.hs.
  512 ppModule :: PPS.Opts -> Module -> PPS.Doc
  513 ppModule opts m =
  514   ppPPModule opts (PPModule (moduleImportNames m) (map toDecl $ moduleDecls m))
  515   where
  516     toDecl (TypeDecl (DataType {..})) =
  517       PPTypeDecl dtName dtParams dtIndices dtSort
  518       (map (\c -> (ctorName c, ctorType c)) dtCtors)
  519     toDecl (DefDecl (Def {..})) =
  520       PPDefDecl defIdent defType defBody
  521     toDecl (InjectCodeDecl ns text) =
  522       PPInjectCode ns text
