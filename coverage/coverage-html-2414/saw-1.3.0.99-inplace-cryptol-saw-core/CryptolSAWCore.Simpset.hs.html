   10 {-# LANGUAGE OverloadedStrings #-}
   11 
   12 module CryptolSAWCore.Simpset
   13   ( mkCryptolSimpset
   14   ) where
   15 
   16 import SAWCore.Module (moduleDefs, Def(..))
   17 import SAWCore.Rewriter
   18 import SAWCore.SharedTerm
   19 import SAWCore.Term.Functor
   20 
   21 mkCryptolSimpset :: SharedContext -> IO (Simpset a)
   22 mkCryptolSimpset sc =
   23   do m <- scFindModule sc cryptolModuleName
   24      scSimpset sc (cryptolDefs m) [] []
   25   where
   26     cryptolDefs m = filter (not . excluded) $ moduleDefs m
   27     excluded d =
   28       case defNameInfo d of
   29         ModuleIdentifier ident -> ident `elem` excludedNames
   30         ImportedName{} -> True
   31 
   32 cryptolModuleName :: ModuleName
   33 cryptolModuleName = mkModuleName ["Cryptol"]
   34 
   35 excludedNames :: [Ident]
   36 excludedNames =
   37   map (mkIdent cryptolModuleName)
   38   [ "fix"
   39   , "pair_cong"
   40   , "seq_cong"
   41   , "pair_cong1"
   42   , "pair_cong2"
   43   , "seq_cong1"
   44   , "fun_cong"
   45   , "seq_TCNum"
   46   , "seq_TCInf"
   47   , "PLiteral"
   48   , "PLogic"
   49   , "PRing"
   50   , "PIntegral"
   51   , "PField"
   52   , "PRound"
   53   , "PEq"
   54   , "PCmp"
   55   , "PSignedCmp"
   56   , "ecEq"
   57   ]
