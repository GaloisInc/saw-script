   64 -- * Certified typed terms
   65 
   66 -- | An abstract datatype pairing a well-formed 'Term' with its type.
   67 data SCTypedTerm =
   68   SCTypedTerm
   69   Term -- ^ value
   70   Term -- ^ type
   71   (IntMap Term) -- ^ typing context
   72 
   73 -- | The raw 'Term' of an 'SCTypedTerm'.
   74 typedVal :: SCTypedTerm -> Term
   75 typedVal (SCTypedTerm trm _ _) = trm
   76 
   77 -- | The type of an 'SCTypedTerm' as a 'Term'.
   78 typedType :: SCTypedTerm -> Term
   79 typedType (SCTypedTerm _ typ _) = typ
   80 
   81 -- | The typing context of an 'SCTypedTerm', keyed by the 'VarIndex'
   82 -- of each 'VarName' in the term.
   83 typedCtx :: SCTypedTerm -> IntMap Term
   84 typedCtx (SCTypedTerm _ _ ctx) = ctx
   85 
   86 -- | Reduce the given 'Term' to WHNF, using all reductions allowed by
   87 -- the SAWCore type system.
   88 scTypeCheckWHNF :: SharedContext -> Term -> IO Term
   89 scTypeCheckWHNF sc t =
   90   do (_, t') <- rewriteSharedTerm sc (addConvs natConversions emptySimpset :: Simpset ()) t
   91      scWhnf sc t'
   92 
   93 -- | Check if two terms are "convertible for type-checking", meaning that they
   94 -- are convertible up to 'natConversions'.
   95 scTypeConvertible :: SharedContext -> Term -> Term -> IO Bool
   96 scTypeConvertible sc t1 t2 = scConvertibleEval sc scTypeCheckWHNF True t1 t2
   97 
   98 -- | Check whether one type is a subtype of another: Either they are
   99 -- convertible, or they are both Pi types with convertible argument
  100 -- types and result sorts @s1@ and @s2@ with @s1 <= s2@.
  101 scSubtype :: SharedContext -> Term -> Term -> IO Bool
  102 scSubtype sc t1 t2
  103   | alphaEquiv t1 t2 = pure True
  104   | otherwise =
  105     do t1' <- scWhnf sc t1
  106        t2' <- scWhnf sc t2
  107        case (t1', t2') of
  108          (asSort -> Just s1, asSort -> Just s2) ->
  109            pure (s1 <= s2)
  110          (unwrapTermF -> Pi x1 a1 b1, unwrapTermF -> Pi x2 a2 b2)
  111            | x1 == x2 ->
  112              (&&) <$> scTypeConvertible sc a1 a2 <*> scSubtype sc b1 b2
  113            | otherwise ->
  114              do conv1 <- scTypeConvertible sc a1 a2
  115                 var1 <- scVariable sc (EC x1 a1)
  116                 b2' <- scInstantiateExt sc (IntMap.singleton (vnIndex x2) var1) b2
  117                 conv2 <- scSubtype sc b1 b2'
  118                 pure (conv1 && conv2)
  119          _ ->
  120            scTypeConvertible sc t1' t2'

  125 -- | Compute the type of an 'SCTypedTerm'.
  126 scTypeOfTypedTerm :: SharedContext -> SCTypedTerm -> IO SCTypedTerm
  127 scTypeOfTypedTerm sc (SCTypedTerm _tm tp ctx) =
  128   do tp_tp <- scTypeOf' sc ctx tp
  129      -- Shrink typing context if possible
  130      let ctx' = pruneContext (freeVars tp_tp) ctx
  131      pure (SCTypedTerm tp tp_tp ctx')
  132 
  133 -- | Reduce an 'SCTypedTerm' to WHNF (see also 'scTypeCheckWHNF').
  134 scTypedTermWHNF :: SharedContext -> SCTypedTerm -> IO SCTypedTerm
  135 scTypedTermWHNF sc (SCTypedTerm tm tp ctx) =
  136   do tm' <- scTypeCheckWHNF sc tm
  137      pure (SCTypedTerm tm' tp ctx)
  138 
  139 scGlobalTypedTerm :: SharedContext -> Ident -> IO SCTypedTerm
  140 scGlobalTypedTerm sc ident =
  141   do tm <- scGlobalDef sc ident
  142      tp <- scTypeOfIdent sc ident
  143      pure (SCTypedTerm tm tp IntMap.empty)

  146 -- * Building typed terms
  147 
  148 -- possible errors: not a pi type, bad argument type, context mismatch
  149 scTypedApply :: SharedContext -> SCTypedTerm -> SCTypedTerm -> IO SCTypedTerm
  150 scTypedApply sc f arg =
  151   do tm <- scApply sc (typedVal f) (typedVal arg)
  152      (vnIndex -> i, t1, t2) <- ensurePi sc (typedType f)
  153      ok <- scSubtype sc (typedType arg) t1
  154      unless ok $ fail $ unlines $
  155        ["Not a subtype", "expected: " ++ showTerm t1, "got: " ++ showTerm (typedType arg)]
  156      tp <- scInstantiateExt sc (IntMap.singleton i (typedVal arg)) t2
  157      ctx <- unifyContexts "scTypedApply" (typedCtx f) (typedCtx arg)
  158      pure (SCTypedTerm tm tp ctx)
  159 
  160 -- possible errors: not a type, context mismatch
  161 scTypedLambda :: SharedContext -> VarName -> SCTypedTerm -> SCTypedTerm -> IO SCTypedTerm
  162 scTypedLambda sc x t body =
  163   do _s <- ensureSort sc (typedType t)
  164      tm <- scLambda sc x (typedVal t) (typedVal body)
  165      _ <- unifyContexts "scTypedLambda" (IntMap.singleton (vnIndex x) (typedVal t)) (typedCtx body)
  166      ctx <- unifyContexts "scTypedLambda" (typedCtx t) (IntMap.delete (vnIndex x) (typedCtx body))
  167      tp <- scPi sc x (typedVal t) (typedType body)
  168      pure (SCTypedTerm tm tp ctx)
  169 
  170 -- possible errors: not a type, context mismatch
  171 scTypedPi :: SharedContext -> VarName -> SCTypedTerm -> SCTypedTerm -> IO SCTypedTerm
  172 scTypedPi sc x t body =
  173   do tm <- scPi sc x (typedVal t) (typedVal body)
  174      _ <- unifyContexts "scTypedPi" (IntMap.singleton (vnIndex x) (typedVal t)) (typedCtx body)
  175      ctx <- unifyContexts "scTypedPi" (typedCtx t) (IntMap.delete (vnIndex x) (typedCtx body))
  176      s1 <- ensureSort sc (typedType t)
  177      s2 <- ensureSort sc (typedType body)
  178      tp <- scSort sc (piSort s1 s2)
  179      pure (SCTypedTerm tm tp ctx)
  180 
  181 -- possible errors: not a type, context mismatch
  182 scTypedFun :: SharedContext -> SCTypedTerm -> SCTypedTerm -> IO SCTypedTerm
  183 scTypedFun sc a b =
  184   do tm <- scFun sc (typedVal a) (typedVal b)
  185      sa <- ensureSort sc (typedType a)
  186      sb <- ensureSort sc (typedType b)
  187      tp <- scSort sc (piSort sa sb)
  188      ctx <- unifyContexts "scTypedFun" (typedCtx a) (typedCtx b)
  189      pure (SCTypedTerm tm tp ctx)

  229 scTypedPairType :: SharedContext -> SCTypedTerm -> SCTypedTerm -> IO SCTypedTerm

  231   do tm <- scPairType sc (typedVal x) (typedVal y)
  232      sx <- ensureSort sc (typedType x)

  234      tp <- scSort sc (max sx sy)
  235      ctx <- unifyContexts "scTypedPairType" (typedCtx x) (typedCtx y)
  236      pure (SCTypedTerm tm tp ctx)

  239 scTypedPairLeft :: SharedContext -> SCTypedTerm -> IO SCTypedTerm
  240 scTypedPairLeft sc x =
  241   do tm <- scPairLeft sc (typedVal x)
  242      tp <- fst <$> ensurePairType sc (typedType x)
  243      let ctx = typedCtx x
  244      pure (SCTypedTerm tm tp ctx)

  246 -- possible errors: not a pair

  248 scTypedPairRight sc x =
  249   do tm <- scPairRight sc (typedVal x)
  250      tp <- snd <$> ensurePairType sc (typedType x)
  251      let ctx = typedCtx x
  252      pure (SCTypedTerm tm tp ctx)

  254 -- possible errors: not a datatype, bad elimination sort
  255 scTypedRecursor :: SharedContext -> Name -> Sort -> IO SCTypedTerm
  256 scTypedRecursor sc nm s =
  257   do mm <- scGetModuleMap sc
  258      case lookupVarIndexInMap (nameIndex nm) mm of
  259        Just (ResolvedDataType dt) ->
  260          do unless (allowedElimSort dt s) $ fail "Disallowed propositional elimination"
  261             let d = dtName dt
  262             let nparams = length (dtParams dt)
  263             let nixs = length (dtIndices dt)
  264             let ctorOrder = map ctorName (dtCtors dt)
  265             let crec = CompiledRecursor d s nparams nixs ctorOrder
  266             tm <- scFlatTermF sc (Recursor crec)
  267             tp <- scRecursorType sc dt s
  268             pure (SCTypedTerm tm tp IntMap.empty)
  269        _ ->
  270          fail "datatype not found"
  271 
  272 -- possible errors: field not a type, context mismatch
  273 scTypedRecordType :: SharedContext -> [(FieldName, SCTypedTerm)] -> IO SCTypedTerm
  274 scTypedRecordType sc fields =
  275   do tm <- scRecordType sc (map (fmap typedVal) fields)
  276      sorts <- traverse (ensureSort sc . typedType . snd) fields
  277      tp <- scSort sc (foldl max (mkSort 0) sorts)
  278      ctx <- unifyContextList "scTypedRecordType" (map (typedCtx . snd) fields)
  279      pure (SCTypedTerm tm tp ctx)
  280 
  281 -- possible errors: duplicate field name
  282 scTypedRecordValue :: SharedContext -> [(FieldName, SCTypedTerm)] -> IO SCTypedTerm
  283 scTypedRecordValue sc fields =
  284   do tm <- scFlatTermF sc $ RecordValue (map (fmap typedVal) fields)
  285      tp <- scRecordType sc (map (fmap typedType) fields)
  286      ctx <- foldM (unifyContexts "scTypedRecordValue") IntMap.empty (map (typedCtx . snd) fields)
  287      pure (SCTypedTerm tm tp ctx)
  288 
  289 -- possible errors: not a record type, field name not found
  290 scTypedRecordProj :: SharedContext -> SCTypedTerm -> FieldName -> IO SCTypedTerm
  291 scTypedRecordProj sc t fname =
  292   do tm <- scRecordSelect sc (typedVal t) fname
  293      let ctx = typedCtx t
  294      tps <- ensureRecordType sc (typedType t)
  295      case Map.lookup fname tps of
  296        Nothing -> fail "scTypedRecordProj: field name not found"
  297        Just tp -> pure (SCTypedTerm tm tp ctx)

  299 -- no possible errors
  300 scTypedSort :: SharedContext -> Sort -> IO SCTypedTerm
  301 scTypedSort sc s = scTypedSort' sc s noFlags

  303 -- | A variant of 'scTypedSort' that also takes a 'SortFlags' argument.

  306 scTypedSort' sc s flags =
  307   do tm <- scFlatTermF sc (Sort s flags)
  308      tp <- scSort sc (sortOf s)

  338 -- | Prune a typing context by dropping indices unreachable from the
  339 -- given 'IntSet'.
  340 pruneContext :: IntSet -> IntMap Term -> IntMap Term
  341 pruneContext vs0 ctx = IntMap.restrictKeys ctx (reachable mempty vs0)
  342   where
  343     fvs = fmap freeVars ctx
  344     reachable old new
  345       | IntSet.null new = old
  346       | otherwise = reachable old' new'
  347           where old' = old <> new
  348                 new' = IntSet.difference (Fold.fold (IntMap.restrictKeys fvs new)) old'
  349 
  350 -- | Two typing contexts are unifiable if they agree perfectly on all
  351 -- entries where they overlap.
  352 unifyContexts :: String -> IntMap Term -> IntMap Term -> IO (IntMap Term)
  353 unifyContexts msg ctx1 ctx2 =
  354   do let check t1 t2 =
  355            unless (t1 == t2) $
  356            fail $ unlines ["unifyContexts: context mismatch", msg,
  357                            "t1: " ++ showTerm t1,
  358                            "t2: " ++ showTerm t2]
  359      sequence_ (IntMap.intersectionWith check ctx1 ctx2)
  360      pure (IntMap.union ctx1 ctx2)
  361 
  362 
  363 unifyContextList :: String -> [IntMap Term] -> IO (IntMap Term)
  364 unifyContextList msg = foldM (unifyContexts msg) IntMap.empty
  365 
  366 ensureRecognizer :: String -> SharedContext -> (Term -> Maybe a) -> Term -> IO a
  367 ensureRecognizer s sc f trm =
  368   case f trm of
  369     Just a -> pure a
  370     Nothing ->
  371       do trm' <- scTypeCheckWHNF sc trm
  372          case f trm' of
  373            Just a -> pure a
  374            Nothing ->
  375              fail $ "ensureRecognizer: Expected " ++ s ++ ", found: " ++ showTerm trm'
  376 
  377 ensureSort :: SharedContext -> Term -> IO Sort
  378 ensureSort sc tp = ensureRecognizer "Sort" sc asSort tp
  379 
  380 ensurePi :: SharedContext -> Term -> IO (VarName, Term, Term)
  381 ensurePi sc tp = ensureRecognizer "Pi" sc asPi tp
  382 
  383 ensurePairType :: SharedContext -> Term -> IO (Term, Term)
  384 ensurePairType sc tp = ensureRecognizer "PairType" sc asPairType tp
  385 
  386 ensureRecordType :: SharedContext -> Term -> IO (Map FieldName Term)
  387 ensureRecordType sc tp = ensureRecognizer "RecordType" sc asRecordType tp

  390 piSort s1 s2 = if s2 == propSort then propSort else max s1 s2
