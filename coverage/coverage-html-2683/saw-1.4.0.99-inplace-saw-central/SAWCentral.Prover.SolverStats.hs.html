   20 -- | Data structure for recording useful information about
   21 --   verification runs on goals.
   22 data SolverStats
   23  = SolverStats
   24    { -- | The names of the solvers that proved this goal.  If verification was
   25      --   skipped for some reason, this should be the empty set.
   26      solverStatsSolvers :: Set Text
   27      -- | The size of the term(s) that represent the proof obligations
   28      --   involved in this verification goal.  Usually, this is the size
   29      --   of the SAWCore term representing the goal, but it might also
   30      --   be the size of an AIG network, etc.
   31    , solverStatsGoalSize :: Integer
   32    }
   33  deriving (Show)
   34 
   35 solverStats :: Text -> Integer -> SolverStats
   36 solverStats nm sz = SolverStats (Set.singleton nm) sz
   37 
   38 instance Semigroup SolverStats where
   39   SolverStats xs n <> SolverStats ys m = SolverStats (mappend xs ys) (n+m)
   40 
   41 instance Monoid SolverStats where
   42   mempty = SolverStats mempty 0
   43   mappend = (<>)
   44 
   45 
   46 ppStats :: SolverStats -> Text
   47 ppStats x = "Goal size " <> Text.pack (show $ solverStatsGoalSize x) <>
   48             ". Provers used: " <>
   49             Text.intercalate "," (Set.toList (solverStatsSolvers x))
