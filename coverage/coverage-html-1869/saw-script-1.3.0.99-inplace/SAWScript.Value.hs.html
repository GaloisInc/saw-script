  133 data Value
  134   = VBool Bool
  135   | VString Text
  136   | VInteger Integer
  137   | VArray [Value]
  138   | VTuple [Value]
  139   | VMaybe (Maybe Value)
  140   | VRecord (Map SS.Name Value)
  141   | VLambda (Value -> TopLevel Value)
  142   | VTerm TypedTerm
  143   | VType Cryptol.Schema
  144   | VReturn Value -- Returned value in unspecified monad
  145   | VBind SS.Pos Value Value
  146     -- ^ Monadic bind in unspecified monad. Requires a source position because
  147     -- operations in these monads can fail at runtime.
  148   | VTopLevel (TopLevel Value)
  149   | VProofScript (ProofScript Value)
  150   | VSimpset SAWSimpset
  151   | VRefnset SAWRefnset
  152   | VTheorem Theorem
  153   | VBisimTheorem BisimTheorem
  154   -----
  155   | VLLVMCrucibleSetup !(LLVMCrucibleSetupM Value)
  156   | VLLVMCrucibleMethodSpec (CMSLLVM.SomeLLVM CMS.ProvedSpec)
  157   | VLLVMCrucibleSetupValue (CMSLLVM.AllLLVM CMS.SetupValue)
  158   -----
  159   | VJVMSetup !(JVMSetupM Value)
  160   | VJVMMethodSpec !(CMS.ProvedSpec CJ.JVM)
  161   | VJVMSetupValue !(CMS.SetupValue CJ.JVM)
  162   -----
  163   | VMIRSetup !(MIRSetupM Value)
  164   | VMIRMethodSpec !(CMS.ProvedSpec MIR)
  165   | VMIRSetupValue !(CMS.SetupValue MIR)
  166   -----
  167   | VLLVMModuleSkeleton ModuleSkeleton
  168   | VLLVMFunctionSkeleton FunctionSkeleton
  169   | VLLVMSkeletonState SkeletonState
  170   | VLLVMFunctionProfile FunctionProfile
  171   -----
  172   | VJavaType JavaType
  173   | VLLVMType LLVM.Type
  174   | VMIRType MIR.Ty
  175   | VCryptolModule CryptolModule
  176   | VJavaClass JSS.Class
  177   | VLLVMModule (Some CMSLLVM.LLVMModule)
  178   | VMIRModule RustModule
  179   | VMIRAdt MIR.Adt
  180   | VHeapsterEnv HeapsterEnv
  181   | VSatResult SatResult
  182   | VProofResult ProofResult
  183   | VUninterp Uninterp
  184   | VAIG AIGNetwork
  185   | VCFG SAW_CFG
  186   | VGhostVar CMS.GhostGlobal
  187   | VYosysModule YosysIR

  283 showsProofResult :: PPOpts -> ProofResult -> ShowS
  284 showsProofResult opts r =
  285   case r of
  286     ValidProof _ _ -> showString "Valid"
  287     InvalidProof _ ts _ -> showString "Invalid: [" . showMulti "" ts
  288     UnfinishedProof st  -> showString "Unfinished: " . shows (length (psGoals st)) . showString " goals remaining"
  289   where
  290     opts' = sawPPOpts opts
  291     showVal t = shows (ppFirstOrderValue opts' t)
  292     showEqn (x, t) = showEC x . showString " = " . showVal t
  293     showEC ec = showString (unpack (toShortName (ecName ec)))
  294 
  295     showMulti _ [] = showString "]"
  296     showMulti s (eqn : eqns) = showString s . showEqn eqn . showMulti ", " eqns

  298 showsSatResult :: PPOpts -> SatResult -> ShowS
  299 showsSatResult opts r =
  300   case r of
  301     Unsat _ -> showString "Unsat"
  302     Sat _ ts -> showString "Sat: [" . showMulti "" ts
  303     SatUnknown  -> showString "Unknown"
  304   where
  305     opts' = sawPPOpts opts
  306     showVal t = shows (ppFirstOrderValue opts' t)
  307     showEC ec = showString (unpack (toShortName (ecName ec)))
  308     showEqn (x, t) = showEC x . showString " = " . showVal t
  309     showMulti _ [] = showString "]"
  310     showMulti s (eqn : eqns) = showString s . showEqn eqn . showMulti ", " eqns

  312 showSimpset :: PPOpts -> Simpset a -> String
  313 showSimpset opts ss =
  314   unlines ("Rewrite Rules" : "=============" : map (show . ppRule) (listRules ss))
  315   where
  316     ppRule r =
  317       PP.pretty '*' PP.<+>
  318       (PP.nest 2 $ PP.fillSep
  319        [ ppTerm (lhsRewriteRule r)
  320        , PP.pretty '=' PP.<+> ppTerm (rhsRewriteRule r) ])
  321     ppTerm t = SAWCorePP.ppTerm opts' t
  322     opts' = sawPPOpts opts
  323 
  324 -- | Pretty-print a 'Refnset' to a 'String'
  325 showRefnset :: PPOpts -> MRSolver.Refnset a -> String
  326 showRefnset opts ss =
  327   unlines ("Refinements" : "=============" : map (show . ppFunAssump)
  328                                                  (MRSolver.listFunAssumps ss))
  329   where
  330     ppFunAssump (MRSolver.FunAssump ctx f args rhs _) =
  331       PP.pretty '*' PP.<+>
  332       (PP.nest 2 $ PP.fillSep
  333        [ ppTermAppInCtx opts' ctx (funNameTerm f) args
  334        , PP.pretty ("|=" :: String) PP.<+> ppFunAssumpRHS ctx rhs ])
  335     ppFunAssumpRHS ctx (OpaqueFunAssump f args) =
  336       ppTermAppInCtx opts' ctx (funNameTerm f) args
  337     ppFunAssumpRHS ctx (RewriteFunAssump rhs) =
  338       SAWCorePP.ppTermInCtx opts' (map fst $ mrVarCtxInnerToOuter ctx) rhs
  339     opts' = sawPPOpts opts

  341 showsPrecValue :: PPOpts -> SAWNamingEnv -> Int -> Value -> ShowS
  342 showsPrecValue opts nenv p v =
  343   case v of
  344     VBool True -> showString "true"
  345     VBool False -> showString "false"
  346     VString s -> shows s
  347     VInteger n -> shows n
  348     VArray vs -> showBrackets $ commaSep $ map (showsPrecValue opts nenv 0) vs
  349     VTuple vs -> showParen True $ commaSep $ map (showsPrecValue opts nenv 0) vs
  350     VMaybe (Just v') -> showString "(Just " . showsPrecValue opts nenv 0 v' . showString ")"
  351     VMaybe Nothing -> showString "Nothing"
  352     VRecord m ->
  353       showBraces $ commaSep $ map showFld (M.toList m)
  354         where
  355           showFld (n, fv) =
  356             showString (unpack n) . showString "=" . showsPrecValue opts nenv 0 fv
  357 
  358     VLambda {} -> showString "<<function>>"
  359     VTerm t -> showString (SAWCorePP.showTermWithNames opts' nenv (ttTerm t))
  360     VType sig -> showString (pretty sig)
  361     VReturn {} -> showString "<<monadic>>"
  362     VBind {} -> showString "<<monadic>>"
  363     VTopLevel {} -> showString "<<TopLevel>>"
  364     VSimpset ss -> showString (showSimpset opts ss)
  365     VRefnset ss -> showString (showRefnset opts ss)
  366     VProofScript {} -> showString "<<proof script>>"
  367     VTheorem thm ->
  368       showString "Theorem " .
  369       showParen True (showString (prettyProp opts' nenv (thmProp thm)))
  370     VBisimTheorem _ -> showString "<<Bisimulation theorem>>"
  371     VLLVMCrucibleSetup{} -> showString "<<Crucible Setup>>"

  411 indexValue :: Value -> Value -> Value
  412 indexValue (VArray vs) (VInteger x)
  413     | i < length vs = vs !! i
  414     | otherwise = error "array index out of bounds"
  415     where i = fromInteger x
  416 indexValue _ _ = error "indexValue"

  418 lookupValue :: Value -> Text -> Value
  419 lookupValue (VRecord vm) name =
  420     case M.lookup name vm of
  421       Nothing -> error $ "no such record field: " ++ unpack name
  422       Just x -> x
  423 lookupValue _ _ = error "lookupValue"

  425 tupleLookupValue :: Value -> Integer -> Value
  426 tupleLookupValue (VTuple vs) i
  427   | 0 <= i && fromIntegral i < length vs = vs !! fromIntegral i
  428   | otherwise = error $ "no such tuple index: " ++ show i

  467 applyValue :: Value -> Value -> TopLevel Value

  471 thenValue :: SS.Pos -> Value -> Value -> Value
  472 thenValue pos v1 v2 = VBind pos v1 (VLambda (const (return v2)))

  474 bindValue :: SS.Pos -> Value -> Value -> TopLevel Value

  477 forValue :: [Value] -> Value -> TopLevel Value
  478 forValue [] _ = return $ VReturn (VArray [])
  479 forValue (x : xs) f =
  480   do m1 <- applyValue f x
  481      m2 <- forValue xs f
  482      bindValue (SS.PosInternal "<for>") m1 (VLambda $ \v1 ->
  483        bindValue (SS.PosInternal "<for>") m2 (VLambda $ \v2 ->
  484          return $ VReturn (VArray (v1 : fromValue v2))))

  510 addTypedef :: SS.Name -> SS.Type -> TopLevelRW -> TopLevelRW
  511 addTypedef name ty rw =
  512   rw { rwNamedTypes = M.insert name (SS.ConcreteType ty') (rwNamedTypes rw) }

  620 -- | A version of 'Control.Exception.bracket' specialized to 'TopLevel'. We
  621 -- can't use 'Control.Monad.Catch.bracket' because it requires 'TopLevel' to
  622 -- implement 'Control.Monad.Catch.MonadMask', which it can't do.

  624 bracketTopLevel acquire release action =

  633 io :: IO a -> TopLevel a
  634 io f = (TopLevel_ (liftIO f))
  635        `catches`
  636        [ Handler (\(ex :: X86Unsupported) -> handleX86Unsupported ex)
  637        , Handler (\(ex :: X86Error) -> handleX86Error ex)
  638        , Handler handleTopLevel
  639        , Handler handleIO
  640        ]
  641   where
  642     rethrow :: X.Exception ex => ex -> TopLevel a
  643     rethrow ex =
  644       do stk <- getStackTrace
  645          throwM (SS.TraceException stk (X.SomeException ex))
  646 
  647     handleTopLevel :: SS.TopLevelException -> TopLevel a
  648     handleTopLevel e = rethrow e
  649 
  650     handleIO :: X.IOException -> TopLevel a
  651     handleIO e
  652       | IOError.isUserError e =
  653           do pos <- getPosition
  654              rethrow (SS.TopLevelException pos (dropEnd 1 . drop 12 $ show e))
  655       | otherwise = rethrow e
  656 
  657     handleX86Unsupported (X86Unsupported path s) =
  658       do let pos = SS.FileOnlyPos path
  659          rethrow (SS.TopLevelException pos ("Unsupported x86 feature: " ++ s))

  661     handleX86Error (X86Error path optfunc s) =
  662       do let pos = case optfunc of
  663                Nothing -> SS.FileOnlyPos path
  664                Just func -> SS.FileAndFunctionPos path func
  665          rethrow (SS.TopLevelException pos ("Error in x86 code: " ++ s))

  829 extendEnv ::
  830   SharedContext ->
  831   SS.LName -> Maybe SS.Schema -> Maybe String -> Value -> TopLevelRW -> IO TopLevelRW
  832 extendEnv sc x mt md v rw =
  833   do ce' <-
  834        case v of
  835          VTerm t ->
  836            pure $ CEnv.bindTypedTerm (ident, t) ce
  837          VType s ->
  838            pure $ CEnv.bindType (ident, s) ce
  839          VInteger n ->
  840            pure $ CEnv.bindInteger (ident, n) ce
  841          VCryptolModule m ->
  842            pure $ CEnv.bindCryptolModule (modname, m) ce
  843          VString s ->
  844            do tt <- typedTermOfString sc (unpack s)
  845               pure $ CEnv.bindTypedTerm (ident, tt) ce
  846          VBool b ->
  847            do tt <- typedTermOfBool sc b
  848               pure $ CEnv.bindTypedTerm (ident, tt) ce
  849          _ ->
  850            pure ce
  851      pure $
  852       rw { rwValues  = M.insert name v (rwValues rw)
  853          , rwValueTypes = maybeInsert name mt (rwValueTypes rw)
  854          , rwDocs    = maybeInsert (SS.getVal name) md (rwDocs rw)
  855          , rwCryptol = ce'
  856          }
  857   where
  858     name = x
  859     -- XXX why is this using getOrig?
  860     ident = T.mkIdent (SS.getOrig x)
  861     modname = T.packModName [SS.getOrig x]
  862     ce = rwCryptol rw

  864 typedTermOfString :: SharedContext -> String -> IO TypedTerm
  865 typedTermOfString sc str =
  866   do let schema = Cryptol.tMono (Cryptol.tString (length str))
  867      bvNat <- scGlobalDef sc "Prelude.bvNat"
  868      bvNat8 <- scApply sc bvNat =<< scNat sc 8
  869      byteT <- scBitvector sc 8
  870      let scChar c = scApply sc bvNat8 =<< scNat sc (fromIntegral (fromEnum c))
  871      ts <- traverse scChar str
  872      trm <- scVector sc byteT ts
  873      pure (TypedTerm (TypedTermSchema schema) trm)
  874 
  875 typedTermOfBool :: SharedContext -> Bool -> IO TypedTerm
  876 typedTermOfBool sc b =
  877   do let schema = Cryptol.tMono Cryptol.tBit
  878      trm <- scBool sc b
  879      pure (TypedTerm (TypedTermSchema schema) trm)

  944 runProofScript ::
  945   ProofScript a ->
  946   Prop ->
  947   ProofGoal ->
  948   Maybe ProgramLoc ->

 1410 underStateT :: (forall b. m b -> m b) -> StateT s m a -> StateT s m a
 1411 underStateT doUnder action = StateT $ \s -> doUnder (runStateT action s)

 1413 underReaderT :: (forall b. m b -> m b) -> ReaderT s m a -> ReaderT s m a
 1414 underReaderT doUnder action = ReaderT $ \env -> doUnder (runReaderT action env)

 1417 underExceptT doUnder action = ExceptT $ doUnder (runExceptT action)

 1419 -- | Implement stack tracing by adding error handlers that rethrow
 1420 -- user errors, prepended with the given string.
 1421 addTrace :: String -> Value -> Value
 1422 addTrace str val =
 1423   case val of
 1424     VLambda        f -> VLambda        (\x -> addTrace str `fmap` addTraceTopLevel str (f x))
 1425     VTopLevel      m -> VTopLevel      (addTrace str `fmap` addTraceTopLevel str m)
 1426     VProofScript   m -> VProofScript   (addTrace str `fmap` addTraceProofScript str m)
 1427     VBind pos v1 v2  -> VBind pos      (addTrace str v1) (addTrace str v2)
 1428     VLLVMCrucibleSetup (LLVMCrucibleSetupM m) -> VLLVMCrucibleSetup $ LLVMCrucibleSetupM $
 1429         addTrace str `fmap` underReaderT (underStateT (addTraceTopLevel str)) m
 1430   -- TODO? JVM setup blocks too?
 1431     _                -> val
 1432 
 1433 addTraceProofScript :: String -> ProofScript a -> ProofScript a
 1434 addTraceProofScript str (ProofScript m) = ProofScript (underExceptT (underStateT (addTraceTopLevel str)) m)
 1435 
 1436 addTraceTopLevel :: String -> TopLevel a -> TopLevel a
 1437 addTraceTopLevel str (TopLevel_ action) =
 1438   TopLevel_ (local (\ro -> ro{ roStackTrace = str : roStackTrace ro }) action)
