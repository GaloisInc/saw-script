   26 -- | Startup function, called from main.
   27 --
   28 --   If a filename is given, the file is loaded and run in batch mode
   29 --   (as if typed into the REPL). Otherwise we take interactive input
   30 --   from stdin.
   31 --
   32 run :: Maybe FilePath -> Options -> IO ()
   33 run mfile opts = do
   34   let batchmode = isJust mfile
   35   unless batchmode $
   36       displayLogo $ useColor opts
   37   rst <- initREPL batchmode opts reenterTopLevel reenterProofScript
   38   (_result, _rst') <- runREPL (repl mfile) rst
   39   return ()
   40 
   41 -- | Function to enter a nested TopLevel REPL, passed down into the
   42 --   interpreter to avoid cyclic dependencies.
   43 --
   44 --   Note that there isn't any REPL state that needs to be transferred
   45 --   to the nested instance, only the interpreter state.
   46 --
   47 reenterTopLevel :: TopLevelRO -> TopLevelRW -> IO TopLevelRW
   48 reenterTopLevel ro rw = do
   49     let rst = resumeREPL ro rw Nothing
   50     (_result, rst') <- runREPL (repl Nothing) rst
   51     return $ rTopLevelRW rst'
   52 
   53 -- | Function to enter a nested ProofScript REPL, passed down into the
   54 --   interpreter to avoid cyclic dependencies.
   55 --
   56 --   Note that there isn't any REPL state that needs to be transferred
   57 --   to the nested instance, only the interpreter state.
   58 --
   59 reenterProofScript ::
   60     TopLevelRO -> TopLevelRW -> ProofState -> IO (TopLevelRW, ProofState)
   61 reenterProofScript ro rw pst = do
   62     let rst = resumeREPL ro rw (Just pst)
   63     (_result, rst') <- runREPL (repl Nothing) rst
   64     let rw' = rTopLevelRW rst'
   65         pst' = case rProofState rst' of
   66           Nothing -> panic "reenterProofScript" ["Proof state disappeared!?"]
   67           Just ps -> ps
   68     return (rw', pst')
