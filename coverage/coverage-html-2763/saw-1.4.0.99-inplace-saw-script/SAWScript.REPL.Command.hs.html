   72 failOn :: [Text] -> REPL a
   73 failOn errs = case (reverse errs) of
   74     [] -> panic "failOn" ["Failure with no error messages"]
   75     lastMsg : revRestMsgs -> do
   76         liftIO $ mapM_ TextIO.putStrLn (reverse revRestMsgs)
   77         fail (Text.unpack lastMsg)

   79 cdCmd :: FilePath -> REPL ()
   80 cdCmd f
   81     | null f =
   82         liftIO $ putStrLn "Error: The :cd command requires a path argument"
   83     | otherwise = do
   84   exists <- liftIO $ doesDirectoryExist f
   85   if exists
   86     then liftIO $ setCurrentDirectory f
   87     else do
   88         let f' = "`" <> Text.pack f <> "'"
   89             msg = "Directory " <> f' <> " not found or not a directory"
   90         liftIO $ TextIO.putStrLn msg

   92 envCmd :: REPL ()
   93 envCmd = do
   94   (rbenv, scopes) <- getSAWScriptVarEnv
   95   liftIO $ do
   96       let blankline = TextIO.putStrLn ""
   97 
   98       unless (null rbenv) $ do
   99           let printrb (x, ty) = do
  100                 let ty' = PPS.pShowText ty
  101                 TextIO.putStrLn (x <> " : rebindable " <> ty')
  102           mapM_ printrb rbenv
  103           blankline
  104 
  105       let printscope entries = do
  106             let printentry (x, ty) = do
  107                   let ty' = PPS.pShowText ty
  108                   TextIO.putStrLn (x <> " : " <> ty')
  109             mapM_ printentry entries
  110 
  111       -- Insert a blank line in the output where there's a scope boundary
  112       sequence_ $ intersperse blankline $ map printscope scopes

  114 helpCmd :: Text -> REPL ()
  115 helpCmd cmd
  116   | Text.null cmd = liftIO (mapM_ TextIO.putStrLn genericHelp)
  117   | otherwise =
  118     do rw <- getTopLevelRW
  119        -- Note: there's no rebindable builtins and thus no way to
  120        -- attach help text to anything rebindable, so we can ignore
  121        -- the rebindables.
  122        let Environ varenv _tyenv _cryenvs = rwEnviron rw
  123        case ScopedMap.lookup cmd varenv of
  124          Just (_pos, _lc, _ty, _v, Just doc) ->
  125            liftIO $ mapM_ TextIO.putStrLn doc
  126          Just (_pos, _lc, _ty, _v, Nothing) -> do
  127            liftIO $ putStrLn $ "// No documentation is available."
  128            typeOfCmd cmd
  129          Nothing ->
  130            liftIO $ putStrLn "// No such command."
  131 -- FIXME? can we restore the ability to lookup doc strings from Cryptol?
  132 --  | Just (ec,_) <- lookup cmd builtIns =
  133 --                liftIO $ print $ helpDoc ec

  144 searchCmd :: Text -> REPL ()
  145 searchCmd str
  146   | Text.null str =
  147       let msg = "Error: The :search command requires at least one argument" in
  148       liftIO $ putStrLn msg
  149   | otherwise = do
  150 
  151      -- FUTURE: it would be nice to be able to use the words
  152      -- "experimental" and "deprecated" in the search term to match
  153      -- against the lifecycle, to allow doing stuff like searching
  154      -- for deprecated functions that take Terms.
  155 
  156      -- Always search the entire environment and recognize all type
  157      -- names in the user's pattern, regardless of whether
  158      -- enable_experimental or enable_deprecated is in effect. It is
  159      -- definitely confusing to search for a hidden type and have
  160      -- search treat that as a free type variable and match all kinds
  161      -- of things you didn't intend. It is also better to retrieve
  162      -- invisible/deprecated items and report their existence than to
  163      -- hide them from the search.
  164      let avail = everythingAvailable
  165 
  166      rw <- getTopLevelRW
  167      let environ = rwEnviron rw
  168          rebindables = rwRebindables rw
  169      errs_or_pat <- liftIO $
  170          Loader.readSchemaPattern replFileName environ rebindables avail str
  171      pat <- case errs_or_pat of
  172            Left errs -> failOn errs
  173            Right p -> return p
  174 
  175      let primsAvail = rwPrimsAvail rw
  176      let Environ varenv tyenv _cryenv = environ
  177 
  178      let search = compileSearchPattern tyenv pat
  179          check (_pos, lc, ty, _v, _docs) =
  180              if matchSearchPattern search ty then
  181                  Just (lc, ty)
  182              else
  183                  Nothing
  184          checkRb (_pos, ty, _v) =
  185              if matchSearchPattern search ty then
  186                  Just (Current, ty)
  187              else
  188                  Nothing
  189          varMatches = Map.mapMaybe check $ ScopedMap.flatten varenv
  190          rbMatches = Map.mapMaybe checkRb rebindables
  191          allMatches = Map.union varMatches rbMatches
  192 
  193          -- Divide the results into visible, experimental-not-visible,
  194          -- and deprecated-not-visible.
  195          inspect ::
  196              SS.Name ->
  197              (PrimitiveLifecycle, SS.Schema) ->
  198              (Map SS.Name (PrimitiveLifecycle, SS.Schema),
  199               Map SS.Name (PrimitiveLifecycle, SS.Schema),
  200               Map SS.Name (PrimitiveLifecycle, SS.Schema)) ->
  201              (Map SS.Name (PrimitiveLifecycle, SS.Schema),
  202               Map SS.Name (PrimitiveLifecycle, SS.Schema),
  203               Map SS.Name (PrimitiveLifecycle, SS.Schema))
  204          inspect name (lc, ty) (vis, ex, dep) =
  205              if Set.member lc primsAvail then
  206                  (Map.insert name (lc, ty) vis, ex, dep)
  207              else case lc of
  208                Current -> oops
  209                WarnDeprecated -> oops
  210                HideDeprecated ->
  211                  (vis, ex, Map.insert name (lc, ty) dep)
  212                Experimental ->
  213                  (vis, Map.insert name (lc, ty) ex, dep)
  214              where
  215                oops =
  216                  panic "searchCmd" [
  217                      "Found non-visible object " <> Text.pack (show name) <>
  218                      " with unexpected lifecycle " <> Text.pack (show lc)
  219                  ]
  220          (visMatches, expMatches, depMatches) =
  221              let empty = (Map.empty, Map.empty, Map.empty) in
  222              Map.foldrWithKey inspect empty allMatches
  223 
  224          printMatch (name, (lc, ty)) = do
  225            let ty' = PPS.pShowText ty
  226                lc' = case lc of
  227                    Current -> ""
  228                    WarnDeprecated -> "  (DEPRECATED AND WILL WARN)"
  229                    HideDeprecated -> "  (DEPRECATED AND UNAVAILABLE BY DEFAULT)"
  230                    Experimental -> "  (EXPERIMENTAL)"
  231            TextIO.putStrLn (name <> " : " <> ty' <> lc')
  232          printMatches matches =
  233            liftIO $ mapM_ printMatch (Map.assocs matches)
  234 
  235          moreMatches matches =
  236              let n = Map.size matches in
  237              if n == 1 then "1 more match"
  238              else show n ++ " more matches"
  239          alsoExperimental =
  240              if not (Map.null expMatches) then
  241                  liftIO $ putStrLn $ moreMatches expMatches ++ " tagged " ++
  242                                  "experimental; use enable_experimental to " ++
  243                                  "see them"
  244              else
  245                  pure ()
  246          alsoDeprecated =
  247              if not (Map.null depMatches) then
  248                  liftIO $ putStrLn $ moreMatches depMatches ++ " tagged " ++
  249                                  "deprecated; use enable_deprecated to " ++
  250                                  "see them"
  251              else
  252                  pure ()
  253 
  254      if not (Map.null visMatches) then do
  255          printMatches visMatches
  256          alsoExperimental
  257          alsoDeprecated
  258      else do
  259          liftIO $ putStrLn "No matches."
  260          if not (Map.null expMatches) then do
  261              liftIO $ putStrLn $ "The following experimental matches " ++
  262                                  "require enable_experimental:"
  263              printMatches expMatches
  264              alsoDeprecated
  265          else if not (Map.null depMatches) then do
  266              liftIO $ putStrLn $ "The following deprecated matches require " ++
  267                                  "enable_deprecated:"
  268              printMatches depMatches
  269          else
  270              pure ()

  287 typeOfCmd :: Text -> REPL ()
  288 typeOfCmd str
  289   | Text.null str =
  290         liftIO $ putStrLn "Error: The :type command requires an argument"
  291   | otherwise = do
  292      rw <- getTopLevelRW
  293      let environ = rwEnviron rw
  294          rebindables = rwRebindables rw
  295          avail = rwPrimsAvail rw
  296      errs_or_expr <- liftIO $
  297          Loader.readExpression replFileName environ rebindables avail str
  298      (schema, _expr) <- case errs_or_expr of
  299          Left errs -> failOn errs
  300          Right info -> return info
  301      liftIO $ TextIO.putStrLn $ PPS.pShowText schema

  307 -- | Command description.
  308 data CommandDescr = CommandDescr
  309   { cName :: Text
  310   , cAliases :: [Text]
  311   , cBody :: CommandBody
  312   , cHelp :: Text
  313   }
  314 
  315 -- When we use `nub` on lists of commands, do the comparison by name.
  316 instance Eq CommandDescr where
  317   (==) = (==) `on` cName
  318 
  319 -- | Schema for argument types of REPL commands.
  320 --
  321 -- XXX: we could distinguish commands that take directories
  322 -- (like :cd) from those that take regular files (though we
  323 -- don't have any)
  324 --
  325 -- XXX: should maybe also distinguish "multiple args of this type"
  326 -- from "one arg of this type".
  327 data CommandBody
  328   = ExprArg     (Text     -> REPL ())
  329   | TypeArgs    (Text     -> REPL ())
  330   | FilenameArg (FilePath -> REPL ())
  331   | ShellArg    (Text     -> REPL ())
  332   | NoArg       (REPL ())
  333 
  334 type CommandMap = Trie CommandDescr
  335 
  336 
  337 -- | Convert the command list to a Trie, expanding aliases.
  338 makeCommands :: [CommandDescr] -> CommandMap
  339 makeCommands list  = foldl insert Trie.empty (concatMap expandAliases list)
  340   where
  341   insert m (name, d) = Trie.insert name d m
  342   expandAliases :: CommandDescr -> [(Text, CommandDescr)]
  343   expandAliases d = (cName d, d) : zip (cAliases d) (repeat d)
  344 
  345 -- | REPL command parsing.
  346 commands :: CommandMap
  347 commands = makeCommands commandList

  350 nbCommandList :: [CommandDescr]

  358   , CommandDescr ":type" [":t"]  (ExprArg typeOfCmd)

  360   , CommandDescr ":?"    []      (ExprArg helpCmd)

  366 commandList :: [CommandDescr]
  367 commandList  =
  368   nbCommandList ++
  369   [ CommandDescr ":quit" []   (NoArg quitCmd)

  371   , CommandDescr ":cd"   []   (FilenameArg cdCmd)

  377 genericHelp :: [Text]
  378 genericHelp = map cmdHelp commandList
  379   where
  380   cmdHelp cmd = Text.concat [ "  ", cName cmd, pad (cName cmd), cHelp cmd ]
  381   padding     = 2 + maximum (map (Text.length . cName) commandList)
  382   pad n       = Text.replicate (max 0 (padding - Text.length n)) " "

  388 -- | Execute some SAWScript text.
  389 executeSAWScriptText :: Text -> REPL ()
  390 executeSAWScriptText str = exceptionProtect $ do
  391   errs_or_stmt <- liftIO $ Loader.readStmtSemiUnchecked replFileName str
  392   case errs_or_stmt of
  393     Left errs -> failOn errs
  394     Right stmt -> do
  395          mbPst <- getProofState
  396          case mbPst of
  397            Nothing -> void $ liftTopLevel (interpretTopStmt True stmt)
  398            Just _  -> void $ liftProofScript (interpretTopStmt True stmt)

  404 -- | Strip leading space.
  405 sanitize :: Text -> Text
  406 sanitize  = Text.dropWhile isSpace
  407 
  408 -- | Find commands that begin with a given prefix.
  409 --
  410 -- If given a string that's both itself a command and a prefix of
  411 -- something else, choose that command.
  412 --
  413 -- Deduplicate the results to avoid silliness with command aliases.
  414 searchCommandsByPrefix :: Text -> [CommandDescr]
  415 searchCommandsByPrefix prefix =
  416     nub $ Trie.lookupWithExact prefix commands
  417 
  418 -- | Find the single command that begins with a given prefix.
  419 --
  420 -- If given a string that's both itself a command and a prefix of
  421 -- something else, choose that command. If given a string that's a
  422 -- prefix of one thing, choose that thing. If given a string that's a
  423 -- prefix of multiple things, fail.
  424 --
  425 -- Deduplicate the results to avoid silliness with command aliases.
  426 searchExactCommandByPrefix :: Text -> Maybe CommandDescr
  427 searchExactCommandByPrefix prefix =
  428     case searchCommandsByPrefix prefix of
  429         [cmd] -> Just cmd
  430         _ -> Nothing
  431 
  432 -- | Do tilde-expansion on filenames.
  433 expandHome :: Text -> REPL FilePath
  434 expandHome path =
  435     case Text.uncons path of
  436         Nothing -> pure ""
  437         Just ('~', more) -> case Text.uncons more of
  438             Just (c, more') | isPathSeparator c -> do
  439                 dir <- liftIO getHomeDirectory
  440                 return (dir </> Text.unpack more')
  441             _ ->
  442                 pure $ Text.unpack path
  443         Just _ ->
  444             pure $ Text.unpack path
  445 
  446 -- | Execute a REPL :-command.
  447 executeReplCommand :: CommandDescr -> [Text] -> REPL ()
  448 executeReplCommand cmd args0 =
  449     let noargs action args = case args of
  450           [] -> action
  451           _ -> do
  452               let msg = "The command " <> cName cmd <> " takes no arguments"
  453               liftIO $ TextIO.putStrLn msg
  454     in
  455     let onearg action args = case args of
  456           [] -> action ""
  457           [arg] -> action arg
  458           _ -> do
  459               let msg = "The command " <> cName cmd <>
  460                         " takes only one argument"
  461               liftIO $ TextIO.putStrLn msg
  462     in
  463     exceptionProtect $ case cBody cmd of
  464         ExprArg action ->
  465             onearg action args0
  466         TypeArgs action ->
  467             action (Text.intercalate " " args0)
  468         FilenameArg action -> do
  469             args' <- mapM expandHome args0
  470             onearg action args'
  471         ShellArg action ->
  472             onearg action args0
  473         NoArg action ->
  474             noargs action args0
  475 
  476 -- | Execute REPL :-command text.
  477 executeReplCommandText :: Text -> REPL ()
  478 executeReplCommandText text =
  479     let textWords =
  480            filter (\w -> not $ Text.null w) $ Text.split isSpace text
  481     in
  482     case textWords of
  483         [] -> pure ()
  484         cmdName : args ->
  485             case searchCommandsByPrefix cmdName of
  486                 [] ->
  487                     -- Historically SAW accepts ":?cmd" without a space
  488                     if Text.isPrefixOf ":?" cmdName then
  489                         executeReplCommandText $ ":? " <> Text.drop 2 cmdName
  490                     else do
  491                         let msg = "Unknown command: " <> cmdName
  492                         liftIO $ TextIO.putStrLn msg
  493                 [cmd] ->
  494                     executeReplCommand cmd args
  495                 cmds -> liftIO $ do
  496                     let msg1 = cmdName <> " is ambiguous; it could mean one of:"
  497                         msg2 = Text.intercalate ", " $ map cName cmds
  498                     TextIO.putStrLn $ msg1
  499                     TextIO.putStrLn $ "\t" <> msg2

  501 -- | Execute some text typed at the REPL.
  502 executeText :: Text -> REPL ()
  503 executeText text = do
  504     -- skip whitespace
  505     let text' = sanitize text
  506     case Text.uncons text' of
  507         Nothing ->
  508             pure ()
  509         Just (':', _) ->
  510             executeReplCommandText text'
  511         Just _ ->
  512             executeSAWScriptText text'
