    6 {- |
    7 Module      : SAWCore.OpenTerm
    8 Copyright   : Galois, Inc. 2018
    9 License     : BSD3
   10 Stability   : experimental
   11 Portability : non-portable (language extensions)
   12 
   13 This module defines a convenient DSL for building terms in a pure way,
   14 where sub-terms can be freely composed and combined into super-terms
   15 without monadic sequencing or 'IO' computations; the 'IO' computation
   16 is only run at the top level when all the term-building is complete.
   17 
   18 Users of this interface can also build binding constructs like lambda-
   19 and pi-abstractions using higher-order abstract syntax, meaning that
   20 the bodies of these term constructs are specified as Haskell functions
   21 that take in terms for the bound variables.
   22 
   23 Since many function names are short and may clash with names from
   24 other modules, this module is intended to be imported @qualified@,
   25 e.g.
   26 
   27 >  import SAWCore.OpenTerm (OpenTerm)
   28 >  import qualified SAWCore.OpenTerm as OpenTerm
   29 
   30 To use the 'OpenTerm' API, the caller builds up 'OpenTerm's using a
   31 variety of combinators that mirror the SAW core 'Term' structure.
   32 As some useful examples of 'OpenTerm' operations, 'apply' applies one
   33 'OpenTerm' to another, 'global' builds an 'OpenTerm' for a global
   34 identifier, and 'lambda' builds a lambda-abstraction. For instance,
   35 the SAW core term
   36 
   37 > \ (f : Bool -> Bool) (x : Bool) -> f x
   38 
   39 can be built with the 'OpenTerm' expression
   40 
   41 > let boolTy = global "Prelude.Bool" in
   42 > lambda "f" (arrow boolTy boolTy) $ \f ->
   43 > lambda "x" boolTy $ \x ->
   44 > apply f x
   45 
   46 Existing SAW core 'Term's can be used in 'OpenTerm' by applying 'term'.
   47 At the top level, 'complete' then "completes" an 'OpenTerm' by running
   48 its underlying 'IO' computation to build and type-check the resulting
   49 SAW core 'Term'.
   50 -}

   90 -- | An open term is represented as a type-checking computation that computes a
   91 -- SAW core term and its type
   92 newtype OpenTerm = OpenTerm { unOpenTerm :: TCM SC.Term }
   93 
   94 -- | \"Complete\" an 'OpenTerm' to a closed term or 'fail' on type-checking
   95 -- error
   96 complete :: SharedContext -> OpenTerm -> IO Term
   97 complete sc (OpenTerm termM) =
   98   either (fail . show) return =<<
   99   runTCM (SC.rawTerm <$> termM) sc
  100 
  101 -- | \"Complete\" an 'OpenTerm' to a closed term for its type
  102 completeType :: SharedContext -> OpenTerm -> IO Term
  103 completeType sc (OpenTerm termM) =
  104   either (fail . show) return =<<
  105   runTCM (SC.rawType <$> termM) sc
  106 
  107 -- | Embed a 'Term' into an 'OpenTerm'.
  108 term :: Term -> OpenTerm
  109 term t = OpenTerm $ typeInferComplete t
  110 
  111 -- | Return type of an 'OpenTerm' as an 'OpenTerm'.
  112 typeOf :: OpenTerm -> OpenTerm
  113 typeOf (OpenTerm m) =
  114   OpenTerm $ do t <- m
  115                 liftTCM SC.scTypeOf t
  116 
  117 -- | Build an 'OpenTerm' from a 'FlatTermF'
  118 flat :: FlatTermF OpenTerm -> OpenTerm
  119 flat ftf = OpenTerm $
  120   (sequence (fmap unOpenTerm ftf) >>= typeInferComplete)
  121 
  122 -- | Build an 'OpenTerm' for a sort
  123 sort :: Sort -> OpenTerm
  124 sort s = flat (Sort s noFlags)
  125 
  126 -- | Build an 'OpenTerm' for a natural number literal
  127 nat :: Natural -> OpenTerm
  128 nat n = OpenTerm $ liftTCM SC.scNat n
  129 
  130 -- | The 'OpenTerm' for the unit value
  131 unit :: OpenTerm
  132 unit = flat UnitValue
  133 
  134 -- | The 'OpenTerm' for the unit type
  135 unitType :: OpenTerm
  136 unitType = flat UnitType

  138 -- | Build a SAW core string literal.
  139 stringLit :: Text -> OpenTerm
  140 stringLit = flat . StringLit

  142 -- | Return the SAW core type @String@ of strings.
  143 stringType :: OpenTerm
  144 stringType = global "Prelude.String"

  146 -- | The 'True' value as a SAW core term
  147 true :: OpenTerm
  148 true = global "Prelude.True"

  150 -- | The 'False' value as a SAW core term

  152 false = global "Prelude.False"

  154 -- | Convert a 'Bool' to a SAW core term
  155 bool :: Bool -> OpenTerm
  156 bool True = global "Prelude.True"
  157 bool False = global "Prelude.False"

  159 -- | The 'Bool' type as a SAW core term
  160 boolType :: OpenTerm
  161 boolType = global "Prelude.Bool"

  163 -- | Build an 'OpenTerm' for an array literal
  164 arrayValue :: OpenTerm -> [OpenTerm] -> OpenTerm
  165 arrayValue tp elems =
  166   flat $ ArrayValue tp $ V.fromList elems
  167 
  168 -- | Create a SAW core term for a bitvector literal
  169 bvLit :: [Bool] -> OpenTerm
  170 bvLit bits =
  171   arrayValue boolType $ map bool bits
  172 
  173 -- | Create a SAW core term for a vector type
  174 vectorType :: OpenTerm -> OpenTerm -> OpenTerm
  175 vectorType n a = applyGlobal "Prelude.Vec" [n,a]
  176 
  177 -- | Create a SAW core term for the type of a bitvector
  178 bvType :: Integral a => a -> OpenTerm
  179 bvType n =
  180   apply (global "Prelude.Vec")
  181   [nat (fromIntegral n), boolType]

  183 -- | Build an 'OpenTerm' for a pair
  184 pair :: OpenTerm -> OpenTerm -> OpenTerm
  185 pair t1 t2 = flat $ PairValue t1 t2
  186 
  187 -- | Build an 'OpenTerm' for a pair type
  188 pairType :: OpenTerm -> OpenTerm -> OpenTerm
  189 pairType t1 t2 = flat $ PairType t1 t2
  190 
  191 -- | Build an 'OpenTerm' for the left projection of a pair
  192 pairLeft :: OpenTerm -> OpenTerm
  193 pairLeft t = flat $ PairLeft t
  194 
  195 -- | Build an 'OpenTerm' for the right projection of a pair
  196 pairRight :: OpenTerm -> OpenTerm
  197 pairRight t = flat $ PairRight t
  198 
  199 -- | Build a right-nested tuple as an 'OpenTerm'
  200 tuple :: [OpenTerm] -> OpenTerm
  201 tuple = foldr pair unit
  202 
  203 -- | Build a right-nested tuple type as an 'OpenTerm'
  204 tupleType :: [OpenTerm] -> OpenTerm
  205 tupleType = foldr pairType unitType
  206 
  207 -- | Project the @n@th element of a right-nested tuple type
  208 projTuple :: Integer -> OpenTerm -> OpenTerm
  209 projTuple 0 t = pairLeft t
  210 projTuple i t = projTuple (i-1) (pairRight t)
  211 
  212 -- | Build a right-nested tuple as an 'OpenTerm' but without adding a final unit
  213 -- as the right-most element
  214 tuple' :: [OpenTerm] -> OpenTerm
  215 tuple' [] = unit
  216 tuple' ts = foldr1 pair ts
  217 
  218 -- | Build a right-nested tuple type as an 'OpenTerm' but without adding a final
  219 -- unit type as the right-most element
  220 tupleType' :: [OpenTerm] -> OpenTerm
  221 tupleType' [] = unitType
  222 tupleType' ts = foldr1 pairType ts
  223 
  224 -- | Project the @i@th element from a term of a right-nested tuple term that
  225 -- does not have a final unit type as the right-most type. The first argument is
  226 -- the number of types used to make the tuple type and the second is the index.
  227 projTuple' :: Natural -> Natural -> OpenTerm -> OpenTerm
  228 projTuple' 0 _ _ =
  229   panic "projTupleOpenTerm'" ["Projection of empty tuple!"]
  230 projTuple' 1 0 tup = tup
  231 projTuple' _ 0 tup = pairLeft tup
  232 projTuple' len i tup =
  233   projTuple' (len-1) (i-1) $ pairRight tup
  234 
  235 -- | Build a record value as an 'OpenTerm'
  236 record :: [(FieldName, OpenTerm)] -> OpenTerm
  237 record flds_ts =
  238   OpenTerm $ do let (flds,ots) = unzip flds_ts
  239                 ts <- mapM unOpenTerm ots
  240                 typeInferComplete $ RecordValue $ zip flds ts
  241 
  242 -- | Build a record type as an 'OpenTerm'
  243 recordType :: [(FieldName, OpenTerm)] -> OpenTerm
  244 recordType flds_ts =
  245   OpenTerm $ do let (flds,ots) = unzip flds_ts
  246                 ts <- mapM unOpenTerm ots
  247                 typeInferComplete $ RecordType $ zip flds ts
  248 
  249 -- | Project a field from a record
  250 projRecord :: OpenTerm -> FieldName -> OpenTerm
  251 projRecord (OpenTerm m) f =
  252   OpenTerm $ do t <- m
  253                 typeInferComplete $ RecordProj t f
  254 
  255 -- | Build an 'OpenTerm' for a global name with a definition
  256 global :: Ident -> OpenTerm
  257 global ident =
  258   OpenTerm (liftTCM SC.scGlobal ident)
  259 
  260 -- | Build an 'OpenTerm' for a named variable.
  261 variable :: VarName -> Term -> OpenTerm
  262 variable x t = OpenTerm (liftTCM scVariable x t >>= typeInferComplete)
  263 
  264 -- | Apply an 'OpenTerm' to another.
  265 app :: OpenTerm -> OpenTerm -> OpenTerm
  266 app (OpenTerm f) (OpenTerm arg) =
  267   OpenTerm ((App <$> f <*> arg) >>= typeInferComplete)
  268 
  269 -- | Apply an 'OpenTerm' to a list of zero or more arguments.
  270 apply :: OpenTerm -> [OpenTerm] -> OpenTerm
  271 apply = foldl app
  272 
  273 -- | Apply a named global to a list of zero or more arguments.
  274 applyGlobal :: Ident -> [OpenTerm] -> OpenTerm
  275 applyGlobal ident = apply (global ident)
  276 
  277 -- | Compute the output type of applying a function of a given type to an
  278 -- argument. That is, given @tp@ and @arg@, compute the type of applying any @f@
  279 -- of type @tp@ to @arg@.
  280 applyPi :: OpenTerm -> OpenTerm -> OpenTerm
  281 applyPi (OpenTerm m_f) (OpenTerm m_arg) =
  282   OpenTerm $
  283   do f <- m_f
  284      arg <- m_arg
  285      ret <- applyPiTyped (NotFuncTypeInApp f arg) (SC.rawTerm f) arg
  286      typeInferComplete ret
  287 
  288 -- | Get the argument type of a function type, 'fail'ing if the input term is
  289 -- not a function type
  290 piArg :: OpenTerm -> OpenTerm
  291 piArg (OpenTerm m) =
  292   OpenTerm $ m >>= \case
  293   (unwrapTermF . SC.rawTerm -> Pi _ tp _) -> typeInferComplete tp
  294   t ->
  295     fail ("piArg: not a pi type: " ++
  296           scPrettyTermInCtx PPS.defaultOpts [] (SC.rawTerm t))
  297 
  298 -- | Build a lambda abstraction as an 'OpenTerm'
  299 lambda :: LocalName -> OpenTerm -> (OpenTerm -> OpenTerm) -> OpenTerm
  300 lambda x (OpenTerm tpM) body_f = OpenTerm $
  301   do tp <- tpM
  302      vn <- liftTCM scFreshVarName x
  303      var <- typeInferComplete $ Variable vn tp
  304      body <- unOpenTerm (body_f (OpenTerm (pure var)))
  305      typeInferComplete $ Lambda vn tp body
  306 
  307 -- | Build a nested sequence of lambda abstractions as an 'OpenTerm'
  308 lambdas :: [(LocalName, OpenTerm)] -> ([OpenTerm] -> OpenTerm) ->
  309                        OpenTerm
  310 lambdas xs_tps body_f =
  311   foldr (\(x,tp) rest_f xs ->
  312           lambda x tp (rest_f . (:xs))) (body_f . reverse) xs_tps []
  313 
  314 -- | Build a Pi abstraction as an 'OpenTerm'
  315 piType :: LocalName -> OpenTerm -> (OpenTerm -> OpenTerm) -> OpenTerm
  316 piType x (OpenTerm tpM) body_f = OpenTerm $
  317   do tp <- tpM
  318      nm <- liftTCM scFreshVarName x
  319      var <- typeInferComplete $ Variable nm tp
  320      body <- unOpenTerm (body_f (OpenTerm (pure var)))
  321      typeInferComplete $ Pi nm tp body
  322 
  323 -- | Build a non-dependent function type.
  324 arrow :: OpenTerm -> OpenTerm -> OpenTerm
  325 arrow t1 t2 =
  326   OpenTerm $
  327   do t1' <- unOpenTerm t1
  328      t2' <- unOpenTerm t2
  329      typeInferComplete $ Pi wildcardVarName t1' t2'
  330 
  331 -- | Build a nested sequence of Pi abstractions as an 'OpenTerm'
  332 piMulti :: [(LocalName, OpenTerm)] -> ([OpenTerm] -> OpenTerm) ->
  333                    OpenTerm
  334 piMulti xs_tps body_f =
  335   foldr (\(x,tp) rest_f xs ->
  336           piType x tp (rest_f . (:xs))) (body_f . reverse) xs_tps []
  337 
  338 -- | Build a let expression as an 'OpenTerm'. This is equivalent to
  339 -- > 'apply' ('lambda' x tp body) rhs
  340 mkLet ::
  341   LocalName -> OpenTerm -> OpenTerm -> (OpenTerm -> OpenTerm) -> OpenTerm
  342 mkLet x tp rhs body_f = app (lambda x tp body_f) rhs
  343 
  344 -- | Build a let expression as an 'OpenTerm'. This is equivalent to
  345 -- > 'apply' ('lambda' x tp body) rhs
  346 sawLet ::
  347   LocalName -> OpenTerm -> OpenTerm -> OpenTerm ->
  348   (OpenTerm -> OpenTerm) -> OpenTerm
  349 sawLet x tp tp_ret rhs body_f =
  350   apply (global "Prelude.sawLet")
  351   [tp, tp_ret, rhs, lambda x tp body_f]
  352 
  353 -- | Build a bitvector type with the given length
  354 bitvectorType :: OpenTerm -> OpenTerm
  355 bitvectorType w =
  356   applyGlobal "Prelude.Vec" [w, global "Prelude.Bool"]
  357 
  358 -- | Build a SAW core term for a list with the given element type
  359 list :: OpenTerm -> [OpenTerm] -> OpenTerm
  360 list tp elems =
  361   foldr (\x l -> applyGlobal "Prelude.Cons" [tp, x, l])
  362   (applyGlobal "Prelude.Nil" [tp]) elems
  363 
  364 -- | Build the type @Either a b@ from types @a@ and @b@
  365 eitherType :: OpenTerm -> OpenTerm -> OpenTerm
  366 eitherType a b = applyGlobal "Prelude.Either" [a, b]
