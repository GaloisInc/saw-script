  200 constMap :: forall sym. Sym sym => sym -> Map Ident (SPrim sym)
  201 constMap sym =
  202   Map.union (Prims.constMap (prims sym)) $
  203   Map.fromList
  204   [
  205   -- Shifts
  206     ("Prelude.bvShl" , bvShLOp sym)
  207   , ("Prelude.bvShr" , bvShROp sym)

  241 swBvWidth :: SWord sym -> Int
  242 swBvWidth x
  243   | w <= toInteger (maxBound :: Int) = fromInteger w
  244   | otherwise = croak
  245  where w = SW.bvWidth x
  246        croak = panic "swBvWidth" [
  247            "Bitvector too long: width " <> Text.pack (show w) <>
  248                ", value " <> Text.pack (show x)
  249         ]

  256   Sym sym => sym -> SValue sym -> IO (SWord sym)

  293 -- primitive intToNat : Integer -> Nat;
  294 -- intToNat x == max 0 x
  295 intToNatOp :: forall sym. Sym sym => sym -> SPrim sym
  296 intToNatOp sym =
  297   Prims.intFun $ \i ->
  298   Prims.Prim $
  299     case W.asInteger i of
  300       Just i'
  301         | 0 <= i'   -> pure (VNat (fromInteger i'))
  302         | otherwise -> pure (VNat 0)
  303       Nothing ->
  304         do z <- W.intLit sym 0
  305            pneg <- W.intLt sym i z
  306            i' <- W.intIte sym pneg z i
  307            pure (VIntToNat i')

  341 -- | Shift left, shifting in copies of the given bit
  342 bvShl :: IsExprBuilder sym => sym -> Pred sym -> SWord sym -> SWord sym -> IO (SWord sym)
  343 bvShl sym z w i =
  344   W.iteM SW.bvIte sym z
  345     (do w' <- SW.bvNot sym w
  346         SW.bvNot sym =<< SW.bvShl sym w' i)
  347     (SW.bvShl sym w i)
  348 
  349 -- | Shift right, shifting in copies of the given bit
  350 bvShr :: IsExprBuilder sym => sym -> Pred sym -> SWord sym -> SWord sym -> IO (SWord sym)
  351 bvShr sym z w i =
  352   W.iteM SW.bvIte sym z
  353     (do w' <- SW.bvNot sym w
  354         SW.bvNot sym =<< SW.bvLshr sym w' i)
  355     (SW.bvLshr sym w i)
  356 
  357 liftShift :: IsExprBuilder sym =>
  358   sym ->
  359   (SWord sym -> SWord sym -> IO (SWord sym)) ->
  360   SWord sym -> Integer -> IO (SWord sym)
  361 liftShift sym f w i =
  362   f w =<< SW.bvLit sym (SW.bvWidth w) (i `min` SW.bvWidth w)
  363 
  364 liftRotate :: IsExprBuilder sym =>
  365   sym ->
  366   (SWord sym -> SWord sym -> IO (SWord sym)) ->
  367   SWord sym -> Integer -> IO (SWord sym)
  368 liftRotate sym f w i =
  369   f w =<< SW.bvLit sym (SW.bvWidth w) (i `mod` SW.bvWidth w)

  372 -- | op : (n : Nat) -> Vec n Bool -> Nat -> Vec n Bool
  373 bvShiftOp :: Sym sym => sym ->
  374              (SWord sym -> SWord sym -> IO (SWord sym)) ->
  375              (SWord sym -> Integer   -> IO (SWord sym)) -> SPrim sym
  376 bvShiftOp sym bvOp natOp =
  377   Prims.constFun  $                  -- additional argument? the size?
  378   wordFun sym $ \x ->                -- word to shift
  379   Prims.strictFun $ \y ->            -- amount to shift as a nat
  380   Prims.Prim $
  381     case y of
  382       VNat i   -> VWord <$> natOp x j
  383         where j = toInteger i `min` SW.bvWidth x
  384       VBVToNat _ v -> VWord <$> (bvOp x =<< toWord sym v)
  385       _        -> error $ unwords ["SAWCoreWhat4.What4.bvShiftOp", show y]
  386 
  387 -- bvShl : (w : Nat) -> Vec w Bool -> Nat -> Vec w Bool;
  388 bvShLOp :: forall sym. Sym sym => sym -> SPrim sym
  389 bvShLOp sym = bvShiftOp sym (SW.bvShl sym)
  390                     (liftShift sym (SW.bvShl sym))

  402 bvForall :: W.IsSymExprBuilder sym =>
  403   sym -> Natural -> (SWord sym -> IO (Pred sym)) -> IO (Pred sym)
  404 bvForall sym n f =
  405   do let indexSymbol = W.safeSymbol "i"
  406      case mkNatRepr n of
  407         Some w
  408           | Just LeqProof <- testLeq (knownNat @1) w ->
  409             withKnownNat w $ do
  410               i <- W.freshBoundVar sym indexSymbol $ W.BaseBVRepr w
  411               body <- f . DBV $ W.varExpr sym i
  412               W.forallPred sym i body
  413           | otherwise -> f ZBV

  429 -- NB: What4's division operation provides SMT-LIB's Euclidean division, which
  430 -- doesn't match the round-to-neg-infinity semantics of Cryptol, so we have to
  431 -- do some work to get the desired semantics.
  432 intDiv ::
  433   IsExprBuilder sym =>
  434   sym ->
  435   SymExpr sym BaseIntegerType ->
  436   SymExpr sym BaseIntegerType ->
  437   IO (SymExpr sym BaseIntegerType)
  438 intDiv sym x y = do
  439   neg <- W.intLt sym y =<< W.intLit sym 0
  440   case W.asConstantPred neg of
  441     Just False -> W.intDiv sym x y
  442     Just True  ->
  443        do xneg <- W.intNeg sym x
  444           yneg <- W.intNeg sym y
  445           W.intDiv sym xneg yneg
  446     Nothing ->
  447        do xneg <- W.intNeg sym x
  448           yneg <- W.intNeg sym y
  449           zneg <- W.intDiv sym xneg yneg
  450           z    <- W.intDiv sym x y
  451           W.intIte sym neg zneg z

  545 streamGet :: forall sym. Sym sym => sym -> SValue sym -> SValue sym -> IO (SValue sym)
  546 streamGet sym xs ix =
  547     case ix of
  548       VNat n -> lookupSStream xs n
  549       VBVToNat _ w ->
  550         do ilv <- toWord sym w
  551            selectV sym (lazyMux @sym (muxBVal sym)) ((2 ^ SW.bvWidth ilv) - 1) (lookupSStream xs) ilv
  552       v -> panic "streamGetOp" ["Expected Nat value, found: " <> Text.pack (show v)]

  554 lookupSStream :: SValue sym -> Natural -> IO (SValue sym)
  555 lookupSStream (VExtra (SStream f r)) n = do
  556    m <- readIORef r
  557    case Map.lookup n m of
  558      Just v  -> return v
  559      Nothing -> do v <- f n
  560                    writeIORef r (Map.insert n v m)
  561                    return v
  562 lookupSStream _ _ = fail "SAWCoreWhat4.What4.lookupSStream: expected Stream"

  603 -- @selectV sym merger maxValue valueFn vx@ treats @vx@ as an index, represented
  604 -- as a big-endian list of bits. It does a binary lookup, using @merger@ as an
  605 -- if-then-else operator. If the index is greater than @maxValue@, then it
  606 -- returns @valueFn maxValue@.
  607 selectV :: forall sym b.
  608   Sym sym =>
  609   sym ->
  610   (SBool sym -> IO b -> IO b -> IO b) -> Natural -> (Natural -> IO b) -> SWord sym -> IO b
  611 selectV sym merger maxValue valueFn vx =
  612   case SW.bvAsUnsignedInteger vx of
  613     Just i  -> valueFn (fromInteger i :: Natural)
  614     Nothing -> impl (swBvWidth vx) 0
  615   where
  616     impl :: Int -> Natural -> IO b
  617     impl _ x | x > maxValue || x < 0 = valueFn maxValue
  618     impl 0 y = valueFn y
  619     impl i y = do
  620       -- NB: `i` counts down in each iteration, so we use bvAtLE (a
  621       -- little-endian indexing function) to ensure that the bits are processed
  622       -- in big-endian order. Alternatively, we could have `i` count up and use
  623       -- bvAtBE (a big-endian indexing function), but we use bvAtLE as it is
  624       -- slightly cheaper to compute.
  625       p <- SW.bvAtLE sym vx (toInteger j)
  626       merger p (impl j (y `setBit` j)) (impl j y) where j = i - 1

  833 w4SolveBasic ::
  834   forall sym. IsSymExprBuilder sym =>
  835   sym ->
  836   SharedContext ->
  837   Map Ident (SPrim sym) {- ^ additional primitives -} ->
  838   Map VarIndex (SValue sym) {- ^ bindings for free variables -} ->
  839   IORef (SymFnCache sym) {- ^ cache for uninterpreted function symbols -} ->
  840   Set VarIndex {- ^ 'unints' Constants in this list are kept uninterpreted -} ->
  841   Term {- ^ term to simulate -} ->
  842   IO (SValue sym)
  843 w4SolveBasic sym sc addlPrims varMap ref unintSet t =
  844   do m <- scGetModuleMap sc
  845      let variable (VarName ix x) ty
  846             | Just v <- Map.lookup ix varMap = return v
  847             | otherwise = parseUninterpreted sym ref (mkUnintApp (Text.unpack x ++ "_" ++ show ix)) ty
  848      let uninterpreted nm ty
  849            | Set.member (nameIndex nm) unintSet =
  850              let vn = VarName (nameIndex nm) (toShortName (nameInfo nm))
  851              in Just (variable vn ty)
  852            | otherwise                          = Nothing
  853      let primHandler = Sim.defaultPrimHandler
  854      let mux = Prims.lazyMuxValue (prims sym)
  855      cfg <-
  856        Sim.evalGlobal m (constMap sym `Map.union` addlPrims)
  857        variable uninterpreted (recursor sym) primHandler mux
  858      Sim.evalSharedTerm cfg t

  886 w4SolveAssert :: forall sym.
  887   IsSymExprBuilder sym =>
  888   sym ->
  889   SharedContext ->
  890   Map VarIndex (SValue sym) {- ^ bindings for free variables -} ->
  891   IORef (SymFnCache sym) {- ^ cache for uninterpreted function symbols -} ->
  892   Set VarIndex {- ^ variables to hold uninterpreted -} ->
  893   SATAssert ->
  894   IO (SBool sym)
  895 w4SolveAssert sym sc varMap ref uninterp (BoolAssert x) =
  896   do bval <- w4SolveBasic sym sc mempty varMap ref uninterp x
  897      case bval of
  898        VBool v -> return v
  899        _ -> fail $ "w4SolveAssert: non-boolean result type. " ++ show bval
  900 
  901 w4SolveAssert sym sc varMap ref uninterp (UniversalAssert vars hyps concl) =
  902   do g <- case hyps of
  903             [] -> return concl
  904             _  -> do h <- scAndList sc hyps
  905                      scImplies sc h concl
  906      (svals,bndvars) <- boundFOTs sym vars
  907      let varMap' = foldl (\m ((vn,_fot), sval) -> Map.insert (vnIndex vn) sval m)
  908                          varMap
  909                          (zip vars svals) -- NB, boundFOTs will construct these lists to be the same length
  910      bval <- w4SolveBasic sym sc mempty varMap' ref uninterp g
  911      case bval of
  912        VBool v ->
  913          do final <- foldM (\p (Some bndvar) -> W.forallPred sym bndvar p) v bndvars
  914             return final
  915 
  916        _ -> fail $ "w4SolveAssert: non-boolean result type. " ++ show bval

  918 -- | Given a list of variable names with first-order types, descend
  919 -- into the structure of those types (as needed) and construct
  920 -- corresponding What4 bound variables so we can bind them using a
  921 -- forall quantifier.
  922 -- At the same time construct @SValue@s containing those variables
  923 -- suitable for passing to the term evaluator as substitutions for the
  924 -- given variables.
  925 -- The length of the @SValue@ list returned will match the list of the
  926 -- input variable list, but the list of What4 @BoundVar@s might not.
  927 --
  928 -- This procedure is capable of handling most first-order types,
  929 -- except that Array types must have base types as index and result
  930 -- types rather than more general first-order types. (TODO? should we
  931 -- actually restrict the @FirstOrderType@ in the same way?)
  932 boundFOTs :: forall sym.
  933   IsSymExprBuilder sym =>
  934   sym ->
  935   [(VarName, FirstOrderType)] ->
  936   IO ([SValue sym], [Some (BoundVar sym)])
  937 boundFOTs sym vars =
  938   do (svals,(bndvars,_)) <- runStateT (mapM (uncurry handleVar) vars) ([], 0)
  939      return (svals, bndvars)
  940 
  941  where
  942    freshBnd :: VarName -> W.BaseTypeRepr tp -> StateT ([Some (BoundVar sym)],Integer) IO (SymExpr sym tp)
  943    freshBnd vn tpr =
  944      do (vs,n) <- get
  945         let nm = Text.unpack (vnName vn) ++ "." ++ show n
  946         bvar <- lift (W.freshBoundVar sym (W.safeSymbol nm) tpr)
  947         put (Some bvar : vs, n+1)
  948         return (W.varExpr sym bvar)
  949 
  950    handleVar :: VarName -> FirstOrderType -> StateT ([Some (BoundVar sym)], Integer) IO (SValue sym)
  951    handleVar x fot =
  952      case fot of
  953        FOTBit -> VBool <$> freshBnd x BaseBoolRepr
  954        FOTInt -> VInt  <$> freshBnd x BaseIntegerRepr
  955        FOTIntMod m -> VIntMod m <$> freshBnd x BaseIntegerRepr
  956        FOTRational ->
  957          do numer <- freshBnd x BaseIntegerRepr
  958             -- TODO(#2433): Assert that the denominator is non-zero.
  959             denom <- freshBnd x BaseIntegerRepr
  960             pure $ VRational numer denom
  961 
  962        FOTVec n FOTBit ->
  963          case somePosNat n of
  964            Nothing -> -- n == 0
  965              return (VWord ZBV)
  966            Just (Some (PosNat nr)) ->
  967              VWord . DBV <$> freshBnd x (BaseBVRepr nr)
  968 
  969        FOTVec n tp -> -- NB, not Bit
  970          do vs  <- V.replicateM (fromIntegral n) (handleVar x tp)
  971             vs' <- traverse (return . ready) vs
  972             return (VVector vs')
  973 
  974        FOTRec tm ->
  975          do vs  <- traverse (handleVar x) tm
  976             vs' <- traverse (return . ready) vs
  977             return (vRecord vs')
  978 
  979        FOTTuple ts ->
  980          do vs  <- traverse (handleVar x) ts
  981             vs' <- traverse (return . ready) vs
  982             return (vTuple vs')
  983 
  984        FOTArray idx res
  985          | Just (Some idx_repr) <- fotToBaseType idx
  986          , Just (Some res_repr) <- fotToBaseType res
  987 
  988          -> VArray . SArray <$> freshBnd x (BaseArrayRepr (Ctx.Empty Ctx.:> idx_repr) res_repr)
  989 
  990          | otherwise -> fail ("boundFOTs: cannot handle " ++ show fot)

 1160 -- | Simplify a saw-core term by evaluating it through the saw backend
 1161 -- of what4. The term may have any first-order monomorphic function
 1162 -- type. Return a term with the same type.
 1163 w4EvalTerm ::
 1164   forall n st fs.
 1165   B.ExprBuilder n st fs ->
 1166   SAWCoreState n ->
 1167   SharedContext ->
 1168   Map Ident (SPrim (B.ExprBuilder n st fs)) ->
 1169   Set VarIndex ->
 1170   Term ->
 1171   IO Term
 1172 w4EvalTerm sym st sc ps unintSet t =
 1173   do modmap <- scGetModuleMap sc
 1174      ref <- newIORef Map.empty
 1175      let eval = w4EvalBasic sym st sc modmap ps mempty ref unintSet
 1176      ty <- eval =<< scTypeOf sc t
 1177      -- evaluate term to an SValue
 1178      val <- eval t
 1179      tytval <- toTValue ty
 1180      -- convert SValue back into a Term
 1181      rebuildTerm sym st sc tytval val

 1183     toTValue :: Value l -> IO (TValue l)

 1188   B.ExprBuilder n st fs ->
 1189   SAWCoreState n ->
 1190   SharedContext ->
 1191   TValue (What4 (B.ExprBuilder n st fs)) ->
 1192   SValue (B.ExprBuilder n st fs) ->
 1193   IO Term
 1194 rebuildTerm sym st sc tv sv =
 1195   let chokeOn what =
 1196         -- XXX: alas, for the time being it looks like all we have for
 1197         -- printing SValue/Value is a show instance
 1198         fail ("saw-core-what4: rebuildTerm: cannot handle " ++ what ++ ": " ++
 1199               show sv)

 1201   case sv of
 1202     VFun {} ->
 1203       chokeOn "lambdas (VFun)"

 1261 -- | Simplify a saw-core term by evaluating it through the saw backend
 1262 -- of what4.
 1263 w4EvalAny ::
 1264   forall n st fs.
 1265   B.ExprBuilder n st fs ->
 1266   SAWCoreState n ->
 1267   SharedContext ->
 1268   Map Ident (SPrim (B.ExprBuilder n st fs)) ->
 1269   Set VarIndex ->
 1270   Term ->
 1271   IO ([String],
 1272       [SValue (B.ExprBuilder n st fs)],
 1273       [Maybe (Labeler (B.ExprBuilder n st fs))],
 1274       SValue (B.ExprBuilder n st fs))
 1275 w4EvalAny sym st sc ps unintSet t =
 1276   do modmap <- scGetModuleMap sc
 1277      ref <- newIORef Map.empty
 1278      let eval = w4EvalBasic sym st sc modmap ps mempty ref unintSet
 1279      ty <- eval =<< scTypeOf sc t
 1280 
 1281      -- get the names of the arguments to the function
 1282      let lamNames = map (Text.unpack . vnName . fst) (fst (R.asLambdaList t))
 1283      let varNames = [ "var" ++ show (i :: Integer) | i <- [0 ..] ]
 1284      let argNames = zipWith (++) varNames (map ("_" ++) lamNames ++ repeat "")
 1285 
 1286      -- and their types
 1287      argTs <- argTypes ty
 1288 
 1289      -- construct symbolic expressions for the variables
 1290      vars' <-
 1291        flip evalStateT 0 $
 1292        sequence (zipWith (newVarsForType sym ref) argTs argNames)
 1293 
 1294      -- symbolically evaluate
 1295      bval <- eval t
 1296 
 1297      -- apply and existentially quantify
 1298      let (bvs, vars) = unzip vars'
 1299      let vars'' = fmap ready vars
 1300      bval' <- applyAll bval vars''
 1301 
 1302      return (argNames, vars, bvs, bval')

 1308   SharedContext ->

 1322   forall n st fs.
 1323   B.ExprBuilder n st fs ->
 1324   SAWCoreState n ->
 1325   SharedContext ->
 1326   ModuleMap ->
 1327   Map Ident (SPrim (B.ExprBuilder n st fs)) {- ^ additional primitives -} ->

 1329   IORef (SymFnCache (B.ExprBuilder n st fs)) {- ^ cache for uninterpreted function symbols -} ->

 1331   Term {- ^ term to simulate -} ->
 1332   IO (SValue (B.ExprBuilder n st fs))
 1333 w4EvalBasic sym st sc m addlPrims varCons ref unintSet t =
 1334   do let variable tf (VarName ix nm) ty
 1335            | Just v <- IntMap.lookup ix varCons = pure v
 1336            | otherwise =
 1337            do trm <- scTermF sc tf

 1339                             Variable {} -> True

 1343                  (mkUnintApp (Text.unpack nm ++ "_" ++ show ix)) ty

 1346            | Set.member (nameIndex nm) unintSet =
 1347              let vn = VarName (nameIndex nm) (toShortName (nameInfo nm))

 1349            | otherwise                          = Nothing
