   54 -- | Internal wrapper for 'TypeRep' that we provide an 'MapF.OrdF'
   55 --   instance for.
   56 newtype TypeRep' a = TypeRep' (TypeRep a)
   57   deriving (TestEquality)
   58 
   59 instance MapF.OrdF TypeRep' where
   60   compareF (TypeRep' tr1) (TypeRep' tr2) =
   61     case testEquality tr1 tr2 of
   62       Just Refl -> MapF.EQF
   63       Nothing -> case compare (SomeTypeRep tr1) (SomeTypeRep tr2) of
   64         LT -> MapF.LTF
   65         GT -> MapF.GTF
   66         EQ -> error "inconsistent TypeRep equality"
   67 
   68 
   69 withRep :: forall a b. TypeRep' a -> (Typeable a => b) -> b
   70 withRep (TypeRep' tr) f = withTypeable tr f
   71 
   72 -- | A store of arbitrary 'Typeable' data, indexed by type.
   73 newtype TypedStore f = TypedStore (MapF TypeRep' f)
   74 
   75 -- | An empty 'TypedStore'.
   76 empty :: TypedStore f
   77 empty = TypedStore MapF.empty
   78 
   79 -- | Lookup the value of type 'f a' from the store.
   80 lookup :: forall a f. Typeable a => TypedStore f -> Maybe (f a)
   81 lookup (TypedStore tm) = MapF.lookup rep tm
   82   where
   83     rep = TypeRep' @a TypeRep
   84 
   85 -- | Alter the data of type 'f a' in the store, inserting or
   86 --   deleting based on the result of 'g'.
   87 alter :: forall a f.
   88   Typeable a =>
   89   (Maybe (f a) -> Maybe (f a)) ->
   90   TypedStore f ->
   91   TypedStore f
   92 alter g ts = runIdentity $ alterM (\ma -> Identity (g ma)) ts
   93 
   94 -- | Alter the data of type 'f a' in the store, inserting or deleting
   95 -- based on the result of the monadic action 'g'.
   96 alterM :: forall a f m.
   97   Typeable a =>
   98   Monad m =>
   99   (Maybe (f a) -> m (Maybe (f a))) ->
  100   TypedStore f ->
  101   m (TypedStore f)
  102 alterM g (TypedStore tm) = 
  103   (TypedStore . MapF.updatedValue) <$> tm'
  104   where
  105     rep = TypeRep' @a TypeRep
  106 
  107     go :: f a -> m (MapF.UpdateRequest (f a))
  108     go a = g (Just a) >>= \case
  109       Nothing -> pure $ MapF.Delete
  110       Just a' -> pure $ MapF.Set a'
  111 
  112     tm' = MapF.updateAtKey rep (g Nothing) go tm
  113 
  114 -- | Insert a value of type 'f a' in the store, overwriting existing
  115 --   data.
  116 insert :: forall a f. Typeable a => f a -> TypedStore f -> TypedStore f
  117 insert a = alter (\_ -> Just a)
  118 
  119 -- | Create a 'TypedStore' with a single entry of type 'f a'.
  120 singleton :: forall a f. (Typeable f, Typeable a) => f a -> TypedStore f
  121 singleton a = TypedStore (MapF.singleton rep a)
  122   where
  123     rep = TypeRep' @a TypeRep
  124 
  125 size :: TypedStore f -> Int
  126 size (TypedStore ts) = MapF.size ts
  127 
  128 -- | Delete the value of type 'f a' from the store, if present.
  129 delete :: forall a f proxy. Typeable a => proxy a -> TypedStore f -> TypedStore f
  130 delete _ tm = alter @a (\_ -> Nothing) tm
  131 
  132 map :: forall f g.
  133   (forall a. (Typeable a) => f a -> g a) -> 
  134   TypedStore f -> 
  135   TypedStore g
  136 map f (TypedStore ts) = TypedStore $ 
  137   MapF.mapWithKey (\rep x -> withRep rep $ f x) ts
  138 
  139 traverse :: forall f g m.
  140   Applicative m => 
  141   (forall a. (Typeable a) => f a -> m (g a)) -> 
  142   TypedStore f -> 
  143   m (TypedStore g)
  144 traverse f (TypedStore ts) = TypedStore <$>
  145   MapF.traverseWithKey (\rep x -> withRep rep $ f x) ts
  146 
  147 traverse_ :: forall f m.
  148   Applicative m => 
  149   (forall a. (Typeable a) => f a -> m ()) -> 
  150   TypedStore f -> 
  151   m ()
  152 traverse_ f (TypedStore ts) =
  153   MapF.traverseWithKey_ (\rep x -> withRep rep $ f x) ts
  154 
  155 merge :: forall f g h.
  156   -- keys in both
  157   (forall a. (Typeable a) => f a -> g a -> Maybe (h a)) ->
  158   -- keys only in left
  159   (forall a. (Typeable a) => f a -> Maybe (h a)) ->
  160   -- keys only in right
  161   (forall a. (Typeable a) => g a -> Maybe (h a)) ->
  162   TypedStore f ->
  163   TypedStore g ->
  164   TypedStore h
  165 merge f g1 g2 (TypedStore ts1) (TypedStore ts2) =
  166   TypedStore $ MapF.mergeWithKey 
  167     (\rep x y -> withRep rep $ f x y)
  168     (MapF.mapMaybeWithKey (\rep x -> withRep rep $ g1 x))
  169     (MapF.mapMaybeWithKey (\rep x -> withRep rep $ g2 x))
  170     ts1 ts2
  171 
  172 mergeM :: forall f g h m.
  173   Applicative m =>
  174   -- keys in both
  175   (forall a. (Typeable a) => f a -> g a -> m (Maybe (h a))) ->
  176   -- keys only in left
  177   (forall a. (Typeable a) => f a -> m (Maybe (h a))) ->
  178   -- keys only in right
  179   (forall a. (Typeable a) => g a -> m (Maybe (h a))) ->
  180   TypedStore f ->
  181   TypedStore g ->
  182   m (TypedStore h)
  183 mergeM f g1 g2 (TypedStore ts1) (TypedStore ts2) =
  184   TypedStore <$> MapF.mergeWithKeyM 
  185     (\rep x y -> withRep rep $ f x y)
  186     (MapF.traverseMaybeWithKey (\rep x -> withRep rep $ g1 x))
  187     (MapF.traverseMaybeWithKey (\rep x -> withRep rep $ g2 x))
  188     ts1 ts2
  189 
  190 -- | Combine two 'TypedStore's, using the given function to merge
  191 --   entries of the same type.
  192 union :: forall f.
  193   (forall a. (Typeable a) => f a -> f a -> Maybe (f a)) ->
  194   TypedStore f ->
  195   TypedStore f ->
  196   TypedStore f
  197 union f (TypedStore ts1) (TypedStore ts2) = TypedStore $
  198   MapF.mergeWithKey (\rep x y -> withRep rep $ f x y) id id ts1 ts2
  199 
  200 -- | Intersect two 'TypedStore's, using the given function to merge
  201 --   entries of the same type, and discarding types which only are
  202 --   only present in one store.
  203 intersection :: forall f.
  204   (forall a. (Typeable a) => f a -> f a -> Maybe (f a)) ->
  205   TypedStore f ->
  206   TypedStore f ->
  207   TypedStore f
  208 intersection f (TypedStore ts1) (TypedStore ts2) = TypedStore $
  209   MapF.mergeWithKey (\rep x y -> withRep rep $ f x y) 
  210     (const MapF.empty) (const MapF.empty) ts1 ts2
  211 
  212 toList :: forall f b.
  213   (forall a. (Typeable a) => f a -> b) -> TypedStore f -> [b]
  214 toList f (TypedStore ts) = 
  215   List.map (\(MapF.Pair rep x) -> withRep rep $ f x) $ MapF.toList ts
  216 
  217 lookupIO :: forall a. Typeable a => TypedStore IORef -> IO (Maybe a)
  218 lookupIO ts =
  219   case lookup @a ts of
  220     Just aref -> Just <$> readIORef aref
  221     Nothing -> return Nothing
  222 
  223 -- | Modify the contents of a 'TypedStore' of IORefs. Returns 'Nothing'
  224 --   if the 'TypedStore' itself is unchanged.
  225 --   NOTE: When an entry is deleted, the contents of the IORef are
  226 --   unmodified, the reference is simply removed in the resulting
  227 --   'TypedStore'.
  228 alterIO :: forall a.
  229   Typeable a => 
  230   (Maybe a -> Maybe a) -> 
  231   TypedStore IORef -> 
  232   IO (Maybe (TypedStore IORef))
  233 alterIO f ts = do
  234   case lookup @a ts of
  235     Just aref -> atomicModifyIORef' aref $ \a -> 
  236       case f (Just a) of
  237         Just a' -> (a', Nothing)
  238         Nothing -> (a, Just $ delete @a aref ts)
  239     Nothing -> case f Nothing of
  240       Just a -> do
  241         aref <- newIORef a
  242         return $ Just $ insert @a aref ts
  243       Nothing -> return Nothing
