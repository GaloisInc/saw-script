   31 type SBool = SVal
   32 type SWord = SVal
   33 type SInteger = SVal
   34 
   35 fromBitsLE :: [SBool] -> SWord
   36 fromBitsLE bs = foldl' f (literalSWord 0 0) bs
   37   where f w b = svJoin (svToWord1 b) w
   38 
   39 literalSWord :: Int -> Integer -> SWord
   40 literalSWord w i = svInteger (KBounded False w) i
   41 
   42 literalSInteger :: Integer -> SWord
   43 literalSInteger i = svInteger KUnbounded i
   44 
   45 forallSWord :: String -> Int -> Symbolic SWord
   46 forallSWord nm w = symbolicEnv >>= liftIO . svMkSymVar (NonQueryVar (Just ALL)) (KBounded False w) (Just nm)
   47 
   48 forallSWord_ :: Int -> Symbolic SWord
   49 forallSWord_ w = symbolicEnv >>= liftIO . svMkSymVar (NonQueryVar (Just ALL)) (KBounded False w) Nothing
   50 
   51 existsSWord :: String -> Int -> Symbolic SWord
   52 existsSWord nm w = symbolicEnv >>= liftIO . svMkSymVar (NonQueryVar (Just EX)) (KBounded False w) (Just nm)
   53 
   54 existsSWord_ :: Int -> Symbolic SWord
   55 existsSWord_ w = symbolicEnv >>= liftIO . svMkSymVar (NonQueryVar (Just EX)) (KBounded False w) Nothing
