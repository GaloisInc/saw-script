   29 type AssumptionReason = (MS.ConditionMetadata, String)
   30 
   31 -- | Checks whether the given list of assumptions contains a contradiction, and
   32 -- if so, computes and displays a minimal set of contradictory assumptions.
   33 checkAssumptionsForContradictions ::
   34   Show (MS.MethodId ext) =>
   35   Common.Sym ->
   36   CrucibleMethodSpecIR ext ->
   37   ProofScript () ->
   38   [Crucible.LabeledPred Term AssumptionReason] ->
   39   TopLevel ()
   40 checkAssumptionsForContradictions sym methodSpec tactic assumptions =
   41   whenM
   42     (assumptionsContainContradiction sym methodSpec tactic assumptions)
   43     (computeMinimalContradictingCore sym methodSpec tactic assumptions)
   44 
   45 -- | Checks for contradictions within the given list of assumptions, by asking
   46 -- the solver about whether their conjunction entails falsehood.
   47 assumptionsContainContradiction ::
   48   Show (MS.MethodId ext) =>
   49   Common.Sym ->
   50   CrucibleMethodSpecIR ext ->
   51   ProofScript () ->
   52   [Crucible.LabeledPred Term AssumptionReason] ->
   53   TopLevel Bool
   54 assumptionsContainContradiction sym methodSpec tactic assumptions = 
   55   do
   56      st <- io $ Common.sawCoreState sym
   57      let sc  = saw_ctx st
   58      let ploc = methodSpec^.MS.csLoc
   59      (goal',pgl) <- io $
   60       do
   61          -- conjunction of all assumptions
   62          assume <- scAndList sc (toListOf (folded . Crucible.labeledPred) assumptions)
   63          -- implies falsehood
   64          goal  <- scImplies sc assume =<< toSC sym st (W4.falsePred sym)
   65          goal' <- boolToProp sc [] goal
   66          return $ (goal',
   67                   ProofGoal
   68                   { goalNum  = 0
   69                   , goalType = "vacuousness check"
   70                   , goalName = show (methodSpec^.MS.csMethod)
   71                   , goalLoc  = show (W4.plSourceLoc ploc) ++ " in " ++ show (W4.plFunction ploc)
   72                   , goalDesc = "vacuousness check"
   73                   , goalSequent = propToSequent goal'
   74                   , goalTags = mempty
   75                   })
   76      res <- runProofScript tactic goal' pgl Nothing
   77               "vacuousness check" False False
   78      case res of
   79        ValidProof _ _     -> return True
   80        InvalidProof _ _ _ -> return False
   81        UnfinishedProof _  ->
   82          -- TODO? is this the right behavior?
   83          do printOutLnTop Warn "Could not determine if preconditions are vacuous"
   84             return True
   85 
   86 -- | Given a list of assumptions, computes and displays a smallest subset of
   87 -- them that are contradictory among each themselves.  This is **not**
   88 -- implemented efficiently.
   89 computeMinimalContradictingCore ::
   90   Show (MS.MethodId ext) =>
   91   Common.Sym ->
   92   CrucibleMethodSpecIR ext ->
   93   ProofScript () ->
   94   [Crucible.LabeledPred Term AssumptionReason] ->
   95   TopLevel ()
   96 computeMinimalContradictingCore sym methodSpec tactic assumes =
   97   do
   98      printOutLnTop Warn "Contradiction detected! Computing minimal core of contradictory assumptions:"
   99      -- test subsets of assumptions of increasing sizes until we find a
  100      -- contradictory one
  101      let cores = sortBy (compare `on` length) (subsequences assumes)
  102      findM (assumptionsContainContradiction sym methodSpec tactic) cores >>= \case
  103       Nothing -> 
  104         printOutLnTop Warn "No minimal core: the assumptions did not contains a contradiction."
  105       Just core ->
  106         forM_ core $ \assume ->
  107           case assume^.Crucible.labeledPredMsg of
  108             (loc, reason) -> printOutLnTop Warn (show loc ++ ": " ++ reason)
  109      printOutLnTop Warn "Because of the contradiction, the following proofs may be vacuous."
