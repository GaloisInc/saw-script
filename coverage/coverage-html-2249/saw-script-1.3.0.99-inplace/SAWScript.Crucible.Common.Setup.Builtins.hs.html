   40 crucible_precond ::
   41   W4.ProgramLoc ->
   42   TypedTerm ->
   43   CrucibleSetup ext ()
   44 crucible_precond loc p = do
   45   st <- get
   46   tags <- view croTags
   47   let md = MS.ConditionMetadata
   48            { MS.conditionLoc = loc
   49            , MS.conditionTags = tags
   50            , MS.conditionType = "specification assertion"
   51            , MS.conditionContext = ""
   52            }
   53   when (st ^. csPrePost == MS.PostState) $
   54     fail "attempt to use `crucible_precond` in post state"
   55   addCondition (MS.SetupCond_Pred md p)
   56

   74 crucible_return ::
   75   MS.SetupValue ext ->
   76   CrucibleSetup ext ()
   77 crucible_return retval = do
   78   ret <- use (csMethodSpec . MS.csRetValue)
   79   case ret of
   80     Just _ -> fail "crucible_return: duplicate return value specification"
   81     Nothing -> csMethodSpec . MS.csRetValue .= Just retval
   82 
   83 crucible_execute_func ::
   84   [MS.SetupValue ext] ->
   85   CrucibleSetup ext ()
   86 crucible_execute_func args = do
   87   tps <- use (csMethodSpec . MS.csArgs)
   88   csPrePost .= MS.PostState
   89   csMethodSpec . MS.csArgBindings .= Map.fromList [ (i, (t,a))
   90                                                   | i <- [0..]
   91                                                   | a <- args
   92                                                   | t <- tps
   93                                                   ]
   94 
   95 crucible_equal ::
   96   W4.ProgramLoc ->
   97   MS.SetupValue ext ->
   98   MS.SetupValue ext ->
   99   CrucibleSetup ext ()
  100 crucible_equal loc val1 val2 = do
  101   tags <- view croTags
  102   let md = MS.ConditionMetadata
  103            { MS.conditionLoc = loc
  104            , MS.conditionTags = tags
  105            , MS.conditionType = "equality specification"
  106            , MS.conditionContext = ""
  107            }
  108   addCondition (MS.SetupCond_Equal md val1 val2)

  113 -- | When invoking a points-to command, against what should SAW check the type
  114 -- of the RHS value?
  115 data CheckPointsToType ty
  116   = CheckAgainstPointerType
  117     -- ^ Check the type of the RHS value against the type that the LHS points to.
  118     --   Used by commands such as @llvm_{conditional_}points_to@ and
  119     --   @mir_points_to@.
  120   | CheckAgainstCastedType ty
  121     -- ^ Check the type of the RHS value against the provided @ty@, which
  122     --   the LHS pointer is casted to.
  123     --   This is currently used by @llvm_{conditional_}points_to_at_type@ only.
  124   deriving (Functor, Foldable, Traversable)
