   21 -- | Names are strings (for which we'll use Text)
   22 type Name = Text
   23 
   24 -- | Arguments are names with (Cryptol) types
   25 data Arg = Arg { argName :: Name
   26                , argType :: Type }
   27                deriving (Eq, Ord)
   28 
   29 instance Show Arg where
   30    showsPrec d (Arg n t) =
   31       showParen (d > app_prec) $
   32          showString (unpack n)
   33          . showString " : "
   34          . showString (pretty t)
   35       where app_prec = 10
   36 
   37 -- | Declarations have a name, a return type, and a list of arguments
   38 data Decl = Decl { declName :: Name
   39                  , declType :: Type
   40                  , declArgs :: [Arg] }
   41                  deriving (Eq, Ord)
   42 
   43 instance Show Decl where
   44    showsPrec d (Decl n t as) =
   45       showParen (d > app_prec) $
   46          showString (unpack n)
   47          . showString " : "
   48          . showString "("
   49          . showString (intercalate ", " (map show as))
   50          . showString ")"
   51          . showString " -> "
   52          . showString (pretty t)
   53       where
   54          app_prec = 10
   55 
   56 -- | The signature of a declaration is the multiset of its argument types crossed with its return type
   57 data Sig = Sig { sigArgs_ :: Map Type Int
   58                , sigOut_  :: Type }
   59                deriving (Eq, Ord)
   60 
   61 sigArgs :: Sig -> Map Type Int
   62 sigArgs = sigArgs_
   63 
   64 sigOut :: Sig -> Type
   65 sigOut  = sigOut_
   66 
   67 -- | Compute the signature of a declaration
   68 declSig :: Decl -> Sig
   69 declSig (Decl _ t as) =
   70   flip Sig t $
   71     foldr (uncurry (Map.insertWith (const succ))) Map.empty $
   72       map (argType &&& const 1) as
