   64 ------------------------------------------------------------
   65 -- REPL state and state monad
   66 
   67 -- | The state maintained by the REPL.
   68 --
   69 -- rContinue starts True and remains true until someone runs @:q@,
   70 -- at which point the main REPL loop exits.
   71 --
   72 -- rIsBatch is True if we're executing the REPL from a file in
   73 -- batch mode (@saw -B@).
   74 --
   75 -- rTopLevelRO and rTopLevelRW are the SAWScript interpreter's
   76 -- TopLevel monad state.
   77 --
   78 -- rProofState is the SAWScript interpreter's ProofScript monad state;
   79 -- it should be `Nothing` when we're running an ordinary TopLevel REPL
   80 -- and `Just` when we're running a ProofScript REPL.
   81 --
   82 -- We carry the state around separately like this (rather than
   83 -- embedding the interpreter's monads) so we can run in either
   84 -- context. This could be done with a monad typeclass instead, though
   85 -- likely that would end up being a lot more complicated, and as we've
   86 -- seen in the interpreter core, it's very easy to mess up code like
   87 -- that so it executes some bits in the wrong monad, which will then
   88 -- generally typecheck and only fail at runtime if you happen to try
   89 -- the right things. So it's probably better this way, although I'd
   90 -- prefer that the internal partitioning of the interpreter state
   91 -- wasn't exposed.
   92 -- 
   93 data REPLState = REPLState
   94   { rContinue   :: Bool
   95   , rIsBatch    :: Bool
   96   , rTopLevelRO :: TopLevelRO
   97   , rTopLevelRW :: TopLevelRW
   98   , rProofState :: Maybe ProofState
   99   }
  100 
  101 -- | Create an initial, empty environment.
  102 initREPL ::
  103     Bool -> Options -> TopLevelShellHook -> ProofScriptShellHook ->
  104     IO REPLState
  105 initREPL isBatch opts tlhook pshook =
  106   do (ro, rw) <- buildTopLevelEnv opts [] tlhook pshook
  107      return REPLState
  108        { rContinue   = True
  109        , rIsBatch    = isBatch
  110        , rTopLevelRO = ro
  111        , rTopLevelRW = rw
  112        , rProofState  = Nothing
  113        }
  114 
  115 -- | Create an environment from an existing interpreter state.
  116 --
  117 -- FUTURE: it might be nice to be able to read subshell input from a
  118 -- file in batch mode. Or from the same file that's feeding the parent
  119 -- shell, though that's probably difficult. For now, assume any
  120 -- subshells are intended to be interactive.
  121 resumeREPL :: TopLevelRO -> TopLevelRW -> Maybe ProofState -> REPLState
  122 resumeREPL ro rw mpst =
  123     REPLState {
  124         rContinue   = True,
  125         rIsBatch    = False,
  126         rTopLevelRO = ro,
  127         rTopLevelRW = rw,
  128         rProofState = mpst
  129     }

  131 -- | REPL monad context.
  132 newtype REPL a = REPL { unREPL :: StateT REPLState IO a }
  133   deriving (
  134       Applicative, Functor, Monad, MonadThrow, MonadCatch, MonadMask, MonadFail
  135   )
  136 
  137 deriving instance MonadState REPLState REPL
  138 
  139 -- | Run a REPL action on a REPL state.
  140 runREPL :: REPL a -> REPLState -> IO (a, REPLState)
  141 runREPL m st = do
  142     runStateT (unREPL m) st
  143 
  144 -- | Run an IO action in a REPL context.
  145 instance MonadIO REPL where
  146   liftIO m = REPL (liftIO m)
  147 
  148 -- | Run a TopLevel action in a REPL context.
  149 liftTopLevel :: TopLevel a -> REPL a
  150 liftTopLevel m = do
  151     ro <- getTopLevelRO
  152     rw <- getTopLevelRW
  153     (result, rw') <- liftIO $ runTopLevel m ro rw
  154     modify (\st -> st { rTopLevelRW = rw' })
  155     return result
  156 
  157 -- | Run a ProofScript action in a REPL context.
  158 liftProofScript :: ProofScript a -> REPL a
  159 liftProofScript m = do
  160     mpst <- gets rProofState
  161     let pst = case mpst of
  162           Nothing -> panic "liftProofScript" ["Not in ProofScript mode"]
  163           Just ps -> ps
  164     (result, pst') <-
  165         liftTopLevel $ runStateT (runExceptT (unProofScript m)) pst
  166     modify (\st -> st { rProofState = Just pst' })
  167     liftTopLevel $ case result of
  168        Left (stats, cex) ->
  169          do ppOpts <- rwPPOpts <$> get
  170             fail (showsProofResult ppOpts (InvalidProof stats cex pst') "")
  171        Right x -> return x

  195 captureError :: TopLevelCheckpoint -> Maybe String -> REPL ()
  196 captureError chk mmsg = do
  197     case mmsg of
  198         Nothing -> pure ()
  199         Just msg -> do
  200             liftIO $ putStrLn ""
  201             liftIO $ putStrLn msg
  202     liftTopLevel $ restoreCheckpoint chk

  204 -- | Inspect and handle or rethrow exceptions.
  205 --
  206 --   Note: this would be cleaner if it used `catches` with multiple
  207 --   cases; however, the docs say nothing about what `catches` does
  208 --   with overlapping cases, or if it actually works currently,
  209 --   whether that's a supported part of the interface.
  210 --
  211 --   FUTURE: with enough cleanup in lower-level code we could make
  212 --   this not need a blanket case and then `catches` becomes viable.
  213 --
  214 --   This logic controls which exceptions the REPL continues executing
  215 --   for. Anything we rethrow here is fatal and will cause the REPL to
  216 --   exit.
  217 --
  218 handleExceptions :: TopLevelCheckpoint -> X.SomeException -> REPL ()
  219 handleExceptions chk e
  220 
  221     -- Cons.Fatal: catch and continue. We don't need to print the
  222     -- exception; the point of `Fatal` vs. anything else is that
  223     -- we already printed the error messages.
  224   | Just (_ :: Cons.Fatal) <- X.fromException e =
  225         captureError chk Nothing
  226 
  227     -- IO exceptions: catch and continue on UserError (this includes
  228     -- `fail` calls), but treat anything else as fatal.
  229   | Just (e' :: X.IOException) <- X.fromException e =
  230         if isUserError e' then
  231             captureError chk $ Just $ ioeGetErrorString e'
  232         else
  233             throwM e'
  234 
  235     -- Async exceptions: treat as fatal.
  236   | Just (e' :: X.AsyncException) <- X.fromException e =
  237         throwM e'
  238 
  239     -- ExitCode arises when someone makes an explicit call to exit
  240     -- (`exitSuccess` or `exitWith`); assume they meant it.
  241   | Just (e' :: ExitCode) <- X.fromException e =
  242         throwM e'
  243 
  244     -- Exceptions generated by panic: treat as fatal.
  245   | Just (e' :: PanicSupport.PanicException) <- X.fromException e =
  246         throwM e'
  247 
  248     -- Trap anything and everything else. XXX: this is way too broad.
  249   | otherwise =
  250         captureError chk $ Just $ X.displayException e
  251

  254 -- | Catch the exceptions we expect to catch while running a REPL
  255 --   action.
  256 exceptionProtect :: REPL () -> REPL ()
  257 exceptionProtect cmd = do
  258     chk <- liftTopLevel makeCheckpoint
  259     cmd `catch` handleExceptions chk
