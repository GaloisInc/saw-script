   43 -- | Optional rewrites to do when resolving a term
   44 data ResolveRewrite = ResolveRewrite {
   45   rrBasicSS :: Maybe (Simpset TheoremNonce),
   46   -- ^ Rewrite terms using these rewrites
   47 
   48   rrWhat4Eval :: Bool
   49   -- ^ Also simplify terms using What4 evaluation
   50 }
   51 
   52 -- | Don't do any rewriting
   53 noResolveRewrite :: ResolveRewrite
   54 noResolveRewrite = ResolveRewrite { rrBasicSS = Nothing, rrWhat4Eval = False }
   55 
   56 -- Convert a term to a What4 expression, trying to simplify it in the process.
   57 -- Currently this expects the terms to be either boolean or bit-vectors, as
   58 -- that's all we use it for.   It should be fairly straightforward to generalize
   59 -- if need be.
   60 resolveTerm ::
   61   Sym {- ^ Backend state -} ->
   62   Set VarIndex {- ^ Keep these opaque -} ->
   63   W4.BaseTypeRepr t  {- ^ Type of term -} ->
   64   ResolveRewrite {- ^ Optional rewrites to do on the term -} ->
   65   Term {- ^ Term to process -} ->
   66   IO (Crucible.RegValue Sym (Crucible.BaseToType t))
   67 resolveTerm sym unint bt rr tm =
   68   do
   69     st       <- sawCoreState sym
   70     checkType st
   71     tm'      <- basicRewrite st tm
   72     case () of
   73       _ | isConstFoldTerm unint tm' ->
   74           do -- Evaluate as constant
   75             modmap <- scGetModuleMap (saw_ctx st)
   76             let v = Concrete.evalSharedTerm modmap mempty mempty tm'
   77             case bt of
   78               W4.BaseBoolRepr -> pure (W4.backendPred sym (Concrete.toBool v))
   79               W4.BaseBVRepr w -> W4.bvLit sym w (BV.mkBV w (Prim.unsigned (Concrete.toWord v)))
   80               _ -> fail "resolveTerm: expected `Bool` or bit-vector"
   81 
   82         | rrWhat4Eval rr ->
   83           do -- Try to use rewrites to simplify the term
   84             let sc = saw_ctx st
   85             cryptol_ss <- Cryptol.mkCryptolSimpset @TheoremNonce sc
   86             tm''       <- snd <$> rewriteSharedTerm sc cryptol_ss tm'
   87             tm'''      <- basicRewrite st tm''
   88             if all isPreludeName (Map.elems (getConstantSet tm''')) then
   89               do
   90                 (_, _, _, p) <- w4EvalAny sym st sc mempty unint tm'''
   91                 case valueToSymExpr p of
   92                   Just (Some y)
   93                     | Just Refl <- testEquality bt ty -> pure y
   94                     | otherwise -> typeError (show ty)
   95                     where ty = W4.exprType y
   96                   _ -> fail ("resolveTerm: unexpected w4Eval result " ++ show p)
   97               else
   98                 bindSAWTerm sym st bt tm'''
   99 
  100           -- Just bind the term
  101         | otherwise -> bindSAWTerm sym st bt tm'
  102 
  103   where
  104   basicRewrite st =
  105     case rrBasicSS rr of
  106       Nothing -> pure
  107       Just ss -> \t -> snd <$> rewriteSharedTerm (saw_ctx st) ss t
  108 
  109   isPreludeName nm =
  110     case nm of
  111       ModuleIdentifier ident -> identModule ident == preludeName
  112       _ -> False
  113 
  114   checkType st =
  115     do
  116       sc <- ttType <$> mkTypedTerm (saw_ctx st) tm
  117       case ttIsMono sc of
  118           Just ty
  119             | tIsBit ty, W4.BaseBoolRepr <- bt -> pure ()
  120             | Just (n,el) <- (tIsSeq ty)
  121             , tIsBit el, Just i <- tIsNum n, W4.BaseBVRepr w <- bt
  122             , intValue w == i -> pure ()
  123             | otherwise -> typeError (show (PP.pp ty)) :: IO ()
  124           Nothing -> typeError (show (ppTypedTermType sc))
  125 
  126   typeError :: String -> IO a
  127   typeError t = fail $ unlines [
  128     "Expected type: " ++ (
  129       case bt of
  130         W4.BaseBoolRepr  -> "Bit"
  131         W4.BaseBVRepr w  -> "[" ++ show w ++ "]"
  132         _             -> show bt),
  133     "Actual type:   " ++ t
  134     ]

  136 -- 'resolveTerm' specialized to booleans.
  137 resolveBoolTerm' :: Sym -> Set VarIndex -> ResolveRewrite -> Term -> IO (W4.Pred Sym)
  138 resolveBoolTerm' sym unint = resolveTerm sym unint W4.BaseBoolRepr
  139 
  140 -- 'resolveTerm' specialized to booleans, without rewriting.
  141 resolveBoolTerm :: Sym -> Set VarIndex -> Term -> IO (W4.Pred Sym)
  142 resolveBoolTerm sym unint = resolveBoolTerm' sym unint noResolveRewrite
  143 
  144 -- 'resolveTerm' specialized to bit-vectors.
  145 resolveBitvectorTerm' ::
  146   (1 W4.<= w) => Sym -> Set VarIndex -> W4.NatRepr w -> ResolveRewrite -> Term -> IO (W4.SymBV Sym w)
  147 resolveBitvectorTerm' sym unint w = resolveTerm sym unint (W4.BaseBVRepr w)
  148                   
  149 -- 'resolveTerm' specialized to bit-vectors, without rewriting.
  150 resolveBitvectorTerm :: 
  151   (1 W4.<= w) => Sym -> Set VarIndex -> W4.NatRepr w -> Term -> IO (W4.SymBV Sym w)
  152 resolveBitvectorTerm sym unint w = resolveBitvectorTerm' sym unint w noResolveRewrite
