  114 -- REPL Environment.
  115 data Refs = Refs
  116   { eContinue   :: IORef Bool
  117   , eIsBatch    :: Bool
  118   , eTopLevelRO :: TopLevelRO
  119   , environment :: IORef TopLevelRW
  120   , proofState  :: Maybe (IORef ProofState)
  121   }
  122 
  123 -- | Initial, empty environment.
  124 defaultRefs :: Bool -> Options -> IO Refs
  125 defaultRefs isBatch opts =
  126   do (_biContext, ro, rw) <- buildTopLevelEnv (AIGProxy AIG.compactProxy) opts []
  127      contRef <- newIORef True
  128      rwRef <- newIORef rw
  129      return Refs
  130        { eContinue   = contRef
  131        , eIsBatch    = isBatch
  132        , eTopLevelRO = ro
  133        , environment = rwRef
  134        , proofState  = Nothing
  135        }
  136 
  137 
  138 mkTitle :: Refs -> String
  139 mkTitle _refs = "sawscript"

  144 -- | REPL_ context with InputT handling.
  145 newtype REPL a = REPL { unREPL :: Refs -> IO a }
  146 
  147 -- | Run a REPL action with a fresh environment.
  148 runREPL :: Bool -> Options -> REPL a -> IO a
  149 runREPL isBatch opts m =
  150   do refs <- defaultRefs isBatch opts
  151      unREPL m refs
  152 
  153 subshell :: REPL () -> TopLevel ()
  154 subshell (REPL m) = TopLevel_ $
  155   do ro <- ask
  156      rw <- get
  157      rw' <- liftIO $
  158        do contRef <- newIORef True
  159           rwRef <- newIORef rw
  160           let refs = Refs
  161                      { eContinue = contRef
  162                      , eIsBatch  = False
  163                      , eTopLevelRO = ro
  164                      , environment = rwRef
  165                      , proofState  = Nothing
  166                      }
  167           m refs
  168           readIORef rwRef
  169      put rw'
  170 
  171 proof_subshell :: REPL () -> ProofScript ()
  172 proof_subshell (REPL m) =
  173   ProofScript $ ExceptT $ StateT $ \proofSt ->
  174   do ro <- ask
  175      rw <- get
  176      (rw', outProofSt) <- liftIO $
  177        do contRef <- newIORef True
  178           rwRef <- newIORef rw
  179           proofRef <- newIORef proofSt
  180           let refs = Refs
  181                      { eContinue = contRef
  182                      , eIsBatch  = False
  183                      , eTopLevelRO = ro
  184                      , environment = rwRef
  185                      , proofState  = Just proofRef
  186                      }
  187           m refs
  188           (,) <$> readIORef rwRef <*> readIORef proofRef
  189      put rw'
  190      return (Right (), outProofSt)
  191 
  192 instance Functor REPL where
  193   {-# INLINE fmap #-}
  194   fmap f m = REPL (\ ref -> fmap f (unREPL m ref))
  195 
  196 instance Monad REPL where
  197   {-# INLINE return #-}
  198   return = pure
  199 
  200   {-# INLINE (>>=) #-}
  201   m >>= f = REPL $ \ref -> do
  202     x <- unREPL m ref
  203     unREPL (f x) ref
  204 
  205 instance Fail.MonadFail REPL where
  206   {-# INLINE fail #-}
  207   fail msg = REPL (\_ -> fail msg)
  208 
  209 instance Applicative REPL where
  210   {-# INLINE pure #-}
  211   pure x = REPL (\_ -> pure x)
  212   {-# INLINE (<*>) #-}
  213   (<*>) = ap

  217 -- | REPL exceptions.
  218 data REPLException
  219   = ParseError ParseError
  220   | FileNotFound FilePath
  221   | DirectoryNotFound FilePath
  222   | NoPatError [Error]
  223   | NoIncludeError [IncludeError]
  224   | EvalError EvalError
  225   | ModuleSystemError M.ModuleError
  226   | EvalPolyError T.Schema
  227   | TypeNotTestable T.Type
  228     deriving (Show,Typeable)
  229 
  230 instance X.Exception REPLException
  231 
  232 instance PP REPLException where
  233   ppPrec _ re = case re of
  234     ParseError e         -> ppError e
  235     FileNotFound path    -> sep [ text "File"
  236                                 , text ("`" ++ path ++ "'")
  237                                 , text"not found"
  238                                 ]
  239     DirectoryNotFound path -> sep [ text "Directory"
  240                                   , text ("`" ++ path ++ "'")
  241                                   , text"not found or not a directory"
  242                                   ]
  243     NoPatError es        -> vcat (map pp es)
  244     NoIncludeError es    -> vcat (map ppIncludeError es)
  245     ModuleSystemError me -> pp me
  246     EvalError e          -> pp e
  247     EvalPolyError s      -> text "Cannot evaluate polymorphic value."
  248                          $$ text "Type:" <+> pp s
  249     TypeNotTestable t    -> text "The expression is not of a testable type."
  250                          $$ text "Type:" <+> pp t
  251 
  252 -- | Raise an exception.
  253 raise :: REPLException -> REPL a
  254 raise exn = io (X.throwIO exn)
  255 
  256 -- | Handle any exception type in 'REPL' actions.
  257 catchEx :: X.Exception e => REPL a -> (e -> REPL a) -> REPL a
  258 catchEx m k = REPL (\ ref -> unREPL m ref `X.catch` \ e -> unREPL (k e) ref)
  259 
  260 -- | Handle 'REPLException' exceptions in 'REPL' actions.
  261 catch :: REPL a -> (REPLException -> REPL a) -> REPL a
  262 catch = catchEx
  263 
  264 -- | Similar to 'catch' above, but catches generic IO exceptions from 'fail'.
  265 catchFail :: REPL a -> (String -> REPL a) -> REPL a
  266 catchFail m k = REPL (\ ref -> X.catchJust sel (unREPL m ref) (\s -> unREPL (k s) ref))
  267   where
  268     sel :: X.IOException -> Maybe String
  269     sel e | isUserError e = Just (ioeGetErrorString e)
  270           | otherwise     = Nothing
  271 
  272 -- | Handle any other exception (except that we ignore async exceptions and exitWith)
  273 catchOther :: REPL a -> (X.SomeException -> REPL a) -> REPL a
  274 catchOther m k = REPL (\ref -> X.catchJust flt (unREPL m ref) (\s -> unREPL (k s) ref))
  275  where
  276   flt e
  277     | Just (_ :: X.AsyncException) <- X.fromException e = Nothing
  278     | Just (_ :: ExitCode)       <- X.fromException e = Nothing
  279     | otherwise = Just e
  280 
  281 rethrowEvalError :: IO a -> IO a
  282 rethrowEvalError m = run `X.catch` rethrow
  283   where
  284   run = do
  285     a <- m
  286     return $! a
  287 
  288   rethrow :: EvalErrorEx -> IO a
  289   rethrow (EvalErrorEx _ exn) = X.throwIO (EvalError exn)

  292 exceptionProtect :: REPL () -> REPL ()
  293 exceptionProtect cmd =
  294       do chk <- io . makeCheckpoint =<< getEnvironment
  295          cmd `catch`      (handlerPP chk)
  296              `catchFail`  (handlerFail chk)
  297              `catchOther` (handlerPrint chk)
  298 
  299     where
  300     handlerPP chk re =
  301       do io (putStrLn "" >> print (pp re))
  302          void $ liftTopLevel (restoreCheckpoint chk)
  303          return ()
  304     handlerPrint chk e =
  305       do io (putStrLn "" >> putStrLn (X.displayException e))
  306          void $ liftTopLevel (restoreCheckpoint chk)
  307 
  308     handlerFail chk s =
  309       do io (putStrLn "" >> putStrLn s)
  310          void $ liftTopLevel (restoreCheckpoint chk)
  311 
  312 liftTopLevel :: TopLevel a -> REPL a
  313 liftTopLevel m =
  314   do ro  <- getTopLevelRO
  315      ref <- getEnvironmentRef
  316      io $ do rw <- readIORef ref
  317              (a, rw') <- runTopLevel m ro rw
  318              writeIORef ref rw'
  319              return a
  320 
  321 liftProofScript :: ProofScript a -> IORef ProofState -> REPL a
  322 liftProofScript m ref =
  323   liftTopLevel $
  324   do st <- liftIO $ readIORef ref
  325      (res, st') <- runStateT (runExceptT (unProofScript m)) st
  326      liftIO $ writeIORef ref st'
  327      case res of
  328        Left (stats, cex) ->
  329          do ppOpts <- rwPPOpts <$> get
  330             fail (showsProofResult ppOpts (InvalidProof stats cex st') "")
  331        Right x -> return x

  347 -- | Construct the prompt for the current environment.
  348 getPrompt :: REPL String
  349 getPrompt =
  350   do batch <- REPL (return . eIsBatch)
  351      if batch then return ""
  352      else
  353        getProofStateRef >>= \case
  354          Nothing -> return "sawscript> "
  355          Just psr ->
  356            do ps <- io (readIORef psr)
  357               return ("proof ("++show (length (psGoals ps))++")> ")

  359 shouldContinue :: REPL Bool

  363 stop = modifyRef eContinue (const False)

  365 unlessBatch :: REPL () -> REPL ()
  366 unlessBatch body =
  367   do batch <- REPL (return . eIsBatch)

  376 getVars :: REPL (Map.Map T.Name M.IfaceDecl)
  377 getVars  = do
  378   me <- getModuleEnv
  379   let decls = getAllIfaceDecls me
  380   let vars1 = M.ifDecls decls
  381   extras <- getExtraTypes
  382   let vars2 = Map.mapWithKey (\q s -> M.IfaceDecl { M.ifDeclName = q
  383                                                   , M.ifDeclSig = s
  384                                                   , M.ifDeclIsPrim = False
  385                                                   , M.ifDeclPragmas = []
  386                                                   , M.ifDeclInfix = False
  387                                                   , M.ifDeclFixity = Nothing
  388                                                   , M.ifDeclDoc = Nothing
  389                                                   }) extras
  390   return (Map.union vars1 vars2)
  391

  493 -- | Get visible variable names for Haskeline completion.
  494 getSAWScriptValueNames :: REPL [String]
  495 getSAWScriptValueNames = do
  496   env <- getEnvironment
  497   let avail = rwPrimsAvail env
  498       visible (_, lc, _, _, _, _) = Set.member lc avail
  499   let rnames = Map.keys $ Map.filter visible $ rwValueInfo env
  500   return (map Text.unpack rnames)
