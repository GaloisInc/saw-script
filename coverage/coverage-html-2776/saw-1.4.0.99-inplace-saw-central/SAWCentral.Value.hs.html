  307 -- | A type to hold the chain of references to a value that arise
  308 --   during execution. This appears in the monadic value types
  309 --   (VTopLevel, VProofScript, etc.) in order to trace let-bindings of
  310 --   monadic objects and treat them as part of the call chain.
  311 --
  312 --   It starts empty for builtins in the builtins table, and gets an
  313 --   entry consed onto it each time a value is either read out of the
  314 --   environment (for plain values) or created by function
  315 --   application.
  316 --
  317 --   Each reference is a position and a name (the position of the
  318 --   reference and the name that was used to refer to the value) and
  319 --   is used as a call site when handling stack traces.
  320 --
  321 --   For example,
  322 --      1: let x = print_stack;
  323 --      2: let y = x;
  324 --      3: y;
  325 --
  326 --   will flow as follows:
  327 --      (a) We load print_stack from the global environment and add an
  328 --          entry containing the reference position (line 1, columns
  329 --          9-20) and the name "print_stack". The resulting value is
  330 --          saved as "x".
  331 --      (b) We load "x" from the environment, and add an entry with
  332 --          the reference position (line 2, columns 9-10) and the name
  333 --          "x". The value already contains the entry from (a), so now
  334 --          there are two, and the one from "x" is on the front of the
  335 --          list.
  336 --      (c) We load "y" from the environment, and add a similar entry
  337 --          with "y".
  338 --      (d) Then we bind y to execute it, and that loads the sequence
  339 --          y -> x -> print_stack onto the call stack.
  340 --      (e) The resulting trace printed by print_stack will look like:
  341 --             (builtin) in print_stack
  342 --             foo.saw:1:9-20 in x
  343 --             foo.saw:2:9-10 in y
  344 --             foo.saw:3:1-2 at top level
  345 --
  346 type RefChain = [(SS.Pos, SS.Name)]
  347 
  348 -- | The SAWScript interpreter's base value type.
  349 --
  350 --   The interpreter executes from AST elements (expressions,
  351 --   statements) to values. Unfortunately, for the time being, values
  352 --   are a (considerable) superset of what the AST can represent, so
  353 --   values can't be injected back into unexecuted AST elements.
  354 --   FUTURE: this should be fixed, but not until it can be done
  355 --   without tying a gigantic knot in the module dependencies.
  356 --
  357 --   Because monadic values are irreducible until executed (they're
  358 --   effectively lambdas) but we also want to be able to know what
  359 --   they are when we execute them (for stack traces and error
  360 --   reporting) we carry around a bunch of metadata with each one.
  361 --
  362 --   Values that can have monadic type are:
  363 --      - the compound monadic values, that reduce to plain ones
  364 --        by executing in the interpreter:
  365 --           VReturn
  366 --           VDo
  367 --           VBindOnce
  368 --      - the plain monadic values, which contain Haskell-level
  369 --        monadic actions:
  370 --           VTopLevel
  371 --           VProofScript
  372 --           VLLVMCrucibleSetup
  373 --           VJVMSetup
  374 --           VMIRSetup
  375 --
  376 --   Furthermore, VLambda and VBuiltin can produce monadic values
  377 --   after application of an argument, and need to carry certain
  378 --   metadata of their own to populate their results.
  379 --
  380 --   In general, monadic values carry the following metadata until
  381 --   executed:
  382 --      - Position of last reference
  383 --      - a RefChain (q.v.)
  384 --
  385 --   Function-class values that might produce monadic values when an
  386 --   argument is applied (VLambda and VBuiltin) carry the following
  387 --   metadata:
  388 --      - a name
  389 --      - an argument list
  390 --
  391 --   The position of last reference is kept for VReturn and the plain
  392 --   monadic values. For plain monadic values, the position is loaded
  393 --   into the interpreter's last position state before the
  394 --   Haskell-level monadic action is executed.
  395 --
  396 --   For VReturn, it propagates to the result. (Whether this is
  397 --   necessary, useful, or even desirable isn't entirely clear -- it
  398 --   has no effect if you return a scalar, obviously, and if you
  399 --   return out a monadic value, that should already have its own
  400 --   reference position and it's not clear that updating it with the
  401 --   return's position is helpful. FUTURE: investigate.)
  402 --
  403 --   VBindOnce contains a position; however, it's the position of the
  404 --   bind operation (comparable to the position in a StmtBind) and not
  405 --   relevant to this discussion.
  406 --
  407 --   The RefChain is kept for plain monadic values, and for VReturn
  408 --   for propagation purposes. (Again, that may or may not be the best
  409 --   way. FUTURE.) VDo (and VBindOnce) also have a RefChain, which
  410 --   propagates to the resultant plain monadic value when the do-block
  411 --   is executed in the interpreter. The do-block RefChain tracks the
  412 --   names the do-block gets passed around under before it gets
  413 --   evaluated. (And then the chain in the resulting value could
  414 --   theoretically accumulate more entries as it gets passed around,
  415 --   except that there's no way to do that: we always execute the
  416 --   resultant Haskell-level action immediately after interpreting any
  417 --   do-block.)
  418 --
  419 --   We could in principle also carry a RefChain in VLambda so if
  420 --   you pass a lambda around it gets traced the same way a monadic
  421 --   value does. There's not as much call for it, but it might be
  422 --   useful. FUTURE.
  423 --
  424 --   VLambda and VBuiltin carry their names. When the last argument is
  425 --   applied, and the resulting value is monadic, the name is used to
  426 --   seed its RefChain. The RefChain then tracks the whole sequence of
  427 --   names used to refer to that value, in order; this is why plain
  428 --   monadic values don't specifically carry their "own" names. Note
  429 --   that for VLambda the name is optional, because lambdas don't
  430 --   necessarily have names. There's no such thing as an anonymous
  431 --   builtin though.
  432 --
  433 --   VBuiltin also carries a list of values (it is actually a
  434 --   snoc-list manifested as a `Seq`) that are the arguments already
  435 --   applied to the builtin. This is purely for printing in stack
  436 --   traces. FUTURE: VLambda should have this as well, but it doesn't
  437 --   make sense to make such a list separate from the variable
  438 --   environment handling, and that needs a through mucking-out of its
  439 --   own first.
  440 --
  441 --   The flow for the position of last reference is as follows:
  442 --      (a) Builtins in the global environment are initialized with
  443 --          `atRestPos`. This should never be seen by a user because
  444 --          stuff in the environment should not be seen without being
  445 --          referenced somehow.
  446 --      (b) When a variable is looked up in the environment, the
  447 --          value returned has its last reference position updated
  448 --          with the position of the variable name.
  449 --      (c) Before an argument is applied to a function (or lambda)
  450 --          the _argument's_ last reference position is updated with
  451 --          the position of the argument expression. For arguments
  452 --          to builtins that are monadic value callbacks, this
  453 --          causes the last reference position to be the point where
  454 --          the value was passed to the builtin. Alas, I no longer
  455 --          remember what special case this was needed to handle,
  456 --          and it's possible it's no longer necessary.
  457 --      (d) Somewhat similarly, the first argument of VBindOnce is
  458 --          updated before it is executed, using the position in
  459 --          the VBindOnce.
  460 --      (e) The same goes for statements in do-blocks, using the
  461 --          position of the BindStmt, or for the last expression in a
  462 --          do-block, the position of the expression.
  463 --      (f) When the Haskell-level monadic action in a plan monadic
  464 --          value is extracted with fromValue, we use the position to
  465 --          call setPosition to update the interpreter's last
  466 --          execution position.
  467 --      (g) The idea is that anytime we leave the interpreter to
  468 --          execute a builtin, there's a setPosition call that sets
  469 --          the last execution position to something that makes sense
  470 --          to report to the user if the builtin errors out.
  471 --      (h) FUTURE: consider clearing a value's position (setting it
  472 --          to `atRestPos`) when added to the environment, so the
  473 --          position of last reference for everything in the
  474 --          environment is the same.
  475 --
  476 --   The flow for names is as follows:
  477 --      (a) toValue is given a name to insert. This allows the
  478 --          typeclass-based construction process for builtins to
  479 --          propagate the name to every VBuiltin. (The name itself is
  480 --          provided from the builtins table.)
  481 --      (b) The names of lambdas are applied by the parser in
  482 --          `buildFunction`. Every let-binding that's written with a
  483 --          function head is converted to a chain of lambdas there,
  484 --          and that conversion inserts the name from the let-binding.
  485 --          Other lambdas are by definition anonymous and get Nothing.
  486 --      (c) As mentioned above, when the last argument is applied to a
  487 --          function, the name is used to seed the RefChain of the
  488 --          resulting value, if there is one. At this time we'll also
  489 --          be generating a stack trace frame while the function
  490 --          executes, using that name.
  491 --
  492 --   See above for notes on the flow for RefChains.
  493 --
  494 --   In principle the position of last reference should be the same as
  495 --   the position at the top of the RefChain, and we should not get
  496 --   empty RefChains (same as we should never see the default position
  497 --   that appears in the builtins table) so it may be sufficient to
  498 --   use the RefChain to call setPosition. However, not all the places
  499 --   that update the position of last reference add to the RefChain
  500 --   (they were tried and found to produce unwanted or incorrect stack
  501 --   traces) so this may not be true and/or those position updates may
  502 --   not be needed, all of which should probably be investigated
  503 --   further in the FUTURE.
  504 --
  505 --   Also note that because the bits of metadata aren't quite the same
  506 --   in each case and don't update in exactly the same places, it
  507 --   currently doesn't work very well to try to wrap them together in
  508 --   their own type. If things become more regular, it might be worth
  509 --   revisiting that proposition.
  510 --
  511 --   In addition to all of the above, VLambda and VDo carry an
  512 --   environment (the interpreter's name -> value environment) which
  513 --   closes in the naming environment they're run against. This
  514 --   environment is collected when the corresponding lambda or do
  515 --   expression is evaluated (basically, where it appears in the input
  516 --   source), and arguments get added to it as they're applied. This
  517 --   produces the expected lexical scoping behavior.
  518 --
  519 --   VBuiltin, even though it's effectively also a lambda, does _not_
  520 --   carry an environment. Closing in a copy of the partly-built
  521 --   default environment that exists when the VBuiltin values are
  522 --   constructed, or even the whole default environment, wouldn't
  523 --   serve any purpose.
  524 
  525 --   Furthermore, VBuiltin is used to bind in Haskell functions, which
  526 --   don't themselves run in SAWScript and don't need or use the
  527 --   naming environment; they just need their arguments. The exception
  528 --   is the subshell and proof_subshell builtins, which intentionally
  529 --   inherit the SAWScript environment they're invoked in. Any
  530 --   otherwise pointless environment capturing we indulged in for
  531 --   uniformity's sake would break that behavior.

  800 -- | Type for the ordinary interpreter environment.

  802 --   There's one environment that maps variable names to values, and

  805 --   appropriate times.

  808 --   they don't get closed in; references to rebindable variables

  823 -- | Enter a scope.
  824 pushScope :: TopLevel ()
  825 pushScope = do
  826     Environ varenv tyenv cryenv <- gets rwEnviron
  827     let varenv' = ScopedMap.push varenv
  828         tyenv' = ScopedMap.push tyenv
  829         cryenv' = cryptolPush cryenv
  830     modifyTopLevelRW (\rw -> rw { rwEnviron = Environ varenv' tyenv' cryenv' })

  832 -- | Leave a scope. This will panic if you try to leave the last scope;
  833 --   pushes and pops should be matched.
  834 popScope  :: TopLevel ()
  835 popScope = do
  836     Environ varenv tyenv cryenv <- gets rwEnviron
  837     let varenv' = ScopedMap.pop varenv
  838         tyenv' = ScopedMap.pop tyenv
  839         cryenv' = cryptolPop cryenv
  840     modifyTopLevelRW (\rw -> rw { rwEnviron = Environ varenv' tyenv' cryenv' })

 1067 io f = (TopLevel_ (liftIO f))
 1068        `catches`
 1069        [ Handler (\(ex :: X86Unsupported) -> handleX86Unsupported ex)
 1070        , Handler (\(ex :: X86Error) -> handleX86Error ex)

 1075     rethrow :: X.Exception ex => ex -> TopLevel a
 1076     rethrow ex =
 1077       do stk <- getStackTrace
 1078          -- XXX: assumes the exception came from inside a builtin somewhere,

 1080          -- true and this logic should be reworked when we're doing the error

 1122 -- | Add multiple stack trace frames. Takes a list of call site and
 1123 --   function name pairs.
 1124 --
 1125 --   The topmost entry in the argument list goes onto the stack first
 1126 --   and becomes the deepest entry and thus the caller of the rest.
 1127 --
 1128 --   This is backwards from what one might expect if the list were
 1129 --   itself a call stack. However, the use case we have produces
 1130 --   entries in that order naturally (it is a chain of references to /
 1131 --   assignments of monadic values, with the most recent and therefore
 1132 --   first/deepest when run at the top of the list), so I'm not
 1133 --   inclined to reverse the list both here and in the caller to match
 1134 --   a rather vague expectation.

 1138   mapM_ (\(pos, name) -> pushTraceFrame pos name) frames

 1306 extendEnv :: SS.Pos -> SS.Name -> SS.Rebindable -> SS.Schema -> Maybe [Text] -> Value -> TopLevel ()
 1307 extendEnv pos name rb ty doc v = do
 1308      sc <- gets rwSharedContext
 1309      let ident = T.mkIdent name
 1310          modname = T.packModName [name]
 1311 
 1312      -- Update the SAWScript environment.
 1313      Environ varenv tyenv cryenvs <- gets rwEnviron
 1314      rbenv <- gets rwRebindables
 1315      let (varenv', rbenv') = case rb of
 1316            SS.ReadOnlyVar ->
 1317              -- If we replace a rebindable variable at the top level with a
 1318              -- readonly one, the readonly version in varenv will hide it
 1319              -- ever after. We ought to remove it from rbenv; however, it's
 1320              -- not easy to know here whether we're at the top level or not.
 1321              -- FUTURE: maybe this will become easier in the long run.
 1322              let ve' = ScopedMap.insert name (pos, SS.Current, ty, v, doc) varenv in
 1323              (ve', rbenv)
 1324            SS.RebindableVar ->
 1325              -- The typechecker restricts this to happen only at the
 1326              -- top level and only if any existing variable is already
 1327              -- rebindable, so we don't have to update varenv.
 1328              let re' = Map.insert name (pos, ty, v) rbenv in
 1329              (varenv, re')
 1330 
 1331      -- Mirror the value into the Cryptol environment if appropriate.
 1332      let CryptolEnvStack ce ces = cryenvs
 1333      ce' <-
 1334        case v of
 1335          VTerm t ->
 1336            pure $ CEnv.bindTypedTerm (ident, t) ce
 1337          VType s ->
 1338            pure $ CEnv.bindType (ident, s) ce
 1339          VInteger n ->
 1340            pure $ CEnv.bindInteger (ident, n) ce
 1341          VCryptolModule m ->
 1342            pure $ CEnv.bindExtCryptolModule (modname, m) ce
 1343          VString s ->
 1344            do tt <- io $ typedTermOfString sc (Text.unpack s)
 1345               pure $ CEnv.bindTypedTerm (ident, tt) ce
 1346          VBool b ->
 1347            do tt <- io $ typedTermOfBool sc b
 1348               pure $ CEnv.bindTypedTerm (ident, tt) ce
 1349          _ ->
 1350            pure ce
 1351      let cryenvs' = CryptolEnvStack ce' ces
 1352 
 1353      -- Drop the new bits into place.
 1354      modify (\rw -> rw {
 1355          rwEnviron = Environ varenv' tyenv cryenvs',
 1356          rwRebindables = rbenv'
 1357      })

 1359 extendEnvMulti :: [(SS.Pos, SS.Name, SS.Rebindable, SS.Schema, Maybe [Text], Environ -> Value)] -> TopLevel ()
 1360 extendEnvMulti bindings = do
 1361 
 1362      -- Update the SAWScript environment.
 1363      Environ varenv tyenv cryenv <- gets rwEnviron
 1364 
 1365      -- Insert all the bindings at once, and feed the final resulting
 1366      -- interpreter environment into each value. This circular
 1367      -- definition only works because of laziness and it's only legal
 1368      -- when the pieces are in a single let-block.
 1369      --
 1370      -- Only allow lambda values because that's the only use case
 1371      -- (functions in mutually recursive "rec" sets) and it lets us
 1372      -- ignore the Cryptol environment.
 1373      --
 1374      -- Be sure to insert v' (rather than v) so the panic check is
 1375      -- actually performed.
 1376      let insert (pos, name, rb, ty, doc, fv) tmpenv =
 1377              let v = fv environ'
 1378                  v' = case v of
 1379                      VLambda{} -> v
 1380                      _ ->
 1381                          panic "extendEnvMulti" [
 1382                              "Non-lambda value: " <> Text.pack (show v),
 1383                              "Source position: " <> Text.pack (show pos)
 1384                          ]
 1385                  v'' = case rb of
 1386                      SS.ReadOnlyVar -> v'
 1387                      SS.RebindableVar ->
 1388                          -- "rec" declarations can't be rebindable
 1389                          panic "extendEnvMulti" [
 1390                              "Rebindable variable: " <> name,
 1391                              "Source position: " <> Text.pack (show pos)
 1392                          ]
 1393              in
 1394              ScopedMap.insert name (pos, SS.Current, ty, v'', doc) tmpenv
 1395          varenv' = foldr insert varenv bindings
 1396          environ' = Environ varenv' tyenv cryenv
 1397 
 1398      -- Drop the new bits into place.
 1399      modify (\rw -> rw { rwEnviron = environ' })
