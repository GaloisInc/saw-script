   28 -- | A SAWCore bitvector term along with its width and whether it should be interpreted as signed.
   29 data CellTerm = CellTerm
   30   { cellTermTerm :: SC.Term
   31   , cellTermWidth :: Natural
   32   , cellTermSigned :: Bool
   33   }
   34 
   35 cellTermNat :: SC.SharedContext -> CellTerm -> IO SC.Term
   36 cellTermNat sc (CellTerm { cellTermTerm = t, cellTermWidth = w }) = SC.scBvToNat sc w t
   37 
   38 -- | Apply the appropriate (possibly signed) extension or truncation to make the given bitvector
   39 -- match the given width.
   40 extTrunc :: SC.SharedContext -> Natural -> CellTerm -> IO CellTerm
   41 extTrunc sc width (CellTerm { cellTermTerm = t, cellTermWidth = bvw, cellTermSigned = signed}) =
   42   do wterm <- SC.scNat sc width
   43      bvwterm <- SC.scNat sc bvw
   44      res <- if
   45        | bvw > width ->
   46            do diffterm <- SC.scNat sc $ bvw - width
   47               SC.scBvTrunc sc diffterm wterm t
   48        | width > bvw && bvw == 0 ->
   49            do zero <- SC.scNat sc 0
   50               SC.scBvNat sc wterm zero
   51        | width > bvw && signed ->
   52            do bvwpredterm <- SC.scNat sc $ bvw - 1
   53               diffterm <- SC.scNat sc $ width - bvw
   54               SC.scBvSExt sc diffterm bvwpredterm t
   55        | width > bvw && not signed ->
   56            do diffterm <- SC.scNat sc $ width - bvw
   57               SC.scBvUExt sc diffterm bvwterm t
   58        | otherwise -> pure t
   59      pure $ CellTerm res width signed
   60 
   61 -- | Given two bitvectors, extend the narrower bitvector to match the wider.
   62 extMax :: SC.SharedContext -> CellTerm -> CellTerm -> IO (CellTerm, CellTerm)
   63 extMax sc c1 c2 =
   64   do let
   65        w1 = cellTermWidth c1
   66        w2 = cellTermWidth c2
   67        w = max w1 w2
   68      res1 <- extTrunc sc w c1
   69      res2 <- extTrunc sc w c2
   70      pure (res1, res2)
   71 
   72 liftUnary ::
   73   SC.SharedContext ->
   74   (SC.Term -> SC.Term -> IO SC.Term) -> -- (w : Nat) -> [w] -> [w]
   75   CellTerm -> IO CellTerm
   76 liftUnary sc f c@(CellTerm { cellTermTerm = t }) =
   77   do wt <- SC.scNat sc $ cellTermWidth c
   78      res <- f wt t
   79      pure $ c { cellTermTerm = res }
   80 
   81 liftBinary ::
   82   SC.SharedContext ->
   83   (SC.Term -> SC.Term -> SC.Term -> IO SC.Term) -> -- (w : Nat) -> [w] -> [w] -> [w]
   84   CellTerm -> CellTerm -> IO CellTerm
   85 liftBinary sc f c1@(CellTerm { cellTermTerm = t1 }) (CellTerm { cellTermTerm = t2 }) =
   86   do wt <- SC.scNat sc $ cellTermWidth c1
   87      res <- f wt t1 t2
   88      pure $ c1 { cellTermTerm = res }
   89 
   90 liftBinaryCmp ::
   91   SC.SharedContext ->
   92   (SC.Term -> SC.Term -> SC.Term -> IO SC.Term) -> -- (w : Nat) -> [w] -> [w] -> Bool
   93   CellTerm -> CellTerm -> IO SC.Term
   94 liftBinaryCmp sc f c1@(CellTerm { cellTermTerm = t1 }) (CellTerm { cellTermTerm = t2 }) =
   95   do wt <- SC.scNat sc $ cellTermWidth c1
   96      f wt t1 t2
   97 
   98 -- | Perform a two's complement negation of a bit vector.
   99 bvneg :: SC.SharedContext -> CellTerm -> IO CellTerm
  100 bvneg sc a =
  101   do w <- SC.scNat sc (cellTermWidth a)
  102      res <- SC.scBvNeg sc w (cellTermTerm a)
  103      pure $ a {cellTermTerm = res}
  104 
  105 -- | Given an input and a shift amount, perform a logical left shift.
  106 shl :: SC.SharedContext -> CellTerm -> CellTerm -> IO CellTerm
  107 shl sc a b =
  108   do w <- SC.scNat sc (cellTermWidth a)
  109      nb <- cellTermNat sc b
  110      res <- SC.scBvShl sc w (cellTermTerm a) nb
  111      pure $ a {cellTermTerm = res}
  112 
  113 -- | Given an input and a shift amount, perform a logical right shift.
  114 shr :: SC.SharedContext -> CellTerm -> CellTerm -> IO CellTerm
  115 shr sc a b =
  116   do w <- SC.scNat sc (cellTermWidth a)
  117      nb <- cellTermNat sc b
  118      res <- SC.scBvShr sc w (cellTermTerm a) nb
  119      pure $ a {cellTermTerm = res}
  120 
  121 -- | Given an input and a shift amount, perform an signed right shift
  122 -- if the input is signed, and a logical right shift if not.
  123 sshr :: SC.SharedContext -> CellTerm -> CellTerm -> IO CellTerm
  124 sshr sc a b
  125   | cellTermSigned a =
  126     do wt1 <- SC.scNat sc (cellTermWidth a - 1)
  127        nb <- cellTermNat sc b
  128        res <- SC.scBvSShr sc wt1 (cellTermTerm a) nb
  129        pure $ a {cellTermTerm = res}
  130   | otherwise =
  131     shr sc a b
  132 
  133 -- | Given an input and a shift amount, perform a logical left shift
  134 -- if the shift amount is negative, and a logical right shift if the
  135 -- shift amount is positive or unsigned.
  136 shift :: SC.SharedContext -> CellTerm -> CellTerm -> IO CellTerm
  137 shift sc a b
  138   | cellTermSigned b =
  139     -- If the shift amount is signed, then shift right by b if b >= 0
  140     -- and left by -b otherwise.
  141     do let wa = cellTermWidth a
  142        let ta = cellTermTerm a
  143        let wb = cellTermWidth b
  144        let tb = cellTermTerm b
  145        wt <- SC.scNat sc wa
  146        wbt <- SC.scNat sc wb
  147        zero <- SC.scBvConst sc wb 0
  148        tbn <- SC.scBvToNat sc wb tb
  149        tbneg <- SC.scBvNeg sc wbt tb
  150        tbnegn <- SC.scBvToNat sc wb tbneg
  151        cond <- SC.scBvSGe sc wbt tb zero
  152        tcase <- SC.scBvShr sc wt ta tbn
  153        ecase <- SC.scBvShl sc wt ta tbnegn
  154        ty <- SC.scBitvector sc wa
  155        res <- SC.scIte sc ty cond tcase ecase
  156        pure $ a {cellTermTerm = res}
  157   | otherwise =
  158     -- If the shift amount is unsigned, then unconditionally shift right.
  159     shr sc a b

  161 -- | Translate a single combinational primitive cell into a SAWCore
  162 -- term, given 'CellTerm's for the inputs.
  163 combCellToTerm ::
  164   SC.SharedContext ->
  165   CellTypeCombinational {- ^ Cell type -} ->
  166   Map Text CellTerm {- ^ Mapping of input names to input terms and bit widths -} ->
  167   Natural {- ^ Output width -} ->
  168   IO SC.Term
  169 combCellToTerm sc ctc args ywidth =
  170   case ctc of
  171     CellTypeNot ->
  172       -- If output size is larger, then we must extend before inverting
  173       -- to compute the high bits correctly.
  174       -- If output size is smaller, truncating first is safe.
  175       do a <- extTrunc sc ywidth =<< input "A"
  176          output =<< liftUnary sc (SC.scBvNot sc) a
  177     CellTypePos ->
  178       do res <- input "A"
  179          output res
  180     CellTypeNeg ->
  181       -- If output size is larger, then we must extend before negating
  182       -- to compute the high bits correctly.
  183       -- If output size is smaller, truncating first is safe.
  184       do a <- extTrunc sc ywidth =<< input "A"
  185          output =<< bvneg sc a
  186     CellTypeAnd -> bvBinaryOp $ SC.scBvAnd sc
  187     CellTypeOr -> bvBinaryOp $ SC.scBvOr sc
  188     CellTypeXor -> bvBinaryOp $ SC.scBvXor sc
  189     CellTypeXnor ->
  190       bvBinaryOp $ \w x y ->
  191       do r <- SC.scBvXor sc w x y
  192          SC.scBvNot sc w r
  193     CellTypeReduceAnd ->
  194       do r <- SC.scGlobalDef sc $ SC.mkIdent SC.preludeName "and"
  195          bvReduce True r
  196     CellTypeReduceOr ->
  197       do r <- SC.scGlobalDef sc $ SC.mkIdent SC.preludeName "or"
  198          bvReduce False r
  199     CellTypeReduceXor ->
  200       do r <- SC.scGlobalDef sc $ SC.mkIdent SC.preludeName "xor"
  201          bvReduce False r
  202     CellTypeReduceXnor ->
  203       -- "~(reduceXor bits)" then extend if necessary.
  204       do r <- SC.scGlobalDef sc $ SC.mkIdent SC.preludeName "xor"
  205          bit <- bvReduce' False r
  206          outputBit =<< SC.scNot sc bit
  207     CellTypeReduceBool ->
  208       do r <- SC.scGlobalDef sc $ SC.mkIdent SC.preludeName "or"
  209          bvReduce False r
  210     CellTypeShl ->
  211       -- If output size is larger, then we must extend before shifting
  212       -- to avoid losing high bits.
  213       -- If output size is smaller, truncating first is safe.
  214       do a <- extTrunc sc ywidth =<< input "A"
  215          b <- input "B"
  216          output =<< shl sc a b
  217     CellTypeShr ->
  218       -- If output size is larger, we must extend before shifting to
  219       -- compute high bits correctly.
  220       -- If output size is smaller, we must shift before truncating.
  221       do a <- input "A"
  222          b <- input "B"
  223          let w = max (cellTermWidth a) ywidth
  224          a' <- extTrunc sc w a
  225          output =<< shr sc a' b
  226     CellTypeSshl ->
  227       -- If output size is larger, then we must extend before shifting
  228       -- to avoid losing high bits.
  229       -- If output size is smaller, truncating first is safe.
  230       do a <- extTrunc sc ywidth =<< input "A"
  231          b <- input "B"
  232          output =<< shl sc a b
  233     CellTypeSshr ->
  234       -- If output size is larger, we must extend before shifting to
  235       -- compute high bits correctly.
  236       -- If output size is smaller, we must shift before truncating.
  237       do a <- input "A"
  238          b <- input "B"
  239          let w = max (cellTermWidth a) ywidth
  240          a' <- extTrunc sc w a
  241          output =<< sshr sc a' b
  242     CellTypeShift ->
  243       -- If output size is larger, we must extend before shifting to
  244       -- compute high bits correctly.
  245       -- If output size is smaller, we must shift before truncating.
  246       do a <- input "A"
  247          b <- input "B"
  248          let w = max (cellTermWidth a) ywidth
  249          a' <- extTrunc sc w a
  250          output =<< shift sc a' b
  251     CellTypeShiftx ->
  252       -- $shiftx is like $shift, but shifts in x bits instead of 0s.
  253       -- For SAW, we model x bits as 0 bits, so we translate the two
  254       -- cell types identically.
  255       do a <- input "A"
  256          b <- input "B"
  257          let w = max (cellTermWidth a) ywidth
  258          a' <- extTrunc sc w a
  259          output =<< shift sc a' b
  260     CellTypeLt -> bvBinaryCmp . liftBinaryCmp sc $
  261       if signed then SC.scBvSLt sc else SC.scBvULt sc
  262     CellTypeLe -> bvBinaryCmp . liftBinaryCmp sc $
  263       if signed then SC.scBvSLe sc else SC.scBvULe sc
  264     CellTypeGt -> bvBinaryCmp . liftBinaryCmp sc $
  265       if signed then SC.scBvSGt sc else SC.scBvUGt sc
  266     CellTypeGe -> bvBinaryCmp . liftBinaryCmp sc $
  267       if signed then SC.scBvSGe sc else SC.scBvUGe sc
  268     CellTypeEq -> bvBinaryCmp . liftBinaryCmp sc $ SC.scBvEq sc
  269     CellTypeNe ->
  270       bvBinaryCmp $ liftBinaryCmp sc $ \w x y ->
  271       do r <- SC.scBvEq sc w x y
  272          SC.scNot sc r
  273     CellTypeEqx -> bvBinaryCmp . liftBinaryCmp sc $ SC.scBvEq sc
  274     CellTypeNex ->
  275       bvBinaryCmp $ liftBinaryCmp sc $ \w x y ->
  276       do r <- SC.scBvEq sc w x y
  277          SC.scNot sc r
  278     CellTypeAdd -> bvBinaryOp $ SC.scBvAdd sc
  279     CellTypeSub -> bvBinaryOp $ SC.scBvSub sc
  280     CellTypeMul -> bvBinaryOp $ SC.scBvMul sc
  281     CellTypeDiv ->
  282       if signed
  283       then bvBinarySOp $ SC.scBvSDiv sc
  284       else bvBinaryOp $ SC.scBvUDiv sc
  285     CellTypeMod ->
  286       if signed
  287       then bvBinarySOp $ SC.scBvSRem sc
  288       else bvBinaryOp $ SC.scBvURem sc
  289     -- "$modfloor" -> _
  290     CellTypeLogicNot ->
  291       do CellTerm ta awidth _ <- input "A"
  292          w <- SC.scNat sc awidth
  293          anz <- SC.scBvNonzero sc w ta
  294          res <- SC.scNot sc anz
  295          outputBit res
  296     CellTypeLogicAnd ->
  297       bvBinaryCmp $ liftBinaryCmp sc $ \w x y ->
  298       do xnz <- SC.scBvNonzero sc w x
  299          ynz <- SC.scBvNonzero sc w y
  300          SC.scAnd sc xnz ynz
  301     CellTypeLogicOr ->
  302       bvBinaryCmp $ liftBinaryCmp sc $ \w x y ->
  303       do xnz <- SC.scBvNonzero sc w x
  304          ynz <- SC.scBvNonzero sc w y
  305          SC.scOr sc xnz ynz
  306     CellTypeMux ->
  307       do CellTerm ta awidth _ <- input "A"
  308          tb <- cellTermTerm <$> input "B"
  309          CellTerm ts swidth _ <- input "S"
  310          swidth' <- SC.scNat sc swidth
  311          snz <- SC.scBvNonzero sc swidth' ts
  312          ty <- SC.scBitvector sc ywidth
  313          res <- SC.scIte sc ty snz tb ta
  314          output $ CellTerm res awidth False
  315     CellTypePmux ->
  316       do CellTerm ta awidth _ <- input "A"
  317          CellTerm tb _ _ <- input "B"
  318          CellTerm ts swidth _ <- input "S"
  319          awidth' <- SC.scNat sc awidth
  320          swidth' <- SC.scNat sc swidth
  321          widthBv <- SC.scBitvector sc awidth
  322          bool <- SC.scBoolType sc
  323          splitb <- SC.scSplit sc swidth' awidth' bool tb
  324          scPmux <- SC.scGlobalDef sc $ SC.mkIdent SC.preludeName "pmux"
  325          res <- SC.scApplyAll sc scPmux [swidth', widthBv, ts, splitb, ta]
  326          output $ CellTerm res awidth False
  327     CellTypeBmux ->
  328       do ia <- input "A"
  329          is <- input "S"
  330          let swidth = cellTermWidth is
  331          -- Split input A into chunks
  332          chunks <- SC.scNat sc (2 ^ swidth)
  333          ywidth' <- SC.scNat sc ywidth
  334          bool <- SC.scBoolType sc
  335          splitA <- SC.scSplit sc chunks ywidth' bool (cellTermTerm ia)
  336          -- reverse to put index 0 on the left
  337          outputType <- SC.scBitvector sc ywidth
  338          revA <- SC.scGlobalApply sc "Prelude.reverse" [chunks, outputType, splitA]
  339          -- Select chunk from output
  340          ixWidth <- SC.scNat sc swidth
  341          elt <- SC.scBvAt sc chunks outputType ixWidth revA (cellTermTerm is)
  342          output (CellTerm elt ywidth False)
  343     -- "$demux" -> _
  344     -- "$lut" -> _
  345     -- "$slice" -> _
  346     -- "$concat" -> _
  347     CellTypeBUF ->
  348       do res <- input "A"
  349          output res
  350   where
  351     nm :: Text
  352     nm = ppCellTypeCombinational ctc
  353 
  354     signed :: Bool
  355     signed = all cellTermSigned args
  356 
  357     input :: Text -> IO CellTerm
  358     input inpNm =
  359       case Map.lookup inpNm args of
  360         -- XXX This should not be a panic, as it is possible to trigger this
  361         -- with a malformed input file.
  362         Nothing -> panic "cellToTerm" [nm <> " missing input " <> inpNm]
  363         Just ctrm -> pure ctrm
  364 
  365     -- | Extend or truncate a cell term as needed to fit the output @Y@ port.
  366     output :: CellTerm -> IO SC.Term
  367     output res = cellTermTerm <$> extTrunc sc ywidth res
  368 
  369     outputBit :: SC.Term -> IO SC.Term
  370     outputBit res =
  371       do bool <- SC.scBoolType sc
  372          vres <- SC.scSingle sc bool res
  373          output $ CellTerm vres 1 False

  375     -- Extend inputs to max of input and output widths, compute with
  376     -- width - 1 for signed output.
  377     bvBinaryOp' ::
  378       Bool -> (SC.Term -> SC.Term -> SC.Term -> IO SC.Term) -> IO SC.Term
  379     bvBinaryOp' isSignedOp f =
  380       do ta <- input "A"
  381          tb <- input "B"
  382          let w = maximum [cellTermWidth ta, cellTermWidth tb, ywidth]
  383          ta' <- extTrunc sc w ta
  384          tb' <- extTrunc sc w tb
  385          if w > 0
  386            then do wt <- SC.scNat sc $ if isSignedOp then w - 1 else w
  387                    res <- f wt (cellTermTerm ta') $ cellTermTerm tb'
  388                    output ta' { cellTermTerm = res }
  389            else output ta'
  390     bvBinaryOp :: (SC.Term -> SC.Term -> SC.Term -> IO SC.Term) -> IO SC.Term
  391     bvBinaryOp = bvBinaryOp' False
  392     bvBinarySOp :: (SC.Term -> SC.Term -> SC.Term -> IO SC.Term) -> IO SC.Term
  393     bvBinarySOp = bvBinaryOp' True
  394     -- extend inputs to max input width, output is a single bit extended to the output width
  395     bvBinaryCmp :: (CellTerm -> CellTerm -> IO SC.Term) -> IO SC.Term
  396     bvBinaryCmp f =
  397       do ta <- input "A"
  398          tb <- input "B"
  399          res <- uncurry f =<< extMax sc ta tb
  400          outputBit res
  401     bvReduce' :: Bool -> SC.Term -> IO SC.Term
  402     bvReduce' boolIdentity boolFun =
  403       do CellTerm t twidth _ <- input "A"
  404          w <- SC.scNat sc twidth
  405          boolTy <- SC.scBoolType sc
  406          identity <- SC.scBool sc boolIdentity
  407          scFoldr <- SC.scGlobalDef sc $ SC.mkIdent SC.preludeName "foldr"
  408          SC.scApplyAll sc scFoldr [boolTy, boolTy, w, boolFun, identity, t]
  409     -- | bvReduce', but extend or truncate output as necessary.
  410     bvReduce :: Bool -> SC.Term -> IO SC.Term
  411     bvReduce boolIdentity boolFun = outputBit =<< bvReduce' boolIdentity boolFun
