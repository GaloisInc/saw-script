   37 -- | The direction of a module port.
   38 data Direction
   39   = DirectionInput
   40   | DirectionOutput
   41   | DirectionInout
   42   deriving (Show, Eq, Ord)
   43 
   44 isInput :: Direction -> Bool
   45 isInput d =
   46   case d of
   47     DirectionInput -> True
   48     DirectionOutput -> False
   49     DirectionInout -> True
   50 
   51 isOutput :: Direction -> Bool
   52 isOutput d =
   53   case d of
   54     DirectionInput -> False
   55     DirectionOutput -> True
   56     DirectionInout -> True
   57 
   58 instance Aeson.FromJSON Direction where
   59   parseJSON (Aeson.String "input") = pure DirectionInput
   60   parseJSON (Aeson.String "output") = pure DirectionOutput
   61   parseJSON (Aeson.String "inout") = pure DirectionInout
   62   parseJSON v = fail $ "Failed to parse direction: " <> show v
   63 
   64 -- | The value of a connection
   65 data Bitrep
   66   = BitrepZero -- ^ Constant zero bit
   67   | BitrepOne -- ^ Constant one bit
   68   | BitrepX -- ^ Undefined bit X
   69   | BitrepZ -- ^ Undefined bit Z
   70   | Bitrep Integer -- ^ The signal bit with the given index
   71   deriving (Show, Eq, Ord)
   72 
   73 instance Aeson.FromJSON Bitrep where
   74   parseJSON (Aeson.String "0") = pure BitrepZero
   75   parseJSON (Aeson.String "1") = pure BitrepOne
   76   parseJSON (Aeson.String "x") = pure BitrepX
   77   parseJSON (Aeson.String "z") = pure BitrepZ
   78   parseJSON vn@Aeson.Number{} = Bitrep <$> Aeson.parseJSON vn
   79   parseJSON v = fail $ "Failed to parse bits: " <> show v

   81 -- ^ A module input/output port.
   82 data Port = Port
   83   { _portDirection :: Direction
   84   , _portBits :: [Bitrep] -- ^ Which bit indices within the module are associated with the port
   85   , _portOffset :: Integer -- currently unused
   86   , _portUpto :: Bool -- currently unused
   87   } deriving (Show, Eq, Ord)
   88 
   89 makeLenses ''Port
   90 
   91 instance Aeson.FromJSON Port where
   92   parseJSON = Aeson.withObject "port" $ \o -> do
   93     _portDirection <- o Aeson..: "direction"
   94     _portBits <- o Aeson..: "bits"
   95     _portOffset <- o Aeson..:? "offset" >>= \case
   96       Just off -> pure off
   97       Nothing -> pure 0
   98     _portUpto <- o Aeson..:? "upto" >>= \case
   99       Just (Aeson.Number 1) -> pure True
  100       _ -> pure False
  101     pure Port{..}

  103 -- | Return 'True' iff a given cell type is a primitive type
  104 cellTypeIsPrimitive :: Text -> Bool
  105 cellTypeIsPrimitive cellType =
  106   case Text.uncons cellType of
  107     Just ('$', _) -> True
  108     _ -> False

  110 -- | Mapping from 'Text' to combinational primitive cell types.
  111 textToCellTypeCombinational :: Map Text CellTypeCombinational
  112 textToCellTypeCombinational =
  113   Map.fromList [ (ppCellTypeCombinational t, t) | t <- [minBound .. maxBound] ]
  114 
  115 -- | Mapping from 'Text' to primitive cell types.
  116 textToPrimitiveCellType :: Map Text CellType
  117 textToPrimitiveCellType =
  118   Map.fromList [ (ppCellType ct, ct) | ct <- [CellTypeDff, CellTypeFf] ] <>
  119   fmap CellTypeCombinational textToCellTypeCombinational
  120 
  121 -- | Mapping from primitive cell types to textual representation
  122 primitiveCellTypeToText :: Map CellType Text
  123 primitiveCellTypeToText =
  124   Map.fromList [(y, x) | (x, y) <- Map.toList textToPrimitiveCellType]

  126 -- | All supported primitive combinational cell types.
  127 data CellTypeCombinational
  128   = CellTypeNot
  129   | CellTypePos
  130   | CellTypeNeg
  131   | CellTypeAnd
  132   | CellTypeOr
  133   | CellTypeXor
  134   | CellTypeXnor
  135   | CellTypeReduceAnd
  136   | CellTypeReduceOr
  137   | CellTypeReduceXor
  138   | CellTypeReduceXnor
  139   | CellTypeReduceBool
  140   | CellTypeShl
  141   | CellTypeShr
  142   | CellTypeSshl
  143   | CellTypeSshr
  144   | CellTypeShift
  145   | CellTypeShiftx
  146   | CellTypeLt
  147   | CellTypeLe
  148   | CellTypeGt
  149   | CellTypeGe
  150   | CellTypeEq
  151   | CellTypeNe
  152   | CellTypeEqx
  153   | CellTypeNex
  154   | CellTypeAdd
  155   | CellTypeSub
  156   | CellTypeMul
  157   | CellTypeDiv
  158   | CellTypeMod
  159   | CellTypeLogicNot
  160   | CellTypeLogicAnd
  161   | CellTypeLogicOr
  162   | CellTypeMux
  163   | CellTypePmux
  164   | CellTypeBmux
  165   | CellTypeBUF
  166   deriving (Eq, Ord, Enum, Bounded)

  168 -- | All supported cell types.
  169 -- All types are primitives except for 'CellTypeUserType' which
  170 -- represents user-defined submodules.
  171 -- Invariants: 'CellTypeUnsupportedPrimitive' should only be applied
  172 -- to names satisfying 'cellTypeIsPrimitive', and 'CellTypeUserType'
  173 -- should only be applied to names *not* satisfying
  174 -- 'cellTypeIsPrimitive'.
  175 data CellType
  176   = CellTypeCombinational CellTypeCombinational
  177   | CellTypeDff
  178   | CellTypeFf
  179   | CellTypeUnsupportedPrimitive Text
  180   | CellTypeUserType Text
  181   deriving (Eq, Ord)
  182 
  183 instance Aeson.FromJSON CellType where
  184   parseJSON (Aeson.String s) =
  185     case s of
  186       "$adff"        -> fail $ show $ YosysErrorUnsupportedFF "$adff"
  187       "$sdff"        -> fail $ show $ YosysErrorUnsupportedFF "$sdff"
  188       "$aldff"       -> fail $ show $ YosysErrorUnsupportedFF "$aldff"
  189       "$dffsr"       -> fail $ show $ YosysErrorUnsupportedFF "$dffsr"
  190       "$dffe"        -> fail $ show $ YosysErrorUnsupportedFF "$dffe"
  191       "$adffe"       -> fail $ show $ YosysErrorUnsupportedFF "$adffe"
  192       "$sdffe"       -> fail $ show $ YosysErrorUnsupportedFF "$sdffe"
  193       "$sdffce"      -> fail $ show $ YosysErrorUnsupportedFF "$sdffce"
  194       "$aldffe"      -> fail $ show $ YosysErrorUnsupportedFF "$aldffe"
  195       "$dffsre"      -> fail $ show $ YosysErrorUnsupportedFF "$dffsre"
  196       _ | cellTypeIsPrimitive s ->
  197           case Map.lookup s textToPrimitiveCellType of
  198             Just cellType -> pure cellType
  199             -- XXX We should probably log a warning when generating CellTypeUnsupportedPrimitive,
  200             -- we can't do that here however.
  201             Nothing -> pure $ CellTypeUnsupportedPrimitive s
  202         | otherwise -> pure $ CellTypeUserType s
  203   parseJSON v = fail $ "Failed to parse cell type: " <> show v
  204 
  205 ppCellTypeCombinational :: CellTypeCombinational -> Text
  206 ppCellTypeCombinational ctc =
  207   case ctc of
  208     CellTypeNot        -> "$not"
  209     CellTypePos        -> "$pos"
  210     CellTypeNeg        -> "$neg"
  211     CellTypeAnd        -> "$and"
  212     CellTypeOr         -> "$or"
  213     CellTypeXor        -> "$xor"
  214     CellTypeXnor       -> "$xnor"
  215     CellTypeReduceAnd  -> "$reduce_and"
  216     CellTypeReduceOr   -> "$reduce_or"
  217     CellTypeReduceXor  -> "$reduce_xor"
  218     CellTypeReduceXnor -> "$reduce_xnor"
  219     CellTypeReduceBool -> "$reduce_bool"
  220     CellTypeShl        -> "$shl"
  221     CellTypeShr        -> "$shr"
  222     CellTypeSshl       -> "$sshl"
  223     CellTypeSshr       -> "$sshr"
  224     CellTypeShift      -> "$shift"
  225     CellTypeShiftx     -> "$shiftx"
  226     CellTypeLt         -> "$lt"
  227     CellTypeLe         -> "$le"
  228     CellTypeGt         -> "$gt"
  229     CellTypeGe         -> "$ge"
  230     CellTypeEq         -> "$eq"
  231     CellTypeNe         -> "$ne"
  232     CellTypeEqx        -> "$eqx"
  233     CellTypeNex        -> "$nex"
  234     CellTypeAdd        -> "$add"
  235     CellTypeSub        -> "$sub"
  236     CellTypeMul        -> "$mul"
  237     CellTypeDiv        -> "$div"
  238     CellTypeMod        -> "$mod"
  239     CellTypeLogicNot   -> "$logic_not"
  240     CellTypeLogicAnd   -> "$logic_and"
  241     CellTypeLogicOr    -> "$logic_or"
  242     CellTypeMux        -> "$mux"
  243     CellTypePmux       -> "$pmux"
  244     CellTypeBmux       -> "$bmux"
  245     CellTypeBUF        -> "$_BUF_"
  246 
  247 instance Show CellTypeCombinational where
  248   show ctc = Text.unpack (ppCellTypeCombinational ctc)
  249 
  250 ppCellType :: CellType -> Text
  251 ppCellType ct =
  252   case ct of
  253     CellTypeCombinational ctc -> ppCellTypeCombinational ctc
  254     CellTypeDff -> "$dff"
  255     CellTypeFf -> "$ff"
  256     CellTypeUnsupportedPrimitive t -> t
  257     CellTypeUserType t -> t
  258 
  259 instance Show CellType where
  260   show ct = Text.unpack (ppCellType ct)
  261 
  262 -- | Extract the name from a user-defined submodule 'CellType'
  263 asUserType :: CellType -> Text
  264 asUserType cellType =
  265   case cellType of
  266     CellTypeUserType t -> t
  267     CellTypeUnsupportedPrimitive t -> t
  268     _ ->
  269       panic "asUserType" [
  270           "Expected a user defined type, but got a primitive type: " <>
  271               Text.pack (show cellType)
  272       ]

  274 -- | A cell within an HDL module.
  275 data Cell bs = Cell
  276   { _cellHideName :: Bool -- ^ Whether the cell's name is human-readable (default: False)
  277   , _cellType :: CellType -- ^ The cell type
  278     -- NB: Yosys's documentation for write_json doesn't impose any restrictions
  279     -- on what type parameter values may take on, so we opt to be as permissive
  280     -- as possible by using Aeson Values. Most of the time, these Values will
  281     -- be strings, but they can also be numbers on occasion (e.g., if you call
  282     -- write_json using the -compat-int flag).
  283   , _cellParameters :: Map Text Aeson.Value -- ^ Metadata parameters
  284   , _cellAttributes :: Maybe Aeson.Value -- currently unused
  285   , _cellPortDirections :: Map Text Direction -- ^ Direction for each cell connection
  286   , _cellConnections :: Map Text bs -- ^ Bitrep for each cell connection
  287   } deriving (Show, Eq, Ord, Functor)
  288 
  289 makeLenses ''Cell
  290 
  291 instance Aeson.FromJSON (Cell [Bitrep]) where
  292   parseJSON = Aeson.withObject "cell" $ \o -> do
  293     _cellHideName <- Maybe.maybe False (/= (0::Int)) <$> o Aeson..:? "hide_name"
  294     _cellType <- o Aeson..: "type"
  295     _cellParameters <- o Aeson..: "parameters"
  296     _cellAttributes <- o Aeson..:? "attributes"
  297     _cellPortDirections <- o Aeson..: "port_directions"
  298     _cellConnections <- o Aeson..: "connections"
  299     pure Cell{..}

  351 -- | Read a collection of HDL modules from a file produced by Yosys' write_json command.
  352 loadYosysIR :: FilePath -> IO YosysIR
  353 loadYosysIR p = Aeson.eitherDecodeFileStrict p >>= \case
  354   Left err -> yosysError $ YosysError $ Text.pack err
  355   Right ir -> pure ir

  357 -- | Return the patterns for all of the input ports of a module
  358 moduleInputPorts :: Module -> Map Text [Bitrep]
  359 moduleInputPorts m =
  360   Map.mapMaybe
  361   ( \ip ->
  362       if ip ^. portDirection == DirectionInput || ip ^. portDirection == DirectionInout
  363       then Just (ip ^. portBits)
  364       else Nothing
  365   )
  366   $ m ^. modulePorts
  367 
  368 -- | Return the patterns for all of the output ports of a module
  369 moduleOutputPorts :: Module -> Map Text [Bitrep]
  370 moduleOutputPorts m =
  371   Map.mapMaybe
  372   ( \ip ->
  373       if isOutput (ip ^. portDirection)
  374       then Just (ip ^. portBits)
  375       else Nothing
  376   )
  377   $ m ^. modulePorts
  378 
  379 -- | Return the patterns for all of the input connections of a cell
  380 cellInputConnections :: Cell [b] -> Map Text [b]
  381 cellInputConnections c = Map.intersection (c ^. cellConnections) inp
  382   where
  383     inp = Map.filter isInput (c ^. cellPortDirections)
  384 
  385 -- | Return the patterns for all of the output connections of a cell
  386 cellOutputConnections :: Ord b => Cell [b] -> Map Text [b]
  387 cellOutputConnections c = Map.intersection (c ^. cellConnections) out
  388   where
  389     out = Map.filter isOutput (c ^. cellPortDirections)

  391 -- | Test whether a 'Cell' is a state element ('CellTypeDff' or 'CellTypeFf').
  392 cellIsRegister :: Cell bs -> Bool
  393 cellIsRegister c =
  394   case c ^. cellType of
  395     CellTypeDff -> True
  396     CellTypeFf -> True
  397     _ -> False
