   34 -- | Haskeline-specific repl implementation.
   35 --
   36 -- XXX this needs to handle Ctrl-C, which at the moment will just cause
   37 -- haskeline to exit.  See the function 'withInterrupt' for more info on how to
   38 -- handle this.
   39 repl :: Maybe FilePath -> Options -> REPL () -> IO ()
   40 repl mbBatch opts begin =
   41   runREPL (isJust mbBatch) opts (replBody mbBatch begin)
   42 
   43 
   44 replBody :: Maybe FilePath -> REPL () -> REPL ()
   45 replBody mbBatch begin =
   46   do settings <- MTL.liftIO (setHistoryFile replSettings)
   47      enableSubshell (runInputTBehavior style settings body)
   48   where
   49   body = withInterrupt $ do
   50     MTL.lift begin
   51     loop
   52 
   53   style = case mbBatch of
   54     Nothing   -> defaultBehavior
   55     Just path -> useFile path
   56 
   57   enableSubshell m =
   58     REPL $ \refs ->
   59       do let ro' = (rTopLevelRO refs){ roSubshell = subshell (runInputT replSettings (withInterrupt loop)) }
   60          unREPL m refs{ rTopLevelRO = ro' }
   61 
   62   loop = do
   63     prompt <- MTL.lift getPrompt
   64     mb     <- handleInterrupt (return (Just "")) (getInputLines prompt [])
   65     case mb of
   66       Just line
   67         | Just cmd <- parseCommand findCommand line -> do
   68           continue <- MTL.lift $ do
   69             handleInterrupt handleCtrlC (runCommand cmd)
   70             shouldContinue
   71           when continue loop
   72 
   73         | otherwise -> loop
   74 
   75       Nothing -> return ()
   76 
   77   getInputLines prompt ls =
   78     do mb <- fmap (filter (/= '\r')) <$> getInputLine prompt
   79        let newPrompt = map (\_ -> ' ') prompt
   80        case mb of
   81           Nothing -> return Nothing
   82           Just l | not (null l) && last l == '\\' ->
   83                                       getInputLines newPrompt (init l : ls)
   84                  | otherwise -> return $ Just $ unlines $ reverse $ l : ls

   88 -- | Try to set the history file.
   89 setHistoryFile :: Settings REPL -> IO (Settings REPL)
   90 setHistoryFile ss =
   91   do dir <- getAppUserDataDirectory "saw"
   92      createDirectoryIfMissing True dir
   93      return ss { historyFile = Just (dir </> "history") }
   94    `X.catch` \(X.SomeException {}) -> return ss
   95 
   96 -- | Haskeline settings for the REPL.
   97 replSettings :: Settings REPL
   98 replSettings  = Settings
   99   { complete       = replComp
  100   , historyFile    = Nothing
  101   , autoAddHistory = True
  102   }

  116       a <- unREPL m ref
  117       return (return a)
  118     unREPL runBody ref

  124 instance E.MonadThrow REPL where
  125   throwM e = REPL $ \_ -> X.throwIO e
  126 
  127 instance E.MonadCatch REPL where
  128   catch repl_op handler = REPL $ \ioref ->
  129     E.catch (unREPL repl_op ioref) (\e -> unREPL (handler e) ioref)
  130 
  131 instance E.MonadMask REPL where
  132   mask repl_op = REPL $ \ioref ->
  133     E.mask (\runIO -> unREPL (repl_op (\f -> REPL (\ioref' -> runIO (unREPL f ioref')))) ioref)
  134   uninterruptibleMask repl_op = REPL $ \ioref ->
  135     E.uninterruptibleMask (\runIO -> unREPL (repl_op (\f -> REPL (\ioref' -> runIO (unREPL f ioref')))) ioref)
  136   generalBracket acquire release repl_op = REPL $ \ioref ->
  137     E.generalBracket
  138     (unREPL acquire ioref)
  139     (\rsrc exitCase -> unREPL (release rsrc exitCase) ioref)
  140     (\rsrc -> unREPL (repl_op rsrc) ioref)
  141 #endif

  146 -- | Top-level completion for the REPL.
  147 replComp :: CompletionFunc REPL
  148 replComp cursor@(l,r)
  149   | ":" `isPrefixOf` l'
  150   , Just (cmd,rest) <- splitCommand l' = case findCommand cmd of
  151 
  152       [c] | null rest && not (any isSpace l') -> do
  153             return (l, [cmdComp cmd c])
  154           | otherwise -> do
  155             (rest',cs) <- cmdArgument (cBody c) (reverse (sanitize rest),r)
  156             return (unwords [rest', reverse cmd],cs)
  157 
  158       cmds ->
  159         return (l, [ cmdComp l' c | c <- cmds ])
  160 
  161   | l' == ":" =
  162       return (l, [ cmdComp l' c | c <- findCommand "" ])
  163   | otherwise = completeSAWScriptValue cursor
  164   where
  165   l' = sanitize (reverse l)
  166 
  167 -- | Generate a completion from a REPL command definition.
  168 cmdComp :: String -> CommandDescr -> Completion
  169 cmdComp prefix c = Completion
  170   { replacement = drop (length prefix) (cName c)
  171   , display     = cName c
  172   , isFinished  = True
  173   }
  174 
  175 -- | Dispatch to a completion function based on the kind of completion the
  176 -- command is expecting.
  177 cmdArgument :: CommandBody -> CompletionFunc REPL
  178 cmdArgument ct cursor@(l,_) = case ct of
  179   ExprArg _     -> completeSAWScriptValue cursor
  180   TypeArg _     -> completeSAWScriptType cursor
  181   FilenameArg _ -> completeFilename cursor
  182   ShellArg _    -> completeFilename cursor
  183   NoArg       _ -> return (l,[])
  184 
  185 data LexerMode = ModeNormal | ModeCryptol | ModeCryType | ModeQuote
  186 
  187 lexerMode :: String -> LexerMode
  188 lexerMode = normal
  189   where
  190     normal [] = ModeNormal
  191     normal ('{' : '{' : s) = cryptol s
  192     normal ('{' : '|' : s) = crytype s
  193     normal ('\"' : s) = quote s
  194     normal (_ : s) = normal s
  195 
  196     cryptol [] = ModeCryptol
  197     cryptol ('}' : '}' : s) = normal s
  198     cryptol (_ : s) = cryptol s
  199 
  200     crytype [] = ModeCryType
  201     crytype ('|' : '}' : s) = normal s
  202     crytype (_ : s) = crytype s
  203 
  204     quote [] = ModeQuote
  205     quote ('\"' : s) = normal s
  206     quote (_ : s) = quote s
  207 
  208 isIdentChar :: Char -> Bool
  209 isIdentChar c = isAlphaNum c || c `elem` "_\'"
  210 
  211 -- | Complete a name from the SAWScript value environment.
  212 completeSAWScriptValue :: CompletionFunc REPL
  213 completeSAWScriptValue cursor@(l, _) = do
  214   ns1 <- getSAWScriptValueNames
  215   ns2 <- getCryptolExprNames
  216   ns3 <- getCryptolTypeNames
  217   let n = reverse (takeWhile isIdentChar l)
  218       nameComps prefix ns = map (nameComp prefix) (filter (prefix `isPrefixOf`) ns)
  219   case lexerMode (reverse l) of
  220     ModeNormal  -> return (l, nameComps n ns1)
  221     ModeCryptol -> return (l, nameComps n ns2)
  222     ModeCryType -> return (l, nameComps n ns3)
  223     ModeQuote   -> completeFilename cursor
  224 
  225 -- | Complete a name from the SAWScript type environment.
  226 completeSAWScriptType :: CompletionFunc REPL
  227 completeSAWScriptType (l, _) = do
  228   ns <- getSAWScriptTypeNames
  229   let prefix = reverse (takeWhile isIdentChar l)
  230       nameComps = map (nameComp prefix) (filter (prefix `isPrefixOf`) ns)
  231   return (l, nameComps)
  232 
  233 -- | Generate a completion from a prefix and a name.
  234 nameComp :: String -> String -> Completion
  235 nameComp prefix c = Completion
  236   { replacement = drop (length prefix) c
  237   , display     = c
  238   , isFinished  = True
  239   }
