   25 -- | Load the 'Stmt's in a @.saw@ file.
   26 loadFile :: Options -> FilePath -> IO [Stmt]
   27 loadFile opts fname = do
   28   printOutLn opts Info $ "Loading file " ++ show fname
   29   ftext <- TextIO.readFile fname
   30   -- XXX: the print functions should take care of printing the position
   31   -- (clean this up when we clean out the printing infrastructure)
   32   -- XXX: the print functions should also take care of exiting on an error
   33   -- (immediately or later). For now, throw errors and print anything else.
   34   case lexSAW fname ftext of
   35     Left (_, pos, txt) -> do
   36       let txt' = show pos ++ ": " ++ Text.unpack txt
   37       throwIO $ userError txt'
   38 
   39     Right (tokens, optmsg) -> do
   40       case optmsg of
   41         Nothing -> return ()
   42         Just (vrb, pos, txt) -> do
   43           let txt' = show pos ++ ": " ++ Text.unpack txt
   44           case vrb of
   45             Error -> throwIO $ userError txt'
   46             _ -> printOutLn opts vrb txt'
   47       either throwIO return (parseFile tokens)
   48 
   49 parseFile :: [Token Pos] -> Either ParseError [Stmt]
   50 parseFile tokens = do
   51   case parseModule tokens of
   52     Left err -> Left err
   53     Right stmts -> Right stmts

   55 -- | Find a file, potentially looking in a list of multiple search paths (as
   56 -- specified via the @SAW_IMPORT_PATH@ environment variable or
   57 -- @-i@/@--import-path@ command-line options). If the file was successfully
   58 -- found, load it. If not, raise an error.
   59 findAndLoadFile :: Options -> FilePath -> IO [Stmt]
   60 findAndLoadFile opts fp = do
   61   let paths = importPath opts
   62   mfname <- findFile paths fp
   63   case mfname of
   64     Nothing -> fail $ unlines $
   65         [ "Couldn't find file: " ++ show fp
   66         , "  Searched in directories:"
   67         ] ++ map ("    " ++) paths
   68     Just fname ->
   69       -- NB: Normalise the path name. The default SAW_IMPORT_PATH contains ".",
   70       -- and the behavior of filepath's 'normalise' function is to prepend a
   71       -- search path to the front of the file path that is found, which can
   72       -- cause paths like "./foo.saw" to be returned. This looks ugly in error
   73       -- messages, where we would rather display "foo.saw" instead.
   74       loadFile opts (normalise fname)
