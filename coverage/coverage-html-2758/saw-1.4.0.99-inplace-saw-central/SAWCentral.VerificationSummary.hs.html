   55 vsVerifSolvers :: VerificationSummary -> Set Text
   56 vsVerifSolvers vs =
   57   Set.unions $
   58   map (\ms -> solverStatsSolvers (ms ^. psSolverStats)) (vsJVMMethodSpecs vs) ++
   59   map (\(CMSLLVM.SomeLLVM ms) -> solverStatsSolvers (ms ^. psSolverStats)) (vsLLVMMethodSpecs vs)
   60 
   61 vsTheoremSolvers :: VerificationSummary -> Set Text
   62 vsTheoremSolvers = Set.unions . map getSolvers . vsTheorems
   63   where getSolvers thm = solverStatsSolvers (thmStats thm)
   64 
   65 vsAllSolvers :: VerificationSummary -> Set Text
   66 vsAllSolvers vs = Set.union (vsVerifSolvers vs) (vsTheoremSolvers vs)

   68 computeVerificationSummary :: TheoremDB -> [JVMTheorem] -> [LLVMTheorem] -> [Theorem] -> VerificationSummary
   69 computeVerificationSummary db js ls thms =
   70   let roots = mconcat (
   71                 [ vcDeps vc | j <- js, vc <- j^.psVCStats ] ++
   72                 [ vcDeps vc | CMSLLVM.SomeLLVM l <- ls, vc <- l^.psVCStats ] ++
   73                 [ Set.singleton (thmNonce t) | t <- thms ])
   74       thms' = Map.elems (reachableTheorems db roots)
   75   in  VerificationSummary js ls thms'

   77 -- TODO: we could make things instances of a ToJSON typeclass instead of using the two methods below.
   78 msToJSON :: forall ext . Pretty (MethodId ext) => CMS.ProvedSpec ext -> Value
   79 msToJSON cms = object [
   80     ("type" .= ("method" :: String))
   81     , ("id" .= (indexValue $ cms ^. psSpecIdent))
   82     , ("method" .= (show $ pretty $ cms ^. psSpec.csMethod))
   83     , ("loc" .= (show $ pretty $ plSourceLoc $ cms ^. psSpec.csLoc))
   84     , ("status" .= case cms ^. psProofMethod of
   85                      SpecAdmitted -> "assumed" :: String
   86                      SpecProved   -> "verified")
   87 --    , ("specification" .= ("unknown" :: String)) -- TODO
   88     , ("dependencies" .= toJSON (map indexValue (Set.toList (cms ^. psSpecDeps))))
   89     , ("vcs"          .= toJSON (map vcToJSON (cms ^. psVCStats)))
   90     , ("elapsedtime"  .= toJSON (cms ^. psElapsedTime))
   91     ]
   92 
   93 vcToJSON :: CMS.VCStats -> Value
   94 vcToJSON vc = object ([
   95   ("type"  .= ("vc" :: String))
   96   , ("id"  .= indexValue (vcIdent vc))
   97   , ("loc" .= show (conditionLoc (vcMetadata vc)))
   98   , ("reason" .= conditionType (vcMetadata vc))
   99   , ("elapsedtime" .= toJSON (vcElapsedTime vc))
  100   , ("dependencies" .= toJSON (map indexValue (Set.toList (vcDeps vc))))
  101   , ("tags" .= toJSON (Set.toList (conditionTags (vcMetadata vc))))
  102   ] ++ theoremStatus (vcThmSummary vc)
  103   )
  104 
  105 thmToJSON :: Theorem -> Value
  106 thmToJSON thm = object ([
  107     ("type" .= ("property" :: String))
  108     , ("id" .= (indexValue $ thmNonce thm))
  109     , ("loc" .= (show $ thmLocation thm))
  110     , ("reason" .= (thmReason thm))
  111     , ("dependencies" .= toJSON (map indexValue (Set.toList (thmDepends thm))))
  112     , ("elapsedtime"  .= toJSON (thmElapsedTime thm))
  113   ] ++ theoremStatus (thmSummary thm)
  114     ++ case thmProgramLoc thm of
  115          Nothing -> []
  116          Just ploc -> [("ploc" .= plocToJSON ploc)]
  117   )
  118 
  119 theoremStatus :: TheoremSummary -> [(Key,Value)]
  120 theoremStatus summary = case summary of
  121       ProvedTheorem stats ->
  122         [ ("status"  .= ("verified" :: String))
  123         , ("provers" .= toJSON (Set.toList (solverStatsSolvers stats)))
  124         ]
  125       TestedTheorem n ->
  126         [ ("status"   .= ("tested" :: String))
  127         , ("numtests" .= toJSON n)
  128         ]
  129       AdmittedTheorem msg ->
  130         [ ("status"   .= ("assumed" :: String))
  131         , ("admitmsg" .= msg)
  132         ]
  133 
  134 --    , ("term" .= (show $ ppProp PPS.defaultOpts $ thmProp thm))
  135 
  136 plocToJSON :: ProgramLoc -> Value
  137 plocToJSON ploc = object
  138   [ "function" .= toJSON (functionName (plFunction ploc))
  139   , "loc"      .= show (plSourceLoc ploc)
  140   ]

  143 jsonVerificationSummary :: VerificationSummary -> String
  144 jsonVerificationSummary (VerificationSummary jspecs lspecs thms) =
  145   BLU.toString $ encode vals where
  146     vals = foldr (++) [] [jvals, lvals, thmvals]
  147     jvals = msToJSON <$> jspecs
  148     lvals = (\(CMSLLVM.SomeLLVM ls) -> msToJSON ls) <$> lspecs -- TODO: why is the type annotation required here?
  149     thmvals = thmToJSON <$> thms

  151 prettyVerificationSummary :: PPS.Opts -> DisplayNameEnv -> VerificationSummary -> String
  152 prettyVerificationSummary ppOpts nenv vs@(VerificationSummary jspecs lspecs thms) =
  153   show $ vsep
  154   [ prettyJVMSpecs jspecs
  155   , prettyLLVMSpecs lspecs
  156   , prettyTheorems thms
  157   , prettySolvers (Set.toList (vsAllSolvers vs))
  158   ] where
  159       section nm = "#" <+> nm
  160       item txt = "*" <+> txt
  161       code txt = vsep ["~~~~", txt, "~~~~"]
  162       subitem = indent 4 . item
  163       sectionWithItems _ _ [] = mempty
  164       sectionWithItems nm prt items =
  165         vsep [section nm, "", vsep (map prt items)]
  166       verifStatus s = case s ^. CMS.psProofMethod of
  167                         CMS.SpecAdmitted -> "assumed"
  168                         CMS.SpecProved   -> "verified"
  169       -- TODO: ultimately the goal is for the following to summarize all
  170       -- preconditions made by this verification, but we need to extract
  171       -- a bunch more information for that to be meaningful.
  172       {-
  173       condStatus s = (if null (s ^. (CMS.csPreState . CMS.csConditions))
  174                       then "without"
  175                       else "with") <+> "conditions"
  176                       -}
  177       prettyJVMSpecs ss =
  178         sectionWithItems "JVM Methods Analyzed" prettyJVMSpec ss
  179       prettyJVMSpec s =
  180         vsep [ item (fromString (s ^. CMS.psSpec.CMSJVM.csMethodName))
  181              -- , subitem (condStatus s)
  182              , subitem (verifStatus s)
  183              ]
  184       prettyLLVMSpecs ss =
  185         sectionWithItems "LLVM Functions Analyzed" prettyLLVMSpec ss
  186       prettyLLVMSpec (CMSLLVM.SomeLLVM s) =
  187         vsep [ item (fromString $ Text.unpack (s ^. CMS.psSpec.CMSLLVM.csName))
  188              -- , subitem (condStatus s)
  189              , subitem (verifStatus s)
  190              ]
  191       prettyTheorems ts =
  192         sectionWithItems "Theorems Proved or Assumed" (item . prettyTheorem) ts
  193       prettyTheorem t =
  194         vsep [ case thmSummary t of
  195                  ProvedTheorem{}   -> "Theorem:"
  196                  TestedTheorem n   -> "Theorem (randomly tested on" <+> viaShow n <+> "samples):"
  197                  AdmittedTheorem{} -> "Axiom:"
  198              , code (indent 2 (ppProp ppOpts nenv (thmProp t)))
  199              , ""
  200              ]
  201       prettySolvers ss =
  202         sectionWithItems "Solvers Used" (item . pretty) ss
