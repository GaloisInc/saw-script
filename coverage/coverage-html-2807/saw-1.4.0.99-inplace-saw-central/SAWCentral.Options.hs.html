   18 -- | Type we use to track the command-line option settings.
   19 data Options = Options
   20   { importPath       :: [FilePath]
   21   , classPath        :: [FilePath]
   22   , jarList          :: [FilePath]
   23   , javaBinDirs      :: [FilePath]
   24   , verbLevel        :: Verbosity
   25   , simVerbose       :: Int
   26   , detectVacuity    :: Bool
   27   , extraChecks      :: Bool
   28   , batchFile        :: Maybe String
   29   , runInteractively :: Bool
   30   , showHelp         :: Bool
   31   , showVersion      :: Bool
   32   , printShowPos     :: Bool
   33   , useColor         :: Bool
   34   , cleanMisVsCache  :: Maybe FilePath
   35   , printOutFn       :: Verbosity -> String -> IO ()
   36   , summaryFile      :: Maybe FilePath
   37   , summaryFormat    :: SummaryFormat
   38   } deriving (Show)
   39 
   40 -- | Verbosity is currently a linear setting (vs a mask or tree).  Any given
   41 -- level includes the outputs of all lower levels.
   42 data Verbosity
   43   = Silent
   44   | OnlyCounterExamples
   45   | Error
   46   | Warn
   47   | Info
   48   | Debug
   49     deriving (Show,Eq,Ord)
   50 
   51 data SummaryFormat
   52   = JSON | Pretty
   53   deriving (Show,Eq,Ord)

   78 printOutWith :: Verbosity -> Verbosity -> String -> IO ()
   79 printOutWith setting level msg
   80     | setting >= level = do
   81       t <- formatTime defaultTimeLocale "%T.%3q" <$> getCurrentTime
   82       putStr $ "[" ++ t ++ "] " ++ msg
   83     | otherwise        = return ()
   84 
   85 printOutLn :: Options -> Verbosity -> String -> IO ()
   86 printOutLn o v s = printOutFn o v (s ++ "\n")

   88 -- | Perform some additional post-processing on an 'Options' value based on
   89 -- whether certain environment variables are defined.
   90 --
   91 -- XXX: this does not belong in Options.hs; it is a top-level
   92 -- operation that's really part of main. But it's shared between saw
   93 -- and saw-remote-api so it has to live somewhere shared, and for the
   94 -- time being there isn't anywhere else.
   95 processEnv :: Options -> IO Options
   96 processEnv opts = do
   97   curEnv <- getEnvironment
   98   return $ foldr addSawOpt opts curEnv
   99   where
  100     addSawOpt :: (String, String) -> Options -> Options
  101     addSawOpt ("SAW_IMPORT_PATH", p) os =
  102       os { importPath = importPath os ++ splitSearchPath p }
  103     addSawOpt ("SAW_JDK_JAR", p) os = os { jarList = p : jarList os }
  104     addSawOpt ("CLASSPATH", p) os = os { jarList = jarList os ++ jars
  105                                        , classPath = classPath os ++ dirs
  106                                        }
  107                                     where
  108                                       es = splitSearchPath p
  109                                       (jars, dirs) = partition (".jar" `isExtensionOf`) es
  110     addSawOpt _ os = os
