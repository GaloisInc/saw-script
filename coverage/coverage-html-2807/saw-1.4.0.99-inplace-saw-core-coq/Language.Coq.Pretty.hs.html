   21 -- | Replace all occurrences of the double quote character @"@ with the string
   22 -- @""@, i.e., two copies of it, as this is how Coq escapes double quote
   23 -- characters.
   24 escapeStringLit :: String -> String
   25 escapeStringLit = concatMap (\c -> if c == '"' then "\"\"" else [c])

   36 -- TODO: import SAWCore pretty-printer?
   37 tightSepList :: Doc ann -> [Doc ann] -> Doc ann
   38 tightSepList _ [] = mempty
   39 tightSepList _ [d] = d
   40 tightSepList s (d:l) = d <> s <+> tightSepList s l
   41 
   42 looseSepList :: Doc ann -> [Doc ann] -> Doc ann
   43 looseSepList _ [] = mempty
   44 looseSepList _ [d] = d
   45 looseSepList s (d:l) = d <+> s <+> looseSepList s l
   46 
   47 commaSepList, starSepList, semiSepList :: [Doc ann] -> Doc ann
   48 commaSepList = tightSepList comma
   49 starSepList = looseSepList (text "*")
   50 semiSepList = tightSepList semi
   51 
   52 period :: Doc ann
   53 period = text "."
   54 
   55 ppIdent :: Ident -> Doc ann
   56 ppIdent (Ident s) = text s
   57 
   58 ppBinder :: Binder -> Doc ann
   59 ppBinder (Binder x Nothing)  = ppIdent x
   60 ppBinder (Binder x (Just t)) = parens (ppIdent x <+> colon <+> ppTerm PrecNone t)
   61 ppBinder (ImplicitBinder x Nothing)  = braces (ppIdent x)
   62 ppBinder (ImplicitBinder x (Just t)) = braces (ppIdent x <+> colon <+> ppTerm PrecNone t)
   63 
   64 ppPiBinder :: PiBinder -> Doc ann
   65 ppPiBinder (PiBinder Nothing t)  = ppTerm PrecApp t <+> text "->"
   66 ppPiBinder (PiBinder (Just x) t) =
   67   text "forall" <+> parens (ppIdent x <+> colon <+> ppTerm PrecNone t) <> comma
   68 ppPiBinder (PiImplicitBinder Nothing t)  = braces (ppTerm PrecApp t) <+> text "->"
   69 ppPiBinder (PiImplicitBinder (Just x) t) =
   70   text "forall" <+> braces (ppIdent x <+> colon <+> ppTerm PrecNone t) <> comma
   71 
   72 ppBinders :: [Binder] -> Doc ann
   73 ppBinders = hsep . map ppBinder
   74 
   75 ppMaybeTy :: Maybe Type -> Doc ann
   76 ppMaybeTy Nothing = mempty
   77 ppMaybeTy (Just ty) = colon <+> ppTerm PrecNone ty
   78 
   79 ppSort :: Sort -> Doc ann
   80 ppSort Prop = text "Prop"
   81 ppSort Set = text "Set"
   82 ppSort Type = text "Type"
   83 
   84 ppPi :: [PiBinder] -> Doc ann
   85 ppPi bs = hsep (map ppPiBinder bs)

   87 data Prec
   88   = PrecNone
   89   | PrecLambda
   90   | PrecApp
   91   | PrecAtom
   92   deriving (Eq, Ord)
   93 
   94 parensIf :: Bool -> Doc ann -> Doc ann
   95 parensIf p d = if p then parens d else d
   96 
   97 ppTerm :: Prec -> Term -> Doc ann
   98 ppTerm p e =
   99   case e of
  100     Lambda bs t ->
  101       parensIf (p > PrecLambda) $
  102       text "fun" <+> ppBinders bs <+> text "=>" <+> ppTerm PrecLambda t
  103     Fix ident binders returnType body ->
  104       parensIf (p > PrecLambda) $
  105       text "fix" <+> ppIdent ident <+> ppBinders binders <+> text ":"
  106         <+> ppTerm PrecNone returnType <+> text ":=" <+> ppTerm PrecLambda body
  107     Pi bs t ->
  108       parensIf (p > PrecLambda) $
  109       ppPi bs <+> ppTerm PrecLambda t
  110     Let x bs mty t body ->
  111       parensIf (p > PrecLambda) $
  112       fillSep
  113       [ text "let" <+> ppIdent x <+> ppBinders bs <+> ppMaybeTy mty <+>
  114         text ":=" <+> ppTerm PrecNone t <+> text "in"
  115       , ppTerm PrecLambda body ]
  116     If c t f ->
  117       parensIf (p > PrecLambda) $
  118       text "if" <+> ppTerm PrecNone c <+>
  119       text "then" <+> ppTerm PrecNone t <+>
  120       text "else" <+> ppTerm PrecLambda f
  121     App f [] ->
  122       ppTerm p f
  123     App f args ->
  124       parensIf (p > PrecApp) $
  125       hsep (ppTerm PrecApp f : map (ppTerm PrecAtom) args)
  126     Sort s ->
  127       ppSort s
  128     Var x -> ppIdent x
  129     ExplVar x ->
  130       parensIf (p > PrecApp) $
  131       string "@" <> ppIdent x
  132     Ascription tm tp ->
  133       parensIf (p > PrecLambda)
  134       (ppTerm PrecApp tm <+> text ":" <+> ppTerm PrecApp tp)
  135     NatLit i ->
  136       if i > 1000 then
  137         -- Explicitly convert from Z if an integer is too big
  138         parensIf (p > PrecLambda) (text "Z.to_nat" <+> integer i <> text "%Z")
  139       else
  140         integer i
  141     ZLit i ->
  142       -- we use hex unless our integer is a positive or negitive digit
  143       if abs i > 9  then let ui = toInteger (fromInteger i :: Word64)
  144                           in text ("0x" ++ showHex ui [] ++ "%Z")
  145       else if i < 0 then text ("(" ++ show i ++ ")%Z")
  146                     else text (show i ++ "%Z")
  147     List ts ->
  148       brackets (semiSepList (map (ppTerm PrecNone) ts))
  149     StringLit s ->
  150       dquotes (string $ escapeStringLit s)
  151     Scope term scope ->
  152       ppTerm PrecAtom term <> text "%" <> text scope
  153     Ltac s ->
  154       text "ltac:" <> parens (string s)
  155 
  156 ppDecl :: Decl -> Doc ann
  157 ppDecl decl = case decl of
  158   Axiom nm ty ->
  159     nest 2 (
  160       hsep [text "Axiom", ppIdent nm, text ":", ppTerm PrecNone ty, period]
  161     ) <> hardline
  162   Parameter nm ty ->
  163     nest 2 (
  164      hsep [text "Parameter", ppIdent nm, text ":", ppTerm PrecNone ty, period]
  165     ) <> hardline
  166   Variable nm ty ->
  167     nest 2 (
  168       hsep [text "Variable", ppIdent nm, text ":", ppTerm PrecNone ty, period]
  169     ) <> hardline
  170   Comment s ->
  171     text "(*" <+> text s <+> text "*)" <> hardline
  172   Definition nm bs mty body ->
  173     nest 2 (
  174       vsep
  175       [ hsep ([text "Definition", ppIdent nm] ++
  176              map ppBinder bs ++
  177              [ppMaybeTy mty, text ":="])
  178       , ppTerm PrecNone body <> period
  179       ]
  180     ) <> hardline
  181   InductiveDecl ind -> ppInductive ind
  182   Section nm ds ->
  183     vsep $ concat
  184      [ [ hsep [text "Section", ppIdent nm, period] ]
  185      , map (indent 2 . ppDecl) ds
  186      , [ hsep [text "End", ppIdent nm, period] ]
  187      ]
  188   Snippet s -> text s
  189 
  190 ppConstructor :: Constructor -> Doc ann
  191 ppConstructor (Constructor {..}) =
  192   nest 2 $
  193   hsep [ text "|"
  194        , ppIdent constructorName
  195        , text ":"
  196        , ppTerm PrecNone constructorType
  197        ]
  198 
  199 ppInductive :: Inductive -> Doc ann
  200 ppInductive (Inductive {..}) =
  201   (vsep
  202    ([ nest 2 $
  203       hsep ([ text "Inductive"
  204             , ppIdent inductiveName
  205             ]
  206             ++ map ppBinder inductiveParameters
  207             ++ [ text ":" ]
  208             ++ map ppPiBinder inductiveIndices
  209             ++ [ ppSort inductiveSort ]
  210             ++ [ text ":="]
  211            )
  212     ]
  213     <> map ppConstructor inductiveConstructors
  214     <> [ period ]
  215    )
  216   ) <> hardline
