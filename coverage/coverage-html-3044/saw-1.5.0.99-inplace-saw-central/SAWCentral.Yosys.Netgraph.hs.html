   45 --------------------------------------------------------------------------------
   46 -- ** Building a network graph from a Yosys module
   47 
   48 -- | A 'Netgraph' represents the data dependencies between 'Cell's in
   49 -- a module. The graph has 'Text' keys which are the cell instance
   50 -- names.
   51 data Netgraph = Netgraph
   52   { _netgraphGraph :: Graph.Graph
   53   , _netgraphNodeFromVertex :: Graph.Vertex -> (Cell [Bitrep], Text, [Text])
   54   -- , _netgraphVertexFromKey :: Text -> Maybe Graph.Vertex
   55   }
   56 makeLenses ''Netgraph
   57 
   58 moduleNetgraph :: Module -> Netgraph
   59 moduleNetgraph m =
   60   let
   61     sources :: Map Bitrep Text
   62     sources =
   63       Map.fromList $
   64       [ (b, cname)
   65       | (cname, c) <- Map.assocs (m ^. moduleCells)
   66       , b <- concat (Map.elems (cellOutputConnections c)) ]
   67 
   68     cellDeps :: Cell [Bitrep] -> [Text]
   69     cellDeps c
   70       | cellIsRegister c = []
   71       | otherwise =
   72         Set.toAscList $ Set.fromList $
   73         Maybe.mapMaybe (flip Map.lookup sources) $
   74         concat $ Map.elems $ cellInputConnections c
   75 
   76     nodes :: [(Cell [Bitrep], Text, [Text])]
   77     nodes = [ (c, cname, cellDeps c) | (cname, c) <- Map.assocs (m ^. moduleCells) ]
   78 
   79     (_netgraphGraph, _netgraphNodeFromVertex, _netgraphVertexFromKey) =
   80       Graph.graphFromEdges nodes
   81   in
   82     Netgraph{..}

   85 -- ** Building a SAWCore term from a network graph
   86 
   87 data ConvertedModule = ConvertedModule
   88   { _convertedModuleTerm :: SC.Term
   89   , _convertedModuleType :: SC.Term
   90   , _convertedModuleCryptolType :: C.Type
   91   }
   92 makeLenses ''ConvertedModule

   94 lookupPatternTerm ::
   95   (Ord b, Show b) =>
   96   SC.SharedContext ->
   97   YosysBitvecConsumer ->
   98   [b] ->
   99   Map [b] Preterm ->
  100   IO SC.Term
  101 lookupPatternTerm sc loc pat ts =
  102   case Map.lookup pat ts of
  103     -- if we can find that pattern exactly, great! use that term
  104     Just t -> scPreterm sc t
  105     -- otherwise, find each individual bit and append the terms
  106     Nothing ->
  107       do bits <-
  108            forM pat $ \b ->
  109            case Map.lookup [b] ts of
  110              Just t -> pure t
  111              Nothing -> yosysError $ YosysErrorNoSuchOutputBitvec (Text.pack $ show b) loc
  112          -- Yosys lists bits in little-endian order, while scVector expects big-endian, so reverse
  113          let ps = fusePreterms (reverse bits)
  114          scPreterms sc ps
  115 
  116 -- | Given a netgraph and an initial map from bit patterns to terms, populate that map with terms
  117 -- generated from the rest of the netgraph.
  118 netgraphToTerms ::
  119   SC.SharedContext ->
  120   Map Text ConvertedModule ->
  121   Netgraph ->
  122   Map [Bitrep] Preterm ->
  123   IO (Map [Bitrep] Preterm)
  124 netgraphToTerms sc env ng inputs
  125   | length (Graph.scc $ ng ^. netgraphGraph) /= length (ng ^. netgraphGraph)
  126   = yosysError $ YosysError "Network graph contains a cycle after splitting on DFFs; SAW does not currently support analysis of this circuit"
  127   | otherwise =
  128       let sorted = reverseTopSort $ ng ^. netgraphGraph
  129       in foldM doVertex inputs sorted
  130   where
  131     doVertex :: Map [Bitrep] Preterm -> Graph.Vertex -> IO (Map [Bitrep] Preterm)
  132     doVertex acc v =
  133       do let (c, cnm, _deps) = ng ^. netgraphNodeFromVertex $ v
  134          let outputFields = Map.filter isOutput (c ^. cellPortDirections)
  135          let lookupConn portname =
  136                case Map.lookup portname (c ^. cellConnections) of
  137                  Nothing ->
  138                    yosysError $
  139                    YosysError $
  140                    "Malformed Yosys file: Missing port " <> portname <> " for cell " <> cnm
  141                  Just bs ->
  142                    pure bs
  143 
  144          case c ^. cellType of
  145            CellTypeCombinational ctc ->
  146              do let doInput inm i =
  147                       do t <- lookupPatternTerm sc (YosysBitvecConsumerCell cnm inm) i acc
  148                          let w = fromIntegral (length i)
  149                          let s =
  150                                case Map.lookup (inm <> "_SIGNED") (c ^. cellParameters) of
  151                                  Just (Aeson.Number n) -> n > 0
  152                                  Just (Aeson.String x) -> textBinNat x > 0
  153                                  Just _ -> False
  154                                  Nothing -> False
  155                          pure (CellTerm t w s)
  156                 args <- Map.traverseWithKey doInput (cellInputConnections c)
  157                 bs <- lookupConn "Y"
  158                 let ywidth = fromIntegral (length bs)
  159                 t <- combCellToTerm sc ctc args ywidth
  160                 ts <- deriveTermsByIndices sc bs t
  161                 pure $ Map.union ts acc
  162            -- special handling for $dff/$ff nodes - we read their
  163            -- /output/ from the inputs map, and later detect and write
  164            -- their /input/ to the state
  165            CellTypeDff ->
  166              do dffout <- lookupConn "Q"
  167                 r <- lookupPatternTerm sc (YosysBitvecConsumerCell cnm "Q") dffout acc
  168                 ts <- deriveTermsByIndices sc dffout r
  169                 pure $ Map.union ts acc
  170            CellTypeFf ->
  171              do ffout <- lookupConn "Q"
  172                 r <- lookupPatternTerm sc (YosysBitvecConsumerCell cnm "Q") ffout acc
  173                 ts <- deriveTermsByIndices sc ffout r
  174                 pure $ Map.union ts acc
  175            CellTypeUnsupportedPrimitive nm
  176              | Map.null outputFields ->
  177                  -- Cells with no output ports are debugging cells, which we can simply ignore.
  178                  pure acc
  179              | otherwise ->
  180                  yosysError $ YosysErrorUnsupportedFF nm
  181            CellTypeUserType submoduleName ->
  182              case Map.lookup submoduleName env of
  183                Nothing ->
  184                  yosysError $ YosysErrorNoSuchSubmodule submoduleName cnm
  185                Just cm ->
  186                  do let doInput inm i = lookupPatternTerm sc (YosysBitvecConsumerCell cnm inm) i acc
  187                     args <- Map.traverseWithKey doInput (cellInputConnections c)
  188                     rin <- cryptolRecord sc args
  189                     r <- SC.scApply sc (cm ^. convertedModuleTerm) rin
  190                     -- once we've built a term, insert it along with each of its bits
  191                     ts <-
  192                       forM (Map.assocs $ cellOutputConnections c) $ \(o, out) ->
  193                       do t <- cryptolRecordSelect sc outputFields r o
  194                          deriveTermsByIndices sc out t
  195                     pure $ Map.union (Map.unions ts) acc

  197 convertModule ::
  198   SC.SharedContext ->
  199   Map Text ConvertedModule ->
  200   Module ->
  201   IO ConvertedModule
  202 convertModule sc env m =
  203   do let ng = moduleNetgraph m
  204 
  205      let inputPorts = moduleInputPorts m
  206      let outputPorts = moduleOutputPorts m
  207 
  208      inputFields <- forM inputPorts
  209        (\inp ->
  210            SC.scBitvector sc . fromIntegral $ length inp
  211        )
  212      outputFields <- forM outputPorts
  213        (\out ->
  214            SC.scBitvector sc . fromIntegral $ length out
  215        )
  216      inputRecordType <- cryptolRecordType sc inputFields
  217      outputRecordType <- cryptolRecordType sc outputFields
  218      inputRecord <- SC.scFreshVariable sc "input" inputRecordType
  219 
  220      derivedInputs <-
  221        forM (Map.assocs inputPorts) $ \(nm, inp) ->
  222        do t <- cryptolRecordSelect sc inputFields inputRecord nm
  223           deriveTermsByIndices sc inp t
  224 
  225      oneBitType <- SC.scBitvector sc 1
  226      xMsg <- SC.scString sc "Attempted to read X bit"
  227      xTerm <- SC.scGlobalApply sc (SC.mkIdent SC.preludeName "error") [oneBitType, xMsg]
  228      zMsg <- SC.scString sc "Attempted to read Z bit"
  229      zTerm <- SC.scGlobalApply sc (SC.mkIdent SC.preludeName "error") [oneBitType, zMsg]
  230      let inputs = Map.unions $ mconcat
  231            [ [ Map.fromList
  232                [ ( [BitrepZero], PretermBvNat 1 0 )
  233                , ( [BitrepOne], PretermBvNat 1 1 )
  234                , ( [BitrepX], PretermSlice 0 1 0 xTerm )
  235                , ( [BitrepZ], PretermSlice 0 1 0 zTerm )
  236                ]
  237              ]
  238            , derivedInputs
  239            ]
  240 
  241      terms <- netgraphToTerms sc env ng inputs
  242      outputRecord <- cryptolRecord sc =<< mapForWithKeyM outputPorts
  243        (\onm out -> lookupPatternTerm sc (YosysBitvecConsumerOutputPort onm) out terms)
  244 
  245      t <- SC.scAbstractTerms sc [inputRecord] outputRecord
  246      ty <- SC.scFun sc inputRecordType outputRecordType
  247 
  248      let toCryptol (nm, rep) = (C.mkIdent nm, C.tWord . C.tNum $ length rep)
  249      let cty = C.tFun
  250            (C.tRec . C.recordFromFields $ toCryptol <$> Map.assocs inputPorts)
  251            (C.tRec . C.recordFromFields $ toCryptol <$> Map.assocs outputPorts)
  252      pure ConvertedModule
  253        { _convertedModuleTerm = t
  254        , _convertedModuleType = ty
  255        , _convertedModuleCryptolType = cty
  256        }
