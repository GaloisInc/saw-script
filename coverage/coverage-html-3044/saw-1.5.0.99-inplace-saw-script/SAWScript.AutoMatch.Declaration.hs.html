   22 -- | Names are strings (for which we'll use Text)
   23 type Name = Text
   24 
   25 -- | Arguments are names with (Cryptol) types
   26 data Arg = Arg { argName :: Name
   27                , argType :: Type }
   28                deriving (Eq, Ord)
   29 
   30 instance Show Arg where
   31    showsPrec d (Arg n t) =
   32       showParen (d > app_prec) $
   33          showString (Text.unpack n)
   34          . showString " : "
   35          . showString (Text.unpack $ CryPP.pp t)
   36       where app_prec = 10
   37 
   38 -- | Declarations have a name, a return type, and a list of arguments
   39 data Decl = Decl { declName :: Name
   40                  , declType :: Type
   41                  , declArgs :: [Arg] }
   42                  deriving (Eq, Ord)
   43 
   44 instance Show Decl where
   45    showsPrec d (Decl n t as) =
   46       showParen (d > app_prec) $
   47          showString (Text.unpack n)
   48          . showString " : "
   49          . showString "("
   50          . showString (intercalate ", " (map show as))
   51          . showString ")"
   52          . showString " -> "
   53          . showString (Text.unpack $ CryPP.pp t)
   54       where
   55          app_prec = 10
   56 
   57 -- | The signature of a declaration is the multiset of its argument types crossed with its return type
   58 data Sig = Sig { sigArgs_ :: Map Type Int
   59                , sigOut_  :: Type }
   60                deriving (Eq, Ord)
   61 
   62 sigArgs :: Sig -> Map Type Int
   63 sigArgs = sigArgs_
   64 
   65 sigOut :: Sig -> Type
   66 sigOut  = sigOut_
   67 
   68 -- | Compute the signature of a declaration
   69 declSig :: Decl -> Sig
   70 declSig (Decl _ t as) =
   71   flip Sig t $
   72     foldr (uncurry (Map.insertWith (const succ))) Map.empty $
   73       map (argType &&& const 1) as
