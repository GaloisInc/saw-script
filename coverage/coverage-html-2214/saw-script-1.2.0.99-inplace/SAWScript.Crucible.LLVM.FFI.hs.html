    8 -- | Commands for verifying Cryptol foreign functions against their Cryptol
    9 -- implementations.
   10 --
   11 -- Since the Cryptol FFI specifies the exact way that foreign function
   12 -- implementations are called
   13 -- (https://galoisinc.github.io/cryptol/master/FFI.html), we can automatically
   14 -- generate the necessary LLVM setup script for verifying Cryptol foreign
   15 -- functions which also have Cryptol reference implementations. However, often
   16 -- the foreign (LLVM) representation of Cryptol data types as specified by the
   17 -- FFI is different from the representation that SAW directly maps the Cryptol
   18 -- type to (e.g. bitvectors are widened to machine-size integers, and
   19 -- multidimensional arrays are flattened). Therefore, if we were writing the
   20 -- setup spec by hand, we would use some Cryptol built in functions to do the
   21 -- appropriate conversions. Here, we use the SAW Core built in functions (which
   22 -- the Cryptol builtins are converted to) directly for these operations.
   23 --
   24 -- In order for SAW to match overrides with the resulting LLVM spec, we always
   25 -- create symbolic variables in the LLVM representation and pass them directly
   26 -- as arguments to the LLVM function, while asserting the necessary
   27 -- preconditions and doing the appropriate conversions to pass the Cryptol
   28 -- version of the variables to Cryptol. If the mapping from Cryptol to LLVM
   29 -- representation is not surjective, then preconditions are necessary to assert
   30 -- that the LLVM input variables start in a state that is in the image of such
   31 -- mapping (a fact which the LLVM implementation of the function may depend on).
   32 -- Similarly, for function outputs where the mapping from Cryptol to LLVM
   33 -- representation is not unique, we cannot directly verify functional
   34 -- correctness by using @llvm_return@ (for LLVM return values) or
   35 -- @llvm_points_to@ (for LLVM output arguments) on the Cryptol function output
   36 -- converted to LLVM representation. Instead, we must create a fresh symbolic
   37 -- variable representing the LLVM output, then convert that to Cryptol
   38 -- representation and compare it with the Cryptol result.
   39 module SAWScript.Crucible.LLVM.FFI
   40   ( llvm_ffi_setup
   41   ) where

   89 -- | Information about a Cryptol 'FFIType' that tells us how to generate a setup
   90 -- spec for values of that type.
   91 data FFITypeInfo = FFITypeInfo
   92   { -- | The representation of the type when passed to the foreign (LLVM)
   93     -- function.
   94     ffiLLVMType     :: LLVM.Type
   95     -- | The same as 'ffiLLVMType' but as a SAWCore term.
   96   , ffiLLVMCoreType :: OpenTerm
   97     -- | If 'Nothing' then the Cryptol and foreign representations are the same
   98     -- and there is no need to convert.
   99   , ffiConv         :: Maybe FFIConv
  100   }
  101 
  102 -- | How to convert between the Cryptol and foreign representations of an
  103 -- 'FFIType'.
  104 data FFIConv = FFIConv
  105   { -- | The representation of the type when passed to the Cryptol function.
  106     ffiCryType :: OpenTerm
  107     -- | Assert any preconditions guaranteed by the Cryptol FFI on the LLVM
  108     -- representation.
  109   , ffiPrecond :: OpenTerm {- : ffiLLVMType -} -> LLVMCrucibleSetupM ()
  110     -- | Convert from the foreign representation to the Cryptol representation.
  111   , ffiToCry :: OpenTerm {- : ffiLLVMType -} -> OpenTerm {- : ffiCryType -}
  112     -- | Convert from the Cryptol representation to the foreign representation.
  113     -- This is only defined when this conversion is unique, that is, when the
  114     -- Cryptol and LLVM representations are isomorphic and this function is the
  115     -- inverse of 'ffiToCry'.
  116   , ffiToLLVM :: Maybe
  117       (OpenTerm {- : ffiCryType -} -> OpenTerm {- : FFILLVMType -})
  118   }
  119 
  120 -- | How to check functional correctness of an 'FFIType' result.
  121 data FFIPostcond
  122   -- | Convert the Cryptol result to the LLVM representation using the given
  123   -- function, for use with @llvm_return@ or @llvm_points_to@.
  124   = FFIPostcondConvToLLVM
  125       (OpenTerm {- : ffiCryType -} -> LLVMCrucibleSetupM (AllLLVM SetupValue))
  126   -- | Given the LLVM result and the Cryptol result, convert the LLVM result to
  127   -- the Cryptol representation then assert as a postcondition that they are
  128   -- equal.
  129   | FFIPostcondConvToCryEq
  130       (TypedTerm {- ffiLLVMType -} -> OpenTerm {- ffiCryType -} ->
  131         LLVMCrucibleSetupM ())

  133 -- | Generate a @LLVMSetup@ spec that can be used to verify that the given
  134 -- monomorphic Cryptol term, consisting of a Cryptol foreign function fully
  135 -- applied to any type arguments, has a correct foreign (LLVM) implementation
  136 -- with respect to its Cryptol implementation.
  137 llvm_ffi_setup :: TypedTerm -> LLVMCrucibleSetupM ()
  138 llvm_ffi_setup TypedTerm { ttTerm = appTerm } = do
  139   let (funTerm, tyArgTerms) = asApplyAll appTerm
  140   sc <- lll getSharedContext
  141   let ?ctx = FFISetupCtx {..}
  142   cryEnv <- lll $ rwCryptol <$> getMergedEnv
  143   case asConstant funTerm of
  144     Just (ec, funDef)
  145       | Just FFIFunType {..} <- Map.lookup (Term.ecName ec) (eFFITypes cryEnv) -> do
  146         when (isNothing funDef) do
  147           throwFFISetup
  148             "Cannot verify foreign function with no Cryptol implementation"
  149         tenv <- buildTypeEnv ffiTParams tyArgTerms
  150         llvmSizeArgs <- lio $ traverse mkSizeArg tyArgTerms
  151         (cryArgs, concat -> llvmInArgs) <- unzip <$> zipWithM
  152           (\i -> setupInArg tenv ("in" <> Text.pack (show i)))
  153           [0 :: Integer ..]
  154           ffiArgTypes
  155         (llvmOutArgs, post) <- setupRet tenv ffiRetType
  156         llvm_execute_func (llvmSizeArgs ++ llvmInArgs ++ llvmOutArgs)
  157         post $ applyOpenTermMulti (closedOpenTerm appTerm) cryArgs
  158     _ ->
  159       throwFFISetup
  160         "Not a (monomorphic instantiation of a) Cryptol foreign function"
  161 
  162 -- | Report an error in generating setup for a foreign function.
  163 throwFFISetup :: Ctx => String -> LLVMCrucibleSetupM a
  164 throwFFISetup msg =
  165   throwLLVMFun "llvm_ffi_setup" $
  166     "Cannot generate FFI setup for " ++ showTerm (funTerm ?ctx) ++ ":\n" ++ msg

  168 -- | Given a list of type parameters and their actual values as terms, create a
  169 -- type environment binding them.
  170 buildTypeEnv :: Ctx => [TParam] -> [Term] -> LLVMCrucibleSetupM TypeEnv
  171 buildTypeEnv [] [] = pure mempty
  172 buildTypeEnv (param:params) (argTerm:argTerms) =
  173   case asCtorParams argTerm of
  174     Just (primName -> "Cryptol.TCNum", [], [asNat -> Just n]) ->
  175       bindTypeVar (TVBound param) (Left (Nat (toInteger n))) <$>
  176         buildTypeEnv params argTerms
  177     _ ->
  178       throwFFISetup $
  179         "Not a numeric literal type argument: " ++ showTerm argTerm
  180 buildTypeEnv params [] = throwFFISetup $
  181   "Foreign function not fully instantiated;\n"
  182   ++ "Missing type arguments for: " ++ intercalate ", " (map pretty params)
  183 buildTypeEnv [] _ = throwFFISetup "Too many (type) arguments"
  184 
  185 -- | Given a Cryptol type argument as a term, return the corresponding size_t
  186 -- LLVM argument.
  187 mkSizeArg :: Ctx => Term -> IO (AllLLVM SetupValue)
  188 mkSizeArg tyArgTerm = do
  189   {- `tyArgTerm : [sizeBitSize]
  190   => Cryptol.ecNumber tyArgTerm (Vec sizeBitSize Bool)
  191                       (Cryptol.PLiteralSeqBool (Cryptol.TCNum sizeBitSize))
  192   -}
  193   openToSetupTerm $
  194     applyGlobalOpenTerm "Cryptol.ecNumber"
  195       [ closedOpenTerm tyArgTerm
  196       , vectorTypeOpenTerm sizeBitSize boolTypeOpenTerm
  197       , applyGlobalOpenTerm "Cryptol.PLiteralSeqBool"
  198           [ctorOpenTerm "Cryptol.TCNum" [sizeBitSize]]
  199       ]
  200   where
  201   sizeBitSize = natOpenTerm $
  202     fromIntegral $ finiteBitSize (undefined :: CSize)

  204 -- | Do setup for an input argument, returning the term to pass to the Cryptol
  205 -- function and a list of arguments to pass to the LLVM function.
  206 setupInArg :: Ctx => TypeEnv -> Text -> FFIType ->
  207   LLVMCrucibleSetupM (OpenTerm, [AllLLVM SetupValue])
  208 setupInArg tenv = go
  209   where
  210   go name ffiType =
  211     case ffiType of
  212       FFIBool ->
  213         valueInArg boolTypeInfo
  214       FFIBasic ffiBasicType ->
  215         valueInArg =<< basicTypeInfo ffiBasicType
  216       FFIArray lengths ffiBasicType -> do
  217         FFITypeInfo {..} <- arrayTypeInfo tenv lengths ffiBasicType
  218         (arr, cryTerm) <- singleInArg FFITypeInfo {..}
  219         ptr <- llvm_alloc_readonly ffiLLVMType
  220         llvm_points_to True ptr (anySetupTerm arr)
  221         pure (cryTerm, [ptr])
  222       FFITuple ffiTypes ->
  223         tupleInArgs <$> setupTupleArgs go name ffiTypes
  224       FFIRecord ffiTypeMap ->
  225         tupleInArgs <$> setupRecordArgs go name ffiTypeMap
  226     where
  227     valueInArg ffiTypeInfo = do
  228       (x, cryTerm) <- singleInArg ffiTypeInfo
  229       pure (cryTerm, [anySetupTerm x])
  230     singleInArg FFITypeInfo {..} = do
  231       x <- llvm_fresh_var name ffiLLVMType
  232       let ox = typedToOpenTerm x
  233       cryTerm <-
  234         case ffiConv of
  235           Just FFIConv {..} -> do
  236             ffiPrecond ox
  237             pure $ ffiToCry ox
  238           Nothing -> pure ox
  239       pure (x, cryTerm)
  240     tupleInArgs (unzip -> (terms, inArgss)) =
  241       (tupleOpenTerm' terms, concat inArgss)
  242 
  243 -- | Do setup for the return value, returning a list of output arguments to pass
  244 -- to the LLVM function and a function that asserts functional correctness given
  245 -- the Cryptol result.
  246 setupRet :: Ctx => TypeEnv -> FFIType ->
  247   LLVMCrucibleSetupM ([AllLLVM SetupValue], OpenTerm -> LLVMCrucibleSetupM ())
  248 setupRet tenv ffiType =
  249   case ffiType of
  250     FFIBool               -> pure $ retValue boolTypeInfo
  251     FFIBasic ffiBasicType -> retValue <$> basicTypeInfo ffiBasicType
  252     _                     -> setupOutArg tenv ffiType
  253   where
  254   retValue FFITypeInfo {..} =
  255     let post cryRet =
  256           case getFFIPostcond ffiConv of
  257             FFIPostcondConvToLLVM toLLVM ->
  258               llvm_return =<< toLLVM cryRet
  259             FFIPostcondConvToCryEq cryEq -> do
  260               llvmRet <- llvm_fresh_var "ret" ffiLLVMType
  261               llvm_return (anySetupTerm llvmRet)
  262               cryEq llvmRet cryRet
  263     in  ([], post)
  264 
  265 -- | Do setup for a return value passed as output arguments to the LLVM
  266 -- function, returning a list of output arguments to pass to the LLVM function
  267 -- and a function that asserts functional correctness given the Cryptol
  268 -- result.
  269 setupOutArg :: Ctx => TypeEnv -> FFIType ->
  270   LLVMCrucibleSetupM ([AllLLVM SetupValue], OpenTerm -> LLVMCrucibleSetupM ())
  271 setupOutArg tenv = go "out"
  272   where
  273   go name ffiType =
  274     case ffiType of
  275       FFIBool ->
  276         singleOutArg boolTypeInfo
  277       FFIBasic ffiBasicType ->
  278         singleOutArg =<< basicTypeInfo ffiBasicType
  279       FFIArray lengths ffiBasicType ->
  280         singleOutArg =<< arrayTypeInfo tenv lengths ffiBasicType
  281       FFITuple ffiTypes -> do
  282         (outArgss, posts) <- unzip <$> setupTupleArgs go name ffiTypes
  283         let len = fromIntegral $ length ffiTypes
  284             post ret = zipWithM_
  285               (\i p -> p (projTupleOpenTerm' len i ret))
  286               [0..]
  287               posts
  288         pure (concat outArgss, post)
  289       FFIRecord ffiTypeMap -> do
  290         -- The FFI passes record elements by display order, while SAW
  291         -- represents records by tuples in canonical order
  292         (outArgss, posts) <- unzip <$> setupRecordArgs go name ffiTypeMap
  293         let canonFields = map fst $ canonicalFields ffiTypeMap
  294             len = fromIntegral $ length canonFields
  295             post ret = zipWithM_
  296               (\field p -> do
  297                 let ix = fromIntegral
  298                       case elemIndex field canonFields of
  299                         Just i -> i
  300                         Nothing -> panic "setupOutArg"
  301                           ["Bad record field access"]
  302                 p (projTupleOpenTerm' len ix ret))
  303               (displayOrder ffiTypeMap)
  304               posts
  305         pure (concat outArgss, post)
  306     where
  307     singleOutArg FFITypeInfo {..} = do
  308       ptr <- llvm_alloc ffiLLVMType
  309       let post cryRet =
  310             case getFFIPostcond ffiConv of
  311               FFIPostcondConvToLLVM toLLVM ->
  312                 llvm_points_to True ptr =<< toLLVM cryRet
  313               FFIPostcondConvToCryEq cryEq -> do
  314                 out <- llvm_fresh_var name ffiLLVMType
  315                 llvm_points_to True ptr (anySetupTerm out)
  316                 cryEq out cryRet
  317       pure ([ptr], post)

  319 -- | Given an 'FFIConv', return a way to assert functional correctness,
  320 -- preferring 'FFIPostcondConvToLLVM' when possible. If no 'FFIConv', then
  321 -- return a trivial conversion to LLVM.
  322 getFFIPostcond :: Ctx => Maybe FFIConv -> FFIPostcond
  323 getFFIPostcond conv =
  324   case conv of
  325     Just FFIConv {..} ->
  326       case ffiToLLVM of
  327         Just toLLVM -> toLLVMWith toLLVM
  328         Nothing ->
  329           FFIPostcondConvToCryEq \llvmTerm cryTerm -> do
  330             -- ffiToCry llvmTerm == cryTerm
  331             lhs <- lio $ completeOpenTerm sc $
  332               ffiToCry $ typedToOpenTerm llvmTerm
  333             rhs <- lio $ completeOpenTerm sc cryTerm
  334             eqTerm <- lio $ mkTypedTerm sc =<< scEq sc lhs rhs
  335             llvm_postcond eqTerm
  336     Nothing -> toLLVMWith id
  337   where
  338   FFISetupCtx {..} = ?ctx
  339   toLLVMWith toLLVM = FFIPostcondConvToLLVM (lio . openToSetupTerm . toLLVM)

  341 -- | Call the given setup function on subparts of the tuple, naming them by
  342 -- index.
  343 setupTupleArgs :: (Text -> FFIType -> LLVMCrucibleSetupM a) ->
  344   Text -> [FFIType] -> LLVMCrucibleSetupM [a]
  345 setupTupleArgs setup name =
  346   zipWithM (\i -> setup (name <> "." <> Text.pack (show i))) [0 :: Integer ..]

  348 -- | Call the given setup function on subparts of the record, naming them by
  349 -- field name.
  350 setupRecordArgs :: (Text -> FFIType -> LLVMCrucibleSetupM a) ->
  351   Text -> RecordMap Cry.Ident FFIType -> LLVMCrucibleSetupM [a]
  352 setupRecordArgs setup name ffiTypeMap =
  353   traverse
  354     (\(field, ty) -> setup (name <> "." <> identText field) ty)
  355     (displayFields ffiTypeMap)

  357 -- | Type info for 'FFIBool'.
  358 boolTypeInfo :: Ctx => FFITypeInfo
  359 boolTypeInfo =
  360   FFITypeInfo
  361     { ffiLLVMType = llvm_int 8
  362     , ffiLLVMCoreType = bvTypeOpenTerm (8 :: Natural)
  363     , ffiConv =
  364         Just FFIConv
  365           { ffiCryType = boolTypeOpenTerm
  366           , ffiPrecond = precondBVZeroPrefix 8 7
  367           , ffiToCry = \x {- : Vec 8 Bool -} -> -- : Bool
  368               {- x != zero
  369               => bvNonzero 8 x -}
  370               applyGlobalOpenTerm "Prelude.bvNonzero" [natOpenTerm 8, x]
  371           , ffiToLLVM = Nothing
  372           }
  373     }

  375 -- | Type info for a 'FFIBasicType'.
  376 basicTypeInfo :: Ctx => FFIBasicType -> LLVMCrucibleSetupM FFITypeInfo
  377 basicTypeInfo (FFIBasicVal ffiBasicValType) = pure
  378   case ffiBasicValType of
  379     FFIWord (fromInteger -> n) ffiWordSize ->
  380       FFITypeInfo
  381         { ffiLLVMType = llvm_int llvmSize
  382         , ffiLLVMCoreType = bvTypeOpenTerm (llvmSize :: Natural)
  383         , ffiConv = do
  384             -- No need for conversion if the word size exactly matches a
  385             -- machine word size.
  386             guard (n < llvmSize)
  387             Just FFIConv
  388               { ffiCryType = bvTypeOpenTerm n
  389               , ffiPrecond =
  390                   precondBVZeroPrefix llvmSize (llvmSize - n)
  391               , ffiToCry = \x {- : Vec llvmSize Bool -} -> -- : Vec n Bool
  392                   {- drop (llvmSize - n) x
  393                   => Prelude.bvTrunc (llvmSize - n) n x -}
  394                   applyGlobalOpenTerm "Prelude.bvTrunc"
  395                     [natOpenTerm (llvmSize - n), natOpenTerm n, x]
  396               , ffiToLLVM = Nothing
  397               }
  398         }
  399       where
  400       llvmSize :: Integral a => a
  401       llvmSize =
  402         case ffiWordSize of
  403           FFIWord8  -> 8
  404           FFIWord16 -> 16
  405           FFIWord32 -> 32
  406           FFIWord64 -> 64
  407     FFIFloat _ _ ffiFloatSize ->
  408       let (ffiLLVMType, ffiLLVMCoreType) =
  409             case ffiFloatSize of
  410               FFIFloat32 -> (llvm_float, globalOpenTerm "Prelude.Float")
  411               FFIFloat64 -> (llvm_double, globalOpenTerm "Prelude.Double")
  412       in  FFITypeInfo
  413             { ffiConv = Nothing
  414             , .. }
  415 basicTypeInfo (FFIBasicRef _) =
  416   throwFFISetup "GMP types (Integer, Rational, Z) not supported"

  418 -- | Assert the precondition that a prefix of the given bitvector is zero.
  419 precondBVZeroPrefix :: Ctx =>
  420   Natural {- totalLen -} -> Natural ->
  421   OpenTerm {- Vec totalLen Bool -} -> LLVMCrucibleSetupM ()
  422 precondBVZeroPrefix totalLen zeroLen x = do
  423   let zeroLenTerm = natOpenTerm zeroLen
  424       precond =
  425         {- take zeroLen x == zero
  426         => Prelude.bvEq zeroLen
  427                         (take Bool zeroLen (totalLen - zeroLen) x)
  428                         (bvNat zeroLen 0) -}
  429         applyGlobalOpenTerm "Prelude.bvEq"
  430           [ zeroLenTerm
  431           , applyGlobalOpenTerm "Prelude.take"
  432               [ boolTypeOpenTerm
  433               , zeroLenTerm
  434               , natOpenTerm (totalLen - zeroLen)
  435               , x
  436               ]
  437           , applyGlobalOpenTerm "Prelude.bvNat"
  438               [zeroLenTerm, natOpenTerm 0]
  439           ]
  440   llvm_precond =<< lio (openToTypedTerm precond)
  441 
  442 -- | Type info for the 'FFIArray' type.
  443 arrayTypeInfo :: Ctx => TypeEnv -> [Type] -> FFIBasicType ->
  444   LLVMCrucibleSetupM FFITypeInfo
  445 arrayTypeInfo tenv lenTypes ffiBasicType = do
  446   let lens :: Integral a => [a]
  447       lens = map (fromInteger . finNat' . evalNumType tenv) lenTypes
  448       lenTerms = map natOpenTerm lens
  449       totalLen :: Integral a => a
  450       totalLen = product lens
  451       totalLenTerm = natOpenTerm $ fromInteger totalLen
  452   FFITypeInfo {..} <- basicTypeInfo ffiBasicType
  453   pure FFITypeInfo
  454     { ffiLLVMType = llvm_array totalLen ffiLLVMType
  455     , ffiLLVMCoreType = vectorTypeOpenTerm totalLenTerm ffiLLVMCoreType
  456     , ffiConv =
  457         case (lenTerms, ffiConv) of
  458           -- If the array is flat and there is no need to convert individual
  459           -- elements, then there is no need to convert the array
  460           ([_], Nothing) -> Nothing
  461           _ -> do
  462             let basicCryType = maybe ffiLLVMCoreType ffiCryType ffiConv
  463                 basicToLLVM = maybe (Just id) ffiToLLVM ffiConv
  464                 cumulLenTerms = map natOpenTerm $ scanl1 (*) lens
  465                 arrCryType :| cumulElemTypes =
  466                   NE.scanr vectorTypeOpenTerm basicCryType lenTerms
  467 
  468                 noArrayLengths :: a
  469                 noArrayLengths =
  470                   panic "arrayTypeInfo"
  471                         ["FFIArray with empty list of length types"]
  472 
  473                 lenTermsTail =
  474                   case lenTerms of
  475                     _:lenTermsTail' -> lenTermsTail'
  476                     [] -> noArrayLengths
  477                 cumulElemTypesTail =
  478                   case cumulElemTypes of
  479                     _:cumulElemTypesTail' -> cumulElemTypesTail'
  480                     [] -> noArrayLengths
  481 
  482                 -- Note that the length of cumulLenTerms will always be one
  483                 -- greater than the lengths of the other two lists, which means
  484                 -- that zip3 will drop the last element of cumulLenTerms. This
  485                 -- is fine, as `cumul` is used in such a way where the last
  486                 -- element is never required. (See the comments next to the uses
  487                 -- of `cumul` below.)
  488                 cumul = zip3 cumulLenTerms lenTermsTail cumulElemTypesTail
  489             Just FFIConv
  490               { ffiCryType = arrCryType
  491               , ffiPrecond = \arr {- : Vec totalLen ffiLLVMCoreType -} ->
  492                   for_ ffiConv \FFIConv {..} ->
  493                     for_ [0 .. totalLen - 1] \i -> do
  494                       {- arr @ i
  495                       => Prelude.at len ffiLLVMCoreType arr i -}
  496                       ffiPrecond $
  497                         applyGlobalOpenTerm "Prelude.at"
  498                           [totalLenTerm, ffiLLVMCoreType, arr, natOpenTerm i]
  499               , ffiToCry = \llvmArr {- : Vec totalLen ffiLLVMCoreType -} ->
  500                   let flatCryArr =
  501                         case ffiConv of
  502                           Just FFIConv {..} ->
  503                             {- map ffiToCry llvmArr
  504                             => Prelude.map ffiLLVMCoreType
  505                                            ffiCryType
  506                                            (\x -> ffiToCry x)
  507                                            totalLen
  508                                            llvmArr -}
  509                             applyGlobalOpenTerm "Prelude.map"
  510                               [ ffiLLVMCoreType
  511                               , ffiCryType
  512                               , lambdaOpenTerm "x" ffiLLVMCoreType ffiToCry
  513                               , totalLenTerm
  514                               , llvmArr
  515                               ]
  516                           Nothing ->
  517                             llvmArr
  518                   in  {- if lens = [x, y, z]
  519                           split x y (Vec z ffiCryType)
  520                                 (split (x * y) z ffiCryType flatCryArr) -}
  521                       foldr
  522                         (\(cumulLen, dimLen, arrElemType) arr ->
  523                           applyGlobalOpenTerm "Prelude.split"
  524                             [cumulLen, dimLen, arrElemType, arr])
  525                         flatCryArr
  526                         cumul
  527               , ffiToLLVM = basicToLLVM <&> \toLLVM cryArr ->
  528                   -- If the element representations are isomorphic, then the
  529                   -- array representations are isomorphic as well
  530                   let flatCryArr =
  531                         {- if lens = [x, y, z]
  532                             join (x * y) z ffiCryType
  533                                  (join x y (Vec z ffiCryType) cryArr) -}
  534                         foldr
  535                           (\(cumulLen, dimLen, arrElemType) arr ->
