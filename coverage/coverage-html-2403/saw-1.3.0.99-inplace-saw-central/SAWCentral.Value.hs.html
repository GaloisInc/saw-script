  403 showsProofResult :: PPS.Opts -> ProofResult -> ShowS
  404 showsProofResult opts r =
  405   case r of
  406     ValidProof _ _ -> showString "Valid"
  407     InvalidProof _ ts _ -> showString "Invalid: [" . showMulti "" ts
  408     UnfinishedProof st  -> showString "Unfinished: " . shows (length (psGoals st)) . showString " goals remaining"
  409   where
  410     showVal t = shows (ppFirstOrderValue opts t)
  411     showEqn (x, t) = showEC x . showString " = " . showVal t
  412     showEC ec = showString (unpack (toShortName (ecName ec)))
  413 
  414     showMulti _ [] = showString "]"
  415     showMulti s (eqn : eqns) = showString s . showEqn eqn . showMulti ", " eqns
  416 
  417 showsSatResult :: PPS.Opts -> SatResult -> ShowS
  418 showsSatResult opts r =
  419   case r of
  420     Unsat _ -> showString "Unsat"
  421     Sat _ ts -> showString "Sat: [" . showMulti "" ts
  422     SatUnknown  -> showString "Unknown"
  423   where
  424     showVal t = shows (ppFirstOrderValue opts t)
  425     showEC ec = showString (unpack (toShortName (ecName ec)))
  426     showEqn (x, t) = showEC x . showString " = " . showVal t
  427     showMulti _ [] = showString "]"
  428     showMulti s (eqn : eqns) = showString s . showEqn eqn . showMulti ", " eqns

  430 showSimpset :: PPS.Opts -> Simpset a -> String
  431 showSimpset opts ss =
  432   unlines ("Rewrite Rules" : "=============" : map (show . ppRule) (listRules ss))
  433   where
  434     ppRule r =
  435       PP.pretty '*' PP.<+>
  436       (PP.nest 2 $ PP.fillSep
  437        [ ppTerm (lhsRewriteRule r)
  438        , PP.pretty '=' PP.<+> ppTerm (rhsRewriteRule r) ])
  439     ppTerm t = SAWCorePP.ppTerm opts t
  440 
  441 -- | Pretty-print a 'Refnset' to a 'String'
  442 showRefnset :: PPS.Opts -> MRSolver.Refnset a -> String
  443 showRefnset opts ss =
  444   unlines ("Refinements" : "=============" : map (show . ppFunAssump)
  445                                                  (MRSolver.listFunAssumps ss))
  446   where
  447     ppFunAssump (MRSolver.FunAssump ctx f args rhs _) =
  448       PP.pretty '*' PP.<+>
  449       (PP.nest 2 $ PP.fillSep
  450        [ ppTermAppInCtx opts ctx (funNameTerm f) args
  451        , PP.pretty ("|=" :: String) PP.<+> ppFunAssumpRHS ctx rhs ])
  452     ppFunAssumpRHS ctx (OpaqueFunAssump f args) =
  453       ppTermAppInCtx opts ctx (funNameTerm f) args
  454     ppFunAssumpRHS ctx (RewriteFunAssump rhs) =
  455       SAWCorePP.ppTermInCtx opts (map fst $ mrVarCtxInnerToOuter ctx) rhs

  457 showsPrecValue :: PPS.Opts -> SAWNamingEnv -> Int -> Value -> ShowS
  458 showsPrecValue opts nenv p v =
  459   case v of
  460     VBool True -> showString "true"
  461     VBool False -> showString "false"
  462     VString s -> shows s
  463     VInteger n -> shows n
  464     VArray vs -> PPS.showBrackets $ PPS.showCommaSep $ map (showsPrecValue opts nenv 0) vs
  465     VTuple vs -> showParen True $ PPS.showCommaSep $ map (showsPrecValue opts nenv 0) vs
  466     VMaybe (Just v') -> showString "(Just " . showsPrecValue opts nenv 0 v' . showString ")"
  467     VMaybe Nothing -> showString "Nothing"
  468     VRecord m ->
  469       PPS.showBraces $ PPS.showCommaSep $ map showFld (M.toList m)
  470         where
  471           showFld (n, fv) =
  472             showString (unpack n) . showString "=" . showsPrecValue opts nenv 0 fv
  473 
  474     VLambda {} -> showString "<<function>>"
  475     VTerm t -> showString (SAWCorePP.showTermWithNames opts nenv (ttTerm t))
  476     VType sig -> showString (pretty sig)
  477     VReturn {} -> showString "<<monadic>>"
  478     VBind {} -> showString "<<monadic>>"
  479     VTopLevel {} -> showString "<<TopLevel>>"
  480     VSimpset ss -> showString (showSimpset opts ss)
  481     VRefnset ss -> showString (showRefnset opts ss)
  482     VProofScript {} -> showString "<<proof script>>"
  483     VTheorem thm ->
  484       showString "Theorem " .
  485       showParen True (showString (prettyProp opts nenv (thmProp thm)))
  486     VBisimTheorem _ -> showString "<<Bisimulation theorem>>"
  487     VLLVMCrucibleSetup{} -> showString "<<Crucible Setup>>"
  488     VLLVMCrucibleSetupValue{} -> showString "<<Crucible SetupValue>>"
  489     VLLVMCrucibleMethodSpec{} -> showString "<<Crucible MethodSpec>>"
  490     VLLVMModuleSkeleton s -> shows s
  491     VLLVMFunctionSkeleton s -> shows s
  492     VLLVMSkeletonState _ -> showString "<<Skeleton state>>"
  493     VLLVMFunctionProfile _ -> showString "<<Array sizes for function>>"
  494     VJavaType {} -> showString "<<Java type>>"
  495     VLLVMType t -> showString (show (Crucible.LLVM.ppType t))
  496     VMIRType t -> showString (show (PP.pretty t))
  497     VCryptolModule m -> showString (showCryptolModule m)
  498     VLLVMModule (Some m) -> showString (CMSLLVM.showLLVMModule m)
  499     VMIRModule m -> shows (PP.pretty (m^.rmCS^.collection))
  500     VMIRAdt adt -> shows (PP.pretty adt)
  501     VHeapsterEnv env -> showString (showHeapsterEnv env)
  502     VJavaClass c -> shows (prettyClass c)

  686 io :: IO a -> TopLevel a
  687 io f = (TopLevel_ (liftIO f))
  688        `catches`
  689        [ Handler (\(ex :: X86Unsupported) -> handleX86Unsupported ex)
  690        , Handler (\(ex :: X86Error) -> handleX86Error ex)
  691        , Handler handleTopLevel
  692        , Handler handleIO
  693        ]
  694   where
  695     rethrow :: X.Exception ex => ex -> TopLevel a
  696     rethrow ex =
  697       do stk <- getStackTrace
  698          throwM (SS.TraceException stk (X.SomeException ex))
  699 
  700     handleTopLevel :: SS.TopLevelException -> TopLevel a
  701     handleTopLevel e = rethrow e
  702 
  703     handleIO :: X.IOException -> TopLevel a
  704     handleIO e
  705       | IOError.isUserError e =
  706           do pos <- getPosition
  707              rethrow (SS.TopLevelException pos (dropEnd 1 . drop 12 $ show e))
  708       | otherwise = rethrow e
  709 
  710     handleX86Unsupported (X86Unsupported path s) =
  711       do let pos = SS.FileOnlyPos path
  712          rethrow (SS.TopLevelException pos ("Unsupported x86 feature: " ++ s))
  713 
  714     handleX86Error (X86Error path optfunc s) =
  715       do let pos = case optfunc of
  716                Nothing -> SS.FileOnlyPos path
  717                Just func -> SS.FileAndFunctionPos path func
  718          rethrow (SS.TopLevelException pos ("Error in x86 code: " ++ s))

  843 extendEnv ::
  844   SharedContext ->
  845   SS.LName -> SS.Schema -> Maybe String -> Value -> TopLevelRW -> IO TopLevelRW
  846 extendEnv sc x ty md v rw =
  847   do ce' <-
  848        case v of
  849          VTerm t ->
  850            pure $ CEnv.bindTypedTerm (ident, t) ce
  851          VType s ->
  852            pure $ CEnv.bindType (ident, s) ce
  853          VInteger n ->
  854            pure $ CEnv.bindInteger (ident, n) ce
  855          VCryptolModule m ->
  856            pure $ CEnv.bindCryptolModule (modname, m) ce
  857          VString s ->
  858            do tt <- typedTermOfString sc (unpack s)
  859               pure $ CEnv.bindTypedTerm (ident, tt) ce
  860          VBool b ->
  861            do tt <- typedTermOfBool sc b
  862               pure $ CEnv.bindTypedTerm (ident, tt) ce
  863          _ ->
  864            pure ce
  865      pure $
  866       rw { rwValueInfo  = M.insert name (SS.Current, ty, v) (rwValueInfo rw)
  867          , rwDocs    = maybeInsert (SS.getVal name) md (rwDocs rw)
  868          , rwCryptol = ce'
  869          }
  870   where
  871     name = x
  872     -- XXX why is this using getOrig?
  873     ident = T.mkIdent (SS.getOrig x)
  874     modname = T.packModName [SS.getOrig x]
  875     ce = rwCryptol rw

  877 typedTermOfString :: SharedContext -> String -> IO TypedTerm
  878 typedTermOfString sc str =
  879   do let schema = Cryptol.tMono (Cryptol.tString (length str))
  880      bvNat <- scGlobalDef sc "Prelude.bvNat"
  881      bvNat8 <- scApply sc bvNat =<< scNat sc 8
  882      byteT <- scBitvector sc 8
  883      let scChar c = scApply sc bvNat8 =<< scNat sc (fromIntegral (fromEnum c))
  884      ts <- traverse scChar str
  885      trm <- scVector sc byteT ts
  886      pure (TypedTerm (TypedTermSchema schema) trm)

  888 typedTermOfBool :: SharedContext -> Bool -> IO TypedTerm
  889 typedTermOfBool sc b =
  890   do let schema = Cryptol.tMono Cryptol.tBit
  891      trm <- scBool sc b
  892      pure (TypedTerm (TypedTermSchema schema) trm)
