   19 prettyClass :: Class -> Doc ann
   20 prettyClass cls = vcat $
   21   [ emptyDoc
   22   , "Class name:" <+> prettyClassName (className cls) <+>
   23     parens (commas attrs)
   24   , "Superclass:" <+> maybe "none" prettyClassName (superClass cls)
   25   , emptyDoc
   26   ] ++
   27   (if null (classInterfaces cls)
   28       then []
   29       else [ "Interfaces:"
   30            , indent 2 (vcat (map prettyClassName (classInterfaces cls)))
   31            , emptyDoc
   32            ]) ++
   33   [ "Fields:"
   34   , indent 2 (vcat (map prettyField (classFields cls)))
   35   , emptyDoc
   36   , "Methods:"
   37   , indent 2 (vcat (map prettyMethod (classMethods cls)))
   38   , emptyDoc
   39   ]
   40   where attrs = concat
   41           [ if classIsPublic cls then ["public"] else []
   42           , if classIsFinal cls then ["final"] else []
   43           , if classHasSuperAttribute cls then ["super"] else []
   44           , if classIsInterface cls then ["interface"] else []
   45           , if classIsAbstract cls then ["abstract"] else []
   46           ]
   47 
   48 -- FIXME: avoid unpacking via String
   49 prettyClassName :: ClassName -> Doc ann
   50 prettyClassName cname = pretty (unClassName cname)
   51 
   52 prettyField :: Field -> Doc ann
   53 prettyField f = hsep $
   54   [ viaShow (fieldVisibility f) ] ++
   55   attrs ++
   56   [ ppType (fieldType f)
   57   , pretty (fieldName f)
   58   ]
   59   where attrs = concat
   60           [ if fieldIsStatic f then ["static"] else []
   61           , if fieldIsFinal f then ["final"] else []
   62           , if fieldIsVolatile f then ["volatile"] else []
   63           , if fieldIsTransient f then ["transient"] else []
   64           ]
   65 
   66 prettyMethod :: Method -> Doc ann
   67 prettyMethod m =
   68   hsep $
   69   (if methodIsStatic m then ["static"] else []) ++
   70   [ maybe "void" ppType ret
   71   , pretty name
   72   , (parens . commas . map ppType) params
   73   ]
   74   where (MethodKey name params ret) = methodKey m
   75 
   76 commas :: [Doc ann] -> Doc ann
   77 commas = sep . punctuate comma
