  375       Just _ ->

  394 -- Push the input through the DFA, remembering the most recent accepting
  395 -- state it encountered.
  396 
  397 alex_scan_tkn user__ orig_input len input__ s last_acc =
  398   input__ `seq` -- strict in the input
  399   let
  400   new_acc = (check_accs (alex_accept `quickIndex` IBOX(s)))
  401   in
  402   new_acc `seq`
  403   case alexGetByte input__ of
  404      Nothing -> (new_acc, input__)
  405      Just (c, new_input) ->
  406 #ifdef ALEX_DEBUG
  407       Debug.Trace.trace ("State: " ++ show IBOX(s) ++ ", char: " ++ show c ++ " " ++ (show . chr . fromIntegral) c) $
  408 #endif
  409       case fromIntegral c of { IBOX(ord_c) ->
  410         let
  411                 base   = alexIndexInt32OffAddr alex_base s
  412                 offset = PLUS(base,ord_c)
  413 
  414                 new_s = if GTE(offset,ILIT(0))
  415                           && let check  = alexIndexInt16OffAddr alex_check offset
  416                              in  EQ(check,ord_c)
  417                           then alexIndexInt16OffAddr alex_table offset
  418                           else alexIndexInt16OffAddr alex_deflt s
  419         in
  420         case new_s of
  421             ILIT(-1) -> (new_acc, input__)
  422                 -- on an error, we want to keep the input *before* the
  423                 -- character that failed, not after.
  424             _ -> alex_scan_tkn user__ orig_input
  425 #ifdef ALEX_LATIN1
  426                    PLUS(len,ILIT(1))
  427                    -- issue 119: in the latin1 encoding, *each* byte is one character
  428 #else
  429                    (if c < 0x80 || c >= 0xC0 then PLUS(len,ILIT(1)) else len)
  430                    -- note that the length is increased ONLY if this is the 1st byte in a char encoding)
  431 #endif
  432                    new_input new_s new_acc
  433       }
  434   where
  435         check_accs (AlexAccNone) = last_acc
  436         check_accs (AlexAcc a  ) = AlexLastAcc a input__ IBOX(len)
  437         check_accs (AlexAccSkip) = AlexLastSkip  input__ IBOX(len)
  438 #ifndef ALEX_NOPRED
  439         check_accs (AlexAccPred a predx rest)
  440            | predx user__ orig_input IBOX(len) input__
  441            = AlexLastAcc a input__ IBOX(len)
  442            | otherwise
  443            = check_accs rest
  444         check_accs (AlexAccSkipPred predx rest)
  445            | predx user__ orig_input IBOX(len) input__
  446            = AlexLastSkip input__ IBOX(len)
  447            | otherwise
  448            = check_accs rest
  449 #endif

  543 -- feed alex a byte describing the current char
  544 -- this came from Cryptol's lexer, which came from LexerUtils, which
  545 -- adapted the technique used in GHC's lexer.
  546 --
  547 -- FUTURE: it would be nice to share this with the saw-script lexer
  548 -- (and maybe also the Cryptol lexer) instead of pasting it repeatedly
  549 byteForChar :: Char -> Word8
  550 byteForChar c
  551   | c <= '\7' = non_graphic
  552   | Char.isAscii c = fromIntegral (Char.ord c)
  553   | otherwise = case Char.generalCategory c of
  554       Char.LowercaseLetter       -> lower
  555       Char.OtherLetter           -> lower
  556       Char.UppercaseLetter       -> upper
  557       Char.TitlecaseLetter       -> upper
  558       Char.DecimalNumber         -> digit
  559       Char.OtherNumber           -> digit
  560       Char.ConnectorPunctuation  -> symbol
  561       Char.DashPunctuation       -> symbol
  562       Char.OtherPunctuation      -> symbol
  563       Char.MathSymbol            -> symbol
  564       Char.CurrencySymbol        -> symbol
  565       Char.ModifierSymbol        -> symbol
  566       Char.OtherSymbol           -> symbol
  567       Char.Space                 -> sp
  568       Char.ModifierLetter        -> other
  569       Char.NonSpacingMark        -> other
  570       Char.SpacingCombiningMark  -> other
  571       Char.EnclosingMark         -> other
  572       Char.LetterNumber          -> other
  573       Char.OpenPunctuation       -> other
  574       Char.ClosePunctuation      -> other
  575       Char.InitialQuote          -> other
  576       Char.FinalQuote            -> tick
  577       _                          -> non_graphic
  578   where
  579   -- CAUTION: these must match the $uni* values at the top of the file
  580   non_graphic     = 0
  581   upper           = 1
  582   lower           = 2
  583   digit           = 3
  584   symbol          = 4
  585   sp              = 5
  586   other           = 6
  587   tick            = 7

  589 -- Get the next byte to feed alex, where the byte is a character class
  590 -- describing the current character as generated above. Update the
  591 -- input state accordingly.
  592 alexGetByte :: AlexInput -> Maybe (Word8, AlexInput)
  593 alexGetByte (PosPair pos (Buffer _ txt)) = fmap fn (LText.uncons txt)
  594   where fn (c, txt') = (byte, PosPair pos' (Buffer c txt'))
  595           where byte  = byteForChar c
  596                 isNew = c == '\n'
  597                 pos'  = if isNew then incLine pos else incCol pos

  600 type MsgList = [(Pos, LexerMessage)]

  602 -- Get the next token.
  603 --
  604 -- Recurses until we have one. If we get AlexError, concatenate all the
  605 -- adjacent invalid characters into a single error message.
  606 --
  607 -- Returns the updated input state, a list of messages, and the token.
  608 scanToken :: AlexInput -> MsgList -> (AlexInput, MsgList, PosPair Token)
  609 scanToken inp0 errors0 =
  610   let go inp errors pendingError =
  611         let (PosPair p (Buffer _ txt)) = inp
  612             finishAnyError = case pendingError of
  613                 Nothing -> errors
  614                 Just (pos, chars) ->
  615                     let badText = Text.pack $ reverse chars in
  616                     (pos, InvalidInput badText) : errors
  617             end =
  618                 (inp, finishAnyError, PosPair p TEnd)
  619         in case alexScan inp 0 of
  620             AlexEOF ->
  621                 end
  622             AlexError _ -> case alexGetByte inp of
  623               Nothing ->
  624                   end
  625               Just (_byte, inp') ->
  626                 -- ignore the classification byte; get the actual failing char
  627                 let badChar = alexInputPrevChar inp' in
  628                 case pendingError of
  629                     Nothing ->
  630                         go inp' errors (Just (p, [badChar]))
  631                     Just (pos, badChars) ->
  632                         go inp' errors (Just (pos, badChar : badChars))
  633             AlexSkip inp' _ ->
  634                 go inp' finishAnyError Nothing
  635             AlexToken inp' l act ->
  636                 let v = act (LText.unpack (LText.take (fromIntegral l) txt)) in
  637                 (inp', finishAnyError, PosPair p v)
  638   in
  639   go inp0 errors0 Nothing

  641 -- | State for processing comments.
  642 --
  643 -- CNone is the ground state (not in a comment)
  644 --
  645 -- CBlock startpos n is when we're within a block comment starting at
  646 -- startpos, and n is the number of additional nested block comments
  647 -- that have been opened.
  648 --
  649 -- CLine is when we're in a line (-- type) comment.
  650 data CommentState = CNone | CBlock Pos !Int | CLine
  651 
  652 -- Get the next token, dropping comments.
  653 --
  654 -- Recurses until we get one.
  655 --
  656 -- Returns the updated input state, a list of messages, and the token.
  657 scanSkipComments :: AlexInput -> (AlexInput, MsgList, PosPair Token)
  658 scanSkipComments inp0 =
  659   let go state inp errors =
  660         let (inp', errors', tok) = scanToken inp errors
  661             again nextState = go nextState inp' errors'
  662             againWith nextState err = go nextState inp' (err : errors')
  663             accept = (inp', errors', tok)
  664             acceptWith err = (inp', err : errors', tok)
  665         in
  666         case state of
  667           CNone -> case val tok of
  668             TEnd ->
  669                 -- plain EOF
  670                 accept
  671             TCommentS ->
  672                 -- open block-comment
  673                 again (CBlock (pos tok) 0)
  674             TCommentE ->
  675                 -- misplaced close-comment
  676                 let err = (pos tok, InvalidInput "-}") in
  677                 againWith CNone err
  678             TCommentL ->
  679                 -- begin line-comment
  680                 again CLine
  681             TEOL ->
  682                 -- ordinary end-of-line, doesn't need to be seen downstream
  683                 again CNone
  684             _ ->
  685                 -- uncommented token, take it
  686                 accept
  687 
  688           CBlock startpos depth -> case val tok of
  689             TEnd ->
  690                 -- EOF in block comment
  691                 let err = (startpos, UnclosedComment) in
  692                 acceptWith err
  693             TCommentS ->
  694                 -- nested open-comment; keep the original start position
  695                 again (CBlock startpos (depth + 1))
  696             TCommentE
  697              | depth == 0 ->
  698                 -- end outer block comment
  699                 again CNone
  700              | otherwise ->
  701                 -- end nested block comment
  702                 again (CBlock startpos (depth - 1))
  703             _ ->
  704                 -- anything else in a block comment; get another token
  705                 again state
  706 
  707           CLine -> case val tok of
  708             TEnd ->
  709                 -- EOF in line-comment; missing a newline, which is a warning
  710                 let err = (pos tok, MissingEOL) in
  711                 acceptWith err
  712             TEOL ->
  713                 -- EOL ending the comment; drop it but end the comment
  714                 again CNone
  715             _ ->
  716                 -- anything else in a line comment; drop it
  717                 again CLine
  718   in
  719   let (inp0', errors, tok) = go CNone inp0 [] in
  720   (inp0', reverse errors, tok)

  724 lexSAWCore :: AlexInput -> (AlexInput, MsgList, PosPair Token)
