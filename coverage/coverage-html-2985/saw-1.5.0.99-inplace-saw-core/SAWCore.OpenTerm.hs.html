    8 Copyright   : Galois, Inc. 2018

   10 Stability   : experimental
   11 Portability : non-portable (language extensions)
   12 
   13 This module defines a convenient DSL for building terms in a pure way,
   14 where sub-terms can be freely composed and combined into super-terms
   15 without monadic sequencing or 'IO' computations; the 'IO' computation
   16 is only run at the top level when all the term-building is complete.
   17 
   18 Users of this interface can also build binding constructs like lambda-
   19 and pi-abstractions using higher-order abstract syntax, meaning that
   20 the bodies of these term constructs are specified as Haskell functions
   21 that take in terms for the bound variables.
   22 
   23 Since many function names are short and may clash with names from
   24 other modules, this module is intended to be imported @qualified@,
   25 e.g.
   26 
   27 >  import SAWCore.OpenTerm (OpenTerm)
   28 >  import qualified SAWCore.OpenTerm as OpenTerm
   29 
   30 To use the 'OpenTerm' API, the caller builds up 'OpenTerm's using a
   31 variety of combinators that mirror the SAW core 'Term' structure.
   32 As some useful examples of 'OpenTerm' operations, 'apply' applies one
   33 'OpenTerm' to another, 'global' builds an 'OpenTerm' for a global
   34 identifier, and 'lambda' builds a lambda-abstraction. For instance,
   35 the SAW core term
   36 
   37 > \ (f : Bool -> Bool) (x : Bool) -> f x
   38 
   39 can be built with the 'OpenTerm' expression
   40 
   41 > let boolTy = global "Prelude.Bool" in
   42 > lambda "f" (arrow boolTy boolTy) $ \f ->
   43 > lambda "x" boolTy $ \x ->
   44 > apply f x
   45 
   46 Existing SAW core 'Term's can be used in 'OpenTerm' by applying 'term'.
   47 At the top level, 'complete' then "completes" an 'OpenTerm' by running
   48 its underlying 'IO' computation to build and type-check the resulting
   49 SAW core 'Term'.
   50 -}

   84 -- | An open term is represented as a type-checking computation that computes a
   85 -- SAW core term.
   86 newtype OpenTerm = OpenTerm { unOpenTerm :: SharedContext -> IO Term }
   87 
   88 -- | \"Complete\" an 'OpenTerm' to a 'Term' or 'fail' on type-checking
   89 -- error.
   90 complete :: SharedContext -> OpenTerm -> IO Term
   91 complete sc (OpenTerm m) = m sc
   92 
   93 -- | \"Complete\" an 'OpenTerm' to a term for its type.
   94 completeType :: SharedContext -> OpenTerm -> IO Term
   95 completeType sc ot = scTypeOf sc =<< complete sc ot
   96 
   97 -- | Embed a 'Term' into an 'OpenTerm'.
   98 term :: Term -> OpenTerm
   99 term t = OpenTerm $ const (pure t)
  100 
  101 -- | Return type of an 'OpenTerm' as an 'OpenTerm'.
  102 typeOf :: OpenTerm -> OpenTerm
  103 typeOf (OpenTerm m) =
  104   OpenTerm $ \sc -> m sc >>= scTypeOf sc
  105 
  106 -- | Build an 'OpenTerm' from a 'FlatTermF'
  107 flat :: FlatTermF OpenTerm -> OpenTerm
  108 flat ftf =
  109   OpenTerm $ \sc -> traverse (complete sc) ftf >>= scFlatTermF sc
  110 
  111 -- | Build an 'OpenTerm' for a sort
  112 sort :: Sort -> OpenTerm
  113 sort s = flat (Sort s noFlags)
  114 
  115 -- | Build an 'OpenTerm' for a natural number literal
  116 nat :: Natural -> OpenTerm
  117 nat n = OpenTerm $ \sc -> scNat sc n
  118 
  119 -- | The 'OpenTerm' for the unit value
  120 unit :: OpenTerm
  121 unit = flat UnitValue
  122 
  123 -- | The 'OpenTerm' for the unit type
  124 unitType :: OpenTerm
  125 unitType = flat UnitType
  126 
  127 -- | Build a SAW core string literal.
  128 stringLit :: Text -> OpenTerm
  129 stringLit = flat . StringLit
  130 
  131 -- | Return the SAW core type @String@ of strings.
  132 stringType :: OpenTerm
  133 stringType = global "Prelude.String"
  134 
  135 -- | The 'True' value as a SAW core term
  136 true :: OpenTerm
  137 true = global "Prelude.True"
  138 
  139 -- | The 'False' value as a SAW core term
  140 false :: OpenTerm
  141 false = global "Prelude.False"
  142 
  143 -- | Convert a 'Bool' to a SAW core term
  144 bool :: Bool -> OpenTerm
  145 bool True = global "Prelude.True"
  146 bool False = global "Prelude.False"
  147 
  148 -- | The 'Bool' type as a SAW core term
  149 boolType :: OpenTerm
  150 boolType = global "Prelude.Bool"
  151 
  152 -- | Build an 'OpenTerm' for an array literal
  153 arrayValue :: OpenTerm -> [OpenTerm] -> OpenTerm
  154 arrayValue tp elems =
  155   flat $ ArrayValue tp $ V.fromList elems
  156 
  157 -- | Create a SAW core term for a bitvector literal
  158 bvLit :: [Bool] -> OpenTerm
  159 bvLit bits =
  160   arrayValue boolType $ map bool bits
  161 
  162 -- | Create a SAW core term for a vector type
  163 vectorType :: OpenTerm -> OpenTerm -> OpenTerm
  164 vectorType n a = applyGlobal "Prelude.Vec" [n,a]
  165 
  166 -- | Create a SAW core term for the type of a bitvector
  167 bvType :: Integral a => a -> OpenTerm
  168 bvType n =
  169   apply (global "Prelude.Vec")
  170   [nat (fromIntegral n), boolType]
  171 
  172 -- | Build an 'OpenTerm' for a pair
  173 pair :: OpenTerm -> OpenTerm -> OpenTerm
  174 pair t1 t2 = flat $ PairValue t1 t2
  175 
  176 -- | Build an 'OpenTerm' for a pair type
  177 pairType :: OpenTerm -> OpenTerm -> OpenTerm
  178 pairType t1 t2 = flat $ PairType t1 t2
  179 
  180 -- | Build an 'OpenTerm' for the left projection of a pair
  181 pairLeft :: OpenTerm -> OpenTerm
  182 pairLeft t = flat $ PairLeft t
  183 
  184 -- | Build an 'OpenTerm' for the right projection of a pair
  185 pairRight :: OpenTerm -> OpenTerm
  186 pairRight t = flat $ PairRight t
  187 
  188 -- | Build a right-nested tuple as an 'OpenTerm'
  189 tuple :: [OpenTerm] -> OpenTerm
  190 tuple = foldr pair unit
  191 
  192 -- | Build a right-nested tuple type as an 'OpenTerm'
  193 tupleType :: [OpenTerm] -> OpenTerm
  194 tupleType = foldr pairType unitType
  195 
  196 -- | Project the @n@th element of a right-nested tuple type
  197 projTuple :: Integer -> OpenTerm -> OpenTerm
  198 projTuple 0 t = pairLeft t
  199 projTuple i t = projTuple (i-1) (pairRight t)
  200 
  201 -- | Build a right-nested tuple as an 'OpenTerm' but without adding a final unit
  202 -- as the right-most element
  203 tuple' :: [OpenTerm] -> OpenTerm
  204 tuple' [] = unit
  205 tuple' ts = foldr1 pair ts
  206 
  207 -- | Build a right-nested tuple type as an 'OpenTerm' but without adding a final
  208 -- unit type as the right-most element
  209 tupleType' :: [OpenTerm] -> OpenTerm
  210 tupleType' [] = unitType
  211 tupleType' ts = foldr1 pairType ts
  212 
  213 -- | Project the @i@th element from a term of a right-nested tuple term that
  214 -- does not have a final unit type as the right-most type. The first argument is
  215 -- the number of types used to make the tuple type and the second is the index.
  216 projTuple' :: Natural -> Natural -> OpenTerm -> OpenTerm
  217 projTuple' 0 _ _ =
  218   panic "projTupleOpenTerm'" ["Projection of empty tuple!"]
  219 projTuple' 1 0 tup = tup
  220 projTuple' _ 0 tup = pairLeft tup
  221 projTuple' len i tup =
  222   projTuple' (len-1) (i-1) $ pairRight tup
  223 
  224 -- | Build a record value as an 'OpenTerm'
  225 record :: [(FieldName, OpenTerm)] -> OpenTerm
  226 record fields =
  227   OpenTerm $ \sc ->
  228   do fields' <- traverse (traverse (complete sc)) fields
  229      scRecordValue sc fields'
  230 
  231 -- | Build a record type as an 'OpenTerm'
  232 recordType :: [(FieldName, OpenTerm)] -> OpenTerm
  233 recordType fields =
  234   OpenTerm $ \sc ->
  235   do fields' <- traverse (traverse (complete sc)) fields
  236      scRecordType sc fields'
  237 
  238 -- | Project a field from a record
  239 projRecord :: OpenTerm -> FieldName -> OpenTerm
  240 projRecord (OpenTerm m) f =
  241   OpenTerm $ \sc ->
  242   do t <- m sc
  243      scRecordSelect sc t f
  244 
  245 -- | Build an 'OpenTerm' for a global name with a definition
  246 global :: Ident -> OpenTerm
  247 global ident =
  248   OpenTerm $ \sc -> scGlobalDef sc ident
  249 
  250 -- | Build an 'OpenTerm' for a named variable.
  251 variable :: VarName -> Term -> OpenTerm
  252 variable x t =
  253   OpenTerm $ \sc -> scVariable sc x t
  254 
  255 -- | Apply an 'OpenTerm' to another.
  256 app :: OpenTerm -> OpenTerm -> OpenTerm
  257 app (OpenTerm f) (OpenTerm arg) =
  258   OpenTerm $ \sc ->
  259   do t1 <- f sc
  260      t2 <- arg sc
  261      scApply sc t1 t2
  262 
  263 -- | Apply an 'OpenTerm' to a list of zero or more arguments.
  264 apply :: OpenTerm -> [OpenTerm] -> OpenTerm
  265 apply = foldl app
  266 
  267 -- | Apply a named global to a list of zero or more arguments.
  268 applyGlobal :: Ident -> [OpenTerm] -> OpenTerm
  269 applyGlobal ident = apply (global ident)
  270 
  271 -- | Build a lambda abstraction as an 'OpenTerm'
  272 lambda :: LocalName -> OpenTerm -> (OpenTerm -> OpenTerm) -> OpenTerm
  273 lambda x (OpenTerm tpM) body_f =
  274   OpenTerm $ \sc ->
  275   do tp <- tpM sc
  276      vn <- scFreshVarName sc x
  277      var <- scVariable sc vn tp
  278      body <- complete sc (body_f (OpenTerm (const (pure var))))
  279      scLambda sc vn tp body
  280 
  281 -- | Build a nested sequence of lambda abstractions as an 'OpenTerm'
  282 lambdas :: [(LocalName, OpenTerm)] -> ([OpenTerm] -> OpenTerm) ->
  283                        OpenTerm
  284 lambdas xs_tps body_f =
  285   foldr (\(x,tp) rest_f xs ->
  286           lambda x tp (rest_f . (:xs))) (body_f . reverse) xs_tps []
  287 
  288 -- | Build a Pi abstraction as an 'OpenTerm'
  289 piType :: LocalName -> OpenTerm -> (OpenTerm -> OpenTerm) -> OpenTerm
  290 piType x (OpenTerm tpM) body_f =
  291   OpenTerm $ \sc ->
  292   do tp <- tpM sc
  293      nm <- scFreshVarName sc x
  294      var <- scVariable sc nm tp
  295      body <- complete sc (body_f (OpenTerm (const (pure var))))
  296      scPi sc nm tp body
  297 
  298 -- | Build a non-dependent function type.
  299 arrow :: OpenTerm -> OpenTerm -> OpenTerm
  300 arrow t1 t2 =
  301   OpenTerm $ \sc ->
  302   do t1' <- complete sc t1
  303      t2' <- complete sc t2
  304      scFun sc t1' t2'
  305 
  306 -- | Build a nested sequence of Pi abstractions as an 'OpenTerm'
  307 piMulti :: [(LocalName, OpenTerm)] -> ([OpenTerm] -> OpenTerm) ->
  308                    OpenTerm
  309 piMulti xs_tps body_f =
  310   foldr (\(x,tp) rest_f xs ->
  311           piType x tp (rest_f . (:xs))) (body_f . reverse) xs_tps []
  312 
  313 -- | Build a let expression as an 'OpenTerm'. This is equivalent to
  314 -- > 'apply' ('lambda' x tp body) rhs
  315 mkLet ::
  316   LocalName -> OpenTerm -> OpenTerm -> (OpenTerm -> OpenTerm) -> OpenTerm
  317 mkLet x tp rhs body_f = app (lambda x tp body_f) rhs
  318 
  319 -- | Build a let expression as an 'OpenTerm'. This is equivalent to
  320 -- > 'apply' ('lambda' x tp body) rhs
  321 sawLet ::
  322   LocalName -> OpenTerm -> OpenTerm -> OpenTerm ->
  323   (OpenTerm -> OpenTerm) -> OpenTerm
  324 sawLet x tp tp_ret rhs body_f =
  325   apply (global "Prelude.sawLet")
  326   [tp, tp_ret, rhs, lambda x tp body_f]
  327 
  328 -- | Build a bitvector type with the given length
  329 bitvectorType :: OpenTerm -> OpenTerm
  330 bitvectorType w =
  331   applyGlobal "Prelude.Vec" [w, global "Prelude.Bool"]
  332 
  333 -- | Build a SAW core term for a list with the given element type
  334 list :: OpenTerm -> [OpenTerm] -> OpenTerm
  335 list tp elems =
  336   foldr (\x l -> applyGlobal "Prelude.Cons" [tp, x, l])
  337   (applyGlobal "Prelude.Nil" [tp]) elems
  338 
  339 -- | Build the type @Either a b@ from types @a@ and @b@
  340 eitherType :: OpenTerm -> OpenTerm -> OpenTerm
  341 eitherType a b = applyGlobal "Prelude.Either" [a, b]
