   49 -- | Get visible Cryptol variable names.
   50 getCryptolExprNames :: REPL [Text]
   51 getCryptolExprNames =
   52   do fNames <- fmap getNamingEnv getCryptolEnv
   53      let keys = Map.keys (MN.namespaceMap NSValue fNames)
   54      return (map (Text.pack . show . pp) keys)
   55 
   56 -- | Get visible Cryptol type names.
   57 getCryptolTypeNames :: REPL [Text]
   58 getCryptolTypeNames =
   59   do fNames <- fmap getNamingEnv getCryptolEnv
   60      let keys = Map.keys (MN.namespaceMap NSType fNames)
   61      return (map (Text.pack . show . pp) keys)

   63 -- | Get visible variable names for Haskeline completion.
   64 getSAWScriptValueNames :: REPL [Text]
   65 getSAWScriptValueNames = do
   66   rw <- getTopLevelRW
   67   let avail = rwPrimsAvail rw
   68       visible (_, lc, _, _, _) = Set.member lc avail
   69       Environ valenv _tyenv _cryenv = rwEnviron rw
   70       rbenv = rwRebindables rw
   71   let rnames1 = ScopedMap.allKeys $ ScopedMap.filter visible valenv
   72       rnames2 = Map.keys rbenv
   73   return (rnames1 ++ rnames2)
   74 
   75 -- | Get visible type names for Haskeline completion.

   88 -- | Get names and printed types from the SAWScript variable
   89 --   environment.
   90 --
   91 --   Returns the rebindable environment and a list of scopes, with the
   92 --   innermost scope last. (That's the normal print order.)
   93 --
   94 getSAWScriptVarEnv :: REPL (VarEnvList, [VarEnvList])
   95 getSAWScriptVarEnv = do
   96   rw <- getTopLevelRW
   97   let Environ varenv _tyenv _cryenv = rwEnviron rw
   98       rbenv = rwRebindables rw
   99       avail = rwPrimsAvail rw
  100 
  101   -- Get the rebindable globals. These are always visible.
  102   let extractRB (x, (_pos, ty, _v)) = (x, ty)
  103       rebindables = map extractRB $ Map.assocs rbenv
  104 
  105   -- Get the scopes.
  106   --
  107   -- Reverse the list so the innermost is last, because that's what
  108   -- people will expect to see when it's printed.
  109   --
  110   let extractVar (x, (_pos, lc, ty, _v, _doc)) =
  111         if Set.member lc avail then Just (x, ty)
  112         else Nothing
  113       extractScope entries = mapMaybe extractVar entries
  114       scopes = map extractScope $ reverse $ ScopedMap.scopedAssocs varenv
  115 
  116   return (rebindables, scopes)
  117 
  118 -- Type shorthand for list of pairs of names and type expansions.
  119 type TypeEnvList = [(Text, AST.NamedType)]
  120 
  121 -- | Get names and printed types from the SAWScript type environment.
  122 --
  123 --   Returns a list of scopes, with the innermost scope last. (That's
  124 --   the normal print order.)
  125 --
  126 getSAWScriptTypeEnv :: REPL [TypeEnvList]
  127 getSAWScriptTypeEnv = do
  128   rw <- getTopLevelRW
  129   let Environ _varenv tyenv _cryenv = rwEnviron rw
  130       avail = rwPrimsAvail rw
  131 
  132   -- Get the scopes.
  133   --
  134   -- Reverse the list so the innermost is last, because that's what
  135   -- people will expect to see when it's printed.
  136   --
  137   let extractTyVar (x, (lc, ty)) =
  138         if Set.member lc avail then Just (x, ty)
  139         else Nothing
  140       extractScope entries = mapMaybe extractTyVar entries
  141       scopes = map extractScope $ reverse $ ScopedMap.scopedAssocs tyenv
  142 
  143   return scopes
