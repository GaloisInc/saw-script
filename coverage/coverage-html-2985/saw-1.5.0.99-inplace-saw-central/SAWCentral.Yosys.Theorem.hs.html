   43 data YosysTheorem = YosysTheorem
   44   { _theoremURI :: URI.URI -- ^ URI identifying overridden module
   45   , _theoremInputCryptolType :: C.Type -- ^ cryptol type of r
   46   , _theoremOutputCryptolType :: C.Type -- ^ cryptol type of (module r)
   47   , _theoremInputType :: SC.Term -- ^ type of r
   48   , _theoremOutputType :: SC.Term -- ^ type of (module r)
   49   , _theoremModule :: SC.Term -- ^ {{ \r -> module r }}
   50   , _theoremPrecond :: Maybe SC.Term -- ^ {{ \r -> precond r }}
   51   , _theoremBody :: SC.Term -- ^ {{ \r -> other r }}
   52   }

   55 -- | Construct a SAWCore proposition for the given theorem.
   56 -- In pseudo-Cryptol, this looks like {{ \r -> precond r ==> (module r == body r) }}
   57 theoremProp ::
   58   SC.SharedContext ->
   59   YosysTheorem ->
   60   IO SC.TypedTerm
   61 theoremProp sc thm =
   62   do r <- SC.scFreshVariable sc "r" $ thm ^. theoremInputType
   63      modr <- SC.scApply sc (thm ^. theoremModule) r
   64      bodyr <- SC.scApply sc (thm ^. theoremBody) r
   65      equality <- eqBvRecords sc (thm ^. theoremOutputCryptolType) modr bodyr
   66      res <-
   67        case thm ^. theoremPrecond of
   68          Nothing -> pure equality
   69          Just pc ->
   70            do pcr <- SC.scApply sc pc r
   71               SC.scImplies sc pcr equality
   72      func <- SC.scAbstractTerms sc [r] res
   73      let cty = C.tFun (thm ^. theoremInputCryptolType) C.tBit
   74      pure (SC.TypedTerm (SC.TypedTermSchema (C.tMono cty)) func)

   76 -- | Construct a SAWCore proposition for the given theorem.
   77 -- In pseudo-Cryptol, this looks like {{ \r -> if precond r then body r else module r }}
   78 theoremReplacement ::
   79   SC.SharedContext ->
   80   YosysTheorem ->
   81   IO SC.Term
   82 theoremReplacement sc thm =
   83   do r <- SC.scFreshVariable sc "r" $ thm ^. theoremInputType
   84      body <-
   85        case thm ^. theoremPrecond of
   86           Nothing -> SC.scApply sc (thm ^. theoremBody) r
   87           Just pc ->
   88             do precond <- SC.scApply sc pc r
   89                thenCase <- SC.scApply sc (thm ^. theoremBody) r
   90                elseCase <- SC.scApply sc (thm ^. theoremModule) r
   91                SC.scIte sc (thm ^. theoremOutputType) precond thenCase elseCase
   92      SC.scAbstractTerms sc [r] body

   94 -- | Given a SAWCore term corresponding to an HDL module, a specification, and a precondition:
   95 -- Construct a theorem summarizing the relationship between the module and the specification.
   96 buildTheorem ::
   97   SC.SharedContext ->
   98   SC.TypedTerm ->
   99   SC.Term ->
  100   Maybe SC.TypedTerm ->
  101   SC.TypedTerm ->
  102   IO YosysTheorem
  103 buildTheorem sc ymod newmod precond body = do
  104   cty <-
  105     case SC.ttType ymod of
  106       SC.TypedTermSchema (C.Forall [] [] cty) -> pure cty
  107       _ -> throw YosysErrorInvalidOverrideTarget
  108   (cinpTy, coutTy) <-
  109     case cty of
  110       C.TCon (C.TC C.TCFun) [ci, co] -> pure (ci, co)
  111       _ -> throw YosysErrorInvalidOverrideTarget
  112   inpTy <- CSC.importType sc CSC.emptyEnv cinpTy
  113   outTy <- CSC.importType sc CSC.emptyEnv coutTy
  114   nmi <-
  115     case reduceSelectors (SC.ttTerm ymod) of
  116       (R.asConstant -> Just (SC.Name _ nmi)) -> pure nmi
  117       _ -> throw YosysErrorInvalidOverrideTarget
  118   uri <-
  119     case nmi of
  120       SC.ImportedName uri _ -> pure uri
  121       _ -> throw YosysErrorInvalidOverrideTarget
  122   pure YosysTheorem
  123     { _theoremURI = uri
  124     , _theoremInputCryptolType = cinpTy
  125     , _theoremOutputCryptolType = coutTy
  126     , _theoremInputType = inpTy
  127     , _theoremOutputType = outTy
  128     , _theoremModule = newmod
  129     , _theoremPrecond = SC.ttTerm <$> precond
  130     , _theoremBody = SC.ttTerm body
  131     }

  133 -- | Reduce nested tuple and record selectors at the top-level of the
  134 -- given term, if possible.
  135 reduceSelectors :: SC.Term -> SC.Term
  136 reduceSelectors t =
  137   case t of
  138     (R.asPairSelector -> Just (t1, b)) ->
  139       case R.asPairValue (reduceSelectors t1) of
  140         Nothing -> t
  141         Just (x, y) -> if b then y else x
  142     (R.asRecordSelector -> Just (t1, fname)) ->
  143       case R.asRecordValue (reduceSelectors t1) of
  144         Nothing -> t
  145         Just tm ->
  146           case lookup fname tm of
  147             Nothing -> t
  148             Just t' -> t'
  149     _ -> t

  151 -- | Applying a theorem thm as an "override" in a Yosys-derived term t proceeds as follows:
  152 --  1) unfold all names except thm.theoremURI in t
  153 --  2) traverse t, looking for constants named thm.theoremURI
  154 --  4) replace the constant term with either thm.theoremBody, or
  155 --     {{ \r -> if thm.theoremPrecond r then thm.theoremBody r else thm.theoremURI r }}
  156 --     in the presence of a precondition
  157 applyOverride ::
  158   SC.SharedContext ->
  159   YosysTheorem ->
  160   SC.Term ->
  161   IO SC.Term
  162 applyOverride sc thm t = do
  163   tidx <-
  164     do result <- SC.scResolveNameByURI sc $ thm ^. theoremURI
  165        case result of
  166          Nothing -> throw . YosysErrorOverrideNameNotFound . URI.render $ thm ^. theoremURI
  167          Just i -> pure i
  168   -- unfold everything except for theoremURI and prelude constants
  169   let isPreludeName (SC.ModuleIdentifier ident) = SC.identModule ident == SC.preludeModuleName
  170       isPreludeName _ = False
  171   let unfold nm = SC.nameIndex nm /= tidx && not (isPreludeName (SC.nameInfo nm))
  172   unfolded <- SC.scUnfoldConstants sc unfold t
  173   cache <- SC.newIntCache
  174   let
  175     go :: SC.Term -> IO SC.Term
  176     go s =
  177       SC.useIntCache cache (SC.termIndex s) $
  178       case SC.unwrapTermF s of
  179         SC.Constant (SC.Name idx _)
  180           | idx == tidx -> theoremReplacement sc thm
  181           | otherwise -> pure s
  182         _ -> SC.scTermF sc =<< traverse go (SC.unwrapTermF s)
  183   go unfolded
