    1 {- |
    2 Module      : SAWSupport.IntRangeSet
    3 Description : A type representing sets of ranges of integers
    4 License     : BSD3
    5 Maintainer  : saw@galois.com
    6 Stability   : provisional
    7 
    8 An 'IntRangeSet' represents a set of 'Int's, much like an @IntSet@,
    9 but it also supports insertion and deletion of whole contiguous ranges
   10 of 'Int's at once.
   11 
   12 Ranges are merged so that every set of 'Int's has a canonical
   13 representation as an 'IntRangeSet'.
   14 
   15 The complexity of operations scales not with the number of 'Int's in a
   16 set, but rather the number of separate contiguous ranges of 'Int's.
   17 -}
   18 
   19 module SAWSupport.IntRangeSet where
   20 
   21 import Data.IntMap.Strict (IntMap)
   22 import qualified Data.IntMap.Strict as IntMap
   23 
   24 -- | An 'IntRangeSet' represents a set of disjoint contiguous
   25 -- intervals of type 'Int'.
   26 -- Each interval @(i, j)@ with @i <= j@ represents the set of values
   27 -- @[i..j]@.
   28 newtype IntRangeSet = IntRangeSet (IntMap Int)
   29   deriving Eq
   30 
   31 instance Show IntRangeSet where
   32   show (IntRangeSet m) = show m
   33 
   34 -- | The empty set.
   35 empty :: IntRangeSet
   36 empty = IntRangeSet IntMap.empty
   37 
   38 -- | A set containing a single interval @[i..j]@.
   39 singleton :: (Int, Int) -> IntRangeSet
   40 singleton (i, j)
   41   | i <= j = IntRangeSet (IntMap.singleton i j)
   42   | otherwise = empty
   43 
   44 -- | Add an interval @[i..j]@ to the set.
   45 insert :: (Int, Int) -> IntRangeSet -> IntRangeSet
   46 insert (i, j) (IntRangeSet m)
   47   | i > j = IntRangeSet m
   48   | otherwise =
   49     let -- m1 contains all intervals strictly below i
   50         -- also possibly one interval starting at i'
   51         m1 = IntMap.takeWhileAntitone (< i) m
   52         -- m2 contains all intervals strictly above j
   53         m2 = IntMap.dropWhileAntitone (<= j+1) m
   54         i' = case IntMap.lookupLE i m of
   55                Nothing -> i
   56                Just (u, v) -> if i <= v+1 then u else i
   57         j' = case IntMap.lookupLE (j+1) m of
   58                Nothing -> j
   59                Just (_, v) -> max v j
   60     in IntRangeSet (IntMap.insert i' j' (IntMap.union m1 m2))
   61     -- Any interval [u..v] with v+1 < i stays unchanged.
   62     -- Any interval [u..v] with u <= i <= v+1 is overwritten.
   63     -- Any interval [u..v] with i < u <= j is removed.
   64     -- Any interval [u..v] with j < u stays unchanged.
   65     -- If i falls inside (or is adjacent to the upper bound of) an
   66     -- interval [u..v], set i' = u; else i' = i.
   67     -- If j falls inside (or is adjacent to the lower bound of) an
   68     -- interval [u..v], let j' = v; else j' = j.
   69     -- Always insert a new interval [i'..j'].
   70 
   71 -- | Remove an interval @[i..j]@ from the set.
   72 delete :: (Int, Int) -> IntRangeSet -> IntRangeSet
   73 delete (i, j) (IntRangeSet m)
   74   | i > j = IntRangeSet m
   75   | otherwise =
   76     let m1 = IntMap.takeWhileAntitone (< i) m
   77         m2 = IntMap.dropWhileAntitone (<= j) m
   78         m1' = case IntMap.lookupLT i m1 of
   79                 Nothing -> m1
   80                 Just (u, v) -> if i <= v then IntMap.insert u (i-1) m1 else m1
   81         m2' = case IntMap.lookupLE j m of
   82                 Nothing -> m2
   83                 Just (_, v) -> if j < v then IntMap.insert (j+1) v m2 else m2
   84     in IntRangeSet (IntMap.union m1' m2')
   85     -- Any interval [u..v] with v < i stays unchanged.
   86     -- Any interval [u..v] with u < i <= v is overwritten.
   87     -- Any interval [u..v] with i <= u <= j is removed.
   88     -- Any interval [u..v] with j < u stays unchanged.
   89     -- If i falls inside an interval [u..v], replace it with interval [u..i-1].
   90     -- If j falls inside an interval [u..v], insert a new interval [j+1..v].
   91 
   92 -- | Is a specific value in the set?
   93 member :: Int -> IntRangeSet -> Bool
   94 member x (IntRangeSet m) =
   95   case IntMap.lookupLE x m of
   96     Nothing -> False
   97     Just (_, j) -> x <= j
   98 
   99 -- | The union of two sets.
  100 union :: IntRangeSet -> IntRangeSet -> IntRangeSet
  101 union s1 s2 = foldr insert s1 (toList s2)
  102 
  103 -- | The difference of two sets: Remove all members of the second set
  104 -- from the first set.
  105 difference :: IntRangeSet -> IntRangeSet -> IntRangeSet
  106 difference s1 s2 = foldr delete s1 (toList s2)
  107 
  108 -- | Create a set from a list of intervals.
  109 fromList :: [(Int, Int)] -> IntRangeSet
  110 fromList = foldr insert empty
  111 
  112 -- | Convert the set to a list of disjoint intervals in ascending
  113 -- order.
  114 toList :: IntRangeSet -> [(Int, Int)]
  115 toList (IntRangeSet m) = IntMap.toList m
