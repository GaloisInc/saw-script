   12 import qualified Data.Parameterized.Map as PM

   14 
   15 type NameDom = Pointed (Constant String)
   16 
   17 nameJoin :: Constant String a -> Constant String a -> NameDom a
   18 nameJoin (Constant x) (Constant y) | x == y = Pointed (Constant x)
   19 nameJoin _ _ = Top
   20 
   21 nameDomain :: Domain (Pointed (Constant String))
   22 nameDomain = pointed nameJoin (==) WTO
   23 
   24 nameInterpretation :: Interpretation LLVM NameDom
   25 nameInterpretation =  Interpretation
   26   { interpExpr        = \_ _ _     names -> (Just names, Bottom)
   27   , interpCall        = \_ _ _ _ _ names -> (Just names, Bottom)
   28   , interpReadGlobal  = \_         names -> (Just names, Bottom)
   29   , interpWriteGlobal = \_ _       names -> Just names
   30   , interpBr          = \_ _ _ _   names -> (Just names, Just names)
   31   , interpMaybe       = \_ _ _     names -> (Just names, Bottom, Just names)
   32   , interpExt         = \_ stmt    names ->
   33       let names' =
   34             case stmt of
   35               LLVM_Debug (LLVM_Dbg_Declare ptr di _) | Just n <- L.dilvName di ->
   36                 modifyAbstractRegValue names ptr (\_ -> Pointed (Constant ("&" ++ n)))
   37               LLVM_Debug (LLVM_Dbg_Addr ptr di _) | Just n <- L.dilvName di ->
   38                 modifyAbstractRegValue names ptr (\_ -> Pointed (Constant ("&" ++ n)))
   39               LLVM_Debug (LLVM_Dbg_Value _ val di _) | Just n <- L.dilvName di ->
   40                 modifyAbstractRegValue names val (\_ -> Pointed (Constant n))
   41               _ -> names
   42       in (Just names', Bottom)             
   43   }
   44 
   45 computeNames ::
   46   forall blocks init ret.
   47   CFG LLVM blocks init ret ->
   48   Ctx.Assignment (Constant [Maybe String]) blocks
   49 computeNames cfg =
   50   case alg of
   51     (_, end, _) -> fmapFC (\(Ignore (Some p)) -> Constant (toListFC flatten (_paRegisters p))) end
   52   where
   53   flatten :: NameDom a -> Maybe String
   54   flatten Top = Nothing
   55   flatten Bottom = Nothing
   56   flatten (Pointed (Constant x)) = Just x
   57 
   58   sz = Ctx.size (handleArgTypes (cfgHandle cfg))
   59   alg =
   60       forwardFixpoint'
   61       nameDomain
   62       nameInterpretation
   63       cfg
   64       PM.empty
   65       (Ctx.replicate sz Bottom)
