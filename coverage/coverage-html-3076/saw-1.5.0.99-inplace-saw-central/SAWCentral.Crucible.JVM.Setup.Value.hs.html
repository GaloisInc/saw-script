   90 -- *** JVMMethodId
   91 
   92 data JVMMethodId =
   93   JVMMethodId
   94     { _jvmMethodKey :: J.MethodKey
   95     , _jvmClassName  :: J.ClassName
   96     }
   97   deriving (Eq, Ord, Show)
   98 
   99 makeLenses ''JVMMethodId
  100 
  101 -- TODO: avoid intermediate 'String' values
  102 instance PPL.Pretty JVMMethodId where
  103   pretty (JVMMethodId methKey className) =
  104     PPL.pretty (concat [J.unClassName className ,".", J.methodKeyName methKey])
  105 
  106 type instance MS.MethodId CJ.JVM = JVMMethodId
