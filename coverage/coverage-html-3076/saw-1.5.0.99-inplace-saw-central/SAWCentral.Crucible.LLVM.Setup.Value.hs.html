  181 -- *** LLVMModule
  182 
  183 -- | An 'LLVMModule' contains an LLVM module that has been parsed from
  184 -- a bitcode file and translated to Crucible.
  185 data LLVMModule arch =
  186   LLVMModule
  187   { _modFilePath :: FilePath
  188   , _modAST :: L.Module
  189   , _modTrans :: CL.ModuleTranslation arch
  190   }
  191 -- NOTE: Type 'LLVMModule' is exported as an abstract type, and we
  192 -- maintain the invariant that the 'FilePath', 'Module', and
  193 -- 'ModuleTranslation' fields are all consistent with each other;
  194 -- 'loadLLVMModule' is the only function that is allowed to create
  195 -- values of type 'LLVMModule'.
  196 
  197 -- | The file path that the LLVM module was loaded from.
  198 modFilePath :: LLVMModule arch -> FilePath
  199 modFilePath = _modFilePath
  200 
  201 -- | The parsed AST of the LLVM module.
  202 modAST :: LLVMModule arch -> L.Module
  203 modAST = _modAST
  204 
  205 -- | The Crucible translation of an LLVM module.
  206 modTrans :: LLVMModule arch -> CL.ModuleTranslation arch
  207 modTrans = _modTrans
  208 
  209 -- | Load an LLVM module from the given bitcode file, then parse and
  210 -- translate to Crucible.
  211 loadLLVMModule ::
  212   (?transOpts :: CL.TranslationOptions) =>
  213   FilePath ->
  214   Crucible.HandleAllocator ->
  215   IO (Either LLVM.Error (Some LLVMModule, Seq LLVM.ParseWarning))
  216 loadLLVMModule file halloc =
  217   do parseResult <- LLVM.parseBitCodeFromFileWithWarnings file
  218      case parseResult of
  219        Left err -> return (Left err)
  220        Right (llvm_mod, warnings) ->
  221          do memVar <- CL.mkMemVar (Text.pack "saw:llvm_memory") halloc
  222             Some mtrans <- CL.translateModule halloc memVar llvm_mod
  223             return (Right (Some (LLVMModule file llvm_mod mtrans), warnings))
  224 
  225 instance TestEquality LLVMModule where
  226   -- As 'LLVMModule' is an abstract type, we know that the values must
  227   -- have been created by a call to 'loadLLVMModule'. Furthermore each
  228   -- call to 'translateModule' generates a 'ModuleTranslation' that
  229   -- contains a fresh nonce; thus comparison of the 'modTrans' fields
  230   -- is sufficient to guarantee equality of two 'LLVMModule' values.
  231   testEquality m1 m2 = testEquality (modTrans m1) (modTrans m2)
  232 
  233 type instance Setup.Codebase (LLVM arch) = LLVMModule arch

  235 prettyLLVMModule :: LLVMModule arch -> PPL.Doc ann
  236 prettyLLVMModule (LLVMModule name m _) =
  237   PPL.vsep [ "Module:" PPL.<+> PPL.pretty name
  238           , "Types:"
  239           , prettyParts (Crucible.LLVM.ppLLVMLatest L.ppTypeDecl) (L.modTypes m)
  240           , "Globals:"
  241           , prettyParts (Crucible.LLVM.ppLLVMLatest prettyGlobal') (L.modGlobals m)
  242           , "External references:"
  243           , prettyParts Crucible.LLVM.ppDeclare (L.modDeclares m)
  244           , "Definitions:"
  245           , prettyParts (Crucible.LLVM.ppLLVMLatest prettyDefine') (L.modDefines m)
  246           ]
  247   where
  248     prettyParts pretty' xs =
  249       let xs' = PP.vcat $ map (PP.nest 2) $ map pretty' xs in
  250       -- Switch from the old prettyprinter to the new one
  251       PPL.pretty $ PP.render xs'
  252     prettyGlobal' g =
  253       L.ppSymbol (L.globalSym g) PP.<+> PP.char '=' PP.<+>
  254       L.ppGlobalAttrs (isJust $ L.globalValue g) (L.globalAttrs g) PP.<+>
  255       L.ppType (L.globalType g)
  256     prettyDefine' d =
  257       L.ppMaybe L.ppLinkage (L.defLinkage d) PP.<+>
  258       L.ppType (L.defRetType d) PP.<+>
  259       L.ppSymbol (L.defName d) PP.<>
  260       L.ppArgList (L.defVarArgs d) (map (L.ppTyped L.ppIdent) (L.defArgs d)) PP.<+>
  261       L.ppMaybe (\gc -> PP.text "gc" PP.<+> L.ppGC gc) (L.defGC d)

  314 data ResolvedPathItem
  315   = ResolvedField Text
  316   | ResolvedElem Int
  317   | ResolvedCast L.Type
  318  deriving (Show, Eq, Ord)
  319 
  320 type ResolvedPath = [ResolvedPathItem]
  321 
  322 -- | A datatype to keep track of which parts of the simulator state
  323 -- have been initialized already. For each allocation unit or global,
  324 -- we keep a list of element-paths that identify the initialized
  325 -- sub-components.
  326 --
  327 -- Note that the data collected and maintained by this datatype
  328 -- represents a \"best-effort\" check that attempts to prevent
  329 -- the user from stating unsatisfiable method specifications.
  330 --
  331 -- It will not prevent all cases of overlapping points-to
  332 -- specifications, especially in the presence of pointer casts.
  333 -- A typical result of overlapping specifications will be
  334 -- successful (vacuous) verifications of functions resulting in
  335 -- overrides that cannot be used at call sites (as their
  336 -- preconditions are unsatisfiable).
  337 data LLVMResolvedState =
  338   ResolvedState
  339     { _rsAllocs :: Map Setup.AllocIndex [ResolvedPath]
  340     , _rsGlobals :: Map Text [ResolvedPath]
  341     }
  342   deriving (Eq, Ord, Show)
  343 
  344 emptyResolvedState :: LLVMResolvedState
  345 emptyResolvedState = ResolvedState Map.empty Map.empty
  346 
  347 makeLenses ''LLVMResolvedState
