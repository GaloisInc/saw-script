  200 prettyLLVMPointsTo :: SharedContext -> PPS.Opts -> LLVMPointsTo arch -> IO PPS.Doc
  201 prettyLLVMPointsTo sc opts (LLVMPointsTo _md cond ptr val) = do
  202     ptr' <- MS.prettySetupValue sc opts ptr
  203     val' <- prettyLLVMPointsToValue sc opts val
  204     cond' <- case cond of
  205         Nothing ->
  206             pure PP.emptyDoc
  207         Just tt -> do
  208             tt' <- prettyTypedTerm sc opts tt
  209             pure $ "if" <+> tt'
  210     pure $ ptr' <+> "points to" <+> val' <+> cond'
  211 prettyLLVMPointsTo sc opts (LLVMPointsToBitfield _md ptr fieldName val) = do
  212     ptr' <- MS.prettySetupValue sc opts ptr
  213     val' <- MS.prettySetupValue sc opts val
  214     pure $ ptr' <> "." <> PP.pretty fieldName <+> "points to (bitfield)" <+> val'
  215 
  216 prettyLLVMPointsToValue :: SharedContext -> PPS.Opts -> (LLVMPointsToValue arch) -> IO PPS.Doc
  217 prettyLLVMPointsToValue sc opts ptv = case ptv of
  218     ConcreteSizeValue val ->
  219         MS.prettySetupValue sc opts val
  220     SymbolicSizeValue arr sz -> do
  221         arr' <- prettyTypedTerm sc opts arr
  222         sz' <- prettyTypedTerm sc opts sz
  223         pure $ arr' <+> "[" <+> sz' <+> "]"

  240 data SetupError
  241   = InvalidReturnType L.Type
  242   | InvalidArgTypes [L.Type]
  243 
  244 prettySetupError :: SetupError -> PP.Doc ann
  245 prettySetupError (InvalidReturnType t) =
  246   let t' = PP.viaShow $ Crucible.LLVM.ppType t in
  247   "Can't lift return type" <+> t' <+> "to a Crucible type."
  248 prettySetupError (InvalidArgTypes ts) =
  249   let ts' = map (\t -> PP.viaShow $ Crucible.LLVM.ppType t) ts
  250       ts'' = PP.parens $ PP.fillSep $ PP.punctuate "," ts'
  251   in
  252   "Can't lift argument types " <+> ts'' <+> "to Crucible types."

  254 resolveArgs ::
  255   (?lc :: CL.TypeContext) =>
  256   [L.Type] ->
  257   Either SetupError [CL.MemType]
  258 resolveArgs args = do
  259   -- TODO: make sure we resolve aliases
  260   let mtys = traverse CL.liftMemType args
  261   -- TODO: should the error message be propagated?
  262   either (\_ -> Left (InvalidArgTypes args)) Right mtys

  264 resolveRetTy ::
  265   (?lc :: CL.TypeContext) =>
  266   L.Type ->
  267   Either SetupError (Maybe CL.MemType)
  268 resolveRetTy ty = do
  269   -- TODO: make sure we resolve aliases
  270   let ret = CL.liftRetType ty
  271   -- TODO: should the error message be propagated?
  272   either (\_ -> Left (InvalidReturnType ty)) Right ret

  274 initialDefCrucibleMethodSpecIR ::
  275   (?lc :: CL.TypeContext) =>
  276   LLVMModule arch ->
  277   L.Define ->
  278   ProgramLoc ->
  279   Maybe Text ->
  280   Either SetupError (MS.CrucibleMethodSpecIR (LLVM arch))
  281 initialDefCrucibleMethodSpecIR llvmModule def loc parent = do
  282   args <- resolveArgs (L.typedType <$> L.defArgs def)
  283   ret <- resolveRetTy (L.defRetType def)
  284   let L.Symbol nm = L.defName def
  285   let methId = LLVMMethodId (Text.pack nm) parent
  286   return $ MS.makeCrucibleMethodSpecIR methId args ret loc llvmModule
  287 
  288 initialDeclCrucibleMethodSpecIR ::
  289   (?lc :: CL.TypeContext) =>
  290   LLVMModule arch ->
  291   L.Declare ->
  292   ProgramLoc ->
  293   Maybe Text ->
  294   Either SetupError (MS.CrucibleMethodSpecIR (LLVM arch))
  295 initialDeclCrucibleMethodSpecIR llvmModule dec loc parent = do
  296   args <- resolveArgs (L.decArgs dec)
  297   ret <- resolveRetTy (L.decRetType dec)
  298   let L.Symbol nm = L.decName dec
  299   let methId = LLVMMethodId (Text.pack nm) parent
  300   return $ MS.makeCrucibleMethodSpecIR methId args ret loc llvmModule

  328 -- *** AllLLVM
  329 
  330 -- | Universal/polymorphic quantification over an 'LLVMArch'
  331 --
  332 -- The following type synonym and associated constructor/destructor are
  333 -- equivalent to this definition:
  334 -- @
  335 -- data AllLLVM t =
  336 --   MkAllLLVM { getAllLLVM :: forall arch. t (LLVM arch) }
  337 -- @
  338 -- But they preserve the instances from 'All' and 'Compose'.
  339 type AllLLVM t = All (Compose t LLVM)
  340 
  341 -- This doesn't work :(
  342 --
  343 -- pattern AllLLVM :: (forall arch. t (LLVM arch)) -> AllLLVM t
  344 -- pattern AllLLVM x = All (Compose x)
  345 
  346 mkAllLLVM :: forall t. (forall arch. t (LLVM arch)) -> AllLLVM t
  347 mkAllLLVM x = All (Compose x)
  348 
  349 getAllLLVM :: forall t. AllLLVM t -> (forall arch. t (LLVM arch))
  350 getAllLLVM (All (Compose x)) = x
  351 
  352 -- Constructors for 'SetupValue' which are architecture-polymorphic
  353 
  354 anySetupTerm :: TypedTerm -> AllLLVM MS.SetupValue
  355 anySetupTerm typedTerm = mkAllLLVM (MS.SetupTerm typedTerm)
  356 
  357 anySetupArray :: [AllLLVM MS.SetupValue] -> AllLLVM MS.SetupValue
  358 anySetupArray vals = mkAllLLVM (MS.SetupArray () $ map (\a -> getAllLLVM a) vals)
  359 
  360 anySetupStruct :: Bool -> [AllLLVM MS.SetupValue] -> AllLLVM MS.SetupValue
  361 anySetupStruct b vals = mkAllLLVM (MS.SetupStruct b $ map (\a -> getAllLLVM a) vals)
  362 
  363 anySetupElem :: AllLLVM MS.SetupValue -> Int -> AllLLVM MS.SetupValue
  364 anySetupElem val idx = mkAllLLVM (MS.SetupElem () (getAllLLVM val) idx)
  365 
  366 anySetupCast :: AllLLVM MS.SetupValue -> L.Type -> AllLLVM MS.SetupValue
  367 anySetupCast val ty = mkAllLLVM (MS.SetupCast ty (getAllLLVM val))
  368 
  369 anySetupField :: AllLLVM MS.SetupValue -> Text -> AllLLVM MS.SetupValue
  370 anySetupField val field = mkAllLLVM (MS.SetupField () (getAllLLVM val) field)
  371 
  372 anySetupUnion :: AllLLVM MS.SetupValue -> Text -> AllLLVM MS.SetupValue
  373 anySetupUnion val uname = mkAllLLVM (MS.SetupUnion () (getAllLLVM val) uname)
  374 
  375 anySetupNull :: AllLLVM MS.SetupValue
  376 anySetupNull = mkAllLLVM (MS.SetupNull ())
  377 
  378 anySetupGlobal :: Text -> AllLLVM MS.SetupValue
  379 anySetupGlobal globalName = mkAllLLVM (MS.SetupGlobal () globalName)
  380 
  381 anySetupGlobalInitializer :: Text -> AllLLVM MS.SetupValue
  382 anySetupGlobalInitializer globalName =
  383   mkAllLLVM (MS.SetupGlobalInitializer () globalName)

  385 --------------------------------------------------------------------------------
  386 -- *** SomeLLVM
  387 
  388 -- | Existential quantification over an 'LLVMArch'
  389 --
  390 -- The following type synonym and associated constructor/destructor are
  391 -- equivalent to this definition:
  392 -- @
  393 -- data SomeLLVM t = forall arch. MkSomeLLVM (t (LLVM arch))
  394 -- @
  395 -- But they preserve the instances from 'Some' and 'Compose'.
  396 type SomeLLVM t = Some (Compose t LLVM)
  397 
  398 pattern SomeLLVM :: t (LLVM arch) -> SomeLLVM t
  399 pattern SomeLLVM x = Some (Compose x)
  400 {-# COMPLETE SomeLLVM #-}
  401 
  402 mkSomeLLVM :: t (LLVM arch) -> SomeLLVM t
  403 mkSomeLLVM x = Some (Compose x)
  404 
  405 getSomeLLVM :: forall t. (forall arch. t (LLVM arch)) -> AllLLVM t
  406 getSomeLLVM x = All (Compose x)

  409 -- *** ResolvedState

  411 -- | Record the initialization of the pointer represented by the given
  412 -- SetupValue.
  413 markResolved ::
  414   MS.SetupValue (LLVM arch) ->
  415   ResolvedPath {-^ path within this object (if any) -} ->
  416   LLVMResolvedState ->
  417   LLVMResolvedState
  418 markResolved val0 path0 rs = go path0 val0
  419   where
  420     go path val =
  421       case val of
  422         MS.SetupVar n         -> rs & rsAllocs %~ Map.alter (ins path) n
  423         MS.SetupGlobal _ name -> rs & rsGlobals %~ Map.alter (ins path) name
  424         MS.SetupElem _ v idx  -> go (ResolvedElem idx : path) v
  425         MS.SetupField _ v fld -> go (ResolvedField fld : path) v
  426         MS.SetupCast tp v     -> go (ResolvedCast tp : path) v
  427         _                     -> rs
  428 
  429     ins path Nothing = Just [path]
  430     ins path (Just paths) = Just (path : paths)
  431 
  432 -- | Test whether the pointer represented by the given SetupValue has
  433 -- been initialized already.
  434 testResolved ::
  435   MS.SetupValue (LLVM arch) ->
  436   ResolvedPath {-^ path within this object (if any) -} ->
  437   LLVMResolvedState ->
  438   Bool
  439 testResolved val0 path0 rs = go path0 val0
  440   where
  441     go path val =
  442       case val of
  443         MS.SetupVar n         -> test path (Map.lookup n (_rsAllocs rs))
  444         MS.SetupGlobal _ c    -> test path (Map.lookup c (_rsGlobals rs))
  445         MS.SetupElem _ v idx  -> go (ResolvedElem idx : path) v
  446         MS.SetupField _ v fld -> go (ResolvedField fld : path) v
  447         MS.SetupCast tp v     -> go (ResolvedCast tp : path) v
  448         _                     -> False
  449 
  450     test _ Nothing = False
  451     test path (Just paths) = any (overlap path) paths
  452 
  453     overlap (x : xs) (y : ys) = x == y && overlap xs ys
  454     overlap [] _ = True
  455     overlap _ [] = True
