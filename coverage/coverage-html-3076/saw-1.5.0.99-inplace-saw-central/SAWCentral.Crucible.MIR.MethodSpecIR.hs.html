  126 prettyMirPointsTo :: SharedContext -> PPS.Opts -> MirPointsTo -> IO PPS.Doc
  127 prettyMirPointsTo sc opts (MirPointsTo _md ref tar) = do
  128     ref' <- MS.prettySetupValue sc opts ref
  129     tar' <- prettyMirPointsToTarget sc opts tar
  130     pure $ PP.parens $ ref' <+> tar'
  131 
  132 prettyMirPointsToTarget :: SharedContext -> PPS.Opts -> MirPointsToTarget -> IO PPS.Doc
  133 prettyMirPointsToTarget sc opts tgt = case tgt of
  134     CrucibleMirCompPointsToTarget svs -> do
  135         svs' <- mapM (MS.prettySetupValue sc opts) svs
  136         pure $ "->" <+> PP.list svs'
  137     MirPointsToSingleTarget sv -> do
  138         sv' <- MS.prettySetupValue sc opts sv
  139         pure $ "->" <+> sv'
  140     MirPointsToMultiTarget svArr -> do
  141         svArr' <- MS.prettySetupValue sc opts svArr
  142         pure $ "->*" <+> svArr'
  143 
  144 ppMirPointsToTarget :: SharedContext -> PPS.Opts -> MirPointsToTarget -> IO Text
  145 ppMirPointsToTarget sc opts tgt = do
  146     PPS.renderText opts <$> prettyMirPointsToTarget sc opts tgt

  148 prettyMirAllocSpec :: MirAllocSpec tp -> PP.Doc ann
  149 prettyMirAllocSpec spec =
  150     let meta = spec ^. maConditionMetadata
  151         _typeRepr = spec ^. maType  -- I think we don't need to print this
  152         ptrKind = spec ^. maPtrKind
  153         mut = spec ^. maMutbl
  154         mirType = spec ^. maMirType
  155         len = spec ^. maLen
  156     in
  157     let meta' = PP.viaShow meta
  158         ptrKind' = case ptrKind of
  159             MirPointerRef -> ""
  160             MirPointerRaw -> "raw "
  161         mut' = PP.viaShow mut
  162         mirType' = PP.viaShow mirType
  163         len' = PP.viaShow len
  164     in
  165     -- XXX: is this ok? it does not I think quite match any Rust syntax
  166     PP.braces meta' <+> "&" <> ptrKind' <> mut' <+> mirType' <+> PP.brackets len'
  167 
  168 ppMirAllocSpec :: PPS.Opts -> MirAllocSpec tp -> Text
  169 ppMirAllocSpec opts spec =
  170     PPS.renderText opts $ prettyMirAllocSpec spec

  172 mutIso :: Iso' M.Mutability Bool
  173 mutIso =
  174   iso
  175     (\case
  176       M.Mut -> True
  177       M.Immut -> False)
  178     (\case
  179       True -> M.Mut
  180       False -> M.Immut)

  182 isMut :: Lens' (MirAllocSpec tp) Bool
  183 isMut = maMutbl . mutIso

  185 -- | Converts a 'MirPointerKind' to a 'M.Ty' constructor.
  186 ptrKindToTy :: MirPointerKind -> M.Ty -> M.Mutability -> M.Ty
  187 ptrKindToTy MirPointerRef = M.TyRef
  188 ptrKindToTy MirPointerRaw = M.TyRawPtr
  189 
  190 -- | Gets the 'MirPointerKind' for a reference or pointer type. Panics if the
  191 -- given 'M.Ty' is not a reference or pointer.
  192 tyToPtrKind :: M.Ty -> MirPointerKind
  193 tyToPtrKind (M.TyRef _ _) = MirPointerRef
  194 tyToPtrKind (M.TyRawPtr _ _) = MirPointerRaw
  195 tyToPtrKind ty = panic "tyToPtrKind"
  196   ["Type is not a reference or pointer: " <> Text.pack (show ty)]

  200 initialDefCrucibleMethodSpecIR ::
  201   CollectionState ->
  202   M.Fn ->
  203   ProgramLoc ->
  204   MS.CrucibleMethodSpecIR MIR
  205 initialDefCrucibleMethodSpecIR rm fn loc =
  206   let fname = fn ^. M.fname
  207       fsig = fn ^. M.fsig
  208       argTys = fsig ^. M.fsarg_tys
  209       retTy = case fsig ^. M.fsreturn_ty of
  210                 M.TyTuple [] -> Nothing
  211                 ty -> Just ty in
  212   MS.makeCrucibleMethodSpecIR fname argTys retTy loc rm
  213 
  214 initialCrucibleSetupState ::
  215   MIRCrucibleContext ->
  216   M.Fn ->
  217   ProgramLoc ->
  218   Setup.CrucibleSetupState MIR
  219 initialCrucibleSetupState cc fn loc =
  220   Setup.makeCrucibleSetupState () cc $
  221     initialDefCrucibleMethodSpecIR
  222       (cc ^. mccRustModule ^. rmCS)
  223       fn
  224       loc
