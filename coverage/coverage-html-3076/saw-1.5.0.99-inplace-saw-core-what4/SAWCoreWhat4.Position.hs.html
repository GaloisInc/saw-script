   12 module SAWCoreWhat4.Position () where

   14 import qualified Prettyprinter as PP
   15 import Prettyprinter ((<+>))
   16 
   17 import qualified What4.ProgramLoc as W4
   18 
   19 import SAWSupport.Position
   20 
   21 -- | SAW IsPosition instance for What4 Position.
   22 --
   23 --   What4 provides a `Pretty` instance for its `Position` type,
   24 --   so we can just use that.
   25 instance IsPosition W4.Position where
   26     prettyPosition pos = PP.pretty pos
   27 
   28 -- | SAW IsPosition instance for What4 ProgramLoc.
   29 --
   30 --   What4 does _not_ provide a `Pretty` instance for `ProgramLoc`, so
   31 --   we need to roll one. (Instead, it provides a derived `Show`
   32 --   instance. Blegh.)
   33 instance IsPosition W4.ProgramLoc where
   34     prettyPosition pl =
   35         let pos = W4.plSourceLoc pl
   36             func = W4.plFunction pl
   37         in
   38         let pos' = prettyPosition pos
   39             func' =
   40                 -- W4.FunctionName is a wrapper around Text, but does
   41                 -- provide a Pretty instance. It isn't clear if we'll
   42                 -- ever get empty function names, but just in case
   43                 -- let's not print nonsense if we do.
   44                 if func == "" then
   45                     PP.emptyDoc
   46                 else
   47                     " in" <+> PP.pretty func
   48         in
   49         pos' <> func'
