   69 data UTerm
   70   = Name (PosPair Text)
   71   | Sort Pos Sort SortFlags
   72   | App UTerm UTerm
   73   | Lambda Pos UTermCtx UTerm
   74   | Let Pos [(UTermVar, UTerm)] UTerm
   75   | Pi Pos UTermCtx UTerm
   76   | Recursor (PosPair Text) Sort
   77   | UnitValue Pos
   78   | UnitType Pos
   79     -- | New-style records
   80   | RecordValue Pos [(PosPair FieldName, UTerm)]
   81   | RecordType Pos [(PosPair FieldName, UTerm)]
   82   | RecordProj UTerm FieldName
   83     -- | Simple pairs
   84   | PairValue Pos UTerm UTerm
   85   | PairType Pos UTerm UTerm
   86   | PairLeft UTerm
   87   | PairRight UTerm
   88     -- | Identifies a type constraint on the term, i.e., a type ascription
   89   | TypeConstraint UTerm Pos UTerm
   90   | NatLit Pos Natural
   91   | StringLit Pos Text
   92     -- | Vector literal.
   93   | VecLit Pos [UTerm]
   94     -- | Bitvector literal.
   95   | BVLit Pos [Bool]
   96   | BadTerm Pos
   97  deriving (Show, TH.Lift)

   99 -- | A pattern used for matching a variable.
  100 data UTermVar
  101   = TermVar (PosPair LocalName)
  102   | UnusedVar Pos
  103   deriving (Eq, Ord, Show, TH.Lift)
  104 
  105 -- | Return the 'String' name associated with a 'UTermVar'
  106 termVarString :: UTermVar -> String
  107 termVarString (TermVar (PosPair _ str)) = Text.unpack str
  108 termVarString (UnusedVar _) = "_"
  109 
  110 -- | Return the 'LocalName' associated with a 'UTermVar'
  111 termVarLocalName :: UTermVar -> LocalName
  112 termVarLocalName (TermVar (PosPair _ str)) = str
  113 termVarLocalName (UnusedVar _) = "_"
  114 
  115 -- | A context of 0 or more variable bindings, with types
  116 type UTermCtx = [(UTermVar, UTerm)]
  117 
  118 instance Positioned UTerm where
  119   pos t =
  120     case t of
  121       Name i               -> pos i
  122       Sort p _ _           -> p
  123       Lambda p _ _         -> p
  124       Let p _ _            -> p
  125       App x _              -> pos x
  126       Pi p _ _             -> p
  127       Recursor i _         -> pos i
  128       UnitValue p          -> p
  129       UnitType p           -> p
  130       RecordValue p _      -> p
  131       RecordType p _       -> p
  132       RecordProj x _       -> pos x
  133       PairValue p _ _      -> p
  134       PairType p _ _       -> p
  135       PairLeft x           -> pos x
  136       PairRight x          -> pos x
  137       TypeConstraint _ p _ -> p
  138       NatLit p _           -> p
  139       StringLit p _        -> p
  140       VecLit p _           -> p
  141       BVLit p _            -> p
  142       BadTerm p            -> p
  143 
  144 instance Positioned UTermVar where
  145   pos (TermVar i) = pos i
  146   pos (UnusedVar p) = p

  148 badTerm :: Pos -> UTerm
  149 badTerm = BadTerm

  163 -- | A top-level declaration in a saw-core file
  164 data Decl
  165    = TypeDecl DefQualifier (PosPair Text) UTerm
  166      -- ^ A declaration of something having a type, where the declaration
  167      -- qualifier states what flavor of thing it is
  168    | DataDecl (PosPair Text) UTermCtx UTerm [CtorDecl]
  169      -- ^ A declaration of an inductive data types, with a name, a parameter
  170      -- context, a return type, and a list of constructor declarations
  171    | TermDef (PosPair Text) [UTermVar] UTerm
  172      -- ^ A declaration of a term having a definition, with variables
  173    | TypedDef (PosPair Text) [(UTermVar, UTerm)] UTerm UTerm
  174      -- ^ A definition of something with a specific type, with parameters
  175    | InjectCodeDecl Text Text
  176      -- ^ Some raw text to inject into a translation
  177   deriving (Show, TH.Lift)

  179 -- | A set of constraints on what 'String' names to import from a module
  180 data ImportConstraint
  181   = SpecificImports [String]
  182     -- ^ Only import the given names
  183   | HidingImports [String]
  184     -- ^ Import all but the given names
  185  deriving (Eq, Ord, Show, TH.Lift)
  186 
  187 -- | An import declaration
  188 data Import = Import { importModName :: PosPair ModuleName
  189                        -- ^ The name of the module to import
  190                      , importConstraints :: Maybe ImportConstraint
  191                        -- ^ The constraints on what to import
  192                      }
  193             deriving (Show, TH.Lift)
  194 
  195 -- | Test whether a 'String' name satisfies the constraints of an 'Import'
  196 nameSatsConstraint :: Maybe ImportConstraint -> String -> Bool
  197 nameSatsConstraint Nothing _ = True
  198 nameSatsConstraint (Just (SpecificImports ns)) n = elem n ns
  199 nameSatsConstraint (Just (HidingImports ns)) n = notElem n ns

  202 -- | A module declaration gives:
  203 -- * A name for the module;
  204 -- * A list of imports; AND
  205 -- * A list of top-level declarations
  206 data Module = Module (PosPair ModuleName) [Import] [Decl]
  207   deriving (Show, TH.Lift)
  208 
  209 moduleName :: Module -> ModuleName
  210 moduleName (Module (PosPair _ mnm) _ _) = mnm
  211 
  212 -- | Get a list of all names (i.e., definitions, axioms, or primitives) declared
  213 -- in a module, along with their types and qualifiers
  214 moduleTypedDecls :: Module -> [(Text, UTerm)]
  215 moduleTypedDecls (Module _ _ decls) = concatMap helper decls where
  216   helper :: Decl -> [(Text, UTerm)]
  217   helper (TypeDecl _ (PosPair _ nm) tm) = [(nm,tm)]
  218   helper _ = []
  219 
  220 -- | Get a list of all datatypes declared in a module
  221 moduleDataDecls :: Module -> [(Text, UTermCtx, UTerm, [CtorDecl])]
  222 moduleDataDecls (Module _ _ decls) = concatMap helper decls where
  223   helper :: Decl -> [(Text, UTermCtx, UTerm, [CtorDecl])]
  224   helper (DataDecl (PosPair _ nm) params tp ctors) = [(nm, params, tp, ctors)]
  225   helper _ = []
  226 
  227 moduleTypedDataDecls :: Module -> [(Text, UTerm)]
  228 moduleTypedDataDecls =
  229   map (\(nm,p_ctx,tp,_) ->
  230         (nm, Pi (pos tp) p_ctx tp)) . moduleDataDecls
  231 
  232 -- | Get a list of all constructors declared in a module, along with the context
  233 -- of parameters for each one
  234 moduleCtorDecls :: Module -> [(UTermCtx, CtorDecl)]
  235 moduleCtorDecls =
  236   concatMap (\(_,p_ctx,_,ctors) -> map (p_ctx,) ctors) . moduleDataDecls
  237 
  238 -- | Get a list of the names and types of all the constructors in a module
  239 moduleTypedCtorDecls :: Module -> [(Text, UTerm)]
  240 moduleTypedCtorDecls =
  241   concatMap (\(_,p_ctx,_,ctors) ->
  242               map (\(Ctor (PosPair _ nm) ctx tp) ->
  243                     (nm, Pi (pos tp) (p_ctx ++ ctx) tp)) ctors)
  244   . moduleDataDecls
  245 
  246 asPiList :: UTerm -> (UTermCtx, UTerm)
  247 asPiList (Pi _ ctx1 body1) =
  248   let (ctx2,body2) = asPiList body1 in
  249   (ctx1 ++ ctx2, body2)
  250 asPiList t = ([], t)
  251 
  252 asApp :: UTerm -> (UTerm, [UTerm])
  253 asApp = go []
  254   where go l (App t u)   = go (u:l) t
  255         go l t = (t,l)
  256 
  257 -- | Build a tuple value @(x1, .., xn)@.
  258 mkTupleValue :: Pos -> [UTerm] -> UTerm
  259 mkTupleValue p [] = UnitValue p
  260 mkTupleValue _ [x] = x
  261 mkTupleValue p (x:xs) = PairValue (pos x) x (mkTupleValue p xs)
  262 
  263 -- | Build a tuple type @#(x1, .., xn)@.
  264 mkTupleType :: Pos -> [UTerm] -> UTerm
  265 mkTupleType p [] = UnitType p
  266 mkTupleType _ [x] = x
  267 mkTupleType p (x:xs) = PairType (pos x) x (mkTupleType p xs)
  268 
  269 -- | Build a projection @t.i@ of a tuple. NOTE: This function does not
  270 -- work to access the last component in a tuple, since it always
  271 -- generates a @PairLeft@.
  272 mkTupleSelector :: UTerm -> Natural -> UTerm
  273 mkTupleSelector t i
  274   | i == 1    = PairLeft t
  275   | i > 1     = mkTupleSelector (PairRight t) (i - 1)
  276   | otherwise = error "mkTupleSelector: non-positive index"

  281 -- | Deconstruct nested left-associated 'App' constructors.
  282 asApps :: UTerm -> (UTerm, [UTerm])
  283 asApps uterm = go uterm []
  284   where
  285     go t ts =
  286       case t of
  287         App t1 t2 -> go t1 (t2 : ts)
  288         _ -> (t, ts)
  289 
  290 -- | Deconstruct nested left-associated 'PairValue' constructors.
  291 asPairValues :: UTerm -> (UTerm, [UTerm])
  292 asPairValues uterm = go uterm []
  293   where
  294     go t ts =
  295       case t of
  296         PairValue _ t1 t2 -> go t1 (t2 : ts)
  297         _ -> (t, ts)

  299 flagsPrefix :: SortFlags -> Text
  300 flagsPrefix (SortFlags i q) =
  301   (if i then "i" else "") <> (if q then "q" else "")
  302 
  303 ppSort :: Sort -> Text
  304 ppSort (TypeSort n) = "sort " <> Text.pack (show n)
  305 ppSort PropSort = "Prop"
  306 
  307 prettyPiBinding :: (UTermVar, UTerm) -> PPS.Doc
  308 prettyPiBinding (v, t) =
  309   case v of
  310     TermVar (PosPair _ str) ->
  311       PP.parens (PP.pretty str PP.<+> PP.colon </> prettyPrecUTerm 1 t)
  312     UnusedVar _ ->
  313       prettyPrecUTerm 2 t
  314 
  315 prettyUTermVar :: UTermVar -> PPS.Doc
  316 prettyUTermVar v =
  317   case v of
  318     TermVar (PosPair _ str) -> PP.pretty str
  319     UnusedVar _ -> "_"
  320 
  321 prettyLambdaBinding :: (UTermVar, UTerm) -> PPS.Doc
  322 prettyLambdaBinding (v, t) =
  323   PP.parens (prettyUTermVar v PP.<+> PP.colon </> prettyPrecUTerm 1 t)
  324 
  325 
  326 prettyUTermCtx :: UTermCtx -> PPS.Doc
  327 prettyUTermCtx ctx = PP.sep (map prettyLambdaBinding ctx)

  329 ppRecursorSuffix :: Sort -> Text
  330 ppRecursorSuffix s =
  331   case s of
  332     PropSort -> "ind"
  333     TypeSort 0 -> "rec"
  334     TypeSort n -> "rec" <> Text.pack (show n)
  335 
  336 prettyField :: Text -> (PosPair FieldName, UTerm) -> PPS.Doc
  337 prettyField op (fname, t) =
  338   PP.pretty (val fname) PP.<+> PP.pretty op </> prettyPrecUTerm 1 t

  340 prettyUTerm :: UTerm -> PPS.Doc
  341 prettyUTerm t = prettyPrecUTerm 0 t
  342 
  343 -- | Precedence values:
  344 -- 0: type ascription
  345 -- 1: lambda or pi expression
  346 -- 2: infix '*' operator
  347 -- 3: function application
  348 -- 4: atomic expression
  349 --
  350 -- Term0 := Term1 | Term1 ':' Term1
  351 -- Term1 := Term2 | Term2 '->' Term1 | '\\' VarCtx '->' Term1
  352 -- Term2 := Term3 | Term3 '*' Term2
  353 -- Term3 := Term4 | Term3 Term4
  354 -- Term4 (atomic)
  355 prettyPrecUTerm :: Int -> UTerm -> PPS.Doc
  356 prettyPrecUTerm prec uterm =
  357   case uterm of
  358     Name x -> PP.pretty (val x)
  359     Sort _ s flags -> PP.pretty (flagsPrefix flags <> ppSort s)
  360     App _ _ ->
  361       let (e, es) = asApps uterm
  362       in PP.nest 2 (wrap prec 3 (foldl (</>) (prettyPrecUTerm 3 e) (map (prettyPrecUTerm 4) es)))
  363     Lambda _ ctx body ->
  364       PP.nest 1 (wrap prec 1 ("\\" PP.<+> prettyUTermCtx ctx PP.<+> "->" </> prettyPrecUTerm 1 body))
  365     Let _ binds body -> PPS.prettyLetBlock
  366       (map (\(var,def) -> (prettyUTermVar var, prettyPrecUTerm 1 def)) binds)
  367       (prettyPrecUTerm 1 body)
  368     Pi _ ctx body ->
  369       wrap prec 1 (foldr (\a b -> a PP.<+> "->" </> b) (prettyPrecUTerm 1 body) (map prettyPiBinding ctx))
  370     Recursor x s -> PP.pretty (val x <> "#" <> ppRecursorSuffix s)
  371     UnitValue _ -> PP.parens mempty
  372     UnitType _ -> "#" <> PP.parens mempty
  373     RecordValue _ fs ->
  374       PP.group (PP.nest 1 (PP.braces (commaSepFill (map (prettyField "=") fs))))
  375     RecordType _ fs ->
  376       PP.group (PP.nest 1 ("#" <> PP.braces (commaSepFill (map (prettyField ":") fs))))
  377     RecordProj t1 fname ->
  378       prettyPrecUTerm 4 t1 <> "." <> PP.pretty fname
  379     PairValue _ _ _ ->
  380       let (t1, ts) = asPairValues uterm
  381       in PP.group (PP.nest 1 (PP.parens (commaSepFill (map prettyUTerm (t1 : ts)))))
  382     PairType _ t1 t2 ->
  383       wrap prec 2 (PP.sep [prettyPrecUTerm 3 t1, "*" PP.<+> prettyPrecUTerm 2 t2])
  384     PairLeft t1 ->
  385       prettyPrecUTerm 4 t1 <> ".1"
  386     PairRight t1 ->
  387       prettyPrecUTerm 4 t1 <> ".2"
  388     TypeConstraint t1 _ t2 ->
  389       wrap prec 0 (PP.sep [prettyPrecUTerm 1 t1 PP.<+> ":", prettyPrecUTerm 1 t2])
  390     NatLit _ n ->
  391       PP.pretty n
  392     StringLit _ s ->
  393       PP.pretty (show s)
  394     VecLit _ ts ->
  395       PP.group (PP.nest 1 (PP.brackets (commaSepFill (map prettyUTerm ts))))
  396     BVLit _ bs -> "0b" <> mconcat [ if b then "1" else "0" | b <- bs ]
  397     BadTerm _ -> "<BadTerm>"
  398 
  399 -- | Print a comma-separated list. Lay out each item on a single line
  400 -- if it will fit. If an item requires multiple lines, then start it
  401 -- on its own line.
  402 commaSepFill :: [PPS.Doc] -> PPS.Doc
  403 commaSepFill xs = fillSep (PP.punctuate PP.comma xs)
  404   where
  405     fillSep [] = mempty
  406     fillSep (d0 : ds) = foldl (\a d -> a <> PP.group (PP.line <> d)) d0 ds
  407 
  408 wrap :: Int -> Int -> PPS.Doc -> PPS.Doc
  409 wrap contextPrec myPrec doc = optParens (myPrec < contextPrec) doc
  410 
  411 optParens :: Bool -> PPS.Doc -> PPS.Doc
  412 optParens b body
  413   | b = PP.nest 1 (PP.parens body)
  414   | otherwise = body
  415 
  416 (</>) :: PPS.Doc -> PPS.Doc -> PPS.Doc
  417 x </> y = x <> PP.group (PP.line <> y)
