  128 prettyMirPointsTo :: SharedContext -> MirPointsTo -> IO PPS.Doc
  129 prettyMirPointsTo sc (MirPointsTo _md ref tar) = do
  130     ref' <- MS.prettySetupValue sc ref
  131     tar' <- prettyMirPointsToTarget sc tar
  132     pure $ PP.parens $ ref' <+> tar'
  133 
  134 prettyMirPointsToTarget :: SharedContext -> MirPointsToTarget -> IO PPS.Doc
  135 prettyMirPointsToTarget sc tgt = case tgt of
  136     CrucibleMirCompPointsToTarget svs -> do
  137         svs' <- mapM (MS.prettySetupValue sc) svs
  138         pure $ "->" <+> PP.list svs'
  139     MirPointsToSingleTarget sv -> do
  140         sv' <- MS.prettySetupValue sc sv
  141         pure $ "->" <+> sv'
  142     MirPointsToMultiTarget svArr -> do
  143         svArr' <- MS.prettySetupValue sc svArr
  144         pure $ "->*" <+> svArr'
  145 
  146 ppMirPointsToTarget :: SharedContext -> MirPointsToTarget -> IO Text
  147 ppMirPointsToTarget sc tgt = do
  148     opts <- scGetPPOpts sc
  149     PPS.renderText opts <$> prettyMirPointsToTarget sc tgt

  151 prettyMirAllocSpec :: MirAllocSpec tp -> PPS.Doc
  152 prettyMirAllocSpec spec =
  153     let meta = spec ^. maConditionMetadata
  154         -- skip spec ^. maType; printing maMirType instead is better
  155         ptrKind = spec ^. maPtrKind
  156         mut = spec ^. maMutbl
  157         mirType = spec ^. maMirType
  158         len = spec ^. maLen
  159     in
  160     let meta' = MS.prettyConditionMetadata meta
  161         ptrKind' = case ptrKind of
  162             MirPointerRef -> ""
  163             MirPointerRaw -> "raw "
  164         mut' = case mut of
  165             M.Mut -> "mut "
  166             M.Immut -> ""
  167         mirType' = PP.pretty mirType
  168         len' = PP.viaShow len  -- len is an Int
  169     in
  170     -- This does not quite match any Rust syntax, but we think it's ok
  171     -- (at least for now)
  172     PP.braces meta' <+> "&" <> ptrKind' <> mut' <> mirType' <+> PP.brackets len'
  173 
  174 ppMirAllocSpec :: PPS.Opts -> MirAllocSpec tp -> Text
  175 ppMirAllocSpec opts spec =
  176     PPS.renderText opts $ prettyMirAllocSpec spec

  178 mutIso :: Iso' M.Mutability Bool
  179 mutIso =
  180   iso
  181     (\case
  182       M.Mut -> True
  183       M.Immut -> False)
  184     (\case
  185       True -> M.Mut
  186       False -> M.Immut)

  188 isMut :: Lens' (MirAllocSpec tp) Bool
  189 isMut = maMutbl . mutIso

  191 -- | Converts a 'MirPointerKind' to a 'M.Ty' constructor.
  192 ptrKindToTy :: MirPointerKind -> M.Ty -> M.Mutability -> M.Ty
  193 ptrKindToTy MirPointerRef = M.TyRef
  194 ptrKindToTy MirPointerRaw = M.TyRawPtr
  195 
  196 -- | Gets the 'MirPointerKind' for a reference or pointer type. Panics if the
  197 -- given 'M.Ty' is not a reference or pointer.
  198 tyToPtrKind :: M.Ty -> MirPointerKind
  199 tyToPtrKind (M.TyRef _ _) = MirPointerRef
  200 tyToPtrKind (M.TyRawPtr _ _) = MirPointerRaw
  201 tyToPtrKind ty = panic "tyToPtrKind"
  202   ["Type is not a reference or pointer: " <> Text.pack (show ty)]

  206 initialDefCrucibleMethodSpecIR ::
  207   CollectionState ->
  208   M.Fn ->
  209   ProgramLoc ->
  210   MS.CrucibleMethodSpecIR MIR
  211 initialDefCrucibleMethodSpecIR rm fn loc =
  212   let fname = fn ^. M.fname
  213       fsig = fn ^. M.fsig
  214       argTys = fsig ^. M.fsarg_tys
  215       retTy = case fsig ^. M.fsreturn_ty of
  216                 M.TyTuple [] -> Nothing
  217                 ty -> Just ty in
  218   MS.makeCrucibleMethodSpecIR fname argTys retTy loc rm
  219 
  220 initialCrucibleSetupState ::
  221   MIRCrucibleContext ->
  222   M.Fn ->
  223   ProgramLoc ->
  224   Setup.CrucibleSetupState MIR
  225 initialCrucibleSetupState cc fn loc =
  226   Setup.makeCrucibleSetupState () cc $
  227     initialDefCrucibleMethodSpecIR
  228       (cc ^. mccRustModule ^. rmCS)
  229       fn
  230       loc
