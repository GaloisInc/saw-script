   51 -- | Evaluator for shared terms.
   52 evalSharedTerm :: ModuleMap -> Map Ident RPrim -> Term -> RValue
   53 evalSharedTerm m addlPrims t =
   54   runIdentity $ do
   55     cfg <- Sim.evalGlobal m (Map.union constMap addlPrims)
   56            variable (\_ _ -> Nothing) recursor primHandler
   57            (Prims.lazyMuxValue prims)
   58     Sim.evalSharedTerm cfg t
   59   where
   60     variable vn _tp = return $ Prim.userError $ "Unimplemented: free variable " ++ show (vnName vn)
   61     primHandler nm msg env =
   62       return $ Prim.userError $ unlines
   63         [ "Could not evaluate primitive " ++ Text.unpack (toAbsoluteName (nameInfo nm))
   64         , "On argument " ++ show (length env)
   65         , Text.unpack msg
   66         ]

   89 vBool :: RME -> RValue
   90 vBool b = VBool b

   92 toBool :: RValue -> RME
   93 toBool (VBool b) = b
   94 toBool x = error $ unwords ["SAWCore.Simulator.RME.toBool", show x]

   96 vWord :: Vector RME -> RValue
   97 vWord x = VWord x

   99 toWord :: RValue -> Vector RME
  100 toWord (VWord x) = x
  101 toWord (VVector vv) = fmap (toBool . runIdentity . force) vv
  102 toWord x = error $ unwords ["SAWCore.Simulator.RME.toWord", show x]

  104 vStream :: IntTrie RValue -> RValue
  105 vStream x = VExtra (AStream x)
  106 
  107 toStream :: RValue -> IntTrie RValue
  108 toStream (VExtra (AStream x)) = x
  109 toStream x = error $ unwords ["SAWCore.Simulator.RME.toStream", show x]

  111 wordFun :: (Vector RME -> RPrim) -> RPrim
  112 wordFun f = Prims.strictFun (\x -> f (toWord x))

  114 genShift :: (a -> b -> b -> b) -> (b -> Integer -> b) -> b -> Vector a -> b
  115 genShift cond f x0 v = go x0 (V.toList v)
  116   where
  117     go x [] = x
  118     go x (y : ys) = go (cond y (f x (2 ^ length ys)) x) ys
  119 
  120 -- | op : (w : Nat) -> Vec w Bool -> Nat -> Vec w Bool;
  121 bvShiftOp :: (Vector RME -> Integer -> Vector RME) -> RPrim
  122 bvShiftOp op =
  123   Prims.constFun $
  124   wordFun $ \x ->
  125   Prims.strictFun $ \y ->
  126   Prims.PrimValue $
  127     case y of
  128       VNat n   -> vWord (op x (toInteger n))
  129       VBVToNat _sz v -> vWord (genShift muxRMEV op x (toWord v))
  130       VIntToNat _i   -> error "RME.shiftOp: intToNat TODO"
  131       _        -> error $ unwords ["SAWCore.Simulator.RME.shiftOp", show y]

  135 pure1 :: Applicative f => (a -> b) -> a -> f b
  136 pure1 f x = pure (f x)
  137 
  138 pure2 :: Applicative f => (a -> b -> c) -> a -> b -> f c
  139 pure2 f x y = pure (f x y)
  140 
  141 pure3 :: Applicative f => (a -> b -> c -> d) -> a -> b -> c -> f d
  142 pure3 f x y z = pure (f x y z)

  264 -- | Recursor overrides for the SAWCore simulator.
  265 recursor :: Name -> sort -> Maybe (Identity RValue)
  266 recursor nm _sort =
  267   case nameInfo nm of
  268     ModuleIdentifier "Prelude.Stream" -> Just (pure streamRecOp)
  269     _ -> Nothing

  279 -- primitive intToBv : (n : Nat) -> Integer -> Vec n Bool;
  280 intToBvOp :: RPrim
  281 intToBvOp =
  282   Prims.natFun $ \n ->
  283   Prims.intFun $ \x -> Prims.PrimValue $
  284     VWord (V.reverse (V.generate (fromIntegral n) (RME.constant . testBit x)))

  286 muxRMEV :: RME -> Vector RME -> Vector RME -> Vector RME
  287 muxRMEV b = V.zipWith (RME.mux b)
  288 
  289 muxInt :: RME -> Integer -> Integer -> Integer
  290 muxInt b x y =
  291   case RME.isBool b of
  292     Just c -> if c then x else y
  293     Nothing -> if x == y then x else error $ "muxRValue: VInt " ++ show (x, y)
  294 
  295 muxExtra :: RME -> RExtra -> RExtra -> RExtra
  296 muxExtra b (AStream xs) (AStream ys) =
  297   AStream (muxRValue b <$> xs <*> ys)
  298 
  299 muxRValue :: RME -> RValue -> RValue -> RValue
  300 muxRValue b x y = runIdentity $ Prims.muxValue prims b x y
  301 
  302 -- | Signed shift right simply copies the high order bit
  303 --   into the shifted places.  We special case the zero
  304 --   length vector to avoid a possible out-of-bounds error.
  305 vSignedShiftR :: V.Vector a -> Integer -> V.Vector a
  306 vSignedShiftR xs i
  307   | V.length xs > 0 = Prims.vShiftR x xs i
  308   | otherwise       = xs
  309  where x = xs V.! 0

  347 -- MkStream :: (a :: sort 0) -> (Nat -> a) -> Stream a;
  348 mkStreamOp :: RPrim
  349 mkStreamOp =
  350   Prims.constFun $
  351   Prims.strictFun $ \f ->
  352     Prims.PrimValue (vStream (fmap (\n -> runIdentity (apply f (ready (VNat n)))) IntTrie.identity))
  353 
  354 -- streamGet :: (a :: sort 0) -> Stream a -> Nat -> a;
  355 streamGetOp :: RPrim
  356 streamGetOp =
  357   Prims.tvalFun   $ \_tp ->
  358   Prims.strictFun $ \xs ->
  359   Prims.strictFun $ \ix -> Prims.Prim $ streamGet xs ix
  360 
  361 streamGet :: RValue -> RValue -> Identity RValue
  362 streamGet xs ix =
  363   case ix of
  364     VNat n -> pure $ IntTrie.apply (toStream xs) (toInteger n)
  365     VIntToNat _i -> error "RME.streamGetOp : symbolic integer TODO"
  366     VBVToNat _sz bv ->
  367       do let trie = toStream xs
  368              loop k [] = IntTrie.apply trie k
  369              loop k (b:bs)
  370                | Just True <- RME.isBool b
  371                = loop k1 bs
  372                | Just False <- RME.isBool b
  373                = loop k0 bs
  374                | otherwise
  375                = muxRValue b (loop k1 bs) (loop k0 bs)
  376               where
  377                k0 = k `shiftL` 1
  378                k1 = k0 + 1
  379          pure $ loop (0::Integer) (V.toList (toWord bv))
  380 
  381     _ -> panic "streamGetOp" ["Expected Nat value; found " <> Text.pack (show ix)]
  382 
  383 -- Stream#rec :
  384 --   (a : sort 0) -> (p : Stream a -> sort 0) ->
  385 --   ((f : Nat -> a) -> p (MkStream a f)) -> (str : Stream a) -> p str
  386 streamRecOp :: RValue
  387 streamRecOp =
  388   VFun $ \_a -> pure $
  389   VFun $ \_p -> pure $
  390   vStrictFun $ \f1 -> pure $
  391   vStrictFun $ \xs ->
  392   do let f = vStrictFun $ \ix -> streamGet xs ix
  393      apply f1 (ready f)

  398 newVars :: FiniteType -> State Int RValue
  399 newVars FTBit = do
  400   i <- get
  401   put (i + 1)
  402   return (vBool (RME.lit i))
  403 newVars (FTVec n t) = VVector <$> V.replicateM (fromIntegral n) (newVars' t)
  404 newVars (FTTuple ts) = vTuple <$> traverse newVars' ts
  405 newVars (FTRec tm) = vRecord <$> traverse newVars' tm
  406 
  407 newVars' :: FiniteType -> State Int RThunk
  408 newVars' shape = ready <$> newVars shape

  411 -- Bit-blasting primitives.
  412 
  413 bitBlastBasic :: ModuleMap
  414               -> Map Ident RPrim
  415               -> Map VarIndex RValue
  416               -> Term
  417               -> RValue
  418 bitBlastBasic m addlPrims varMap t = runIdentity $ do
  419   let primHandler nm msg env =
  420          return $ Prim.userError $ unlines
  421            [ "Could not evaluate primitive " ++ Text.unpack (toAbsoluteName (nameInfo nm))
  422            , "On argument " ++ show (length env)
  423            , Text.unpack msg
  424            ]
  425   cfg <- Sim.evalGlobal m (Map.union constMap addlPrims)
  426          (\vn _ -> case Map.lookup (vnIndex vn) varMap of
  427                    Just v -> pure v
  428                    Nothing -> error ("RME: unknown variable: " ++ show (vnName vn)))
  429          (\_ _ -> Nothing)
  430          recursor
  431          primHandler
  432          (Prims.lazyMuxValue prims)
  433   Sim.evalSharedTerm cfg t

  436 processVar ::
  437   (VarName, FirstOrderType) ->
  438   IO (VarName, FiniteType)
  439 processVar (vn, fot) =
  440   case toFiniteType fot of
  441     Nothing -> fail ("RME solver does not support variables of type " ++ show fot)
  442     Just ft -> pure (vn, ft)

  444 withBitBlastedSATQuery ::
  445   SharedContext ->
  446   Map Ident RPrim ->
  447   SATQuery ->
  448   (RME -> [(VarName, FiniteType)] -> IO a) ->
  449   IO a
  450 withBitBlastedSATQuery sc addlPrims satq cont =
  451   do unless (Set.null (satUninterp satq)) $ fail
  452         "RME prover does not support uninterpreted symbols"
  453      t <- satQueryAsTerm sc satq
  454      varShapes <- mapM processVar (Map.toList (satVariables satq))
  455      modmap <- scGetModuleMap sc
  456      let vars = evalState (traverse (traverse newVars) varShapes) 0
  457      let varMap = Map.fromList [ (vnIndex x, v) | (x, v) <- vars ]
  458      let bval = bitBlastBasic modmap addlPrims varMap t
  459      case bval of
  460        VBool anf -> cont anf varShapes
  461        _ -> panic "withBitBlastedSATQuery" ["Non-boolean result type " <> Text.pack (show bval)]
