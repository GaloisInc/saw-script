   33 -- Term Datatype ---------------------------------------------------------------
   34 
   35 type TermIndex = Int -- Word64
   36 
   37 -- | For more information on the semantics of 'Term's, see the
   38 -- [manual](https://saw.galois.com/manual.html). 'Term' and 'TermF' are split
   39 -- into two structures to facilitate mutual structural recursion (sometimes
   40 -- referred to as the ["knot-tying"](https://wiki.haskell.org/Tying_the_Knot)
   41 -- pattern, sometimes referred to in terms of ["recursion
   42 -- schemes"](https://blog.sumtypeofway.com/posts/introduction-to-recursion-schemes.html))
   43 -- and term object reuse via hash-consing.
   44 data Term
   45   = STApp
   46     -- ^ This constructor \"wraps\" a 'TermF' 'Term', assigning it a
   47     -- guaranteed-unique integer identifier and caching its likely-unique hash.
   48     -- 'Term's are constructed via 'STApp'. When a fresh 'TermF' is evinced
   49     -- in the course of a SAW invocation and needs to be lifted into a 'Term',
   50     -- we can see if we've already created a 'Term' wrapper for an identical
   51     -- 'TermF', and reuse it if so. The implementation of hash-consed 'Term'
   52     -- construction exists in 'SAWCore.SharedTerm', in particular in the
   53     -- 'SAWCore.SharedTerm.scTermF' field of the
   54     -- t'SAWCore.SharedTerm.SharedContext' object.
   55      { stAppIndex    :: {-# UNPACK #-} !TermIndex
   56        -- ^ The UID associated with a 'Term'. It is guaranteed unique across a
   57        -- universe of properly-constructed 'Term's within a single SAW
   58        -- invocation.
   59      , stAppHash     :: {-# UNPACK #-} !Int
   60        -- ^ The hash, according to 'hash', of the 'stAppTermF' field associated
   61        -- with this 'Term'. This should be as unique as a hash can be, but is
   62        -- not guaranteed unique as 'stAppIndex' is.
   63      , stAppVarTypes :: !(IntMap Term)
   64        -- ^ A map relating the 'VarIndex' of each free 'Variable' in
   65        -- the term to the type attached to the 'Variable' constructor.
   66        -- As an invariant, all free occurrences of the same variable
   67        -- must be tagged with the same type.
   68      , stAppTermF    :: !(TermF Term)
   69        -- ^ The underlying 'TermF' that this 'Term' wraps. This field "ties the
   70        -- knot" of the 'Term'/'TermF' recursion scheme.
   71      , stAppType     :: !(Either Sort Term)
   72        -- ^ The type of this term, represented a 'Sort' or another 'Term'.
   73        -- Invariant: This field must never contain a 'Right' with a term
   74        -- that is a 'Sort'; if the type is a 'Sort' then 'Left' is required.
   75        -- Making 'Sort' a special case lets us avoid storing an infinite
   76        -- tower of types inside every term.
   77      }
   78   deriving (Show)
   79 
   80 instance Hashable Term where
   81   -- The hash of an 'STApp' depends on its not-necessarily-unique
   82   -- 'stAppHash' instead of its necessarily-unique 'stAppIndex'.
   83   -- The reason is that per #1830 (PR) and #1831 (issue), we want to
   84   -- to derive references to terms based solely on their shape.
   85   -- Indices have nothing to do with a term's shape - they're assigned
   86   -- sequentially when building terms, according to the (arbitrary)
   87   -- order in which a term is built.
   88   -- As for uniqueness, though hashing a term based on its subterms'
   89   -- hashes introduces less randomness/freshness, it maintains plenty,
   90   -- and provides benefits as described above.
   91   -- No code should ever rely on total uniqueness of hashes, and terms
   92   -- are no exception.
   93   --
   94   -- Note: Nevertheless, we do take some minor liberties with the
   95   -- contract of 'hashWithSalt'. The contract states that if two
   96   -- values are equal according to '(==)', then they must have the
   97   -- same hash.
   98   -- For terms constructed by/within SAW, this will hold, because
   99   -- SAW's handling of index generation and assignment ensures that
  100   -- equality of indices implies equality of terms and term hashes
  101   -- (see 'SAWCore.SharedTerm.getTerm').
  102   -- However, if terms are constructed outside this standard procedure
  103   -- or in a way that does not respect index uniqueness rules,
  104   -- 'hashWithSalt''s contract could be violated.
  105   hash STApp{ stAppHash = h } = h
  106   hashWithSalt salt = hashWithSalt salt . hash
  107 
  108 instance Eq Term where
  109   (==) = equalTerm
  110 
  111 equalTerm :: Term -> Term -> Bool
  112 equalTerm (STApp{stAppIndex = i1, stAppHash = h1, stAppTermF = tf1, stAppType = mty1})
  113           (STApp{stAppIndex = i2, stAppHash = h2, stAppTermF = tf2, stAppType = mty2}) =
  114   i1 == i2 || (h1 == h2 && tf1 == tf2 && mty1 == mty2)
  115   -- The hash check (^) is merely an optimization that enables us to
  116   -- quickly return 'False' in most cases. Since we're assuming the
  117   -- contract of 'hashWithSalt' holds, then we know @tf1 == tf2@
  118   -- implies @h1 == h2@. Thus we could safely remove @h1 == h2@ without
  119   -- changing the behavior of this function, but keeping it in enables
  120   -- us to utilize the fact that we save 'STApp' hashes to get away
  121   -- with not traversing the 'stAppTermF' fields in most cases of
  122   -- inequality.
  123 
  124 -- | Return 'True' iff the given terms are equal modulo alpha equivalence (i.e.
  125 -- 'VarName's in 'Lambda' and 'Pi' expressions).
  126 -- The types of the terms are not inspected.
  127 alphaEquiv :: Term -> Term -> Bool
  128 alphaEquiv = term IntMap.empty
  129   where
  130     term :: IntMap VarIndex -> Term -> Term -> Bool
  131     term vm
  132       (STApp{stAppIndex = i1, stAppTermF = tf1, stAppVarTypes = vt1})
  133       (STApp{stAppIndex = i2, stAppTermF = tf2}) =
  134       (IntMap.disjoint vt1 vm && i1 == i2) || termf vm tf1 tf2
  135 
  136     termf :: IntMap VarIndex -> TermF Term -> TermF Term -> Bool
  137     termf vm (FTermF ftf1) (FTermF ftf2) = ftermf vm ftf1 ftf2
  138     termf vm (App t1 u1) (App t2 u2) = term vm t1 t2 && term vm u1 u2
  139     termf vm (Lambda (vnIndex -> i1) t1 u1) (Lambda (vnIndex -> i2) t2 u2) =
  140       let vm' = if i1 == i2 then vm else IntMap.insert i1 i2 vm
  141       in term vm t1 t2 && term vm' u1 u2
  142     termf vm (Pi (vnIndex -> i1) t1 u1) (Pi (vnIndex -> i2) t2 u2) =
  143       let vm' = if i1 == i2 then vm else IntMap.insert i1 i2 vm
  144       in term vm t1 t2 && term vm' u1 u2
  145     termf _vm (Constant x1) (Constant x2) = x1 == x2
  146     termf vm (Variable x1 _t1) (Variable x2 _t2) =
  147       case IntMap.lookup (vnIndex x1) vm of
  148         Just i -> vnIndex x2 == i
  149         Nothing -> x1 == x2
  150     termf _ FTermF{}   _ = False
  151     termf _ App{}      _ = False
  152     termf _ Lambda{}   _ = False
  153     termf _ Pi{}       _ = False
  154     termf _ Constant{} _ = False
  155     termf _ Variable{} _ = False
  156 
  157     ftermf :: IntMap Int -> FlatTermF Term -> FlatTermF Term -> Bool
  158     ftermf vm ftf1 ftf2 =
  159       case zipWithFlatTermF (term vm) ftf1 ftf2 of
  160         Nothing -> False
  161         Just ftf3 -> Foldable.and ftf3
  162 
  163 unwrapTermF :: Term -> TermF Term
  164 unwrapTermF STApp{stAppTermF = tf} = tf
  165 
  166 termIndex :: Term -> TermIndex
  167 termIndex STApp{ stAppIndex = i } = i
  168 
  169 termSortOrType :: Term -> Either Sort Term
  170 termSortOrType STApp{ stAppType = ty } = ty
  171 
  172 instance Ord Term where
  173   compare (STApp{stAppIndex = i}) (STApp{stAppIndex = j}) | i == j = EQ
  174   compare x y = compare (unwrapTermF x) (unwrapTermF y)

  176 -- Free Variables --------------------------------------------------------------
  177 
  178 -- | Return an 'IntMap' relating the 'VarIndex' of each free variable
  179 -- of a term to its type.
  180 varTypes :: Term -> IntMap Term
  181 varTypes STApp{ stAppVarTypes = vt } = vt
  182 
  183 -- | Return an 'IntSet' containing the 'VarIndex' of all free
  184 -- variables in the 'Term'.
  185 freeVars :: Term -> IntSet
  186 freeVars t = IntMap.keysSet (varTypes t)
  187 
  188 -- | Test whether a 'Term' is closed, i.e., it has no free variables.
  189 closedTerm :: Term -> Bool
  190 closedTerm t = IntMap.null (varTypes t)
