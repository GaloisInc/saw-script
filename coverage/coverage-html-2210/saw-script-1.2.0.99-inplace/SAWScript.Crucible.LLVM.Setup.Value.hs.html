  184 -- *** LLVMModule
  185 
  186 -- | An 'LLVMModule' contains an LLVM module that has been parsed from
  187 -- a bitcode file and translated to Crucible.
  188 data LLVMModule arch =
  189   LLVMModule
  190   { _modFilePath :: FilePath
  191   , _modAST :: L.Module
  192   , _modTrans :: CL.ModuleTranslation arch
  193   }
  194 -- NOTE: Type 'LLVMModule' is exported as an abstract type, and we
  195 -- maintain the invariant that the 'FilePath', 'Module', and
  196 -- 'ModuleTranslation' fields are all consistent with each other;
  197 -- 'loadLLVMModule' is the only function that is allowed to create
  198 -- values of type 'LLVMModule'.
  199 
  200 -- | The file path that the LLVM module was loaded from.
  201 modFilePath :: LLVMModule arch -> FilePath
  202 modFilePath = _modFilePath
  203 
  204 -- | The parsed AST of the LLVM module.
  205 modAST :: LLVMModule arch -> L.Module
  206 modAST = _modAST
  207 
  208 -- | The Crucible translation of an LLVM module.
  209 modTrans :: LLVMModule arch -> CL.ModuleTranslation arch
  210 modTrans = _modTrans
  211 
  212 -- | Load an LLVM module from the given bitcode file, then parse and
  213 -- translate to Crucible.
  214 loadLLVMModule ::
  215   (?transOpts :: CL.TranslationOptions) =>
  216   FilePath ->
  217   Crucible.HandleAllocator ->
  218   IO (Either LLVM.Error (Some LLVMModule))
  219 loadLLVMModule file halloc =
  220   do parseResult <- LLVM.parseBitCodeFromFile file
  221      case parseResult of
  222        Left err -> return (Left err)
  223        Right llvm_mod ->
  224          do memVar <- CL.mkMemVar (Text.pack "saw:llvm_memory") halloc
  225             Some mtrans <- CL.translateModule halloc memVar llvm_mod
  226             return (Right (Some (LLVMModule file llvm_mod mtrans)))
  227 
  228 instance TestEquality LLVMModule where
  229   -- As 'LLVMModule' is an abstract type, we know that the values must
  230   -- have been created by a call to 'loadLLVMModule'. Furthermore each
  231   -- call to 'translateModule' generates a 'ModuleTranslation' that
  232   -- contains a fresh nonce; thus comparison of the 'modTrans' fields
  233   -- is sufficient to guarantee equality of two 'LLVMModule' values.
  234   testEquality m1 m2 = testEquality (modTrans m1) (modTrans m2)
  235 
  236 type instance Setup.Codebase (LLVM arch) = LLVMModule arch

  238 showLLVMModule :: LLVMModule arch -> String
  239 showLLVMModule (LLVMModule name m _) =
  240   unlines [ "Module: " ++ name
  241           , "Types:"
  242           , showParts (Crucible.LLVM.ppLLVMLatest L.ppTypeDecl) (L.modTypes m)
  243           , "Globals:"
  244           , showParts (Crucible.LLVM.ppLLVMLatest ppGlobal') (L.modGlobals m)
  245           , "External references:"
  246           , showParts Crucible.LLVM.ppDeclare (L.modDeclares m)
  247           , "Definitions:"
  248           , showParts (Crucible.LLVM.ppLLVMLatest ppDefine') (L.modDefines m)
  249           ]
  250   where
  251     showParts pp xs = unlines $ map (show . PP.nest 2 . pp) xs
  252     ppGlobal' g =
  253       L.ppSymbol (L.globalSym g) PP.<+> PP.char '=' PP.<+>
  254       L.ppGlobalAttrs (isJust $ L.globalValue g) (L.globalAttrs g) PP.<+>
  255       L.ppType (L.globalType g)
  256     ppDefine' d =
  257       L.ppMaybe L.ppLinkage (L.defLinkage d) PP.<+>
  258       L.ppType (L.defRetType d) PP.<+>
  259       L.ppSymbol (L.defName d) PP.<>
  260       L.ppArgList (L.defVarArgs d) (map (L.ppTyped L.ppIdent) (L.defArgs d)) PP.<+>
  261       L.ppMaybe (\gc -> PP.text "gc" PP.<+> L.ppGC gc) (L.defGC d)

  315 data ResolvedPathItem
  316   = ResolvedField String
  317   | ResolvedElem Int
  318   | ResolvedCast L.Type
  319  deriving (Show, Eq, Ord)
  320 
  321 type ResolvedPath = [ResolvedPathItem]
  322 
  323 -- | A datatype to keep track of which parts of the simulator state
  324 -- have been initialized already. For each allocation unit or global,
  325 -- we keep a list of element-paths that identify the initialized
  326 -- sub-components.
  327 --
  328 -- Note that the data collected and maintained by this datatype
  329 -- represents a \"best-effort\" check that attempts to prevent
  330 -- the user from stating unsatisfiable method specifications.
  331 --
  332 -- It will not prevent all cases of overlapping points-to
  333 -- specifications, especially in the presence of pointer casts.
  334 -- A typical result of overlapping specifications will be
  335 -- successful (vacuous) verifications of functions resulting in
  336 -- overrides that cannot be used at call sites (as their
  337 -- preconditions are unsatisfiable).
  338 data LLVMResolvedState =
  339   ResolvedState
  340     { _rsAllocs :: Map Setup.AllocIndex [ResolvedPath]
  341     , _rsGlobals :: Map String [ResolvedPath]
  342     }
  343   deriving (Eq, Ord, Show)
  344 
  345 emptyResolvedState :: LLVMResolvedState
  346 emptyResolvedState = ResolvedState Map.empty Map.empty
  347 
  348 makeLenses ''LLVMResolvedState
