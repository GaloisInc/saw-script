   21 -- Natural numbers

   23 -- | width(n) = 1 + floor(log_2(n))
   24 widthNat :: Natural -> Natural
   25 widthNat 0 = 0
   26 widthNat n = 1 + widthNat (n `div` 2)

   29 data Vec t a = Vec t !(Vector a)

   31 ------------------------------------------------------------
   32 -- Unsigned, variable-width bit vectors
   33 
   34 data BitVector = BV { width :: !Int, unsigned :: !Integer }
   35     deriving Show
   36 -- ^ Invariant: BV w x requires that 0 <= x < 2^w.
   37 
   38 bitMask :: Int -> Integer
   39 bitMask w = bit w - 1
   40 
   41 -- | Smart constructor for bitvectors.
   42 bv :: Int -> Integer -> BitVector
   43 bv w x = BV w (x .&. bitMask w)
   44 
   45 signed :: BitVector -> Integer
   46 signed (BV w x)
   47   | w > 0 && testBit x (w - 1) = x - bit w
   48   | otherwise                  = x
   49 
   50 bvAt :: BitVector -> Int -> Bool
   51 bvAt (BV w x) i = testBit x (w - 1 - i)
   52 
   53 -- | Conversion from list of bits to integer (big-endian)
   54 bvToInteger :: Vector Bool -> Integer
   55 bvToInteger = V.foldl' (\x b -> if b then 2*x+1 else 2*x) 0
   56 
   57 unpackBitVector :: BitVector -> Vector Bool
   58 unpackBitVector x = V.generate (width x) (bvAt x)
   59 
   60 packBitVector :: Vector Bool -> BitVector
   61 packBitVector v = BV (V.length v) (bvToInteger v)

   64 -- Primitive operations
   65 
   66 -- NOTE: Many of the primitive operations in this section are written
   67 -- with arguments of unit type '()'.
   68 -- This is done so that the Haskell function arguments line up in a
   69 -- 1-to-1 correspondence with the SAWCore function arguments.
   70 -- Type classes are used to generate matching code for SAWCore terms
   71 -- based on the Haskell type.
   72 -- SAWCore uses a binary representation for numeric literals that has
   73 -- a non-trivial cost to match against; for this reason we use '()'
   74 -- instead of 'Int' or 'Natural' or 'Integer' for all unused arguments
   75 -- of SAWCore type 'Nat'.
   76 
   77 -- coerce :: (y x :: sort 0) -> Eq (sort 0) x y -> x -> y;
   78 coerce :: () -> () -> () -> a -> a
   79 coerce _ _ _ x = x
   80 
   81 -- ite :: ?(a :: sort 1) -> Bool -> a -> a -> a;
   82 ite :: t -> Bool -> a -> a -> a
   83 ite _ b x y = if b then x else y
   84 
   85 -- Succ :: Nat -> Nat;
   86 succNat :: Integer -> Integer
   87 succNat = succ
   88 
   89 -- addNat :: Nat -> Nat -> Nat;
   90 addNat :: Integer -> Integer -> Integer
   91 addNat = (+)

   93 -- get :: (n :: Nat) -> (e :: sort 0) -> Vec n e -> Fin n -> e;
   94 --get :: Int -> t -> Vec t e -> Fin -> e
   95 --get _ _ (Vec _ v) i = v ! fromEnum i
   96 
   97 -- append :: (m n :: Nat) -> (e :: sort 0) -> Vec m e -> Vec n e -> Vec (addNat m n) e;
   98 append :: Int -> Int -> t -> Vec t e -> Vec t e -> Vec t e
   99 append _ _ _ (Vec t xv) (Vec _ yv) = Vec t ((V.++) xv yv)
  100 
  101 -- at :: (n :: Nat) -> (a :: sort 0) -> Vec n a -> Nat -> a;
  102 at :: () -> () -> Vec t e -> Int -> Maybe e
  103 at _ _ (Vec _ v) i = v V.!? i
  104 
  105 -- atWithDefault :: (n :: Nat) -> (a :: sort 0) -> a -> Vec n a -> Nat -> a;
  106 atWithDefault :: () -> () -> e -> Vec t e -> Int -> e
  107 atWithDefault _ _ z (Vec _ v) i =
  108   case v V.!? i of
  109     Just y -> y
  110     Nothing -> z
  111 
  112 -- upd :: (n :: Nat) -> (a :: sort 0) -> Vec n a -> Nat -> a -> Vec n a;
  113 upd :: () -> t -> Vec t e -> Int -> e -> Vec t e
  114 upd _ _ (Vec t v) i e = Vec t (v V.// [(i, e)])

  117 -- Bitvector operations
  118 
  119 -- bvNat : (n : Nat) -> Nat -> Vec n Bool;
  120 bvNat :: Int -> Integer -> BitVector
  121 bvNat w x = bv w x
  122 
  123 -- bvAdd : (n : Nat) -> Vec n Bool -> Vec n Bool -> Vec n Bool;
  124 bvAdd, bvSub, bvMul :: Natural -> BitVector -> BitVector -> BitVector
  125 bvAdd _ (BV w x) (BV _ y) = bv w (x + y)
  126 bvSub _ (BV w x) (BV _ y) = bv w (x - y)
  127 bvMul _ (BV w x) (BV _ y) = bv w (x * y)
  128 
  129 bvNeg :: Natural -> BitVector -> BitVector
  130 bvNeg _ x@(BV w _) = bv w $ negate $ signed x
  131 
  132 bvAnd, bvOr, bvXor :: Int -> BitVector -> BitVector -> BitVector
  133 bvAnd _ (BV w x) (BV _ y) = BV w (x .&. y)
  134 bvOr  _ (BV w x) (BV _ y) = BV w (x .|. y)
  135 bvXor _ (BV w x) (BV _ y) = BV w (x `xor` y)
  136 
  137 bvNot :: Int -> BitVector -> BitVector
  138 bvNot _ (BV w x) = BV w (x `xor` bitMask w)
  139 
  140 bvEq, bvult, bvule, bvugt, bvuge, bvsgt, bvsge, bvslt, bvsle
  141     :: () -> BitVector -> BitVector -> Bool
  142 bvEq  _ x y = unsigned x == unsigned y
  143 bvugt _ x y = unsigned x >  unsigned y
  144 bvuge _ x y = unsigned x >= unsigned y
  145 bvult _ x y = unsigned x <  unsigned y
  146 bvule _ x y = unsigned x <= unsigned y
  147 bvsgt _ x y = signed x >  signed y
  148 bvsge _ x y = signed x >= signed y
  149 bvslt _ x y = signed x <  signed y
  150 bvsle _ x y = signed x <= signed y
  151 
  152 bvPopcount :: () -> BitVector -> BitVector
  153 bvPopcount _ (BV w x) = BV w (toInteger (popCount x))
  154 
  155 bvCountLeadingZeros :: () -> BitVector -> BitVector
  156 bvCountLeadingZeros _ (BV w x) = BV w (toInteger (go 0))
  157  where
  158  go !i
  159    | i < w && testBit x (w - i - 1) == False = go (i+1)
  160    | otherwise = i
  161 
  162 bvCountTrailingZeros :: () -> BitVector -> BitVector
  163 bvCountTrailingZeros _ (BV w x) = BV w (toInteger (go 0))
  164  where
  165  go !i
  166    | i < w && testBit x i == False = go (i+1)
  167    | otherwise = i
  168 
  169 -- | @get@ specialized to BitVector (big-endian)
  170 -- get :: (n :: Nat) -> (a :: sort 0) -> Vec n a -> Fin n -> a;
  171 --get_bv :: Int -> () -> BitVector -> Fin -> Bool
  172 --get_bv _ _ x i = testBit (unsigned x) (width x - 1 - fromEnum i)
  173 -- little-endian version:
  174 -- get_bv _ _ x i = testBit (unsigned x) (fromEnum i)

  176 -- | @at@ specialized to BitVector (big-endian)
  177 -- at :: (n :: Nat) -> (a :: sort 0) -> Vec n a -> Nat -> a;
  178 at_bv :: () -> () -> BitVector -> Int -> Maybe Bool
  179 at_bv _ _ x i
  180   | i < width x = Just (testBit (unsigned x) (width x - 1 - i))
  181   | otherwise = Nothing

  187 --  where f = if b then setBit else clearBit

  189 -- | @append@ specialized to BitVector (big-endian)
  190 -- append :: (m n :: Nat) -> (a :: sort 0) -> Vec m a -> Vec n a -> Vec (addNat m n) a;
  191 append_bv :: () -> () -> () -> BitVector -> BitVector -> BitVector
  192 append_bv _ _ _ (BV m x) (BV n y) = BV (m + n) (shiftL x n .|. y)
  193 -- little-endian version:
  194 -- append_bv _ _ _ (BV m x) (BV n y) = BV (m + n) (x .|. shiftL y m)
  195 
  196 -- bvToNat : (n : Nat) -> Vec n Bool -> Nat;
  197 bvToNat :: () -> BitVector -> Integer
  198 bvToNat _ (BV _ x) = x
  199 
  200 -- bvAddWithCarry : (n : Nat) -> Vec n Bool -> Vec n Bool -> Bool * Vec n Bool;
  201 bvAddWithCarry :: () -> BitVector -> BitVector -> (Bool, BitVector)
  202 bvAddWithCarry _ (BV w x) (BV _ y) = (testBit z w, bv w z)
  203     where z = x + y
  204 
  205 bvUDiv :: () -> BitVector -> BitVector -> Maybe BitVector
  206 bvUDiv _ (BV w x) (BV _ y)
  207   | y == 0    = Nothing
  208   | otherwise = Just (bv w (x `quot` y))
  209 
  210 bvURem :: () -> BitVector -> BitVector -> Maybe BitVector
  211 bvURem _ (BV w x) (BV _ y)
  212   | y == 0    = Nothing
  213   | otherwise = Just (bv w (x `rem` y))
  214 
  215 bvSDiv :: () -> BitVector -> BitVector -> Maybe BitVector
  216 bvSDiv _ x y
  217   | unsigned y == 0 = Nothing
  218   | otherwise       = Just (bv (width x) (signed x `quot` signed y))
  219 
  220 bvSRem :: () -> BitVector -> BitVector -> Maybe BitVector
  221 bvSRem _ x y
  222   | unsigned y == 0 = Nothing
  223   | otherwise       = Just (bv (width x) (signed x `rem` signed y))
  224 
  225 bvShl :: () -> BitVector -> Int -> BitVector
  226 bvShl _ (BV w x) i = bv w (x `shiftL` i)
  227 
  228 bvShr :: () -> BitVector -> Int -> BitVector
  229 bvShr _ (BV w x) i = bv w (x `shiftR` i)
  230 
  231 bvSShr :: () -> BitVector -> Int -> BitVector
  232 bvSShr _ x i = bv (width x) (signed x `shiftR` i)
  233 
  234 -- bvTrunc : (m n : Nat) -> Vec (addNat m n) Bool -> Vec n Bool;
  235 bvTrunc :: () -> Int -> BitVector -> BitVector
  236 bvTrunc _ n (BV _ x) = bv n x
  237 
  238 -- bvUExt : (m n : Nat) -> Vec n Bool -> Vec (addNat m n) Bool;
  239 bvUExt :: Int -> Int -> BitVector -> BitVector
  240 bvUExt m n x = BV (m + n) (unsigned x)
  241 
  242 -- bvSExt : (m n : Nat) -> Vec (Succ n) Bool -> Vec (addNat m (Succ n)) Bool;
  243 bvSExt :: Int -> Int -> BitVector -> BitVector
  244 bvSExt m n x = bv (m + n + 1) (signed x)

  246 -- | @take@ specialized to BitVector (big-endian)
  247 -- take :: (a :: sort 0) -> (m n :: Nat) -> Vec (addNat m n) a -> Vec m a;
  248 take_bv :: () -> Int -> Int -> BitVector -> BitVector
  249 take_bv _ m n (BV _ x) = bv m (x `shiftR` n)
  250 -- little-endian version:
  251 -- take_bv _ m _ (BV _ x) = bv m x
  252 
  253 -- | @vDrop@ specialized to BitVector (big-endian)
  254 -- drop :: (a :: sort 0) -> (m n :: Nat) -> Vec (addNat m n) a -> Vec n a;
  255 drop_bv :: () -> () -> Int -> BitVector -> BitVector
  256 drop_bv _ _ n (BV _ x) = bv n x
  257 -- little-endian version:
  258 -- drop_bv _ m n (BV _ x) = BV n (x `shiftR` m)
  259 
  260 -- | @slice@ specialized to BitVector
  261 slice_bv :: () -> Int -> Int -> Int -> BitVector -> BitVector
  262 slice_bv _ _ n o (BV _ x) = bv n (shiftR x o)
  263 -- little-endian version:
  264 -- slice_bv _ i n _ (BV _ x) = bv n (shiftR x i)

  266 ------------------------------------------------------------
  267 -- Base 2 logarithm
  268 
  269 bvLg2 :: BitVector -> BitVector
  270 bvLg2 (BV m x) = BV m (if d > 0 then k+1 else k)
  271   where (k, d) = lg2rem x
  272 
  273 -- | lg2rem n = (k, d) <--> n = 2^k + d, with d < 2^k.
  274 lg2rem :: Integer -> (Integer, Integer)
  275 lg2rem 0 = (0, -1)
  276 lg2rem 1 = (0, 0)
  277 lg2rem n = (k+1, 2*d+r)
  278   where (q, r) = n `divMod` 2
  279         (k, d) = lg2rem q

  282 -- BitVector shift/rotate
  283 
  284 bvRotateL :: BitVector -> Integer -> BitVector
  285 bvRotateL (BV w x) i = bv w ((x `shiftL` j) .|. (x `shiftR` (w - j)))
  286   where j = fromInteger (i `mod` toInteger w)
  287 
  288 bvRotateR :: BitVector -> Integer -> BitVector
  289 bvRotateR w i = bvRotateL w (- i)
  290 
  291 bvShiftL ::
  292   Bool {- ^ bit value to shift in -} ->
  293   BitVector {- ^ value to shift -} ->
  294   Integer {- ^ amount to shift by -} ->
  295   BitVector
  296 bvShiftL c (BV w x) i = bv w ((x `shiftL` j) .|. c')
  297   where c' = if c then (1 `shiftL` j) - 1 else 0
  298         j = fromInteger (i `min` toInteger w)
  299 
  300 bvShiftR ::
  301   Bool {- ^ bit value to shift in -} ->
  302   BitVector {- ^ value to shift -} ->
  303   Integer {- ^ amount to shift by -} ->
  304   BitVector
  305 bvShiftR c (BV w x) i = bv w (c' .|. (x `shiftR` j))
  306   where c' = if c then (full `shiftL` (w - j)) .&. full else 0
  307         full = (1 `shiftL` w) - 1
  308         j = fromInteger (i `min` toInteger w)

  312 -- Errors
  313 
  314 data EvalError
  315   = InvalidIndex Integer
  316   | DivideByZero
  317   | UnsupportedPrimitive String String
  318   | UserError String
  319   deriving (Eq)
  320 
  321 instance X.Exception EvalError
  322 
  323 instance Show EvalError where
  324   show e = case e of
  325     InvalidIndex i -> "invalid sequence index: " ++ show i
  326     DivideByZero -> "division by 0"
  327     UnsupportedPrimitive b p -> "unsupported primitive " ++ p ++ " in " ++ b ++ " backend"
  328     UserError msg -> "Run-time error: " ++ msg
  329 
  330 -- | A sequencing operation has gotten an invalid index.
  331 invalidIndex :: Integer -> a
  332 invalidIndex i = X.throw (InvalidIndex i)
  333 
  334 -- | For division by 0.
  335 divideByZero :: a
  336 divideByZero = X.throw DivideByZero
  337 
  338 -- | A backend with a unsupported primitive.
  339 unsupportedPrimitive :: String -> String -> a
  340 unsupportedPrimitive backend primitive =
  341   X.throw $ UnsupportedPrimitive backend primitive
  342 
  343 -- | For `error`
  344 userError :: String -> a
  345 userError msg = X.throw (UserError msg)
  346 
  347 -- | Convert asynchronous EvalError exceptions into IO exceptions.
  348 rethrowEvalError :: IO a -> IO a
  349 rethrowEvalError m = run `X.catch` rethrow
  350   where
  351     run = do
  352       a <- m
  353       return $! a
  354 
  355     rethrow :: EvalError -> IO a
  356     rethrow exn = fail (show exn) -- X.throwIO (EvalError exn)
