   13 module SAWCore.Parser.Position
   14   ( Pos(..)
   15   , ppPos
   16   , incLine
   17   , incCol
   18   , Positioned(..)
   19   , PosPair(..)
   20   ) where
   21 
   22 import qualified Language.Haskell.TH.Syntax as TH
   23 import System.FilePath (makeRelative)
   24 
   25 data Pos = Pos { -- | Base directory to use for pretty printing purposes
   26                  posBase :: !FilePath
   27                , posPath  :: !FilePath
   28                , posLine  :: !Int
   29                , posCol   :: !Int
   30                }
   31   deriving (Show, TH.Lift)
   32 
   33 posTuple :: Pos -> (Int,Int,FilePath)
   34 posTuple x = (posLine x, posCol x, posPath x)
   35 
   36 -- Eq instance  overridden to compare positions in the same file more efficiently.
   37 instance Eq Pos where
   38   x == y = posTuple x == posTuple y
   39 
   40 -- Ord instance  overridden to compare positions in the same file more efficiently.
   41 instance Ord Pos where
   42   compare x y = compare (posTuple x) (posTuple y)
   43 
   44 ppPos :: Pos -> String
   45 ppPos p = rp ++ ':' : show (posLine p) ++ ':' : show (posCol p) ++ ":"
   46   where rp = makeRelative (posBase p) (posPath p)
   47 
   48 incLine :: Pos -> Pos
   49 incLine p = p { posLine = 1 + posLine p, posCol = 0 }
   50 
   51 incCol :: Pos -> Pos
   52 incCol p = p { posCol = 1 + posCol p }
   53 
   54 class Positioned v where
   55   pos :: v -> Pos
   56 
   57 data PosPair v = PosPair { _pos :: !Pos, val :: !v }
   58   deriving (Eq, Ord, Functor, Show, TH.Lift)
   59 
   60 instance Positioned (PosPair v) where
   61   pos (PosPair p _) = p
