  282 -- | Compare pointers that are not valid LLVM pointers. Comparing the offsets
  283 -- as unsigned bitvectors is not sound, because of overflow (e.g. `base - 1` is
  284 -- less than `base`, but -1 is not less than 0 when compared as unsigned). It
  285 -- is safe to allow a small negative offset, because each pointer base is
  286 -- mapped to an address that is not in the first page (4K), which is never
  287 -- mapped on X86_64 Linux. Specifically, assume pointer1 = (base1, offset1) and
  288 -- pointer2 = (base2, offset2), and size1 is the size of the allocation of
  289 -- base1 and size2 is the size of the allocation of base2. If offset1 is in the
  290 -- interval [-4096, size1], and offset2 is in the interval [-4096, size2], then
  291 -- the unsigned comparison between pointer1 and pointer2 is equivalent with the
  292 -- unsigned comparison between offset1 + 4096 and offset2 + 4096.
  293 doPtrCmp ::
  294   FixpointSelect ->
  295   (sym -> W4.SymBV sym w -> W4.SymBV sym w -> IO (W4.Pred sym)) ->
  296   Macaw.PtrOp sym w (C.RegValue sym C.BoolType)
  297 doPtrCmp fixpointSelect f = Macaw.ptrOp $ \bak mem w xPtr xBits yPtr yBits x y -> do
  298   let sym = backendGetSym bak
  299   let ptr_as_bv_for_cmp ptr = do
  300         page_size <- W4.bvLit sym (W4.bvWidth $ llvmPointerOffset ptr) $
  301           BV.mkBV (W4.bvWidth $ llvmPointerOffset ptr) 4096
  302         ptr_as_bv <- W4.bvAdd sym (llvmPointerOffset ptr) page_size
  303         is_valid <- Macaw.isValidPtr sym mem w ptr
  304         is_negative_offset <- W4.bvIsNeg sym (llvmPointerOffset ptr)
  305         is_not_overflow <- W4.notPred sym =<< W4.bvIsNeg sym ptr_as_bv
  306         ok <- W4.orPred sym is_valid
  307           =<< W4.andPred sym is_negative_offset is_not_overflow
  308         return (ptr_as_bv, ok)
  309         -- is_valid <- Macaw.isValidPtr sym mem w ptr
  310         -- return (llvmPointerOffset ptr, is_valid)
  311   both_bits <- W4.andPred sym xBits yBits
  312   both_ptrs <- W4.andPred sym xPtr yPtr
  313   same_region <- W4.natEq sym (llvmPointerBlock x) (llvmPointerBlock y)
  314   (x_ptr_as_bv, ok_x) <- ptr_as_bv_for_cmp x
  315   (y_ptr_as_bv, ok_y) <- ptr_as_bv_for_cmp y
  316   ok_both_ptrs <- W4.andPred sym both_ptrs
  317     =<< W4.andPred sym same_region
  318     =<< W4.andPred sym ok_x ok_y
  319   res_both_bits <- f sym (llvmPointerOffset x) (llvmPointerOffset y)
  320   res_both_ptrs <- f sym x_ptr_as_bv y_ptr_as_bv
  321   let ptrCmpDefault = do
  322         undef <- Macaw.mkUndefinedBool sym "ptr_cmp"
  323         W4.itePred sym both_bits res_both_bits
  324           =<< W4.itePred sym ok_both_ptrs res_both_ptrs undef
  325   case fixpointSelect of
  326     SimpleFixpointCHC{} ->
  327       if | Just True <- W4.asConstantPred both_bits ->
  328            return res_both_bits
  329          | Just True <- W4.asConstantPred both_ptrs -> do
  330            C.assert bak ok_both_ptrs $ C.AssertFailureSimError "" ""
  331            return res_both_ptrs
  332          | otherwise ->
  333            ptrCmpDefault
  334     _ -> ptrCmpDefault

  583           liftIO $
  584             forM_ assumes $ \(C.LabeledPred p (md, reason)) ->

  632           gp <- getGlobalPair opts pr

  654         Nothing -> return (MapF.empty, [])

  656       liftIO $ printOutLn opts Info
  657         "Examining specification to determine postconditions"
  658       liftIO $ void $ runX86Sim finalState $
  659         assertPost path nm env (preState ^. x86Mem) (preState ^. x86Regs) mdMap
  660 
  661       (stats,vcstats) <- checkGoals bak opts nm (methodSpec ^. MS.csLoc) sc tactic mdMap invSubst loopFunEquivConds
  662 
  663       -- putTopLevelRW =<< liftIO (readIORef rw_ref)
  664 
  665       end <- io getCurrentTime
  666       let diff = diffUTCTime end start
  667       ps <- io (MS.mkProvedSpec MS.SpecProved methodSpec stats vcstats mempty diff)
  668       returnProof $ SomeLLVM ps

  670   | otherwise = fail "LLVM module must be 64-bit"

  672 setupSimpleLoopFixpointFeature ::
  673   ( sym ~ W4.B.ExprBuilder n st fs
  674   , C.IsSymInterface sym
  675   , ?memOpts::C.LLVM.MemOptions
  676   , C.LLVM.HasLLVMAnn sym
  677   , C.IsSyntaxExtension ext
  678   ) =>
  679   sym ->
  680   SharedContext ->
  681   SAWCoreState n ->
  682   C.CFG ext blocks init ret ->
  683   C.GlobalVar C.LLVM.Mem ->
  684   TypedTerm ->
  685   IO (C.ExecutionFeature p sym ext rtp)
  686 
  687 setupSimpleLoopFixpointFeature sym sc sawst cfg mvar func =
  688   Crucible.LLVM.Fixpoint.simpleLoopFixpoint sym cfg mvar fixpoint_func
  689 
  690  where
  691   fixpoint_func fixpoint_substitution condition =
  692     do let fixpoint_substitution_as_list = reverse $ MapF.toList fixpoint_substitution
  693        let body_exprs = map (mapSome $ Crucible.LLVM.Fixpoint.bodyValue) (MapF.elems fixpoint_substitution)
  694        let uninterpreted_constants = foldMap
  695              (viewSome $ Set.map (mapSome $ W4.varExpr sym) . W4.exprUninterpConstants sym)
  696              (Some condition : body_exprs)
  697        let filtered_uninterpreted_constants = Set.toList $ Set.filter
  698              (\(Some variable) ->
  699                not (List.isPrefixOf "creg_join_var" $ show $ W4.printSymExpr variable)
  700                && not (List.isPrefixOf "cmem_join_var" $ show $ W4.printSymExpr variable)
  701                && not (List.isPrefixOf "cundefined" $ show $ W4.printSymExpr variable)
  702                && not (List.isPrefixOf "calign_amount" $ show $ W4.printSymExpr variable))
  703              uninterpreted_constants
  704        body_tms <- mapM (viewSome $ toSC sym sawst) filtered_uninterpreted_constants
  705        implicit_parameters <- mapM (scExtCns sc) $ Set.toList $ foldMap getAllExtSet body_tms
  706 
  707        arguments <- forM fixpoint_substitution_as_list $ \(MapF.Pair _ fixpoint_entry) ->
  708          toSC sym sawst $ Crucible.LLVM.Fixpoint.headerValue fixpoint_entry
  709        applied_func <- scApplyAll sc (ttTerm func) $ implicit_parameters ++ arguments
  710        applied_func_selectors <- forM [1 .. (length fixpoint_substitution_as_list)] $ \i ->
  711          scTupleSelector sc applied_func i (length fixpoint_substitution_as_list)
  712        result_substitution <- MapF.fromList <$> zipWithM
  713          (\(MapF.Pair variable _) applied_func_selector ->
  714            MapF.Pair variable <$> bindSAWTerm sym sawst (W4.exprType variable) applied_func_selector)
  715          fixpoint_substitution_as_list
  716          applied_func_selectors
  717 
  718        explicit_parameters <- forM fixpoint_substitution_as_list $ \(MapF.Pair variable _) ->
  719          toSC sym sawst variable
  720        inner_func <- case asConstant (ttTerm func) of
  721          Just (_, Just (asApplyAll -> (isGlobalDef "Prelude.fix" -> Just (), [_, inner_func]))) ->
  722            return inner_func
  723          _ -> fail $ "not Prelude.fix: " ++ showTerm (ttTerm func)
  724        func_body <- betaNormalize sc
  725          =<< scApplyAll sc inner_func ((ttTerm func) : (implicit_parameters ++ explicit_parameters))
  726 
  727        step_arguments <- forM fixpoint_substitution_as_list $ \(MapF.Pair _ fixpoint_entry) ->
  728          toSC sym sawst $ Crucible.LLVM.Fixpoint.bodyValue fixpoint_entry
  729        tail_applied_func <- scApplyAll sc (ttTerm func) $ implicit_parameters ++ step_arguments
  730        explicit_parameters_tuple <- scTuple sc explicit_parameters
  731        let lhs = Prelude.last step_arguments
  732        w <- scNat sc 64
  733        let implicit_parameter_head =
  734              case implicit_parameters of
  735                ip:_ -> ip
  736                [] -> panic "setupSimpleLoopFixpointFeature"
  737                            ["No implicit parameters"]
  738        rhs <- scBvMul sc w implicit_parameter_head =<< scBvNat sc w =<< scNat sc 128
  739        loop_condition <- scBvULt sc w lhs rhs
  740        output_tuple_type <- scTupleType sc =<< mapM (scTypeOf sc) explicit_parameters
  741        loop_body <- scIte sc output_tuple_type loop_condition tail_applied_func explicit_parameters_tuple
  742 
  743        induction_step_condition <- scEq sc loop_body func_body
  744        result_condition <- bindSAWTerm sym sawst W4.BaseBoolRepr induction_step_condition
  745 
  746        return (result_substitution, result_condition)

  823 -- | This procedure sets up the simple loop fixpoint feature.
  824 --   Its main task is to massage the user-provided invariant
  825 --   term into the form expected by the Crucible exeuction
  826 --   feature.
  827 setupSimpleLoopInvariantFeature ::
  828   ( sym ~ W4.B.ExprBuilder n st fs
  829   , C.IsSymInterface sym
  830   , n ~ GlobalNonceGenerator
  831   , ?memOpts::C.LLVM.MemOptions
  832   , C.LLVM.HasLLVMAnn sym
  833   , C.IsSyntaxExtension ext
  834   ) =>
  835   sym ->
  836   (String -> IO ()) {- ^ logging function -} ->
  837   Integer {- ^ Which loop are we targeting? -} ->
  838   SharedContext ->
  839   SAWCoreState n ->
  840   IORef MetadataMap ->
  841   C.CFG ext blocks init ret ->
  842   C.GlobalVar C.LLVM.Mem ->
  843   TypedTerm {- ^ user-provided invariant term -} ->
  844   IO (C.ExecutionFeature p sym ext rtp)
  845 
  846 setupSimpleLoopInvariantFeature sym printFn loopNum sc sawst mdMap cfg mvar func =
  847   SimpleInvariant.simpleLoopInvariant sym loopNum cfg mvar invariant_func
  848 
  849  where
  850   invariant_func phase implicit_params invariant_substitution =
  851     do let subst_pairs = reverse (MapF.toList invariant_substitution)
  852        let filtered_implicit_params = filter
  853              (\ (Some variable) ->
  854                -- We filter variables with the following special names from appearing as
  855                -- implicit arguments to the loop invariant.  These are basically all various
  856                -- kinds of underspecification appearing from various underlying components
  857                -- and don't make sense as arguments to the loop invariant.
  858                not (List.isPrefixOf "creg_join_var" $ show $ W4.printSymExpr variable)
  859                && not (List.isPrefixOf "cmem_join_var" $ show $ W4.printSymExpr variable)
  860                && not (List.isPrefixOf "cundefined" $ show $ W4.printSymExpr variable)
  861                && not (List.isPrefixOf "calign_amount" $ show $ W4.printSymExpr variable)
  862                && not (List.isPrefixOf "cnoSatisfyingWrite" $ show $ W4.printSymExpr variable)
  863              )
  864              implicit_params
  865        body_tms <- mapM (viewSome $ toSC sym sawst) filtered_implicit_params
  866        implicit_params' <- mapM (scExtCns sc) $ Set.toList $ foldMap getAllExtSet body_tms
  867        initial_exprs <-
  868          forM subst_pairs $
  869            \ (MapF.Pair _var (SimpleInvariant.InvariantEntry initVal _current)) ->
  870                toSC sym sawst initVal
  871        current_exprs <-
  872          forM subst_pairs $
  873            \ (MapF.Pair _var (SimpleInvariant.InvariantEntry _init current)) ->
  874                toSC sym sawst current
  875 
  876        initial_tuple <- scTuple sc initial_exprs
  877        current_tuple <- scTuple sc current_exprs
  878 
  879        -- use the provided logging function to print the discovered
  880        -- implicit parameters
  881        when (phase == SimpleInvariant.InitialInvariant) $
  882          do printFn "Loop invariant implicit parameters!"
  883             forM_ implicit_params' $ \x ->
  884                 do printFn (showTerm x)
  885                    tp <- scTypeOf sc x
  886                    printFn (showTerm tp)
  887 
  888        -- actually apply the arguments to the given term
  889        inv <- scApplyAll sc (ttTerm func) (implicit_params' ++ [initial_tuple, current_tuple])
  890 
  891        -- check that the produced term is type-correct
  892        res <- scTypeCheck sc Nothing inv
  893        case res of
  894          Left _tcErr ->
  895            do tpType <- scTypeOf sc initial_tuple
  896               fail $ unlines [ "Loop invariant has incorrect type! State tuple has type:"
  897                              , showTerm tpType
  898                              ]
  899          Right tp ->
  900            do ok <- scConvertible sc True tp =<< scBoolType sc
  901               unless ok $
  902                 fail $ unlines [ "Loop invariant must return a boolean value, but got:"
  903                                , showTerm tp
  904                                   -- TODO, get ppOpts from the right place
  905                                ]
  906        b <- bindSAWTerm sym sawst W4.BaseBoolRepr inv
  907 
  908        -- Add goal metadata for the initial and inductive invariants
  909        case phase of
  910          SimpleInvariant.HypotheticalInvariant -> return b
  911          SimpleInvariant.InitialInvariant ->
  912            do (ann,b') <- W4.annotateTerm sym b
  913               loc <- W4.getCurrentProgramLoc sym
  914               let md = MS.ConditionMetadata
  915                        { MS.conditionLoc = loc
  916                        , MS.conditionTags = Set.singleton "initial loop invariant"
  917                        , MS.conditionType = "initial loop invariant"
  918                        , MS.conditionContext = ""
  919                        }
  920               modifyIORef mdMap (Map.insert ann md)
  921               return b'
  922          SimpleInvariant.InductiveInvariant ->
  923            do (ann,b') <- W4.annotateTerm sym b
  924               loc <- W4.getCurrentProgramLoc sym
  925               let md = MS.ConditionMetadata
  926                        { MS.conditionLoc = loc
  927                        , MS.conditionTags = Set.singleton "inductive loop invariant"
  928                        , MS.conditionType = "inductive loop invariant"
  929                        , MS.conditionContext = ""
  930                        }
  931               modifyIORef mdMap (Map.insert ann md)
  932               return b'

  947        (Macaw.ArchRegStruct Macaw.X86_64)
  948      , Elf.ElfHeaderInfo 64
  949      , RelevantElf
  950      , Macaw.MemSegmentOff 64
  951      , Map
  952        (Macaw.MemSegmentOff 64)
  953        (C.SomeCFG
  954          (Macaw.MacawExt Macaw.X86_64)
  955          (EmptyCtx ::> Macaw.ArchRegStruct Macaw.X86_64)
  956          (Macaw.ArchRegStruct Macaw.X86_64))
  957      )
  958 buildCFG opts halloc preserved path nm = do
  959   printOutLn opts Info $ mconcat ["Finding symbol for \"", nm, "\""]
  960   elf <- getElf path
  961   relf <- getRelevant path elf
  962   (addr :: Macaw.MemSegmentOff 64) <-
  963     case findSymbols (symMap relf) . encodeUtf8 $ Text.pack nm of
  964       (addr:_) -> pure addr
  965       _ -> fail $ mconcat ["Could not find symbol \"", nm, "\""]
  966   printOutLn opts Info $ mconcat ["Found symbol at address ", show addr, ", building CFG"]
  967   let
  968     preservedRegs = Set.union preserved Macaw.x86CalleeSavedRegs
  969     preserveFn :: forall t. Macaw.X86Reg t -> Bool
  970     preserveFn r = Set.member (Some r) preservedRegs
  971     macawCallParams = Macaw.x86_64CallParams { Macaw.preserveReg = preserveFn }
  972     macawArchInfo = (Macaw.x86_64_info preserveFn)
  973       { Macaw.archCallParams = macawCallParams
  974       }
  975     initialDiscoveryState =
  976       Macaw.emptyDiscoveryState (memory relf) (funSymMap relf) macawArchInfo
  977       -- "inline" any function addresses that we happen to jump to
  978       & Macaw.trustedFunctionEntryPoints .~ Map.empty
  979     finalState = Macaw.cfgFromAddrsAndState initialDiscoveryState [addr] []
  980     finfos = finalState ^. Macaw.funInfo
  981   cfgs <- forM finfos $ \(Some finfo) ->
  982     Macaw.mkFunCFG Macaw.x86_64MacawSymbolicFns halloc
  983     (W4.functionNameFromText . decodeUtf8 $ Macaw.discoveredFunName finfo)
  984     posFn finfo
  985 
  986   case Map.lookup addr cfgs of
  987     Nothing -> throwX86 path $ "Unable to discover CFG from address " <> show addr
  988     Just scfg -> pure (scfg, elf, relf, addr, cfgs)

 1056 initialState ::
 1057   (X86Constraints, OnlineSolver solver) =>
 1058   Backend solver ->
 1059   Options ->

 1061   LLVMCrucibleContext LLVMArch ->
 1062   FilePath ->
 1063   Elf.ElfHeaderInfo 64 ->
 1064   RelevantElf ->
 1065   MS.CrucibleMethodSpecIR LLVM ->
 1066   [(String, Integer)] {- ^ Global variable symbol names and sizes (in bytes) -} ->
 1067   Integer {- ^ Minimum size of the global allocation (here, the end of the .text segment -} ->

 1069 initialState bak opts sc cc path elf relf ms globs maxAddr = do
 1070   let sym = backendGetSym bak
 1071   emptyMem <- C.LLVM.emptyMem C.LLVM.LittleEndian

 1193 -- | Push a fresh pointer as the return address.
 1194 pushFreshReturnAddress ::
 1195   X86Constraints =>
 1196   FilePath -> String ->
 1197   X86Sim ()
 1198 pushFreshReturnAddress path func = do
 1199   SomeOnlineBackend bak <- use x86Backend
 1200   sym <- use x86Sym
 1201   mem <- use x86Mem
 1202   regs <- use x86Regs
 1203   sn <- case W4.userSymbol "stack" of
 1204     Left err -> throwX86 path $ "Invalid symbol for stack: " <> show err
 1205     Right sn -> pure sn
 1206   fresh <- liftIO $ C.LLVM.LLVMPointer
 1207     <$> W4.natLit sym 0
 1208     <*> W4.freshConstant sym sn (W4.BaseBVRepr $ knownNat @64)
 1209   rsp <- getReg path func Macaw.RSP regs
 1210 
 1211   -- x86-64 System V ABI specifies that: "In other words, the stack needs to be
 1212   -- 16 (32 or 64) byte aligned immediately before the call instruction is
 1213   -- executed. Once control has been transferred to the function entry point,
 1214   -- i.e. immediately after the return address has been pushed, %rsp points to
 1215   -- the return address, and the value of (%rsp + 8) is a multiple of 16 (32 or
 1216   -- 64)."
 1217   stackAlign <- integerToAlignment path func defaultStackBaseAlign
 1218   is_aligned <- liftIO $ C.LLVM.isAligned sym (knownNat @64) rsp stackAlign
 1219   when (W4.asConstantPred is_aligned /= Just True) $
 1220     panic "SAWCentral.Crucible.LLVM.X86.pushFreshReturnAddress" ["%rsp is not 16 byte aligned before the call instruction is executed."]
 1221 
 1222   ptr <- liftIO $ C.LLVM.doPtrAddOffset bak mem rsp =<< W4.bvLit sym knownNat (BV.mkBV knownNat (-8))
 1223   let writeAlign = C.LLVM.noAlignment
 1224   finalMem <- liftIO $ C.LLVM.doStore bak mem ptr
 1225     (C.LLVM.LLVMPointerRepr $ knownNat @64)
 1226     (C.LLVM.bitvectorType 8) writeAlign fresh
 1227   x86Mem .= finalMem
 1228   finalRegs <- setReg path func Macaw.RSP ptr regs
 1229   x86Regs .= finalRegs

 1341 -- | Write each SetupValue passed to llvm_execute_func to the appropriate
 1342 -- x86_64 register from the calling convention.
 1343 setArgs ::
 1344   X86Constraints =>
 1345   FilePath -> {- ^ File we're in -}
 1346   String -> {- ^ Function we're in -}
 1347   Map MS.AllocIndex Ptr {- ^ Associates each AllocIndex with the corresponding allocation -} ->
 1348   Map MS.AllocIndex LLVMAllocSpec {- ^ Associates each AllocIndex with its specification -} ->
 1349   Map MS.AllocIndex C.LLVM.Ident {- ^ Associates each AllocIndex with its name -} ->
 1350   [MS.SetupValue LLVM] {- ^ Arguments passed to llvm_execute_func -} ->
 1351   X86Sim ()
 1352 setArgs path func env tyenv nameEnv args = do
 1353   SomeOnlineBackend bak <- use x86Backend
 1354   sym <- use x86Sym
 1355   cc <- use x86CrucibleContext
 1356   mem <- use x86Mem
 1357   let
 1358     setRegSetupValue rs (reg, sval) =
 1359       exceptToFail (typeOfSetupValue cc tyenv nameEnv sval) >>= \case
 1360         ty | C.LLVM.isPointerMemType ty -> do
 1361           val <- C.LLVM.unpackMemValue sym (C.LLVM.LLVMPointerRepr $ knownNat @64)
 1362             =<< resolveSetupVal cc mem env tyenv nameEnv sval
 1363           setReg path func reg val rs
 1364         C.LLVM.IntType 64 -> do
 1365           val <- C.LLVM.unpackMemValue sym (C.LLVM.LLVMPointerRepr $ knownNat @64)
 1366             =<< resolveSetupVal cc mem env tyenv nameEnv sval
 1367           setReg path func reg val rs
 1368         C.LLVM.IntType _ -> do
 1369           C.LLVM.LLVMValInt base off <- resolveSetupVal cc mem env tyenv nameEnv sval
 1370           case testLeq (incNat $ W4.bvWidth off) (knownNat @64) of
 1371             Nothing -> fail "Argument bitvector does not fit in a single general-purpose register"
 1372             Just LeqProof -> do
 1373               off' <- W4.bvZext sym (knownNat @64) off
 1374               val <- C.LLVM.unpackMemValue sym (C.LLVM.LLVMPointerRepr $ knownNat @64)
 1375                 $ C.LLVM.LLVMValInt base off'
 1376               setReg path func reg val rs
 1377         _ -> fail "Argument does not fit into a single general-purpose register"
 1378   regs <- use x86Regs
 1379   newRegs <- liftIO . foldlM setRegSetupValue regs $ zip argRegs args
 1380   x86Regs .= newRegs
 1381 
 1382   -- x86-64 System V ABI specifies that: "Once registers are assigned, the
 1383   -- arguments passed in memory are pushed on the stack in reversed
 1384   -- (right-to-left21) order."
 1385   let stackArgs = reverse $ Prelude.drop (length argRegs) args
 1386   forM_ stackArgs $ \sval -> do

 1388       C.LLVM.PtrType _ -> pure ()

 1392     regs' <- use x86Regs
 1393     rsp <- getReg path func Macaw.RSP regs'
 1394     rsp' <- liftIO $ C.LLVM.doPtrAddOffset bak mem rsp =<< W4.bvLit sym knownNat (BV.mkBV knownNat (-8))
 1395     newRegs' <- setReg path func Macaw.RSP rsp' regs'
 1396     x86Regs .= newRegs'
 1397 
 1398     val <- liftIO $ C.LLVM.unpackMemValue sym (C.LLVM.LLVMPointerRepr $ knownNat @64)
 1399       =<< resolveSetupVal cc mem env tyenv nameEnv sval
 1400 
 1401     mem' <- use x86Mem
 1402     mem'' <- liftIO $
 1403       C.LLVM.doStore bak mem' rsp' (C.LLVM.LLVMPointerRepr $ knownNat @64) (C.LLVM.bitvectorType 8) C.LLVM.noAlignment val
 1404     x86Mem .= mem''

 1438   actualIP <- getReg path func Macaw.X86_IP postregs
 1439   correctRetAddr <- liftIO $ C.LLVM.ptrEq sym C.LLVM.PtrWidth actualIP expectedIP
 1440   liftIO . C.addAssertion bak . C.LabeledPred correctRetAddr . C.SimError W4.initializationLoc
 1441     $ C.AssertFailureSimError "Instruction pointer not set to return address" ""
 1442 
 1443   stack <- liftIO $ C.LLVM.doPtrAddOffset bak premem prersp =<< W4.bvLit sym knownNat (BV.mkBV knownNat 8)
 1444   postrsp <- getReg path func Macaw.RSP postregs
 1445   correctStack <- liftIO $ C.LLVM.ptrEq sym C.LLVM.PtrWidth stack postrsp
 1446   liftIO $ C.addAssertion bak . C.LabeledPred correctStack . C.SimError W4.initializationLoc
 1447     $ C.AssertFailureSimError "Stack not preserved" ""
 1448 
 1449   returnMatches <- case (ms ^. MS.csRetValue, ms ^. MS.csRet) of
 1450     (Just expectedRet, Just retTy) -> do
 1451       postRAX <- C.LLVM.ptrToPtrVal <$> getReg path func Macaw.RAX postregs
 1452       case (postRAX, C.LLVM.memTypeBitwidth retTy) of
 1453         (C.LLVM.LLVMValInt base off, Just retTyBits) -> do
 1454           let
 1455             truncateRAX :: forall r. NatRepr r -> X86Sim (C.LLVM.LLVMVal Sym)
 1456             truncateRAX rsz =
 1457               case (testLeq (knownNat @1) rsz, testLeq rsz (W4.bvWidth off)) of
 1458                 (Just LeqProof, Just LeqProof) ->
 1459                   case testStrictLeq rsz (W4.bvWidth off) of
 1460                     Left LeqProof -> do
 1461                       offTrunc <- liftIO $ W4.bvTrunc sym rsz off
 1462                       pure $ C.LLVM.LLVMValInt base offTrunc
 1463                     _ -> pure $ C.LLVM.LLVMValInt base off
 1464                 _ -> throwX86func path func "Width of return type is zero bits"
 1465           postRAXTrunc <- viewSome truncateRAX (mkNatRepr retTyBits)
 1466           let md = MS.ConditionMetadata
 1467                    { MS.conditionLoc = ms ^. MS.csLoc
 1468                    , MS.conditionTags = mempty
 1469                    , MS.conditionType = "return value matching"
 1470                    , MS.conditionContext = ""
 1471                    }
 1472           pure [LO.matchArg opts sc cc ms MS.PostState md postRAXTrunc retTy expectedRet]
 1473         _ -> throwX86func path func $ "Invalid return type: " <> show (C.LLVM.ppMemType retTy)
 1474     _ -> pure []
 1475 
 1476 
 1477   pointsToMatches <- forM (ms ^. MS.csPostState . MS.csPointsTos)
 1478     $ assertPointsTo path func env tyenv nameEnv
 1479 
 1480   let setupConditionMatchesPost = fmap -- assert postconditions
 1481         (LO.learnSetupCondition opts sc cc ms MS.PostState)
 1482         $ ms ^. MS.csPostState . MS.csConditions
 1483 
 1484   let
 1485     initialECs = Map.fromList
 1486       [ (ecVarIndex ec, ec)
 1487       | tt <- ms ^. MS.csPreState . MS.csFreshVars
 1488       , let ec = tecExt tt
 1489       ]
 1490     initialFree = Set.fromList . fmap (ecVarIndex . tecExt) $ ms ^. MS.csPostState . MS.csFreshVars
 1491 
 1492   initialTerms <- liftIO $ traverse (scExtCns sc) initialECs
 1493 
 1494   result <- liftIO
 1495     . O.runOverrideMatcher sym globals env initialTerms initialFree (ms ^. MS.csLoc)
 1496     . sequence_ $ mconcat
 1497     [ returnMatches
 1498     , pointsToMatches
 1499     , setupConditionMatchesPost
 1500     , [LO.assertTermEqualities sc cc]
 1501     ]
 1502   st <- case result of
 1503     Left err -> throwX86func path func $ show err
 1504     Right (_, st) -> pure st
 1505   liftIO . forM_ (view LO.osAsserts st) $ \(md, W4.LabeledPred p r) ->
 1506        do (ann,p') <- W4.annotateTerm sym p
 1507           modifyIORef mdMap (Map.insert ann md)
 1508           C.addAssertion bak (W4.LabeledPred p' r)

 1551 -- | Gather and run the solver on goals from the simulator.
 1552 checkGoals ::
 1553   IsSymBackend Sym bak =>
 1554   bak ->
 1555   Options ->
 1556   String ->
 1557   W4.ProgramLoc ->
 1558   SharedContext ->
 1559   ProofScript () ->
 1560   IORef MetadataMap {- ^ metadata map -} ->
 1561   MapF (W4.SymFnWrapper Sym) (W4.SymFnWrapper Sym) ->
 1562   [W4.Pred Sym] ->
 1563   TopLevel (SolverStats, [MS.VCStats])
 1564 checkGoals bak opts nm loc sc tactic mdMap invSubst loopFunEquivConds = do
 1565   poststate_gs <- liftIO $ getPoststateObligations sc bak mdMap invSubst
 1566   loop_gs <- liftIO $ forM loopFunEquivConds $ \cond -> do
 1567     let sym = C.backendGetSym bak
 1568     st <- Common.sawCoreState sym
 1569     condTerm <- toSC sym st =<< W4.substituteSymFns sym invSubst cond
 1570     let defaultMd = MS.ConditionMetadata
 1571           { MS.conditionLoc = loc
 1572           , MS.conditionTags = mempty
 1573           , MS.conditionType = "loop function equivalence"
 1574           , MS.conditionContext = ""
 1575           }
 1576     return ("", defaultMd, condTerm)
 1577   let gs = poststate_gs ++ loop_gs
 1578 
 1579   liftIO . printOutLn opts Info $ mconcat
 1580     [ "Simulation finished, running solver on "
 1581     , show $ length gs
 1582     , " goals"
 1583     ]
 1584   outs <- forM (zip [0..] gs) $ \(n, (msg, md, term)) -> do
 1585     g <- liftIO $ boolToProp sc [] term
 1586     let ploc = MS.conditionLoc md
 1587     let gloc = (unwords [show (W4.plSourceLoc ploc)
 1588                        ,"in"
 1589                        , show (W4.plFunction ploc)]) ++
 1590                (if Prelude.null (MS.conditionContext md) then [] else
 1591                   "\n" ++ MS.conditionContext md)
 1592     let proofgoal = ProofGoal
 1593                     { goalNum  = n
 1594                     , goalType = MS.conditionType md
 1595                     , goalName = nm
 1596                     , goalLoc  = gloc
 1597                     , goalDesc = msg
 1598                     , goalSequent = propToSequent g
 1599                     , goalTags = MS.conditionTags md
 1600                     }
 1601     res <- runProofScript tactic g proofgoal (Just ploc)
 1602              (Text.unwords
 1603               ["X86 verification condition", Text.pack (show n), Text.pack msg])
 1604              False -- do not record this theorem in the database
 1605              False -- TODO! useSequentGoals
 1606     case res of
 1607       ValidProof stats thm ->
 1608         return (stats, MS.VCStats md stats (thmSummary thm) (thmNonce thm) (thmDepends thm) (thmElapsedTime thm))
 1609       UnfinishedProof pst -> do
 1610         printOutLnTop Info $ unwords ["Subgoal failed:", msg]
 1611         printOutLnTop Info (show (psStats pst))
 1612         throwTopLevel $ "Proof failed: " ++ show (length (psGoals pst)) ++ " goals remaining."
 1613       InvalidProof stats vals _pst -> do
 1614         printOutLnTop Info $ unwords ["Subgoal failed:", msg]
 1615         printOutLnTop Info (show stats)
 1616         printOutLnTop OnlyCounterExamples "----------Counterexample----------"
 1617         ppOpts <- rwPPOpts <$> getTopLevelRW
 1618         case vals of
 1619           [] -> printOutLnTop OnlyCounterExamples "<<All settings of the symbolic variables constitute a counterexample>>"
 1620           _ -> let showEC ec = Text.unpack (toShortName (ecName ec)) in
 1621                let showAssignment (ec, val) =
 1622                      mconcat [ " ", showEC ec, ": ", show $ ppFirstOrderValue ppOpts val ]
 1623                in mapM_ (printOutLnTop OnlyCounterExamples . showAssignment) vals
 1624         printOutLnTop OnlyCounterExamples "----------------------------------"
 1625         throwTopLevel "Proof failed."
 1626   liftIO $ printOutLn opts Info "All goals succeeded"
 1627 
 1628   let stats = mconcat (map fst outs)
 1629   let vcstats = map snd outs
 1630   return (stats, vcstats)
