  407 -- Push the input through the DFA, remembering the most recent accepting
  408 -- state it encountered.
  409 
  410 alex_scan_tkn user__ orig_input len input__ s last_acc =
  411   input__ `seq` -- strict in the input
  412   let
  413   new_acc = (check_accs (alex_accept `quickIndex` IBOX(s)))
  414   in
  415   new_acc `seq`
  416   case alexGetByte input__ of
  417      Nothing -> (new_acc, input__)
  418      Just (c, new_input) ->
  419 #ifdef ALEX_DEBUG
  420       Debug.Trace.trace ("State: " ++ show IBOX(s) ++ ", char: " ++ show c ++ " " ++ (show . Data.Char.chr . fromIntegral) c) $
  421 #endif
  422       case fromIntegral c of { IBOX(ord_c) ->
  423         let
  424                 base   = alexIndexInt32OffAddr alex_base s
  425                 offset = PLUS(base,ord_c)
  426 
  427                 new_s = if GTE(offset,ILIT(0))
  428                           && let check  = alexIndexInt16OffAddr alex_check offset
  429                              in  EQ(check,ord_c)
  430                           then alexIndexInt16OffAddr alex_table offset
  431                           else alexIndexInt16OffAddr alex_deflt s
  432         in
  433         case new_s of
  434             ILIT(-1) -> (new_acc, input__)
  435                 -- on an error, we want to keep the input *before* the
  436                 -- character that failed, not after.
  437             _ -> alex_scan_tkn user__ orig_input
  438 #ifdef ALEX_LATIN1
  439                    PLUS(len,ILIT(1))
  440                    -- issue 119: in the latin1 encoding, *each* byte is one character
  441 #else
  442                    (if c < 0x80 || c >= 0xC0 then PLUS(len,ILIT(1)) else len)
  443                    -- note that the length is increased ONLY if this is the 1st byte in a char encoding)
  444 #endif
  445                    new_input new_s new_acc
  446       }
  447   where
  448         check_accs (AlexAccNone) = last_acc
  449         check_accs (AlexAcc a  ) = AlexLastAcc a input__ IBOX(len)
  450         check_accs (AlexAccSkip) = AlexLastSkip  input__ IBOX(len)
  451 #ifndef ALEX_NOPRED
  452         check_accs (AlexAccPred a predx rest)
  453            | predx user__ orig_input IBOX(len) input__
  454            = AlexLastAcc a input__ IBOX(len)
  455            | otherwise
  456            = check_accs rest
  457         check_accs (AlexAccSkipPred predx rest)
  458            | predx user__ orig_input IBOX(len) input__
  459            = AlexLastSkip input__ IBOX(len)
  460            | otherwise
  461            = check_accs rest
  462 #endif

  571 -- feed alex a byte describing the current char
  572 -- this came from Cryptol's lexer, which came from LexerUtils, which
  573 -- adapted the technique used in GHC's lexer.

  575 -- FUTURE: it would be nice to share this with the saw-script lexer

  577 byteForChar :: Char -> Word8
  578 byteForChar c
  579   | c <= '\7' = non_graphic
  580   | Char.isAscii c = fromIntegral (Char.ord c)
  581   | otherwise = case Char.generalCategory c of
  582       Char.LowercaseLetter       -> lower
  583       Char.OtherLetter           -> lower
  584       Char.UppercaseLetter       -> upper
  585       Char.TitlecaseLetter       -> upper
  586       Char.DecimalNumber         -> digit
  587       Char.OtherNumber           -> digit
  588       Char.ConnectorPunctuation  -> symbol
  589       Char.DashPunctuation       -> symbol
  590       Char.OtherPunctuation      -> symbol
  591       Char.MathSymbol            -> symbol
  592       Char.CurrencySymbol        -> symbol
  593       Char.ModifierSymbol        -> symbol
  594       Char.OtherSymbol           -> symbol
  595       Char.Space                 -> sp
  596       Char.ModifierLetter        -> other
  597       Char.NonSpacingMark        -> other
  598       Char.SpacingCombiningMark  -> other
  599       Char.EnclosingMark         -> other
  600       Char.LetterNumber          -> other
  601       Char.OpenPunctuation       -> other
  602       Char.ClosePunctuation      -> other
  603       Char.InitialQuote          -> other
  604       Char.FinalQuote            -> tick
  605       _                          -> non_graphic
  606   where
  607   -- CAUTION: these must match the $uni* values at the top of the file
  608   non_graphic     = 0
  609   upper           = 1
  610   lower           = 2
  611   digit           = 3
  612   symbol          = 4
  613   sp              = 5
  614   other           = 6
  615   tick            = 7

  617 -- Get the next byte to feed alex, where the byte is a character class
  618 -- describing the current character as generated above. Update the
  619 -- input state accordingly.
  620 alexGetByte :: AlexInput -> Maybe (Word8, AlexInput)
  621 alexGetByte (PosPair pos (Buffer _ txt)) = fmap fn (LText.uncons txt)
  622   where fn (c, txt') = (byte, PosPair pos' (Buffer c txt'))
  623           where byte  = byteForChar c
  624                 isNew = c == '\n'
  625                 pos'  = if isNew then incLine pos else incCol pos

  630 -- Get the next token.
  631 --
  632 -- Recurses until we have one. If we get AlexError, concatenate all the
  633 -- adjacent invalid characters into a single error message.
  634 --
  635 -- Returns the updated input state, a list of messages, and the token.
  636 scanToken :: AlexInput -> MsgList -> (AlexInput, MsgList, PosPair Token)
  637 scanToken inp0 errors0 =
  638   let go inp errors pendingError =
  639         let (PosPair p (Buffer _ txt)) = inp
  640             finishAnyError = case pendingError of
  641                 Nothing -> errors
  642                 Just (pos, chars) ->
  643                     let badText = Text.pack $ reverse chars in
  644                     (pos, InvalidInput badText) : errors
  645             end =
  646                 (inp, finishAnyError, PosPair p TEnd)
  647         in case alexScan inp 0 of
  648             AlexEOF ->
  649                 end
  650             AlexError _ -> case alexGetByte inp of
  651               Nothing ->
  652                   end
  653               Just (_byte, inp') ->
  654                 -- ignore the classification byte; get the actual failing char
  655                 let badChar = alexInputPrevChar inp' in
  656                 case pendingError of
  657                     Nothing ->
  658                         go inp' errors (Just (p, [badChar]))
  659                     Just (pos, badChars) ->
  660                         go inp' errors (Just (pos, badChar : badChars))
  661             AlexSkip inp' _ ->
  662                 go inp' finishAnyError Nothing
  663             AlexToken inp' l act ->
  664                 let v = act (LText.unpack (LText.take (fromIntegral l) txt)) in
  665                 (inp', finishAnyError, PosPair p v)
  666   in
  667   go inp0 errors0 Nothing

  669 -- | State for processing comments.
  670 --
  671 -- CNone is the ground state (not in a comment)
  672 --
  673 -- CBlock startpos n is when we're within a block comment starting at
  674 -- startpos, and n is the number of additional nested block comments
  675 -- that have been opened.
  676 --
  677 -- CLine is when we're in a line (-- type) comment.
  678 data CommentState = CNone | CBlock Pos !Int | CLine
  679 
  680 -- Get the next token, dropping comments.
  681 --
  682 -- Recurses until we get one.
  683 --
  684 -- Returns the updated input state, a list of messages, and the token.
  685 scanSkipComments :: AlexInput -> (AlexInput, MsgList, PosPair Token)
  686 scanSkipComments inp0 =
  687   let go state inp errors =
  688         let (inp', errors', tok) = scanToken inp errors
  689             again nextState = go nextState inp' errors'
  690             againWith nextState err = go nextState inp' (err : errors')
  691             accept = (inp', errors', tok)
  692             acceptWith err = (inp', err : errors', tok)
  693         in
  694         case state of
  695           CNone -> case val tok of
  696             TEnd ->
  697                 -- plain EOF
  698                 accept
  699             TCommentS ->
  700                 -- open block-comment
  701                 again (CBlock (pos tok) 0)
  702             TCommentE ->
  703                 -- misplaced close-comment
  704                 let err = (pos tok, InvalidInput "-}") in
  705                 againWith CNone err
  706             TCommentL ->
  707                 -- begin line-comment
  708                 again CLine
  709             TEOL ->
  710                 -- ordinary end-of-line, doesn't need to be seen downstream
  711                 again CNone
  712             _ ->
  713                 -- uncommented token, take it
  714                 accept
  715 
  716           CBlock startpos depth -> case val tok of
  717             TEnd ->
  718                 -- EOF in block comment
  719                 let err = (startpos, UnclosedComment) in
  720                 acceptWith err
  721             TCommentS ->
  722                 -- nested open-comment; keep the original start position
  723                 again (CBlock startpos (depth + 1))
  724             TCommentE
  725              | depth == 0 ->
  726                 -- end outer block comment
  727                 again CNone
  728              | otherwise ->
  729                 -- end nested block comment
  730                 again (CBlock startpos (depth - 1))
  731             _ ->
  732                 -- anything else in a block comment; get another token
  733                 again state
  734 
  735           CLine -> case val tok of
  736             TEnd ->
  737                 -- EOF in line-comment; missing a newline, which is a warning
  738                 let err = (pos tok, MissingEOL) in
  739                 acceptWith err
  740             TEOL ->
  741                 -- EOL ending the comment; drop it but end the comment
  742                 again CNone
  743             _ ->
  744                 -- anything else in a line comment; drop it
  745                 again CLine
  746   in
  747   let (inp0', errors, tok) = go CNone inp0 [] in
  748   (inp0', reverse errors, tok)
