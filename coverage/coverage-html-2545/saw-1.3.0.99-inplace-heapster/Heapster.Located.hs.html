   12 -- | A thing paired with its location
   13 data Located a = Located
   14   { locPos :: !Pos      -- ^ location
   15   , locThing :: a       -- ^ thing
   16   }
   17   deriving (Functor, Foldable, Traversable, Show)
   18 
   19 -- | A position in a text-file
   20 data Pos = Pos
   21   { posChar :: !Int     -- ^ 0-based absolute index
   22   , posLine :: !Int     -- ^ 1-based line number
   23   , posCol  :: !Int     -- ^ 1-based column number
   24   }
   25   deriving Show
   26 
   27 -- | Convenience class for types of things with a known location
   28 class HasPos a where
   29   -- | Get contained position
   30   pos :: a -> Pos
   31 
   32 -- | Returns 'locPos'
   33 instance HasPos (Located a) where
   34   pos = locPos
   35 
   36 -- | Returns itself
   37 instance HasPos Pos where
