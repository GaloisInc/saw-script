   10 module SAWCore.Unique (getUniqueInt) where
   11 
   12 import Data.IORef
   13 import System.IO.Unsafe (unsafePerformIO)
   14 
   15 globalRef :: IORef Int
   16 {-# NOINLINE globalRef #-}
   17 globalRef = unsafePerformIO (newIORef 0)
   18 
   19 -- | Get the next unique integer, and increment the global counter
   20 getUniqueInt :: IO Int
   21 getUniqueInt = atomicModifyIORef' globalRef (\x -> (x+1, x))
