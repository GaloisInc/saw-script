   69 -- ** Building the module graph from Yosys IR

   71 data Modgraph = Modgraph
   72   { _modgraphGraph :: Graph.Graph
   73   , _modgraphNodeFromVertex :: Graph.Vertex -> (Module, Text, [Text])
   74   -- , _modgraphVertexFromKey :: Text -> Maybe Graph.Vertex
   75   }
   76 makeLenses ''Modgraph
   77 
   78 -- | Given a Yosys IR, construct a graph of intermodule dependencies.
   79 yosysIRModgraph :: YosysIR -> Modgraph
   80 yosysIRModgraph ir =
   81   let
   82     moduleToNode :: (Text, Module) -> (Module, Text, [Text])
   83     moduleToNode (nm, m) = (m, nm, deps)
   84       where
   85         deps = Text.pack . show . view cellType <$> Map.elems (m ^. moduleCells)
   86     nodes = moduleToNode <$> Map.assocs (ir ^. yosysModules)
   87     (_modgraphGraph, _modgraphNodeFromVertex, _modgraphVertexFromKey)
   88       = Graph.graphFromEdges nodes
   89   in Modgraph{..}

   91 -- | Given a Yosys IR, construct a map from module names to SAWCore terms alongside SAWCore and Cryptol types
   92 convertYosysIR ::
   93   MonadIO m =>
   94   SC.SharedContext ->
   95   YosysIR ->
   96   m (Map Text ConvertedModule)
   97 convertYosysIR sc ir = do
   98   let mg = yosysIRModgraph ir
   99   let sorted = reverseTopSort $ mg ^. modgraphGraph
  100   foldM
  101     (\env v -> do
  102         let (m, nm, _) = mg ^. modgraphNodeFromVertex $ v
  103         cm <- convertModule sc env m
  104         _ <- validateTerm sc ("translating the combinational circuit \"" <> nm <> "\"") $ cm ^. convertedModuleTerm
  105         n <- liftIO $ Nonce.freshNonce Nonce.globalNonceGenerator
  106         let frag = Text.pack . show $ Nonce.indexValue n
  107         let uri = URI.URI
  108               { URI.uriScheme = URI.mkScheme "yosys"
  109               , URI.uriAuthority = Left True
  110               , URI.uriPath = (False,) <$> mapM URI.mkPathPiece (nm NE.:| [])
  111               , URI.uriQuery = []
  112               , URI.uriFragment = URI.mkFragment frag
  113               }
  114         let ni = SC.ImportedName uri [nm]
  115         tc <- liftIO $ SC.scConstant' sc ni (cm ^. convertedModuleTerm) (cm ^. convertedModuleType)
  116         let cm' = cm { _convertedModuleTerm = tc }
  117         pure $ Map.insert nm cm' env
  118     )
  119     Map.empty
  120     sorted

  122 -- | Given a Yosys IR, construct a map from module names to TypedTerms
  123 yosysIRToTypedTerms ::
  124   MonadIO m =>
  125   SC.SharedContext ->
  126   YosysIR ->
  127   m (Map Text SC.TypedTerm)
  128 yosysIRToTypedTerms sc ir = do
  129   env <- convertYosysIR sc ir
  130   pure . flip fmap env $ \cm ->
  131     SC.TypedTerm
  132     (SC.TypedTermSchema $ C.tMono $ cm ^. convertedModuleCryptolType)
  133     $ cm ^. convertedModuleTerm
  134 
  135 -- | Given a Yosys IR, construct a SAWCore record containing terms for each module
  136 yosysIRToRecordTerm ::
  137   MonadIO m =>
  138   SC.SharedContext ->
  139   YosysIR ->
  140   m SC.TypedTerm
  141 yosysIRToRecordTerm sc ir = do
  142   env <- convertYosysIR sc ir
  143   record <- cryptolRecord sc $ view convertedModuleTerm <$> env
  144   let cty = C.tRec . C.recordFromFields $ (\(nm, cm) -> (C.mkIdent nm, cm ^. convertedModuleCryptolType)) <$> Map.assocs env
  145   let tt = SC.TypedTerm (SC.TypedTermSchema $ C.tMono cty) record
  146   pure tt

  148 -- | Given a Yosys IR, construct a value representing a specific module with all submodules inlined
  149 yosysIRToSequential ::
  150   MonadIO m =>
  151   SC.SharedContext ->
  152   YosysIR ->
  153   Text ->
  154   m YosysSequential
  155 yosysIRToSequential sc ir nm = do
  156   case Map.lookup nm $ ir ^. yosysModules of
  157     Nothing -> throw . YosysError $ mconcat
  158       [ "Could not find module: "
  159       , nm
  160       ]
  161     Just m -> convertModuleInline sc m

  168 -- | Produces a Term given the path to a JSON file produced by the
  169 -- Yosys write_json command. The resulting term is a Cryptol record,
  170 -- where each field corresponds to one HDL module exported by Yosys.
  171 -- Each HDL module is in turn represented by a function from a record
  172 -- of input port values to a record of output port values.
  173 yosys_import :: FilePath -> TopLevel SC.TypedTerm
  174 yosys_import path = do
  175   sc <- getSharedContext
  176   ir <- loadYosysIR path
  177   tt <- yosysIRToRecordTerm sc ir
  178   _ <- validateTerm sc "translating combinational circuits" $ SC.ttTerm tt
  179   pure tt
  180 
  181 -- | Proves equality between a combinational HDL module and a
  182 -- specification. Note that terms derived from HDL modules are first
  183 -- class, and are not restricted to yosys_verify: they may also be
  184 -- used with SAW's typical Term infrastructure like sat, prove_print,
  185 -- term rewriting, etc. yosys_verify simply provides a convenient and
  186 -- familiar interface, similar to llvm_verify or jvm_verify.
  187 yosys_verify ::
  188   SC.TypedTerm {- ^ Term corresponding to the HDL module -} ->
  189   [SC.TypedTerm] {- ^ Preconditions for the equality -} ->
  190   SC.TypedTerm {- ^ Specification term of the same type as the HDL module -} ->
  191   [YosysTheorem] {- ^ Overrides to apply -} ->
  192   ProofScript () ->
  193   TopLevel YosysTheorem
  194 yosys_verify ymod preconds other specs tactic = do
  195   sc <- getSharedContext
  196   newmod <- foldM (flip $ applyOverride sc)
  197     (SC.ttTerm ymod)
  198     specs
  199   mpc <- case preconds of
  200     [] -> pure Nothing
  201     (pc:pcs) -> do
  202       t <- foldM (\a b -> liftIO $ SC.scAnd sc a b) (SC.ttTerm pc) (SC.ttTerm <$> pcs)
  203       pure . Just $ SC.TypedTerm (SC.ttType pc) t
  204   thm <- buildTheorem sc ymod newmod mpc other
  205   prop <- theoremProp sc thm
  206   _ <- Builtins.provePrintPrim tactic prop
  207   pure thm

  209 -- | Import a single sequential HDL module. N.B. SAW expects the
  210 -- sequential module to exist entirely within a single Yosys module.
  211 yosys_import_sequential ::
  212   Text {- ^ Name of the HDL module -} ->
  213   FilePath {- ^ Path to the Yosys JSON file -} ->
  214   TopLevel YosysSequential
  215 yosys_import_sequential nm path = do
  216   sc <- getSharedContext
  217   ir <- loadYosysIR path
  218   yosysIRToSequential sc ir nm

  220 -- | Extracts a term from the given sequential module with the state
  221 -- eliminated by iterating the term over the given concrete number of
  222 -- cycles. The resulting term has no state field in the inputs or
  223 -- outputs. Each input and output field is replaced with an array of
  224 -- that field's type (array length being the number of cycles
  225 -- specified).
  226 yosys_extract_sequential ::
  227   YosysSequential ->
  228   Integer {- ^ Number of cycles to iterate term -} ->
  229   TopLevel SC.TypedTerm
  230 yosys_extract_sequential s n = do
  231   sc <- getSharedContext
  232   tt <- composeYosysSequential sc s n
  233   _ <- validateTerm sc "composing a sequential term" $ SC.ttTerm tt
  234   pure tt
  235 
  236 -- | Like `yosys_extract_sequential`, but the resulting term has an
  237 -- additional parameter to specify the initial state.
  238 yosys_extract_sequential_with_state ::
  239   YosysSequential ->
  240   Integer {- ^ Number of cycles to iterate term -} ->
  241   TopLevel SC.TypedTerm
  242 yosys_extract_sequential_with_state s n = do
  243   sc <- getSharedContext
  244   tt <- composeYosysSequentialWithState sc s n
  245   _ <- validateTerm sc "composing a sequential term with state" $ SC.ttTerm tt
  246   pure tt

  248 -- | Extracts a term from the given sequential module. This term has
  249 -- explicit fields for the state of the circuit in the input and
  250 -- output record types.
  251 yosys_extract_sequential_raw :: YosysSequential -> TopLevel SC.TypedTerm
  252 yosys_extract_sequential_raw s = pure $ s ^. yosysSequentialTerm

  254 -- | Export a query over the given sequential module to an input file for the Sally model checker.
  255 yosys_verify_sequential_sally ::
  256   YosysSequential ->
  257   FilePath {- ^ Path to write the resulting Sally input -} ->
  258   SC.TypedTerm {- ^ A boolean function of three parameters: an 8-bit cycle counter, a record of "fixed" inputs, and a record of circuit outputs -} ->
  259   [String] {- ^ Names of circuit inputs that are fixed -} ->
  260   TopLevel ()
  261 yosys_verify_sequential_sally s path q fixed = do
  262   sc <- getSharedContext
  263   sym <- liftIO $ Common.newSAWCoreExprBuilder sc False
  264   queryModelChecker sym sc s path q . Set.fromList $ Text.pack <$> fixed
