   24 -- | Unchecked types
   25 data AstType

   27   | TyBool Pos                  -- ^ bool type
   28   | TyNat Pos                   -- ^ nat type
   29   | TyBV Pos Natural            -- ^ bitvector type
   30   | TyLlvmPtr Pos Natural       -- ^ llvm pointer with width
   31   | TyLlvmFrame Pos Natural     -- ^ llvm frame with width
   32   | TyLlvmBlock Pos Natural     -- ^ llvm block with width
   33   | TyLlvmShape Pos Natural     -- ^ llvm shape with width
   34   | TyLifetime Pos              -- ^ lifetime
   35   | TyRwModality Pos            -- ^ rwmodality
   36   | TyPermList Pos              -- ^ permlist
   37   | TyStruct Pos [AstType]      -- ^ struct(types)
   38   | TyPerm Pos AstType          -- ^ perm(type)
   39   deriving Show

   41 -- | Unchecked expressions
   42 data AstExpr
   43   = ExUnit Pos                  -- ^ unit
   44   | ExAlways Pos                -- ^ always
   45   | ExNat Pos Natural           -- ^ number literal
   46   | ExVar Pos String (Maybe [AstExpr]) (Maybe AstExpr) -- ^ identifier, shape arguments, offset
   47   | ExAdd Pos AstExpr AstExpr   -- ^ addition
   48   | ExNeg Pos AstExpr           -- ^ negation
   49   | ExMul Pos AstExpr AstExpr   -- ^ multiplication or permission conjunction
   50   | ExRead Pos                  -- ^ read modality
   51   | ExWrite Pos                 -- ^ read/write modality
   52   | ExStruct Pos [AstExpr]      -- ^ struct literal with field expressions
   53   | ExLlvmWord Pos AstExpr      -- ^ llvmword with value
   54   | ExLlvmFrame Pos [(AstExpr, Natural)] -- ^ llvmframe literal
   55   | ExOr Pos AstExpr AstExpr    -- ^ or permission
   56   | ExFalse Pos                 -- ^ false permission
   57   | ExAny Pos                   -- ^ any permission
   58 
   59   | ExEmptySh Pos               -- ^ empty shape
   60   | ExEqSh Pos AstExpr AstExpr  -- ^ equal shape
   61   | ExTrue Pos                  -- ^ trivial permission
   62   | ExExists Pos String AstType AstExpr -- ^ existentially quantified value
   63   | ExSeqSh Pos AstExpr AstExpr -- ^ sequenced shapes
   64   | ExOrSh Pos AstExpr AstExpr  -- ^ alternative shapes
   65   | ExExSh Pos String AstType AstExpr -- ^ existentially quantified shape
   66   | ExFieldSh Pos (Maybe AstExpr) AstExpr -- ^ field shape
   67   | ExPtrSh Pos (Maybe AstExpr) (Maybe AstExpr) AstExpr -- ^ pointer shape
   68   | ExArraySh Pos AstExpr AstExpr AstExpr -- ^ array shape
   69   | ExTupleSh Pos AstExpr -- ^ field shape
   70   | ExFalseSh Pos               -- ^ false shape
   71 
   72   | ExEqual Pos AstExpr AstExpr -- ^ equal bitvector proposition
   73   | ExNotEqual Pos AstExpr AstExpr -- ^ not-equal bitvector proposition
   74   | ExLessThan Pos AstExpr AstExpr -- ^ less-than bitvector proposition
   75   | ExLessEqual Pos AstExpr AstExpr -- ^ less-than or equal-to bitvector proposition
   76 
   77   | ExEq Pos AstExpr            -- ^ equal permission
   78   | ExLOwned Pos [AstExpr] [(Located String, AstExpr)] [(Located String, AstExpr)] -- ^ owned permission
   79   | ExLCurrent Pos (Maybe AstExpr) -- ^ current permission
   80   | ExLFinished Pos -- ^ finished permission
   81   | ExShape Pos AstExpr -- ^ shape literal
   82   | ExFree Pos AstExpr -- ^ free literal
   83   | ExPtr Pos (Maybe AstExpr) AstExpr AstExpr (Maybe AstExpr) AstExpr -- ^ pointer permission
   84   | ExMemblock Pos (Maybe AstExpr) AstExpr AstExpr AstExpr AstExpr -- ^ memblock permission
   85   | ExLlvmFunPtr Pos AstExpr AstExpr AstFunPerm -- ^ function pointer permission
   86   | ExArray Pos (Maybe AstExpr) AstExpr AstExpr AstExpr AstExpr AstExpr -- ^ array permission
   87   deriving Show
