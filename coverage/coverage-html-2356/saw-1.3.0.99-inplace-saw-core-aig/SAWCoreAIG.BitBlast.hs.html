   55 ------------------------------------------------------------
   56 -- Vector operations
   57 
   58 lvFromV :: V.Vector l -> LitVector l
   59 lvFromV v = AIG.generate_msb0 (V.length v) ((V.!) v)
   60 
   61 vFromLV :: LitVector l -> V.Vector l
   62 vFromLV lv = V.generate (AIG.length lv) (AIG.at lv)
   63 
   64 lvRotateL :: LitVector l -> Integer -> LitVector l
   65 lvRotateL xs i
   66   | AIG.length xs == 0 = xs
   67   | otherwise = (AIG.++) (AIG.drop j xs) (AIG.take j xs)
   68   where j = fromInteger (i `mod` toInteger (AIG.length xs))
   69 
   70 lvRotateR :: LitVector l -> Integer -> LitVector l
   71 lvRotateR xs i = lvRotateL xs (- i)
   72 
   73 lvShiftL :: l -> LitVector l -> Integer -> LitVector l
   74 lvShiftL x xs i = (AIG.++) (AIG.drop j xs) (AIG.replicate j x)
   75   where j = fromInteger (min i (toInteger (AIG.length xs)))
   76 
   77 lvShl :: l -> LitVector l -> Natural -> LitVector l
   78 lvShl l v i = AIG.slice v j (n-j) AIG.++ AIG.replicate j l
   79   where n = AIG.length v
   80         j = fromIntegral i `min` n
   81 
   82 lvShiftR :: l -> LitVector l -> Integer -> LitVector l
   83 lvShiftR x xs i = (AIG.++) (AIG.replicate j x) (AIG.take (AIG.length xs - j) xs)
   84   where j = fromInteger (min i (toInteger (AIG.length xs)))
   85 
   86 lvShr :: l -> LitVector l -> Natural -> LitVector l
   87 lvShr l v i = AIG.replicate j l AIG.++ AIG.slice v 0 (n-j)
   88   where n = AIG.length v
   89         j = fromIntegral i `min` n

  106 data BExtra l

  112 vBool :: l -> BValue l
  113 vBool l = VBool l
  114 
  115 toBool :: BValue l -> l
  116 toBool (VBool l) = l
  117 toBool x = error $ unwords ["SAWCoreAIG.BitBlast.toBool", show x]
  118 
  119 vWord :: LitVector l -> BValue l
  120 vWord lv = VWord lv
  121 
  122 toWord :: BValue l -> IO (LitVector l)
  123 toWord (VWord lv) = return lv
  124 toWord (VVector vv) = lvFromV <$> traverse (fmap toBool . force) vv
  125 toWord x = fail $ unwords ["SAWCoreAIG.BitBlast.toWord", show x]
  126 
  127 flattenBValue :: BValue l -> IO (LitVector l)
  128 flattenBValue (VBool l) = return (AIG.replicate 1 l)
  129 flattenBValue (VWord lv) = return lv
  130 flattenBValue (VExtra (BStream _ _)) = error "SAWCoreAIG.BitBlast.flattenBValue: BStream"
  131 flattenBValue (VVector vv) =
  132   AIG.concat <$> traverse (flattenBValue <=< force) (V.toList vv)
  133 flattenBValue VUnit = return $ AIG.concat []
  134 flattenBValue (VPair x y) = do
  135   vx <- flattenBValue =<< force x
  136   vy <- flattenBValue =<< force y
  137   return $ AIG.concat [vx, vy]
  138 flattenBValue (VRecordValue elems) = do
  139   AIG.concat <$> mapM (flattenBValue <=< force . snd) elems
  140 flattenBValue _ = error $ unwords ["SAWCoreAIG.BitBlast.flattenBValue: unsupported value"]

  144 -- | op : (n : Nat) -> Vec n Bool -> Nat -> Vec n Bool
  145 bvShiftOp :: (LitVector l -> LitVector l -> IO (LitVector l))
  146           -> (LitVector l -> Natural -> LitVector l)
  147           -> BPrim l
  148 bvShiftOp bvOp natOp =
  149   Prims.constFun $
  150   Prims.wordFun (pure1 lvFromV) $ \x ->
  151   Prims.strictFun $ \y -> Prims.Prim $
  152     case y of
  153       VNat n   -> return (vWord (natOp x n))
  154       VBVToNat _ v -> fmap vWord (bvOp x =<< toWord v)
  155       _        -> error $ unwords ["SAWCoreAIG.BitBlast.shiftOp", show y]
  156 
  157 lvSShr :: LitVector l -> Natural -> LitVector l
  158 lvSShr v i = lvShr (AIG.msb v) v i

  169 pure3 f x y z = pure (f x y z)

  294 -- | Lifts a strict mux operation to a lazy mux
  295 lazyMux :: AIG.IsAIG l g => g s -> (l s -> a -> a -> IO a) -> l s -> IO a -> IO a -> IO a
  296 lazyMux be muxFn c tm fm
  297   | (AIG.===) c (AIG.trueLit be) = tm
  298   | (AIG.===) c (AIG.falseLit be) = fm
  299   | otherwise = do
  300       t <- tm
  301       f <- fm
  302       muxFn c t f

  304 muxBVal :: AIG.IsAIG l g => g s -> TValue (BitBlast (l s)) -> l s -> BValue (l s) -> BValue (l s) -> IO (BValue (l s))
  305 muxBVal be = Prims.muxValue (prims be)
  306 
  307 muxInt :: a -> Integer -> Integer -> IO Integer
  308 muxInt _ x y = if x == y then return x else fail $ "muxBVal: VInt " ++ show (x, y)
  309 
  310 muxBExtra :: AIG.IsAIG l g => g s ->
  311   TValue (BitBlast (l s)) -> l s -> BExtra (l s) -> BExtra (l s) -> IO (BExtra (l s))
  312 muxBExtra be (VDataType (primName -> "Prelude.Stream") [TValue tp] []) c x y =
  313   do let f i = do xi <- lookupBStream (VExtra x) i
  314                   yi <- lookupBStream (VExtra y) i
  315                   muxBVal be tp c xi yi
  316      r <- newIORef Map.empty
  317      return (BStream f r)
  318 muxBExtra _ tp _ _ _ = panic "muxBExtra" ["Type mismatch: " <> Text.pack (show tp)]

  320 -- | Barrel-shifter algorithm. Takes a list of bits in big-endian order.
  321 genShift ::
  322   AIG.IsAIG l g => g s -> (LitVector (l s) -> Integer -> LitVector (l s)) ->
  323   LitVector (l s) -> LitVector (l s) -> IO (LitVector (l s))
  324 genShift be op x y = Prims.shifter (AIG.ite be) (pure2 op) x (AIG.bvToList y)
  325 
  326 -- | rounded-up log base 2, where we complete the function by setting:
  327 --   lg2 0 = 0
  328 bitblastLogBase2 :: AIG.IsAIG l g => g s -> LitVector (l s) -> IO (LitVector (l s))
  329 bitblastLogBase2 g x = do
  330   z <- AIG.isZero g x
  331   AIG.iteM g z (return x) (AIG.logBase2_up g x)

  336 -- primitive bvToInt : (n : Nat) -> Vec n Bool -> Integer;
  337 bvToIntOp :: AIG.IsAIG l g => g s -> BPrim (l s)
  338 bvToIntOp g =
  339   Prims.constFun $
  340   Prims.wordFun (pure1 lvFromV) $ \v ->
  341   Prims.Prim $
  342    case AIG.asUnsigned g v of
  343       Just i -> return $ VInt i
  344       Nothing -> fail "Cannot convert symbolic bitvector to integer"
  345 
  346 -- primitive sbvToInt : (n : Nat) -> Vec n Bool -> Integer;
  347 sbvToIntOp :: AIG.IsAIG l g => g s -> BPrim (l s)
  348 sbvToIntOp g =
  349   Prims.constFun $
  350   Prims.wordFun (pure1 lvFromV) $ \v ->
  351   Prims.Prim $
  352    case AIG.asSigned g v of
  353       Just i -> return $ VInt i
  354       Nothing -> fail "Cannot convert symbolic bitvector to integer"
  355 
  356 -- primitive intToBv : (n : Nat) -> Integer -> Vec n Bool;
  357 intToBvOp :: AIG.IsAIG l g => g s -> BPrim (l s)
  358 intToBvOp g =
  359   Prims.natFun $ \n ->
  360   Prims.intFun $ \x -> Prims.Prim 
  361     (VWord <$>
  362      if n >= 0 then return (AIG.bvFromInteger g (fromIntegral n) x)
  363                else AIG.neg g (AIG.bvFromInteger g (fromIntegral n) (negate x)))

  381   Prims.constFun $

  386 intModBinOp :: (Integer -> Integer -> Integer) -> BPrim l
  387 intModBinOp f =
  388   Prims.natFun $ \n ->
  389   Prims.intModFun $ \x ->
  390   Prims.intModFun $ \y ->
  391     Prims.PrimValue (VIntMod n (f x y `mod` toInteger n))

  401 -- MkStream :: (a :: sort 0) -> (Nat -> a) -> Stream a;
  402 mkStreamOp :: BPrim l
  403 mkStreamOp =
  404   Prims.constFun $
  405   Prims.strictFun $ \f ->
  406   Prims.Prim $
  407     do r <- newIORef Map.empty
  408        return $ VExtra (BStream (\n -> apply f (ready (VNat n))) r)
  409 
  410 -- streamGet :: (a :: sort 0) -> Stream a -> Nat -> a;
  411 streamGetOp :: AIG.IsAIG l g => g s -> BPrim (l s)
  412 streamGetOp be =
  413   Prims.tvalFun   $ \tp ->
  414   Prims.strictFun $ \xs ->
  415   Prims.strictFun $ \ix ->
  416   Prims.Prim $ case ix of
  417     VNat n -> lookupBStream xs n
  418     VBVToNat _ w ->
  419        do bs <- toWord w
  420           AIG.muxInteger (lazyMux be (muxBVal be tp)) ((2 ^ AIG.length bs) - 1) bs (lookupBStream xs)
  421     v -> fail (unlines ["SAWCoreAIG.BitBlast.streamGetOp", "Expected Nat value", show v])
  422 
  423 
  424 lookupBStream :: BValue l -> Natural -> IO (BValue l)
  425 lookupBStream (VExtra (BStream f r)) n = do
  426    m <- readIORef r
  427    case Map.lookup n m of
  428      Just v  -> return v
  429      Nothing -> do v <- f n
  430                    writeIORef r (Map.insert n v m)
  431                    return v
  432 lookupBStream _ _ = fail "SAWCoreAIG.BitBlast.lookupBStream: expected Stream"

  447 -- Bit-blasting primitives.
  448 --
  449 -- NB: It doesn't make sense to bit blast more than one term using the
  450 -- same bit engine, so the primitives 'withBitBlasted*' create their
  451 -- own bit engine internally, instead of receiving it from the caller,
  452 -- and pass it to the caller-provided continuation.
  453 
  454 type PrimMap l g = forall s. g s -> Map Ident (BPrim (l s))
  455 
  456 bitBlastBasic :: AIG.IsAIG l g
  457               => g s
  458               -> ModuleMap
  459               -> PrimMap l g
  460               -> Map VarIndex (BValue (l s))
  461               -> Term
  462               -> IO (BValue (l s))
  463 bitBlastBasic be m addlPrims ecMap t = do
  464   let neutral _env nt = fail ("bitBlastBasic: could not evaluate neutral term: " ++ show nt)
  465   let primHandler pn msg env _tv =
  466          fail $ unlines
  467            [ "Could not evaluate primitive " ++ show (primName pn)
  468            , "On argument " ++ show (length env)
  469            , Text.unpack msg
  470            ]
  471   cfg <- Sim.evalGlobal m (Map.union (beConstMap be) (addlPrims be))
  472          (bitBlastExtCns ecMap)
  473          (const Nothing)
  474          neutral
  475          primHandler
  476   Sim.evalSharedTerm cfg t
  477 
  478 bitBlastExtCns ::
  479   Map VarIndex (BValue (l s)) -> ExtCns (TValue (BitBlast (l s))) ->
  480   IO (BValue (l s))
  481 bitBlastExtCns ecMap (EC idx name _v) =
  482   case Map.lookup idx ecMap of
  483     Just var -> return var
  484     Nothing -> fail $
  485       "SAWCoreAIG.BitBlast: can't translate variable " ++
  486       show name ++ "(index: " ++ show idx ++ ")"
  487 
  488 asAIGType :: SharedContext -> Term -> IO [(String, Term)]
  489 asAIGType sc t = do
  490   t' <- scWhnf sc t
  491   case t' of
  492     (R.asPi -> Just (n, t1, t2)) -> ((Text.unpack n, t1) :) <$> asAIGType sc t2
  493     (R.asBoolType -> Just ())    -> return []
  494     (R.asVecType -> Just _)      -> return []
  495     (R.asTupleType -> Just _)    -> return []
  496     (R.asRecordType -> Just _)   -> return []
  497     _                            -> fail $ "SAWCoreAIG.BitBlast.adAIGType: invalid AIG type: "
  498                                     ++ scPrettyTerm defaultPPOpts t'

  500 bitBlastTerm ::
  501   AIG.IsAIG l g =>
  502   g s ->
  503   SharedContext ->
  504   PrimMap l g ->
  505   Term ->
  506   IO (BValue (l s), [(String, FiniteType)])
  507 bitBlastTerm be sc addlPrims t = do
  508   ty <- scTypeOf sc t
  509   args <- asAIGType sc ty
  510   let ecs = getAllExts t
  511   argShapes <- traverse (asFiniteType sc) (map snd args)
  512   ecShapes <- traverse (asFiniteType sc) (map ecType ecs)
  513   argVars <- traverse (newVars' be) argShapes
  514   ecVars <- traverse (newVars be) ecShapes
  515   let ecMap = Map.fromList $ zip (map ecVarIndex ecs) ecVars
  516   modmap <- scGetModuleMap sc
  517   bval <- bitBlastBasic be modmap addlPrims ecMap t
  518   bval' <- applyAll bval argVars
  519   let names =  map fst args ++ map (Text.unpack . toShortName . ecName) ecs
  520       shapes = argShapes ++ ecShapes
  521   return (bval', zip names shapes)
  522 
  523 -- | Bitblast a term and apply a function to the result.
  524 withBitBlastedTerm :: AIG.IsAIG l g => AIG.Proxy l g ->
  525   SharedContext ->
  526   PrimMap l g ->
  527   Term ->
  528   (forall s. g s -> LitVector (l s) -> IO a) -> IO a
  529 withBitBlastedTerm proxy sc addlPrims t c = AIG.withNewGraph proxy $ \be -> do
  530   (bval, _) <- bitBlastTerm be sc addlPrims t
  531   v <- flattenBValue bval
  532   c be v
  533 
  534 asFiniteType :: SharedContext -> Term -> IO FiniteType
  535 asFiniteType sc t =
  536   scGetModuleMap sc >>= \modmap ->
  537   case asFiniteTypeValue (Concrete.evalSharedTerm modmap Map.empty Map.empty t) of
  538     Just ft -> return ft
  539     Nothing ->
  540       fail $ "asFiniteType: unsupported type " ++ scPrettyTerm defaultPPOpts t
  541 
  542 processVar ::
  543   (ExtCns Term, FirstOrderType) ->
  544   IO (ExtCns Term, FiniteType)
  545 processVar (ec, fot) =
  546   case toFiniteType fot of
  547     Nothing -> fail ("ABC solver does not support variables of type " ++ show fot)
  548     Just ft -> pure (ec, ft)

  551 withBitBlastedSATQuery ::
  552   AIG.IsAIG l g =>
  553   AIG.Proxy l g ->
  554   SharedContext ->
  555   PrimMap l g ->
  556   SATQuery ->
  557   (forall s. g s -> l s -> [(ExtCns Term, FiniteType)] -> IO a) ->
  558   IO a
  559 withBitBlastedSATQuery proxy sc addlPrims satq cont =
  560   do unless (Set.null (satUninterp satq)) $ fail
  561         "RME prover does not support uninterpreted symbols"
  562      t <- satQueryAsTerm sc satq
  563      varShapes <- mapM processVar (Map.toList (satVariables satq))
  564      modmap <- scGetModuleMap sc
  565      AIG.withNewGraph proxy $ \be ->
  566        do vars <- traverse (traverse (newVars be)) varShapes
  567           let varMap = Map.fromList [ (ecVarIndex ec, v) | (ec,v) <- vars ]
  568           x <- bitBlastBasic be modmap addlPrims varMap t
  569           case x of
  570             VBool l -> cont be l varShapes
  571             _ -> fail "SAWCoreAIG.BitBlast.withBitBlastedSATQuery: non-boolean result type."
