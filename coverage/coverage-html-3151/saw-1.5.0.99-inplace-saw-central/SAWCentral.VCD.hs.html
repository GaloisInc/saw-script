   46 -- | Signals in a VCD file use internal identifiers that consist of
   47 -- strings of printable ASCII characters ranging from character code
   48 -- 33 '!' to code 126 '~'.
   49 -- We represent internal identifiers as non-negative integers.
   50 type WireId = Int
   51 
   52 -- | Signals in a VCD file use internal identifiers that consist of
   53 -- strings of printable ASCII characters ranging from character code
   54 -- 33 '!' to code 126 '~'.
   55 prettyWireId :: WireId -> Text
   56 prettyWireId n
   57   | n < 0 = ""
   58   | otherwise =
   59     let c = chr (33 + (n `mod` 94))
   60         s = prettyWireId (n `div` 94 - 1)
   61     in Text.cons c s
   62 
   63 -- | A hierarchical set of VCD wire declarations. The leaves are named
   64 -- signals with a bit width and a name; these may be grouped
   65 -- hierarchically into named scopes.
   66 data Decl
   67   = Wire Int Text -- ^ Bit width, user-visible signal name
   68   | Scope Text [Decl]
   69 
   70 -- | Given a lazy stream of internal identifiers, a top name, and an
   71 -- evaluated type, either compute the set of VCD wire declarations or
   72 -- fail with an error message if the type is not supported.
   73 mkDecls :: Text -> C.TValue -> Either String [Decl]
   74 mkDecls name tv =
   75   case tv of
   76     C.TVBit           -> pure [Wire 1 name]
   77     C.TVSeq n C.TVBit -> pure [Wire (fromIntegral n) name]
   78     C.TVSeq n tv'     -> do let name' i = name <> Text.pack ("[" ++ show i ++ "]")
   79                             concat <$> sequence [ mkDecls (name' i) tv' | i <- [0 .. n-1] ]
   80     C.TVTuple tvs     -> do let names = [ Text.pack (show i) | i <- [(0::Int)..] ]
   81                             dss <- sequence (zipWith mkDecls names tvs)
   82                             pure [Scope name (concat dss)]
   83     C.TVRec tvm       -> do let fields = C.displayFields tvm
   84                             dss <- sequence [ mkDecls (C.identText i) tv' | (i, tv') <- fields ]
   85                             pure [Scope name (concat dss)]
   86     C.TVStream _      -> Left "stream type"
   87     C.TVInteger       -> Left "integer type"
   88     C.TVFloat _ _     -> Left "float type"
   89     C.TVIntMod _      -> Left "Z type"
   90     C.TVRational      -> Left "rational type"
   91     C.TVArray _ _     -> Left "array type"
   92     C.TVFun _ _       -> Left "function type"
   93     C.TVNominal{}     -> Left "nominal type"
   94 
   95 -- | Given a internal identifier supply and a 'Decl', return the
   96 -- remaining supply and the pretty-printed output.
   97 prettyDecl :: WireId -> Decl -> (WireId, [Text])
   98 prettyDecl i decl =
   99   case decl of
  100     Wire w name ->
  101       (i+1, [Text.unwords ["$var wire", Text.pack (show w), prettyWireId i, name, "$end"]])
  102     Scope name decls ->
  103       let (i', outputs) = prettyDecls i decls
  104           start = Text.unwords ["$scope module", name, "$end"]
  105           end = "$upscope $end"
  106       in (i', [start] ++ outputs ++ [end])
  107 
  108 prettyDecls :: WireId -> [Decl] -> (WireId, [Text])
  109 prettyDecls i decls = concat <$> mapAccumL prettyDecl i decls
  110 
  111 -- | Flatten a compound Cryptol value of an appropriate type to a list
  112 -- of concrete booleans and bit vectors.
  113 flatten :: C.Value -> C.Eval [Either Bool Integer]
  114 flatten v =
  115   case v of
  116     C.VRecord rm ->
  117       do vs <- sequence (C.displayElements rm)
  118          concat <$> traverse flatten vs
  119     C.VTuple xs ->
  120       do vs <- sequence xs
  121          concat <$> traverse flatten vs
  122     C.VBit b ->
  123       pure [Left b]
  124     C.VSeq n xs ->
  125       do vs <- sequence (C.enumerateSeqMap n xs)
  126          concat <$> traverse flatten vs
  127     C.VWord w ->
  128       do bv <- C.asWordVal C.Concrete w
  129          pure [Right (C.bvVal bv)]
  130     C.VInteger _   -> unsupported
  131     C.VRational _  -> unsupported
  132     C.VFloat _     -> unsupported
  133     C.VStream _    -> unsupported
  134     C.VFun _ _     -> unsupported
  135     C.VPoly _ _    -> unsupported
  136     C.VNumPoly _ _ -> unsupported
  137     C.VEnum _ _    -> unsupported
  138   where
  139     unsupported = panic "write_vcd" ["unsupported value type"]

  142 -- | Pretty-print a bit-vector literal in binary.
  143 prettyInteger :: Integer -> Text
  144 prettyInteger 0 = "b0"
  145 prettyInteger n0 = go "" n0
  146   where
  147     go :: Text -> Integer -> Text
  148     go s 0 = "b" <> s
  149     go s n =
  150       let c = if even n then '0' else '1'
  151       in go (Text.cons c s) (n `div` 2)
  152 
  153 -- | Pretty-print a single wire update.
  154 prettyUpdate :: (WireId, Either Bool Integer) -> Text
  155 prettyUpdate (i, x) =
  156   case x of
  157     Left b -> (if b then "1" else "0") <> prettyWireId i
  158     Right n -> Text.unwords [prettyInteger n, prettyWireId i]
  159 
  160 -- | Pretty-print all the wire updates for the given clock cycle.
  161 prettyUpdates :: Int -> [(WireId, Either Bool Integer)] -> [Text]
  162 prettyUpdates n upds =
  163   ("#" <> Text.pack (show n)) : map prettyUpdate upds
  164 
  165 -- | Given a list of wire values from the previous cycle and another
  166 -- list from the current cycle, return the list of all list positions
  167 -- and elements that have changed.
  168 filterUpdates :: [Either Bool Integer] -> [Either Bool Integer] -> [(WireId, Either Bool Integer)]
  169 filterUpdates = go 0
  170   where
  171     go !i (x : xs) (y : ys)
  172       | x == y = go (i+1) xs ys
  173       | otherwise = (i, y) : go (i+1) xs ys
  174     go _ _ _ = []
  175 
  176 generateVCD :: TypedTerm -> TopLevel [Text]
  177 generateVCD t =
  178   do sc <- getSharedContext
  179      unless (closedTerm (ttTerm t)) $
  180        fail "write_vcd: term contains symbolic variables"
  181      tv0 <-
  182        case ttType t of
  183          TypedTermSchema (C.Forall [] [] ty) -> pure (C.evalValType mempty ty)
  184          _ -> fail "write_vcd: term is not monomorphic"
  185      (_len, tv) <-
  186        case tv0 of
  187          C.TVSeq n tv -> pure (n, tv)
  188          _ -> fail "write_vcd: term is not a finite sequence"
  189      decls <-
  190        case mkDecls "top" tv of
  191          Left msg -> fail $ "write_vcd: unsupported " ++ msg
  192          Right ds -> pure ds
  193      v <- io $ rethrowEvalError $ evaluateTypedTerm sc t
  194      vs <-
  195        case v of
  196          C.VSeq n xs -> io $ C.runEval mempty $ sequence (C.enumerateSeqMap n xs)
  197          _ -> fail "write_vcd: term is not a finite sequence"
  198      cycles <- io $ C.runEval mempty $ traverse flatten vs
  199      -- The first cycle should be printed in its entirety.
  200      -- On subsequent cycles, we should filter out any signals whose
  201      -- values didn't change.
  202      let updates =
  203            case cycles of
  204              [] -> []
  205              cycle0 : cycles' ->
  206                zip [0..] cycle0 : zipWith filterUpdates cycles cycles'
  207      let out1 = ["$version SAW write_vcd $end", "$timescale 1ns $end"]
  208      let out2 = snd $ prettyDecls 0 decls
  209      let out3 = concat (zipWith prettyUpdates [0..] updates)
  210      pure (out1 ++ out2 ++ out3)
