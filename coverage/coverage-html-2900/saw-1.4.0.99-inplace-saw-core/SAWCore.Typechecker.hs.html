   56 -- | Infer the type of an untyped term and complete it to a 'Term'.
   57 inferCompleteTerm ::
   58   SharedContext -> Maybe ModuleName -> Un.UTerm -> IO (Either PPS.Doc Term)
   59 inferCompleteTerm sc mnm t =
   60   do res <- runCheckM (typeInferCompleteUTerm t) sc mnm
   61      case res of
   62        -- TODO: avoid intermediate 'String's from 'prettyTCError'
   63        Left err -> return $ Left $ vsep $ map pretty $ TC.prettyTCError err
   64        Right t' -> return $ Right t'
   65 
   66 -- | The 'ReaderT' environment for a computation to typecheck a
   67 -- SAWCore parser AST.
   68 data CheckEnv =
   69   CheckEnv
   70   { tcModName :: Maybe ModuleName -- ^ the current module name
   71   , tcLocals :: Map LocalName (VarName, Term) -- ^ the mapping of display names to variables
   72   }
   73 
   74 -- | The monad for computations to typecheck a SAWCore parser AST.
   75 type CheckM = ReaderT CheckEnv TC.TCM
   76 
   77 runCheckM ::
   78   CheckM a -> SharedContext -> Maybe ModuleName -> IO (Either TC.TCError a)
   79 runCheckM m sc mnm =
   80   TC.runTCM (runReaderT m (CheckEnv mnm Map.empty)) sc
   81 
   82 -- | Read the current module name
   83 askModName :: CheckM (Maybe ModuleName)
   84 askModName = asks tcModName
   85 
   86 -- | Read the current context of named variables
   87 askLocals :: CheckM (Map LocalName (VarName, Term))
   88 askLocals = asks tcLocals
   89 
   90 -- | Look up the current module name, raising an error if it is not set
   91 getModuleName :: CheckM ModuleName
   92 getModuleName =
   93   do maybe_modname <- askModName
   94      case maybe_modname of
   95        Just mnm -> return mnm
   96        Nothing ->
   97          panic "getModuleName" ["Current module name not set during typechecking"]

  101 inferTermF :: TermF Term -> CheckM SC.Term
  102 inferTermF x = lift $ TC.inferTermF x
  103 
  104 inferFlatTermF :: FlatTermF Term -> CheckM SC.Term
  105 inferFlatTermF x = lift $ TC.inferFlatTermF x
  106 
  107 -- | Build a multi-arity application of 'SC.Term's
  108 inferApplyAll :: SC.Term -> [SC.Term] -> CheckM SC.Term
  109 inferApplyAll t [] = return t
  110 inferApplyAll t (arg:args) =
  111   do app1 <- inferTermF (App t arg)
  112      inferApplyAll app1 args
  113 
  114 -- | Resolve a name.
  115 inferResolveName :: Text -> CheckM SC.Term
  116 inferResolveName n =
  117   do nctx <- askLocals
  118      mnm <- getModuleName
  119      mm <- lift $ TC.liftTCM scGetModuleMap
  120      let ident = mkIdent mnm n
  121      case (Map.lookup n nctx, resolveNameInMap mm ident) of
  122        (Just (vn, tp), _) ->
  123          inferTermF (Variable vn tp)
  124        (_, Just rn) ->
  125          do let c = resolvedNameName rn
  126             inferTermF (Constant c)
  127        (Nothing, Nothing) ->
  128          throwTCError $ UnboundName n
  129 
  130 -- | Match an untyped term as a name applied to 0 or more arguments
  131 matchAppliedName :: Un.UTerm -> Maybe (Text, [Un.UTerm])
  132 matchAppliedName (Un.Name (PosPair _ n)) = Just (n, [])
  133 matchAppliedName (Un.App f arg) =
  134   do (n, args) <- matchAppliedName f
  135      return (n, args++[arg])
  136 matchAppliedName _ = Nothing
  137 
  138 -- | Match an untyped term as a recursor applied to 0 or more arguments
  139 matchAppliedRecursor :: Un.UTerm -> Maybe (Text, Sort, [Un.UTerm])
  140 matchAppliedRecursor (Un.Recursor (PosPair _ n) s) = Just (n, s, [])
  141 matchAppliedRecursor (Un.App f arg) =
  142   do (n, s, args) <- matchAppliedRecursor f
  143      return (n, s, args++[arg])
  144 matchAppliedRecursor _ = Nothing

  146 -- | The debugging level
  147 debugLevel :: Int
  148 debugLevel = 0

  150 -- | Print debugging output if 'debugLevel' is greater than 0
  151 typeInferDebug :: String -> CheckM ()
  152 typeInferDebug str | debugLevel > 0 = liftIO $ traceIO str
  153 typeInferDebug _ = return ()

  155 -- | Completely typecheck an untyped SAWCore AST.
  156 typeInferCompleteUTerm :: Un.UTerm -> CheckM SC.Term
  157 typeInferCompleteUTerm t =
  158   do typeInferDebug ("typechecking term: " ++ show t)
  159      res <- atPos (pos t) $ TC.withErrorUTerm t $ typeInferCompleteTerm t
  160      ty <- lift $ TC.liftTCM SC.scTypeOf res
  161      typeInferDebug ("completed typechecking term: " ++ show t ++ "\n"
  162                      ++ "type = " ++ show ty)
  163      return res
  164 
  165 -- | Main workhorse function for type inference on untyped terms
  166 typeInferCompleteTerm :: Un.UTerm -> CheckM SC.Term
  167 
  168 -- Names
  169 typeInferCompleteTerm (matchAppliedName -> Just (n, args)) =
  170   do t <- inferResolveName n
  171      ts <- traverse typeInferCompleteUTerm args
  172      inferApplyAll t ts
  173 typeInferCompleteTerm (Un.Name (PosPair _ n)) =
  174   -- NOTE: this is actually covered by the previous case, but we put it here
  175   -- so GHC doesn't complain about coverage
  176   inferResolveName n
  177 
  178 -- Sorts
  179 typeInferCompleteTerm (Un.Sort _ srt h) =
  180   inferFlatTermF (Sort srt h)
  181 
  182 -- Recursors (must come before applications)
  183 typeInferCompleteTerm (matchAppliedRecursor -> Just (str, s, args)) =
  184   do mnm <- getModuleName
  185      mm <- lift $ TC.liftTCM scGetModuleMap
  186      let dt_ident = mkIdent mnm str
  187      dt <- case findDataTypeInMap dt_ident mm of
  188        Just d -> return d
  189        Nothing -> throwTCError $ NoSuchDataType (ModuleIdentifier dt_ident)
  190      typed_args <- mapM typeInferCompleteUTerm args
  191      crec <- lift $ TC.compileRecursor dt s
  192      r <- inferFlatTermF (Recursor crec)
  193      inferApplyAll r typed_args
  194 
  195 typeInferCompleteTerm (Un.Recursor _ _) =
  196   error "typeInferComplete: found a bare Recursor, which should never happen!"
  197 
  198 -- Applications, lambdas, and pis
  199 typeInferCompleteTerm (Un.App f arg) =
  200   (App <$> typeInferCompleteUTerm f <*> typeInferCompleteUTerm arg)
  201   >>= inferTermF
  202 typeInferCompleteTerm (Un.Lambda _ [] t) = typeInferCompleteUTerm t
  203 typeInferCompleteTerm (Un.Lambda p ((Un.termVarLocalName -> x, tp) : ctx) t) =
  204   do tp_trm <- typeInferCompleteUTerm tp
  205      _ <- lift $ TC.ensureSortType tp_trm
  206      -- NOTE: we need the type of x to be normalized when we add it to the
  207      -- context in withVar, but we do not want to normalize this type in the
  208      -- output, as the contract for typeInferComplete only normalizes the type,
  209      -- so we use the unnormalized tp_trm in the return
  210      -- tp_whnf <- lift $ TC.typeCheckWHNF tp_trm
  211      vn <- lift $ TC.liftTCM scFreshVarName x
  212      body <- withVar x vn tp_trm $
  213        typeInferCompleteUTerm $ Un.Lambda p ctx t
  214      inferTermF (Lambda vn tp_trm body)
  215 typeInferCompleteTerm (Un.Pi _ [] t) = typeInferCompleteUTerm t
  216 typeInferCompleteTerm (Un.Pi p ((Un.termVarLocalName -> x, tp) : ctx) t) =
  217   do tp_trm <- typeInferCompleteUTerm tp
  218      -- NOTE: we need the type of x to be normalized when we add it to the
  219      -- context in withVar, but we do not want to normalize this type in the
  220      -- output, as the contract for typeInferComplete only normalizes the type,
  221      -- so we use the unnormalized tp_trm in the return
  222      tp_whnf <- lift $ TC.typeCheckWHNF tp_trm
  223      vn <- lift $ TC.liftTCM scFreshVarName x
  224      body <- withVar x vn tp_whnf $
  225        typeInferCompleteUTerm $ Un.Pi p ctx t
  226      result <- inferTermF (Pi vn tp_trm body)
  227      pure result
  228 
  229 -- Non-dependent records
  230 typeInferCompleteTerm (Un.RecordValue _ elems) =
  231   do typed_elems <-
  232        mapM (\(PosPair _ fld, t) -> (fld,) <$> typeInferCompleteUTerm t) elems
  233      inferFlatTermF (RecordValue typed_elems)
  234 typeInferCompleteTerm (Un.RecordType _ elems) =
  235   do typed_elems <-
  236        mapM (\(PosPair _ fld, t) -> (fld,) <$> typeInferCompleteUTerm t) elems
  237      inferFlatTermF (RecordType typed_elems)
  238 typeInferCompleteTerm (Un.RecordProj t prj) =
  239   (RecordProj <$> typeInferCompleteUTerm t <*> return prj) >>= inferFlatTermF
  240 
  241 -- Unit
  242 typeInferCompleteTerm (Un.UnitValue _) =
  243   inferFlatTermF UnitValue
  244 typeInferCompleteTerm (Un.UnitType _) =
  245   inferFlatTermF UnitType
  246 
  247 -- Simple pairs
  248 typeInferCompleteTerm (Un.PairValue _ t1 t2) =
  249   (PairValue <$> typeInferCompleteUTerm t1 <*> typeInferCompleteUTerm t2)
  250   >>= inferFlatTermF
  251 typeInferCompleteTerm (Un.PairType _ t1 t2) =
  252   (PairType <$> typeInferCompleteUTerm t1 <*> typeInferCompleteUTerm t2)
  253   >>= inferFlatTermF
  254 typeInferCompleteTerm (Un.PairLeft t) =
  255   (PairLeft <$> typeInferCompleteUTerm t) >>= inferFlatTermF
  256 typeInferCompleteTerm (Un.PairRight t) =
  257   (PairRight <$> typeInferCompleteUTerm t) >>= inferFlatTermF
  258 
  259 -- Type ascriptions
  260 typeInferCompleteTerm (Un.TypeConstraint t _ tp) =
  261   do typed_t <- typeInferCompleteUTerm t
  262      typed_tp <- typeInferCompleteUTerm tp
  263      _ <- lift $ TC.ensureSortType typed_tp
  264      lift $ TC.checkSubtype typed_t typed_tp
  265      return typed_t
  266 
  267 -- Literals
  268 typeInferCompleteTerm (Un.NatLit _ i) =
  269   lift $ TC.liftTCM SC.scNat i
  270 typeInferCompleteTerm (Un.StringLit _ str) =
  271   inferFlatTermF (StringLit str)
  272 typeInferCompleteTerm (Un.VecLit _ []) = throwTCError EmptyVectorLit
  273 typeInferCompleteTerm (Un.VecLit _ ts) =
  274   do typed_ts <- mapM typeInferCompleteUTerm ts
  275      typed_tp <-
  276        case typed_ts of
  277          (t1:_) -> lift $ TC.liftTCM SC.scTypeOf t1
  278          [] -> throwTCError $ EmptyVectorLit
  279      inferFlatTermF $ ArrayValue typed_tp $ V.fromList typed_ts
  280 typeInferCompleteTerm (Un.BVLit _ []) = throwTCError EmptyVectorLit
  281 typeInferCompleteTerm (Un.BVLit _ bits) =
  282   do tp <- lift $ TC.liftTCM scBoolType
  283      bit_tms <- lift $ mapM (TC.liftTCM scBool) bits
  284      inferFlatTermF $ ArrayValue tp $ V.fromList bit_tms
  285 
  286 typeInferCompleteTerm (Un.BadTerm _) =
  287   -- Should be unreachable, since BadTerms represent parse errors, that should
  288   -- already have been signaled before type inference
  289   panic "typeInferCompleteTerm" ["Type inference encountered a BadTerm"]

  296 -- | Type-check a list of declarations and insert them into the current module
  297 processDecls :: [Un.Decl] -> CheckM ()
  298 processDecls [] = return ()
  299 
  300 processDecls (Un.InjectCodeDecl ns txt : rest) =
  301   do mnm <- getModuleName
  302      lift $ TC.liftTCM scInjectCode mnm ns txt
  303      processDecls rest
  304 
  305 processDecls (Un.TypedDef nm params rty body : rest) =
  306   processDecls (Un.TypeDecl NoQualifier nm (Un.Pi (pos nm) params rty) :
  307                 Un.TermDef nm (map fst params) body : rest)
  308 
  309 processDecls (Un.TypeDecl NoQualifier (PosPair p nm) tp :
  310               Un.TermDef (PosPair _ ((== nm) -> True)) vars body : rest) =
  311   -- Type-checking for definitions
  312   (atPos p $
  313    do
  314      -- Step 1: type-check the type annotation, and make sure it is a type
  315      typed_tp <- typeInferCompleteUTerm tp
  316      void $ lift $ TC.ensureSortType typed_tp
  317      let def_tp = typed_tp
  318      def_tp_whnf <- lift $ TC.liftTCM scWhnf def_tp
  319 
  320      -- Step 2: assign types to the bound variables of the definition, by
  321      -- peeling off the pi-abstraction variables in the type annotation. Any
  322      -- remaining body of the pi-type is the required type for the def body.
  323      matchResult <-
  324        runMaybeT $ matchPiWithNames (map Un.termVarLocalName vars) def_tp_whnf
  325      (ctx, req_body_tp) <-
  326        case matchResult of
  327          Just x -> return x
  328          Nothing ->
  329              throwTCError $
  330              DeclError nm ("More variables " ++ show (map Un.termVarLocalName vars) ++
  331                            " than length of function type:\n" ++
  332                            showTerm typed_tp)
  333 
  334      -- Step 3: type-check the body of the definition in the context of its
  335      -- variables, and build a function that takes in those variables
  336      def_tm <-
  337        withCtx ctx $
  338        do typed_body <- typeInferCompleteUTerm body
  339           lift $ TC.checkSubtype typed_body req_body_tp
  340           let vts = map (\(_, v, t) -> (v, t)) ctx
  341           result <- lift $ TC.liftTCM scLambdaList vts typed_body
  342           pure result
  343 
  344      -- Step 4: add the definition to the current module
  345      mnm <- getModuleName
  346      let nmi = ModuleIdentifier (mkIdent mnm nm)
  347      void $ lift $ TC.liftTCM scDefineConstant nmi def_tm def_tp) >>
  348   processDecls rest
  349 
  350 processDecls (Un.TypeDecl NoQualifier (PosPair p nm) _ : _) =
  351   atPos p $ throwTCError $ DeclError nm "Definition without defining equation"
  352 
  353 processDecls (Un.TypeDecl _ (PosPair p nm) _ :
  354               Un.TermDef (PosPair _ ((== nm) -> True)) _ _ : _) =
  355   atPos p $ throwTCError $ DeclError nm "Primitive or axiom with definition"
  356 
  357 processDecls (Un.TypeDecl q (PosPair p nm) tp : rest) =
  358   (atPos p $
  359    do typed_tp <- typeInferCompleteUTerm tp
  360       void $ lift $ TC.ensureSortType typed_tp
  361       mnm <- getModuleName
  362       let ident = mkIdent mnm nm
  363       let def_tp = typed_tp
  364       lift $ TC.liftTCM scDeclarePrim ident q def_tp) >>
  365   processDecls rest
  366 
  367 processDecls (Un.TermDef (PosPair p nm) _ _ : _) =
  368   atPos p $ throwTCError $ DeclError nm "Dangling definition without a type"
  369 
  370 processDecls (Un.DataDecl (PosPair p nm) param_ctx dt_tp c_decls : rest) =
  371   let param_ctx' = map (\(x, t) -> (Un.termVarLocalName x, t)) param_ctx in
  372   -- This top line makes sure that we process the rest of the decls after the
  373   -- main body of the code below, which processes just the current data decl
  374   (>> processDecls rest) $ atPos p $
  375   -- Step 1: type-check the parameters
  376   typeInferCompleteInCtx param_ctx' $ \params -> do
  377   let dtParams = map (\(_, v, t, _) -> (v, t)) params
  378   let param_sort = maxSort (map (\(_, _, _, s) -> s) params)
  379   let err :: String -> CheckM a
  380       err msg = throwTCError $ DeclError nm msg
  381 
  382   -- Step 2: type-check the type given for d, and make sure it is of the form
  383   -- (i1:ix1) -> ... -> (in:ixn) -> Type s for some sort s. Then form the full
  384   -- type of d as (p1:param1) -> ... -> (i1:ix1) -> ... -> Type s
  385   (dt_ixs, dtSort) <-
  386     case Un.asPiList dt_tp of
  387       (ixs, Un.Sort _ s h) | h == noFlags ->
  388         return (ixs, s) -- NB, don't allow `isort`, etc.
  389       _ -> err "Wrong form for type of datatype"
  390   let dt_ixs' = map (\(x, t) -> (Un.termVarLocalName x, t)) dt_ixs
  391   dt_ixs_typed <- typeInferCompleteCtx dt_ixs'
  392   let dtIndices = map (\(_, v, t, _) -> (v, t)) dt_ixs_typed
  393       ixs_max_sort = maxSort (map (\(_, _, _, s) -> s) dt_ixs_typed)
  394 
  395   -- Step 3: do the necessary universe inclusion checking for any predicative
  396   -- (non-Prop) inductive type, which includes:
  397   --
  398   -- 1. All ix types must be of sort dtSort; AND
  399   -- 2. All param types must be of sort dtSort+1
  400   if dtSort /= propSort && param_sort > sortOf dtSort then
  401     err ("Universe level of parameters should be no greater" ++
  402          " than that of the datatype")
  403     else return ()
  404   if dtSort /= propSort && ixs_max_sort > dtSort then
  405     err ("Universe level of indices should be strictly contained" ++
  406          " in that of the datatype")
  407     else return ()
  408 
  409   -- Step 4: Add d as an empty datatype, so we can typecheck the constructors
  410   mnm <- getModuleName
  411   let dtIdent = mkIdent mnm nm
  412   pn <- lift $ TC.liftTCM scBeginDataType dtIdent dtParams dtIndices dtSort
  413 
  414   -- Step 5: typecheck the constructors, and build Ctors for them
  415   typed_ctors <-
  416     mapM (\(Un.Ctor (PosPair p' c) ctx body) ->
  417            (c,) <$> typeInferCompleteUTerm (Un.Pi p' ctx body)) c_decls
  418   ctors <-
  419     forM typed_ctors $ \(c, typed_tp) ->
  420     -- Check that the universe level of the type of each constructor
  421     do case termSortOrType typed_tp of
  422            Left ctor_sort
  423              | dtSort /= propSort && ctor_sort > dtSort ->
  424                err ("Universe level of constructors should be strictly" ++
  425                     " contained in that of the datatype")
  426            Left _ ->
  427                return ()
  428            Right ty ->
  429                panic "processDecls" [
  430                    "Type of the type of constructor is not a sort!",
  431                    "Constructor type: " <> Text.pack (showTerm typed_tp),
  432                    "Type of that type: " <> Text.pack (showTerm ty)
  433                ]
  434        let tp = typed_tp
  435        let result = mkCtorArgStruct pn dtParams dtIndices tp
  436        case result of
  437          Just arg_struct ->
  438            lift $ TC.liftTCM scBuildCtor pn (mkIdent mnm c) arg_struct
  439          Nothing -> err ("Malformed type form constructor: " ++ show c)
  440 
  441   -- Step 6: complete the datatype with the given ctors
  442   lift $ TC.liftTCM scCompleteDataType dtIdent ctors
  443

  445 -- | Typecheck a module and, on success, insert it into the current context
  446 tcInsertModule :: SharedContext -> Un.Module -> IO ()
  447 tcInsertModule sc (Un.Module (PosPair _ mnm) imports decls) = do
  448   -- First, insert an empty module for mnm
  449   scLoadModule sc $ emptyModule mnm
  450   -- Next, process all the imports
  451   forM_ imports $ \imp ->
  452     do let imn = val $ Un.importModName imp
  453        i_exists <- scModuleIsLoaded sc imn
  454        unless i_exists $ fail $ "Imported module not found: " ++ show imn
  455        scImportModule sc (Un.nameSatsConstraint (Un.importConstraints imp) . Text.unpack) imn mnm
  456   -- Finally, process all the decls
  457   decls_res <- runCheckM (processDecls decls) sc (Just mnm)
  458   case decls_res of
  459     Left err -> fail $ unlines $ TC.prettyTCError err
  460     Right _ -> return ()

  467 -- | Pattern match a nested pi-abstraction, like 'asPiList', but only match as
  468 -- far as the supplied list of variables, and use them as the new names
  469 matchPiWithNames ::
  470   [LocalName] -> Term -> MaybeT CheckM ([(LocalName, VarName, Term)], Term)
  471 matchPiWithNames [] tp = pure ([], tp)
  472 matchPiWithNames (var : vars) tp =
  473   case asPi tp of
  474     Nothing -> mzero
  475     Just (x, arg_tp, body_tp) ->
  476       do vn <-
  477            if IntSet.member (vnIndex x) (freeVars body_tp)
  478            then -- dependent function type: use name from Pi type
  479              pure x
  480            else -- non-dependent function: use 'var' as base name
  481              lift $ lift $ TC.liftTCM scFreshVarName var
  482          (ctx, body) <- matchPiWithNames vars body_tp
  483          pure ((var, vn, arg_tp) : ctx, body)

  485 -- | Run a type-checking computation in a typing context extended with a new
  486 -- variable with the given name and type.
  487 withVar :: LocalName -> VarName -> Term -> CheckM a -> CheckM a
  488 withVar x vn tp m =
  489   local (\env -> env { tcLocals = Map.insert x (vn, tp) (tcLocals env) }) m
  490 
  491 -- | Run a type-checking computation in a typing context extended by a list of
  492 -- variables and their types. See 'withVar'.
  493 withCtx :: [(LocalName, VarName, Term)] -> CheckM a -> CheckM a
  494 withCtx = flip (foldr (\(x, vn, tp) -> withVar x vn tp))
  495 
  496 -- | Perform type inference on a context, i.e., a list of variable names and
  497 -- their associated types. This will give us 'Term's for each type, as
  498 -- well as their 'Sort's, since the type of any type is a 'Sort'.
  499 typeInferCompleteCtx ::
  500   [(LocalName, Un.UTerm)] -> CheckM [(LocalName, VarName, Term, Sort)]
  501 typeInferCompleteCtx [] = pure []
  502 typeInferCompleteCtx ((x, tp) : ctx) =
  503   do typed_tp <- typeInferCompleteUTerm tp
  504      s <- lift $ TC.ensureSortType typed_tp
  505      vn <- lift $ TC.liftTCM scFreshVarName x
  506      let t' = typed_tp
  507      ((x, vn, t', s) :) <$> withVar x vn t' (typeInferCompleteCtx ctx)
  508 
  509 -- | Perform type inference on a context via 'typeInferCompleteCtx', and then
  510 -- run a computation in that context via 'withCtx', also passing in that context
  511 -- to the computation
  512 typeInferCompleteInCtx ::
  513   [(LocalName, Un.UTerm)] ->
  514   ([(LocalName, VarName, Term, Sort)] -> CheckM a) -> CheckM a
  515 typeInferCompleteInCtx ctx f =
  516   do typed_ctx <- typeInferCompleteCtx ctx
  517      withCtx (map (\(x, v, t, _) -> (x, v, t)) typed_ctx) (f typed_ctx)
