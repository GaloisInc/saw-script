   54 -- | Internal wrapper for 'TypeRep' that we provide an 'OrdF'
   55 --   instance for.
   56 newtype TypeRep' a = TypeRep' (TypeRep a)
   57   deriving (TestEquality, Show)
   58 
   59 instance OrdF TypeRep' where
   60   compareF (TypeRep' tr1) (TypeRep' tr2) =
   61     case testEquality tr1 tr2 of
   62       Just Refl -> EQF
   63       Nothing -> case compare (SomeTypeRep tr1) (SomeTypeRep tr2) of
   64         LT -> LTF
   65         GT -> GTF
   66         EQ -> error "inconsistent TypeRep equality"
   67 
   68 instance ShowF TypeRep'
   69 
   70 withRep :: forall a b. TypeRep' a -> (Typeable a => b) -> b
   71 withRep (TypeRep' tr) f = withTypeable tr f
   72 
   73 -- | A store of arbitrary 'Typeable' data, indexed by type.
   74 newtype TypedStore f = TypedStore (MapF TypeRep' f)
   75 
   76 -- | An empty 'TypedStore'.
   77 empty :: TypedStore f
   78 empty = TypedStore MapF.empty
   79 
   80 -- | Lookup the value of type 'f a' from the store.
   81 lookup :: forall a f. Typeable a => TypedStore f -> Maybe (f a)
   82 lookup (TypedStore tm) = MapF.lookup rep tm
   83   where
   84     rep = TypeRep' @a TypeRep
   85 
   86 -- | Alter the data of type 'f a' in the store, inserting or
   87 --   deleting based on the result of 'g'.
   88 alter :: forall a f.
   89   Typeable a =>
   90   (Maybe (f a) -> Maybe (f a)) ->
   91   TypedStore f ->
   92   TypedStore f
   93 alter g ts = runIdentity $ alterM (\ma -> Identity (g ma)) ts
   94 
   95 -- | Alter the data of type 'f a' in the store, inserting or deleting
   96 -- based on the result of the monadic action 'g'.
   97 alterM :: forall a f m.
   98   Typeable a =>
   99   Monad m =>
  100   (Maybe (f a) -> m (Maybe (f a))) ->
  101   TypedStore f ->
  102   m (TypedStore f)
  103 alterM g (TypedStore tm) = 
  104   (TypedStore . MapF.updatedValue) <$> tm'
  105   where
  106     rep = TypeRep' @a TypeRep
  107 
  108     go :: f a -> m (MapF.UpdateRequest (f a))
  109     go a = g (Just a) >>= \case
  110       Nothing -> pure $ MapF.Delete
  111       Just a' -> pure $ MapF.Set a'
  112 
  113     tm' = MapF.updateAtKey rep (g Nothing) go tm
  114 
  115 -- | Insert a value of type 'f a' in the store, overwriting existing
  116 --   data.
  117 insert :: forall a f. Typeable a => f a -> TypedStore f -> TypedStore f
  118 insert a = alter (\_ -> Just a)
  119 
  120 -- | Create a 'TypedStore' with a single entry of type 'f a'.
  121 singleton :: forall a f. (Typeable f, Typeable a) => f a -> TypedStore f
  122 singleton a = TypedStore (MapF.singleton rep a)
  123   where
  124     rep = TypeRep' @a TypeRep
  125 
  126 size :: TypedStore f -> Int
  127 size (TypedStore ts) = MapF.size ts
  128 
  129 -- | Delete the value of type 'f a' from the store, if present.
  130 delete :: forall a f proxy. Typeable a => proxy a -> TypedStore f -> TypedStore f
  131 delete _ tm = alter @a (\_ -> Nothing) tm
  132 
  133 map :: forall f g.
  134   (forall a. (Typeable a) => f a -> g a) -> 
  135   TypedStore f -> 
  136   TypedStore g
  137 map f (TypedStore ts) = TypedStore $ 
  138   MapF.mapWithKey (\rep x -> withRep rep $ f x) ts
  139 
  140 traverse :: forall f g m.
  141   Applicative m => 
  142   (forall a. (Typeable a) => f a -> m (g a)) -> 
  143   TypedStore f -> 
  144   m (TypedStore g)
  145 traverse f (TypedStore ts) = TypedStore <$>
  146   MapF.traverseWithKey (\rep x -> withRep rep $ f x) ts
  147 
  148 traverse_ :: forall f m.
  149   Applicative m => 
  150   (forall a. (Typeable a) => f a -> m ()) -> 
  151   TypedStore f -> 
  152   m ()
  153 traverse_ f (TypedStore ts) =
  154   MapF.traverseWithKey_ (\rep x -> withRep rep $ f x) ts
  155 
  156 merge :: forall f g h.
  157   -- keys in both
  158   (forall a. (Typeable a) => f a -> g a -> Maybe (h a)) ->
  159   -- keys only in left
  160   (forall a. (Typeable a) => f a -> Maybe (h a)) ->
  161   -- keys only in right
  162   (forall a. (Typeable a) => g a -> Maybe (h a)) ->
  163   TypedStore f ->
  164   TypedStore g ->
  165   TypedStore h
  166 merge f g1 g2 (TypedStore ts1) (TypedStore ts2) =
  167   TypedStore $ MapF.mergeWithKey 
  168     (\rep x y -> withRep rep $ f x y)
  169     (MapF.mapMaybeWithKey (\rep x -> withRep rep $ g1 x))
  170     (MapF.mapMaybeWithKey (\rep x -> withRep rep $ g2 x))
  171     ts1 ts2
  172 
  173 mergeM :: forall f g h m.
  174   Applicative m =>
  175   -- keys in both
  176   (forall a. (Typeable a) => f a -> g a -> m (Maybe (h a))) ->
  177   -- keys only in left
  178   (forall a. (Typeable a) => f a -> m (Maybe (h a))) ->
  179   -- keys only in right
  180   (forall a. (Typeable a) => g a -> m (Maybe (h a))) ->
  181   TypedStore f ->
  182   TypedStore g ->
  183   m (TypedStore h)
  184 mergeM f g1 g2 (TypedStore ts1) (TypedStore ts2) =
  185   TypedStore <$> MapF.mergeWithKeyM 
  186     (\rep x y -> withRep rep $ f x y)
  187     (MapF.traverseMaybeWithKey (\rep x -> withRep rep $ g1 x))
  188     (MapF.traverseMaybeWithKey (\rep x -> withRep rep $ g2 x))
  189     ts1 ts2
  190 
  191 -- | Combine two 'TypedStore's, using the given function to merge
  192 --   entries of the same type.
  193 union :: forall f.
  194   (forall a. (Typeable a) => f a -> f a -> Maybe (f a)) ->
  195   TypedStore f ->
  196   TypedStore f ->
  197   TypedStore f
  198 union f (TypedStore ts1) (TypedStore ts2) = TypedStore $
  199   MapF.mergeWithKey (\rep x y -> withRep rep $ f x y) id id ts1 ts2
  200 
  201 -- | Intersect two 'TypedStore's, using the given function to merge
  202 --   entries of the same type, and discarding types which only are
  203 --   only present in one store.
  204 intersection :: forall f.
  205   (forall a. (Typeable a) => f a -> f a -> Maybe (f a)) ->
  206   TypedStore f ->
  207   TypedStore f ->
  208   TypedStore f
  209 intersection f (TypedStore ts1) (TypedStore ts2) = TypedStore $
  210   MapF.mergeWithKey (\rep x y -> withRep rep $ f x y) 
  211     (const MapF.empty) (const MapF.empty) ts1 ts2
  212 
  213 toList :: forall f b.
  214   (forall a. (Typeable a) => f a -> b) -> TypedStore f -> [b]
  215 toList f (TypedStore ts) = 
  216   List.map (\(Pair rep x) -> withRep rep $ f x) $ MapF.toList ts
  217 
  218 instance FunctorF TypedStore where
  219   fmapF = map
  220 
  221 instance FoldableF TypedStore where
  222   foldrF f z (TypedStore ts) = foldrF f z ts
  223 
  224 instance TraversableF TypedStore where
  225   traverseF = traverse
  226 
  227 instance EqF f => Eq (TypedStore f) where
  228   (TypedStore ts1) == (TypedStore ts2) = ts1 == ts2
  229 
  230 newtype Pair' f g = Pair' (Pair f g)
  231   deriving (Eq)
  232 
  233 instance (EqF g, OrdF f, OrdF g) => Ord (Pair' f g) where
  234   compare (Pair' (Pair f1 g1)) (Pair' (Pair f2 g2)) =
  235     toOrdering (compareF f1 f2) <> toOrdering (compareF g1 g2)
  236 
  237 instance (EqF f, OrdF f) => Ord (TypedStore f) where
  238   compare (TypedStore ts1) (TypedStore ts2) = 
  239     compare 
  240       (coerce (MapF.toList ts1) :: [Pair' TypeRep' f]) 
  241       (coerce (MapF.toList ts2) :: [Pair' TypeRep' f])
  242 
  243 instance (EqF f, HashableF f) => Hashable (TypedStore f) where
  244   hashWithSalt i = foldrF (\x y -> hashWithSaltF y x) i
  245 
  246 instance ShowF f => Show (TypedStore f) where
  247   show (TypedStore ts) = show ts
