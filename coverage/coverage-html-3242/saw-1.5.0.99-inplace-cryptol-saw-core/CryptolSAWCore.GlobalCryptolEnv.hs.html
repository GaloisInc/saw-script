   82 -- | ImportVisibility is an enumeration that indicates how we handle
   83 -- the visibility of the symbols in an imported module.
   84 --
   85 -- `OnlyPublic` makes only the public/exported symbols of a module
   86 -- visible from SAW. `PublicAndPrivate` instead makes all symbols
   87 -- visible, as if one is working inside it. The latter is often useful
   88 -- (or necessary) for verification and code generation.
   89 --
   90 -- `PublicAndPrivate` is akin to setting the module focus in the
   91 -- Cryptol REPL; however, it uses a simpler internal mechanism and is
   92 -- only settable at import time.
   93 --
   94 -- (See 'CryptolEnv.importCryptolModule'.)
   95 --
   96 -- NOTE: this notion of public vs. private symbols is specific to
   97 -- SAWScript and distinct from Cryptol's notion of private
   98 -- definitions.
   99 --
  100 -- FUTURE: this should probably be replaced with a way to manipulate
  101 -- the module focus like the Cryptol REPL uses. What you really want
  102 -- is not to expose module innards that weren't meant to be exposed,
  103 -- but to go inside to prove things in the module's internal context.
  104 --
  105 data ImportVisibility
  106   = OnlyPublic       -- ^ behaves like a normal Cryptol @import@
  107   | PublicAndPrivate -- ^ allows viewing of both @private@ sections
  108                      --   and (arbitrarily nested) submodules.
  109   deriving (Eq, Show)

  112 -- | The global environment for capturing the Cryptol state, both
  113 --   Cryptol's own state and the state associated with
  114 --   importing/translating into SAWCore.
  115 --   This is intended to be a write-once record of any work done
  116 --   during translation, analagous to the 'SharedContext' from
  117 ---  SAWCore. Rather than directly accessing this environment,
  118 --   operations take/return a 'CryptolEnv', which additionally
  119 --   includes a scoped naming environment via 'CryptolScope'.
  120 
  121 
  122 --
  123 -- Note that prior to 202603 there were two environment types,
  124 -- `CryptolEnv` carrying around the persistent bits and generally
  125 -- being (in most places) the external interface; and another type
  126 -- called (far too generically) @Env@ used by the import logic in this
  127 -- file. There was a bunch of code for copying bits from `CryptolEnv`
  128 -- into an empty @Env@ on the fly, calling into here, then pouring the
  129 -- results back. This code was arbitrary and in some cases possibly
  130 -- wrong. Furthermore, having the import code tied to an incompatible
  131 -- type made a bunch of external code calling directly into it pass an
  132 -- empty environment instead, which caused further problems.
  133 --
  134 -- While this was being fixed the prior @Env@ type got renamed to
  135 -- @ImportEnv@. There should be no references to it or its field names
  136 -- (@imp*@ rather than @env*@) left, but in case some are hiding in
  137 -- comments the transitional field names are also documented below.
  138 --
  139 -- There is now one environment type. The history above remains
  140 -- relevant until all the leftover warts and weaknesses arising from
  141 -- the old structure get cleaned out, which may take a while.
  142 --
  143 -- (FUTURE: once that's done, remove the historical notes; they are
  144 -- only of value while they remain relevant to the current code.)
  145 data GlobalCryptolEnv = GlobalCryptolEnv
  146   { geModuleEnv   :: ME.ModuleEnv
  147   -- | Invariant: This is a subset of 'geAllVars', which is
  148   --   enforced by 'addExtraVars'
  149   , geExtraVars   :: Map C.Name C.Schema
  150   , geExtraTySyns :: Map C.Name C.TySyn
  151   , geAllVars     :: Map C.Name C.Schema
  152   , geTyVars      :: Map Int Term
  153   , geTyProps     :: Map C.Prop (Term, [FieldName])
  154   , geAllTerms    :: Map C.Name Term
  155   , geRefPrims    :: Map C.PrimIdent C.Expr
  156   , gePrims       :: Map C.PrimIdent Term
  157   , gePrimTypes   :: Map C.PrimIdent Term
  158   , geFFITypes    :: Map NameInfo C.FFI
  159   }

  169 -- | A scope frame that captures which Cryptol names are accessible.
  170 --  `fNamingEnv` is the local naming environment, which can be extended
  171 --  ad-hoc with additional declarations. `fImports` is a list of all
  172 --  the modules that have been imported, and a visibility setting for
  173 --  each. This does not include, for example, builtin modules that
  174 --  exist but that have not been imported.
  175 
  176 data CryptolFrame =
  177   CryptolFrame { fNamingEnv :: MR.NamingEnv
  178                , fImports :: [(ImportVisibility, C.Import)] 
  179                }
  180 
  181 initFrame :: CryptolFrame
  182 initFrame = CryptolFrame mempty mempty
  183 
  184 -- | A nonempty list of 'CryptolFrame's, where the first element is the
  185 --   "bottom" frame that takes highest precedence when looking up
  186 --   names. Each individual frame only contains values declared at
  187 --   exactly that level. The full scope is computed by collecting
  188 --   everything in this stack, via 'eExtraNaming' and 'eImports'.
  189 newtype CryptolScope = CryptolScope (NonEmpty CryptolFrame)
  190 
  191 initScope :: CryptolScope
  192 initScope = CryptolScope (initFrame :| [])
  193 
  194 isToplevelScope :: CryptolScope -> Bool
  195 isToplevelScope (CryptolScope (_ :| frames)) = null frames
  196 
  197 isToplevel :: CryptolEnv -> Bool
  198 isToplevel env = isToplevelScope (eScope env)
  199 
  200 -- | Test if the scopes have the same number of frames pushed.
  201 sameHeight :: CryptolScope -> CryptolScope -> Bool
  202 sameHeight (CryptolScope scope1) (CryptolScope scope2) = 
  203   NE.length scope1 == NE.length scope2
  204 
  205 mapCurFrame :: 
  206   (CryptolFrame -> CryptolFrame) -> 
  207   CryptolScope -> 
  208   CryptolScope
  209 mapCurFrame f (CryptolScope (frame :| frames)) =
  210   CryptolScope (f frame :| frames)
  211 
  212 -- | Push a fresh frame onto the stack.
  213 pushScope :: CryptolScope -> CryptolScope
  214 pushScope (CryptolScope frames) = CryptolScope (initFrame <| frames)
  215 
  216 -- | Pop the current frame from the stack, discarding its
  217 --   contents. Panics if this is the only frame.
  218 popScope :: CryptolScope -> CryptolScope
  219 popScope (CryptolScope frames) = case snd (NE.uncons frames) of
  220   Nothing -> panic "popCryptolScope" [ "Popping topmost scope"]
  221   Just frames' -> CryptolScope frames'
  222 
  223 -- | The full translation and Cryptol environment 'GlobalCryptolEnv',
  224 --   paired with a 'CryptolScope' indicating which names are currently
  225 --   in scope. Although the fields may be independently accessed, most
  226 --   operations are expected to operate on the full 'CryptolEnv'. 
  227 --
  228 --   It is generally safe to pair a previously-created 'CryptolScope'
  229 --   with a more recent 'GlobalCryptolEnv', as the names in the
  230 --   previous scope should remain valid. Conversely, it is *not* safe
  231 --   to pair a more recent 'CryptolScope' with an old
  232 --   'GlobalCryptolEnv', as the scope may contain entries which do not
  233 --   exist in the global environment.
  234 data CryptolEnv = CryptolEnv 
  235   { eGlobalEnv :: GlobalCryptolEnv
  236   , eScope :: CryptolScope 
  237   }
  238 
  239 initEnv :: ME.ModuleEnv -> CryptolEnv
  240 initEnv modEnv = CryptolEnv (initGlobalEnv modEnv) initScope

  242 -- | Map the naming environment of the frame currently in scope.
  243 mapScopeNaming :: 
  244   (MR.NamingEnv -> MR.NamingEnv) -> 
  245   CryptolScope -> 
  246   CryptolScope
  247 mapScopeNaming f = mapCurFrame $ 
  248       \fr -> fr {fNamingEnv = f (fNamingEnv fr) }
  249 
  250 -- | Map the module imports of the frame currently in scope.
  251 mapScopeImports :: 
  252   ([(ImportVisibility, C.Import)] -> [(ImportVisibility, C.Import)] ) -> 
  253   CryptolScope -> 
  254   CryptolScope
  255 mapScopeImports f = mapCurFrame $ 
  256       \fr -> fr {fImports = f (fImports fr) }
  257 
  258 -- | Map the naming environment currently in scope.
  259 mapNaming:: (MR.NamingEnv -> MR.NamingEnv) -> CryptolEnv -> CryptolEnv
  260 mapNaming f env = env { eScope = mapScopeNaming f (eScope env) }
  261 
  262 -- | Map the module imports currently in scope.
  263 mapImports :: 
  264   ([(ImportVisibility, C.Import)] -> [(ImportVisibility, C.Import)] ) -> 
  265   CryptolEnv -> 
  266   CryptolEnv
  267 mapImports f env = env { eScope = mapScopeImports f (eScope env) }

  270 -- | Run the inner action bracketed new frame pushed/popped on
  271 --   the 'CryptolScope' stack.
  272 --   Fails if the inner action changes the scope height
  273 --   (i.e. it does not properly bracket its pushes and pops).
  274 withFreshScope :: 
  275   MonadFail m =>
  276   CryptolEnv -> 
  277   (CryptolEnv -> 
  278   m (a, CryptolEnv)) -> 
  279   m (a, CryptolEnv)
  280 withFreshScope env0 f = do
  281   let env1 = env0 { eScope = pushScope (eScope env0) }
  282   (a, env2) <- f env1
  283   unless (sameHeight (eScope env1) (eScope env2)) $
  284     fail "withFreshScope: mismatched push/pops"
  285   let env3 = env2 { eScope = popScope (eScope env2) }
  286   return (a, env3)

  288 -- | Access the 'C.Supply' in the global 'ME.ModuleEnv' for generating
  289 --   fresh names. More efficient than directly modifying the
  290 --   environment and using 'setModuleEnv', as it avoids any other
  291 --   bookkeeping.
  292 withModEnvSupply :: CryptolEnv -> (C.Supply -> (a, C.Supply)) -> (a, CryptolEnv)
  293 withModEnvSupply env f = 
  294   let (a, supply) = f $ ME.meSupply $ eModuleEnv env
  295   in (a, mapModEnv (\modEnv -> modEnv { ME.meSupply = supply }) env)

  309 -- The "getters" below were historically fields in 'CryptolEnv', which
  310 -- are now defined functions that access either the 'GlobalCryptolEnv'
  311 -- or the 'CryptolScope' as appropriate. In contrast, updates were
  312 -- historically made by directly accessing 'CryptolEnv' fields, and are
  313 -- now restricted to only adding new entries to the maps in
  314 -- 'GlobalCryptolEnv' (e.g. 'addExtraVars'). This enforces the policy
  315 -- that the global environment is intended to be a complete record of
  316 -- all work done during translation/import, regardless of scope. Since
  317 -- the maps in the global environment are all keyed on globally-unique
  318 -- values, these operations are expected to only add entries and never
  319 -- overwrite existing ones.
  320 --
  321 -- NOTE: We could enforce a write-once policy here, but it's
  322 -- not immediately obvious if we need to support key clashes
  323 -- for equal entries (not possible for
  324 -- some times like 'C.Expr'), or if it is even useful to do so.

  526 -- | The "extra" naming environment that captures Cryptol names
  527 --   which don't correspond to any imported module. Generally these
  528 --   result from names created in SAW which have been reflected into
  529 --   the Cryptol environment.
  530 --   This is scoped content, where the accessible names are expected
  531 --   to be managed by SAW and correspond to the same SAW values that
  532 --   are in scope.
  533 
  534 -- FUTURE: Cryptol has its own functionality for additional bindings
  535 -- (it uses it for things created from the Cryptol REPL) and we ought
  536 -- to be able to use it instead of bolting on our own additional layer
  537 -- of material. Doing so would avoid various inconsistencies and
  538 -- irregularities that can creep in when we reimplement Cryptol name
  539 -- resolution.
  540 eExtraNaming :: CryptolEnv -> MR.NamingEnv
  541 eExtraNaming (eScope -> CryptolScope (frame :| frames)) = 
  542   foldr (\fr ne -> ne `MR.shadowing` (fNamingEnv fr)) (fNamingEnv frame) frames

  544 -- | The list of Cryptol modules which have been brought into the
  545 --   current scope. This essentially acts as a filter on the full set
  546 --   of loaded modules in the global module environment ('eModuleEnv'),
  547 --   where the contents of the selected modules are brought into scope
  548 --   according to the associated 'ImportVisibility'. The modules here
  549 --   should only correspond to modules that are present in the module
  550 --   environment *and* have been translated into SAWCore.
  551 eImports :: CryptolEnv -> [(ImportVisibility, C.Import)]
  552 eImports (eScope -> CryptolScope frames) = 
  553   concat $ map fImports $ NE.toList frames

  567 -- | Refresh 'geAllVars' after updating the module environment.
  568 --   Previously (before 'GlobalCryptolEnv'), this would overwrite the
  569 --   'eAllVars' field. Now this will add new vars (i.e. from
  570 --   newly-added modules) but keep existing ones, as the
  571 --   'GlobalCryptolEnv' should never drop entries. If the module
  572 --   environment has been managed properly, we expect any clashing keys
  573 --   to have equal elements, but this is not checked/enforced. Not
  574 --   exported.
  575 refreshCryptolEnv :: GlobalCryptolEnv -> GlobalCryptolEnv
  576 refreshCryptolEnv genv =
  577      let modEnv = geModuleEnv genv
  578          ifaceDecls = getAllIfaceDecls modEnv
  579          newtypeCons = Map.fromList
  580                          [ con
  581                          | nt <- Map.elems (MI.ifNominalTypes ifaceDecls)
  582                          , con <- C.nominalTypeConTypes nt
  583                          ]
  584          vars = Map.map MI.ifDeclSig $ MI.ifDecls ifaceDecls
  585          -- note that we don't need to re-add geExtraVars, because
  586          -- it is already included in the existing geAllVars
  587          allvars = newtypeCons `Map.union` vars
  588          allvars' = Map.union allvars (geAllVars genv)
  589      in genv { geAllVars = allvars' }
  590

  593 getAllIfaceDecls :: ME.ModuleEnv -> MI.IfaceDecls
  594 getAllIfaceDecls me =
  595   mconcat
  596     (map (MI.ifDefines . ME.lmInterface)
  597          (ME.getLoadedModules (ME.meLoadedModules me)))

  599 -- | Restore a `CryptolEnv` from a checkpoint. The first argument
  600 --   @chkEnv@ is the `CryptolEnv` saved by / copied into the
  601 --   checkpoint; the second argument @newEnv@ is the current one
  602 --   we wish to overwrite by rolling back to the checkpoint.
  603 --   The 'ME.meNameSeeds' and 'ME.meSupply' from the
  604 --   module environment are not rolled back, to avoid re-using old
  605 --   names.
  606 --   NOTE: This reverts the 'GlobalCryptolEnv', which effectively
  607 --   invalidates any translated 'Term's or Cryptol expressions created
  608 --   after the checkpoint. Attempting to use them in the restored
  609 --   environment will have unpredictable results, and likely will
  610 --   result in a panic. Similarly, 'CryptolScope's captured after the
  611 --   checkpoint are no longer safe to use in the resulting environment.
  612 
  613 --   We also ought to invalidate terms constructed since the checkpoint
  614 --   was taken, like SAWCore does. See #2859.
  615 
  616 --   We could, for example, have 'CryptolScope' track which
  617 --   identifiers it references, and check that they are in a valid
  618 --   range with respect to the corresponding global environment
  619 --   before combining them into a 'CryptolEnv'.
  620 restoreCryptolEnv :: CryptolEnv -> CryptolEnv -> CryptolEnv
  621 restoreCryptolEnv chkEnv newEnv =
  622     let newMEnv = eModuleEnv newEnv
  623         chkMEnv = eModuleEnv chkEnv
  624         menv' = chkMEnv { ME.meNameSeeds = ME.meNameSeeds newMEnv
  625                         , ME.meSupply = ME.meSupply newMEnv 
  626                         }
  627     in mapGlobal (\genv -> genv { geModuleEnv = menv' }) chkEnv
