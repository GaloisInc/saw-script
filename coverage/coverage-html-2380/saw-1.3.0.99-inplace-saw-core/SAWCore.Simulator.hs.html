   81 -- Simulator configuration

   83 data SimulatorConfig l =
   84   SimulatorConfig
   85   { simPrimitive :: PrimName (TValue l) -> MValue l
   86   -- ^ Interpretation of 'Primitive' terms.
   87   , simExtCns :: TermF Term -> ExtCns (TValue l) -> MValue l
   88   -- ^ Interpretation of 'ExtCns' terms.
   89   , simConstant :: TermF Term -> ExtCns (TValue l) -> Maybe (MValue l)
   90   -- ^ Interpretation of 'Constant' terms. 'Nothing' indicates that
   91   -- the body of the constant should be evaluated. 'Just' indicates
   92   -- that the constant's definition should be overridden.
   93   , simCtorApp :: PrimName (TValue l) -> Maybe (MValue l)
   94   -- ^ Interpretation of constructor terms. 'Nothing' indicates that
   95   -- the constructor is treated as normal. 'Just' replaces the
   96   -- constructor with a custom implementation.
   97   , simNeutral :: Env l -> NeutralTerm -> MValue l
   98   -- ^ Handler that fires if the simulator encounters a term that it
   99   --   cannot otherwise evaluate because it is blocked. For some simulators,
  100   --   this is just an error condition; for others, sensible action can
  101   --   be taken.
  102   , simModMap :: ModuleMap
  103   }

  123   evalTermF :: Show (Extra l) =>

  130 -- | Generic evaluator for TermFs.
  131 evalTermF :: forall l. (VMonadLazy l, Show (Extra l)) =>
  132              SimulatorConfig l          -- ^ Evaluator for global constants
  133           -> (Term -> OpenValue l)      -- ^ Evaluator for subterms under binders
  134           -> (Term -> MValue l)         -- ^ Evaluator for subterms in the same bound variable context
  135           -> TermF Term -> OpenValue l
  136 evalTermF cfg lam recEval tf env =
  137   case tf of
  138     App t1 t2               -> recEval t1 >>= \case
  139                                  VFun _ f ->
  140                                    do x <- recEvalDelay t2
  141                                       f x
  142                                  _ -> simNeutral cfg env (NeutralApp (NeutralBox t1) t2)
  143     Lambda nm tp t          -> do v <- evalType tp
  144                                   return $ VFun nm (\x -> lam t ((x,v) : env))
  145     Pi nm t1 t2             -> do v <- evalType t1
  146                                   body <-
  147                                     if inBitSet 0 (looseVars t2) then
  148                                       pure (VDependentPi (\x -> toTValue <$> lam t2 ((x,v) : env)))
  149                                     else
  150                                       do -- put dummy values in the environment; the term should never reference them
  151                                          let val = ready VUnit
  152                                          let tp  = VUnitType
  153                                          VNondependentPi . toTValue <$> lam t2 ((val,tp):env)
  154                                   return $ TValue $ VPiType nm v body
  155 
  156     LocalVar i              -> force (fst (env !! i))
  157     Constant ec t           -> do ec' <- traverse evalType ec
  158                                   fromMaybe
  159                                     (simNeutral cfg env (NeutralConstant ec))
  160                                     (simConstant cfg tf ec' <|> (recEval <$> t))
  161     FTermF ftf              ->
  162       case ftf of
  163         Primitive pn ->
  164           do pn' <- traverse evalType pn
  165              case simConstant cfg tf (primNameToExtCns pn') of
  166                Just m  -> m
  167                Nothing -> simPrimitive cfg pn'
  168 
  169         UnitValue           -> return VUnit
  170 
  171         UnitType            -> return $ TValue VUnitType
  172 
  173         PairValue x y       -> do tx <- recEvalDelay x
  174                                   ty <- recEvalDelay y
  175                                   return $ VPair tx ty
  176 
  177         PairType x y        -> do vx <- evalType x
  178                                   vy <- evalType y
  179                                   return $ TValue $ VPairType vx vy
  180 
  181         PairLeft x          -> recEval x >>= \case
  182                                  VPair l _r -> force l
  183                                  _ -> simNeutral cfg env (NeutralPairLeft (NeutralBox x))
  184 
  185         PairRight x         -> recEval x >>= \case
  186                                  VPair _l r -> force r
  187                                  _ -> simNeutral cfg env (NeutralPairRight (NeutralBox x))
  188 
  189         CtorApp c ps ts     -> do c'  <- traverse evalType c
  190                                   ps' <- mapM recEvalDelay ps
  191                                   ts' <- mapM recEvalDelay ts
  192                                   case simCtorApp cfg c' of
  193                                     Just mv ->
  194                                       do v <- mv
  195                                          foldM apply v (ps' ++ ts')
  196                                     Nothing ->
  197                                       pure $ VCtorApp c' ps' ts'
  198 
  199         DataTypeApp d ps ts -> do d' <- traverse evalType d
  200                                   ps' <- mapM recEval ps
  201                                   ts' <- mapM recEval ts
  202                                   pure (TValue (VDataType d' ps' ts'))
  203 
  204         RecursorType d ps m mtp ->
  205           TValue <$> (VRecursorType <$>
  206             traverse evalType d <*>
  207             mapM recEval ps <*>
  208             recEval m <*>
  209             (evalType mtp))
  210 
  211         Recursor r ->
  212           do let f (e,ety) = do v  <- recEvalDelay e
  213                                 ty <- evalType ety
  214                                 pure (v,ty)
  215              d   <- traverse evalType (recursorDataType r)
  216              ps  <- traverse recEval (recursorParams r)
  217              m   <- recEval (recursorMotive r)
  218              mty <- evalType (recursorMotiveTy r)
  219              es  <- traverse f (recursorElims r)
  220              pure (VRecursor d ps m mty es)
  221 
  222         RecursorApp rectm ixs arg ->
  223           do r <- recEval rectm
  224              case r of
  225                VRecursor d ps motive motiveTy ps_fs ->
  226                  do argv <- recEval arg
  227                     case evalConstructor argv of
  228                       Just (ctor, args)
  229                         | Just (elim,elimTy) <- Map.lookup (ctorVarIndex ctor) ps_fs
  230                         -> do let rTy = VRecursorType d ps motive motiveTy
  231                               ctorTy <- toTValue <$> lam (ctorType ctor) []
  232                               allArgs <- processRecArgs ps args ctorTy [(elim,elimTy),(ready r,rTy)]
  233                               lam (ctorIotaTemplate ctor) allArgs
  234 
  235                         | otherwise -> panic "evalTermF / RecursorApp" ["could not find info for constructor: " <> Text.pack (show ctor)]
  236                       Nothing -> simNeutral cfg env (NeutralRecursorArg rectm ixs (NeutralBox arg))
  237                _ -> simNeutral cfg env (NeutralRecursor (NeutralBox rectm) ixs arg)
  238 
  239         RecordType elem_tps ->
  240           TValue . VRecordType <$> traverse (traverse evalType) elem_tps
  241 
  242         RecordValue elems   ->
  243           VRecordValue <$> mapM (\(fld,t) -> (fld,) <$> recEvalDelay t) elems
  244 
  245         RecordProj t fld    -> recEval t >>= \case
  246                                  v@VRecordValue{} -> valRecordProj v fld
  247                                  _ -> simNeutral cfg env (NeutralRecordProj (NeutralBox t) fld)
  248 
  249         Sort s _h           -> return $ TValue (VSort s)
  250 
  251         NatLit n            -> return $ VNat n
  252 
  253         ArrayValue _ tv     -> liftM VVector $ mapM recEvalDelay tv
  254 
  255         StringLit s         -> return $ VString s
  256 
  257         ExtCns ec           -> do ec' <- traverse evalType ec
  258                                   simExtCns cfg tf ec'
  259   where
  260     evalType :: Term -> EvalM l (TValue l)
  261     evalType t = toTValue <$> recEval t
  262 
  263     toTValue :: HasCallStack => Value l -> TValue l
  264     toTValue (TValue x) = x
  265     toTValue t = panic "evalTermF / toTValue" ["Not a type value: " <> Text.pack (show t)]
  266 
  267     evalConstructor :: Value l -> Maybe (Ctor, [Thunk l])
  268     evalConstructor (VCtorApp c _ps args) =
  269        do ctor <- findCtorInMap (simModMap cfg) (primName c)
  270           Just (ctor, args)
  271     evalConstructor (VNat 0) =
  272        do ctor <- findCtorInMap (simModMap cfg) preludeZeroIdent
  273           Just (ctor, [])
  274     evalConstructor (VNat n) =
  275        do ctor <- findCtorInMap (simModMap cfg) preludeSuccIdent
  276           Just (ctor, [ ready (VNat (pred n)) ])
  277     evalConstructor _ =
  278        Nothing
  279 
  280     recEvalDelay :: Term -> EvalM l (Thunk l)
  281     recEvalDelay = delay . recEval

  284 processRecArgs ::
  285   (VMonadLazy l, Show (Extra l)) =>
  286   [Value l] ->
  287   [Thunk l] ->
  288   TValue l ->
  289   Env l ->
  290   EvalM l (Env l)
  291 processRecArgs (p:ps) args (VPiType _ _ body) env =
  292   do tp' <- applyPiBody body (ready p)
  293      processRecArgs ps args tp' env
  294 processRecArgs [] (x:xs) (VPiType _ tp body) env =
  295   do tp' <- applyPiBody body x
  296      processRecArgs [] xs tp' ((x,tp):env)
  297 processRecArgs [] [] _ env = pure env
  298 processRecArgs _ _ ty _ =
  299   panic "processRegArgs" [
  300      "Expected Pi type!",
  301      "Found: " <> Text.pack (show ty) 
  302   ]

  352 -- | A variant of 'evalGlobal' that lets the uninterpreted function
  353 -- symbol and external-constant callbacks have access to the 'TermF'.
  354 evalGlobal' ::
  355   forall l. (VMonadLazy l, Show (Extra l)) =>
  356   ModuleMap ->
  357   -- | Implementations of 'Primitive' terms, plus overrides for 'Constant' and 'CtorApp' terms
  358   Map Ident (Prims.Prim l) ->
  359   -- | Implementations of ExtCns terms
  360   (TermF Term -> ExtCns (TValue l) -> MValue l) ->
  361   -- | Overrides for Constant terms (e.g. uninterpreted functions)
  362   (TermF Term -> ExtCns (TValue l) -> Maybe (MValue l)) ->
  363   -- | Handler for neutral terms
  364   (Env l -> NeutralTerm -> MValue l) ->
  365   -- | Handler for stuck primitives
  366   (PrimName (TValue l) -> Text -> Env l -> TValue l -> MValue l) ->
  367   EvalM l (SimulatorConfig l)
  368 evalGlobal' modmap prims extcns constant neutral primHandler =
  369   do checkPrimitives modmap prims
  370      return (SimulatorConfig primitive extcns constant' ctors neutral modmap)
  371   where
  372     constant' :: TermF Term -> ExtCns (TValue l) -> Maybe (MValue l)
  373     constant' tf ec =
  374       case constant tf ec of
  375         Just v -> Just v
  376         Nothing ->
  377           case ecName ec of
  378             ModuleIdentifier ident ->
  379               let pn = PrimName (ecVarIndex ec) ident (ecType ec)
  380                in evalPrim (primHandler pn) pn <$> Map.lookup ident prims
  381             _ -> Nothing
  382 
  383     ctors :: PrimName (TValue l) -> Maybe (MValue l)
  384     ctors pn = evalPrim (primHandler pn) pn <$> Map.lookup (primName pn) prims
  385 
  386     primitive :: PrimName (TValue l) -> MValue l
  387     primitive pn =
  388       case Map.lookup (primName pn) prims of
  389         Just v  -> evalPrim (primHandler pn) pn v
  390         Nothing -> panic "evalGlobal'" ["Unimplemented global: " <> identText (primName pn)]

  392 -- | Check that all the primitives declared in the given module
  393 --   are implemented, and that terms with implementations are not
  394 --   overridden.
  395 checkPrimitives :: forall l. (VMonadLazy l, Show (Extra l))
  396                 => ModuleMap
  397                 -> Map Ident (Prims.Prim l)
  398                 -> EvalM l ()
  399 checkPrimitives modmap prims = do
  400    -- FIXME this is downgraded to a warning temporarily while we work out a
  401    -- solution to issue GaloisInc/saw-script#48
  402    --   when (not $ null unimplementedPrims) (panic $ unimplementedMsg)
  403    -- (if null unimplementedPrims then id else Debug.trace (unimplementedMsg++"\n")) $
  404 --   (if null overridePrims then id else Debug.trace (overrideMsg++"\n")) $
  405      return ()
  406 
  407   where _unimplementedMsg = unwords $
  408             ("WARNING unimplemented primitives:" : (map show unimplementedPrims))
  409         _overrideMsg = unwords $
  410             ("WARNING overridden definitions:" : (map show overridePrims))
  411 
  412         primSet = Set.fromList $ map defIdent $ allModulePrimitives modmap

  416         unimplementedPrims = Set.toList $ Set.difference primSet implementedPrims
  417         overridePrims = Set.toList $ Set.intersection defSet implementedPrims

  420 -- The evaluation strategy for SharedTerms involves two memo tables:
  421 -- The first, @memoClosed@, is precomputed and contains the result of
  422 -- evaluating all _closed_ subterms. The same @memoClosed@ table is
  423 -- used for evaluation under lambdas, since the meaning of a closed
  424 -- term does not depend on the local variable context. The second memo
  425 -- table is @memoLocal@, which additionally includes the result of
  426 -- evaluating _open_ terms in the current variable context. It is
  427 -- reinitialized to @memoClosed@ whenever we descend under a lambda
  428 -- binder.
  429 
  430 {-# SPECIALIZE evalSharedTerm ::
  431   Show (Extra l) => SimulatorConfigIn Id l -> Term -> MValueIn Id l #-}
  432 {-# SPECIALIZE evalSharedTerm ::
  433   Show (Extra l) => SimulatorConfigIn IO l -> Term -> MValueIn IO l #-}
  434 
  435 -- | Evaluator for shared terms.
  436 evalSharedTerm :: (VMonadLazy l, MonadFix (EvalM l), Show (Extra l)) =>
  437                   SimulatorConfig l -> Term -> MValue l
  438 evalSharedTerm cfg t = do
  439   memoClosed <- mkMemoClosed cfg t
  440   evalOpen cfg memoClosed t []
  441 
  442 {-# SPECIALIZE mkMemoClosed ::
  443   Show (Extra l) =>
  444   SimulatorConfigIn Id l -> Term -> Id (IntMap (ThunkIn Id l)) #-}
  445 {-# SPECIALIZE mkMemoClosed ::
  446   Show (Extra l) =>
  447   SimulatorConfigIn IO l -> Term -> IO (IntMap (ThunkIn IO l)) #-}
  448 
  449 -- | Precomputing the memo table for closed subterms.
  450 mkMemoClosed :: forall l. (VMonadLazy l, MonadFix (EvalM l), Show (Extra l)) =>
  451                 SimulatorConfig l -> Term -> EvalM l (IntMap (Thunk l))
  452 mkMemoClosed cfg t =
  453   mfix $ \memoClosed -> mapM (delay . evalClosedTermF cfg memoClosed) subterms
  454   where
  455     -- | Map of all closed subterms of t.
  456     subterms :: IntMap (TermF Term)
  457     subterms = fmap fst $ IMap.filter ((== emptyBitSet) . snd) $ State.execState (go t) IMap.empty
  458 
  459     go :: Term -> State.State (IntMap (TermF Term, BitSet)) BitSet
  460     go (Unshared tf) = freesTermF <$> traverse go tf
  461     go (STApp{ stAppIndex = i, stAppTermF = tf }) = do
  462       memo <- State.get
  463       case IMap.lookup i memo of
  464         Just (_, b) -> return b
  465         Nothing -> do
  466           b <- freesTermF <$> traverse go tf
  467           State.modify (IMap.insert i (tf, b))
  468           return b

  470 {-# SPECIALIZE evalClosedTermF ::
  471   Show (Extra l) =>

  473   IntMap (ThunkIn Id l) ->

  475   MValueIn Id l #-}
  476 {-# SPECIALIZE evalClosedTermF ::

  483 -- | Evaluator for closed terms, used to populate @memoClosed@.
  484 evalClosedTermF :: (VMonadLazy l, Show (Extra l)) =>
  485                    SimulatorConfig l
  486                 -> IntMap (Thunk l)
  487                 -> TermF Term -> MValue l
  488 evalClosedTermF cfg memoClosed tf = evalTermF cfg lam recEval tf []
  489   where
  490     lam = evalOpen cfg memoClosed
  491     recEval (Unshared tf') = evalTermF cfg lam recEval tf' []
  492     recEval (STApp{ stAppIndex = i }) =
  493       case IMap.lookup i memoClosed of
  494         Just x -> force x
  495         Nothing -> panic "evalClosedTermF" ["internal error"]

  497 {-# SPECIALIZE mkMemoLocal ::
  498   Show (Extra l) =>
  499   SimulatorConfigIn Id l ->
  500   IntMap (ThunkIn Id l) ->
  501   Term ->
  502   EnvIn Id l ->
  503   Id (IntMap (ThunkIn Id l)) #-}
  504 {-# SPECIALIZE mkMemoLocal ::
  505   Show (Extra l) =>
  506   SimulatorConfigIn IO l ->
  507   IntMap (ThunkIn IO l) ->
  508   Term ->
  509   EnvIn IO l ->
  510   IO (IntMap (ThunkIn IO l)) #-}

  512 -- | Precomputing the memo table for open subterms in the current context.
  513 mkMemoLocal :: forall l. (VMonadLazy l, Show (Extra l)) =>
  514                SimulatorConfig l -> IntMap (Thunk l) ->
  515                Term -> Env l -> EvalM l (IntMap (Thunk l))
  516 mkMemoLocal cfg memoClosed t env = go mempty t
  517   where
  518     go :: IntMap (Thunk l) -> Term -> EvalM l (IntMap (Thunk l))
  519     go memo (Unshared tf) = goTermF memo tf
  520     go memo (t'@STApp{ stAppIndex = i, stAppFreeVars = _, stAppTermF = tf })
  521       | termIsClosed t' = pure memo
  522       | otherwise =
  523         case IMap.lookup i memo of
  524           Just _ -> pure memo
  525           Nothing ->
  526             do memo' <- goTermF memo tf
  527                thunk <- delay (evalLocalTermF cfg memoClosed memo' tf env)
  528                pure (IMap.insert i thunk memo')
  529     goTermF :: IntMap (Thunk l) -> TermF Term -> EvalM l (IntMap (Thunk l))
  530     goTermF memo tf =
  531       case tf of
  532         FTermF ftf      -> foldlM go memo ftf
  533         App t1 t2       -> do memo' <- goTermF memo (unwrapTermF t1)
  534                               go memo' t2
  535         Lambda _ t1 _   -> go memo t1
  536         Pi _ t1 _       -> go memo t1
  537         LocalVar _      -> return memo
  538         Constant _ Nothing -> return memo -- TODO? is this right?
  539         Constant _ (Just t1) -> go memo t1

  584 -- | Evaluator for open terms; parameterized by a precomputed table @memoClosed@.
  585 evalOpen :: forall l. (VMonadLazy l, Show (Extra l)) =>
  586             SimulatorConfig l
  587          -> IntMap (Thunk l)
  588          -> Term -> OpenValue l
  589 evalOpen cfg memoClosed t env = do
  590   memoLocal <- mkMemoLocal cfg memoClosed t env
  591   let eval :: Term -> MValue l
  592       eval (Unshared tf) = evalF tf
  593       eval (t'@STApp{ stAppIndex = i, stAppFreeVars = _, stAppTermF = tf }) =
  594         case IMap.lookup i memo of
  595           Just x -> force x
  596           Nothing -> evalF tf
  597         where memo = if termIsClosed t' then memoClosed else memoLocal
  598       evalF :: TermF Term -> MValue l
  599       evalF tf = evalTermF cfg (evalOpen cfg memoClosed) eval tf env
  600   eval t

  617 evalPrim :: forall l. (VMonadLazy l, Show (Extra l)) =>
  618   (Text -> Env l -> TValue l -> MValue l) ->
  619   PrimName (TValue l) ->
  620   Prims.Prim l ->
  621   MValue l
  622 evalPrim fallback pn = loop [] (primType pn)
  623   where
  624     loop :: Env l -> TValue l -> Prims.Prim l -> MValue l
  625     loop env (VPiType nm t body) (Prims.PrimFun f) =
  626       pure $ VFun nm $ \x ->
  627         do tp' <- applyPiBody body x
  628            loop ((x,t):env) tp' (f x)
  629 
  630     loop env (VPiType nm t body) (Prims.PrimStrict f) =
  631       pure $ VFun nm $ \x ->
  632         do tp' <- applyPiBody body x
  633            x'  <- force x
  634            loop ((ready x',t):env) tp' (f x')
  635 
  636     loop env (VPiType nm t body) (Prims.PrimFilterFun msg r f) =
  637       pure $ VFun nm $ \x ->
  638         do tp' <- applyPiBody body x
  639            x'  <- force x
  640            runMaybeT (r x') >>= \case
  641              Just v -> loop ((ready x',t):env) tp' (f v)
  642              _ -> fallback msg ((ready x',t):env) tp'
  643 
  644     loop env tp (Prims.PrimExcept m) =
  645       runExceptT m >>= \case
  646         Right v  -> pure v
  647         Left msg -> fallback msg env tp
  648 
  649     loop _env _tp (Prims.Prim m) = m
  650     loop _env _tp (Prims.PrimValue v) = pure v
