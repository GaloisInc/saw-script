   23 -- | Resolve a SAWCore 'Term' into a What4 'W4.Pred'.
   24 resolveBoolTerm :: Sym -> Term -> IO (W4.Pred Sym)
   25 resolveBoolTerm sym tm =
   26   do st <- sawCoreState sym
   27      let sc = saw_ctx st
   28      mx <- case getAllExts tm of
   29              -- concretely evaluate if it is a closed term
   30              [] ->
   31                do modmap <- scGetModuleMap sc
   32                   let v = Concrete.evalSharedTerm modmap mempty mempty tm
   33                   pure (Just (Concrete.toBool v))
   34              _ -> return Nothing
   35      case mx of
   36        Just x  -> return (W4.backendPred sym x)
   37        Nothing -> bindSAWTerm sym st W4.BaseBoolRepr tm
   38 
   39 -- | Ensure that the term has Cryptol type @Bit@.
   40 checkBooleanType :: SharedContext -> Term -> IO ()
   41 checkBooleanType sc tm = do
   42   tt <- mkTypedTerm sc tm
   43   case ttIsMono (ttType tt) of
   44     Just ty | tIsBit ty -> pure ()
   45             | otherwise -> fail $ unlines
   46                  [ "Expected type: Bit"
   47                  , "Actual type:   " ++ show (PP.pp ty)
   48                  ]
   49     Nothing -> fail "Expected monomorphic Cryptol type, got polymorphic or unrecognized type"
