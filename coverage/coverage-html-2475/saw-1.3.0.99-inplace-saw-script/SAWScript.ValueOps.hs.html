   78 isVUnit :: Value -> Bool
   79 isVUnit (VTuple []) = True
   80 isVUnit _ = False
   81 
   82 indexValue :: Value -> Value -> Value
   83 indexValue (VArray vs) (VInteger x)
   84     | i < length vs = vs !! i
   85     | otherwise = error "array index out of bounds"
   86     where i = fromInteger x
   87 indexValue _ _ = error "indexValue"
   88 
   89 lookupValue :: Value -> Text -> Value
   90 lookupValue (VRecord vm) name =
   91     case M.lookup name vm of
   92       Nothing -> error $ "no such record field: " ++ Text.unpack name
   93       Just x -> x
   94 lookupValue _ _ = error "lookupValue"
   95 
   96 tupleLookupValue :: Value -> Integer -> Value
   97 tupleLookupValue (VTuple vs) i
   98   | 0 <= i && fromIntegral i < length vs = vs !! fromIntegral i
   99   | otherwise = error $ "no such tuple index: " ++ show i
  100 tupleLookupValue _ _ = error "tupleLookupValue"

  111 -- | A version of 'Control.Exception.bracket' specialized to 'TopLevel'. We
  112 -- can't use 'Control.Monad.Catch.bracket' because it requires 'TopLevel' to
  113 -- implement 'Control.Monad.Catch.MonadMask', which it can't do.
  114 bracketTopLevel :: TopLevel a -> (a -> TopLevel b) -> (a -> TopLevel c) -> TopLevel c
  115 bracketTopLevel acquire release action =
  116   do  resource <- acquire
  117       try (action resource) >>= \case
  118         Left (bad :: X.SomeException) -> release resource >> throwM bad
  119         Right good -> release resource >> pure good

  121 combineRW :: TopLevelCheckpoint -> TopLevelRW -> IO TopLevelRW
  122 combineRW (TopLevelCheckpoint chk scc) rw =
  123   do cenv' <- CEnv.combineCryptolEnv (rwCryptol chk) (rwCryptol rw)
  124      sc' <- restoreSharedContext scc (rwSharedContext rw)
  125      return chk{ rwCryptol = cenv'
  126                , rwSharedContext = sc'
  127                }

  129 -- | Represents the mutable state of the TopLevel monad
  130 --   that can later be restored.
  131 data TopLevelCheckpoint =
  132   TopLevelCheckpoint
  133     TopLevelRW
  134     SharedContextCheckpoint
  135 
  136 makeCheckpoint :: TopLevelRW -> IO TopLevelCheckpoint
  137 makeCheckpoint rw =
  138   do scc <- checkpointSharedContext (rwSharedContext rw)
  139      return (TopLevelCheckpoint rw scc)
  140 
  141 restoreCheckpoint :: TopLevelCheckpoint -> TopLevel ()
  142 restoreCheckpoint chk =
  143   do rw <- getTopLevelRW
  144      rw' <- io (combineRW chk rw)
  145      putTopLevelRW rw'
  146 
  147 -- | Capture the current state of the TopLevel monad
  148 --   and return an action that, if invoked, resets
  149 --   the state back to that point.
  150 checkpoint :: TopLevel (() -> TopLevel ())
  151 checkpoint = TopLevel_ $
  152   do chk <- liftIO . makeCheckpoint =<< get
  153      return $ \_ ->
  154        do printOutLnTop Info "Restoring state from checkpoint"
  155           restoreCheckpoint chk
  156 
  157 -- | Capture the current proof state and return an
  158 --   action that, if invoked, resets the state back to that point.
  159 proof_checkpoint :: ProofScript (() -> ProofScript ())
  160 proof_checkpoint =
  161   do ps <- get
  162      return $ \_ ->
  163        do scriptTopLevel (printOutLnTop Info "Restoring proof state from checkpoint")
  164           put ps

  166 withLocalEnv :: LocalEnv -> TopLevel a -> TopLevel a
  167 withLocalEnv env m = do
  168   prevEnv <- gets rwLocalEnv
  169   modify (\rw -> rw { rwLocalEnv = env })
  170   result <- m
  171   modify (\rw -> rw { rwLocalEnv = prevEnv })
  172   return result
  173 
  174 withLocalEnvProof :: LocalEnv -> ProofScript a -> ProofScript a
  175 withLocalEnvProof env (ProofScript m) = do
  176   ProofScript (underExceptT (underStateT (withLocalEnv env)) m)
  177 
  178 withOptions :: (Options -> Options) -> TopLevel a -> TopLevel a
  179 withOptions f (TopLevel_ m) =
  180   TopLevel_ (local (\x -> x {roOptions = f (roOptions x)}) m)
