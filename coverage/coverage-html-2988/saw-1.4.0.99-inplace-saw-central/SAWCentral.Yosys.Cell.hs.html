   34 -- | A SAWCore bitvector term along with its width and whether it should be interpreted as signed.
   35 data CellTerm = CellTerm
   36   { cellTermTerm :: SC.Term
   37   , cellTermWidth :: Natural
   38   , cellTermSigned :: Bool
   39   }
   40 
   41 cellTermNat :: SC.SharedContext -> CellTerm -> IO SC.Term
   42 cellTermNat sc (CellTerm { cellTermTerm = t, cellTermWidth = w }) = SC.scBvToNat sc w t
   43 
   44 -- | Apply the appropriate (possibly signed) extension or truncation to make the given bitvector
   45 -- match the given width.
   46 extTrunc :: SC.SharedContext -> Natural -> CellTerm -> IO CellTerm
   47 extTrunc sc width (CellTerm { cellTermTerm = t, cellTermWidth = bvw, cellTermSigned = signed}) =
   48   do wterm <- SC.scNat sc width
   49      bvwterm <- SC.scNat sc bvw
   50      res <- if
   51        | bvw > width ->
   52            do diffterm <- SC.scNat sc $ bvw - width
   53               SC.scBvTrunc sc diffterm wterm t
   54        | width > bvw && signed ->
   55            do bvwpredterm <- SC.scNat sc $ bvw - 1
   56               diffterm <- SC.scNat sc $ width - bvw
   57               SC.scBvSExt sc diffterm bvwpredterm t
   58        | width > bvw && not signed ->
   59            do diffterm <- SC.scNat sc $ width - bvw
   60               SC.scBvUExt sc diffterm bvwterm t
   61        | otherwise -> pure t
   62      pure $ CellTerm res width signed
   63 
   64 -- | Given two bitvectors, extend the narrower bitvector to match the wider.
   65 extMax :: SC.SharedContext -> CellTerm -> CellTerm -> IO (CellTerm, CellTerm)
   66 extMax sc c1 c2 =
   67   do let
   68        w1 = cellTermWidth c1
   69        w2 = cellTermWidth c2
   70        w = max w1 w2
   71      res1 <- extTrunc sc w c1
   72      res2 <- extTrunc sc w c2
   73      pure (res1, res2)
   74 
   75 liftUnary ::
   76   SC.SharedContext ->
   77   (SC.Term -> SC.Term -> IO SC.Term) -> -- (w : Nat) -> [w] -> [w]
   78   CellTerm -> IO CellTerm
   79 liftUnary sc f c@(CellTerm { cellTermTerm = t }) =
   80   do wt <- SC.scNat sc $ cellTermWidth c
   81      res <- f wt t
   82      pure $ c { cellTermTerm = res }
   83 
   84 liftBinary ::
   85   SC.SharedContext ->
   86   (SC.Term -> SC.Term -> SC.Term -> IO SC.Term) -> -- (w : Nat) -> [w] -> [w] -> [w]
   87   CellTerm -> CellTerm -> IO CellTerm
   88 liftBinary sc f c1@(CellTerm { cellTermTerm = t1 }) (CellTerm { cellTermTerm = t2 }) =
   89   do wt <- SC.scNat sc $ cellTermWidth c1
   90      res <- f wt t1 t2
   91      pure $ c1 { cellTermTerm = res }
   92 
   93 liftShift ::
   94   SC.SharedContext ->
   95   (SC.Term -> SC.Term -> SC.Term -> IO SC.Term) -> -- (w : Nat) -> [w] -> Nat -> [w]
   96   CellTerm -> CellTerm -> IO CellTerm
   97 liftShift sc f a b =
   98   do wt <- SC.scNat sc $ cellTermWidth a
   99      nb <- cellTermNat sc b
  100      res <- f wt (cellTermTerm a) nb
  101      pure $ a { cellTermTerm = res }
  102 
  103 liftBinaryCmp ::
  104   SC.SharedContext ->
  105   (SC.Term -> SC.Term -> SC.Term -> IO SC.Term) -> -- (w : Nat) -> [w] -> [w] -> Bool
  106   CellTerm -> CellTerm -> IO SC.Term
  107 liftBinaryCmp sc f c1@(CellTerm { cellTermTerm = t1 }) (CellTerm { cellTermTerm = t2 }) =
  108   do wt <- SC.scNat sc $ cellTermWidth c1
  109      f wt t1 t2

  111 -- | Given a primitive Yosys cell and a map of terms for its
  112 -- arguments, construct a record term representing the output. If the
  113 -- provided cell is not a primitive, return Nothing.
  114 primCellToTerm ::
  115   forall b.
  116   SC.SharedContext ->
  117   Cell [b] {- ^ Cell type -} ->
  118   Map Text SC.Term {- ^ Mapping of input names to input terms -} ->
  119   IO (Maybe SC.Term)
  120 primCellToTerm sc c args =
  121   do mm <- primCellToMap sc c args
  122      traverse (cryptolRecord sc) mm
  123 
  124 primCellToMap ::
  125   forall b.
  126   SC.SharedContext ->
  127   Cell [b] {- ^ Cell type -} ->
  128   Map Text SC.Term {- ^ Mapping of input names to input terms -} ->
  129   IO (Maybe (Map Text SC.Term))
  130 primCellToMap sc c args =
  131   case c ^. cellType of
  132     CellTypeNot -> bvUnaryOp . liftUnary sc $ SC.scBvNot sc
  133     CellTypePos ->
  134       do res <- input "A"
  135          output res
  136     CellTypeNeg -> bvUnaryOp . liftUnary sc $ SC.scBvNeg sc
  137     CellTypeAnd -> bvBinaryOp . liftBinary sc $ SC.scBvAnd sc
  138     CellTypeOr -> bvBinaryOp . liftBinary sc $ SC.scBvOr sc
  139     CellTypeXor -> bvBinaryOp . liftBinary sc $ SC.scBvXor sc
  140     CellTypeXnor ->
  141       bvBinaryOp $ liftBinary sc $ \w x y ->
  142       do r <- SC.scBvXor sc w x y
  143          SC.scBvNot sc w r
  144     CellTypeReduceAnd ->
  145       do r <- SC.scGlobalDef sc $ SC.mkIdent SC.preludeName "and"
  146          bvReduce True r
  147     CellTypeReduceOr ->
  148       do r <- SC.scGlobalDef sc $ SC.mkIdent SC.preludeName "or"
  149          bvReduce False r
  150     CellTypeReduceXor ->
  151       do r <- SC.scGlobalDef sc $ SC.mkIdent SC.preludeName "xor"
  152          bvReduce False r
  153     CellTypeReduceXnor ->
  154       do boolTy <- SC.scBoolType sc
  155          x <- SC.scFreshVariable sc "x" boolTy
  156          y <- SC.scFreshVariable sc "y" boolTy
  157          r <- SC.scXor sc x y
  158          res <- SC.scNot sc r
  159          t <- SC.scAbstractTerms sc [x, y] res
  160          bvReduce True t
  161     CellTypeReduceBool ->
  162       do r <- SC.scGlobalDef sc $ SC.mkIdent SC.preludeName "or"
  163          bvReduce False r
  164     CellTypeShl -> bvShiftOp $ liftShift sc (SC.scBvShl sc)
  165     CellTypeShr -> bvShiftOp $ liftShift sc (SC.scBvShr sc)
  166     CellTypeSshl -> bvShiftOp $ liftShift sc (SC.scBvShl sc)
  167     CellTypeSshr -> bvShiftOp $ liftShift sc (SC.scBvSShr sc)
  168     -- "$shift" -> _
  169     CellTypeShiftx ->
  170       do let w = max (connWidthNat "A") (connWidthNat "B")
  171          wt <- SC.scNat sc w
  172          CellTerm ta _ _ <- extTrunc sc w =<< input "A"
  173          CellTerm tb _ _ <- extTrunc sc w =<< input "B"
  174          zero <- SC.scBvConst sc w 0
  175          tbn <- SC.scBvToNat sc w tb
  176          tbneg <- SC.scBvNeg sc wt tb
  177          tbnegn <- SC.scBvToNat sc w tbneg
  178          cond <- SC.scBvSGe sc wt tb zero
  179          tcase <- SC.scBvShr sc wt ta tbn
  180          ecase <- SC.scBvShl sc wt ta tbnegn
  181          ty <- SC.scBitvector sc $ connWidthNat "A"
  182          res <-
  183            if connSigned "B"
  184            then SC.scIte sc ty cond tcase ecase
  185            else pure tcase
  186          output (CellTerm res (connWidthNat "A") (connSigned "A"))
  187     CellTypeLt -> bvBinaryCmp . liftBinaryCmp sc $ SC.scBvULt sc
  188     CellTypeLe -> bvBinaryCmp . liftBinaryCmp sc $ SC.scBvULe sc
  189     CellTypeGt -> bvBinaryCmp . liftBinaryCmp sc $ SC.scBvUGt sc
  190     CellTypeGe -> bvBinaryCmp . liftBinaryCmp sc $ SC.scBvUGe sc
  191     CellTypeEq -> bvBinaryCmp . liftBinaryCmp sc $ SC.scBvEq sc
  192     CellTypeNe ->
  193       bvBinaryCmp $ liftBinaryCmp sc $ \w x y ->
  194       do r <- SC.scBvEq sc w x y
  195          SC.scNot sc r
  196     CellTypeEqx -> bvBinaryCmp . liftBinaryCmp sc $ SC.scBvEq sc
  197     CellTypeNex ->
  198       bvBinaryCmp $ liftBinaryCmp sc $ \w x y ->
  199       do r <- SC.scBvEq sc w x y
  200          SC.scNot sc r
  201     CellTypeAdd -> bvBinaryOp . liftBinary sc $ SC.scBvAdd sc
  202     CellTypeSub -> bvBinaryOp . liftBinary sc $ SC.scBvSub sc
  203     CellTypeMul -> bvBinaryOp . liftBinary sc $ SC.scBvMul sc
  204     CellTypeDiv -> bvBinaryOp . liftBinary sc $ SC.scBvUDiv sc
  205     CellTypeMod -> bvBinaryOp . liftBinary sc $ SC.scBvURem sc
  206     -- "$modfloor" -> _
  207     CellTypeLogicNot ->
  208       do w <- connWidth "A"
  209          ta <- cellTermTerm <$> input "A"
  210          anz <- SC.scBvNonzero sc w ta
  211          res <- SC.scNot sc anz
  212          outputBit res
  213     CellTypeLogicAnd ->
  214       bvBinaryCmp $ liftBinaryCmp sc $ \w x y ->
  215       do xnz <- SC.scBvNonzero sc w x
  216          ynz <- SC.scBvNonzero sc w y
  217          SC.scAnd sc xnz ynz
  218     CellTypeLogicOr ->
  219       bvBinaryCmp $ liftBinaryCmp sc $ \w x y ->
  220       do xnz <- SC.scBvNonzero sc w x
  221          ynz <- SC.scBvNonzero sc w y
  222          SC.scOr sc xnz ynz
  223     CellTypeMux ->
  224       do ta <- cellTermTerm <$> input "A"
  225          tb <- cellTermTerm <$> input "B"
  226          ts <- cellTermTerm <$> input "S"
  227          swidth <- connWidth "S"
  228          snz <- SC.scBvNonzero sc swidth ts
  229          let width = connWidthNat "Y"
  230          ty <- SC.scBitvector sc width
  231          res <- SC.scIte sc ty snz tb ta
  232          output $ CellTerm res (connWidthNat "A") (connSigned "A")
  233     CellTypePmux ->
  234       do ta <- cellTermTerm <$> input "A"
  235          tb <- cellTermTerm <$> input "B"
  236          ts <- cellTermTerm <$> input "S"
  237          width <- connWidth "A"
  238          widthBv <- SC.scBitvector sc $ connWidthNat "A"
  239          swidth <- connWidth "S"
  240          bool <- SC.scBoolType sc
  241          splitb <- SC.scSplit sc swidth width bool tb
  242          scPmux <- SC.scGlobalDef sc $ SC.mkIdent SC.preludeName "pmux"
  243          res <- SC.scApplyAll sc scPmux [swidth, widthBv, ts, splitb, ta]
  244          output $ CellTerm res (connWidthNat "A") (connSigned "Y")
  245     CellTypeBmux ->
  246       do ia <- input "A"
  247          is <- input "S"
  248          let swidth = cellTermWidth is
  249          let ywidth = connWidthNat "Y"
  250          -- Split input A into chunks
  251          chunks <- SC.scNat sc (2 ^ swidth)
  252          ywidth' <- SC.scNat sc ywidth
  253          bool <- SC.scBoolType sc
  254          splitA <- SC.scSplit sc chunks ywidth' bool (cellTermTerm ia)
  255          -- reverse to put index 0 on the left
  256          outputType <- SC.scBitvector sc ywidth
  257          revA <- SC.scGlobalApply sc "Prelude.reverse" [chunks, outputType, splitA]
  258          -- Select chunk from output
  259          ixWidth <- SC.scNat sc swidth
  260          elt <- SC.scBvAt sc chunks outputType ixWidth revA (cellTermTerm is)
  261          output (CellTerm elt ywidth (connSigned "Y"))
  262     -- "$demux" -> _
  263     -- "$lut" -> _
  264     -- "$slice" -> _
  265     -- "$concat" -> _
  266     CellTypeDff -> pure Nothing
  267     CellTypeFf -> pure Nothing
  268     CellTypeBUF ->
  269       do res <- input "A"
  270          output res
  271     CellTypeCheck -> pure Nothing
  272     CellTypePrint -> pure Nothing
  273     CellTypeScopeinfo -> pure Nothing
  274     CellTypeUnsupportedPrimitive _ -> pure Nothing
  275     CellTypeUserType _ -> pure Nothing
  276   where
  277     nm :: Text
  278     nm = Text.pack $ show $ c ^. cellType
  279 
  280     textBinNat :: Text -> Natural
  281     textBinNat = fromIntegral . Text.foldl' (\a x -> digitToInt x + a * 2) 0
  282     connSigned :: Text -> Bool
  283     connSigned onm =
  284       case Map.lookup (onm <> "_SIGNED") $ c ^. cellParameters of
  285         Just (Aeson.Number n) -> n > 0
  286         Just (Aeson.String t) -> textBinNat t > 0
  287         Just v ->
  288           -- XXX This should not be a panic, as it is possible to trigger this
  289           -- with a malformed input file.
  290           panic "cellToTerm"
  291             [ "Expected SIGNED parameter to be a number or a string,"
  292             , "but encountered " <> Text.pack (show v)
  293             ]
  294         Nothing -> False
  295     connWidthNat :: Text -> Natural
  296     connWidthNat onm =
  297       case Map.lookup onm $ c ^. cellConnections of
  298         -- XXX This should not be a panic, as it is possible to trigger this
  299         -- with a malformed input file.
  300         Nothing -> panic "cellToTerm" ["Missing expected output name for " <> nm <> " cell"]
  301         Just bits -> fromIntegral $ length bits
  302     connWidth :: Text -> IO SC.Term
  303     connWidth onm = SC.scNat sc $ connWidthNat onm
  304 
  305     input :: Text -> IO CellTerm
  306     input inpNm =
  307       case Map.lookup inpNm args of
  308         -- XXX This should not be a panic, as it is possible to trigger this
  309         -- with a malformed input file.
  310         Nothing -> panic "cellToTerm" [nm <> " missing input " <> inpNm]
  311         Just a -> pure $ CellTerm a (connWidthNat inpNm) (connSigned inpNm)
  312 
  313     output :: CellTerm -> IO (Maybe (Map Text SC.Term))
  314     output (CellTerm ct cw _) =
  315       do let res = CellTerm ct cw (connSigned "Y")
  316          CellTerm t _ _ <- extTrunc sc (connWidthNat "Y") res
  317          pure . Just $ Map.fromList
  318            [ ("Y", t)
  319            ]
  320 
  321     outputBit :: SC.Term -> IO (Maybe (Map Text SC.Term))
  322     outputBit res =
  323       do bool <- SC.scBoolType sc
  324          vres <- SC.scSingle sc bool res
  325          pure . Just $ Map.fromList
  326            [ ("Y", vres)
  327            ]
  328 
  329     -- convert input to big endian
  330     bvUnaryOp :: (CellTerm -> IO CellTerm) -> IO (Maybe (Map Text SC.Term))
  331     bvUnaryOp f =
  332       do t <- input "A"
  333          res <- f t
  334          output res
  335     -- extend inputs to output width
  336     bvBinaryOp :: (CellTerm -> CellTerm -> IO CellTerm) -> IO (Maybe (Map Text SC.Term))
  337     bvBinaryOp f =
  338       do let w = connWidthNat "Y"
  339          ta <- extTrunc sc w =<< input "A"
  340          tb <- extTrunc sc w =<< input "B"
  341          res <- f ta tb
  342          output res
  343     -- extend inputs to max input width, output is a single bit
  344     bvBinaryCmp :: (CellTerm -> CellTerm -> IO SC.Term) -> IO (Maybe (Map Text SC.Term))
  345     bvBinaryCmp f =
  346       do ta <- input "A"
  347          tb <- input "B"
  348          res <- uncurry f =<< extMax sc ta tb
  349          outputBit res
  350     bvReduce :: Bool -> SC.Term -> IO (Maybe (Map Text SC.Term))
  351     bvReduce boolIdentity boolFun =
  352       do CellTerm t _ _ <- input "A"
  353          w <- connWidth "A"
  354          boolTy <- SC.scBoolType sc
  355          identity <- SC.scBool sc boolIdentity
  356          scFoldr <- SC.scGlobalDef sc $ SC.mkIdent SC.preludeName "foldr"
  357          bit <- SC.scApplyAll sc scFoldr [boolTy, boolTy, w, boolFun, identity, t]
  358          outputBit bit
  359     -- extend input A to max(A_WIDTH, Y_WIDTH)
  360     bvShiftOp :: (CellTerm -> CellTerm -> IO CellTerm) -> IO (Maybe (Map Text SC.Term))
  361     bvShiftOp f =
  362       do let w = max (connWidthNat "A") (connWidthNat "Y")
  363          a <- extTrunc sc w =<< input "A"
  364          b <- input "B"
  365          res <- f a b
  366          output res
