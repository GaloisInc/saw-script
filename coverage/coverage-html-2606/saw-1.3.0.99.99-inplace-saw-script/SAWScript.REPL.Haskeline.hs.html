   36 -- | Haskeline-specific repl implementation.
   37 --
   38 -- XXX this needs to handle Ctrl-C, which at the moment will just cause
   39 -- haskeline to exit.  See the function 'withInterrupt' for more info on how to
   40 -- handle this.
   41 repl :: Maybe FilePath -> Options -> REPL () -> IO ()
   42 repl mbBatch opts begin =
   43   runREPL (isJust mbBatch) opts (replBody mbBatch begin)
   44 
   45 
   46 replBody :: Maybe FilePath -> REPL () -> REPL ()
   47 replBody mbBatch begin =
   48   do settings <- MTL.liftIO (setHistoryFile replSettings)
   49      enableSubshell (runInputTBehavior style settings body)
   50   where
   51   body = withInterrupt $ do
   52     MTL.lift begin
   53     loop
   54 
   55   style = case mbBatch of
   56     Nothing   -> defaultBehavior
   57     Just path -> useFile path
   58 
   59   enableSubshell m =
   60     REPL $ \refs ->
   61       do let ro' = (eTopLevelRO refs){ roSubshell = subshell (runInputT replSettings (withInterrupt loop)) }
   62          unREPL m refs{ eTopLevelRO = ro' }
   63 
   64   loop = do
   65     prompt <- MTL.lift getPrompt
   66     mb     <- handleInterrupt (return (Just "")) (getInputLines prompt [])
   67     case mb of
   68       Just line
   69         | Just cmd <- parseCommand findCommand line -> do
   70           continue <- MTL.lift $ do
   71             handleInterrupt handleCtrlC (runCommand cmd)
   72             shouldContinue
   73           when continue loop
   74 
   75         | otherwise -> loop
   76 
   77       Nothing -> return ()
   78 
   79   getInputLines prompt ls =
   80     do mb <- fmap (filter (/= '\r')) <$> getInputLine prompt
   81        let newPrompt = map (\_ -> ' ') prompt
   82        case mb of
   83           Nothing -> return Nothing
   84           Just l | not (null l) && last l == '\\' ->
   85                                       getInputLines newPrompt (init l : ls)
   86                  | otherwise -> return $ Just $ unlines $ reverse $ l : ls

   90 -- | Try to set the history file.
   91 setHistoryFile :: Settings REPL -> IO (Settings REPL)
   92 setHistoryFile ss =
   93   do dir <- getAppUserDataDirectory "saw"
   94      createDirectoryIfMissing True dir
   95      return ss { historyFile = Just (dir </> "history") }
   96    `X.catch` \(X.SomeException {}) -> return ss

   98 -- | Haskeline settings for the REPL.
   99 replSettings :: Settings REPL
  100 replSettings  = Settings
  101   { complete       = cryptolCommand
  102   , historyFile    = Nothing
  103   , autoAddHistory = True
  104   }

  126 instance E.MonadThrow REPL where
  127   throwM e = REPL $ \_ -> X.throwIO e
  128 
  129 instance E.MonadCatch REPL where
  130   catch repl_op handler = REPL $ \ioref ->
  131     E.catch (unREPL repl_op ioref) (\e -> unREPL (handler e) ioref)
  132 
  133 instance E.MonadMask REPL where
  134   mask repl_op = REPL $ \ioref ->
  135     E.mask (\runIO -> unREPL (repl_op (\f -> REPL (\ioref' -> runIO (unREPL f ioref')))) ioref)
  136   uninterruptibleMask repl_op = REPL $ \ioref ->
  137     E.uninterruptibleMask (\runIO -> unREPL (repl_op (\f -> REPL (\ioref' -> runIO (unREPL f ioref')))) ioref)
  138   generalBracket acquire release repl_op = REPL $ \ioref ->
  139     E.generalBracket
  140     (unREPL acquire ioref)
  141     (\rsrc exitCase -> unREPL (release rsrc exitCase) ioref)
  142     (\rsrc -> unREPL (repl_op rsrc) ioref)

  148 -- | Completion for cryptol commands.
  149 cryptolCommand :: CompletionFunc REPL
  150 cryptolCommand cursor@(l,r)
  151   | ":" `isPrefixOf` l'
  152   , Just (cmd,rest) <- splitCommand l' = case findCommand cmd of
  153 
  154       [c] | null rest && not (any isSpace l') -> do
  155             return (l, [cmdComp cmd c])
  156           | otherwise -> do
  157             (rest',cs) <- cmdArgument (cBody c) (reverse (sanitize rest),r)
  158             return (unwords [rest', reverse cmd],cs)
  159 
  160       cmds ->
  161         return (l, [ cmdComp l' c | c <- cmds ])
  162 
  163   | otherwise = completeSAWScript cursor
  164   where
  165   l' = sanitize (reverse l)
  166 
  167 -- | Generate a completion from a REPL command definition.
  168 cmdComp :: String -> CommandDescr -> Completion
  169 cmdComp prefix c = Completion
  170   { replacement = drop (length prefix) (cName c)
  171   , display     = cName c
  172   , isFinished  = True
  173   }
  174 
  175 -- | Dispatch to a completion function based on the kind of completion the
  176 -- command is expecting.
  177 cmdArgument :: CommandBody -> CompletionFunc REPL
  178 cmdArgument ct cursor@(l,_) = case ct of
  179   ExprArg     _ -> completeSAWScript cursor
  180   FilenameArg _ -> completeFilename cursor
  181   ShellArg _    -> completeFilename cursor
  182   NoArg       _ -> return (l,[])
  183 
  184 -- | Complete a name from the expression environment.
  185 completeExpr :: CompletionFunc REPL
  186 completeExpr (l,_) = do
  187   ns <- getExprNames
  188   let n    = reverse l
  189       vars = filter (n `isPrefixOf`) ns
  190   return (l,map (nameComp n) vars)
  191 
  192 -- | Complete a name from the type synonym environment.
  193 completeType :: CompletionFunc REPL
  194 completeType (l,_) = do
  195   ns <- getTypeNames
  196   let n    = reverse l
  197       vars = filter (n `isPrefixOf`) ns
  198   return (l,map (nameComp n) vars)

  200 data LexerMode = ModeNormal | ModeCryptol | ModeCryType | ModeQuote
  201 
  202 lexerMode :: String -> LexerMode
  203 lexerMode = normal
  204   where
  205     normal [] = ModeNormal
  206     normal ('{' : '{' : s) = cryptol s
  207     normal ('{' : '|' : s) = crytype s
  208     normal ('\"' : s) = quote s
  209     normal (_ : s) = normal s
  210 
  211     cryptol [] = ModeCryptol
  212     cryptol ('}' : '}' : s) = normal s
  213     cryptol (_ : s) = cryptol s
  214 
  215     crytype [] = ModeCryType
  216     crytype ('|' : '}' : s) = normal s
  217     crytype (_ : s) = crytype s
  218 
  219     quote [] = ModeQuote
  220     quote ('\"' : s) = normal s
  221     quote (_ : s) = quote s
  222 
  223 isIdentChar :: Char -> Bool
  224 isIdentChar c = isAlphaNum c || c `elem` "_\'"
  225 
  226 -- | Complete a name from the sawscript environment.
  227 completeSAWScript :: CompletionFunc REPL
  228 completeSAWScript cursor@(l, _) = do
  229   ns1 <- getSAWScriptNames
  230   ns2 <- getExprNames
  231   ns3 <- getTypeNames
  232   let n = reverse (takeWhile isIdentChar l)
  233       nameComps prefix ns = map (nameComp prefix) (filter (prefix `isPrefixOf`) ns)
  234   case lexerMode (reverse l) of
  235     ModeNormal  -> return (l, nameComps n ns1)
  236     ModeCryptol -> return (l, nameComps n ns2)
  237     ModeCryType -> return (l, nameComps n ns3)
  238     ModeQuote   -> completeFilename cursor
  239 
  240 -- | Generate a completion from a prefix and a name.
  241 nameComp :: String -> String -> Completion
  242 nameComp prefix c = Completion
  243   { replacement = drop (length prefix) c
  244   , display     = c
  245   , isFinished  = True
  246   }

  249 -- | Join two completion functions together, merging and sorting their results.
  250 (+++) :: CompletionFunc REPL -> CompletionFunc REPL -> CompletionFunc REPL
  251 (as +++ bs) cursor = do
  252   (_,acs) <- as cursor
  253   (_,bcs) <- bs cursor
  254   return (fst cursor, sortBy (compare `on` replacement) (acs ++ bcs))
