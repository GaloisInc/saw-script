   11 --
   12 -- namedTyVars is a type-class-polymorphic function for extracting named
   13 -- type variables from a type or type schema. It returns a set of Name
   14 -- (Name is just Text) manifested as a map from those Names to their
   15 -- positions/provenance.
   16 --
   17 
   18 class NamedTyVars t where
   19   namedTyVars :: t -> Map Name Pos
   20 
   21 instance (Ord k, NamedTyVars a) => NamedTyVars (Map k a) where
   22   namedTyVars = namedTyVars . M.elems
   23 
   24 instance (NamedTyVars a) => NamedTyVars [a] where
   25   namedTyVars = M.unionsWith choosePos . map namedTyVars
   26 
   27 instance NamedTyVars Type where
   28   namedTyVars t = case t of
   29     TyCon _ _ ts      -> namedTyVars ts
   30     TyRecord _ tm     -> namedTyVars tm
   31     TyVar pos n       -> M.singleton n pos
   32     TyUnifyVar _ _    -> M.empty
   33 
   34 instance NamedTyVars Schema where
   35   namedTyVars (Forall ns t) = namedTyVars t M.\\ M.fromList ns'
   36     where ns' = map (\(pos, n) -> (n, pos)) ns
