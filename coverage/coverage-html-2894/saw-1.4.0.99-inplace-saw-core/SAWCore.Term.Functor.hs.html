   55 -- Sorts -----------------------------------------------------------------------
   56 
   57 -- | The sorts, also known as universes, which can either be a predicative
   58 -- universe with level i or the impredicative universe Prop.
   59 data Sort
   60   = TypeSort Natural
   61   | PropSort
   62   deriving (Eq, Generic, TH.Lift)
   63 
   64 -- Prop is the lowest sort
   65 instance Ord Sort where
   66   PropSort <= _ = True
   67   (TypeSort _) <= PropSort = False
   68   (TypeSort i) <= (TypeSort j) = i <= j
   69 
   70 instance Hashable Sort -- automatically derived
   71 
   72 instance Show Sort where
   73   showsPrec p (TypeSort i) = showParen (p >= 10) (showString "sort " . shows i)
   74   showsPrec _ PropSort = showString "Prop"
   75 
   76 -- | Create sort @Type i@ for the given natural number @i@.
   77 mkSort :: Natural -> Sort
   78 mkSort i = TypeSort i
   79 
   80 -- | Wrapper around 'PropSort', for export
   81 propSort :: Sort
   82 propSort = PropSort
   83 
   84 -- | Returns sort of the given sort.
   85 sortOf :: Sort -> Sort
   86 sortOf (TypeSort i) = TypeSort (i + 1)
   87 sortOf PropSort = TypeSort 0
   88 
   89 -- | Get the maximum sort in a list, returning Prop for the empty list
   90 maxSort :: [Sort] -> Sort
   91 maxSort [] = propSort
   92 maxSort ss = maximum ss
   93 
   94 -- | This type represents a set of advisory flags for 'Sort's that are mostly
   95 -- ignored, but are used in the Coq export process to indicate where various
   96 -- typeclass instances are necessary in function definitions. In the concrete
   97 -- syntax "isort", "qsort", etc. is used to indicate cases where these flags
   98 -- are set. Note in particular that these flags do not affect typechecking,
   99 -- so missing or overeager "isort"/"qsort" annotations will only be detected
  100 -- via the Coq export.
  101 --
  102 -- * If 'flagInhabited' is 'True', an implicit @Inhabited@ typeclass argument
  103 --   will be added during Coq translation. In the concrete syntax, an "i" is
  104 --   prepended to the sort (e.g. "isort").
  105 -- * If 'flagQuantType' is 'True', an implicit @QuantType@ typeclass argument
  106 --   will be added during Coq translation. In the concrete syntax, an "q" is
  107 --   prepended to the sort (e.g. "qsort", "qisort").
  108 data SortFlags = SortFlags { flagInhabited :: Bool
  109                            , flagQuantType :: Bool }
  110     deriving (Eq, Ord, Generic, TH.Lift)
  111 
  112 instance Hashable SortFlags -- automatically derived
  113 
  114 instance Show SortFlags where
  115   showsPrec _ (SortFlags i q) = showString $
  116     concatMap (\(b,s) -> if b then s else "")
  117               [(q,"q"), (i,"i")]
  118 
  119 -- | The 'SortFlags' object with no flags set
  120 noFlags :: SortFlags
  121 noFlags = SortFlags False False
  122 
  123 -- | Apply a binary operation to corresponding flags of two 'SortFlags'
  124 sortFlagsLift2 :: (Bool -> Bool -> Bool) -> SortFlags -> SortFlags -> SortFlags
  125 sortFlagsLift2 f (SortFlags i1 q1) (SortFlags i2 q2) = SortFlags (f i1 i2) (f q1 q2)
  126 
  127 -- | Convert a 'SortFlags' to a list of 'Bool's, indicating which flags are set
  128 sortFlagsToList :: SortFlags -> [Bool]
  129 sortFlagsToList (SortFlags i q) = [i, q]
  130 
  131 -- | Build a 'SortFlags' from a list of 'Bool's indicating which flags are set
  132 sortFlagsFromList :: [Bool] -> SortFlags
  133 sortFlagsFromList bs = SortFlags (isSet 0) (isSet 1)
  134   where isSet i = i < length bs && bs !! i

  137 -- Flat Terms ------------------------------------------------------------------
  138 
  139 -- | The "flat terms", which are the built-in atomic constructs of SAW core.
  140 --
  141 -- NB: If you add constructors to FlatTermF, make sure you update
  142 --     zipWithFlatTermF!
  143 data FlatTermF e
  144     -- Tuples are represented as nested pairs, grouped to the right,
  145     -- terminated with unit at the end.
  146   = UnitValue
  147   | UnitType
  148   | PairValue e e
  149   | PairType e e
  150   | PairLeft e
  151   | PairRight e
  152 
  153     -- | A recursor, which is specified by a 'CompiledRecursor'
  154     -- comprising the datatype name, elimination sort, and other data
  155     -- about the recursor.
  156   | Recursor CompiledRecursor
  157 
  158     -- | Non-dependent record types, i.e., N-ary tuple types with named
  159     -- fields. These are considered equal up to reordering of fields. Actual
  160     -- tuple types are represented with field names @"1"@, @"2"@, etc.
  161   | RecordType ![(FieldName, e)]
  162     -- | Non-dependent records, i.e., N-ary tuples with named fields. These are
  163     -- considered equal up to reordering of fields. Actual tuples are
  164     -- represented with field names @"1"@, @"2"@, etc.
  165   | RecordValue ![(FieldName, e)]
  166     -- | Non-dependent record projection
  167   | RecordProj e !FieldName
  168 
  169     -- | Sorts, aka universes, are the types of types; i.e., an object is a
  170     -- "type" iff it has type @Sort s@ for some s. See 'SortFlags' for an
  171     -- explanation of the extra argument.
  172   | Sort !Sort !SortFlags
  173 
  174     -- Primitive builtin values
  175     -- | Array value includes type of elements followed by elements.
  176   | ArrayValue e (Vector e)
  177     -- | String literal
  178   | StringLit !Text
  179   deriving (Eq, Ord, Show, Functor, Foldable, Traversable, Generic)
  180 
  181 instance Hashable e => Hashable (FlatTermF e) -- automatically derived
  182 
  183 -- | A 'CompiledRecursor' comprises the datatype name and elimination
  184 -- sort of a recursor, along with some other data derived from details
  185 -- of the datatype definition.
  186 data CompiledRecursor =
  187   CompiledRecursor
  188   { recursorDataType  :: Name
  189   , recursorSort      :: Sort
  190   , recursorNumParams :: Int
  191   , recursorNumIxs    :: Int
  192   , recursorCtorOrder :: [Name]
  193   }
  194  deriving (Eq, Ord, Show, Generic)
  195 
  196 instance Hashable CompiledRecursor -- automatically derived
  197 
  198 -- | Test if the association list used in a 'RecordType' or 'RecordValue' uses
  199 -- precisely the given field names and no more. If so, return the values
  200 -- associated with those field names, in the order given in the input, and
  201 -- otherwise return 'Nothing'
  202 alistAllFields :: Eq k => [k] -> [(k, a)] -> Maybe [a]
  203 alistAllFields [] [] = Just []
  204 alistAllFields (fld:flds) alist
  205   | Just val <- lookup fld alist =
  206     (val :) <$> alistAllFields flds (deleteField fld alist)
  207   where
  208     deleteField _ [] = error "deleteField"
  209     deleteField f ((f',_):rest) | f == f' = rest
  210     deleteField f (x:rest) = x : deleteField f rest
  211 alistAllFields _ _ = Nothing
  212 
  213 zipName :: Name -> Name -> Maybe Name
  214 zipName x y
  215   | x == y = Just x
  216   | otherwise = Nothing
  217 
  218 zipRec :: CompiledRecursor -> CompiledRecursor -> Maybe CompiledRecursor
  219 zipRec (CompiledRecursor d1 s1 ps1 n1 ord1) (CompiledRecursor d2 s2 _ n2 ord2)
  220   | n1 == n2 && s1 == s2
  221   = do d <- zipName d1 d2
  222        ord <- sequence (zipWith zipName ord1 ord2)
  223        pure $ CompiledRecursor
  224               d
  225               s1
  226               ps1
  227               n1
  228               ord
  229   | otherwise = Nothing
  230 
  231 -- | Zip a binary function @f@ over a pair of 'FlatTermF's by applying @f@
  232 -- pointwise to immediate subterms, if the two 'FlatTermF's are the same
  233 -- constructor; otherwise, return 'Nothing' if they use different constructors
  234 zipWithFlatTermF :: (x -> y -> z) -> FlatTermF x -> FlatTermF y ->
  235                     Maybe (FlatTermF z)
  236 zipWithFlatTermF f = go
  237   where
  238     go UnitValue UnitValue = Just UnitValue
  239     go UnitType UnitType = Just UnitType
  240     go (PairValue x1 x2) (PairValue y1 y2) = Just (PairValue (f x1 y1) (f x2 y2))
  241     go (PairType x1 x2) (PairType y1 y2) = Just (PairType (f x1 y1) (f x2 y2))
  242     go (PairLeft x) (PairLeft y) = Just (PairLeft (f x y))
  243     go (PairRight x) (PairRight y) = Just (PairLeft (f x y))
  244 
  245     go (Recursor rec1) (Recursor rec2) =
  246       Recursor <$> zipRec rec1 rec2
  247 
  248     go (RecordType elems1) (RecordType elems2)
  249       | Just vals2 <- alistAllFields (map fst elems1) elems2 =
  250         Just $ RecordType $ zipWith (\(fld,x) y -> (fld, f x y)) elems1 vals2
  251     go (RecordValue elems1) (RecordValue elems2)
  252       | Just vals2 <- alistAllFields (map fst elems1) elems2 =
  253         Just $ RecordValue $ zipWith (\(fld,x) y -> (fld, f x y)) elems1 vals2
  254     go (RecordProj e1 fld1) (RecordProj e2 fld2)
  255       | fld1 == fld2 = Just $ RecordProj (f e1 e2) fld1
  256 
  257     go (Sort sx hx) (Sort sy hy) | sx == sy = Just (Sort sx (sortFlagsLift2 (&&) hx hy))
  258          -- /\ NB, it's not entirely clear how the flags should be propagated
  259     go (StringLit s) (StringLit t) | s == t = Just (StringLit s)
  260     go (ArrayValue tx vx) (ArrayValue ty vy)
  261       | V.length vx == V.length vy
  262       = Just $ ArrayValue (f tx ty) (V.zipWith f vx vy)
  263 
  264     go UnitValue      _ = Nothing
  265     go UnitType       _ = Nothing
  266     go PairValue{}    _ = Nothing
  267     go PairType{}     _ = Nothing
  268     go PairLeft{}     _ = Nothing
  269     go PairRight{}    _ = Nothing
  270     go Recursor{}     _ = Nothing
  271     go RecordType{}   _ = Nothing
  272     go RecordValue{}  _ = Nothing
  273     go RecordProj{}   _ = Nothing
  274     go Sort{}         _ = Nothing
  275     go ArrayValue{}   _ = Nothing
  276     go StringLit{}    _ = Nothing

  279 -- Term Functor ----------------------------------------------------------------
  280 
  281 -- | A \"knot-tying\" structure for representing terms and term-like things.
  282 -- Often, this appears in context as the type \"'TermF' 'Term'\", in which case
  283 -- it represents a full 'Term' AST. The \"F\" stands for 'Functor', or
  284 -- occasionally for \"Former\".
  285 data TermF e
  286     = FTermF !(FlatTermF e)
  287       -- ^ The atomic, or builtin, term constructs
  288     | App !e !e
  289       -- ^ Applications of functions
  290     | Lambda !VarName !e !e
  291       -- ^ Function abstractions
  292     | Pi !VarName !e !e
  293       -- ^ The type of a (possibly) dependent function
  294     | Constant !Name
  295       -- ^ A global constant identified by its name.
  296     | Variable !VarName !e
  297       -- ^ A named variable with a type.
  298   deriving (Eq, Ord, Show, Functor, Foldable, Traversable, Generic)
  299 
  300 instance Hashable e => Hashable (TermF e) -- automatically derived

  303 -- Free Named Variables --------------------------------------------------------
  304 
  305 -- | Compute an 'IntSet' containing the 'VarIndex' of the free
  306 -- variables of a term, given the free variables for its immediate
  307 -- subterms.
  308 freesTermF :: TermF IntSet -> IntSet
  309 freesTermF tf =
  310   case tf of
  311     FTermF ftf -> Foldable.foldl' IntSet.union IntSet.empty ftf
  312     App l r -> IntSet.union l r
  313     Lambda nm tp rhs -> IntSet.union tp (IntSet.delete (vnIndex nm) rhs)
  314     Pi nm lhs rhs -> IntSet.union lhs (IntSet.delete (vnIndex nm) rhs)
  315     Constant {} -> IntSet.empty
  316     Variable nm tp -> IntSet.insert (vnIndex nm) tp
