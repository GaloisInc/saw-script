   26 type SBool = SVal
   27 type SWord = SVal
   28 type SInteger = SVal
   29 
   30 fromBitsLE :: [SBool] -> SWord
   31 fromBitsLE bs = foldl' f (literalSWord 0 0) bs
   32   where f w b = svJoin (svToWord1 b) w
   33 
   34 literalSWord :: Int -> Integer -> SWord
   35 literalSWord w i = svInteger (KBounded False w) i
   36 
   37 literalSInteger :: Integer -> SWord
   38 literalSInteger i = svInteger KUnbounded i
   39 
   40 forallSWord :: String -> Int -> Symbolic SWord
   41 forallSWord nm w = symbolicEnv >>= liftIO . svMkSymVar (NonQueryVar (Just ALL)) (KBounded False w) (Just nm)
   42 
   43 forallSWord_ :: Int -> Symbolic SWord
   44 forallSWord_ w = symbolicEnv >>= liftIO . svMkSymVar (NonQueryVar (Just ALL)) (KBounded False w) Nothing
   45 
   46 existsSWord :: String -> Int -> Symbolic SWord
   47 existsSWord nm w = symbolicEnv >>= liftIO . svMkSymVar (NonQueryVar (Just EX)) (KBounded False w) (Just nm)
   48 
   49 existsSWord_ :: Int -> Symbolic SWord
   50 existsSWord_ w = symbolicEnv >>= liftIO . svMkSymVar (NonQueryVar (Just EX)) (KBounded False w) Nothing
