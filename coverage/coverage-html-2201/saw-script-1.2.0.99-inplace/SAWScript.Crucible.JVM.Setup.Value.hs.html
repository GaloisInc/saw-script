  101 -- *** JVMMethodId
  102 
  103 data JVMMethodId =
  104   JVMMethodId
  105     { _jvmMethodKey :: J.MethodKey
  106     , _jvmClassName  :: J.ClassName
  107     }
  108   deriving (Eq, Ord, Show)
  109 
  110 makeLenses ''JVMMethodId
  111 
  112 -- TODO: avoid intermediate 'String' values
  113 instance PPL.Pretty JVMMethodId where
  114   pretty (JVMMethodId methKey className) =
  115     PPL.pretty (concat [J.unClassName className ,".", J.methodKeyName methKey])
  116 
  117 type instance MS.MethodId CJ.JVM = JVMMethodId
  118

  120 -- *** Allocation

  122 data Allocation
  123   = AllocObject J.ClassName
