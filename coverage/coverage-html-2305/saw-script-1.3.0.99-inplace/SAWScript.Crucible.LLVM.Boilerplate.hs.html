   35 sortByDeps :: [FunctionSkeleton] -> [FunctionSkeleton]
   36 sortByDeps skels = reverse $ (\(f, _, _) -> f) . fromVertex <$> Graph.topSort g
   37   where
   38     (g, fromVertex, _) = Graph.graphFromEdges $ adjacency <$> skels
   39     adjacency :: FunctionSkeleton -> (FunctionSkeleton, Text, [Text])
   40     adjacency s = (s, s ^. funSkelName, Set.toList $ s ^. funSkelCalls)

   42 preludeBoilerplate :: Bool -> ModuleSkeleton -> Text
   43 preludeBoilerplate True _mskel = "enable_experimental;\nMODULE_SKEL <- module_skeleton MODULE;\n\n"
   44 preludeBoilerplate False _mskel = "\n";
   45 
   46 globalsPreBoilerplate :: Bool -> ModuleSkeleton -> Text
   47 globalsPreBoilerplate True _mskel = "skeleton_globals_pre MODULE_SKEL;\n"
   48 globalsPreBoilerplate False _mskel = "\n"
   49 
   50 globalsPostBoilerplate :: Bool -> ModuleSkeleton -> Text
   51 globalsPostBoilerplate True _mskel = "skeleton_globals_post MODULE_SKEL;\n"
   52 globalsPostBoilerplate False _mskel = "\n"

   54 llvmTypeBoilerplate :: LLVM.Type -> Text
   55 llvmTypeBoilerplate (LLVM.PrimType (LLVM.Integer n)) = mconcat
   56   [ "(llvm_int ", Text.pack $ show n, ")"
   57   ]
   58 llvmTypeBoilerplate (LLVM.PrimType (LLVM.FloatType LLVM.Float)) = "llvm_float"
   59 llvmTypeBoilerplate (LLVM.PrimType (LLVM.FloatType LLVM.Double)) = "llvm_double"
   60 llvmTypeBoilerplate (LLVM.Alias (LLVM.Ident n)) = mconcat
   61   [ "(llvm_type \"%", Text.pack n, ")"
   62   ]
   63 llvmTypeBoilerplate _ = "undefined"
   64 
   65 typeBoilerplate :: TypeSkeleton -> Text
   66 typeBoilerplate t
   67   | t ^. typeSkelIsPointer
   68   , (s:_) <- t ^. typeSkelSizeGuesses
   69   = mconcat
   70     [ "(llvm_array "
   71     , Text.pack . show $ s ^. sizeGuessElems
   72     , " "
   73     , llvmTypeBoilerplate $ t ^. typeSkelLLVMType
   74     , ")"
   75     ]
   76   | otherwise = llvmTypeBoilerplate $ t ^. typeSkelLLVMType
   77 
   78 bodyBoilerplate :: Bool -> FunctionSkeleton -> Text -> Text
   79 bodyBoilerplate True _fskel skelName = mconcat
   80   [ "  prestate <- skeleton_prestate ", skelName, ";\n"
   81   , "  skeleton_exec prestate;\n"
   82   , "  poststate <- skeleton_poststate ", skelName,  " prestate;\n"
   83   ]
   84 bodyBoilerplate False fskel _ = mconcat
   85   [ mconcat $ uncurry argPrecondition <$> numberedArgs
   86   , "  llvm_execute_func ["
   87   , Text.intercalate ", " $ uncurry argTerm <$> numberedArgs
   88   , "];\n\n"
   89   , if | LLVM.PrimType LLVM.Void <- fskel ^. funSkelRet . typeSkelLLVMType -> ""
   90        | fskel ^. funSkelRet . typeSkelIsPointer -> "  llvm_return undefined;\n\n"
   91        | otherwise -> mconcat
   92          [ "  __return <- llvm_fresh_var \"(return ", fskel ^. funSkelName
   93          , ")\" ", typeBoilerplate $ fskel ^. funSkelRet, ";\n"
   94          , "  llvm_return (llvm_term __return);\n\n"
   95          ]
   96   , mconcat $ uncurry argPostcondition <$> numberedArgs
   97   ]
   98   where
   99     numberedArgs = zip [0..] $ fskel ^. funSkelArgs
  100     argName :: Int -> ArgSkeleton -> Text
  101     argName i a
  102       | Just n <- a ^. argSkelName = n
  103       | otherwise = "arg" <> Text.pack (show i)
  104     argPrecondition :: Int -> ArgSkeleton -> Text
  105     argPrecondition i a
  106       | a ^. argSkelType . typeSkelIsPointer
  107       , (s:_) <- a ^. argSkelType . typeSkelSizeGuesses
  108       , s ^. sizeGuessInitialized
  109       = mconcat
  110         [ "  ", argName i a, " <- llvm_fresh_var \"", argName i a, "\" ", typeBoilerplate (a ^. argSkelType), ";\n"
  111         , "  ", argName i a, "_ptr <- llvm_alloc ", typeBoilerplate (a ^. argSkelType), ";\n"
  112         , "  llvm_points_to ", argName i a, "_ptr (llvm_term ", argName i a, ");\n\n"
  113         ]
  114       | a ^. argSkelType . typeSkelIsPointer
  115       = mconcat
  116         [ "  ", argName i a, "_ptr <- llvm_alloc ", typeBoilerplate (a ^. argSkelType), ";\n\n"
  117         ]
  118       | otherwise
  119       = mconcat
  120         [ "  ", argName i a, " <- llvm_fresh_var \"", argName i a, "\" ", typeBoilerplate (a ^. argSkelType), ";\n\n"
  121         ]
  122     argPostcondition :: Int -> ArgSkeleton -> Text
  123     argPostcondition i a
  124       | a ^. argSkelType . typeSkelIsPointer
  125       = mconcat
  126         [ "  ", argName i a, "_post <- llvm_fresh_var \"", argName i a, "_post\" ", typeBoilerplate (a ^. argSkelType), ";\n"
  127         , "  llvm_points_to ", argName i a, "_ptr (llvm_term ", argName i a, "_post);\n\n"
  128         ]
  129       | otherwise = ""
  130     argTerm :: Int -> ArgSkeleton -> Text
  131     argTerm i a
  132       | a ^. argSkelType . typeSkelIsPointer
  133       = argName i a <> "_ptr"
  134       | otherwise
  135       = "(llvm_term " <> argName i a <> ")"

  137 functionBoilerplate :: Bool -> ModuleSkeleton -> FunctionSkeleton -> Text
  138 functionBoilerplate skelBuiltins mskel fskel = mconcat
  139   [ if skelBuiltins
  140     then mconcat [skelName, " <- function_skeleton MODULE_SKEL \"", name, "\";\n"]
  141     else ""
  142   , "let ", specName, " = do {\n"
  143   , "  ", globalsPreBoilerplate skelBuiltins mskel
  144   , bodyBoilerplate skelBuiltins fskel skelName
  145   , "  ", globalsPostBoilerplate skelBuiltins mskel
  146   , "};\n"
  147   , overrideName, " <- llvm_verify MODULE \"", name, "\" ["
  148   , Text.intercalate ", " $ (<>"_override") <$> (Set.toList $ fskel ^. funSkelCalls)
  149   , "] false ", specName, " z3;\n"
  150   ]
  151   where
  152     name = fskel ^. funSkelName
  153     specName = name <> "_spec"
  154     skelName = name <> "_skel"
  155     overrideName = name <> "_override"
  156 
  157 llvm_boilerplate :: FilePath -> ModuleSkeleton -> Bool -> TopLevel ()
  158 llvm_boilerplate path mskel skelBuiltins = do
  159   let fskels = sortByDeps . Map.elems $ mskel ^. modSkelFunctions
  160   liftIO . withFile path WriteMode $ \h -> Text.IO.hPutStrLn h $ mconcat
  161     [ preludeBoilerplate skelBuiltins mskel
  162     , Text.unlines $ functionBoilerplate skelBuiltins mskel <$> fskels
  163     ]
