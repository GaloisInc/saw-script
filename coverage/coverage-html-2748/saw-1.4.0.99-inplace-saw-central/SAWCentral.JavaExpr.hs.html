   87 data JavaExprF v
   88   = ReturnVal JSS.Type
   89   | Local String JSS.LocalVariableIndex JSS.Type
   90   | InstanceField v JSS.FieldId
   91   | StaticField JSS.FieldId
   92   deriving (Functor, CC.Foldable, CC.Traversable)
   93 
   94 instance CC.EqFoldable JavaExprF where
   95   fequal (ReturnVal _) (ReturnVal _) = True
   96   fequal (Local _ i _)(Local _ j _) = i == j
   97   fequal (InstanceField xr xf) (InstanceField yr yf) = xf == yf && (xr == yr)
   98   fequal (StaticField xf) (StaticField yf) = xf == yf
   99   fequal _ _ = False
  100 
  101 instance CC.OrdFoldable JavaExprF where
  102   ReturnVal _ `fcompare` ReturnVal _  = EQ
  103   ReturnVal _ `fcompare` _            = LT
  104   _           `fcompare` ReturnVal _  = GT
  105   Local _ i _ `fcompare` Local _ i' _ = i `compare` i'
  106   Local _ _ _ `fcompare` _           = LT
  107   _           `fcompare` Local _ _ _ = GT
  108   InstanceField r1 f1 `fcompare` InstanceField r2 f2 =
  109         case r1 `compare` r2 of
  110           EQ -> f1 `compare` f2
  111           r  -> r
  112   StaticField f1 `fcompare` StaticField f2 = f1 `compare` f2
  113   StaticField _ `fcompare` _ = GT
  114   _ `fcompare` StaticField _ = LT
  115 
  116 instance CC.ShowFoldable JavaExprF where
  117   fshow (ReturnVal _) = "return"
  118   fshow (Local nm _ _) = nm
  119   fshow (InstanceField r f) = show r ++ "." ++ JSS.fieldIdName f
  120   fshow (StaticField f) = ppFldId f

  122 -- | Typechecked JavaExpr
  123 type JavaExpr = CC.Term JavaExprF

  125 thisJavaExpr :: JSS.Class -> JavaExpr
  126 thisJavaExpr cl = CC.Term (Local "this" 0 (JSS.ClassType (JSS.className cl)))
  127 
  128 returnJavaExpr :: JSS.Method -> Maybe (JavaExpr)
  129 returnJavaExpr meth =
  130   (CC.Term . ReturnVal) <$> JSS.methodReturnType meth
  131 
  132 isReturnExpr :: JavaExpr -> Bool
  133 isReturnExpr (CC.Term (ReturnVal _)) = True
  134 isReturnExpr _ = False
  135 
  136 containsReturn :: JavaExpr -> Bool
  137 containsReturn (CC.Term e) =
  138   case e of
  139     ReturnVal _ -> True
  140     InstanceField e' _ -> containsReturn e'
  141     _ -> False
  142 
  143 -- | Pretty print a Java expression.
  144 ppJavaExpr :: JavaExpr -> String
  145 ppJavaExpr (CC.Term exprF) =
  146   case exprF of
  147     ReturnVal _ -> "return"
  148     Local nm _ _ -> nm
  149     InstanceField r f -> ppJavaExpr r ++ "." ++ JSS.fieldIdName f
  150     StaticField f -> ppFldId f
  151 
  152 -- | Turn a Java expression into a valid SAWCore variable name.
  153 jeVarName :: JavaExpr -> String
  154 jeVarName = map dotToUnderscore . ppJavaExpr
  155   where dotToUnderscore '.' = '_'
  156         dotToUnderscore c = c
  157 
  158 asJavaExpr :: Term -> Maybe String
  159 asJavaExpr (asVariable -> Just (x, _)) = Just (Text.unpack (vnName x))
  160 asJavaExpr _ = Nothing
  161 
  162 isRefJavaExpr :: JavaExpr -> Bool
  163 isRefJavaExpr = JSS.isRefType . exprType
  164 
  165 isClassJavaExpr :: JavaExpr -> Bool
  166 isClassJavaExpr = isClassType . exprType
  167   where isClassType (JSS.ClassType _) = True
  168         isClassType _ = False
  169 
  170 maxArg :: JSS.Method -> Int
  171 maxArg meth = length (JSS.methodParameterTypes meth) - 1
  172 
  173 isArg :: JSS.Method -> JavaExpr -> Bool
  174 isArg meth (CC.Term (Local _ idx _)) =
  175   idx <= JSS.localIndexOfParameter meth (maxArg meth)
  176 isArg _ _ = False
  177 
  178 exprType :: JavaExpr -> JSS.Type
  179 exprType (CC.Term (ReturnVal ty)) = ty
  180 exprType (CC.Term (Local _ _ ty)) = ty
  181 exprType (CC.Term (InstanceField _ f)) = JSS.fieldIdType f
  182 exprType (CC.Term (StaticField f)) = JSS.fieldIdType f
  183 
  184 exprDepth :: JavaExpr -> Int
  185 exprDepth (CC.Term (InstanceField e _)) = 1 + exprDepth e
  186 exprDepth _ = 1
  187 
  188 isScalarExpr :: JavaExpr -> Bool
  189 isScalarExpr = JSS.isPrimitiveType . exprType

  226 -- | Identifies concrete type of a Java expression.
  227 data JavaActualType
  228   = ClassInstance JSS.Class
  229   -- TODO: eventually change JSS.Type to JavaActualType, for more flexibility
  230   | ArrayInstance Int JSS.Type
  231   | PrimitiveType JSS.Type
  232   deriving (Show)
  233 
  234 instance Eq JavaActualType where
  235   ClassInstance c1 == ClassInstance c2 = JSS.className c1 == JSS.className c2
  236   ArrayInstance l1 tp1 == ArrayInstance l2 tp2 = l1 == l2 && tp1 == tp2
  237   PrimitiveType tp1 == PrimitiveType tp2 = tp1 == tp2
  238   _ == _ = False
  239 
  240 -- | Returns true if this represents a reference.
  241 isActualRef :: JavaActualType -> Bool
  242 isActualRef ClassInstance{} = True
  243 isActualRef ArrayInstance{} = True
  244 isActualRef PrimitiveType{} = False
  245 
  246 -- | Check whether a Type and a JavaActualType are compatible.
  247 actualTypeCompatible :: JSS.Type -> JavaActualType -> Bool
  248 actualTypeCompatible ty aty = jssTypeOfActual aty == ty
  249 
  250 -- | Returns Java symbolic simulator type that actual type represents.
  251 jssTypeOfActual :: JavaActualType -> JSS.Type
  252 jssTypeOfActual (ClassInstance x) = JSS.ClassType (JSS.className x)
  253 jssTypeOfActual (ArrayInstance _ tp) = JSS.ArrayType tp
  254 jssTypeOfActual (PrimitiveType tp) = tp
  255 
  256 javaTypeToActual :: JSS.Type -> Maybe JavaActualType
  257 javaTypeToActual tp
  258   | JSS.isPrimitiveType tp = Just (PrimitiveType tp)
  259   | otherwise = Nothing

  261 narrowTypeOfActual :: SharedContext -> JavaActualType -> IO (Maybe Term)
  262 narrowTypeOfActual sc at =
  263   case cryptolTypeOfActual at of
  264     Nothing -> return Nothing
  265     Just cty ->
  266       do t <- importType sc emptyEnv cty
  267          return (Just t)
  268 
  269 cryptolTypeOfActual :: JavaActualType -> Maybe Cryptol.Type
  270 cryptolTypeOfActual ty =
  271   case ty of
  272     ClassInstance _ -> Nothing
  273     ArrayInstance l tp ->
  274       do at <- javaTypeToActual tp
  275          ct <- cryptolTypeOfActual at
  276          Just (Cryptol.tSeq (Cryptol.tNum l) ct)
  277     PrimitiveType pt ->
  278       case pt of
  279         JSS.BooleanType -> Just $ Cryptol.tBit
  280         JSS.ByteType    -> Just $ Cryptol.tWord (Cryptol.tNum (8 :: Integer))
  281         JSS.CharType    -> Just $ Cryptol.tWord (Cryptol.tNum (16 :: Integer))
  282         JSS.ShortType   -> Just $ Cryptol.tWord (Cryptol.tNum (16 :: Integer))
  283         JSS.IntType     -> Just $ Cryptol.tWord (Cryptol.tNum (32 :: Integer))
  284         JSS.LongType    -> Just $ Cryptol.tWord (Cryptol.tNum (64 :: Integer))
  285         JSS.ArrayType _ -> Nothing
  286         JSS.ClassType _ -> Nothing
  287         JSS.DoubleType  -> Nothing
  288         JSS.FloatType   -> Nothing
  289 
  290 ppActualType :: JavaActualType -> String
  291 ppActualType (ClassInstance x) = JSS.slashesToDots (JSS.unClassName (JSS.className x))
  292 ppActualType (ArrayInstance l tp) = show tp ++ "[" ++ show l ++ "]"
  293 ppActualType (PrimitiveType tp) = show tp

  295 parseJavaExpr :: JSS.Codebase -> JSS.Class -> JSS.Method -> String
  296               -> ExceptT String IO JavaExpr
  297 parseJavaExpr cb cls meth estr = parseParts eparts
  298   where parseParts :: [String] -> ExceptT String IO JavaExpr
  299         parseParts [] = throwE "empty Java expression"
  300         parseParts [s] =
  301           case s of
  302             "this" | JSS.methodIsStatic meth -> throwE $
  303                      "can't use 'this' in static method " ++
  304                      JSS.methodName meth
  305                    | otherwise -> return $ thisJavaExpr cls
  306             "return" -> case returnJavaExpr meth of
  307                           Just e -> return e
  308                           Nothing -> throwE $
  309                             "no return value for " ++ methodName meth
  310             ('a':'r':'g':'s':'[':rest) -> do
  311               let num = fst (break (==']') rest)
  312               case readMaybe num of
  313                 Just (n :: Int) -> do
  314                   let i = localIndexOfParameter meth n
  315                       mlv = lookupLocalVariableByIdx meth 0 i
  316                       paramTypes = V.fromList .
  317                                    methodKeyParameterTypes .
  318                                    methodKey $
  319                                    meth
  320                   case mlv of
  321                     Nothing
  322                       | n < V.length paramTypes ->
  323                         return $ CC.Term $ Local s i $ paramTypes V.! (fromIntegral n)
  324                       | otherwise -> throwE $
  325                           "(zero-based) local variable index " ++ show i ++
  326                           " for parameter " ++ show n ++ " doesn't exist"
  327                     Just lv -> return $ CC.Term $ Local s i $ localType lv
  328                 Nothing -> fail $ "bad Java expression syntax: " ++ s
  329             _ | hasDebugInfo meth -> do
  330                   let mlv = lookupLocalVariableByName meth 0 s
  331                   case mlv of
  332                     Nothing -> throwE $ "can't parse expression: " ++ estr
  333                     Just lv -> return $ CC.Term $ Local s i ty
  334                       where i = JSS.localIdx lv
  335                             ty = JSS.localType lv
  336               | otherwise -> throwE $
  337                   "variable " ++ s ++
  338                   " referenced by name, but no debug info available"
  339         parseParts (fname:rest) = do
  340           let cname = JSS.mkClassName (intercalate "/" (reverse rest))
  341           mc <- lift $ tryLookupClass cb cname
  342           case (filter ((== fname) . fieldName) . filter fieldIsStatic . classFields) <$> mc of
  343             Just [fld] -> do
  344               let fid = FieldId cname fname (fieldType fld)
  345               return $ CC.Term $ StaticField fid
  346             _ -> do
  347               e <- parseParts rest
  348               let jt = exprType e
  349                   pos = PosInternal "FIXME" -- TODO
  350               fid <- findField cb pos jt fname
  351               return $ CC.Term $ InstanceField e fid
  352         eparts = reverse $ splitOn "." estr
