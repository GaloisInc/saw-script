   64 data UTerm
   65   = Name (PosPair Text)
   66   | Sort Pos Sort SortFlags
   67   | App UTerm UTerm
   68   | Lambda Pos UTermCtx UTerm
   69   | Pi Pos UTermCtx UTerm
   70   | Recursor (PosPair Text) Sort
   71   | UnitValue Pos
   72   | UnitType Pos
   73     -- | New-style records
   74   | RecordValue Pos [(PosPair FieldName, UTerm)]
   75   | RecordType Pos [(PosPair FieldName, UTerm)]
   76   | RecordProj UTerm FieldName
   77     -- | Simple pairs
   78   | PairValue Pos UTerm UTerm
   79   | PairType Pos UTerm UTerm
   80   | PairLeft UTerm
   81   | PairRight UTerm
   82     -- | Identifies a type constraint on the term, i.e., a type ascription
   83   | TypeConstraint UTerm Pos UTerm
   84   | NatLit Pos Natural
   85   | StringLit Pos Text
   86     -- | Vector literal.
   87   | VecLit Pos [UTerm]
   88     -- | Bitvector literal.
   89   | BVLit Pos [Bool]
   90   | BadTerm Pos
   91  deriving (Show, TH.Lift)
   92 
   93 -- | A pattern used for matching a variable.
   94 data UTermVar
   95   = TermVar (PosPair LocalName)
   96   | UnusedVar Pos
   97   deriving (Eq, Ord, Show, TH.Lift)
   98 
   99 -- | Return the 'String' name associated with a 'UTermVar'
  100 termVarString :: UTermVar -> String
  101 termVarString (TermVar (PosPair _ str)) = Text.unpack str
  102 termVarString (UnusedVar _) = "_"
  103 
  104 -- | Return the 'LocalName' associated with a 'UTermVar'
  105 termVarLocalName :: UTermVar -> LocalName
  106 termVarLocalName (TermVar (PosPair _ str)) = str
  107 termVarLocalName (UnusedVar _) = "_"
  108 
  109 -- | A context of 0 or more variable bindings, with types
  110 type UTermCtx = [(UTermVar, UTerm)]
  111 
  112 instance Positioned UTerm where
  113   pos t =
  114     case t of
  115       Name i               -> pos i
  116       Sort p _ _           -> p
  117       Lambda p _ _         -> p
  118       App x _              -> pos x
  119       Pi p _ _             -> p
  120       Recursor i _         -> pos i
  121       UnitValue p          -> p
  122       UnitType p           -> p
  123       RecordValue p _      -> p
  124       RecordType p _       -> p
  125       RecordProj x _       -> pos x
  126       PairValue p _ _      -> p
  127       PairType p _ _       -> p
  128       PairLeft x           -> pos x
  129       PairRight x          -> pos x
  130       TypeConstraint _ p _ -> p
  131       NatLit p _           -> p
  132       StringLit p _        -> p
  133       VecLit p _           -> p
  134       BVLit p _            -> p
  135       BadTerm p            -> p
  136 
  137 instance Positioned UTermVar where
  138   pos (TermVar i) = pos i
  139   pos (UnusedVar p) = p
  140 
  141 badTerm :: Pos -> UTerm
  142 badTerm = BadTerm

  144 -- | A constructor declaration of the form @c (x1 :: tp1) .. (xn :: tpn) :: tp@
  145 data CtorDecl = Ctor (PosPair Text) UTermCtx UTerm
  146   deriving (Show, TH.Lift)

  148 data DefQualifier
  149   = NoQualifier
  150   | PrimQualifier
  151   | AxiomQualifier
  152  deriving (Eq, Show, Generic, TH.Lift)
  153 
  154 instance Hashable DefQualifier -- automatically derived

  156 -- | A top-level declaration in a saw-core file
  157 data Decl
  158    = TypeDecl DefQualifier (PosPair Text) UTerm
  159      -- ^ A declaration of something having a type, where the declaration
  160      -- qualifier states what flavor of thing it is
  161    | DataDecl (PosPair Text) UTermCtx UTerm [CtorDecl]
  162      -- ^ A declaration of an inductive data types, with a name, a parameter
  163      -- context, a return type, and a list of constructor declarations
  164    | TermDef (PosPair Text) [UTermVar] UTerm
  165      -- ^ A declaration of a term having a definition, with variables
  166    | TypedDef (PosPair Text) [(UTermVar, UTerm)] UTerm UTerm
  167      -- ^ A definition of something with a specific type, with parameters
  168    | InjectCodeDecl Text Text
  169      -- ^ Some raw text to inject into a translation
  170   deriving (Show, TH.Lift)

  172 -- | A set of constraints on what 'String' names to import from a module
  173 data ImportConstraint
  174   = SpecificImports [String]
  175     -- ^ Only import the given names
  176   | HidingImports [String]
  177     -- ^ Import all but the given names
  178  deriving (Eq, Ord, Show, TH.Lift)
  179 
  180 -- | An import declaration
  181 data Import = Import { importModName :: PosPair ModuleName
  182                        -- ^ The name of the module to import
  183                      , importConstraints :: Maybe ImportConstraint
  184                        -- ^ The constraints on what to import
  185                      }
  186             deriving (Show, TH.Lift)
  187 
  188 -- | Test whether a 'String' name satisfies the constraints of an 'Import'
  189 nameSatsConstraint :: Maybe ImportConstraint -> String -> Bool
  190 nameSatsConstraint Nothing _ = True
  191 nameSatsConstraint (Just (SpecificImports ns)) n = elem n ns
  192 nameSatsConstraint (Just (HidingImports ns)) n = notElem n ns
  193

  195 -- | A module declaration gives:
  196 -- * A name for the module;
  197 -- * A list of imports; AND
  198 -- * A list of top-level declarations
  199 data Module = Module (PosPair ModuleName) [Import] [Decl]
  200   deriving (Show, TH.Lift)
  201 
  202 moduleName :: Module -> ModuleName
  203 moduleName (Module (PosPair _ mnm) _ _) = mnm
  204 
  205 -- | Get a list of all names (i.e., definitions, axioms, or primitives) declared
  206 -- in a module, along with their types and qualifiers
  207 moduleTypedDecls :: Module -> [(Text, UTerm)]
  208 moduleTypedDecls (Module _ _ decls) = concatMap helper decls where
  209   helper :: Decl -> [(Text, UTerm)]
  210   helper (TypeDecl _ (PosPair _ nm) tm) = [(nm,tm)]
  211   helper _ = []
  212 
  213 -- | Get a list of all datatypes declared in a module
  214 moduleDataDecls :: Module -> [(Text, UTermCtx, UTerm, [CtorDecl])]
  215 moduleDataDecls (Module _ _ decls) = concatMap helper decls where
  216   helper :: Decl -> [(Text, UTermCtx, UTerm, [CtorDecl])]
  217   helper (DataDecl (PosPair _ nm) params tp ctors) = [(nm, params, tp, ctors)]
  218   helper _ = []
  219 
  220 moduleTypedDataDecls :: Module -> [(Text, UTerm)]
  221 moduleTypedDataDecls =
  222   map (\(nm,p_ctx,tp,_) ->
  223         (nm, Pi (pos tp) p_ctx tp)) . moduleDataDecls
  224 
  225 -- | Get a list of all constructors declared in a module, along with the context
  226 -- of parameters for each one
  227 moduleCtorDecls :: Module -> [(UTermCtx, CtorDecl)]
  228 moduleCtorDecls =
  229   concatMap (\(_,p_ctx,_,ctors) -> map (p_ctx,) ctors) . moduleDataDecls
  230 
  231 -- | Get a list of the names and types of all the constructors in a module
  232 moduleTypedCtorDecls :: Module -> [(Text, UTerm)]
  233 moduleTypedCtorDecls =
  234   concatMap (\(_,p_ctx,_,ctors) ->
  235               map (\(Ctor (PosPair _ nm) ctx tp) ->
  236                     (nm, Pi (pos tp) (p_ctx ++ ctx) tp)) ctors)
  237   . moduleDataDecls
  238 
  239 asPiList :: UTerm -> (UTermCtx, UTerm)
  240 asPiList (Pi _ ctx1 body1) =
  241   let (ctx2,body2) = asPiList body1 in
  242   (ctx1 ++ ctx2, body2)
  243 asPiList t = ([], t)
  244 
  245 asApp :: UTerm -> (UTerm, [UTerm])
  246 asApp = go []
  247   where go l (App t u)   = go (u:l) t
  248         go l t = (t,l)
  249 
  250 -- | Build a tuple value @(x1, .., xn)@.
  251 mkTupleValue :: Pos -> [UTerm] -> UTerm
  252 mkTupleValue p [] = UnitValue p
  253 mkTupleValue _ [x] = x
  254 mkTupleValue p (x:xs) = PairValue (pos x) x (mkTupleValue p xs)
  255 
  256 -- | Build a tuple type @#(x1, .., xn)@.
  257 mkTupleType :: Pos -> [UTerm] -> UTerm
  258 mkTupleType p [] = UnitType p
  259 mkTupleType _ [x] = x
  260 mkTupleType p (x:xs) = PairType (pos x) x (mkTupleType p xs)
  261 
  262 -- | Build a projection @t.i@ of a tuple. NOTE: This function does not
  263 -- work to access the last component in a tuple, since it always
  264 -- generates a @PairLeft@.
  265 mkTupleSelector :: UTerm -> Natural -> UTerm
  266 mkTupleSelector t i
  267   | i == 1    = PairLeft t
  268   | i > 1     = mkTupleSelector (PairRight t) (i - 1)
  269   | otherwise = error "mkTupleSelector: non-positive index"
