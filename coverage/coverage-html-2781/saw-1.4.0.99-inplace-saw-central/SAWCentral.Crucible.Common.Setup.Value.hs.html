   82 nextAllocIndex (AllocIndex n) = AllocIndex (n + 1)

  107 --------------------------------------------------------------------------------
  108 -- ** SetupValue
  109 
  110 -- The following type families are extension fields, which describe what
  111 -- SetupValues are legal for which language backends. Instances for these type
  112 -- families adhere to the following conventions:
  113 --
  114 -- - If a SetupValue constructor is used in a particular backend and there is
  115 --   some data that is /only/ used by that backend, then the corresponding
  116 --   extension field should have a @type instance@ that evaluates to the type of
  117 --   that data. For instance, the @XSetupCast@ instance for @LLVM@ might
  118 --   evaluate to an LLVM @Type@ to represent the type to cast to.
  119 --
  120 -- - If a SetupValue constructor is used in a particular backend but there is
  121 --   no backend-specific information, then the corresponding extension field
  122 --   should have a @type instance@ that evaluates to @()@.
  123 --
  124 -- - If a SetupValue constructor is not used in a particular backend, its
  125 --   corresponding extension field should have a @type instance@ that evaluates
  126 --   to Void. Any code that pattern matches on the constructor should then
  127 --   dispatch on the Void value using the 'absurd' function.
  128 
  129 type family XSetupNull ext
  130 type family XSetupStruct ext
  131 type family XSetupEnum ext
  132 type family XSetupTuple ext
  133 type family XSetupSlice ext
  134 type family XSetupArray ext
  135 type family XSetupElem ext
  136 type family XSetupField ext
  137 type family XSetupGlobal ext
  138 type family XSetupCast ext
  139 type family XSetupUnion ext
  140 type family XSetupGlobalInitializer ext
  141 type family XSetupMux ext
  142 
  143 -- | From the manual: \"The SetupValue type corresponds to values that can occur
  144 -- during symbolic execution, which includes both 'Term' values, pointers, and
  145 -- composite types consisting of either of these (both structures and arrays).\"
  146 data SetupValue ext where
  147   SetupVar    :: AllocIndex -> SetupValue ext
  148   SetupTerm   :: TypedTerm -> SetupValue ext
  149   SetupNull   :: XSetupNull ext -> SetupValue ext
  150   SetupStruct :: XSetupStruct ext -> [SetupValue ext] -> SetupValue ext
  151   SetupArray  :: XSetupArray ext -> [SetupValue ext] -> SetupValue ext
  152   SetupElem   :: XSetupElem ext -> SetupValue ext -> Int -> SetupValue ext
  153   SetupField  :: XSetupField ext -> SetupValue ext -> Text -> SetupValue ext
  154   SetupCast   :: XSetupCast ext -> SetupValue ext -> SetupValue ext
  155   SetupUnion  :: XSetupUnion ext -> SetupValue ext -> Text -> SetupValue ext
  156 
  157   -- | A tuple value. At the moment, this is only ever used for MIR
  158   -- verification.
  159   SetupTuple :: XSetupTuple ext -> [SetupValue ext] -> SetupValue ext
  160   -- | A slice value. At the moment, this is only ever used for MIR
  161   -- verification.
  162   SetupSlice :: XSetupSlice ext -> SetupValue ext
  163   -- | An enumeration value. At the moment, this is only ever used for MIR
  164   -- verification.
  165   SetupEnum :: XSetupEnum ext -> SetupValue ext
  166 
  167   -- | A pointer to a global variable
  168   SetupGlobal :: XSetupGlobal ext -> Text -> SetupValue ext
  169   -- | This represents the value of a global's initializer.
  170   SetupGlobalInitializer ::
  171     XSetupGlobalInitializer ext -> Text -> SetupValue ext
  172   -- | A mux of two 'SetupValue's based on a boolean-typed 'TypedTerm'.
  173   SetupMux ::
  174     XSetupMux ext ->
  175     TypedTerm
  176       {- ^ The conditional expression. Must be of Cryptol type @Bit@. -} ->
  177     SetupValue ext
  178       {- ^ The value to use if the conditional expression is true. -} ->
  179     SetupValue ext
  180       {- ^ The value to use if the conditional expression is false. -} ->
  181     SetupValue ext
  182 
  183 -- | This constraint can be solved for any ext so long as '()', 'Void', and any
  184 --   other types used in the right-hand sides of extension field
  185 --   @type instance@s have the constraint. Unfortunately, GHC can't (yet?)
  186 --   reason over the equations in our closed type family and realize that, so we
  187 --   must explicitly abstract this reasoning out into a type synonym.
  188 type SetupValueHas (c :: Type -> Constraint) ext =
  189   ( c (XSetupNull ext)
  190   , c (XSetupStruct ext)
  191   , c (XSetupEnum ext)
  192   , c (XSetupTuple ext)
  193   , c (XSetupSlice ext)
  194   , c (XSetupArray ext)
  195   , c (XSetupElem ext)
  196   , c (XSetupField ext)
  197   , c (XSetupCast ext)
  198   , c (XSetupUnion ext)
  199   , c (XSetupGlobal ext)
  200   , c (XSetupGlobalInitializer ext)
  201   , c (XSetupMux ext)
  202   )
  203 
  204 deriving instance (SetupValueHas Show ext) => Show (SetupValue ext)
  205 
  206 -- TypedTerm is neither Eq nor Ord
  207 -- deriving instance (SetupValueHas Eq ext) => Eq (SetupValue ext)
  208 -- deriving instance (SetupValueHas Ord ext) => Ord (SetupValue ext)
