   38 -- | The direction of a module port.
   39 data Direction
   40   = DirectionInput
   41   | DirectionOutput
   42   | DirectionInout
   43   deriving (Show, Eq, Ord)
   44 
   45 instance Aeson.FromJSON Direction where
   46   parseJSON (Aeson.String "input") = pure DirectionInput
   47   parseJSON (Aeson.String "output") = pure DirectionOutput
   48   parseJSON (Aeson.String "inout") = pure DirectionInout
   49   parseJSON v = fail $ "Failed to parse direction: " <> show v
   50 
   51 -- | The value of a connection
   52 data Bitrep
   53   = BitrepZero -- ^ Constant zero bit
   54   | BitrepOne -- ^ Constant one bit
   55   | BitrepX -- ^ Undefined bit X
   56   | BitrepZ -- ^ Undefined bit Z
   57   | Bitrep Integer -- ^ The signal bit with the given index
   58   deriving (Show, Eq, Ord)
   59 
   60 instance Aeson.FromJSON Bitrep where
   61   parseJSON (Aeson.String "0") = pure BitrepZero
   62   parseJSON (Aeson.String "1") = pure BitrepOne
   63   parseJSON (Aeson.String "x") = pure BitrepX
   64   parseJSON (Aeson.String "z") = pure BitrepZ
   65   parseJSON vn@Aeson.Number{} = Bitrep <$> Aeson.parseJSON vn
   66   parseJSON v = fail $ "Failed to parse bits: " <> show v

   68 -- ^ A module input/output port.
   69 data Port = Port
   70   { _portDirection :: Direction
   71   , _portBits :: [Bitrep] -- ^ Which bit indices within the module are associated with the port
   72   , _portOffset :: Integer -- currently unused
   73   , _portUpto :: Bool -- currently unused
   74   } deriving (Show, Eq, Ord)
   75 
   76 makeLenses ''Port
   77 
   78 instance Aeson.FromJSON Port where
   79   parseJSON = Aeson.withObject "port" $ \o -> do
   80     _portDirection <- o Aeson..: "direction"
   81     _portBits <- o Aeson..: "bits"
   82     _portOffset <- o Aeson..:? "offset" >>= \case
   83       Just off -> pure off
   84       Nothing -> pure 0
   85     _portUpto <- o Aeson..:? "upto" >>= \case
   86       Just (Aeson.Number 1) -> pure True
   87       _ -> pure False
   88     pure Port{..}

   90 -- | Return 'True' iff a given cell type is a primitive type
   91 cellTypeIsPrimitive :: Text -> Bool
   92 cellTypeIsPrimitive cellType =
   93   case Text.uncons cellType of
   94     Just ('$', _) -> True
   95     _ -> False

  150 -- | All supported cell types. All types are primitives except for
  151 -- 'CellTypeUserType' which represents user-defined submodules
  152 data CellType
  153   = CellTypeNot
  154   | CellTypePos
  155   | CellTypeNeg
  156   | CellTypeAnd
  157   | CellTypeOr
  158   | CellTypeXor
  159   | CellTypeXnor
  160   | CellTypeReduceAnd
  161   | CellTypeReduceOr
  162   | CellTypeReduceXor
  163   | CellTypeReduceXnor
  164   | CellTypeReduceBool
  165   | CellTypeShl
  166   | CellTypeShr
  167   | CellTypeSshl
  168   | CellTypeSshr
  169   | CellTypeShift
  170   | CellTypeShiftx
  171   | CellTypeLt
  172   | CellTypeLe
  173   | CellTypeGt
  174   | CellTypeGe
  175   | CellTypeEq
  176   | CellTypeNe
  177   | CellTypeEqx
  178   | CellTypeNex
  179   | CellTypeAdd
  180   | CellTypeSub
  181   | CellTypeMul
  182   | CellTypeDiv
  183   | CellTypeMod
  184   | CellTypeLogicNot
  185   | CellTypeLogicAnd
  186   | CellTypeLogicOr
  187   | CellTypeMux
  188   | CellTypePmux
  189   | CellTypeBmux
  190   | CellTypeDff
  191   | CellTypeFf
  192   | CellTypeBUF
  193   | CellTypeCheck
  194   | CellTypePrint
  195   | CellTypeScopeinfo
  196   | CellTypeUnsupportedPrimitive Text
  197   | CellTypeUserType Text
  198   deriving (Eq, Ord)
  199 
  200 instance Aeson.FromJSON CellType where
  201   parseJSON (Aeson.String s) =
  202     case s of
  203       "$adff"        -> throw $ YosysErrorUnsupportedFF "$adff"
  204       "$sdff"        -> throw $ YosysErrorUnsupportedFF "$sdff"
  205       "$aldff"       -> throw $ YosysErrorUnsupportedFF "$aldff"
  206       "$dffsr"       -> throw $ YosysErrorUnsupportedFF "$dffsr"
  207       "$dffe"        -> throw $ YosysErrorUnsupportedFF "$dffe"
  208       "$adffe"       -> throw $ YosysErrorUnsupportedFF "$adffe"
  209       "$sdffe"       -> throw $ YosysErrorUnsupportedFF "$sdffe"
  210       "$sdffce"      -> throw $ YosysErrorUnsupportedFF "$sdffce"
  211       "$aldffe"      -> throw $ YosysErrorUnsupportedFF "$aldffe"
  212       "$dffsre"      -> throw $ YosysErrorUnsupportedFF "$dffsre"
  213       _ | cellTypeIsPrimitive s ->
  214           case Map.lookup s textToPrimitiveCellType of
  215             Just cellType -> pure cellType
  216             -- XXX We should probably log a warning when generating CellTypeUnsupportedPrimitive,
  217             -- we can't do that here however.
  218             Nothing -> pure $ CellTypeUnsupportedPrimitive s
  219         | otherwise -> pure $ CellTypeUserType s
  220   parseJSON v = fail $ "Failed to parse cell type: " <> show v
  221 
  222 instance Show CellType where
  223   show ct = Text.unpack $
  224     case ct of
  225       CellTypeUserType ut -> ut
  226       CellTypeUnsupportedPrimitive t -> t
  227       _ | Just t <- Map.lookup ct primitiveCellTypeToText -> t
  228         | otherwise -> panic "Show CellType" ["Unknown primitive cell type"]
  229 
  230 -- | Extract the name from a user-defined submodule 'CellType'
  231 asUserType :: CellType -> Text
  232 asUserType cellType =
  233   case cellType of
  234     CellTypeUserType t -> t
  235     CellTypeUnsupportedPrimitive t -> t
  236     _ ->
  237       panic "asUserType" [
  238           "Expected a user defined type, but got a primitive type: " <>
  239               Text.pack (show cellType)
  240       ]

  242 -- | A cell within an HDL module.
  243 data Cell bs = Cell
  244   { _cellHideName :: Bool -- ^ Whether the cell's name is human-readable
  245   , _cellType :: CellType -- ^ The cell type
  246     -- NB: Yosys's documentation for write_json doesn't impose any restrictions
  247     -- on what type parameter values may take on, so we opt to be as permissive
  248     -- as possible by using Aeson Values. Most of the time, these Values will
  249     -- be strings, but they can also be numbers on occasion (e.g., if you call
  250     -- write_json using the -compat-int flag).
  251   , _cellParameters :: Map Text Aeson.Value -- ^ Metadata parameters
  252   , _cellAttributes :: Maybe Aeson.Value -- currently unused
  253   , _cellPortDirections :: Map Text Direction -- ^ Direction for each cell connection
  254   , _cellConnections :: Map Text bs -- ^ Bitrep for each cell connection
  255   } deriving (Show, Eq, Ord, Functor)
  256 
  257 makeLenses ''Cell
  258 
  259 instance Aeson.FromJSON (Cell [Bitrep]) where
  260   parseJSON = Aeson.withObject "cell" $ \o -> do
  261     _cellHideName <- o Aeson..:? "hide_name" >>= \case
  262       Just (Aeson.Number 1) -> pure True
  263       _ -> pure False
  264     _cellType <- o Aeson..: "type"
  265     _cellParameters <- o Aeson..: "parameters"
  266     _cellAttributes <- o Aeson..:? "attributes"
  267     _cellPortDirections <- o Aeson..: "port_directions"
  268     _cellConnections <- o Aeson..: "connections"
  269     pure Cell{..}

  289 -- | A single HDL module.

  291   { _moduleAttributes :: Maybe Aeson.Value -- currently unused
  292   , _modulePorts :: Map Text Port
  293   , _moduleCells :: Map Text (Cell [Bitrep])
  294   , _moduleNetnames :: Map Text Netname
  295   } deriving (Show, Eq, Ord)

  297 makeLenses ''Module

  299 instance Aeson.FromJSON Module where
  300   parseJSON = Aeson.withObject "module" $ \o -> do
  301     _moduleAttributes <- o Aeson..:? "attributes"
  302     _modulePorts <- Maybe.fromMaybe mempty <$> o Aeson..:? "ports"
  303     _moduleCells <- Maybe.fromMaybe mempty <$> o Aeson..:? "cells"
  304     _moduleNetnames <- Maybe.fromMaybe mempty <$> o Aeson..:? "netnames"
  305     pure Module{..}

  307 -- | A collection of multiple HDL modules (possibly with dependencies on each other).

  317     _yosysCreator <- Maybe.fromMaybe mempty <$> o Aeson..:? "creator"

  321 -- | Read a collection of HDL modules from a file produced by Yosys' write_json command.
  322 loadYosysIR :: FilePath -> IO YosysIR
  323 loadYosysIR p = Aeson.eitherDecodeFileStrict p >>= \case
  324   Left err -> throw . YosysError $ Text.pack err
  325   Right ir -> pure ir

  327 -- | Return the patterns for all of the input ports of a module
  328 moduleInputPorts :: Module -> Map Text [Bitrep]
  329 moduleInputPorts m =
  330   Map.fromList
  331   . Maybe.mapMaybe
  332   ( \(nm, ip) ->
  333       if ip ^. portDirection == DirectionInput || ip ^. portDirection == DirectionInout
  334       then Just (nm, ip ^. portBits)
  335       else Nothing
  336   )
  337   . Map.assocs
  338   $ m ^. modulePorts
  339 
  340 -- | Return the patterns for all of the output ports of a module
  341 moduleOutputPorts :: Module -> Map Text [Bitrep]
  342 moduleOutputPorts m =
  343   Map.fromList
  344   . Maybe.mapMaybe
  345   ( \(nm, ip) ->
  346       if ip ^. portDirection == DirectionOutput || ip ^. portDirection == DirectionInout
  347       then Just (nm, ip ^. portBits)
  348       else Nothing
  349   )
  350   . Map.assocs
  351   $ m ^. modulePorts
  352 
  353 -- | Return the patterns for all of the input connections of a cell
  354 cellInputConnections :: Cell [b] -> Map Text [b]
  355 cellInputConnections c = Map.intersection (c ^. cellConnections) inp
  356   where
  357     inp = Map.filter (\d -> d == DirectionInput || d == DirectionInout) $ c ^. cellPortDirections
  358 
  359 -- | Return the patterns for all of the output connections of a cell
  360 cellOutputConnections :: Ord b => Cell [b] -> Map Text [b]
  361 cellOutputConnections c = Map.intersection (c ^. cellConnections) out
  362   where
  363     out = Map.filter (\d -> d == DirectionOutput || d == DirectionInout) $ c ^. cellPortDirections

  366 cellIsRegister :: Cell bs -> Bool
