  109 -- | A Definition contains an identifier, the type of the definition, and an
  110 -- optional body (axioms and primitives do not have a body)

  112   Def
  113   { defIdent :: Ident
  114   , defVarIndex :: VarIndex

  117   , defBody :: Maybe Term

  126 -- | A specification of a constructor
  127 data Ctor =
  128   forall d params ixs.
  129   Ctor
  130   { ctorName :: !Ident
  131     -- ^ The name of this constructor
  132   , ctorVarIndex :: !VarIndex
  133     -- ^ Unique var index for this constructor
  134   , ctorArgStruct :: CtorArgStruct d params ixs
  135     -- ^ Arguments to the constructor
  136   , ctorDataType :: !(PrimName Term)
  137     -- ^ The datatype this constructor belongs to
  138   , ctorType :: Term
  139     -- ^ Cached type of the constructor, which should always be equal to
  140     --
  141     -- > (p1::P1) -> .. -> (pn::Pn) -> (x1::arg1) -> .. -> (xm::argm) ->
  142     -- >  d p1 .. pn ix1 .. ixk
  143     --
  144     -- where the @pi@ are the 'ctorParams', the @argi@ are the types specified
  145     -- by the 'ctorArgs', and the @ixi@ are the 'ctorDataTypeIndices'. Note that
  146     -- this type should always be top-level, i.e., have no free variables.
  147   , ctorElimTypeFun :: [Term] -> Term -> IO Term
  148     -- ^ Cached function for generating the type of an eliminator for this
  149     -- constructor by passing it a list of parameters and a @p_ret@ function,
  150     -- also known as the "motive function", which itself must have type
  151     --
  152     -- > (x1::ix1) -> .. -> (xm::ixm) -> d p1 .. pn x1 .. xm -> Type i
  153     --
  154     -- where the @ps@ are the parameters and the @ix@s are the indices of
  155     -- datatype @d@
  156   , ctorIotaReduction ::
  157        Term {- ^ recursor term -} ->
  158        Map VarIndex Term {- ^ constructor eliminators -} ->
  159        [Term] {- ^ constructor arguments -} ->
  160        IO Term
  161     -- ^ Cached function for computing the result of one step of iota
  162     --   reduction of the term
  163     --
  164     -- > RecursorApp rec ixs (c params args)
  165     --
  166     --   The arguments to this function are the recusor value, the
  167     --   the map from the recursor that maps constructors to eliminator
  168     --   functions, and the arguments to the constructor.
  169 
  170   , ctorIotaTemplate :: Term
  171     -- ^ Cached term used for computing iota reductions.  It has free variables
  172     --   @rec@, @elim@ and @args@, in that order so that the last @arg@ is the
  173     --   most recently-bound variable with deBruijn index 0.  The @rec@ variable
  174     --   represents the recursor value, @elim@ represents the eliminator function
  175     --   for the constructor, and @args@ represent the arguments to this constructor.
  176   }
  177 
  178 -- | Return the number of parameters of a constructor
  179 ctorNumParams :: Ctor -> Int
  180 ctorNumParams (Ctor { ctorArgStruct = CtorArgStruct {..}}) =
  181   bindingsLength ctorParams
  182 
  183 -- | Return the number of non-parameter arguments of a constructor
  184 ctorNumArgs :: Ctor -> Int
  185 ctorNumArgs (Ctor { ctorArgStruct = CtorArgStruct {..}}) =
  186   bindingsLength ctorArgs
  187 
  188 -- | Compute the ExtCns that uniquely references a constructor
  189 ctorPrimName :: Ctor -> PrimName Term
  190 ctorPrimName ctor = PrimName (ctorVarIndex ctor) (ctorName ctor) (ctorType ctor)
  191 
  192 lift2 :: (a -> b) -> (b -> b -> c) -> a -> a -> c
  193 lift2 f h x y = h (f x) (f y)
  194 
  195 instance Eq Ctor where
  196   (==) = lift2 ctorName (==)
  197 
  198 instance Ord Ctor where
  199   compare = lift2 ctorName compare
  200 
  201 instance Show Ctor where
  202   show = show . ctorName

  207 -- | An inductively-defined datatype
  208 data DataType =
  209   DataType
  210   { dtName :: Ident
  211     -- ^ The name of this datatype
  212   , dtVarIndex :: !VarIndex
  213     -- ^ Unique var index for this data type
  214   , dtParams :: [(LocalName, Term)]
  215     -- ^ The context of parameters of this datatype
  216   , dtIndices :: [(LocalName, Term)]
  217     -- ^ The context of indices of this datatype
  218   , dtSort :: Sort
  219     -- ^ The universe of this datatype
  220   , dtCtors :: [Ctor]
  221     -- ^ The list of constructors of this datatype
  222   , dtType :: Term
  223     -- ^ The cached type of this datatype, which should always be equal to
  224     --
  225     -- > (p1::P1) -> .. -> (pn::Pn) -> (i1::I1) -> .. -> (im::Im) -> 'dtSort'
  226     --
  227     -- where the @pi@ are the 'dtParams' and the @ii@ are the 'dtIndices'. Note
  228     -- that this type should always be top-level, i.e., have no free variables.
  229   }
  230 
  231 -- | Return the number of parameters of a datatype
  232 dtNumParams :: DataType -> Int
  233 dtNumParams dt = length $ dtParams dt
  234 
  235 -- | Return the number of indices of a datatype
  236 dtNumIndices :: DataType -> Int
  237 dtNumIndices dt = length $ dtIndices dt
  238 
  239 -- | Compute the ExtCns that uniquely references a datatype
  240 dtPrimName :: DataType -> PrimName Term
  241 dtPrimName dt = PrimName (dtVarIndex dt) (dtName dt) (dtType dt)
  242 
  243 instance Eq DataType where
  244   (==) = lift2 dtName (==)
  245 
  246 instance Ord DataType where
  247   compare = lift2 dtName compare
  248 
  249 instance Show DataType where
  250   show = show . dtName

  255 -- | Declarations that can occur in a module
  256 data ModuleDecl = TypeDecl DataType
  257                 | DefDecl Def
  258                     -- | Code to inject directly in-stream when
  259                     --   doing translations.
  260                 | InjectCodeDecl
  261                     Text {- ^ Code namespace -}
  262                     Text {- ^ Code to inject -}

  264 -- | The different sorts of things that a 'Text' name can be resolved to
  265 data ResolvedName
  266   = ResolvedCtor Ctor
  267   | ResolvedDataType DataType
  268   | ResolvedDef Def
  269 
  270 -- | Get the 'Ident' for a 'ResolvedName'
  271 resolvedNameIdent :: ResolvedName -> Ident
  272 resolvedNameIdent (ResolvedCtor ctor) = ctorName ctor
  273 resolvedNameIdent (ResolvedDataType dt) = dtName dt
  274 resolvedNameIdent (ResolvedDef d) = defIdent d
  275 
  276 -- | Get the 'VarIndex' for a 'ResolvedName'.
  277 resolvedNameVarIndex :: ResolvedName -> VarIndex
  278 resolvedNameVarIndex r =
  279   case r of
  280     ResolvedCtor ctor -> ctorVarIndex ctor
  281     ResolvedDataType dt -> dtVarIndex dt
  282     ResolvedDef def -> defVarIndex def
  283 
  284 -- | Modules define namespaces of datatypes, constructors, and definitions,
  285 -- i.e., mappings from 'Text' names to these objects. A module is allowed to
  286 -- map a 'Text' name to an object defined in a different module. Modules also
  287 -- keep a record of the top-level declarations and the imports that were used to
  288 -- build them.
  289 data Module = Module {
  290           moduleName    :: !ModuleName
  291         , moduleResolveMap :: !(Map Text ResolvedName)
  292         , moduleRDecls   :: [ModuleDecl] -- ^ All declarations in reverse order they were added.
  293         }
  294 
  295 emptyModule :: ModuleName -> Module
  296 emptyModule nm =
  297   Module { moduleName = nm
  298          , moduleResolveMap = Map.empty
  299          , moduleRDecls = []
  300          }
  301 
  302 
  303 -- | Resolve a 'Text' name in the namespace defined by a 'Module', to either a
  304 -- 'Ctor', 'DataType', or 'Def'
  305 resolveName :: Module -> Text -> Maybe ResolvedName
  306 resolveName m str = Map.lookup str (moduleResolveMap m)
  307 
  308 asResolvedCtor :: ResolvedName -> Maybe Ctor
  309 asResolvedCtor = \case { ResolvedCtor ctor -> Just ctor; _ -> Nothing }
  310 
  311 asResolvedDataType :: ResolvedName -> Maybe DataType
  312 asResolvedDataType = \case { ResolvedDataType d -> Just d; _ -> Nothing }
  313 
  314 asResolvedDef :: ResolvedName -> Maybe Def
  315 asResolvedDef = \case { ResolvedDef d -> Just d; _ -> Nothing }
  316 
  317 -- | Resolve a 'Text' name to a 'Ctor'
  318 findCtor :: Module -> Text -> Maybe Ctor
  319 findCtor m str = resolveName m str >>= asResolvedCtor
  320 
  321 -- | Resolve a 'Text' name to a 'DataType'
  322 findDataType :: Module -> Text -> Maybe DataType
  323 findDataType m str = resolveName m str >>= asResolvedDataType
  324 
  325 -- | Resolve a 'Text' name to a 'Def'
  326 findDef :: Module -> Text -> Maybe Def
  327 findDef m str = resolveName m str >>= asResolvedDef
  328

  330 -- | Insert a 'ResolvedName' into a 'Module', adding a mapping from the 'Text'
  331 -- name of that resolved name to it. Signal an error in the case of a name
  332 -- clash, i.e., an existing binding for the same 'Text' name.
  333 insResolvedName :: Module -> ResolvedName -> Module
  334 insResolvedName m nm =
  335   let str = identBaseName $ resolvedNameIdent nm in
  336   if Map.member str (moduleResolveMap m) then
  337     panic "insResolvedName" [
  338         "inserting duplicate name " <> str <> " into module " <>
  339             moduleNameText (moduleName m)
  340     ]
  341   else
  342     m { moduleResolveMap = Map.insert str nm (moduleResolveMap m) }
  343 
  344 -- | @insImport i m@ returns the module obtained by importing @i@ into @m@,
  345 -- using a predicate to specify which names are imported from @i@ into @m@. In
  346 -- the case of name clashes, an error is signaled.
  347 insImport :: (ResolvedName -> Bool) -> Module -> Module -> Module
  348 insImport name_p i m =
  349   (foldl' insResolvedName m $ Map.elems $
  350    Map.filter name_p (moduleResolveMap i))
  351 
  352 -- | Insert a 'DataType' declaration, along with its 'Ctor's, into a module
  353 insDataType :: Module -> DataType -> Module
  354 insDataType m dt =
  355   foldl' insResolvedName (m { moduleRDecls = TypeDecl dt : moduleRDecls m}) $
  356   (ResolvedDataType dt : map ResolvedCtor (dtCtors dt))
  357 
  358 -- | Insert an "incomplete" datatype, used as part of building up a 'DataType'
  359 -- to typecheck its constructors. This incomplete datatype must have no
  360 -- constructors, and it will not be added to the 'moduleRDecls' list until it is
  361 -- completed by 'completeDataType'.
  362 beginDataType :: Module -> DataType -> Module
  363 beginDataType m dt =
  364    if null (dtCtors dt) then
  365      insResolvedName m (ResolvedDataType dt)
  366    else
  367      panic "insTempDataType" ["attempt to insert a non-empty temporary datatype"]
  368 
  369 -- | Complete a datatype, by adding its constructors
  370 completeDataType :: Module -> Ident -> [Ctor] -> Module
  371 completeDataType m (identBaseName -> str) ctors =
  372   case resolveName m str of
  373     Just (ResolvedDataType dt)
  374       | null (dtCtors dt) ->
  375         let dt' = dt {dtCtors = ctors} in
  376         flip (foldl' insResolvedName) (map ResolvedCtor ctors) $
  377         m { moduleResolveMap =
  378               Map.insert str (ResolvedDataType dt') (moduleResolveMap m),
  379               moduleRDecls = TypeDecl dt' : moduleRDecls m }
  380     Just (ResolvedDataType _) ->
  381       panic "completeDataType" ["datatype already completed: " <> str]
  382     Just _ ->
  383       panic "completeDataType" ["not a datatype: " <> str]
  384     Nothing ->
  385       panic "completeDataType" ["datatype not found: " <> str]
  386

  388 -- | Insert an "injectCode" declaration
  389 insInjectCode :: Module -> Text -> Text -> Module
  390 insInjectCode m ns txt =
  391   m{ moduleRDecls = InjectCodeDecl ns txt : moduleRDecls m }

  393 -- | Insert a definition into a module
  394 insDef :: Module -> Def -> Module
  395 insDef m d =
  396   insResolvedName
  397   (m { moduleRDecls = DefDecl d : moduleRDecls m })
  398   (ResolvedDef d)

  400 -- | Get the resolved names that are local to a module
  401 localResolvedNames :: Module -> [ResolvedName]
  402 localResolvedNames m =
  403   filter ((== moduleName m) . identModule . resolvedNameIdent)
  404   (Map.elems $ moduleResolveMap m)
  405 
  406 -- | Get all data types defined in a module
  407 moduleDataTypes :: Module -> [DataType]
  408 moduleDataTypes =
  409   foldr' (\case { ResolvedDataType dt -> (dt :); _ -> id } ) [] .
  410   localResolvedNames
  411 
  412 -- | Get all constructors defined in a module
  413 moduleCtors :: Module -> [Ctor]
  414 moduleCtors =
  415   foldr' (\case { ResolvedCtor ctor -> (ctor :); _ -> id } ) [] .
  416   localResolvedNames
  417 
  418 -- | Get all definitions defined in a module
  419 moduleDefs :: Module -> [Def]
  420 moduleDefs =
  421   foldr' (\case { ResolvedDef d -> (d :); _ -> id } ) [] .
  422   localResolvedNames
  423 
  424 -- | Get all declarations that are local to a module, i.e., that defined names
  425 -- that were not imported from some other module
  426 moduleDecls :: Module -> [ModuleDecl]
  427 moduleDecls = reverse . moduleRDecls
  428 
  429 -- | Get all local declarations in a module that are marked as primitives
  430 modulePrimitives :: Module -> [Def]
  431 modulePrimitives m =
  432     [ def
  433     | DefDecl def <- moduleDecls m
  434     , defQualifier def == PrimQualifier
  435     ]
  436 
  437 -- | Get all local declarations in a module that are marked as axioms
  438 moduleAxioms :: Module -> [Def]
  439 moduleAxioms m =
  440     [ def
  441     | DefDecl def <- moduleDecls m
  442     , defQualifier def == AxiomQualifier
  443     ]
  444 
  445 -- | Get all local declarations in a module that are not marked as primitives or
  446 -- axioms
  447 moduleActualDefs :: Module -> [Def]
  448 moduleActualDefs m =
  449     [ def
  450     | DefDecl def <- moduleDecls m
  451     , defQualifier def == NoQualifier
  452     ]

  454 -- | The type of mappings from module names to modules
  455 data ModuleMap =
  456   ModuleMap
  457   { mmIdentMap :: !(Map Ident VarIndex)
  458   , mmIndexMap :: !(Map VarIndex ResolvedName)
  459   , mmRDecls :: !(Map ModuleName [ModuleDecl])
  460   }
  461 
  462 emptyModuleMap :: ModuleMap
  463 emptyModuleMap =
  464   ModuleMap
  465   { mmIdentMap = Map.empty
  466   , mmIndexMap = Map.empty
  467   , mmRDecls = Map.empty
  468   }
  469 
  470 -- | Test whether a 'ModuleName' is in a 'ModuleMap'.
  471 moduleIsLoaded :: ModuleName -> ModuleMap -> Bool
  472 moduleIsLoaded mn mm = Map.member mn (mmRDecls mm)
  473 
  474 loadModule :: Module -> ModuleMap -> ModuleMap
  475 loadModule m mm =
  476   ModuleMap
  477   { mmIdentMap = Map.union identMap (mmIdentMap mm)
  478   , mmIndexMap = Map.union indexMap (mmIndexMap mm)
  479   , mmRDecls = Map.insert (moduleName m) (moduleRDecls m) (mmRDecls mm)
  480   }
  481   where
  482     identMap = Map.mapKeys (mkIdent (moduleName m)) (fmap resolvedNameVarIndex (moduleResolveMap m))
  483     indexMap = Map.fromList [ (resolvedNameVarIndex r, r) | r <- Map.elems (moduleResolveMap m) ]
  484 
  485 modifyModule :: ModuleName -> (Module -> Module) -> ModuleMap -> ModuleMap
  486 modifyModule mnm f mm =
  487   case findModule mnm mm of
  488     Just m -> loadModule (f m) mm
  489     Nothing -> mm
  490 
  491 findModule :: ModuleName -> ModuleMap -> Maybe Module
  492 findModule mnm mm =
  493   do decls <- Map.lookup mnm (mmRDecls mm)
  494      let rmap =
  495            Map.mapMaybe (\vi -> Map.lookup vi (mmIndexMap mm)) $
  496            Map.mapKeys identBaseName $
  497            Map.filterWithKey (\i _ -> identModule i == mnm) $
  498            mmIdentMap mm
  499      Just $ Module { moduleName = mnm, moduleResolveMap = rmap, moduleRDecls = decls }
  500 
  501 resolveNameInMap :: ModuleMap -> Ident -> Maybe ResolvedName
  502 resolveNameInMap mm i =
  503   do vi <- Map.lookup i (mmIdentMap mm)
  504      Map.lookup vi (mmIndexMap mm)
  505 
  506 -- | Resolve an 'Ident' to a 'Ctor' in a 'ModuleMap'
  507 findCtorInMap :: ModuleMap -> Ident -> Maybe Ctor
  508 findCtorInMap mm i = resolveNameInMap mm i >>= asResolvedCtor
  509 
  510 -- | Resolve an 'Ident' to a 'DataType' in a 'ModuleMap'
  511 findDataTypeInMap :: ModuleMap -> Ident -> Maybe DataType
  512 findDataTypeInMap mm i = resolveNameInMap mm i >>= asResolvedDataType
  513 
  514 -- | Get all definitions defined in any module in an entire module map. Note
  515 -- that the returned list might have redundancies if a definition is visible /
  516 -- imported in multiple modules in the module map.
  517 allModuleDefs :: ModuleMap -> [Def]
  518 allModuleDefs mm = mapMaybe asResolvedDef (Map.elems (mmIndexMap mm))
  519 
  520 -- | Get all local declarations from all modules in an entire module map
  521 allModuleDecls :: ModuleMap -> [ModuleDecl]
  522 allModuleDecls mm = concat (Map.elems (mmRDecls mm))
  523 
  524 -- | Get all local declarations from all modules in an entire module map that
  525 -- are marked as primitives
  526 allModulePrimitives :: ModuleMap -> [Def]
  527 allModulePrimitives modmap =
  528     [ def
  529     | DefDecl def <- allModuleDecls modmap
  530     , defQualifier def == PrimQualifier
  531     ]
  532 
  533 -- | Get all local declarations from all modules in an entire module map that
  534 -- are marked as axioms
  535 allModuleAxioms :: ModuleMap -> [Def]
  536 allModuleAxioms modmap =
  537     [ def
  538     | DefDecl def <- allModuleDecls modmap
  539     , defQualifier def == AxiomQualifier
  540     ]
  541 
  542 -- | Get all local declarations from all modules in an entire module map that
  543 -- are marked as neither primitives nor axioms
  544 allModuleActualDefs :: ModuleMap -> [Def]
  545 allModuleActualDefs modmap =
  546     [ def
  547     | DefDecl def <- allModuleDecls modmap
  548     , defQualifier def == NoQualifier
  549     ]
  550 
  551 -- | Get all datatypes in all modules in a module map
  552 allModuleDataTypes :: ModuleMap -> [DataType]
  553 allModuleDataTypes mm = mapMaybe asResolvedDataType (Map.elems (mmIndexMap mm))
  554 
  555 -- | Get all constructors in all modules in a module map
  556 allModuleCtors :: ModuleMap -> [Ctor]
  557 allModuleCtors mm = mapMaybe asResolvedCtor (Map.elems (mmIndexMap mm))
