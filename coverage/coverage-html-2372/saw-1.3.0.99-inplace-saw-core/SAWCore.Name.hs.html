   74 -- Module Names ----------------------------------------------------------------
   75 
   76 newtype ModuleName = ModuleName Text
   77   deriving (Eq, Ord, Generic, TH.Lift)
   78 
   79 instance Hashable ModuleName -- automatically derived
   80 
   81 instance Show ModuleName where
   82   show (ModuleName s) = Text.unpack s
   83 
   84 
   85 moduleNameText :: ModuleName -> Text
   86 moduleNameText (ModuleName x) = x
   87 
   88 moduleNamePieces :: ModuleName -> [Text]
   89 moduleNamePieces (ModuleName x) = Text.splitOn "." x
   90 
   91 -- | Create a module name given a list of strings with the top-most
   92 -- module name given first.
   93 mkModuleName :: [Text] -> ModuleName
   94 mkModuleName [] = panic "mkModuleName" ["Empty module name"]
   95 mkModuleName nms = ModuleName $ Text.intercalate "." (reverse nms)
   96 
   97 preludeName :: ModuleName
   98 preludeName = mkModuleName ["Prelude"]

  101 -- Identifiers -----------------------------------------------------------------
  102 
  103 data Ident =
  104   Ident
  105   { identModule :: ModuleName
  106   , identBaseName :: Text
  107   }
  108   deriving (Eq, Ord, Generic)
  109 
  110 instance Hashable Ident -- automatically derived
  111 
  112 instance Show Ident where
  113   show (Ident m s) = shows m ('.' : Text.unpack s)
  114 
  115 identText :: Ident -> Text
  116 identText i = moduleNameText (identModule i) <> Text.pack "." <> identBaseName i
  117 
  118 identPieces :: Ident -> NonEmpty Text
  119 identPieces i =
  120   case moduleNamePieces (identModule i) of
  121     [] -> identBaseName i :| []
  122     (x:xs) -> x :| (xs ++ [identBaseName i])
  123 
  124 identName :: Ident -> String
  125 identName = Text.unpack . identBaseName
  126 
  127 instance Read Ident where
  128   readsPrec _ str =
  129     let (str1, str2) = break (not . isIdChar) str in
  130     [(parseIdent str1, str2)]
  131 
  132 mkIdent :: ModuleName -> Text -> Ident
  133 mkIdent m s = Ident m s
  134 
  135 -- | Make a \"coq-safe\" identifier from a string that might contain
  136 -- non-identifier characters, where we use the SAW core notion of identifier
  137 -- characters as letters, digits, underscore and primes. Any disallowed
  138 -- character is mapped to the string @__xNN@, where @NN@ is the hexadecimal code
  139 -- for that character. Additionally, a SAW core identifier is not allowed to
  140 -- start with a prime, so a leading underscore is added in such a case.
  141 mkSafeIdent :: ModuleName -> String -> Ident
  142 mkSafeIdent _ [] = fromString "_"
  143 mkSafeIdent mnm nm =
  144   let is_safe_char c = isAlphaNum c || c == '_' || c == '\'' in
  145   mkIdent mnm $ Text.pack $
  146   (if nm!!0 == '\'' then ('_' :) else id) $
  147   concatMap
  148   (\c -> if is_safe_char c then [c] else
  149            "__x" ++ showHex (ord c) "")
  150   nm
  151 
  152 -- | Parse a fully qualified identifier.
  153 parseIdent :: String -> Ident
  154 parseIdent s0 =
  155     case reverse (breakEach s0) of
  156       (_:[]) -> panic "parseIdent" ["Empty module name"]
  157       (nm:rMod) -> mkIdent (mkModuleName (reverse rMod)) nm
  158       _ -> panic "parseIdent" ["Bad identifier " <> Text.pack s0]
  159   where breakEach s =
  160           case break (=='.') s of
  161             (h, []) -> [Text.pack h]
  162             (h, _ : r) -> Text.pack h : breakEach r
  163 
  164 instance IsString Ident where
  165   fromString = parseIdent
  166 
  167 isIdent :: String -> Bool
  168 isIdent (c:l) = isAlpha c && all isIdChar l
  169 isIdent [] = False
  170 
  171 -- | Returns true if character can appear in identifier.
  172 isIdChar :: Char -> Bool
  173 isIdChar c = isAlphaNum c || (c == '_') || (c == '\'') || (c == '.')

  177 -- NameInfo
  178 
  179 
  180 -- | Descriptions of the origins of names that may be in scope
  181 data NameInfo
  182   = -- | This name arises from an exported declaration from a module
  183     ModuleIdentifier Ident
  184 
  185   | -- | This name was imported from some other programming language/scope
  186     ImportedName
  187       URI      -- ^ An absolutely-qualified name, which is required to be unique
  188       [Text]   -- ^ A collection of aliases for this name.  Shorter or "less-qualified"
  189                --   aliases should be nearer the front of the list
  190 
  191  deriving (Eq,Ord,Show)
  192 
  193 nameURI :: NameInfo -> URI
  194 nameURI =
  195   \case
  196     ModuleIdentifier i -> moduleIdentToURI i
  197     ImportedName uri _ -> uri
  198 
  199 nameAliases :: NameInfo -> [Text]
  200 nameAliases =
  201   \case
  202     ModuleIdentifier i -> [identBaseName i, identText i]
  203     ImportedName _ aliases -> aliases
  204 
  205 toShortName :: NameInfo -> Text
  206 toShortName (ModuleIdentifier i) = identBaseName i
  207 toShortName (ImportedName uri []) = render uri
  208 toShortName (ImportedName _ (x:_)) = x
  209 
  210 toAbsoluteName :: NameInfo -> Text
  211 toAbsoluteName (ModuleIdentifier i) = identText i
  212 toAbsoluteName (ImportedName uri _) = render uri
  213 
  214 moduleIdentToURI :: Ident -> URI
  215 moduleIdentToURI ident = fromMaybe (panic "moduleIdentToURI" ["Failed to construct ident URI: " <> identText ident]) $
  216   do sch  <- mkScheme "sawcore"
  217      path <- mapM mkPathPiece (identPieces ident)
  218      pure URI
  219        { uriScheme = Just sch
  220        , uriAuthority = Left True -- absolute path
  221        , uriPath   = Just (False, path)
  222        , uriQuery  = []
  223        , uriFragment = Nothing
  224        }

  227 -- External Constants ----------------------------------------------------------
  228 
  229 type VarIndex = Int
  230 
  231 -- | An external constant with a name.
  232 -- Names are not necessarily unique, but the var index should be.
  233 data ExtCns e =
  234   EC
  235   { ecVarIndex :: !VarIndex
  236   , ecName :: !NameInfo
  237   , ecType :: !e
  238   }
  239   deriving (Show, Functor, Foldable, Traversable)
  240 
  241 instance Eq (ExtCns e) where
  242   x == y = ecVarIndex x == ecVarIndex y
  243 
  244 instance Ord (ExtCns e) where
  245   compare x y = compare (ecVarIndex x) (ecVarIndex y)
  246 
  247 instance Hashable (ExtCns e) where
  248   hashWithSalt x ec = hashWithSalt x (ecVarIndex ec)
  249

  251 -- Primitive Names ------------------------------------------------------------
  252 
  253 -- | Names of SAWCore primitives, data types and data type constructors.
  254 data PrimName e =
  255   PrimName
  256   { primVarIndex :: !VarIndex
  257   , primName     :: !Ident
  258   , primType     :: e
  259   }
  260   deriving (Show, Functor, Foldable, Traversable)
  261 
  262 instance Eq (PrimName e) where
  263   x == y = primName x == primName y
  264 
  265 instance Ord (PrimName e) where
  266   compare x y = compare (primName x) (primName y)
  267 
  268 instance Hashable (PrimName e) where
  269   hashWithSalt x pn = hashWithSalt x (primName pn)
  270 
  271 primNameToExtCns :: PrimName e -> ExtCns e
  272 primNameToExtCns (PrimName varIdx nm tp) = EC varIdx (ModuleIdentifier nm) tp
  273 
  274 scFreshNameURI :: Text -> VarIndex -> URI
  275 scFreshNameURI nm i = fromMaybe (panic "scFreshNameURI" ["Failed to construct name URI: <> " <> nm <> "  " <> Text.pack (show i)]) $
  276   do sch <- mkScheme "fresh"
  277      nm' <- mkPathPiece (if Text.null nm then "_" else nm)
  278      i'  <- mkFragment (Text.pack (show i))
  279      pure URI
  280        { uriScheme = Just sch
  281        , uriAuthority = Left False -- relative path
  282        , uriPath   = Just (False, (nm' :| []))
  283        , uriQuery  = []
  284        , uriFragment = Just i'
  285        }

  288 -- Naming Environments ---------------------------------------------------------
  289 
  290 data SAWNamingEnv = SAWNamingEnv
  291   { resolvedNames :: !(Map VarIndex NameInfo)
  292   , absoluteNames :: !(Map URI VarIndex)
  293   , aliasNames    :: !(Map Text (Set VarIndex))
  294   }
  295 -- Invariants: The 'resolvedNames' and 'absoluteNames' maps should be
  296 -- inverses of each other. That is, 'resolvedNames' maps @i@ to @n@ if
  297 -- and only if 'absoluteNames' maps @nameURI n@ to @i@. Also, every
  298 -- 'VarIndex' appearing in 'aliasNames' must be present as a key in
  299 -- 'resolvedNames'.
  300 
  301 emptySAWNamingEnv :: SAWNamingEnv
  302 emptySAWNamingEnv = SAWNamingEnv mempty mempty mempty
  303 
  304 -- | Add a new name entry in a 'SAWNamingEnv'. Returns 'Left' if
  305 -- there is already an entry under the same URI.
  306 registerName :: VarIndex -> NameInfo -> SAWNamingEnv -> Either URI SAWNamingEnv
  307 registerName i nmi env =
  308   case Map.lookup uri (absoluteNames env) of
  309     Just _ -> Left uri
  310     Nothing ->
  311       Right $
  312       SAWNamingEnv
  313       { resolvedNames = Map.insert i nmi (resolvedNames env)
  314       , absoluteNames = Map.insert uri i (absoluteNames env)
  315       , aliasNames    = foldr insertAlias (aliasNames env) aliases
  316       }
  317   where
  318     uri = nameURI nmi
  319     aliases = render uri : nameAliases nmi
  320 
  321     insertAlias :: Text -> Map Text (Set VarIndex) -> Map Text (Set VarIndex)
  322     insertAlias x m = Map.insertWith Set.union x (Set.singleton i) m
  323 
  324 resolveURI :: SAWNamingEnv -> URI -> Maybe VarIndex
  325 resolveURI env uri = Map.lookup uri (absoluteNames env)
  326 
  327 resolveName :: SAWNamingEnv -> Text -> [(VarIndex, NameInfo)]
  328 resolveName env nm =
  329   case Map.lookup nm (aliasNames env) of
  330     Nothing -> []
  331     Just vs -> [ (v, findName v (resolvedNames env)) | v <- Set.toList vs ]
  332   where
  333     findName v m =
  334       case Map.lookup v m of
  335         Just nmi -> nmi
  336         Nothing ->
  337             panic "resolveName" [
  338                 "Unbound VarIndex when resolving name: " <> nm,
  339                 "Index: " <> Text.pack (show v)
  340             ]
  341 
  342 -- | Return the first alias (according to 'nameAliases') that is
  343 -- unambiguous in the naming environment. If there is no unambiguous
  344 -- alias, then return the URI.
  345 bestAlias :: SAWNamingEnv -> NameInfo -> Either URI Text
  346 bestAlias env nmi = go (nameAliases nmi)
  347   where
  348     go [] = Left (nameURI nmi)
  349     go (x : xs) =
  350       case Map.lookup x (aliasNames env) of
  351         Nothing -> go xs
  352         Just vs
  353           | Set.size vs == 1 -> Right x
  354           | otherwise -> go xs
