   10 Stability   : experimental
   11 Portability : non-portable (language extensions)
   12 
   13 This module defines an interface to building SAW core terms in an incrementally
   14 type-checked way, meaning that type-checking is performed as the terms are
   15 built. The interface provides a convenient DSL for building terms in a pure way,
   16 where sub-terms can be freely composed and combined into super-terms without
   17 monadic sequencing or 'IO' computations; the 'IO' computation is only run at the
   18 top level when all the term-building is complete. Users of this interface can
   19 also build binding constructs like lambda- and pi-abstractions without worrying
   20 about deBruijn indices, lifting, and free variables. Instead, a key feature of
   21 this interface is that it uses higher-order abstract syntax for lambda- and
   22 pi-abstractions, meaning that the bodies of these term constructs are specified
   23 as Haskell functions that take in terms for the bound variables. The library
   24 takes care of all the deBruijn indices under the hood.
   25 
   26 To use the 'OpenTerm' API, the caller builds up 'OpenTerm's using a variety of
   27 combinators that mirror the SAW core 'Term' structure. As some useful examples
   28 of 'OpenTerm' operations, 'applyOpenTerm' applies one 'OpenTerm' to another,
   29 'globalOpenTerm' builds an 'OpenTerm' for a global identifier, and
   30 'lambdaOpenTerm' builds a lambda-abstraction. For instance, the SAW core term
   31 
   32 > \ (f : Bool -> Bool) (x : Bool) -> f x
   33 
   34 can be built with the 'OpenTerm' expression
   35 
   36 > let bool = globalOpenTerm "Prelude.Bool" in
   37 > lambdaOpenTerm "f" (arrowOpenTerm bool bool) $ \f ->
   38 > lambdaOpenTerm "x" (globalOpenTerm "Prelude.Bool") $ \x ->
   39 > applyOpenTerm f x
   40 
   41 Existing SAW core 'Term's can be used in 'OpenTerm' by applying 'mkOpenTerm'.
   42 At the top level, 'completeOpenTerm' then "completes" an 'OpenTerm'
   43 by running its underlying 'IO' computation to build and type-check the resulting
   44 SAW core 'Term'.
   45 -}

  100 -- | An open term is represented as a type-checking computation that computes a
  101 -- SAW core term and its type
  102 newtype OpenTerm = OpenTerm { unOpenTerm :: TCM SC.Term }
  103 
  104 -- | \"Complete\" an 'OpenTerm' to a closed term or 'fail' on type-checking
  105 -- error
  106 completeOpenTerm :: SharedContext -> OpenTerm -> IO Term
  107 completeOpenTerm sc (OpenTerm termM) =
  108   either (fail . show) return =<<
  109   runTCM (SC.rawTerm <$> termM) sc
  110 
  111 -- | \"Complete\" an 'OpenTerm' to a closed term for its type
  112 completeOpenTermType :: SharedContext -> OpenTerm -> IO Term
  113 completeOpenTermType sc (OpenTerm termM) =
  114   either (fail . show) return =<<
  115   runTCM (SC.rawType <$> termM) sc

  117 -- | Embed a closed 'Term' into an 'OpenTerm'
  118 mkOpenTerm :: Term -> OpenTerm
  119 mkOpenTerm t = OpenTerm $ typeInferComplete t
  120 
  121 -- | Build an 'OpenTerm' that 'fail's in the underlying monad when completed
  122 failOpenTerm :: String -> OpenTerm
  123 failOpenTerm str = OpenTerm $ fail str
  124 
  125 -- | Bind the result of a type-checking computation in building an 'OpenTerm'.
  126 -- NOTE: this operation should be considered \"unsafe\" because it can create
  127 -- malformed 'OpenTerm's if the result of the 'TCM' computation is used as part
  128 -- of the resulting 'OpenTerm'. For instance, @a@ should not be 'OpenTerm'.
  129 bindTCMOpenTerm :: TCM a -> (a -> OpenTerm) -> OpenTerm
  130 bindTCMOpenTerm m f = OpenTerm (m >>= unOpenTerm . f)
  131 
  132 -- | Bind the result of pretty-printing an 'OpenTerm' while building another
  133 bindPPOpenTerm :: OpenTerm -> (String -> OpenTerm) -> OpenTerm
  134 bindPPOpenTerm (OpenTerm m) f =
  135   OpenTerm $
  136   do t <- SC.rawTerm <$> m
  137      -- XXX: this could use scPrettyTermInCtx (which builds in the call to
  138      -- PPS.render) except that it's slightly different under the covers
  139      -- (in its use of the "global" flag, and it isn't entirely clear what
  140      -- that actually does)
  141      unOpenTerm $ f $ PPS.render PPS.defaultOpts $
  142        ppTermInCtx PPS.defaultOpts [] t
  143 
  144 -- | Return type type of an 'OpenTerm' as an 'OpenTerm
  145 openTermType :: OpenTerm -> OpenTerm
  146 openTermType (OpenTerm m) =
  147   OpenTerm $ do t <- m
  148                 liftTCM SC.scTypeOf t

  150 -- | Build an 'OpenTerm' from a 'FlatTermF'
  151 flatOpenTerm :: FlatTermF OpenTerm -> OpenTerm
  152 flatOpenTerm ftf = OpenTerm $
  153   (sequence (fmap unOpenTerm ftf) >>= typeInferComplete)

  216 -- | Build an 'OpenTerm' for a pair
  217 pairOpenTerm :: OpenTerm -> OpenTerm -> OpenTerm
  218 pairOpenTerm t1 t2 = flatOpenTerm $ PairValue t1 t2

  220 -- | Build an 'OpenTerm' for a pair type
  221 pairTypeOpenTerm :: OpenTerm -> OpenTerm -> OpenTerm
  222 pairTypeOpenTerm t1 t2 = flatOpenTerm $ PairType t1 t2

  224 -- | Build an 'OpenTerm' for the left projection of a pair
  225 pairLeftOpenTerm :: OpenTerm -> OpenTerm
  226 pairLeftOpenTerm t = flatOpenTerm $ PairLeft t

  228 -- | Build an 'OpenTerm' for the right projection of a pair
  229 pairRightOpenTerm :: OpenTerm -> OpenTerm
  230 pairRightOpenTerm t = flatOpenTerm $ PairRight t

  232 -- | Build a right-nested tuple as an 'OpenTerm'
  233 tupleOpenTerm :: [OpenTerm] -> OpenTerm
  234 tupleOpenTerm = foldr pairOpenTerm unitOpenTerm
  235 
  236 -- | Build a right-nested tuple type as an 'OpenTerm'
  237 tupleTypeOpenTerm :: [OpenTerm] -> OpenTerm
  238 tupleTypeOpenTerm = foldr pairTypeOpenTerm unitTypeOpenTerm

  240 -- | Project the @n@th element of a right-nested tuple type
  241 projTupleOpenTerm :: Integer -> OpenTerm -> OpenTerm
  242 projTupleOpenTerm 0 t = pairLeftOpenTerm t
  243 projTupleOpenTerm i t = projTupleOpenTerm (i-1) (pairRightOpenTerm t)
  244 
  245 -- | Build a right-nested tuple as an 'OpenTerm' but without adding a final unit
  246 -- as the right-most element
  247 tupleOpenTerm' :: [OpenTerm] -> OpenTerm
  248 tupleOpenTerm' [] = unitOpenTerm
  249 tupleOpenTerm' ts = foldr1 pairOpenTerm ts
  250 
  251 -- | Build a right-nested tuple type as an 'OpenTerm' but without adding a final
  252 -- unit type as the right-most element
  253 tupleTypeOpenTerm' :: [OpenTerm] -> OpenTerm
  254 tupleTypeOpenTerm' [] = unitTypeOpenTerm
  255 tupleTypeOpenTerm' ts = foldr1 pairTypeOpenTerm ts
  256 
  257 -- | Project the @i@th element from a term of a right-nested tuple term that
  258 -- does not have a final unit type as the right-most type. The first argument is
  259 -- the number of types used to make the tuple type and the second is the index.
  260 projTupleOpenTerm' :: Natural -> Natural -> OpenTerm -> OpenTerm
  261 projTupleOpenTerm' 0 _ _ =
  262   panic "projTupleOpenTerm'" ["Projection of empty tuple!"]
  263 projTupleOpenTerm' 1 0 tup = tup
  264 projTupleOpenTerm' _ 0 tup = pairLeftOpenTerm tup
  265 projTupleOpenTerm' len i tup =
  266   projTupleOpenTerm' (len-1) (i-1) $ pairRightOpenTerm tup
  267 
  268 -- | Build a record value as an 'OpenTerm'
  269 recordOpenTerm :: [(FieldName, OpenTerm)] -> OpenTerm
  270 recordOpenTerm flds_ts =
  271   OpenTerm $ do let (flds,ots) = unzip flds_ts
  272                 ts <- mapM unOpenTerm ots
  273                 typeInferComplete $ RecordValue $ zip flds ts
  274 
  275 -- | Build a record type as an 'OpenTerm'
  276 recordTypeOpenTerm :: [(FieldName, OpenTerm)] -> OpenTerm
  277 recordTypeOpenTerm flds_ts =
  278   OpenTerm $ do let (flds,ots) = unzip flds_ts
  279                 ts <- mapM unOpenTerm ots
  280                 typeInferComplete $ RecordType $ zip flds ts

  282 -- | Project a field from a record
  283 projRecordOpenTerm :: OpenTerm -> FieldName -> OpenTerm
  284 projRecordOpenTerm (OpenTerm m) f =
  285   OpenTerm $ do t <- m
  286                 typeInferComplete $ RecordProj t f

  288 -- | Build an 'OpenTerm' for a constructor applied to its arguments
  289 ctorOpenTerm :: Ident -> [OpenTerm] -> OpenTerm
  290 ctorOpenTerm c all_args = applyOpenTermMulti ctor_open_term all_args
  291   where
  292     ctor_open_term =
  293       OpenTerm $
  294       do maybe_ctor <- liftTCM scFindCtor c
  295          ctor <- case maybe_ctor of
  296                    Just ctor -> pure ctor
  297                    Nothing -> throwTCError $ NoSuchCtor (ModuleIdentifier c)
  298          typeInferComplete (Constant (ctorName ctor) :: TermF Term)
  299 
  300 -- | Build an 'OpenTerm' for a datatype applied to its arguments
  301 dataTypeOpenTerm :: Ident -> [OpenTerm] -> OpenTerm
  302 dataTypeOpenTerm d all_args = applyOpenTermMulti dt_open_term all_args
  303   where
  304     dt_open_term =
  305       OpenTerm $
  306       do maybe_dt <- liftTCM scFindDataType d
  307          dt <- case maybe_dt of
  308                  Just dt -> pure dt
  309                  Nothing -> throwTCError $ NoSuchDataType (ModuleIdentifier d)
  310          typeInferComplete (Constant (dtName dt) :: TermF Term)
  311 
  312 -- | Build an 'OpenTerm' for a global name with a definition

  323 -- | Build an 'OpenTerm' for a named variable.

  327 -- | Apply an 'OpenTerm' to another
  328 applyOpenTerm :: OpenTerm -> OpenTerm -> OpenTerm
  329 applyOpenTerm (OpenTerm f) (OpenTerm arg) =
  330   OpenTerm ((App <$> f <*> arg) >>= typeInferComplete)
  331 
  332 -- | Apply an 'OpenTerm' to 0 or more arguments
  333 applyOpenTermMulti :: OpenTerm -> [OpenTerm] -> OpenTerm
  334 applyOpenTermMulti = foldl applyOpenTerm
  335 
  336 -- | Apply a named global to 0 or more arguments
  337 applyGlobalOpenTerm :: Ident -> [OpenTerm] -> OpenTerm
  338 applyGlobalOpenTerm ident = applyOpenTermMulti (globalOpenTerm ident)

  340 -- | Compute the output type of applying a function of a given type to an
  341 -- argument. That is, given @tp@ and @arg@, compute the type of applying any @f@
  342 -- of type @tp@ to @arg@.
  343 applyPiOpenTerm :: OpenTerm -> OpenTerm -> OpenTerm
  344 applyPiOpenTerm (OpenTerm m_f) (OpenTerm m_arg) =
  345   OpenTerm $
  346   do f <- m_f
  347      arg <- m_arg
  348      ret <- applyPiTyped (NotFuncTypeInApp f arg) (SC.rawTerm f) arg
  349      typeInferComplete ret
  350 
  351 -- | Get the argument type of a function type, 'fail'ing if the input term is
  352 -- not a function type
  353 piArgOpenTerm :: OpenTerm -> OpenTerm
  354 piArgOpenTerm (OpenTerm m) =
  355   OpenTerm $ m >>= \case
  356   (unwrapTermF . SC.rawTerm -> Pi _ tp _) -> typeInferComplete tp
  357   t ->
  358     fail ("piArgOpenTerm: not a pi type: " ++
  359           scPrettyTermInCtx PPS.defaultOpts [] (SC.rawTerm t))
  360 
  361 -- | Build a lambda abstraction as an 'OpenTerm'
  362 lambdaOpenTerm :: LocalName -> OpenTerm -> (OpenTerm -> OpenTerm) -> OpenTerm
  363 lambdaOpenTerm x (OpenTerm tpM) body_f = OpenTerm $
  364   do tp <- tpM
  365      vn <- liftTCM scFreshVarName x
  366      var <- typeInferComplete $ Variable vn tp
  367      body <- unOpenTerm (body_f (OpenTerm (pure var)))
  368      typeInferComplete $ Lambda vn tp body
  369 
  370 -- | Build a nested sequence of lambda abstractions as an 'OpenTerm'
  371 lambdaOpenTermMulti :: [(LocalName, OpenTerm)] -> ([OpenTerm] -> OpenTerm) ->
  372                        OpenTerm
  373 lambdaOpenTermMulti xs_tps body_f =
  374   foldr (\(x,tp) rest_f xs ->
  375           lambdaOpenTerm x tp (rest_f . (:xs))) (body_f . reverse) xs_tps []
  376 
  377 -- | Build a Pi abstraction as an 'OpenTerm'
  378 piOpenTerm :: LocalName -> OpenTerm -> (OpenTerm -> OpenTerm) -> OpenTerm
  379 piOpenTerm x (OpenTerm tpM) body_f = OpenTerm $
  380   do tp <- tpM
  381      nm <- liftTCM scFreshVarName x
  382      var <- typeInferComplete $ Variable nm tp
  383      body <- unOpenTerm (body_f (OpenTerm (pure var)))
  384      typeInferComplete $ Pi nm tp body
  385 
  386 -- | Build a non-dependent function type.
  387 arrowOpenTerm :: LocalName -> OpenTerm -> OpenTerm -> OpenTerm
  388 arrowOpenTerm x tp body = piOpenTerm x tp (const body)
  389 
  390 -- | Build a nested sequence of Pi abstractions as an 'OpenTerm'
  391 piOpenTermMulti :: [(LocalName, OpenTerm)] -> ([OpenTerm] -> OpenTerm) ->
  392                    OpenTerm
  393 piOpenTermMulti xs_tps body_f =
  394   foldr (\(x,tp) rest_f xs ->
  395           piOpenTerm x tp (rest_f . (:xs))) (body_f . reverse) xs_tps []

  397 -- | Build a let expression as an 'OpenTerm'. This is equivalent to
  398 -- > 'applyOpenTerm' ('lambdaOpenTerm' x tp body) rhs
  399 letOpenTerm :: LocalName -> OpenTerm -> OpenTerm -> (OpenTerm -> OpenTerm) ->
  400                OpenTerm
  401 letOpenTerm x tp rhs body_f = applyOpenTerm (lambdaOpenTerm x tp body_f) rhs
  402 
  403 -- | Build a let expression as an 'OpenTerm'. This is equivalent to
  404 -- > 'applyOpenTerm' ('lambdaOpenTerm' x tp body) rhs
  405 sawLetOpenTerm :: LocalName -> OpenTerm -> OpenTerm -> OpenTerm ->
  406                   (OpenTerm -> OpenTerm) -> OpenTerm
  407 sawLetOpenTerm x tp tp_ret rhs body_f =
  408   applyOpenTermMulti (globalOpenTerm "Prelude.sawLet")
  409   [tp, tp_ret, rhs, lambdaOpenTerm x tp body_f]
  410 
  411 -- | Build a bitvector type with the given length
  412 bitvectorTypeOpenTerm :: OpenTerm -> OpenTerm
  413 bitvectorTypeOpenTerm w =
  414   applyGlobalOpenTerm "Prelude.Vec" [w, globalOpenTerm "Prelude.Bool"]

  416 -- | Build a SAW core term for a list with the given element type
  417 listOpenTerm :: OpenTerm -> [OpenTerm] -> OpenTerm
  418 listOpenTerm tp elems =
  419   foldr (\x l -> ctorOpenTerm "Prelude.Cons" [tp, x, l])
  420   (ctorOpenTerm "Prelude.Nil" [tp]) elems

  422 -- | Build the type @Either a b@ from types @a@ and @b@

  429 class OpenTermLike t where
  430   -- | Convert an 'OpenTerm' to a @t@
  431   openTermLike :: OpenTerm -> t
  432   -- | Get the type of a @t@
  433   typeOfTermLike :: t -> t
  434   -- | Build a @t@ from a 'FlatTermF'
  435   flatTermLike :: FlatTermF t -> t
  436   -- | Apply a @t@ to another @t@
  437   applyTermLike :: t -> t -> t
  438   -- | Build a lambda abstraction as a @t@
  439   lambdaTermLike :: LocalName -> t -> (t -> t) -> t
  440   -- | Build a pi abstraction as a @t@
  441   piTermLike :: LocalName -> t -> (t -> t) -> t
  442   -- | Build a @t@ for a constructor applied to its arguments
  443   ctorTermLike :: Ident -> [t] -> t
  444   -- | Build a @t@ for a datatype applied to its arguments
  445   dataTypeTermLike :: Ident -> [t] -> t
  446 
  447 instance OpenTermLike OpenTerm where
  448   openTermLike = id
  449   typeOfTermLike = openTermType
  450   flatTermLike = flatOpenTerm
  451   applyTermLike = applyOpenTerm
  452   lambdaTermLike = lambdaOpenTerm
  453   piTermLike = piOpenTerm
  454   ctorTermLike = ctorOpenTerm
  455   dataTypeTermLike = dataTypeOpenTerm
  456 
  457 -- Lift an OpenTermLike instance from t to functions from some type a to t,
  458 -- where the OpenTermLike methods pass the same input a argument to all subterms
  459 instance OpenTermLike t => OpenTermLike (a -> t) where
  460   openTermLike t = const $ openTermLike t
  461   typeOfTermLike t = \x -> typeOfTermLike (t x)
  462   flatTermLike ftf = \x -> flatTermLike (fmap ($ x) ftf)
  463   applyTermLike f arg = \x -> applyTermLike (f x) (arg x)
  464   lambdaTermLike nm tp bodyF =
  465     \x -> lambdaTermLike nm (tp x) (\y -> bodyF (const y) x)
  466   piTermLike nm tp bodyF =
  467     \x -> piTermLike nm (tp x) (\y -> bodyF (const y) x)
  468   ctorTermLike c args = \x -> ctorTermLike c (map ($ x) args)
  469   dataTypeTermLike d args = \x -> dataTypeTermLike d (map ($ x) args)
  470 
  471 -- This is the same as the function instance above
  472 instance OpenTermLike t => OpenTermLike (Reader r t) where
  473   openTermLike t = reader $ openTermLike t
  474   typeOfTermLike t = reader $ typeOfTermLike $ runReader t
  475   flatTermLike ftf = reader $ flatTermLike $ fmap runReader ftf
  476   applyTermLike f arg = reader $ applyTermLike (runReader f) (runReader arg)
  477   lambdaTermLike x tp body =
  478     reader $ lambdaTermLike x (runReader tp) (runReader . body . reader)
  479   piTermLike x tp body =
  480     reader $ piTermLike x (runReader tp) (runReader . body . reader)
  481   ctorTermLike c args = reader $ ctorTermLike c $ map runReader args
  482   dataTypeTermLike d args = reader $ dataTypeTermLike d $ map runReader args
  483

  540 boolTypeTermLike :: OpenTermLike t => t
  541 boolTypeTermLike = globalTermLike "Prelude.Bool"

  543 -- | Build an term for an array literal
  544 arrayValueTermLike :: OpenTermLike t => t -> [t] -> t
  545 arrayValueTermLike tp elems =
  546   flatTermLike $ ArrayValue tp $ V.fromList elems

  550 bvLitTermLike bits =

  555 vectorTypeTermLike n a = applyGlobalTermLike "Prelude.Vec" [n,a]

  563 -- | Build a term for a pair
  564 pairTermLike :: OpenTermLike t => t -> t -> t
  565 pairTermLike t1 t2 = flatTermLike $ PairValue t1 t2
  566 
  567 -- | Build a term for a pair type
  568 pairTypeTermLike :: OpenTermLike t => t -> t -> t
  569 pairTypeTermLike t1 t2 = flatTermLike $ PairType t1 t2

  571 -- | Build a term for the left projection of a pair
  572 pairLeftTermLike :: OpenTermLike t => t -> t
  573 pairLeftTermLike t = flatTermLike $ PairLeft t

  576 pairRightTermLike :: OpenTermLike t => t -> t
  577 pairRightTermLike t = flatTermLike $ PairRight t

  579 -- | Build a right-nested tuple as a term
  580 tupleTermLike :: OpenTermLike t => [t] -> t
  581 tupleTermLike = foldr pairTermLike unitTermLike
  582 
  583 -- | Build a right-nested tuple type as a term
  584 tupleTypeTermLike :: OpenTermLike t => [t] -> t
  585 tupleTypeTermLike = foldr pairTypeTermLike unitTypeTermLike

  587 -- | Project the @n@th element of a right-nested tuple type
  588 projTupleTermLike :: OpenTermLike t => Integer -> t -> t
  589 projTupleTermLike 0 t = pairLeftTermLike t
  590 projTupleTermLike i t = projTupleTermLike (i-1) (pairRightTermLike t)

  592 -- | Build a let expression as a term. This is equivalent to
  593 -- > 'applyTermLike' ('lambdaTermLike' x tp body) rhs
  594 letTermLike :: OpenTermLike t => LocalName -> t -> t -> (t -> t) -> t
  595 letTermLike x tp rhs body_f = applyTermLike (lambdaTermLike x tp body_f) rhs
  596 
  597 -- | Build a let expression as a term using the @sawLet@ combinator. This
  598 -- is equivalent to the term @sawLet tp tp_ret rhs (\ (x : tp) -> body_f)@
  599 sawLetTermLike :: OpenTermLike t => LocalName -> t -> t -> t -> (t -> t) -> t
  600 sawLetTermLike x tp tp_ret rhs body_f =
  601   applyTermLikeMulti (globalTermLike "Prelude.sawLet")
  602   [tp, tp_ret, rhs, lambdaTermLike x tp body_f]
