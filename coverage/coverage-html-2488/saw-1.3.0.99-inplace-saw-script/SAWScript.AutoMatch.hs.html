   62 -- | The interactive procedure for matching two given function declarations
   63 matchingProcedure :: Decl -> Decl -> Interaction (Assignments, Mappings)
   64 matchingProcedure left right =
   65    execMatchDecls (left, right) . sequence_ $
   66       [ initialInfo
   67       , checkReturnTypeCompat
   68       , checkSignatureCompat
   69       , processExactMatches
   70       , checkNameClashes
   71       , matchInteractively ]
   72 
   73 -- | Print information about the two declarations about to be matched
   74 initialInfo :: ArgMatch ()
   75 initialInfo = do
   76    (left, right) <- ask
   77    info (Just "Comparing") $ show left ++
   78           "\n           " ++ show right
   79 
   80 -- | Fail matching if the two declarations differ in return type
   81 checkReturnTypeCompat :: ArgMatch ()
   82 checkReturnTypeCompat = do
   83    (left, right) <- ask
   84    when (declType left /= declType right) $
   85       failure True $
   86          "The declarations (" ++
   87          Text.unpack (declName left) ++ " : ... " ++ show (declType left) ++
   88          ") and (" ++
   89          Text.unpack (declName right) ++ " : ... " ++ show (declType right) ++
   90          ") do not match in return type, and so cannot be reconciled."
   91 
   92 -- | Fail matching if the two declarations have argument types which aren't permutations of each other
   93 checkSignatureCompat :: ArgMatch ()
   94 checkSignatureCompat = do
   95    (left, right) <- ask
   96    r <- get
   97    when ((uncurry (/=) . both (fmap Set.size . typeBins)) r) $ do
   98       warning $
   99          "The signatures for '" ++ Text.unpack (declName left) ++
  100          "' and '" ++ Text.unpack (declName right) ++
  101          "' cannot be aligned by permutation."
  102       confirmOrQuit "Proceed with matching anyway?"
  103 
  104 -- | Automatically equate all arguments which match exactly in type and name
  105 processExactMatches :: ArgMatch ()
  106 processExactMatches = do
  107    exactMatches <- findExactMatches <$> get
  108    forM_ exactMatches $ uncurry matchArgs
  109    separator ThinSep
  110    when (not . null $ exactMatches) $ do
  111       info Nothing "Exact matches:"
  112       bulleted . for exactMatches $
  113          \((arg1, i1), (_arg2, i2)) ->
  114             "(" ++ show arg1 ++ ")" ++ " at " ++
  115             formatIndex i1 ++ corresponds ++ formatIndex i2
  116       separator ThinSep
  117 
  118 -- | Interactively warn when any arguments of *different* types have the same name -- this is likely a user error
  119 checkNameClashes :: ArgMatch ()
  120 checkNameClashes = do
  121    sharedNames <- getIntersect (Map.keys . nameLocs)
  122    fencepostForM_ (separator ThinSep) sharedNames $ \name -> do
  123       ((li, lt), (ri, rt)) <- (both $ assertJust . lookupName name) <$> get -- assertJust *just*ified
  124       warning $                                                             -- b/c name is definitely in map
  125          "Arguments " ++ formatIndexedArg False name lt li ++
  126          corresponds  ++ formatIndexedArg False name rt ri ++
  127          " are named identically, but have differing types."
  128       confirmOrQuit "Proceed with matching by considering them distinct?"
  129 
  130 -- | Walk through the remaining unmatched declarations asking the user to align them
  131 matchInteractively :: ArgMatch ()
  132 matchInteractively = do
  133    sharedTypes <- getIntersect (Map.keys . typeBins)
  134    forM_ sharedTypes $ \typ ->
  135       both (argsForType typ) <$> get >>=
  136          (fix $ \loop -> \case
  137              ([],_) -> separator ThinSep >> return ()
  138              (_,[]) -> separator ThinSep >> return ()
  139              (leftArgs@((ln, li):_), rightArgs) -> do
  140                 separator ThinSep
  141                 offerChoice
  142                    ("Match " ++ formatIndexedArg True ln typ li ++ corresponds ++ "___:")
  143                    ((for rightArgs $ \(rn, ri) ->
  144                       (formatIndexedArg False rn typ ri,
  145                        do matchArgs (Arg ln typ, li) (Arg rn typ, ri)
  146                           loop (delete (ln, li) leftArgs,
  147                                 delete (rn, ri) rightArgs)))
  148                      ++ [("Quit", userQuit)]))
  149    where
  150       argsForType typ =
  151          sortBy (comparing snd) . Set.toList . assertJust . lookupType typ
  152          --                    *just*ified use ^^^^^^^^^^ because typ is
  153          --                    definitely in map when we call this function
  154 
  155 -- | Project from the (s,s) state a list of the intersection of return values
  156 --   from the function on both parts of the state
  157 getIntersect :: (Ord b) => (s -> [b]) -> Match r w (s,s) [b]
  158 getIntersect f =
  159    Set.toList . uncurry Set.intersection . (both $ Set.fromList . f) <$> get
  160 
  161 -- | Compute exact matches (name & type) between two ArgMappings:
  162 findExactMatches :: (ArgMapping, ArgMapping) -> [((Arg, Int), (Arg, Int))]
  163 findExactMatches (leftMapping, rightMapping) =
  164    concat $
  165       for sharedTypes $ \typ ->
  166          for (sharedNamesWithType typ) $ \name ->
  167               let (li, _) = assertJust $ lookupName name leftMapping  -- assertJust is (pun intended) *just*ified here,
  168                   (ri, _) = assertJust $ lookupName name rightMapping -- b/c we have already confirmed the keys exist
  169               in ((Arg name typ, li), (Arg name typ, ri))
  170    where
  171       sharedTypes =
  172          Set.toList $
  173             Set.intersection (Map.keysSet (typeBins leftMapping))
  174                              (Map.keysSet (typeBins rightMapping))
  175       namesWithType typ =
  176          fromMaybe Set.empty
  177          . fmap (Set.fromList . map fst . Set.toList)
  178          . lookupType typ
  179       sharedNamesWithType typ =
  180          Set.toList $
  181             Set.intersection (namesWithType typ leftMapping)
  182                              (namesWithType typ rightMapping)

  184 -- | The interactive procedure to match declarations between two whole modules
  185 matchModules :: [Decl] -> [Decl] -> Interaction [(Decl, Decl, Assignments)]
  186 matchModules leftModule rightModule =
  187    runMatchModules leftModule rightModule $ do
  188       sharedSigs <- gets $ uncurry sharedKeys
  189       forM_ sharedSigs $ \sig -> do
  190          declsByApproxName <- gets (both $ tabulateBy (approximateName . declName) .
  191                                            Set.toList . assertJust . Map.lookup sig)
  192          let matchingNames = uncurry sharedKeys $ declsByApproxName
  193          fencepostForM_ (separator ThickSep) matchingNames $ \name -> do
  194             case both (Set.toList . assertJust . Map.lookup name) declsByApproxName of
  195                ([leftDecl], [rightDecl]) -> do
  196                   (assigns, leftovers) <- lift . lift . lift . lift $
  197                                           matchingProcedure leftDecl rightDecl
  198                   if uncurry (&&) (both isEmptyArgMapping leftovers)
  199                      then matchDecls leftDecl rightDecl assigns
  200                      else return ()
  201                _ -> return ()
  202 
  203             -- TODO: provide interactive matching of remaining functions binned in signature
  204 
  205          -- TODO: more inexact name matching
  206 
  207       -- Report unmatched declarations
  208       (unselectedL, unselectedR) <- gets (both $ concatMap Set.toList . Map.elems)
  209       when (not . null $ unselectedL) $ do
  210          warning "Did not find matches for left-side declarations:"
  211          bulleted $ map show unselectedL
  212       separator ThinSep
  213       when (not . null $ unselectedR) $ do
  214          warning "Did not find matches for right-side declarations:"
  215          bulleted $ map show unselectedR

  217 -- | Names are approximately equal if they are equal modulo
  218 --   spaces, underscores, hyphens, and capitalization
  219 approximateName :: Name -> Name
  220 approximateName =
  221    Text.filter (not . (`Text.elem` "_- ")) . Text.toCaseFold

  226 -- | A filepath tagged with its language of origin
  227 data TaggedSourceFile =
  228    TaggedSourceFile { sourceLanguage :: SourceLanguage
  229                     , sourceFilePath :: FilePath } deriving (Eq, Ord, Show)

  231 -- | Try to recognize the source language of a filepath, and tag it if we know how
  232 autoTagSourceFile :: FilePath -> Either String TaggedSourceFile
  233 autoTagSourceFile path = case takeExtension path of
  234    ".cry"   -> Right $ TaggedSourceFile Cryptol path
  235    ".bc"    -> Right $ TaggedSourceFile LLVM    path
  236    ".class" -> Right $ TaggedSourceFile JVM     path
  237    ext      -> Left ext
  238 
  239 -- | Try to tag both files and fail with a descriptive error if either cannot be recognized
  240 autoTagSourceFiles :: FilePath -> FilePath -> Either String (TaggedSourceFile, TaggedSourceFile)
  241 autoTagSourceFiles leftPath rightPath =
  242    case both autoTagSourceFile (leftPath, rightPath) of
  243       (Left leftExt, Left rightExt) ->
  244          Left $
  245             "Could handle neither file extension " ++
  246             leftExt ++ " of " ++ leftPath ++ " nor " ++
  247             rightExt ++ " of " ++ rightPath
  248       (Left leftExt, Right _) ->
  249          Left $
  250             "Couldn't handle file extension " ++
  251             leftExt ++ " of left-hand file " ++ leftPath
  252       (Right _, Left rightExt) ->
  253          Left $
  254             "Couldn't handle file extension " ++
  255             rightExt ++ " of right-hand file " ++ rightPath
  256       (Right leftSource, Right rightSource) ->
  257          return (leftSource, rightSource)

  259 -- | Take two tagged source files and load them up to generate an interaction which matches the modules together
  260 autoMatchFiles :: TaggedSourceFile -> TaggedSourceFile -> TopLevel (Interaction MatchResult)
  261 autoMatchFiles leftSource@(TaggedSourceFile _ leftPath) rightSource@(TaggedSourceFile _ rightPath) = do
  262    leftModInteract  <- loadDecls leftSource
  263    rightModInteract <- loadDecls rightSource
  264    return . frame (separator SuperThickSep) $ do
  265       info Nothing $ "Aligning declarations between " ++ leftPath ++ corresponds ++ rightPath
  266       separator ThickSep
  267       maybe (return $ MatchResult [] Nothing False False)
  268             (processResults leftSource rightSource <=< uncurry matchModules) =<<
  269          pairA <$> leftModInteract <*> rightModInteract

  271 -- | Load the declarations from the given file, dispatching on the source language to determine how to do this
  272 loadDecls :: TaggedSourceFile -> TopLevel (Interaction (Maybe [Decl]))
  273 loadDecls (TaggedSourceFile lang path) = do
  274    sc <- getSharedContext
  275    AIGProxy proxy <- getProxy
  276    case lang of
  277       Cryptol -> io $ getDeclsCryptol path
  278       LLVM    -> do
  279         Some m <- llvm_load_module path
  280         io $ getDeclsLLVM proxy sc m
  281       JVM     -> loadJavaClassTopLevel (dropExtension path) >>= io . getDeclsJVM
  282    where
  283       loadJavaClassTopLevel cls = do
  284          javaCodebase <- getJavaCodebase
  285          let pos = FileOnlyPos path
  286          io . lookupClass javaCodebase pos . mkClassName . dotsToSlashes $ cls

  288 -- A description of the result of matching: some generated SAWScript, and some flags determining what to do now
  289 data MatchResult =
  290    MatchResult { generatedScript   :: [SAWScript.Stmt]
  291                , afterMatchSave    :: Maybe FilePath
  292                , afterMatchPrint   :: Bool
  293                , afterMatchExecute :: Bool }

  295 -- | The type of the line-by-line interpreter, which needs to be passed in another module to avoid circularity
  296 type StmtInterpreter = TopLevelRO -> TopLevelRW -> [SAWScript.Stmt] -> IO ()
  297 
  298 -- | How to interpret a MatchResult to the TopLevel monad
  299 actAfterMatch :: StmtInterpreter -> MatchResult -> TopLevel ()
  300 actAfterMatch interpretStmts MatchResult{..} =
  301    let renderedScript = SAWScript.prettyWholeModule generatedScript
  302    in do io . awhen afterMatchSave $ \file ->
  303                  withFile file WriteMode $ \handle ->
  304                      hPutDoc handle renderedScript
  305          io . when afterMatchPrint $ putDoc renderedScript
  306          when afterMatchExecute $ interpret generatedScript
  307    where
  308       interpret script = do
  309          ro <- getTopLevelRO
  310          rw <- getTopLevelRW
  311          io . void $ interpretStmts ro rw script

  313 -- | The top level entry-point for AutoMatch
  314 --   Requires a StmtInterpreter to be passed as input to resolve import cycle
  315 autoMatch :: StmtInterpreter -> FilePath -> FilePath -> TopLevel ()
  316 autoMatch interpreter leftFile rightFile =
  317    autoTagSourceFiles leftFile rightFile &
  318       (either (io . putStrLn) $
  319          uncurry autoMatchFiles >=> io . interactIO >=> actAfterMatch interpreter)

  321 -- | Just a bunch of SAWScript statments as output and a supply of unique identifiers
  322 type ScriptWriter s tp = WriterT [SAWScript.Stmt] (ReaderT (NonceGenerator (ST s) tp) (ST s))
  323 
  324 runScriptWriter :: (forall s tp. ScriptWriter s tp a)
  325                 -> (a, [SAWScript.Stmt])
  326 runScriptWriter c =
  327   ST.runST $ withSTNonceGenerator $ runReaderT (runWriterT c)

  329 execScriptWriter :: (forall s tp. ScriptWriter s tp a)
  330                  -> [SAWScript.Stmt]
  331 execScriptWriter c = snd (runScriptWriter c)

  409       equivalenceTheorem :: (Text -> Text) -> SAWScript.LName -> SAWScript.LName -> Assignments -> ScriptWriter s tp (SAWScript.LName)
  410       equivalenceTheorem prefix leftFunction rightFunction assigns = do
  411          theoremName <- newNameWith prefix
  412          (leftArgs, rightArgs) <-
  413             fmap (both (map snd . sortBy (comparing fst)) . unzip) .
  414             forM assigns $
  415                \((Arg leftName _, leftIndex), (Arg rightName _, rightIndex)) -> do
  416                   name <- newNameWith (nameCenter (leftName <> "_" <> rightName))
  417                   return ((leftIndex, name), (rightIndex, name))
  418          returning theoremName . tell $
  419             [SAWScript.StmtBind triggerPos (SAWScript.PVar triggerPos theoremName Nothing) .
  420                 SAWScript.Code . locate .
  421                    Text.pack . show . Cryptol.ppPrec 0 .
  422                       cryptolAbstractNamesSAW leftArgs .
  423                          cryptolApplyFunction (Cryptol.EParens . Cryptol.EVar . nameCryptolFromSAW . locate $ "==") $
  424                             [ cryptolApplyFunctionSAW leftFunction  leftArgs
  425                             , cryptolApplyFunctionSAW rightFunction rightArgs ]]
  426 
  427       nameCryptolFromSAW :: SAWScript.LName -> Cryptol.PName
  428       nameCryptolFromSAW = Cryptol.mkUnqual . Cryptol.mkIdent . SAWScript.getVal

  430       cryptolAbstractNamesSAW :: [SAWScript.LName] -> Cryptol.Expr Cryptol.PName -> Cryptol.Expr Cryptol.PName
  431       cryptolAbstractNamesSAW names expr =
  432          Cryptol.EFun Cryptol.emptyFunDesc
  433          (for names $ Cryptol.PVar . cryptolLocate . SAWScript.getVal) expr

  435       cryptolApplyFunctionSAW :: SAWScript.LName -> [SAWScript.LName] -> Cryptol.Expr Cryptol.PName
  436       cryptolApplyFunctionSAW function args =
  437          cryptolApplyFunction (Cryptol.EVar $ nameCryptolFromSAW function)
  438                               (map (Cryptol.EVar . nameCryptolFromSAW) args)

  440       cryptolApplyFunction :: Cryptol.Expr Cryptol.PName -> [Cryptol.Expr Cryptol.PName] -> Cryptol.Expr Cryptol.PName
  441       cryptolApplyFunction function args =
  442          foldl Cryptol.EApp function args

  479       -- Here's the template for the output script:
  480       script :: [SAWScript.Stmt]
  481       script = execScriptWriter $ do
  482         leftModule  <- loadFile (nameLeft  "module") leftLang  leftFile
  483         rightModule <- loadFile (nameRight "module") rightLang rightFile
  484         forM_ matchings $ \(leftDecl, rightDecl, assigns) -> do
  485           leftFunction  <- extractFunction (nameLeft  (declName leftDecl))  leftLang  (declName leftDecl)  leftModule
  486           rightFunction <- extractFunction (nameRight (declName rightDecl)) rightLang (declName rightDecl) rightModule
  487           theorem       <- equivalenceTheorem (nameCenter "theorem") leftFunction rightFunction assigns
  488           let leftFile' = Text.pack leftFile
  489               rightFile' = Text.pack rightFile
  490           printString $ "Proving '" <> declName leftDecl  <> "' (" <> leftFile'  <> ")" <>
  491                             " == '" <> declName rightDecl <> "' (" <> rightFile' <> ")"
  492           prove theorem
