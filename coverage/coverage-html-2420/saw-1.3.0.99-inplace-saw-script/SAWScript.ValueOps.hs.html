  131 isVUnit :: Value -> Bool
  132 isVUnit (VTuple []) = True
  133 isVUnit _ = False

  135 indexValue :: Value -> Value -> Value
  136 indexValue (VArray vs) (VInteger x)
  137     | i < length vs = vs !! i
  138     | otherwise = error "array index out of bounds"
  139     where i = fromInteger x
  140 indexValue _ _ = error "indexValue"
  141 
  142 lookupValue :: Value -> Text -> Value
  143 lookupValue (VRecord vm) name =
  144     case M.lookup name vm of
  145       Nothing -> error $ "no such record field: " ++ Text.unpack name
  146       Just x -> x
  147 lookupValue _ _ = error "lookupValue"
  148 
  149 tupleLookupValue :: Value -> Integer -> Value
  150 tupleLookupValue (VTuple vs) i
  151   | 0 <= i && fromIntegral i < length vs = vs !! fromIntegral i
  152   | otherwise = error $ "no such tuple index: " ++ show i
  153 tupleLookupValue _ _ = error "tupleLookupValue"

  155 toplevelSubshell :: Value
  156 toplevelSubshell = VLambda $ \_ ->
  157   do m <- roSubshell <$> ask
  158      env <- getLocalEnv
  159      return (VTopLevel (toValue <$> withLocalEnv env m))

  161 proofScriptSubshell :: Value
  162 proofScriptSubshell = VLambda $ \_ ->
  163   do m <- roProofSubshell <$> ask
  164      env <- getLocalEnv
  165      return (VProofScript (toValue <$> withLocalEnvProof env m))

  167 applyValue :: Value -> Value -> TopLevel Value
  168 applyValue (VLambda f) x = f x
  169 applyValue _ _ = throwTopLevel "applyValue"
  170 
  171 thenValue :: SS.Pos -> Value -> Value -> Value
  172 thenValue pos v1 v2 = VBind pos v1 (VLambda (const (return v2)))
  173 
  174 bindValue :: SS.Pos -> Value -> Value -> TopLevel Value
  175 bindValue pos v1 v2 = return (VBind pos v1 v2)
  176 
  177 forValue :: [Value] -> Value -> TopLevel Value
  178 forValue [] _ = return $ VReturn (VArray [])
  179 forValue (x : xs) f =
  180   do m1 <- applyValue f x
  181      m2 <- forValue xs f
  182      bindValue (SS.PosInternal "<for>") m1 (VLambda $ \v1 ->
  183        bindValue (SS.PosInternal "<for>") m2 (VLambda $ \v2 ->
  184          return $ VReturn (VArray (v1 : fromValue v2))))

  192 -- | A version of 'Control.Exception.bracket' specialized to 'TopLevel'. We
  193 -- can't use 'Control.Monad.Catch.bracket' because it requires 'TopLevel' to
  194 -- implement 'Control.Monad.Catch.MonadMask', which it can't do.
  195 bracketTopLevel :: TopLevel a -> (a -> TopLevel b) -> (a -> TopLevel c) -> TopLevel c
  196 bracketTopLevel acquire release action =
  197   do  resource <- acquire
  198       try (action resource) >>= \case
  199         Left (bad :: X.SomeException) -> release resource >> throwM bad
  200         Right good -> release resource >> pure good

  202 combineRW :: TopLevelCheckpoint -> TopLevelRW -> IO TopLevelRW
  203 combineRW (TopLevelCheckpoint chk scc) rw =
  204   do cenv' <- CEnv.combineCryptolEnv (rwCryptol chk) (rwCryptol rw)
  205      sc' <- restoreSharedContext scc (rwSharedContext rw)
  206      return chk{ rwCryptol = cenv'
  207                , rwSharedContext = sc'
  208                }

  210 -- | Represents the mutable state of the TopLevel monad
  211 --   that can later be restored.
  212 data TopLevelCheckpoint =
  213   TopLevelCheckpoint
  214     TopLevelRW
  215     SharedContextCheckpoint
  216 
  217 makeCheckpoint :: TopLevelRW -> IO TopLevelCheckpoint
  218 makeCheckpoint rw =
  219   do scc <- checkpointSharedContext (rwSharedContext rw)
  220      return (TopLevelCheckpoint rw scc)
  221 
  222 restoreCheckpoint :: TopLevelCheckpoint -> TopLevel ()
  223 restoreCheckpoint chk =
  224   do rw <- getTopLevelRW
  225      rw' <- io (combineRW chk rw)
  226      putTopLevelRW rw'

  228 -- | Capture the current state of the TopLevel monad
  229 --   and return an action that, if invoked, resets
  230 --   the state back to that point.
  231 checkpoint :: TopLevel (() -> TopLevel ())
  232 checkpoint = TopLevel_ $
  233   do chk <- liftIO . makeCheckpoint =<< get
  234      return $ \_ ->
  235        do printOutLnTop Info "Restoring state from checkpoint"
  236           restoreCheckpoint chk
  237 
  238 -- | Capture the current proof state and return an
  239 --   action that, if invoked, resets the state back to that point.
  240 proof_checkpoint :: ProofScript (() -> ProofScript ())
  241 proof_checkpoint =
  242   do ps <- get
  243      return $ \_ ->
  244        do scriptTopLevel (printOutLnTop Info "Restoring proof state from checkpoint")
  245           put ps

  247 withLocalEnv :: LocalEnv -> TopLevel a -> TopLevel a
  248 withLocalEnv env m = do
  249   prevEnv <- gets rwLocalEnv
  250   modify (\rw -> rw { rwLocalEnv = env })
  251   result <- m
  252   modify (\rw -> rw { rwLocalEnv = prevEnv })
  253   return result
  254 
  255 withLocalEnvProof :: LocalEnv -> ProofScript a -> ProofScript a
  256 withLocalEnvProof env (ProofScript m) = do
  257   ProofScript (underExceptT (underStateT (withLocalEnv env)) m)
  258 
  259 withOptions :: (Options -> Options) -> TopLevel a -> TopLevel a
  260 withOptions f (TopLevel_ m) =
  261   TopLevel_ (local (\x -> x {roOptions = f (roOptions x)}) m)

  263 -- | Implement stack tracing by adding error handlers that rethrow
  264 -- user errors, prepended with the given string.
  265 withStackTraceFrame :: String -> Value -> Value
  266 withStackTraceFrame str val =
  267   let doTopLevel :: TopLevel a -> TopLevel a
  268       doTopLevel action = do
  269         trace <- gets rwStackTrace
  270         modify (\rw -> rw { rwStackTrace = str : trace })
  271         result <- action
  272         modify (\rw -> rw { rwStackTrace = trace })
  273         return result
  274       doProofScript :: ProofScript a -> ProofScript a
  275       doProofScript (ProofScript m) =
  276         let m' =
  277               underExceptT (underStateT doTopLevel) m
  278         in
  279         ProofScript m'
  280       doLLVM :: LLVMCrucibleSetupM a -> LLVMCrucibleSetupM a
  281       doLLVM (LLVMCrucibleSetupM m) =
  282         LLVMCrucibleSetupM (underReaderT (underStateT doTopLevel) m)
  283       doJVM :: JVMSetupM a -> JVMSetupM a
  284       doJVM (JVMSetupM m) =
  285         JVMSetupM (underReaderT (underStateT doTopLevel) m)
  286       doMIR :: MIRSetupM a -> MIRSetupM a
  287       doMIR (MIRSetupM m) =
  288         MIRSetupM (underReaderT (underStateT doTopLevel) m)
  289   in
  290   case val of
  291     VLambda f ->
  292       let wrap x =
  293             withStackTraceFrame str `fmap` doTopLevel (f x)
  294       in
  295       VLambda wrap
  296     VTopLevel m ->
  297       let m' =
  298             withStackTraceFrame str `fmap` doTopLevel m
  299       in
  300       VTopLevel m'
  301     VProofScript m ->
  302       let m' =
  303             withStackTraceFrame str `fmap` doProofScript m
  304       in
  305       VProofScript m'
  306     VBind pos v1 v2 ->
  307       let v1' = withStackTraceFrame str v1
  308           v2' = withStackTraceFrame str v2
  309       in
  310       VBind pos v1' v2'
  311     VLLVMCrucibleSetup m ->
  312       let m' =
  313             withStackTraceFrame str `fmap` doLLVM m
  314       in
  315       VLLVMCrucibleSetup m'
  316     VJVMSetup m ->
  317       let m' =
  318             withStackTraceFrame str `fmap` doJVM m
  319       in
  320       VJVMSetup m'
  321     VMIRSetup m ->
  322       let m' =
  323             withStackTraceFrame str `fmap` doMIR m
  324       in
  325       VMIRSetup m'
  326     _ ->
  327       val

  329 toValueCase :: (FromValue b) =>
  330                (b -> Value -> Value -> TopLevel Value)
  331             -> Value
  332 toValueCase prim =
  333   VLambda $ \b -> return $
  334   VLambda $ \v1 -> return $
  335   VLambda $ \v2 ->
  336   prim (fromValue b) v1 v2

  338 caseProofResultPrim ::
  339   ProofResult ->
  340   Value {- ^ valid case -} ->
  341   Value {- ^ invalid/unknown case -} ->
  342   TopLevel Value
  343 caseProofResultPrim pr vValid vInvalid = do
  344   sc <- getSharedContext
  345   case pr of
  346     ValidProof _ thm ->
  347       applyValue vValid (toValue thm)
  348     InvalidProof _ pairs _pst -> do
  349       let fov = FOVTuple (map snd pairs)
  350       tt <- io $ typedTermOfFirstOrderValue sc fov
  351       applyValue vInvalid (toValue tt)
  352     UnfinishedProof _ -> do
  353       tt <- io $ typedTermOfFirstOrderValue sc (FOVTuple [])
  354       applyValue vInvalid (toValue tt)
  355 
  356 caseSatResultPrim ::
  357   SatResult ->
  358   Value {- ^ unsat case -} ->
  359   Value {- ^ sat/unknown case -} ->
  360   TopLevel Value
  361 caseSatResultPrim sr vUnsat vSat = do
  362   sc <- getSharedContext
  363   case sr of
  364     Unsat _ -> return vUnsat
  365     Sat _ pairs -> do
  366       let fov = FOVTuple (map snd pairs)
  367       tt <- io $ typedTermOfFirstOrderValue sc fov
  368       applyValue vSat (toValue tt)
  369     SatUnknown -> do
  370       let fov = FOVTuple []
  371       tt <- io $ typedTermOfFirstOrderValue sc fov
  372       applyValue vSat (toValue tt)

  441 instance FromValue a => FromValue (TopLevel a) where
  442     fromValue (VTopLevel action) = fmap fromValue action
  443     fromValue (VReturn v) = return (fromValue v)
  444     fromValue (VBind pos m1 v2) = do
  445       savepos <- pushPosition pos
  446       v1 <- fromValue m1
  447       popPosition savepos
  448       m2 <- applyValue v2 v1
  449       fromValue m2

  452 instance IsValue a => IsValue (ProofScript a) where

  455 instance FromValue a => FromValue (ProofScript a) where
  456     fromValue (VProofScript m) = fmap fromValue m
  457     -- Inject top-level computations automatically into proof scripts.
  458     -- This should really only possible in interactive subshell mode; otherwise
  459     --  the type system should keep this from happening.
  460     fromValue (VTopLevel m) = ProofScript (lift (lift (fmap fromValue m)))
  461     fromValue (VReturn v) = return (fromValue v)
  462     fromValue (VBind pos m1 v2) = ProofScript $ do
  463       savepos <- lift $ lift $ pushPosition pos
  464       v1 <- unProofScript (fromValue m1)
  465       lift $ lift $ popPosition savepos
  466       m2 <- lift $ lift $ applyValue v2 v1
  467       unProofScript (fromValue m2)
  468     fromValue _ = error "fromValue ProofScript"
