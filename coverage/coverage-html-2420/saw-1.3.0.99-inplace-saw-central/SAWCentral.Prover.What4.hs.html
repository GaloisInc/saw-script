   47 setupWhat4_sym ::
   48   Bool ->
   49   Bool ->
   50   IO (B.ExprBuilder
   51       GlobalNonceGenerator
   52       St
   53       (B.Flags B.FloatReal))
   54 setupWhat4_sym hashConsing what4PushMuxOps =
   55   do -- TODO: get rid of GlobalNonceGenerator ???
   56      sym <- B.newExprBuilder B.FloatRealRepr St globalNonceGenerator
   57      let cfg = getConfiguration sym
   58      cacheTermsSetting <- getOptionSetting B.cacheTerms cfg
   59      _ <- setOpt cacheTermsSetting hashConsing
   60      pushMuxOpsSetting <- getOptionSetting B.pushMuxOpsOption cfg
   61      _ <- setOpt pushMuxOpsSetting what4PushMuxOps
   62      return sym

   64 what4Theories ::
   65   Set VarIndex ->
   66   Bool ->
   67   Sequent ->
   68   TopLevel [Text]
   69 what4Theories unintSet hashConsing goal = do
   70   sc <- getSharedContext
   71   what4PushMuxOps <- gets rwWhat4PushMuxOps
   72   io $ do
   73      sym <- setupWhat4_sym hashConsing what4PushMuxOps
   74      satq <- sequentToSATQuery sc unintSet goal
   75      (_varMap, lits) <- W.w4Solve sym sc satq
   76      let pf lit = (predicateVarInfo lit)^.problemFeatures
   77      return (nub (concatMap evalTheories (map pf lits)))
   78 
   79 evalTheories :: ProblemFeatures -> [Text]
   80 evalTheories pf = [ nm | (nm,f) <- xs, hasProblemFeature pf f ]
   81  where
   82   xs = [ ("LinearArithmetic", useLinearArithmetic)
   83        , ("NonlinearArithmetic", useNonlinearArithmetic)
   84        , ("TranscendentalFunctions", useComputableReals)
   85        , ("Integers", useIntegerArithmetic)
   86        , ("Bitvectors", useBitvectors)
   87        , ("ExistsForall", useExistForall)
   88        , ("FirstOrderQuantifiers", useQuantifiers)
   89        , ("Arrays", useSymbolicArrays)
   90        , ("Structs", useStructs)
   91        , ("Strings", useStrings)
   92        , ("FloatingPoint", useFloatingPoint)
   93        ]

   95 proveWhat4_sym ::
   96   SolverAdapter St ->
   97   Bool ->
   98   SATQuery ->
   99   TopLevel (Maybe CEX, Text)
  100 proveWhat4_sym solver hashConsing satq = do
  101   sc <- getSharedContext
  102   what4PushMuxOps <- gets rwWhat4PushMuxOps
  103   io $ do
  104      sym <- setupWhat4_sym hashConsing what4PushMuxOps
  105      proveWhat4_solver solver sym sc satq (return ())

  107 proveExportWhat4_sym ::
  108   SolverAdapter St ->
  109   Bool ->
  110   FilePath ->
  111   SATQuery ->
  112   TopLevel (Maybe CEX, Text)
  113 proveExportWhat4_sym solver hashConsing outFilePath satq = do
  114   sc <- getSharedContext
  115   what4PushMuxOps <- gets rwWhat4PushMuxOps
  116   io $ do
  117      sym <- setupWhat4_sym hashConsing what4PushMuxOps
  118 
  119      -- Write smt out
  120      (_, _, lits, solver_name) <- setupWhat4_solver solver sym sc satq
  121      withFile outFilePath WriteMode $ \handle ->
  122        solver_adapter_write_smt2 solver sym handle lits
  123 
  124      -- Assume unsat
  125      return (Nothing, solver_name)

  180 setupWhat4_solver :: forall st t ff.
  181   SolverAdapter st   {- ^ Which solver to use -} ->
  182   B.ExprBuilder t st ff {- ^ The glorious sym -}  ->
  183   SharedContext      {- ^ Context for working with terms -} ->
  184   SATQuery           {- ^ The query to be proved/checked. -} ->
  185   IO ( [ExtCns Term]
  186      , [W.Labeler (B.ExprBuilder t st ff)]
  187      , [Pred (B.ExprBuilder t st ff)]
  188      , Text)
  189 setupWhat4_solver solver sym sc satq =
  190   do
  191      -- symbolically evaluate
  192      let varList  = Map.toList (satVariables satq)
  193      let argNames = map fst varList
  194      (varMap, lits) <- W.w4Solve sym sc satq
  195      let bvs = map (fst . snd) varMap
  196 
  197      extendConfig (solver_adapter_config_options solver)
  198                   (getConfiguration sym)
  199 
  200      let solver_name = "W4 ->" <> (Text.pack $ solver_adapter_name solver)
  201 
  202      return (argNames, bvs, lits, solver_name)

  205 -- | Check the validity of a proposition using What4.
  206 proveWhat4_solver :: forall st t ff.
  207   SolverAdapter st   {- ^ Which solver to use -} ->
  208   B.ExprBuilder t st ff {- ^ The glorious sym -}  ->
  209   SharedContext      {- ^ Context for working with terms -} ->
  210   SATQuery           {- ^ The query to be proved/checked. -} ->
  211   IO ()              {- ^ Extra setup actions -} ->
  212   IO (Maybe CEX, Text)
  213   -- ^ (example/counter-example, solver statistics)
  214 proveWhat4_solver solver sym sc satq extraSetup =
  215   do
  216      (argNames, bvs, lits, solver_name) <- setupWhat4_solver solver sym sc satq
  217      extraSetup
  218 
  219      -- log to stdout
  220      let logger _ str = putStrLn str
  221      let logData = defaultLogData { logCallbackVerbose = logger
  222                                   , logReason = "SAW proof" }
  223 
  224      -- run solver
  225      solver_adapter_check_sat solver sym logData lits $ \ r -> case r of
  226          Sat (gndEvalFcn,_) -> do
  227            mvals <- mapM (getValues @(B.ExprBuilder t st ff) gndEvalFcn)
  228                          (zip bvs argNames)
  229            return (Just mvals, solver_name) where
  230 
  231          Unsat _ -> return (Nothing, solver_name)
  232 
  233          Unknown -> fail "Prover returned Unknown"

  237 getValues :: forall sym gt a. (SymExpr sym ~ B.Expr gt) => GroundEvalFn gt ->
  238   (W.Labeler sym, a) -> IO (a, FirstOrderValue)
  239 getValues f (labeler, orig) = do
  240   fov <- W.getLabelValues f labeler
  241   return (orig, fov)

  244 -- | For debugging
  245 printValue :: (B.ExprBuilder t st ff) -> GroundEvalFn t ->
  246   (Maybe (W.TypedExpr (B.ExprBuilder t st ff)), String) -> IO ()
  247 printValue _ _ (Nothing, _) = return ()
  248 printValue _ f (Just (W.TypedExpr (ty :: BaseTypeRepr ty) (bv :: B.Expr t ty)), orig) = do
  249   gv <- groundEval f @ty bv
  250   putStr $ orig ++ "=?"
  251   print (groundToFOV ty gv)
