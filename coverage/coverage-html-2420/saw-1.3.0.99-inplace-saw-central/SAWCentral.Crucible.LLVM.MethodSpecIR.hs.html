  206 ppPointsTo :: LLVMPointsTo arch -> PPL.Doc ann
  207 ppPointsTo (LLVMPointsTo _md cond ptr val) =
  208   MS.ppSetupValue ptr
  209   PPL.<+> PPL.pretty "points to"
  210   PPL.<+> PPL.pretty val
  211   PPL.<+> maybe PPL.emptyDoc (\tt -> PPL.pretty "if" PPL.<+> ppTypedTerm tt) cond
  212 ppPointsTo (LLVMPointsToBitfield _md ptr fieldName val) =
  213   MS.ppSetupValue ptr <> PPL.pretty ("." ++ fieldName)
  214   PPL.<+> PPL.pretty "points to (bitfield)"
  215   PPL.<+> MS.ppSetupValue val

  241 data SetupError
  242   = InvalidReturnType L.Type
  243   | InvalidArgTypes [L.Type]
  244 
  245 ppSetupError :: SetupError -> PPL.Doc ann
  246 ppSetupError (InvalidReturnType t) =
  247   PPL.pretty "Can't lift return type" PPL.<+>
  248   PPL.viaShow (Crucible.LLVM.ppType t) PPL.<+>
  249   PPL.pretty "to a Crucible type."
  250 ppSetupError (InvalidArgTypes ts) =
  251   PPL.pretty "Can't lift argument types " PPL.<+>
  252   PPL.encloseSep PPL.lparen PPL.rparen PPL.comma (map (PPL.viaShow . Crucible.LLVM.ppType) ts) PPL.<+>
  253   PPL.pretty "to Crucible types."

  255 resolveArgs ::
  256   (?lc :: CL.TypeContext) =>
  257   [L.Type] ->
  258   Either SetupError [CL.MemType]
  259 resolveArgs args = do
  260   -- TODO: make sure we resolve aliases
  261   let mtys = traverse CL.liftMemType args
  262   -- TODO: should the error message be propagated?
  263   either (\_ -> Left (InvalidArgTypes args)) Right mtys
  264 
  265 resolveRetTy ::
  266   (?lc :: CL.TypeContext) =>
  267   L.Type ->
  268   Either SetupError (Maybe CL.MemType)
  269 resolveRetTy ty = do
  270   -- TODO: make sure we resolve aliases
  271   let ret = CL.liftRetType ty
  272   -- TODO: should the error message be propagated?
  273   either (\_ -> Left (InvalidReturnType ty)) Right ret

  275 initialDefCrucibleMethodSpecIR ::
  276   (?lc :: CL.TypeContext) =>
  277   LLVMModule arch ->
  278   L.Define ->
  279   ProgramLoc ->
  280   Maybe Text ->
  281   Either SetupError (MS.CrucibleMethodSpecIR (LLVM arch))
  282 initialDefCrucibleMethodSpecIR llvmModule def loc parent = do
  283   args <- resolveArgs (L.typedType <$> L.defArgs def)
  284   ret <- resolveRetTy (L.defRetType def)
  285   let L.Symbol nm = L.defName def
  286   let methId = LLVMMethodId (Text.pack nm) parent
  287   return $ MS.makeCrucibleMethodSpecIR methId args ret loc llvmModule
  288 
  289 initialDeclCrucibleMethodSpecIR ::
  290   (?lc :: CL.TypeContext) =>
  291   LLVMModule arch ->
  292   L.Declare ->
  293   ProgramLoc ->
  294   Maybe Text ->
  295   Either SetupError (MS.CrucibleMethodSpecIR (LLVM arch))
  296 initialDeclCrucibleMethodSpecIR llvmModule dec loc parent = do
  297   args <- resolveArgs (L.decArgs dec)
  298   ret <- resolveRetTy (L.decRetType dec)
  299   let L.Symbol nm = L.decName dec
  300   let methId = LLVMMethodId (Text.pack nm) parent
  301   return $ MS.makeCrucibleMethodSpecIR methId args ret loc llvmModule

  329 -- *** AllLLVM
  330 
  331 -- | Universal/polymorphic quantification over an 'LLVMArch'
  332 --
  333 -- The following type synonym and associated constructor/destructor are
  334 -- equivalent to this definition:
  335 -- @
  336 -- data AllLLVM t =
  337 --   MkAllLLVM { getAllLLVM :: forall arch. t (LLVM arch) }
  338 -- @
  339 -- But they preserve the instances from 'All' and 'Compose'.
  340 type AllLLVM t = All (Compose t LLVM)
  341 
  342 -- This doesn't work :(
  343 --
  344 -- pattern AllLLVM :: (forall arch. t (LLVM arch)) -> AllLLVM t
  345 -- pattern AllLLVM x = All (Compose x)
  346 
  347 mkAllLLVM :: forall t. (forall arch. t (LLVM arch)) -> AllLLVM t
  348 mkAllLLVM x = All (Compose x)
  349 
  350 getAllLLVM :: forall t. AllLLVM t -> (forall arch. t (LLVM arch))
  351 getAllLLVM (All (Compose x)) = x
  352 
  353 -- Constructors for 'SetupValue' which are architecture-polymorphic
  354 
  355 anySetupTerm :: TypedTerm -> AllLLVM MS.SetupValue
  356 anySetupTerm typedTerm = mkAllLLVM (MS.SetupTerm typedTerm)
  357 
  358 anySetupArray :: [AllLLVM MS.SetupValue] -> AllLLVM MS.SetupValue
  359 anySetupArray vals = mkAllLLVM (MS.SetupArray () $ map (\a -> getAllLLVM a) vals)
  360 
  361 anySetupStruct :: Bool -> [AllLLVM MS.SetupValue] -> AllLLVM MS.SetupValue
  362 anySetupStruct b vals = mkAllLLVM (MS.SetupStruct b $ map (\a -> getAllLLVM a) vals)
  363 
  364 anySetupElem :: AllLLVM MS.SetupValue -> Int -> AllLLVM MS.SetupValue
  365 anySetupElem val idx = mkAllLLVM (MS.SetupElem () (getAllLLVM val) idx)
  366 
  367 anySetupCast :: AllLLVM MS.SetupValue -> L.Type -> AllLLVM MS.SetupValue
  368 anySetupCast val ty = mkAllLLVM (MS.SetupCast ty (getAllLLVM val))
  369 
  370 anySetupField :: AllLLVM MS.SetupValue -> Text -> AllLLVM MS.SetupValue
  371 anySetupField val field = mkAllLLVM (MS.SetupField () (getAllLLVM val) field)
  372 
  373 anySetupUnion :: AllLLVM MS.SetupValue -> Text -> AllLLVM MS.SetupValue
  374 anySetupUnion val uname = mkAllLLVM (MS.SetupUnion () (getAllLLVM val) uname)
  375 
  376 anySetupNull :: AllLLVM MS.SetupValue
  377 anySetupNull = mkAllLLVM (MS.SetupNull ())
  378 
  379 anySetupGlobal :: Text -> AllLLVM MS.SetupValue
  380 anySetupGlobal globalName = mkAllLLVM (MS.SetupGlobal () globalName)
  381 
  382 anySetupGlobalInitializer :: Text -> AllLLVM MS.SetupValue
  383 anySetupGlobalInitializer globalName =
  384   mkAllLLVM (MS.SetupGlobalInitializer () globalName)

  387 -- *** SomeLLVM
  388 
  389 -- | Existential quantification over an 'LLVMArch'
  390 --
  391 -- The following type synonym and associated constructor/destructor are
  392 -- equivalent to this definition:
  393 -- @
  394 -- data SomeLLVM t = forall arch. MkSomeLLVM (t (LLVM arch))
  395 -- @
  396 -- But they preserve the instances from 'Some' and 'Compose'.
  397 type SomeLLVM t = Some (Compose t LLVM)
  398 
  399 pattern SomeLLVM :: t (LLVM arch) -> SomeLLVM t
  400 pattern SomeLLVM x = Some (Compose x)
  401 {-# COMPLETE SomeLLVM #-}
  402 
  403 mkSomeLLVM :: t (LLVM arch) -> SomeLLVM t
  404 mkSomeLLVM x = Some (Compose x)
  405 
  406 getSomeLLVM :: forall t. (forall arch. t (LLVM arch)) -> AllLLVM t
  407 getSomeLLVM x = All (Compose x)

  410 -- *** ResolvedState

  412 -- | Record the initialization of the pointer represented by the given
  413 -- SetupValue.
  414 markResolved ::
  415   MS.SetupValue (LLVM arch) ->
  416   ResolvedPath {-^ path within this object (if any) -} ->
  417   LLVMResolvedState ->
  418   LLVMResolvedState
  419 markResolved val0 path0 rs = go path0 val0
  420   where
  421     go path val =
  422       case val of
  423         MS.SetupVar n         -> rs & rsAllocs %~ Map.alter (ins path) n
  424         MS.SetupGlobal _ name -> rs & rsGlobals %~ Map.alter (ins path) name
  425         MS.SetupElem _ v idx  -> go (ResolvedElem idx : path) v
  426         MS.SetupField _ v fld -> go (ResolvedField fld : path) v
  427         MS.SetupCast tp v     -> go (ResolvedCast tp : path) v
  428         _                     -> rs
  429 
  430     ins path Nothing = Just [path]
  431     ins path (Just paths) = Just (path : paths)
  432 
  433 -- | Test whether the pointer represented by the given SetupValue has
  434 -- been initialized already.
  435 testResolved ::
  436   MS.SetupValue (LLVM arch) ->
  437   ResolvedPath {-^ path within this object (if any) -} ->
  438   LLVMResolvedState ->
  439   Bool
  440 testResolved val0 path0 rs = go path0 val0
  441   where
  442     go path val =
  443       case val of
  444         MS.SetupVar n         -> test path (Map.lookup n (_rsAllocs rs))
  445         MS.SetupGlobal _ c    -> test path (Map.lookup c (_rsGlobals rs))
  446         MS.SetupElem _ v idx  -> go (ResolvedElem idx : path) v
  447         MS.SetupField _ v fld -> go (ResolvedField fld : path) v
  448         MS.SetupCast tp v     -> go (ResolvedCast tp : path) v
  449         _                     -> False
  450 
  451     test _ Nothing = False
  452     test path (Just paths) = any (overlap path) paths
  453 
  454     overlap (x : xs) (y : ys) = x == y && overlap xs ys
  455     overlap [] _ = True
  456     overlap _ [] = True
