   50 {- | SAWCore Values. Values are parameterized by the /name/ of an
   51 instantiation.
   52 The concrete parameters to use are computed from the name using
   53 a collection of type families (e.g., 'EvalM', 'VBool', etc.). -}
   54 data Value l
   55   = VFun !LocalName !(Thunk l -> MValue l)
   56   | VUnit
   57   | VPair (Thunk l) (Thunk l) -- TODO: should second component be strict?
   58   | VCtorApp !(ExtCns (TValue l)) ![Thunk l] ![Thunk l]
   59   | VVector !(Vector (Thunk l))
   60   | VBool (VBool l)
   61   | VWord (VWord l)
   62   | VBVToNat !Int (Value l) -- TODO: don't use @Int@ for this, use @Natural@
   63   | VIntToNat (Value l)
   64   | VNat !Natural
   65   | VInt (VInt l)
   66   | VIntMod !Natural (VInt l)
   67   | VArray (VArray l)
   68   | VString !Text
   69   | VRecordValue ![(FieldName, Thunk l)]
   70   | VRecursor
   71      !(ExtCns (TValue l)) -- data type ident
   72      ![Value l]  -- data type parameters
   73      !(Value l)  -- motive function
   74      !(TValue l) -- type of motive
   75      !(Map VarIndex (Thunk l, TValue l)) -- constructor eliminators and their types
   76   | VExtra (Extra l)
   77   | TValue (TValue l)
   78 
   79 -- | The subset of values that represent types.
   80 data TValue l
   81   = VVecType !Natural !(TValue l)
   82   | VBoolType
   83   | VIntType
   84   | VIntModType !Natural
   85   | VArrayType !(TValue l) !(TValue l)
   86   | VPiType LocalName !(TValue l) !(PiBody l)
   87   | VStringType
   88   | VUnitType
   89   | VPairType !(TValue l) !(TValue l)
   90   | VDataType !(ExtCns (TValue l)) ![Value l] ![Value l]
   91   | VRecordType ![(FieldName, TValue l)]
   92   | VSort !Sort
   93   | VRecursorType
   94      !(ExtCns (TValue l)) -- data type name
   95      ![Value l]  -- data type parameters
   96      !(Value l)  -- motive function
   97      !(TValue l) -- type of motive function
   98   | VTyTerm !Sort !Term
   99 
  100 data PiBody l
  101   = VDependentPi !(Thunk l -> EvalM l (TValue l))
  102   | VNondependentPi !(TValue l)
  103 
  104 -- | Neutral terms represent computations that are blocked
  105 --   because some internal term cannot be evaluated
  106 --   (e.g., because it is a variable, because it's definition
  107 --   is being hidden, etc.)
  108 data NeutralTerm
  109   = NeutralBox Term -- the thing blocking evaluation
  110   | NeutralPairLeft NeutralTerm   -- left pair projection
  111   | NeutralPairRight NeutralTerm  -- right pair projection
  112   | NeutralRecordProj NeutralTerm FieldName -- record projection
  113   | NeutralApp NeutralTerm Term -- function application
  114   | NeutralConstant -- A constant value with no definition
  115       (ExtCns Term)
  116 
  117 type Thunk l = Lazy (EvalM l) (Value l)
  118 
  119 -- | Evaluation monad for value instantiation 'l'
  120 type family EvalM l :: Type -> Type
  121 -- | Booleans for value instantiation 'l'
  122 type family VBool l :: Type
  123 -- | Words for value instantiation 'l'
  124 type family VWord l :: Type
  125 -- | Integers for value instantiation 'l'
  126 type family VInt  l :: Type
  127 -- | SMT arrays for value instantiation 'l'
  128 type family VArray l :: Type
  129 -- | Additional constructors for instantiation 'l'
  130 type family Extra l :: Type
  131 
  132 -- | Short-hand for a monadic value.
  133 type MValue l     = EvalM l (Value l)
  134 
  135 -- | Short-hand for a monadic boolean.
  136 type MBool  l     = EvalM l (VBool l)
  137 
  138 -- | Short-hand for a monadic word.
  139 type MWord l      = EvalM l (VWord l)
  140 
  141 -- | Short-hand for a monadic integer.
  142 type MInt l       = EvalM l (VInt  l)
  143 
  144 -- | Short-hand for a monadic array.
  145 type MArray l     = EvalM l (VArray l)
  146 
  147 -- | Short hand to specify that the evaluation monad is a monad (very common)
  148 type VMonad l     = Monad (EvalM l)
  149 
  150 -- | Short hand to specify that the evaluation monad is a lazy monad.
  151 type VMonadLazy l = MonadLazy (EvalM l)

  156 -- | Language instantiations with a specific monad.
  157 data WithM (m :: Type -> Type) l
  158 type instance EvalM (WithM m l) = m
  159 type instance VBool (WithM m l) = VBool l
  160 type instance VWord (WithM m l) = VWord l
  161 type instance VInt  (WithM m l) = VInt l
  162 type instance VArray (WithM m l) = VArray l
  163 type instance Extra (WithM m l) = Extra l

  167 instance Show (Extra l) => Show (Value l) where
  168   showsPrec p v =
  169     case v of
  170       VFun {}        -> showString "<<fun>>"
  171       VUnit          -> showString "()"
  172       VPair{}        -> showString "<<tuple>>"
  173       VCtorApp s _ps _xv -> shows (toAbsoluteName (ecName s))
  174       VVector xv     -> showList (toList xv)
  175       VBool _        -> showString "<<boolean>>"
  176       VWord _        -> showString "<<bitvector>>"
  177       VBVToNat n x   -> showString "bvToNat " . shows n . showString " " . showParen True (shows x)
  178       VIntToNat x    -> showString "intToNat " . showParen True (shows x)
  179       VNat n         -> shows n
  180       VInt _         -> showString "<<integer>>"
  181       VIntMod n _    -> showString ("<<Z " ++ show n ++ ">>")
  182       VArray{}       -> showString "<<array>>"
  183       VString s      -> shows s
  184       VRecordValue [] -> showString "{}"
  185       VRecordValue ((fld,_):_) ->
  186         showString "{" . showString (Text.unpack fld) . showString " = _, ...}"
  187       VRecursor d _ _ _ _
  188                      -> showString "<<recursor: " . shows d . showString ">>"
  189       VExtra x       -> showsPrec p x
  190       TValue x       -> showsPrec p x
  191     where
  192       toList = map (const Nil) . V.toList
  193 
  194 instance Show (Extra l) => Show (TValue l) where
  195   showsPrec p v =
  196     case v of
  197       VBoolType      -> showString "Bool"
  198       VStringType    -> showString "String"
  199       VIntType       -> showString "Integer"
  200       VIntModType n  -> showParen True (showString "IntMod " . shows n)
  201       VArrayType{}   -> showString "Array"
  202       VPiType _ t _    -> showParen True
  203                         (shows t . showString " -> ...")
  204       VUnitType      -> showString "#()"
  205       VPairType x y  -> showParen True (shows x . showString " * " . shows y)
  206       VDataType s ps vs
  207         | null (ps++vs) -> shows s
  208         | otherwise  -> shows s . showList (ps++vs)
  209       VRecordType [] -> showString "{}"
  210       VRecordType ((fld,_):_) ->
  211         showString "{" . showString (Text.unpack fld) . showString " :: _, ...}"
  212       VVecType n a   -> showString "Vec " . shows n
  213                         . showString " " . showParen True (showsPrec p a)
  214       VSort s        -> shows s
  215       VRecursorType{} -> showString "RecursorType"
  216 
  217       VTyTerm _ tm   -> showString "TyTerm (" . (\x -> showTerm tm ++ x) . showString ")"

  222   show Nil = "_"

  225 -- Basic operations on values

  227 vTuple :: VMonad l => [Thunk l] -> Value l
  228 vTuple [] = VUnit
  229 vTuple [_] = error "vTuple: unsupported 1-tuple"
  230 vTuple [x, y] = VPair x y
  231 vTuple (x : xs) = VPair x (ready (vTuple xs))
  232 
  233 vTupleType :: VMonad l => [TValue l] -> TValue l
  234 vTupleType [] = VUnitType
  235 vTupleType [t] = t
  236 vTupleType (t : ts) = VPairType t (vTupleType ts)
  237 
  238 valPairLeft :: (HasCallStack, VMonad l, Show (Extra l)) => Value l -> MValue l
  239 valPairLeft (VPair t1 _) = force t1
  240 valPairLeft v = panic "valPairLeft" ["Not a pair value: " <> Text.pack (show v)]
  241 
  242 valPairRight :: (HasCallStack, VMonad l, Show (Extra l)) => Value l -> MValue l
  243 valPairRight (VPair _ t2) = force t2
  244 valPairRight v = panic "valPairRight" ["Not a pair value: " <> Text.pack (show v)]
  245 
  246 vRecord :: Map FieldName (Thunk l) -> Value l
  247 vRecord m = VRecordValue (Map.assocs m)
  248 
  249 valRecordProj :: (HasCallStack, VMonad l, Show (Extra l)) => Value l -> FieldName -> MValue l
  250 valRecordProj (VRecordValue fld_map) fld
  251   | Just t <- lookup fld fld_map = force t
  252 valRecordProj v@(VRecordValue _) fld =
  253   panic "valRecordProj" [
  254       "Record field " <> Text.pack (show fld) <> " not found in value: " <>
  255           Text.pack (show v)
  256   ]
  257 valRecordProj v _ =
  258   panic "valRecordProj" ["Not a record value: " <> Text.pack (show v)]
  259 
  260 apply :: (HasCallStack, VMonad l, Show (Extra l)) => Value l -> Thunk l -> MValue l
  261 apply (VFun _ f) x = f x
  262 apply (TValue (VPiType _ _ body)) x = TValue <$> applyPiBody body x
  263 
  264 apply v _x = panic "apply" ["Not a function value: " <> Text.pack (show v)]
  265 
  266 applyAll :: (VMonad l, Show (Extra l)) => Value l -> [Thunk l] -> MValue l
  267 applyAll = foldM apply
  268 
  269 {-# INLINE applyPiBody #-}
  270 applyPiBody :: VMonad l => PiBody l -> Thunk l -> EvalM l (TValue l)
  271 applyPiBody (VDependentPi f) x    = f x
  272 applyPiBody (VNondependentPi t) _ = pure t

  274 asFiniteTypeValue :: Value l -> Maybe FiniteType
  275 asFiniteTypeValue v =
  276   case v of
  277     TValue tv -> asFiniteTypeTValue tv
  278     _ -> Nothing
  279 
  280 asFiniteTypeTValue :: TValue l -> Maybe FiniteType
  281 asFiniteTypeTValue v =
  282   case v of
  283     VBoolType -> return FTBit
  284     VVecType n v1 -> do
  285       t1 <- asFiniteTypeTValue v1
  286       return (FTVec n t1)
  287     VUnitType -> return (FTTuple [])
  288     VPairType v1 v2 -> do
  289       t1 <- asFiniteTypeTValue v1
  290       t2 <- asFiniteTypeTValue v2
  291       case t2 of
  292         FTTuple ts -> return (FTTuple (t1 : ts))
  293         _ -> return (FTTuple [t1, t2])
  294     VRecordType elem_tps ->
  295       FTRec <$> Map.fromList <$>
  296       mapM (\(fld,tp) -> (fld,) <$> asFiniteTypeTValue tp) elem_tps
  297     _ -> Nothing
  298 
  299 asFirstOrderTypeValue :: Value l -> Maybe FirstOrderType
  300 asFirstOrderTypeValue v =
  301   case v of
  302     TValue tv -> asFirstOrderTypeTValue tv
  303     _ -> Nothing
  304 
  305 asFirstOrderTypeTValue :: TValue l -> Maybe FirstOrderType
  306 asFirstOrderTypeTValue v =
  307   case v of
  308     VBoolType     -> return FOTBit
  309     VVecType n v1 -> FOTVec n <$> asFirstOrderTypeTValue v1
  310     VIntType      -> return FOTInt
  311     VIntModType m -> return (FOTIntMod m)
  312     VArrayType a b ->
  313       FOTArray <$> asFirstOrderTypeTValue a <*> asFirstOrderTypeTValue b
  314     VUnitType -> return (FOTTuple [])
  315     VPairType v1 v2 -> do
  316       t1 <- asFirstOrderTypeTValue v1
  317       t2 <- asFirstOrderTypeTValue v2
  318       case t2 of
  319         FOTTuple ts -> return (FOTTuple (t1 : ts))
  320         _ -> return (FOTTuple [t1, t2])
  321     VRecordType elem_tps ->
  322       FOTRec . Map.fromList <$>
  323         mapM (traverse asFirstOrderTypeTValue) elem_tps
  324 
  325     VStringType   -> Nothing
  326     VPiType{}   -> Nothing
  327     VDataType{} -> Nothing
  328     VSort{}     -> Nothing
  329     VRecursorType{} -> Nothing
  330     VTyTerm{}   -> Nothing
  331 
  332 -- | A (partial) injective mapping from type values to strings. These
  333 -- are intended to be useful as suffixes for names of type instances
  334 -- of uninterpreted constants.
  335 suffixTValue :: TValue sym -> Maybe String
  336 suffixTValue tv =
  337   case tv of
  338     VVecType n a ->
  339       do a' <- suffixTValue a
  340          Just ("_Vec_" ++ show n ++ a')
  341     VBoolType -> Just "_Bool"
  342     VIntType -> Just "_Int"
  343     VIntModType n -> Just ("_IntMod_" ++ show n)
  344     VArrayType a b ->
  345       do a' <- suffixTValue a
  346          b' <- suffixTValue b
  347          Just ("_Array" ++ a' ++ b')
  348     VPiType _ _ _ -> Nothing
  349     VUnitType -> Just "_Unit"
  350     VPairType a b ->
  351       do a' <- suffixTValue a
  352          b' <- suffixTValue b
  353          Just ("_Pair" ++ a' ++ b')
  354 
  355     VStringType -> Nothing
  356     VDataType {} -> Nothing
  357     VRecordType {} -> Nothing
  358     VSort {} -> Nothing
  359     VRecursorType{} -> Nothing
  360     VTyTerm{} -> Nothing

  363 neutralToTerm :: NeutralTerm -> Term
  364 neutralToTerm = loop
  365   where
  366   loop (NeutralBox tm) = tm
  367   loop (NeutralPairLeft nt) =
  368     Unshared (FTermF (PairLeft (loop nt)))
  369   loop (NeutralPairRight nt) =
  370     Unshared (FTermF (PairRight (loop nt)))
  371   loop (NeutralRecordProj nt f) =
  372     Unshared (FTermF (RecordProj (loop nt) f))
  373   loop (NeutralApp nt arg) =
  374     Unshared (App (loop nt) arg)
  375   loop (NeutralConstant ec) =
  376     Unshared (Constant ec)
  377 
  378 neutralToSharedTerm :: SharedContext -> NeutralTerm -> IO Term
  379 neutralToSharedTerm sc = loop
  380   where
  381   loop (NeutralBox tm) = pure tm
  382   loop (NeutralPairLeft nt) =
  383     scFlatTermF sc . PairLeft =<< loop nt
  384   loop (NeutralPairRight nt) =
  385     scFlatTermF sc . PairRight =<< loop nt
  386   loop (NeutralRecordProj nt f) =
  387     do tm <- loop nt
  388        scFlatTermF sc (RecordProj tm f)
  389   loop (NeutralApp nt arg) =
  390     do tm <- loop nt
  391        scApply sc tm arg
  392   loop (NeutralConstant ec) =
  393     do scTermF sc (Constant ec)
  394 
  395 ppNeutral :: PPS.Opts -> NeutralTerm -> PPS.Doc
  396 ppNeutral opts = ppTerm opts . neutralToTerm
