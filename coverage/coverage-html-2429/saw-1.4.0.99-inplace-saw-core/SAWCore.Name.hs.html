   82 -- Module Names ----------------------------------------------------------------
   83 
   84 newtype ModuleName = ModuleName Text
   85   deriving (Eq, Ord, Generic, TH.Lift)
   86 
   87 instance Hashable ModuleName -- automatically derived
   88 
   89 instance Show ModuleName where
   90   show (ModuleName s) = Text.unpack s
   91 
   92 
   93 moduleNameText :: ModuleName -> Text
   94 moduleNameText (ModuleName x) = x
   95 
   96 moduleNamePieces :: ModuleName -> [Text]
   97 moduleNamePieces (ModuleName x) = Text.splitOn "." x
   98 
   99 -- | Create a module name given a list of strings with the top-most
  100 -- module name given first.
  101 mkModuleName :: [Text] -> ModuleName
  102 mkModuleName nms = ModuleName $ Text.intercalate "." nms
  103 
  104 preludeName :: ModuleName
  105 preludeName = mkModuleName ["Prelude"]

  108 -- Identifiers -----------------------------------------------------------------
  109 
  110 data Ident =
  111   Ident
  112   { identModule :: ModuleName
  113   , identBaseName :: Text
  114   }
  115   deriving (Eq, Ord, Generic)
  116 
  117 instance Hashable Ident -- automatically derived
  118 
  119 instance Show Ident where
  120   show (Ident m s) = shows m ('.' : Text.unpack s)
  121 
  122 identText :: Ident -> Text
  123 identText i = moduleNameText (identModule i) <> Text.pack "." <> identBaseName i
  124 
  125 identPieces :: Ident -> NonEmpty Text
  126 identPieces i =
  127   case moduleNamePieces (identModule i) of
  128     [] -> identBaseName i :| []
  129     (x:xs) -> x :| (xs ++ [identBaseName i])
  130 
  131 identName :: Ident -> String
  132 identName = Text.unpack . identBaseName
  133 
  134 instance Read Ident where
  135   readsPrec _ str =
  136     let (str1, str2) = break (not . isIdChar) str in
  137     [(parseIdent str1, str2)]
  138 
  139 mkIdent :: ModuleName -> Text -> Ident
  140 mkIdent m s = Ident m s
  141 
  142 -- | Make a \"coq-safe\" identifier from a string that might contain
  143 -- non-identifier characters, where we use the SAW core notion of identifier
  144 -- characters as letters, digits, underscore and primes. Any disallowed
  145 -- character is mapped to the string @__xNN@, where @NN@ is the hexadecimal code
  146 -- for that character. Additionally, a SAW core identifier is not allowed to
  147 -- start with a prime, so a leading underscore is added in such a case.
  148 mkSafeIdent :: ModuleName -> String -> Ident
  149 mkSafeIdent _ [] = fromString "_"
  150 mkSafeIdent mnm nm =
  151   let is_safe_char c = isAlphaNum c || c == '_' || c == '\'' in
  152   mkIdent mnm $ Text.pack $
  153   (if nm!!0 == '\'' then ('_' :) else id) $
  154   concatMap
  155   (\c -> if is_safe_char c then [c] else
  156            "__x" ++ showHex (ord c) "")
  157   nm
  158 
  159 -- | Parse a fully qualified identifier.
  160 parseIdent :: String -> Ident
  161 parseIdent s0 =
  162     case reverse (breakEach s0) of
  163       (_:[]) -> panic "parseIdent" ["Empty module name"]
  164       (nm:rMod) -> mkIdent (mkModuleName (reverse rMod)) nm
  165       _ -> panic "parseIdent" ["Bad identifier " <> Text.pack s0]
  166   where breakEach s =
  167           case break (=='.') s of
  168             (h, []) -> [Text.pack h]
  169             (h, _ : r) -> Text.pack h : breakEach r
  170 
  171 instance IsString Ident where
  172   fromString = parseIdent
  173 
  174 isIdent :: String -> Bool
  175 isIdent (c:l) = isAlpha c && all isIdChar l
  176 isIdent [] = False
  177 
  178 -- | Returns true if character can appear in identifier.
  179 isIdChar :: Char -> Bool
  180 isIdChar c = isAlphaNum c || (c == '_') || (c == '\'') || (c == '.')

  184 -- NameInfo
  185 
  186 
  187 -- | Descriptions of the origins of names that may be in scope
  188 data NameInfo
  189   = -- | This name arises from an exported declaration from a module
  190     ModuleIdentifier Ident
  191 
  192   | -- | This name was imported from some other programming language/scope
  193     ImportedName
  194       URI      -- ^ An absolutely-qualified name, which is required to be unique
  195       [Text]   -- ^ A collection of aliases for this name.  Shorter or "less-qualified"
  196                --   aliases should be nearer the front of the list
  197 
  198  deriving (Eq,Ord,Show)
  199 
  200 instance Hashable NameInfo where
  201   hashWithSalt x nmi =
  202     case nmi of
  203       ModuleIdentifier ident -> hashWithSalt x ident
  204       ImportedName uri _ -> hashWithSalt x uri
  205 
  206 nameURI :: NameInfo -> URI
  207 nameURI =
  208   \case
  209     ModuleIdentifier i -> moduleIdentToURI i
  210     ImportedName uri _ -> uri
  211 
  212 nameAliases :: NameInfo -> [Text]
  213 nameAliases =
  214   \case
  215     ModuleIdentifier i -> [identBaseName i, identText i]
  216     ImportedName _ aliases -> aliases
  217 
  218 toShortName :: NameInfo -> Text
  219 toShortName (ModuleIdentifier i) = identBaseName i
  220 toShortName (ImportedName uri []) = render uri
  221 toShortName (ImportedName _ (x:_)) = x
  222 
  223 toAbsoluteName :: NameInfo -> Text
  224 toAbsoluteName (ModuleIdentifier i) = identText i
  225 toAbsoluteName (ImportedName uri _) = render uri
  226 
  227 moduleIdentToURI :: Ident -> URI
  228 moduleIdentToURI ident = fromMaybe (panic "moduleIdentToURI" ["Failed to construct ident URI: " <> identText ident]) $
  229   do sch  <- mkScheme "sawcore"
  230      path <- mapM mkPathPiece (identPieces ident)
  231      pure URI
  232        { uriScheme = Just sch
  233        , uriAuthority = Left True -- absolute path
  234        , uriPath   = Just (False, path)
  235        , uriQuery  = []
  236        , uriFragment = Nothing
  237        }

  240 -- Global Names ----------------------------------------------------------------
  241 
  242 type VarIndex = Int
  243 
  244 -- | A global name with a unique ID. We maintain a global invariant
  245 -- that the 'VarIndex' and the 'NameInfo' must be in a strict
  246 -- one-to-one correspondence: Each 'VarIndex' is paired with a unique
  247 -- 'NameInfo', and each 'NameInfo' is paired with a unique 'VarIndex'.
  248 data Name =
  249   Name
  250   { nameIndex :: !VarIndex
  251   , nameInfo :: !NameInfo
  252   }
  253   deriving (Show)
  254 
  255 -- | Because of the global uniqueness invariant, comparing the
  256 -- 'VarIndex' is sufficient to ensure equality of names.
  257 instance Eq Name where
  258   x == y = nameIndex x == nameIndex y
  259 
  260 instance Ord Name where
  261   compare x y = compare (nameIndex x) (nameIndex y)
  262 
  263 -- | For hashing, we consider only the 'NameInfo' and not the
  264 -- 'VarIndex'; this gives a stable hash value for a particular name,
  265 -- even if the unique IDs are assigned differently from run to run.
  266 instance Hashable Name where
  267   hashWithSalt x nm = hashWithSalt x (nameInfo nm)
  268

  270 -- Variable Names --------------------------------------------------------------
  271 
  272 -- | A variable name with a unique ID.
  273 -- We assume a global uniqueness invariant: The 'VarIndex' determines the name.
  274 data VarName =
  275   VarName
  276   { vnIndex :: !VarIndex
  277   , vnName :: !Text
  278   }
  279   deriving (Show)
  280 
  281 -- | Because of the global uniqueness invariant, comparing the
  282 -- 'VarIndex' is sufficient to ensure equality of variable names.
  283 instance Eq VarName where
  284   x == y = vnIndex x == vnIndex y
  285 
  286 instance Ord VarName where
  287   compare x y = compare (vnIndex x) (vnIndex y)
  288 
  289 -- | For hashing, we consider only the base name and not the
  290 -- 'VarIndex'; this gives a stable hash value for a particular name,
  291 -- even if the unique IDs are assigned differently from run to run.
  292 instance Hashable VarName where
  293  hashWithSalt x vn = hashWithSalt x (vnName vn)

  297 
  298 -- | A global name paired with a type.
  299 -- We assume a global invariant: the type should be uniquely
  300 -- determined by the name.
  301 data ExtCns e =
  302   EC
  303   { ecName :: !VarName
  304   , ecType :: !e
  305   }
  306   deriving (Show, Functor, Foldable, Traversable)
  307 
  308 -- | Because of the global uniqueness invariant, comparing the
  309 -- 'Name' is sufficient to ensure equality.
  310 instance Eq (ExtCns e) where
  311   x == y = ecName x == ecName y
  312 
  313 instance Ord (ExtCns e) where
  314   compare x y = compare (ecName x) (ecName y)
  315 
  316 instance Hashable (ExtCns e) where
  317   hashWithSalt x ec = hashWithSalt x (ecName ec)
  318 
  319 ecVarIndex :: ExtCns e -> VarIndex
  320 ecVarIndex ec = vnIndex (ecName ec)
  321 
  322 ecShortName :: ExtCns e -> Text
  323 ecShortName ec = vnName (ecName ec)
  324 
  325 scFreshNameURI :: Text -> VarIndex -> URI
  326 scFreshNameURI nm i = fromMaybe (panic "scFreshNameURI" ["Failed to construct name URI: <> " <> nm <> "  " <> Text.pack (show i)]) $
  327   do sch <- mkScheme "fresh"
  328      nm' <- mkPathPiece (if Text.null nm then "_" else nm)
  329      i'  <- mkFragment (Text.pack (show i))
  330      pure URI
  331        { uriScheme = Just sch
  332        , uriAuthority = Left False -- relative path
  333        , uriPath   = Just (False, (nm' :| []))
  334        , uriQuery  = []
  335        , uriFragment = Just i'
  336        }

  339 -- Display Name Environments --------------------------------------------------------
  340 
  341 -- | A 'DisplayNameEnv' encodes the mappings between display names (type
  342 -- 'Text') and internal names (type 'VarIndex'). Multiple display
  343 -- names may be associated with each internal name; these are stored
  344 -- in priority order, with preferred display names first.
  345 data DisplayNameEnv =
  346   DisplayNameEnv
  347   { displayNames :: !(IntMap [Text]) -- Keyed by VarIndex; preferred names come first.
  348   , displayIndexes :: !(Map Text IntSet)
  349   }
  350 -- Invariants: The 'displayNames' and 'displayIndexes' maps should be
  351 -- inverses of each other. That is, 'displayNames' maps @i@ to a list
  352 -- containing @x@ if and only if 'displayIndexes' maps @x@ to a set
  353 -- containing @i@.
  354 
  355 emptyDisplayNameEnv :: DisplayNameEnv
  356 emptyDisplayNameEnv = DisplayNameEnv mempty mempty
  357 
  358 -- | Extend a 'DisplayNameEnv' by providing new 'Text' display names
  359 -- for a given 'VarIndex'. Display names should be provided in
  360 -- priority order, with preferred names first. If the 'VarIndex'
  361 -- already exists in the map, then add the new display names with
  362 -- higher priority while keeping the old ones.
  363 extendDisplayNameEnv :: VarIndex -> [Text] -> DisplayNameEnv -> DisplayNameEnv
  364 extendDisplayNameEnv i aliases env =
  365   DisplayNameEnv
  366   { displayNames = IntMap.insertWith List.union i aliases (displayNames env)
  367   , displayIndexes = foldr insertAlias (displayIndexes env) aliases
  368   }
  369   where
  370     insertAlias :: Text -> Map Text IntSet -> Map Text IntSet
  371     insertAlias x m = Map.insertWith IntSet.union x (IntSet.singleton i) m
  372 
  373 -- | Filter display names in a 'DisplayNameEnv' that satisfy a predicate.
  374 filterDisplayNameEnv :: (Text -> Bool) -> DisplayNameEnv -> DisplayNameEnv
  375 filterDisplayNameEnv p env =
  376   DisplayNameEnv
  377   { displayNames = IntMap.filter (not . null) $ fmap (filter p) $ displayNames env
  378   , displayIndexes = Map.filterWithKey (\k _ -> p k) $ displayIndexes env
  379   }
  380 
  381 -- | Merge two 'DisplayNameEnv's, giving higher priority to display
  382 -- names from the first argument.
  383 mergeDisplayNameEnv :: DisplayNameEnv -> DisplayNameEnv -> DisplayNameEnv
  384 mergeDisplayNameEnv env1 env2 =
  385   DisplayNameEnv
  386   { displayNames = IntMap.unionWith List.union (displayNames env1) (displayNames env2)
  387   , displayIndexes = Map.unionWith IntSet.union (displayIndexes env1) (displayIndexes env2)
  388   }
  389 
  390 -- | Look up a 'Text' display name in an environment, returning the
  391 -- list of indexes it could resolve to.
  392 resolveDisplayName :: DisplayNameEnv -> Text -> [VarIndex]
  393 resolveDisplayName env x =
  394   case Map.lookup x (displayIndexes env) of
  395     Nothing -> []
  396     Just s -> IntSet.elems s
  397 
  398 -- | Return the first display name for the given 'VarIndex' that is
  399 -- unambiguous in the naming environment. If there is no unambiguous
  400 -- alias, then return 'Nothing'.
  401 bestDisplayName :: DisplayNameEnv -> VarIndex -> Maybe Text
  402 bestDisplayName env i =
  403   do aliases <- IntMap.lookup i (displayNames env)
  404      go aliases
  405   where
  406     go [] = Nothing
  407     go (x : xs) =
  408       case Map.lookup x (displayIndexes env) of
  409         Nothing -> panic "bestDisplayName" ["Invariant violated: Missing key " <> x]
  410         Just vs
  411           | IntSet.size vs == 1 -> Just x
  412           | otherwise -> go xs
