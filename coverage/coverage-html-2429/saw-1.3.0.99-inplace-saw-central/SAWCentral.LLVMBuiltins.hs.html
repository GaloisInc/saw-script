   41 llvm_load_module :: FilePath -> TopLevel (Some CMS.LLVMModule)
   42 llvm_load_module file =
   43   do laxArith <- gets rwLaxArith
   44      debugIntrinsics <- gets rwDebugIntrinsics
   45      let ?transOpts = CL.defaultTranslationOptions
   46                         { CL.laxArith = laxArith
   47                         , CL.debugIntrinsics = debugIntrinsics
   48                         }
   49      halloc <- getHandleAlloc
   50      io (CMS.loadLLVMModule file halloc) >>= \case
   51        Left err -> fail (LLVM.formatError err)
   52        Right (llvm_mod, warnings) -> do
   53          unless (null warnings) $
   54            printOutLnTop Warn $ show $ LLVM.ppParseWarnings warnings
   55          return llvm_mod

   57 llvm_type :: Text -> TopLevel LLVM.Type
   58 llvm_type str =
   59   case LLVM.parseType (Text.unpack str) of
   60     Left e -> fail (show e)
   61     Right t -> return t
   62
