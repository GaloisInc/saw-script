  170 -- | A singleton type that encodes the extensions used for each SAW backend.
  171 -- Matching on a value of type @'SAWExt' ext@ allows one to learn which backend
  172 -- is currently in use.
  173 data SAWExt :: Type -> Type where
  174   LLVMExt :: SAWExt (LLVM arch)
  175   JVMExt  :: SAWExt JVM
  176   MIRExt  :: SAWExt MIR
  177 
  178 -- | This allows checking which SAW backend we are using at runtime.
  179 class IsExt ext where
  180   sawExt :: SAWExt ext
  181 
  182 instance IsExt (LLVM arch) where
  183   sawExt = LLVMExt
  184 instance IsExt JVM where
  185   sawExt = JVMExt
  186 instance IsExt MIR where
  187   sawExt = MIRExt
  188 
  189 -- | Note that most 'SetupValue' concepts (like allocation indices)
  190 --   are implementation details and won't be familiar to users.
  191 --   Consider using 'resolveSetupValue' and printing the language-specific value
  192 --   (e.g., an 'LLVMVal') with @PP.pretty@ instead.
  193 ppSetupValue :: forall ext ann. IsExt ext => SetupValue ext -> PP.Doc ann
  194 ppSetupValue setupval = case setupval of
  195   SetupTerm tm   -> ppTypedTerm tm
  196   SetupVar i     -> ppAllocIndex i
  197   SetupNull _    -> "NULL"
  198   SetupStruct x vs ->
  199     case (ext, x) of
  200       (LLVMExt, packed) ->
  201         ppSetupStructLLVM packed vs
  202       (JVMExt, empty) ->
  203         absurd empty
  204       (MIRExt, _defId) ->
  205         ppSetupStructDefault vs
  206   SetupEnum x ->
  207     case (ext, x) of
  208       (LLVMExt, empty) ->
  209         absurd empty
  210       (JVMExt, empty) ->
  211         absurd empty
  212       (MIRExt, enum_) ->
  213         ppMirSetupEnum enum_
  214   SetupTuple x vs ->
  215     case (ext, x) of
  216       (LLVMExt, empty) ->
  217         absurd empty
  218       (JVMExt, empty) ->
  219         absurd empty
  220       (MIRExt, ()) ->
  221         ppSetupTuple vs
  222   SetupSlice x ->
  223     case (ext, x) of
  224       (LLVMExt, empty) ->
  225         absurd empty
  226       (JVMExt, empty) ->
  227         absurd empty
  228       (MIRExt, slice) ->
  229         ppMirSetupSlice slice
  230   SetupArray _ vs  -> PP.brackets (commaList (map ppSetupValue vs))
  231   SetupElem _ v i  -> PP.parens (ppSetupValue v) PP.<> PP.pretty ("." ++ show i)
  232   SetupField _ v f -> PP.parens (ppSetupValue v) PP.<> PP.pretty ("." <> f)
  233   SetupUnion _ v u -> PP.parens (ppSetupValue v) PP.<> PP.pretty ("." <> u)
  234   SetupCast x v ->
  235     case (ext, x) of
  236       (LLVMExt, tp) ->
  237         ppCast v tp
  238       (JVMExt, empty) ->
  239         absurd empty
  240       (MIRExt, ty) ->
  241         ppCast v ty
  242   SetupGlobal _ nm -> PP.pretty ("global(" <> nm <> ")")
  243   SetupGlobalInitializer _ nm -> PP.pretty ("global_initializer(" <> nm <> ")")
  244   SetupMux x c t f ->
  245     case (ext, x) of
  246       (LLVMExt, empty) ->
  247         absurd empty
  248       (JVMExt, empty) ->
  249         absurd empty
  250       (MIRExt, ()) ->
  251         "mux" <>
  252         PP.parens (ppTypedTerm c <> PP.comma PP.<+> ppSetupValue t <> PP.comma PP.<+> ppSetupValue f)
  253   where
  254     ext :: SAWExt ext
  255     ext = sawExt @ext
  256 
  257     commaList :: [PP.Doc ann] -> PP.Doc ann
  258     commaList []     = PP.emptyDoc
  259     commaList (x:xs) = x PP.<> PP.hcat (map (\y -> PP.comma PP.<+> y) xs)
  260 
  261     ppSetupStructLLVM ::
  262          Bool
  263          -- ^ 'True' if this is an LLVM packed struct, 'False' otherwise.
  264       -> [SetupValue ext] -> PP.Doc ann
  265     ppSetupStructLLVM packed vs
  266       | packed    = PP.angles (ppSetupStructDefault vs)
  267       | otherwise = ppSetupStructDefault vs
  268 
  269     ppSetupStructDefault :: forall ext'. IsExt ext' => [SetupValue ext'] -> PP.Doc ann
  270     ppSetupStructDefault vs = PP.braces (commaList (map ppSetupValue vs))
  271 
  272     ppSetupTuple :: [SetupValue MIR] -> PP.Doc ann
  273     ppSetupTuple vs = PP.parens (commaList (map ppSetupValue vs))
  274 
  275     ppMirSetupEnum :: MirSetupEnum -> PP.Doc ann
  276     ppMirSetupEnum (MirSetupEnumVariant _defId variantName _varIdx fields) =
  277       PP.pretty variantName PP.<+> ppSetupStructDefault fields
  278     ppMirSetupEnum (MirSetupEnumSymbolic _defId _discr _variants) =
  279       "<symbolic enum>"
  280 
  281     ppMirSetupSlice :: MirSetupSlice -> PP.Doc ann
  282     ppMirSetupSlice (MirSetupSliceRaw ref len) =
  283       "SliceRaw" <> ppSetupTuple [ref, len]
  284     ppMirSetupSlice (MirSetupSlice _ arr) =
  285       ppSetupValue arr <> "[..]"
  286     ppMirSetupSlice (MirSetupSliceRange _ arr start end) =
  287       ppSetupValue arr <> "[" <> PP.pretty start <>
  288       ".." <> PP.pretty end <> "]"
  289 
  290     ppCast :: Show ty => SetupValue ext -> ty -> PP.Doc ann
  291     ppCast v ty = PP.parens (ppSetupValue v) PP.<> PP.pretty (" AS " ++ show ty)

  293 setupToTypedTerm ::
  294   Options {-^ Printing options -} ->
  295   SharedContext ->
  296   SetupValue ext ->
  297   MaybeT IO TypedTerm
  298 setupToTypedTerm opts sc sv =
  299   case sv of
  300     SetupTerm term -> return term
  301     _ -> do t <- setupToTerm opts sc sv
  302             lift $ mkTypedTerm sc t
  303 
  304 -- | Convert a setup value to a SAW-Core term. This is a partial
  305 -- function, as certain setup values ---SetupVar, SetupNull and
  306 -- SetupGlobal--- don't have semantics outside of the symbolic
  307 -- simulator.
  308 setupToTerm ::
  309   Options ->
  310   SharedContext ->
  311   SetupValue ext ->
  312   MaybeT IO Term
  313 setupToTerm opts sc =
  314   \case
  315     SetupTerm term -> return (ttTerm term)
  316 
  317     SetupStruct _ fields ->
  318       do ts <- mapM (setupToTerm opts sc) fields
  319          lift $ scTuple sc ts
  320 
  321     SetupArray _ elems@(_:_) ->
  322       do ts@(t:_) <- mapM (setupToTerm opts sc) elems
  323          typt <- lift $ scTypeOf sc t
  324          vec <- lift $ scVector sc typt ts
  325          typ <- lift $ scTypeOf sc vec
  326          lift $ printOutLn opts Info $ show vec
  327          lift $ printOutLn opts Info $ show typ
  328          return vec
  329 
  330     SetupElem _ base ind ->
  331       case base of
  332         SetupArray _ elems@(e:_) ->
  333           do let intToNat = fromInteger . toInteger
  334              art <- setupToTerm opts sc base
  335              ixt <- lift $ scNat sc $ intToNat ind
  336              lent <- lift $ scNat sc $ intToNat $ length elems
  337              et <- setupToTerm opts sc e
  338              typ <- lift $ scTypeOf sc et
  339              lift $ scAt sc lent typ art ixt
  340 
  341         SetupStruct _ fs ->
  342           do st <- setupToTerm opts sc base
  343              lift $ scTupleSelector sc st ind (length fs)
  344 
  345         _ -> MaybeT $ return Nothing
  346 
  347     -- SetupVar, SetupNull, SetupGlobal
  348     _ -> MaybeT $ return Nothing

  351 -- ** Ghost state
  352 
  353 type GhostValue  = "GhostValue"
  354 type GhostType   = Crucible.IntrinsicType GhostValue Crucible.EmptyCtx
  355 type GhostGlobal = Crucible.GlobalVar GhostType
  356 
  357 instance Crucible.IntrinsicClass Sym GhostValue where
  358   type Intrinsic Sym GhostValue ctx = (Cryptol.Schema, Term)
  359   muxIntrinsic sym _ _namerep _ctx prd (thnSch,thn) (elsSch,els) =
  360     do when (thnSch /= elsSch) $ fail $ unlines $
  361          [ "Attempted to mux ghost variables of different types:"
  362          , show (Cryptol.pp thnSch)
  363          , show (Cryptol.pp elsSch)
  364          ]
  365        st <- sawCoreState sym
  366        let sc  = saw_ctx st
  367        prd' <- toSC sym st prd
  368        typ  <- scTypeOf sc thn
  369        res  <- scIte sc typ prd' thn els
  370        return (thnSch, res)

  385 -- | Verification state (either pre- or post-) specification
  386 data StateSpec ext = StateSpec
  387   { _csAllocs        :: Map AllocIndex (AllocSpec ext)
  388     -- ^ allocated or declared pointers
  389   , _csPointsTos     :: [PointsTo ext]
  390     -- ^ points-to statements
  391   , _csConditions    :: [SetupCondition ext]
  392     -- ^ equality, propositions, and ghost-variable conditions
  393   , _csFreshVars     :: [TypedExtCns]
  394     -- ^ fresh variables created in this state
  395   , _csVarTypeNames  :: !(Map AllocIndex (TypeName ext))
  396     -- ^ names for types of variables, for diagnostics
  397   }
  398 
  399 makeLenses ''StateSpec
  400 
  401 initialStateSpec :: StateSpec ext
  402 initialStateSpec =  StateSpec
  403   { _csAllocs        = Map.empty
  404   , _csPointsTos     = []
  405   , _csConditions    = []
  406   , _csFreshVars     = []
  407   , _csVarTypeNames  = Map.empty
  408   }

  433 type SpecNonce ext = Nonce GlobalNonceGenerator (ProvedSpec ext)

  435 -- | Data collected about discharged verification conditions (VCs).
  436 --   Verification conditions arise when proving function specifications
  437 --   due to, e.g., safety conditions, specification postconditions, and
  438 --   preconditions of called override functions.
  439 data VCStats =
  440   VCStats
  441   { vcMetadata    :: ConditionMetadata -- ^ Metadata describing why the VC arose
  442   , vcSolverStats :: SolverStats       -- ^ Statistics about any solvers used when proving this VC
  443   , vcThmSummary  :: TheoremSummary    -- ^ A summary of the proof status of this VC
  444   , vcIdent       :: TheoremNonce      -- ^ A unique identifier for this VC
  445   , vcDeps        :: Set TheoremNonce  -- ^ A collection of the theorems the proof of this VC relied on
  446   , vcElapsedTime :: NominalDiffTime   -- ^ The time required to prove this VC
  447   }
  448 
  449 data ProvedSpec ext =
  450   ProvedSpec
  451   { _psSpecIdent   :: Nonce GlobalNonceGenerator (ProvedSpec ext)
  452   , _psProofMethod :: ProofMethod
  453   , _psSpec        :: CrucibleMethodSpecIR ext
  454   , _psSolverStats :: SolverStats -- ^ statistics about the proof that produced this
  455   , _psVCStats     :: [VCStats]
  456        -- ^ Stats about the individual verification conditions produced
  457        --   by the proof of this specification
  458   , _psSpecDeps    :: Set (SpecNonce ext)
  459                         -- ^ Other proved specifications this proof depends on
  460   , _psElapsedTime :: NominalDiffTime -- ^ The time elapsed during the proof of this specification
  461   }
  462 
  463 makeLenses ''ProvedSpec

  465 mkProvedSpec ::
  466   ProofMethod ->
  467   CrucibleMethodSpecIR ext ->
  468   SolverStats ->
  469   [VCStats] ->
  470   Set (SpecNonce ext) ->
  471   NominalDiffTime ->
  472   IO (ProvedSpec ext)
  473 mkProvedSpec m mspec stats vcStats sps elapsed =
  474   do n <- freshNonce globalNonceGenerator
  475      let ps = ProvedSpec n m mspec stats vcStats sps elapsed
  476      return ps

  482 ppMethodSpec ::
  483   ( PP.Pretty (MethodId ext)
  484   , PP.Pretty (ExtType ext)
  485   ) =>
  486   CrucibleMethodSpecIR ext ->
  487   PP.Doc ann
  488 ppMethodSpec methodSpec =
  489   PP.vcat
  490   [ "Name: " <> PP.pretty (methodSpec ^. csMethod)
  491   , "Location: " <> prettyPosition (plSourceLoc (methodSpec ^. csLoc))
  492   , "Argument types: "
  493   , bullets '-' (map PP.pretty (methodSpec ^. csArgs))
  494   , "Return type: " <> case methodSpec ^. csRet of
  495                                        Nothing  -> "<void>"
  496                                        Just ret -> PP.pretty ret
  497   ]

  499 csAllocations :: CrucibleMethodSpecIR ext -> Map AllocIndex (AllocSpec ext)
  500 csAllocations
  501   = Map.unions
  502   . toListOf ((csPreState <> csPostState) . csAllocs)

  504 csTypeNames :: CrucibleMethodSpecIR ext -> Map AllocIndex (TypeName ext)
  505 csTypeNames
  506   = Map.unions
  507   . toListOf ((csPreState <> csPostState) . csVarTypeNames)

  509 makeCrucibleMethodSpecIR ::
  510   MethodId ext ->
  511   [ExtType ext] ->
  512   Maybe (ExtType ext) ->
  513   ProgramLoc ->
  514   Codebase ext ->
  515   CrucibleMethodSpecIR ext
  516 makeCrucibleMethodSpecIR meth args ret loc code = do
  517   CrucibleMethodSpec
  518     {_csMethod          = meth
  519     ,_csArgs            = args
  520     ,_csRet             = ret
  521     ,_csPreState        = initialStateSpec
  522     ,_csPostState       = initialStateSpec
  523     ,_csArgBindings     = Map.empty
  524     ,_csRetValue        = Nothing
  525     ,_csGlobalAllocs    = []
  526     ,_csLoc             = loc
  527     ,_csCodebase        = code
  528     }
