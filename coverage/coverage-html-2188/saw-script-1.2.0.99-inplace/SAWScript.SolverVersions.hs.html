   26 -- | Given an 'SBV.Solver' from @SBV@, attempt to query the solver for its
   27 -- version and return the result as a string.
   28 -- Adapted from @what4/test/ProbeSolvers.hs@
   29 getSolverVersion :: SBV.Solver -> IO (Maybe String)
   30 getSolverVersion s =
   31   let s' = SBV.solver $ SBV.defaultSolverConfig s
   32       (args, pref) = case SBV.name s' of
   33         -- n.b. abc will return a non-zero exit code if asked for command usage.
   34         SBV.ABC       -> (["s", "-q", "version;quit"], "UC Berkeley, ABC ")
   35         SBV.Boolector -> (["--version"]              , "")
   36         SBV.Bitwuzla  -> (["--version"]              , "")
   37         SBV.CVC4      -> (["--version"]              , "This is CVC4 version ")
   38         SBV.CVC5      -> (["--version"]              , "This is cvc5 version ")
   39         SBV.DReal     -> (["--version"]              , "dReal v")
   40         SBV.MathSAT   -> (["-version"]               , "MathSAT5 version ")
   41         SBV.OpenSMT   -> error "opensmt not currently supported"
   42         SBV.Yices     -> (["--version"]              , "Yices ")
   43         SBV.Z3        -> (["--version"]              , "Z3 version ")
   44   in try (readProcessWithExitCode (SBV.executable s') args "") >>= \case
   45     Right (ExitSuccess,o,_) | (l:_) <- lines o ->
   46       return $ Just $ dropPrefix pref l
   47     Right _                   -> return Nothing
   48     Left (_ :: SomeException) -> return Nothing
   49   where dropPrefix (x:xs) (y:ys) | x == y = dropPrefix xs ys
   50         dropPrefix _ ys = ys

   52 -- | Get the 'SolverBackendVersion' of a 'SolverBackend'
   53 getSolverBackendVersion :: SolverBackend -> IO (Maybe String)
   54 getSolverBackendVersion backend = case backend of
   55   What4     -> return what4Hash
   56   SBV       -> return (Just VERSION_sbv)
   57   AIG       -> return aigHash
   58   RME       -> return rmeHash
   59   -- We use individual cases for the remaining constructors to ensure that if
   60   -- a new backend is added, a warning is generated for this pattern match
   61   ABC       -> getSolverVersion SBV.ABC
   62   Boolector -> getSolverVersion SBV.Boolector
   63   Bitwuzla  -> getSolverVersion SBV.Bitwuzla
   64   CVC4      -> getSolverVersion SBV.CVC4
   65   CVC5      -> getSolverVersion SBV.CVC5
   66   DReal     -> getSolverVersion SBV.DReal
   67   MathSAT   -> getSolverVersion SBV.MathSAT
   68   OpenSMT   -> getSolverVersion SBV.OpenSMT
   69   Yices     -> getSolverVersion SBV.Yices
   70   Z3        -> getSolverVersion SBV.Z3
   71 
   72 -- | Get the set of 'SolverBackendVersions' of a list of 'SolverBackend's
   73 getSolverBackendVersions :: [SolverBackend] -> IO SolverBackendVersions
   74 getSolverBackendVersions backends = Map.fromList <$>
   75   mapM (\backend -> (backend,) <$> getSolverBackendVersion backend) backends
