   48 -- | Convert a type expression from SAW-core to What4
   49 fotToBaseType :: FirstOrderType -> Maybe (Some BaseTypeRepr)
   50 fotToBaseType FOTBit
   51   = Just (Some BaseBoolRepr)
   52 fotToBaseType FOTInt
   53   = Just (Some BaseIntegerRepr)
   54 fotToBaseType (FOTIntMod _n)
   55   = Just (Some BaseIntegerRepr)
   56 fotToBaseType (FOTVec nat FOTBit)
   57   | Just (Some (PosNat nr)) <- somePosNat nat
   58   = Just (Some (BaseBVRepr nr))
   59   | otherwise = Nothing
   60 
   61 fotToBaseType (FOTVec nat fot)
   62   | Just (Some assn) <- listToAssn (replicate (fromIntegral nat) fot)
   63   = Just (Some (BaseStructRepr assn))
   64 fotToBaseType (FOTArray fot1 fot2)
   65   | Just (Some ty1) <- fotToBaseType fot1
   66   , Just (Some ty2) <- fotToBaseType fot2
   67   = Just (Some (BaseArrayRepr (Empty :> ty1) ty2))
   68 fotToBaseType (FOTTuple fots)
   69   | Just (Some assn) <- listToAssn fots
   70   = Just (Some (BaseStructRepr assn))
   71 
   72 -- TODO: convert to What4 records ?!
   73 fotToBaseType _ = Nothing
   74 
   75 listToAssn :: [FirstOrderType] -> Maybe (Some (Assignment BaseTypeRepr))
   76 listToAssn [] = Just (Some empty)
   77 listToAssn (fot:rest) =
   78   case (fotToBaseType fot, listToAssn rest) of
   79     (Just (Some bt), Just (Some assn)) -> Just (Some (extend assn bt))
   80     _ -> Nothing

   83 -- | Convert a type expression from What4 to SAW-core
   84 
   85 typeReprToFOT :: BaseTypeRepr ty -> Either String FirstOrderType
   86 typeReprToFOT BaseBoolRepr            = pure FOTBit
   87 typeReprToFOT BaseIntegerRepr         = pure FOTInt
   88 typeReprToFOT (BaseBVRepr w)          = pure $ FOTVec (natValue w) FOTBit
   89 typeReprToFOT BaseRealRepr            = Left "No FO Real"
   90 typeReprToFOT BaseComplexRepr         = Left "No FO Complex"
   91 typeReprToFOT (BaseStringRepr _)      = Left "No FO String"
   92 typeReprToFOT (BaseArrayRepr (Empty :> ty) b)
   93   | Right fot1 <- typeReprToFOT ty
   94   , Right fot2 <- typeReprToFOT b
   95   = pure $ FOTArray fot1 fot2
   96 typeReprToFOT ty@(BaseArrayRepr _ctx _b) = Left $ "Unsupported FO Array: " ++ show ty
   97 typeReprToFOT (BaseFloatRepr _)       = Left "No FO Floating point"
   98 typeReprToFOT (BaseStructRepr ctx)    = FOTTuple <$> assnToList ctx
   99 
  100 assnToList :: Assignment BaseTypeRepr ctx -> Either String [FirstOrderType]
  101 assnToList = foldrFC g (Right []) where
  102   g :: BaseTypeRepr x -> Either String [FirstOrderType] -> Either String [FirstOrderType]
  103   g ty l = (:) <$> typeReprToFOT ty <*> l

  107 -- | Convert between What4 GroundValues and saw-core FirstOrderValues
  108 
  109 groundToFOV :: BaseTypeRepr ty -> GroundValue ty -> Either String FirstOrderValue
  110 groundToFOV BaseBoolRepr    b         = pure $ FOVBit b
  111 groundToFOV BaseIntegerRepr i         = pure $ FOVInt i
  112 groundToFOV (BaseBVRepr w) bv         = pure $ FOVWord (natValue w) (BV.asUnsigned bv)
  113 groundToFOV BaseRealRepr    _         = Left "Real is not FOV"
  114 groundToFOV BaseComplexRepr         _ = Left "Complex is not FOV"
  115 groundToFOV (BaseStringRepr _)      _ = Left "String is not FOV"
  116 groundToFOV (BaseFloatRepr _)       _ = Left "Floating point is not FOV"
  117 groundToFOV (BaseArrayRepr (Empty :> ty_idx) ty_val) (ArrayMapping _) = do
  118     -- ArrayMapping is an array represented as a function call we can
  119     -- use to extract values. We can't do anything useful with this
  120     -- because we don't have any clue what key values to look at and
  121     -- no good way to figure that out either. So return a placeholder
  122     -- value in FirstOrderValue.
  123     --
  124     -- (The frustrating part is that in many cases the _user_ will
  125     -- know what key values to look at, but has no way to tell us.)
  126     --
  127     -- In principle we could change groundToFOV to return any of an
  128     -- error, a FirstOrderValue, or a separate FunctionArrayValue
  129     -- type, and use that channel to return the actual function
  130     -- handle. (Including the function handle in a FirstOrderValue
  131     -- value is problematic as things stand.) However, there's nothing
  132     -- to be gained by doing this without a way to figure out what
  133     -- key values to look at. And realistically, if we ever _do_
  134     -- come up with a way to figure out what key values to look at,
  135     -- it should be implemented in What4 so What4 never returns
  136     -- ArrayMapping.
  137     --
  138     -- (See Note [FOVArray] in SAWCore.FiniteValue
  139     -- where FirstOrderValue is defined.)
  140     ty_idx' <- typeReprToFOT ty_idx
  141     ty_val' <- typeReprToFOT ty_val
  142     pure $ FOVOpaqueArray ty_idx' ty_val'
  143 groundToFOV (BaseArrayRepr (Empty :> ty_idx) ty_val) (ArrayConcrete dfl values) = do
  144     ty_idx' <- typeReprToFOT ty_idx
  145     dfl' <- groundToFOV ty_val dfl
  146     let convert idx values' v = do
  147           let idx' = indexToFOV idx
  148           v' <- groundToFOV ty_val v
  149           return $ Map.insert idx' v' values'
  150     values' <- ifoldlM convert Map.empty values
  151     pure $ FOVArray ty_idx' dfl' values'
  152 groundToFOV (BaseArrayRepr _ _) _ =
  153     Left "Unsupported FOV array (unexpected key type)"
  154 groundToFOV (BaseStructRepr ctx) tup  = FOVTuple <$> tupleToList ctx tup
  155 
  156 indexToFOV :: Assignment IndexLit (EmptyCtx ::> ty) -> FirstOrderValue
  157 indexToFOV (Empty :> IntIndexLit k) =
  158     FOVInt k
  159 indexToFOV (Empty :> BVIndexLit w k) =
  160     FOVWord (natValue w) (BV.asUnsigned k)
  161

  163 tupleToList :: Assignment BaseTypeRepr ctx ->
  164              Assignment GroundValueWrapper ctx -> Either String [FirstOrderValue]
  165 tupleToList (viewAssign -> AssignEmpty) (viewAssign -> AssignEmpty) = Right []
  166 tupleToList (viewAssign -> AssignExtend rs r) (viewAssign -> AssignExtend gvs gv) =
  167   (:) <$> groundToFOV r (unGVW gv) <*> tupleToList rs gvs
  168 #if !MIN_VERSION_GLASGOW_HASKELL(8,10,0,0)
  169 tupleToList _ _ = error "GADTs should rule this out."
  170 #endif
