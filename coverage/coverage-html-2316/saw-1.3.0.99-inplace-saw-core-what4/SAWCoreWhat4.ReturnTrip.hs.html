   69 data SAWCoreState n
   70   = SAWCoreState
   71     { saw_ctx       :: SC.SharedContext                         -- ^ the main SAWCore datastructure for building shared terms
   72     , saw_inputs    :: IORef (Seq (SC.ExtCns SC.Term ))
   73       -- ^ a record of all the symbolic input variables created so far,
   74       --   in the order they were created
   75 
   76     , saw_symMap :: IORef (Map Word64 (SC.SharedContext -> [SC.Term] -> IO SC.Term))
   77       -- ^ What to do with uninterpreted functions.
   78       -- The key is the "indexValue" of the "symFunId" for the function
   79 
   80     , saw_elt_cache :: B.IdxCache n SAWExpr
   81       -- ^ cache mapping a What4 variable nonce to its corresponding SAWCore term.
   82 
   83     , saw_elt_cache_r :: IORef (IntMap (Some (B.Expr n)))
   84       -- ^ reverse cache mapping a SAWCore TermIndex to its corresponding What4 variable.
   85       -- 'saw_elt_cache' and 'saw_elt_cache_r' implement a bidirectional map between
   86       -- SAWCore terms and What4 variables.
   87 
   88     }

   90 newSAWCoreState ::
   91   SC.SharedContext ->
   92   IO (SAWCoreState n)
   93 newSAWCoreState sc =
   94   do inpr <- newIORef Seq.empty
   95      ch   <- B.newIdxCache
   96      ch_r <- newIORef IntMap.empty
   97      mr   <- newIORef Map.empty
   98      return SAWCoreState
   99             { saw_ctx = sc
  100             , saw_inputs = inpr
  101             , saw_symMap = mr
  102             , saw_elt_cache = ch
  103             , saw_elt_cache_r = ch_r
  104             }

  116 baseSCType ::
  117   sym ->
  118   SC.SharedContext ->
  119   BaseTypeRepr tp ->
  120   IO SC.Term
  121 baseSCType sym sc bt =
  122   case bt of
  123     BaseBoolRepr -> SC.scBoolType sc
  124     BaseBVRepr w -> SC.scBitvector sc $ fromIntegral (natValue w)
  125     BaseIntegerRepr -> SC.scIntegerType sc
  126     BaseArrayRepr indexTypes range
  127       | Ctx.Empty Ctx.:> idx_type <- indexTypes ->
  128         do sc_idx_type <- baseSCType sym sc idx_type
  129            sc_elm_type <- baseSCType sym sc range
  130            SC.scArrayType sc sc_idx_type sc_elm_type
  131       | otherwise ->
  132         unsupported sym "SAW backend does not support multidimensional Arrays: baseSCType"
  133     BaseFloatRepr _ ->
  134       unsupported sym "SAW backend does not support IEEE-754 floating point values: baseSCType"
  135     BaseStringRepr _ ->
  136       unsupported sym "SAW backend does not support string values: baseSCType"
  137     BaseComplexRepr  ->
  138       unsupported sym "SAW backend does not support complex values: baseSCType"
  139     BaseRealRepr     ->
  140       unsupported sym "SAW backend does not support real values: baseSCType"
  141     BaseStructRepr ts ->
  142       SC.scTupleType sc =<< baseSCTypes ts
  143   where
  144     baseSCTypes :: Ctx.Assignment BaseTypeRepr args -> IO [SC.Term]
  145     baseSCTypes Ctx.Empty = return []
  146     baseSCTypes (xs Ctx.:> x) =
  147       do ts <- baseSCTypes xs
  148          t <- baseSCType sym sc x
  149          return (ts ++ [t])

  151 -- | Create a new symbolic variable.
  152 sawCreateVar :: SAWCoreState n
  153              -> Text -- ^ the name of the variable
  154              -> SC.Term
  155              -> IO SC.Term
  156 sawCreateVar st nm tp = do
  157   let sc = saw_ctx st
  158   ec <- SC.scFreshEC sc nm tp
  159   t <- SC.scFlatTermF sc (SC.ExtCns ec)
  160   modifyIORef (saw_inputs st) (\xs -> xs Seq.|> ec)
  161   return t

  163 bindSAWTerm ::
  164   B.ExprBuilder n st fs ->
  165   SAWCoreState n ->
  166   BaseTypeRepr bt ->
  167   SC.Term ->
  168   IO (B.Expr n bt)
  169 bindSAWTerm sym st bt t = do
  170   ch_r <- readIORef $ saw_elt_cache_r st
  171   let midx =
  172         case t of
  173           SC.STApp { SC.stAppIndex = idx } -> Just idx
  174           SC.Unshared _ -> Nothing
  175   case midx >>= flip IntMap.lookup ch_r of
  176     Just (Some var) -> do
  177       Just Refl <- return $ testEquality bt (B.exprType var)
  178       return var
  179     Nothing -> do
  180       sbVar@(B.BoundVarExpr bv) <- freshConstant sym emptySymbol bt
  181       B.insertIdxValue (saw_elt_cache st) (B.bvarId bv) (SAWExpr t)
  182       case midx of
  183         Just i -> modifyIORef' (saw_elt_cache_r st) $ IntMap.insert i (Some sbVar)
  184         Nothing -> pure ()
  185       return sbVar

  187 -- | Register an interpretation for a symbolic function. This is not
  188 -- used during simulation, but rather, when we translate Crucible
  189 -- values back into SAW. The interpretation function takes a list of
  190 -- arguments in regular (left-to-right) order.
  191 sawRegisterSymFunInterp ::
  192   SAWCoreState n ->
  193   B.ExprSymFn n args ret ->
  194   (SC.SharedContext -> [SC.Term] -> IO SC.Term) ->
  195   IO ()
  196 sawRegisterSymFunInterp st f i =
  197   modifyIORef (saw_symMap st) (Map.insert (indexValue (B.symFnId f)) i)

  199 toSC :: B.ExprBuilder n st fs -> SAWCoreState n -> B.Expr n tp -> IO SC.Term
  200 toSC sym st elt = evaluateExpr sym st (saw_ctx st) (saw_elt_cache st) elt

  203 -- | Create a Real SAWELT value from a Rational.
  204 --
  205 -- This fails on non-integer expressions.
  206 scRealLit ::
  207   sym ->
  208   SC.SharedContext ->
  209   Rational ->
  210   IO (SAWExpr BaseRealType)
  211 scRealLit sym sc r
  212   | denominator r /= 1 =
  213     unsupported sym "SAW backend does not support real values"
  214   | otherwise =
  215     IntToRealSAWExpr <$> scIntLit sc (numerator r)
  216 
  217 -- | Create a SAWCore term with type 'Integer' from a Haskell Integer.
  218 scIntLit :: SC.SharedContext -> Integer -> IO (SAWExpr BaseIntegerType)
  219 scIntLit sc i
  220   | i >= 0 =
  221     SAWExpr <$> (SC.scNatToInt sc =<< SC.scNat sc (fromInteger i))
  222   | otherwise =
  223     SAWExpr <$> (SC.scIntNeg sc =<< SC.scNatToInt sc =<< SC.scNat sc (fromInteger (negate i)))
  224 
  225 scBvLit :: SC.SharedContext -> NatRepr w -> BV.BV w -> IO (SAWExpr (BaseBVType w))
  226 scBvLit sc w bv = SAWExpr <$> SC.scBvConst sc (natValue w) (BV.asUnsigned bv)

  319 scIte ::
  320   sym ->
  321   SC.SharedContext ->
  322   BaseTypeRepr tp ->
  323   SAWExpr BaseBoolType ->
  324   SAWExpr tp ->
  325   SAWExpr tp ->
  326   IO (SAWExpr tp)
  327 scIte sym sc tp (SAWExpr p) x y =
  328   case tp of
  329     BaseRealRepr    -> scIteReal sym sc p x y
  330     BaseIntegerRepr -> scIteInt sc p x y
  331     _ ->
  332       do tp' <- baseSCType sym sc tp
  333          x' <- termOfSAWExpr sym sc x
  334          y' <- termOfSAWExpr sym sc y
  335          SAWExpr <$> SC.scIte sc tp' p x' y'

  359 scEq ::
  360   sym ->
  361   SC.SharedContext ->
  362   BaseTypeRepr tp ->
  363   SAWExpr tp ->
  364   SAWExpr tp ->
  365   IO (SAWExpr BaseBoolType)
  366 scEq sym sc tp x y =
  367   case tp of
  368     BaseBoolRepr    -> scBoolEq sc x y
  369     BaseRealRepr    -> scRealEq sym sc x y
  370     BaseIntegerRepr -> scIntEq sc x y
  371     BaseArrayRepr idxTypes range
  372       | Ctx.Empty Ctx.:> idx_type <- idxTypes ->
  373         do let SAWExpr x' = x
  374            let SAWExpr y' = y
  375            sc_idx_type <- baseSCType sym sc idx_type
  376            sc_elm_type <- baseSCType sym sc range
  377            SAWExpr <$> SC.scArrayEq sc sc_idx_type sc_elm_type x' y'
  378       | otherwise ->
  379         unsupported sym ("SAW backend: equality comparison on unsupported multidimensional array type:" ++ show tp)
  380     BaseBVRepr w    ->
  381       do let SAWExpr x' = x
  382          let SAWExpr y' = y
  383          w' <- SC.scNat sc $ fromIntegral (natValue w)
  384          SAWExpr <$> SC.scBvEq sc w' x' y'
  385     _ -> unsupported sym ("SAW backend: equality comparison on unsupported type:" ++ show tp)

  454 termOfSAWExpr ::
  455   sym ->
  456   SC.SharedContext ->
  457   SAWExpr tp -> IO SC.Term
  458 termOfSAWExpr sym _sc expr =
  459   case expr of
  460     SAWExpr t -> return t
  461     IntToRealSAWExpr _
  462       -> unsupported sym "SAW backend does not support real values"

  464 applyExprSymFn ::
  465   forall n st fs args ret.
  466   B.ExprBuilder n st fs ->
  467   SAWCoreState n ->
  468   SC.SharedContext ->
  469   B.ExprSymFn n args ret ->
  470   Ctx.Assignment SAWExpr args ->
  471   IO (SAWExpr ret)
  472 applyExprSymFn sym st sc fn args =
  473   do mp <- readIORef (saw_symMap st)
  474      mk <-
  475        case Map.lookup (indexValue (B.symFnId fn)) mp of
  476          Nothing -> panic "applyExprSymFn" [
  477                         "Unknown symbolic function " <> Text.pack (show fn)
  478                     ]
  479          Just mk -> return mk
  480      ts <- evaluateAsgn args
  481      SAWExpr <$> mk sc (reverse ts)
  482   where
  483     evaluateAsgn :: Ctx.Assignment SAWExpr args' -> IO [SC.Term]
  484     evaluateAsgn Ctx.Empty = return []
  485     evaluateAsgn (xs Ctx.:> x) =
  486       do vs <- evaluateAsgn xs
  487          v <- termOfSAWExpr sym sc x
  488          return (v : vs)

  490 {- | Declare that we don't support something or other.
  491 This aborts the current path of execution, and adds a proof
  492 obligation to ensure that we won't get there.
  493 These proof obligations are all tagged with "Unsupported", so that
  494 users of the library can choose if they will try to discharge them,
  495 fail in some other way, or just ignore them. -}
  496 unsupported :: {- OnlineSolver solver => -} sym -> String -> IO a
  497 unsupported _sym x = fail ("Unsupported " <> x) -- TODO, something better here....

  500 evaluateExpr :: forall n st tp fs.
  501   B.ExprBuilder n st fs->
  502   SAWCoreState n ->
  503   SC.SharedContext ->
  504   B.IdxCache n SAWExpr ->
  505   B.Expr n tp ->
  506   IO SC.Term
  507 evaluateExpr sym st sc cache = f []
  508   where
  509     -- Evaluate the element, and expect the result to have the same type.
  510     f :: [Maybe SolverSymbol] -> B.Expr n tp' -> IO SC.Term
  511     f env elt = termOfSAWExpr sym sc =<< eval env elt
  512 
  513     eval :: [Maybe SolverSymbol] -> B.Expr n tp' -> IO (SAWExpr tp')
  514     eval env elt = B.idxCacheEval cache elt (go env elt)
  515 
  516     realFail :: IO a
  517     realFail = unsupported sym "SAW backend does not support real values"
  518 
  519     cplxFail :: IO a
  520     cplxFail = unsupported sym "SAW backend does not support complex values"
  521 
  522     floatFail :: IO a
  523     floatFail = unsupported sym "SAW backend does not support floating-point values"
  524 
  525     stringFail :: IO a
  526     stringFail = unsupported sym "SAW backend does not support string values"
  527 
  528     unimplemented :: String -> IO a
  529     unimplemented x = unsupported sym $ "SAW backend: not implemented: " ++ x
  530 
  531     go :: [Maybe SolverSymbol] -> B.Expr n tp' -> IO (SAWExpr tp')
  532 
  533     go _ (B.BoolExpr b _) = SAWExpr <$> SC.scBool sc b
  534 
  535     go _ (B.SemiRingLiteral sr x _) =
  536       case sr of
  537         B.SemiRingBVRepr _ w  -> scBvLit sc w x
  538         B.SemiRingIntegerRepr -> scIntLit sc x
  539         B.SemiRingRealRepr    -> scRealLit sym sc x
  540 
  541     go _ (B.StringExpr{}) =
  542       unsupported sym "SAW backend does not support string values"
  543 
  544     go _ (B.FloatExpr{}) =
  545       unsupported sym "SAW backend does not support floating-point values"
  546 
  547     go env (B.BoundVarExpr bv) =
  548       case B.bvarKind bv of
  549         B.UninterpVarKind -> do
  550           tp <- baseSCType sym sc (B.bvarType bv)
  551           SAWExpr <$> sawCreateVar st nm tp
  552             where nm = solverSymbolAsText $ B.bvarName bv
  553         B.LatchVarKind ->
  554           unsupported sym "SAW backend does not support latch variables"
  555         B.QuantifierVarKind -> do
  556           case elemIndex (Just $ B.bvarName bv) env of
  557             Nothing -> unsupported sym $ "unbound quantifier variable " <> nm
  558             Just idx -> SAWExpr <$> SC.scLocalVar sc idx
  559             where nm = Text.unpack $ solverSymbolAsText $ B.bvarName bv
  560 
  561     go env (B.NonceAppExpr p) =
  562       case B.nonceExprApp p of
  563         B.Annotation _tpr _n x ->
  564           eval env x
  565 
  566         B.Forall bvar body ->
  567           case B.bvarType bvar of
  568             BaseBVRepr wrepr -> do
  569               w <- SC.scNat sc $ natValue wrepr
  570               ty <- SC.scVecType sc w =<< SC.scBoolType sc
  571               SAWExpr <$>
  572                 (SC.scBvForall sc w
  573                  =<< SC.scLambda sc nm ty =<< f (Just (B.bvarName bvar):env) body)
  574               where nm = solverSymbolAsText $ B.bvarName bvar
  575             _ -> unsupported sym "SAW backend only supports universal quantifiers over bitvectors"
  576         B.Exists{} ->
  577           unsupported sym "SAW backend does not support existential quantifiers"
  578         B.ArrayFromFn{} -> unimplemented "ArrayFromFn"
  579         B.MapOverArrays{} -> unimplemented "MapOverArrays"
  580         B.ArrayTrueOnEntries{} -> unimplemented "ArrayTrueOnEntries"
  581         B.FnApp fn asgn ->
  582           do args <- traverseFC (eval env) asgn
  583              applyExprSymFn sym st sc fn args
  584 
  585     go env a0@(B.AppExpr a) =
  586       let nyi = unsupported sym $
  587                   "Expression form not yet implemented in SAWCore backend:\n"
  588                         ++ show a0
  589       in
  590       case B.appExprApp a of
  591         B.BaseIte bt _ c xe ye -> join (scIte sym sc bt <$> eval env c <*> eval env xe <*> eval env ye)
  592         B.BaseEq bt xe ye -> join (scEq sym sc bt <$> eval env xe <*> eval env ye)
  593 
  594         B.SemiRingLe sr xe ye ->
  595           case sr of
  596             B.OrderedSemiRingRealRepr    -> join (scRealLe sym sc <$> eval env xe <*> eval env ye)
  597             B.OrderedSemiRingIntegerRepr -> join (scIntLe sc <$> eval env xe <*> eval env ye)
  598 
  599         B.NotPred x ->
  600           goNeg env x
  601 
  602         B.ConjPred xs ->
  603           case BM.viewConjMap xs of
  604             BM.ConjTrue -> SAWExpr <$> SC.scBool sc True
  605             BM.ConjFalse -> SAWExpr <$> SC.scBool sc False
  606             BM.Conjuncts (t:|ts) ->
  607               let pol (x,BM.Positive) = f env x
  608                   pol (x,BM.Negative) = termOfSAWExpr sym sc =<< goNeg env x
  609               in SAWExpr <$> join (foldM (SC.scAnd sc) <$> pol t <*> mapM pol ts)
  610 
  611         B.SemiRingProd pd ->
  612            case WSum.prodRepr pd of
  613              B.SemiRingRealRepr ->
  614                do pd' <- WSum.prodEvalM (scMulReal sym sc) (eval env) pd
  615                   maybe (scRealLit sym sc 1) return pd'
  616              B.SemiRingIntegerRepr ->
  617                do pd' <- WSum.prodEvalM (scMulInt sc) (eval env) pd
  618                   maybe (scIntLit sc 1) return pd'
  619              B.SemiRingBVRepr B.BVArithRepr w ->
  620                do n <- SC.scNat sc (natValue w)
  621                   pd' <- WSum.prodEvalM (SC.scBvMul sc n) (f env) pd
  622                   maybe (scBvLit sc w (BV.one w)) (return . SAWExpr) pd'
  623              B.SemiRingBVRepr B.BVBitsRepr w ->
  624                do n <- SC.scNat sc (natValue w)
  625                   pd' <- WSum.prodEvalM (SC.scBvAnd sc n) (f env) pd
  626                   maybe (scBvLit sc w (BV.maxUnsigned w)) (return . SAWExpr) pd'
  627 
  628         B.SemiRingSum ss ->
  629           case WSum.sumRepr ss of
  630             B.SemiRingRealRepr -> WSum.evalM add smul (scRealLit sym sc) ss
  631                where add x y = scAddReal sym sc x y
  632                      smul 1  e = eval env e
  633                      smul sm e = join $ scMulReal sym sc <$> scRealLit sym sc sm <*> eval env e
  634             B.SemiRingIntegerRepr -> WSum.evalM add smul (scIntLit sc) ss
  635                where add x y = scAddInt sc x y
  636                      smul 1  e = eval env e
  637                      smul sm e = join $ scMulInt sc <$> scIntLit sc sm <*> eval env e
  638             B.SemiRingBVRepr B.BVArithRepr w -> WSum.evalM add smul (scBvLit sc w) ss
  639                where add x y          = scBvAdd sc w x y
  640                      smul (BV.BV 1) e = eval env e
  641                      smul sm e        = join (scBvMul sc w <$> scBvLit sc w sm <*> eval env e)
  642             B.SemiRingBVRepr B.BVBitsRepr w
  643                | ss^.WSum.sumOffset == one -> scBvNot sc w =<< bitwise_eval (ss & WSum.sumOffset .~ BV.zero w)
  644                | otherwise -> bitwise_eval ss
  645 
  646               where one = BV.maxUnsigned w
  647                     bitwise_eval = WSum.evalM add smul (scBvLit sc w)
  648                     add x y = scBvXor sc w x y
  649                     smul sm e
  650                        | sm == one = eval env e
  651                        | otherwise = join (scBvAnd sc w <$> scBvLit sc w sm <*> eval env e)
  652 
  653         B.RealIsInteger{} -> unsupported sym "SAW backend does not support real values"
  654 
  655         B.BVOrBits w bs ->
  656           do n <- SC.scNat sc (natValue w)

  659                [] -> scBvLit sc w (BV.zero w)

  664              n <- SC.scNat sc (natValue w)

  669              w <- SC.scNat sc (natValue (bvWidth bv))
  670              bit <- SC.scBoolType sc
  671              -- NB, SAWCore's `scAt` is big endian

  692            -- need to subtract 1 to make the types match.

  694            join (SC.scBvSDiv sc n <$> f env x <*> f env y)
  695         B.BVSrem _ x y -> fmap SAWExpr $ do
  696            -- NB: bvSDiv applies 'Succ' to its width argument, so we
  697            -- need to subtract 1 to make the types match.
  698            n <- SC.scNat sc (natValue (bvWidth x) - 1)
  699            join (SC.scBvSRem sc n <$> f env x <*> f env y)
  700         B.BVShl _ x y -> fmap SAWExpr $ do
  701            let w = natValue (bvWidth x)

  704         B.BVLshr _ x y -> fmap SAWExpr $ do

  706            n <- SC.scNat sc w

  708         B.BVAshr _ x y -> fmap SAWExpr $ do
  709            let w = natValue (bvWidth x)
  710            -- NB: bvSShr applies a `Succ` to its width argument, so we subtract
  711            --     1 here to make things match up.
  712            n <- SC.scNat sc (w - 1)
  713            join (SC.scBvSShr sc n <$> f env x <*> (SC.scBvToNat sc w =<< f env y))
  714         B.BVRol w x y -> fmap SAWExpr $ do
  715            n <- SC.scNat sc (natValue w)
  716            bit <- SC.scBoolType sc
  717            x' <- f env x
  718            y' <- SC.scBvToNat sc (natValue w) =<< f env y
  719            SC.scGlobalApply sc (SC.mkIdent SC.preludeName "rotateL") [n,bit,x',y']
  720         B.BVRor w x y -> fmap SAWExpr $ do
  721            n <- SC.scNat sc (natValue w)
  722            bit <- SC.scBoolType sc
  723            x' <- f env x
  724            y' <- SC.scBvToNat sc (natValue w) =<< f env y
  725            SC.scGlobalApply sc (SC.mkIdent SC.preludeName "rotateR") [n,bit,x',y']
  726         B.BVConcat _ x y -> fmap SAWExpr $ do
  727            m <- SC.scNat sc (natValue (bvWidth x))
  728            n <- SC.scNat sc (natValue (bvWidth y))
  729            t <- SC.scBoolType sc
  730            join (SC.scAppend sc m n t <$> f env x <*> f env y)
  731         B.BVSelect start num x -> fmap SAWExpr $ do
  732            i <- SC.scNat sc (natValue (bvWidth x) - natValue num - natValue start)
  733            n <- SC.scNat sc (natValue num)
  734            o <- SC.scNat sc (natValue start)
  735            t <- SC.scBoolType sc
  736            x' <- f env x
  737            SC.scSlice sc t i n o x'
  738         B.BVZext w' x -> fmap SAWExpr $ do
  739           let w = bvWidth x
  740           n <- SC.scNat sc (natValue w)
  741           m <- SC.scNat sc (natValue w' - natValue w)
  742           x' <- f env x
  743           SC.scBvUExt sc m n x'
  744         B.BVSext w' x -> fmap SAWExpr $ do
  745           let w = bvWidth x
  746           -- NB: width - 1 to make SAWCore types work out
  747           n <- SC.scNat sc (natValue w - 1)
  748           m <- SC.scNat sc (natValue w' - natValue w)
  749           x' <- f env x
  750           SC.scBvSExt sc m n x'
  751         B.BVPopcount w x ->
  752           do n  <- SC.scNat sc (natValue w)
  753              x' <- f env x

  936     -- returns the logical negation of the result of 'go'
  937     -- negations are pushed inside conjunctions and less-than-or-equal
  938     goNeg :: [Maybe SolverSymbol] -> B.Expr n BaseBoolType -> IO (SAWExpr BaseBoolType)
  939     goNeg env expr =
  940       case expr of
  941         -- negation of (x /\ y) becomes (~x \/ ~y)
  942         B.AppExpr (B.appExprApp -> B.ConjPred xs) ->
  943           case BM.viewConjMap xs of
  944             BM.ConjTrue -> SAWExpr <$> SC.scBool sc False
  945             BM.ConjFalse -> SAWExpr <$> SC.scBool sc True
  946             BM.Conjuncts (t:|ts) ->
  947               let pol (x, BM.Positive) = termOfSAWExpr sym sc =<< goNegAtom env x
  948                   pol (x, BM.Negative) = f env x
  949               in SAWExpr <$> join (foldM (SC.scOr sc) <$> pol t <*> mapM pol ts)
  950         _ -> goNegAtom env expr
  951 
  952     -- returns the logical negation of the result of 'go'
  953     -- negations are pushed inside less-than-or-equal
  954     goNegAtom :: [Maybe SolverSymbol] -> B.Expr n BaseBoolType -> IO (SAWExpr BaseBoolType)
  955     goNegAtom env expr =
  956       case expr of
  957         -- negation of (x <= y) becomes (y < x)
  958         B.AppExpr (B.appExprApp -> B.SemiRingLe sr xe ye) ->
  959           case sr of
  960             B.OrderedSemiRingRealRepr    -> join (scRealLt sym sc <$> eval env ye <*> eval env xe)
  961             B.OrderedSemiRingIntegerRepr -> join (scIntLt sc <$> eval env ye <*> eval env xe)
  962         _ -> SAWExpr <$> (SC.scNot sc =<< f env expr)
  963
