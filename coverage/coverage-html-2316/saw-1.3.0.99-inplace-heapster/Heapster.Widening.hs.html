   64 -- | A sequence of permissions for some list of variables that extends @vars@
   65 data ExtVarPerms vars
   66   = ExtVarPerms_Base (ValuePerms vars)
   67   | forall var. ExtVarPerms_Mb (TypeRepr var) (Binding var
   68                                                (ExtVarPerms (vars :> var)))
   69 
   70 $(mkNuMatching [t| forall vars. ExtVarPerms vars |])

   72 extVarPerms_MbMulti :: CruCtx ctx -> Mb ctx (ExtVarPerms (vars :++: ctx)) ->
   73                        ExtVarPerms vars
   74 extVarPerms_MbMulti CruCtxNil mb_ewid = elimEmptyMb mb_ewid
   75 extVarPerms_MbMulti (CruCtxCons ctx tp) mb_ewid =
   76   extVarPerms_MbMulti ctx $
   77   fmap (ExtVarPerms_Mb tp) $ mbSeparate (MNil :>: Proxy) mb_ewid

   83 -- | A map from free variables to their permissions and whether they have been
   84 -- \"visited\" yet
   85 type WidNameMap = NameMap (Product ValuePerm (Constant Bool))
   86 
   87 -- | Modify the entry in a 'WidNameMap' associated with a particular free
   88 -- variable, starting from the default entry of @('ValPerm_True','False')@ if
   89 -- the variable has not been entered into the map yet
   90 wnMapAlter :: (Product ValuePerm (Constant Bool) a ->
   91                Product ValuePerm (Constant Bool) a) -> ExprVar a ->
   92               WidNameMap -> WidNameMap
   93 wnMapAlter f =
   94   NameMap.alter $ \case
   95   Just entry -> Just $ f entry
   96   Nothing -> Just $ f (Pair ValPerm_True (Constant False))
   97 
   98 -- | Look up the permission for a name in a 'WidNameMap'
   99 wnMapGetPerm :: Name a -> WidNameMap -> ValuePerm a
  100 wnMapGetPerm n nmap | Just (Pair p _) <- NameMap.lookup n nmap = p
  101 wnMapGetPerm _ _ = ValPerm_True
  102 
  103 -- | Build an 'ExtVarPermsFun' from a widening name map
  104 wnMapExtWidFun :: WidNameMap -> ExtVarPermsFun vars
  105 wnMapExtWidFun wnmap =
  106   ExtVarPermsFun $ \ns -> ExtVarPerms_Base $ RL.map (flip wnMapGetPerm wnmap) ns
  107 
  108 -- | Assign the trivial @true@ permission to any variable that has not yet been
  109 -- visited
  110 wnMapDropUnvisiteds :: WidNameMap -> WidNameMap
  111 wnMapDropUnvisiteds =
  112   NameMap.map $ \case
  113   p@(Pair _ (Constant True)) -> p
  114   (Pair _ (Constant False)) -> Pair ValPerm_True (Constant False)
  115 
  116 -- | Look up a variable of block type in a 'WidNameMap' to see if it has an
  117 -- associated @shape(sh)@ permission
  118 wnMapBlockShape :: WidNameMap -> ExprVar (LLVMBlockType w) ->
  119                    Maybe (PermExpr (LLVMShapeType w))
  120 wnMapBlockShape nmap n =
  121   NameMap.lookup n nmap >>= \case
  122   Pair (ValPerm_Conj1 (Perm_LLVMBlockShape sh)) _ -> Just sh
  123   _ -> Nothing

  125 newtype PolyContT r m a =
  126   PolyContT { runPolyContT :: forall x. (forall y. a -> m (r y)) -> m (r x) }
  127 
  128 instance Functor (PolyContT r m) where
  129   fmap f m = m >>= return . f
  130 
  131 instance Applicative (PolyContT r m) where
  132   pure x = PolyContT $ \k -> k x
  133   (<*>) = ap
  134 
  135 instance Monad (PolyContT r m) where
  136   return = pure
  137   (PolyContT m) >>= f =
  138     PolyContT $ \k -> m $ \a -> runPolyContT (f a) k

  151 runWideningM :: WideningM () -> DebugLevel -> PermEnv -> RAssign Name args ->
  152                 ExtVarPerms args
  153 runWideningM m dlevel env =
  154   applyExtVarPermsFun $ runIdentity $
  155   runPolyContT (runStateT m $
  156                 WidState NameMap.empty emptyPPInfo env dlevel RecNone)
  157   (Identity . wnMapExtWidFun . _wsNameMap . snd)

  159 openMb :: CruCtx ctx -> Mb ctx a -> WideningM (RAssign Name ctx, a)
  160 openMb ctx mb_a =
  161   lift $ PolyContT $ \k -> Identity $ ExtVarPermsFun $ \ns ->
  162   extVarPerms_MbMulti ctx $
  163   mbMap2 (\ns' a ->
  164            applyExtVarPermsFun (runIdentity $ k (ns',a)) (RL.append ns ns'))
  165   (nuMulti (cruCtxProxies ctx) id) mb_a

  167 bindFreshVar :: TypeRepr tp -> WideningM (ExprVar tp)
  168 bindFreshVar tp =
  169   (snd <$> openMb (singletonCruCtx tp) (nu id)) >>= \n ->
  170   setVarNameM "var" n >>
  171   return n

  173 visitM :: ExprVar a -> WideningM ()
  174 visitM n = modify $ over wsNameMap $ wnMapAlter (\(Pair p _) ->
  175                                                   Pair p (Constant True)) n

  177 isVisitedM :: ExprVar a -> WideningM Bool
  178 isVisitedM n =
  179   maybe False (\(Pair _ (Constant b)) -> b) <$>
  180   NameMap.lookup n <$> view wsNameMap <$> get

  182 getVarPermM :: ExprVar a -> WideningM (ValuePerm a)
  183 getVarPermM n = wnMapGetPerm n <$> view wsNameMap <$> get

  185 setVarPermM :: ExprVar a -> ValuePerm a -> WideningM ()
  186 setVarPermM n p =
  187   modify $ over wsNameMap $ wnMapAlter (\(Pair _ isv) -> Pair p isv) n

  189 -- | Set the permissions for @x &+ off@ to @p@, by setting the permissions for
  190 -- @x@ to @p - off@
  191 setOffVarPermM :: ExprVar a -> PermOffset a -> ValuePerm a -> WideningM ()
  192 setOffVarPermM x off p =
  193   setVarPermM x (offsetPerm (negatePermOffset off) p)

  195 setVarPermsM :: RAssign Name ctx -> RAssign ValuePerm ctx -> WideningM ()
  196 setVarPermsM MNil MNil = return ()
  197 setVarPermsM (ns :>: n) (ps :>: p) = setVarPermsM ns ps >> setVarPermM n p

  226 -- | Test if an expression in a binding is a free variable plus offset
  227 mbAsOffsetVar :: KnownCruCtx vars -> Mb vars (PermExpr a) ->
  228                  Maybe (Name a, PermOffset a)
  229 mbAsOffsetVar vars [nuP| PExpr_Var mb_x |]
  230   | Right n <- mbNameBoundP mb_x = Just (n, NoPermOffset)
  231 mbAsOffsetVar vars [nuP| PExpr_LLVMOffset mb_x mb_off |]
  232   | Right n <- mbNameBoundP mb_x
  233   , Just off <- partialSubst (emptyPSubst vars) mb_off
  234   = Just (n, LLVMPermOffset off)
  235 mbAsOffsetVar _ _ = Nothing

  238 -- | Take a permission @p1@ at some existing location and split it into some
  239 -- @p1'*p1''@ such that @p1'@ will remain at the existing location and @p1''@
  240 -- will be widened against @p1''@. Return @p1'@ and the result of widening
  241 -- @p1''@ against @p2@.
  242 splitWidenPerm :: TypeRepr a -> ValuePerm a -> ValuePerm a ->
  243                   WideningM (ValuePerm a, ValuePerm a)
  244 splitWidenPerm tp p1 p2
  245   | permIsCopyable p1 = (p1,) <$> widenPerm tp p1 p2
  246 splitWidenPerm _ p1 _ = return (p1, ValPerm_True)
  247 
  248 -- | Take permissions @p1@ and @p2@ that are both on existing locations and
  249 -- split them both into @p1'*p1''@ and @p2'*p2''@ such that @p1'@ and @p2'@
  250 -- remain at the existing locations and @p1''@ and @p2''@ are widened against
  251 -- each other. Return @p1'@, @p2'@, and the result of the further widening of
  252 -- @p1''@ against @p2''@.
  253 doubleSplitWidenPerm :: TypeRepr a -> ValuePerm a -> ValuePerm a ->
  254                         WideningM ((ValuePerm a, ValuePerm a), ValuePerm a)
  255 doubleSplitWidenPerm tp p1 p2
  256   | permIsCopyable p1 && permIsCopyable p2
  257   = ((p1,p2),) <$> widenPerm tp p1 p2
  258 doubleSplitWidenPerm _ p1 p2 =
  259   return ((p1, p2), ValPerm_True)

  262 -- | Replace all variables @x@ in an expression or permission that have an
  263 -- equality permission @eq(e)@ with the expression @e@
  264 substEqVars :: Substable PermSubst a Identity => PermPretty a =>
  265                AbstractVars a => FreeVars a => WidNameMap -> a -> a
  266 substEqVars wnmap a
  267   | AbsObj vars cl_mb_a <- abstractFreeVars a =
  268     subst (substOfExprs $
  269            RL.map (\x -> case wnMapGetPerm x wnmap of
  270                       ValPerm_Eq e -> substEqVars wnmap e
  271                       _ -> PExpr_Var x)
  272            vars) $
  273     unClosed cl_mb_a

  276 -- | Widen two expressions against each other
  277 --
  278 -- FIXME: document this more
  279 widenExpr :: TypeRepr a -> PermExpr a -> PermExpr a -> WideningM (PermExpr a)
  280 widenExpr tp e1 e2 =
  281   traceM (\i ->
  282            fillSep [pretty "widenExpr", permPretty i e1, permPretty i e2]) >>
  283   widenExpr' tp e1 e2
  284 
  285 widenExpr' :: TypeRepr a -> PermExpr a -> PermExpr a -> WideningM (PermExpr a)
  286 
  287 -- If both sides are equal, return one of the sides
  288 widenExpr' _ e1 e2 | e1 == e2 = return e1
  289 
  290 -- If both sides are variables, look up their permissions and whether they have
  291 -- been visited and use that information to decide what to do
  292 widenExpr' tp e1@(asVarOffset -> Just (x1, off1)) e2@(asVarOffset ->
  293                                                      Just (x2, off2)) =
  294   do p1 <- getVarPermM x1
  295      p2 <- getVarPermM x2
  296      isv1 <- isVisitedM x1
  297      isv2 <- isVisitedM x2
  298      case (p1, p2, isv1, isv2) of
  299 
  300        -- If we have the same variable with the same offsets (it can avoid the
  301        -- case above of e1 == e2 if the offsets are offsetsEq but not ==) then
  302        -- we are done, though we do want to visit the variable
  303        _ | x1 == x2 && offsetsEq off1 off2 ->
  304            visitM x1 >> return e1
  305 
  306        -- If we have the same variable but different offsets, we widen them
  307        -- using 'widenOffsets'. Note that we cannot have the same variable
  308        -- x on both sides unless they have been visited, so we can safely
  309        -- ignore the isv1 and isv2 flags. The complexity of having these two
  310        -- cases is to find the BVType of one of off1 or off2; because the
  311        -- previous case did not match, we know at least one is LLVMPermOffset.
  312        _ | x1 == x2, LLVMPermOffset (exprType -> off_tp) <- off1 ->
  313            PExpr_LLVMOffset x1 <$> widenOffsets off_tp off1 off2
  314        _ | x1 == x2, LLVMPermOffset (exprType -> off_tp) <- off2 ->
  315            PExpr_LLVMOffset x1 <$> widenOffsets off_tp off1 off2
  316 
  317        -- If a variable has an eq(e) permission, replace it with e and recurse
  318        (ValPerm_Eq e1', _, _, _) ->
  319          visitM x1 >> widenExpr tp (offsetExpr off1 e1') e2
  320        (_, ValPerm_Eq e2', _, _) ->
  321          visitM x2 >> widenExpr tp e1 (offsetExpr off2 e2')
  322 
  323        -- If both variables have been visited and are not equal and do not have
  324        -- eq permissions, then they are equal to different locations elsewhere
  325        -- in our widening, and so this location should not be equated to either
  326        -- of them; thus we make a fresh variable
  327        (_, _, True, True) ->
  328          do x <- bindFreshVar tp
  329             visitM x
  330             ((p1',p2'), p') <-
  331               doubleSplitWidenPerm tp (offsetPerm off1 p1) (offsetPerm off2 p2)
  332             setOffVarPermM x1 off1 p1'
  333             setOffVarPermM x2 off2 p2'
  334             setVarPermM x p'
  335             return $ PExpr_Var x
  336 
  337        -- If only one variable has been visited, its perms need to be split
  338        -- between its other location(s) and here
  339        (_, _, True, _) ->
  340          do (p1', p2') <-
  341               splitWidenPerm tp (offsetPerm off1 p1) (offsetPerm off2 p2)
  342             setVarPermM x1 (offsetPerm (negatePermOffset off1) p1')
  343             setVarPermM x2 (offsetPerm (negatePermOffset off2) p2')
  344             return e2
  345        (_, _, _, True) ->
  346          do (p2', p1') <-
  347               splitWidenPerm tp (offsetPerm off2 p2) (offsetPerm off1 p2)
  348             setVarPermM x1 (offsetPerm (negatePermOffset off1) p1')
  349             setVarPermM x2 (offsetPerm (negatePermOffset off2) p2')
  350             return e1
  351 
  352        -- If we get here, then neither x1 nor x2 has been visited, so choose x1,
  353        -- set x2 equal to x1 &+ (off1 - off2), and set x1's permissions to be
  354        -- the result of widening p1 against p2
  355        _ ->
  356          do visitM x1 >> visitM x2
  357             setVarPermM x2 (ValPerm_Eq $
  358                             offsetExpr (addPermOffsets off1 $
  359                                         negatePermOffset off2) $ PExpr_Var x1)
  360             p' <- widenPerm tp (offsetPerm off1 p1) (offsetPerm off2 p2)
  361             setVarPermM x1 (offsetPerm (negatePermOffset off1) p')
  362             return e1
  363 
  364 
  365 -- If one side is a variable x and the other is not, then the non-variable side
  366 -- cannot have any permissions, and there are fewer cases than the above
  367 widenExpr' tp (asVarOffset -> Just (x1, off1)) e2 =
  368   do p1 <- getVarPermM x1
  369      case p1 of
  370 
  371        -- If x1 has an eq(e) permission, replace it with e and recurse
  372        ValPerm_Eq e1' ->
  373          visitM x1 >> widenExpr tp (offsetExpr off1 e1') e2
  374 
  375        -- Otherwise bind a fresh variable, because even if x1 has not been
  376        -- visited before, it could still occur somewhere we haven't visited yet
  377        _ ->
  378          do x <- bindFreshVar tp
  379             visitM x
  380             return $ PExpr_Var x
  381 
  382 -- Similar to the previous case, but with the variable on the right
  383 widenExpr' tp e1 (asVarOffset -> Just (x2, off2)) =
  384   do p2 <- getVarPermM x2
  385      case p2 of
  386 
  387        -- If x2 has an eq(e) permission, replace it with e and recurse
  388        ValPerm_Eq e2' ->
  389          visitM x2 >> widenExpr tp e1 (offsetExpr off2 e2')
  390 
  391        -- Otherwise bind a fresh variable, because even if x1 has not been
  392        -- visited before, it could still occur somewhere we haven't visited yet
  393        _ ->
  394          do x <- bindFreshVar tp
  395             visitM x
  396             return $ PExpr_Var x

  406 -- Widen named shapes with the same names
  407 --
  408 -- FIXME: we currently only handle shapes with no modalities, because the
  409 -- modalities only come about when proving equality shapes, which themselves are
  410 -- only really used by memcpy and similar functions
  411 widenExpr' _ (PExpr_NamedShape Nothing Nothing nmsh1 args1)
  412   (PExpr_NamedShape Nothing Nothing nmsh2 args2)
  413   | Just (Refl,Refl) <- namedShapeEq nmsh1 nmsh2 =
  414     PExpr_NamedShape Nothing Nothing nmsh1 <$>
  415     widenExprs (namedShapeArgs nmsh1) args1 args2

  417 widenExpr' (LLVMShapeRepr w) (PExpr_EqShape len1 e1) (PExpr_EqShape len2 e2)
  418   | bvEq len1 len2
  419   = PExpr_EqShape len1 <$> widenExpr (LLVMBlockRepr w) e1 e2
  420 
  421 widenExpr' tp (PExpr_PtrShape Nothing Nothing sh1)
  422   (PExpr_PtrShape Nothing Nothing sh2) =
  423   PExpr_PtrShape Nothing Nothing <$> widenExpr tp sh1 sh2

  425 widenExpr' _ (PExpr_FieldShape (LLVMFieldShape p1)) (PExpr_FieldShape
  426                                                     (LLVMFieldShape p2))
  427   | Just Refl <- testEquality (exprLLVMTypeWidth p1) (exprLLVMTypeWidth p2) =
  428     PExpr_FieldShape <$> LLVMFieldShape <$> widenPerm knownRepr p1 p2
  429 
  430 -- Array shapes can be widened if they have the same length and stride
  431 widenExpr' _ (PExpr_ArrayShape
  432               len1 stride1 sh1) (PExpr_ArrayShape len2 stride2 sh2)
  433   | bvEq len1 len2 && stride1 == stride2 =
  434     PExpr_ArrayShape len1 stride1 <$> widenExpr knownRepr sh1 sh2
  435 
  436 -- An array shape of length 1 can be replaced by its sole cell
  437 widenExpr' tp (PExpr_ArrayShape len1 _ sh1) sh2
  438   | bvEq len1 (bvInt 1) = widenExpr' tp sh1 sh2
  439 widenExpr' tp sh1 (PExpr_ArrayShape len2 _ sh2)
  440   | bvEq len2 (bvInt 1) = widenExpr' tp sh1 sh2
  441 
  442 -- FIXME: there should be some check that the first shapes have the same length,
  443 -- though this is more complex if they might have free variables...?
  444 widenExpr' tp (PExpr_SeqShape sh1 sh1') (PExpr_SeqShape sh2 sh2') =
  445   PExpr_SeqShape <$> widenExpr tp sh1 sh2 <*> widenExpr tp sh1' sh2'
  446 
  447 widenExpr' tp (PExpr_OrShape sh1 sh1') (PExpr_OrShape sh2 sh2') =
  448   PExpr_OrShape <$> widenExpr tp sh1 sh2 <*> widenExpr tp sh1' sh2'
  449 
  450 widenExpr' tp (PExpr_ExShape mb_sh1) sh2 =
  451   do x <- bindFreshVar knownRepr
  452      widenExpr tp (varSubst (singletonVarSubst x) mb_sh1) sh2
  453 
  454 widenExpr' tp sh1 (PExpr_ExShape mb_sh2) =
  455   do x <- bindFreshVar knownRepr
  456      widenExpr tp sh1 (varSubst (singletonVarSubst x) mb_sh2)
  457 
  458 -- For two shapes that don't match any of the above cases, return the most
  459 -- general shape, which is the empty shape
  460 widenExpr' (LLVMShapeRepr _) _ _ = return $ PExpr_EmptyShape

  462 -- NOTE: this assumes that permission expressions only occur in covariant

  483 -- | Widen two bitvector offsets by trying to widen them additively
  484 -- ('widenBVsAddy'), or if that is not possible, by widening them
  485 -- multiplicatively ('widenBVsMulty')
  486 widenOffsets :: (1 <= w, KnownNat w) => TypeRepr (BVType w) ->
  487                 PermOffset (LLVMPointerType w) ->
  488                 PermOffset (LLVMPointerType w) ->
  489                 WideningM (PermExpr (BVType w))
  490 widenOffsets tp (llvmPermOffsetExpr -> off1) (llvmPermOffsetExpr -> off2) =
  491   widenBVsAddy tp off1 off2 >>= maybe (widenBVsMulty tp off1 off2) return
  492 
  493 -- | Widen two bitvectors @bv1@ and @bv2@ additively, i.e. bind a fresh
  494 -- variable @bv@ and return @(bv2 - bv1) * bv + bv1@, assuming @bv2 - bv1@
  495 -- is a constant
  496 widenBVsAddy :: (1 <= w, KnownNat w) => TypeRepr (BVType w) ->
  497                 PermExpr (BVType w) -> PermExpr (BVType w) ->
  498                 WideningM (Maybe (PermExpr (BVType w)))
  499 widenBVsAddy tp bv1 bv2 =
  500   case bvMatchConst (bvSub bv2 bv1) of
  501     Just d -> do x <- bindFreshVar tp
  502                  visitM x
  503                  return $ Just (bvAdd (bvFactorExpr d x) bv1)
  504     _ -> return Nothing

  509                  PermExpr (BVType w) -> PermExpr (BVType w) ->

  514      return $ bvFactorExpr (bvGCD bv1 bv2) x

  517 -- | Take two block permissions @bp1@ and @bp2@ with the same offset and use
  518 -- 'splitLLVMBlockPerm' to remove any parts of them that do not overlap,
  519 -- returning some @bp1'@ and @bp2'@ with the same range, along with additional
  520 -- portions of @bp1@ and @bp2@ that were removed
  521 equalizeLLVMBlockRanges' :: (1 <= w, KnownNat w) =>
  522                             WidNameMap -> LLVMBlockPerm w -> LLVMBlockPerm w ->
  523                             Maybe (LLVMBlockPerm w, LLVMBlockPerm w,
  524                                    [LLVMBlockPerm w], [LLVMBlockPerm w])
  525 equalizeLLVMBlockRanges' _ bp1 bp2
  526   | not (bvEq (llvmBlockOffset bp1) (llvmBlockOffset bp2)) =
  527     error "equalizeLLVMBlockRanges'"
  528 equalizeLLVMBlockRanges' _ bp1 bp2
  529   | bvEq (llvmBlockLen bp1) (llvmBlockLen bp2) =
  530     return (bp1, bp2, [], [])
  531 equalizeLLVMBlockRanges' wnmap bp1 bp2
  532   | bvLt (llvmBlockLen bp1) (llvmBlockLen bp2) =
  533     do let blsubst = wnMapBlockShape wnmap
  534        (bp2', bp2'') <- splitLLVMBlockPerm blsubst (llvmBlockEndOffset bp1) bp2
  535        return (bp1, bp2', [], [bp2''])
  536 equalizeLLVMBlockRanges' wnmap bp1 bp2
  537   | bvLt (llvmBlockLen bp2) (llvmBlockLen bp1) =
  538     do let blsubst = wnMapBlockShape wnmap
  539        (bp1', bp1'') <- splitLLVMBlockPerm blsubst (llvmBlockEndOffset bp2) bp1
  540        return (bp1', bp2, [bp1''], [])
  541 equalizeLLVMBlockRanges' _ _ _ = Nothing
  542 
  543 -- | Take two block permissions @bp1@ and @bp2@ whose ranges overlap and use
  544 -- 'splitLLVMBlockPerm' to remove any parts of them that do not overlap,
  545 -- returning some @bp1'@ and @bp2'@ with the same range, along with additional
  546 -- portions of @bp1@ and @bp2@ that were removed
  547 equalizeLLVMBlockRanges :: (1 <= w, KnownNat w) =>
  548                            WidNameMap -> LLVMBlockPerm w -> LLVMBlockPerm w ->
  549                            Maybe (LLVMBlockPerm w, LLVMBlockPerm w,
  550                                   [LLVMBlockPerm w], [LLVMBlockPerm w])
  551 equalizeLLVMBlockRanges wnmap bp1 bp2
  552   | bvEq (llvmBlockOffset bp1) (llvmBlockOffset bp2) =
  553     equalizeLLVMBlockRanges' wnmap bp1 bp2
  554 equalizeLLVMBlockRanges wnmap bp1 bp2
  555   | bvLt (llvmBlockOffset bp1) (llvmBlockOffset bp2) =
  556     do let blsubst = wnMapBlockShape wnmap
  557        (bp1', bp1'') <- splitLLVMBlockPerm blsubst (llvmBlockOffset bp2) bp1
  558        (bp1_ret, bp2_ret, bps1, bps2) <- equalizeLLVMBlockRanges' wnmap bp1'' bp2
  559        return (bp1_ret, bp2_ret, bp1':bps1, bps2)
  560 equalizeLLVMBlockRanges wnmap bp1 bp2
  561   | bvLt (llvmBlockOffset bp2) (llvmBlockOffset bp1) =
  562     do let blsubst = wnMapBlockShape wnmap
  563        (bp2', bp2'') <- splitLLVMBlockPerm blsubst (llvmBlockOffset bp1) bp2
  564        (bp1_ret, bp2_ret, bps1, bps2) <- equalizeLLVMBlockRanges' wnmap bp1 bp2''
  565        return (bp1_ret, bp2_ret, bps1, bp2':bps2)
  566 equalizeLLVMBlockRanges _ _ _ = Nothing

  569 -- | Widen two block permissions against each other, assuming they already have
  570 -- the same range
  571 widenBlockPerm :: (1 <= w, KnownNat w) => LLVMBlockPerm w -> LLVMBlockPerm w ->
  572                   WideningM (LLVMBlockPerm w)
  573 widenBlockPerm bp1 bp2 =
  574   LLVMBlockPerm <$>
  575   widenExpr knownRepr (llvmBlockRW bp1) (llvmBlockRW bp2) <*>
  576   widenExpr knownRepr (llvmBlockLifetime bp1) (llvmBlockLifetime bp2) <*>
  577   return (llvmBlockOffset bp1) <*> return (llvmBlockLen bp1) <*>
  578   widenExpr knownRepr (llvmBlockShape bp1) (llvmBlockShape bp2)

  581 -- | Widen a sequence of atomic permissions against each other
  582 widenAtomicPerms :: TypeRepr a -> [AtomicPerm a] -> [AtomicPerm a] ->
  583                     WideningM [AtomicPerm a]
  584 widenAtomicPerms tp ps1 ps2 =
  585   do traceM (\i ->
  586               fillSep [pretty "widenAtomicPerms",
  587                        permPretty i ps1, permPretty i ps2])
  588      wnmap <- view wsNameMap <$> get
  589      widenAtomicPerms' wnmap tp ps1 ps2
  590 
  591 widenAtomicPerms' :: WidNameMap -> TypeRepr a ->
  592                      [AtomicPerm a] -> [AtomicPerm a] ->
  593                      WideningM [AtomicPerm a]
  594 
  595 -- If one side is empty, we return the empty list, i.e., true
  596 widenAtomicPerms' _ _ [] _ = return []
  597 widenAtomicPerms' _ _ _ [] = return []
  598 
  599 -- If there is a permission on the right that equals p1, use p1, and recursively
  600 -- widen the remaining permissions
  601 widenAtomicPerms' _ tp (p1 : ps1) ps2
  602   | Just i <- findIndex (== p1) ps2 =
  603     (p1 :) <$> widenAtomicPerms tp ps1 (deleteNth i ps2)
  604 
  605 -- If we have array permissions with the same offset, length, and stride on both
  606 -- sides, check that their fields are the same and equalize their borrows
  607 --
  608 -- FIXME: handle arrays with different lengths, and widen their fields
  609 widenAtomicPerms' wnmap tp (Perm_LLVMArray ap1 : ps1) ps2 =
  610   case findIndex
  611        (\case
  612            Perm_LLVMArray ap2 ->
  613              substEqVars wnmap (llvmArrayOffset ap1)
  614              == substEqVars wnmap (llvmArrayOffset ap2) &&
  615              substEqVars wnmap (llvmArrayLen ap1)
  616              == substEqVars wnmap (llvmArrayLen ap2) &&
  617              llvmArrayStride ap1 == llvmArrayStride ap2 &&
  618              -- FIXME: widen the rw modalities?
  619              substEqVars wnmap (llvmArrayRW ap1)
  620              == substEqVars wnmap (llvmArrayRW ap2) &&
  621              substEqVars wnmap (llvmArrayLifetime ap1)
  622              == substEqVars wnmap (llvmArrayLifetime ap2)
  623            _ -> False) ps2 of
  624     Just i
  625       | Perm_LLVMArray ap2 <- ps2!!i ->
  626         -- NOTE: at this point, ap1 and ap2 are equal except for perhaps their
  627         -- borrows and shapes, so we just filter out the borrows in ap1 that are
  628         -- also in ap2 and widen the shapes
  629         widenExpr knownRepr (llvmArrayCellShape ap1) (llvmArrayCellShape ap2)
  630         >>= \sh ->
  631         (Perm_LLVMArray (ap1 { llvmArrayCellShape = sh,
  632                                llvmArrayBorrows =
  633                                  filter (flip elem (llvmArrayBorrows ap2))
  634                                  (llvmArrayBorrows ap1) }) :) <$>
  635         widenAtomicPerms tp ps1 (deleteNth i ps2)
  636     _ ->
  637       -- We did not find an appropriate array on the RHS, so drop this one
  638       widenAtomicPerms tp ps1 ps2
  639 
  640 -- If the first permission on the left is an LLVM permission overlaps with some
  641 -- permission on the right, widen these against each other
  642 widenAtomicPerms' wnmap tp@(LLVMPointerRepr w) (p1 : ps1) ps2
  643   | Just bp1 <- llvmAtomicPermToBlock p1
  644   , rng1 <- llvmBlockRange bp1
  645   , Just i <-
  646       withKnownNat w (findIndex ((== Just True) . fmap (bvRangesOverlap rng1)
  647                                  . llvmAtomicPermRange) ps2)
  648   , Just bp2 <- llvmAtomicPermToBlock (ps2!!i)
  649   , Just (bp1', bp2', bps1_rem, bps2_rem) <-
  650       withKnownNat w (equalizeLLVMBlockRanges wnmap bp1 bp2)
  651   = withKnownNat w (
  652       (:) <$> (Perm_LLVMBlock <$> widenBlockPerm bp1' bp2') <*>
  653       widenAtomicPerms tp (map Perm_LLVMBlock bps1_rem ++ ps1)
  654       (map Perm_LLVMBlock bps2_rem ++ deleteNth i ps2))
  655 
  656 -- If the LHS is a frame permission such that there is a frame permission on the
  657 -- RHS with the same list of lengths, widen the expressions
  658 widenAtomicPerms' _ tp@(LLVMFrameRepr w) (Perm_LLVMFrame frmps1 : ps1) ps2
  659   | Just i <- findIndex (\case
  660                             Perm_LLVMFrame _ -> True
  661                             _ -> False) ps2
  662   , Perm_LLVMFrame frmps2 <- ps2 !! i
  663   , map snd frmps1 == map snd frmps2 =
  664     do es <- zipWithM (widenExpr
  665                        (LLVMPointerRepr w)) (map fst frmps1) (map fst frmps2)
  666        (Perm_LLVMFrame (zip es (map snd frmps1)) :) <$>
  667          widenAtomicPerms tp ps1 (deleteNth i ps2)
  668 
  669 -- If either side has unfoldable named permissions, unfold them and recurse
  670 widenAtomicPerms' _ tp ps1 ps2
  671   | any isFoldableConjPerm (ps1 ++ ps2)
  672   = do ps1' <- concatMapM widUnfoldConjPerm ps1
  673        ps2' <- concatMapM widUnfoldConjPerm ps2
  674        widenAtomicPerms tp ps1' ps2'
  675 
  676 -- Default: cannot widen p1 against any p2 on the right, so drop it and recurse
  677 widenAtomicPerms' _ tp (_ : ps1) ps2 = widenAtomicPerms tp ps1 ps2
  678

  680 -- | Widen permissions against each other
  681 widenPerm :: TypeRepr a -> ValuePerm a -> ValuePerm a -> WideningM (ValuePerm a)
  682 widenPerm tp p1 p2 =
  683   traceM (\i ->
  684            fillSep [pretty "widenPerm", permPretty i p1, permPretty i p2]) >>
  685   widenPerm' tp p1 p2
  686 
  687 widenPerm' :: TypeRepr a -> ValuePerm a -> ValuePerm a ->
  688               WideningM (ValuePerm a)
  689 
  690 widenPerm' tp (ValPerm_Eq e1) (ValPerm_Eq e2) =
  691   ValPerm_Eq <$> widenExpr tp e1 e2
  692 
  693 widenPerm' tp (ValPerm_Eq (asVarOffset -> Just (x1, off1))) p2 =
  694   do p1 <- getVarPermM x1
  695      isv1 <- isVisitedM x1
  696      case (p1, isv1) of
  697        (ValPerm_Eq e1, _) ->
  698          visitM x1 >> widenPerm tp (ValPerm_Eq $ offsetExpr off1 e1) p2
  699        (_, False) ->
  700          do visitM x1
  701             p1' <- widenPerm tp (offsetPerm off1 p1) p2
  702             setVarPermM x1 (offsetPerm (negatePermOffset off1) p1')
  703             return (ValPerm_Eq $ offsetExpr off1 $ PExpr_Var x1)
  704        (_, True) ->
  705          do x <- bindFreshVar tp
  706             visitM x
  707             (p1', p1'') <- splitWidenPerm tp (offsetPerm off1 p1) p2
  708             setVarPermM x1 p1'
  709             setVarPermM x p1''
  710             return (ValPerm_Eq $ PExpr_Var x)
  711 
  712 widenPerm' tp p1 (ValPerm_Eq (asVarOffset -> Just (x2, off2))) =
  713   do p2 <- getVarPermM x2
  714      isv2 <- isVisitedM x2
  715      case (p2, isv2) of
  716        (ValPerm_Eq e2, _) ->
  717          visitM x2 >> widenPerm tp p1 (ValPerm_Eq $ offsetExpr off2 e2)
  718        (_, False) ->
  719          do visitM x2
  720             p2' <- widenPerm tp p1 (offsetPerm off2 p2)
  721             setVarPermM x2 (offsetPerm (negatePermOffset off2) p2')
  722             return (ValPerm_Eq $ offsetExpr off2 $ PExpr_Var x2)
  723        (_, True) ->
  724          do x <- bindFreshVar tp
  725             visitM x
  726             (p2', p2'') <- splitWidenPerm tp (offsetPerm off2 p2) p1
  727             setVarPermM x2 p2'
  728             setVarPermM x p2''
  729             return (ValPerm_Eq $ PExpr_Var x)
  730 
  731 widenPerm' tp (ValPerm_Or p1 p1') (ValPerm_Or p2 p2') =
  732   ValPerm_Or <$> widenPerm tp p1 p2 <*> widenPerm tp p1' p2'
  733 widenPerm' tp (ValPerm_Exists mb_p1) p2 =
  734   do x <- bindFreshVar knownRepr
  735      widenPerm tp (varSubst (singletonVarSubst x) mb_p1) p2
  736 widenPerm' tp p1 (ValPerm_Exists mb_p2) =
  737   do x <- bindFreshVar knownRepr
  738      widenPerm tp p1 (varSubst (singletonVarSubst x) mb_p2)
  739 widenPerm' _ (ValPerm_Named npn1 args1 off1) (ValPerm_Named npn2 args2 off2)
  740   | Just (Refl, Refl, Refl) <- testNamedPermNameEq npn1 npn2
  741   , offsetsEq off1 off2 =
  742     (\args -> ValPerm_Named npn1 args off1) <$>
  743     widenExprs (namedPermNameArgs npn1) args1 args2
  744 widenPerm' tp (ValPerm_Named npn1 args1 off1) p2
  745   | DefinedSortRepr _ <- namedPermNameSort npn1 =
  746     do env <- use wsPermEnv
  747        let np1 = requireNamedPerm env npn1
  748        widenPerm tp (unfoldPerm np1 args1 off1) p2
  749 widenPerm' tp p1 (ValPerm_Named npn2 args2 off2)
  750   | DefinedSortRepr _ <- namedPermNameSort npn2 =
  751     do env <- use wsPermEnv
  752        let np2 = requireNamedPerm env npn2
  753        widenPerm tp p1 (unfoldPerm np2 args2 off2)
  754 widenPerm' tp (ValPerm_Named npn1 args1 off1) p2
  755   | RecursiveSortRepr _ _ <- namedPermNameSort npn1 =
  756     use wsRecFlag >>= \case
  757       RecRight ->
  758         -- If we have already unfolded on the right, don't unfold on the left
  759         -- (for termination reasons); instead just give up and return true
  760         return ValPerm_True
  761       _ ->
  762         do wsRecFlag .= RecLeft
  763            env <- use wsPermEnv
  764            let np1 = requireNamedPerm env npn1
  765            widenPerm tp (unfoldPerm np1 args1 off1) p2
  766 widenPerm' tp p1 (ValPerm_Named npn2 args2 off2)
  767   | RecursiveSortRepr _ _ <- namedPermNameSort npn2 =
  768     use wsRecFlag >>= \case
  769       RecLeft ->
  770         -- If we have already unfolded on the left, don't unfold on the right
  771         -- (for termination reasons); instead just give up and return true
  772         return ValPerm_True
  773       _ ->
  774         do wsRecFlag .= RecRight
  775            env <- use wsPermEnv
  776            let np2 = requireNamedPerm env npn2
  777            widenPerm tp p1 (unfoldPerm np2 args2 off2)
  778 widenPerm' _ (ValPerm_Var x1 off1) (ValPerm_Var x2 off2)
  779   | x1 == x2 && offsetsEq off1 off2 = return $ ValPerm_Var x1 off1
  780 widenPerm' tp (ValPerm_Conj ps1) (ValPerm_Conj ps2) =
  781   ValPerm_Conj <$> widenAtomicPerms tp ps1 ps2
  782 widenPerm' _ _ _ = return ValPerm_True

  939              Mb (ctx1 :> a :++: ctx2) b -> Mb (ctx1 :++: ctx2) b
  940 subst1Mid ctx1 ctx2 mb_e mb_b =
  941   mbMap2 subst1 mb_e $ mbSwapMidToEnd ctx1 ctx2 Proxy mb_b
  942 
  943 tryLift1Mid :: NuMatching b => Substable PartialSubst b Maybe =>
  944                RAssign Proxy ctx1 -> RAssign Proxy ctx2 ->
  945                Proxy (a :: CrucibleType) ->
  946                Mb (ctx1 :> a :++: ctx2) b ->
  947                Maybe (Mb (ctx1 :++: ctx2) b)
  948 tryLift1Mid ctx1 ctx2 a mb_e =
  949   mbMaybe $ fmap (partialSubst $ emptyPSubst' (MNil :>: a)) $
  950   mbSwapMidToEnd ctx1 ctx2 a mb_e
  951 
  952 emptyPSubst' :: RAssign any ctx -> PartialSubst ctx
  953 emptyPSubst' = PartialSubst . RL.map (PSubstElem . const Nothing)
  954 
  955 mbExprProxy :: Mb ctx (f a) -> Proxy a
  956 mbExprProxy _ = Proxy

  958 simplify1GhostPerm :: RAssign Proxy args ->
  959                       FoundPerm (args :++: vars) vars ->
  960                       ArgVarPerms args vars ->
  961                       Maybe (Some (ArgVarPerms args))
  962 simplify1GhostPerm args (FoundPerm vars1 vars2
  963                          [nuP| ValPerm_Eq mb_e |]) (ArgVarPerms
  964                                                     vars mb_perms)
  965   | a <- mbExprProxy mb_e
  966   , Refl <- RL.appendAssoc args vars1 vars2
  967   , Refl <- RL.appendAssoc args (vars1 :>: a) vars2
  968   , args_vars1 <- RL.append args vars1
  969   , Just mb_e' <- tryLift1Mid args_vars1 vars2 a mb_e =
  970     Just (Some $ ArgVarPerms (cruCtxRemMid vars1 vars2 a vars)
  971           (subst1Mid args_vars1 vars2 mb_e' $
  972            fmap (rlRemMid args_vars1 vars2 a) mb_perms))
  973 simplify1GhostPerm args (FoundPerm vars1 vars2
  974                          p@[nuP| ValPerm_True |]) (ArgVarPerms
  975                                                    vars mb_perms)
  976   | a <- mbExprProxy p
  977   , Refl <- RL.appendAssoc args vars1 vars2
  978   , Refl <- RL.appendAssoc args (vars1 :>: a) vars2
  979   , args_vars1 <- RL.append args vars1
  980   , Just mb_perms' <- (tryLift1Mid args_vars1 vars2 a $
  981                        fmap (rlRemMid args_vars1 vars2 a) mb_perms) =
  982     Just $ Some $ ArgVarPerms (cruCtxRemMid vars1 vars2 a vars) mb_perms'
  983 simplify1GhostPerm _ _ _ = Nothing
  984

  986 simplifyGhostPerms :: RAssign Proxy args -> Some (ArgVarPerms args) ->
  987                       Some (ArgVarPerms args)
  988 simplifyGhostPerms args (Some avps)
  989   | some_avps':_ <-
  990       mapMaybe (flip (simplify1GhostPerm args) avps) (findGhostSimplPerm avps)
  991   = simplifyGhostPerms args some_avps'
  992 simplifyGhostPerms _ some_avps = some_avps

  999 -- | Widen two lists of permissions-in-bindings
 1000 widen :: DebugLevel -> PermEnv -> CruCtx tops -> CruCtx args ->
 1001          Some (ArgVarPerms (tops :++: args)) ->
 1002          Some (ArgVarPerms (tops :++: args)) ->
 1003          Some (ArgVarPerms (tops :++: args))
 1004 widen dlevel env tops args (Some (ArgVarPerms
 1005                                   vars1 mb_perms1)) (Some (ArgVarPerms
 1006                                                            vars2 mb_perms2)) =
 1007   let all_args = appendCruCtx tops args
 1008       prxs1 = cruCtxProxies vars1
 1009       prxs2 = cruCtxProxies vars2
 1010       mb_mb_perms1 = mbSeparate prxs1 mb_perms1 in
 1011   simplifyGhostPerms (cruCtxProxies all_args) $
 1012   completeArgVarPerms $ flip nuMultiWithElim1 mb_mb_perms1 $
 1013   \args_ns1 mb_perms1' ->
 1014   (\m -> runWideningM m dlevel env args_ns1) $
 1015   do (vars1_ns, ps1) <- openMb vars1 mb_perms1'
 1016      (ns2, ps2) <- openMb (appendCruCtx all_args vars2) mb_perms2
 1017      let (args_ns2, vars2_ns) = RL.split all_args prxs2 ns2
 1018      setVarPermsM (RL.append args_ns1 vars1_ns) ps1
 1019      setVarPermsM ns2 ps2
 1020      let (tops1, locals1) = RL.split tops (cruCtxProxies args) args_ns1
 1021      let (tops2, locals2) = RL.split tops (cruCtxProxies args) args_ns2
 1022      setVarNamesM "topL" tops1
 1023      setVarNamesM "topR" tops2
 1024      setVarNamesM "localL" locals1
 1025      setVarNamesM "localR" locals2
 1026      setVarNamesM "varL" vars1_ns
 1027      setVarNamesM "varR" vars2_ns
 1028      let dist_ps1 = RL.map2 VarAndPerm (RL.append args_ns1 vars1_ns) ps1
 1029      let dist_ps2 = RL.map2 VarAndPerm ns2 ps2
 1030      traceM (\i ->
 1031               pretty "Widening" <> line <>
 1032               indent 2 (permPretty i dist_ps1) <> line <>
 1033               pretty "Against" <> line <>
 1034               indent 2 (permPretty i dist_ps2))
 1035      void $ widenExprs all_args (RL.map PExpr_Var args_ns1) (RL.map
 1036                                                              PExpr_Var args_ns2)
 1037      widenExtGhostVars vars1 vars1_ns vars2 vars2_ns
 1038      modifying wsNameMap wnMapDropUnvisiteds
 1039      wnmap <- view wsNameMap <$> get
 1040      traceM (\i ->
 1041               pretty "Widening returning:" <> line <>
 1042               indent 2 (fillSep $
 1043                         map (\(NameAndElem x (Pair p _)) ->
 1044                               permPretty i x <> colon <> permPretty i p) $
 1045                         NameMap.assocs wnmap))
 1046      return ()
