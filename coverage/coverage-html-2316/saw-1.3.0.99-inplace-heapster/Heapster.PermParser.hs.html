   45 -- | One of the generated parsers from "Heapster.Parser"
   46 -- which is intended to be used with 'Heapster.Lexer.lexer'
   47 type Parser p = [Located Token] -> Either ParseError p
   48 
   49 -- | Harness for running the lexer, parser, and type-checker.
   50 -- Human-readable errors are raised through 'fail'.
   51 runParser ::
   52   Fail.MonadFail m =>
   53   String                {- ^ object name                -} ->
   54   PermEnv               {- ^ permission environment     -} ->
   55   Parser p              {- ^ parser                     -} ->
   56   (p -> Tc a)           {- ^ checker                    -} ->
   57   String                {- ^ input                      -} ->
   58   m a
   59 runParser obj env parser checker str =
   60   case parser (lexer str) of
   61     Left UnexpectedEnd ->
   62       fail $ unlines
   63         [ "Error parsing " ++ obj
   64         , "Unexpected end of input"
   65         , pointEnd str
   66         ]
   67     Left (UnexpectedToken p t) ->
   68       fail $ unlines
   69         [ "Error parsing " ++ obj ++ " at " ++ renderPos p
   70         , "Unexpected " ++ describeToken t
   71         , point p str
   72         ]
   73     Right ast ->
   74       case startTc (checker ast) env of
   75         Left (TypeError p e) ->
   76           fail $ unlines
   77             [ "Error checking " ++ obj ++ " at " ++ renderPos p
   78             , e
   79             , point p str
   80             ]
   81         Right x -> pure x
   82 
   83 -- | Human-readable rendering of error location
   84 renderPos ::
   85   Pos                   {- ^ error position             -} ->
   86   String                {- ^ rendered output            -}
   87 renderPos p = "line " ++ show (posLine p) ++ " column " ++ show (posCol p)
   88 
   89 -- | Point to the error in the line with an error.
   90 point ::
   91   Pos                   {- ^ error position             -} ->
   92   String                {- ^ full input string          -} ->
   93   String                {- ^ rendered output            -}
   94 point p str =
   95   lines str !! (posLine p - 1) ++ "\n" ++
   96   Data.List.replicate (posCol p - 1) ' ' ++ "^"
   97 
   98 -- | Point to the end of the file
   99 pointEnd ::
  100   String                {- ^ full input string          -} ->
  101   String                {- ^ rendered output            -}
  102 pointEnd "" = "<<empty input>>"
  103 pointEnd str = end ++ "\n" ++ (' ' <$ end) ++ "^"
  104   where
  105     end = last (lines str)

  107 -- | Parse a permission set @x1:p1, x2:p2, ...@ for the variables in the
  108 -- supplied context
  109 parsePermsString ::
  110   Fail.MonadFail m =>
  111   String                {- ^ object name                -} ->
  112   PermEnv               {- ^ permission environment     -} ->
  113   ParsedCtx ctx         {- ^ parsed context             -} ->
  114   String                {- ^ input text                 -} ->
  115   m (MbValuePerms ctx)
  116 parsePermsString nm env ctx =
  117   runParser nm env parseValuePerms (tcSortedMbValuePerms ctx)
  118 
  119 -- | Parse a permission of the given type within the given context and with
  120 -- the given named permission variables in scope
  121 parsePermInCtxString ::
  122   Fail.MonadFail m =>
  123   String                {- ^ object name                -} ->
  124   PermEnv               {- ^ permission environment     -} ->
  125   ParsedCtx ctx         {- ^ parsed context             -} ->
  126   TypeRepr a            {- ^ expected permission type   -} ->
  127   String                {- ^ input text                 -} ->
  128   m (Mb ctx (ValuePerm a))
  129 parsePermInCtxString nm env ctx tp =
  130   runParser nm env parseExpr (inParsedCtxM ctx . const . tcValPerm tp)
  131 
  132 -- | Parse a sequence of atomic permissions within the given context and with
  133 -- the given named permission variables in scope
  134 parseAtomicPermsInCtxString ::
  135   Fail.MonadFail m =>
  136   String                {- ^ object name                -} ->
  137   PermEnv               {- ^ permission environment     -} ->
  138   ParsedCtx ctx         {- ^ parsed context             -} ->
  139   TypeRepr a            {- ^ expected permission type   -} ->
  140   String                {- ^ input text                 -} ->
  141   m (Mb ctx [AtomicPerm a])
  142 parseAtomicPermsInCtxString nm env ctx tp =
  143   runParser nm env parseExpr (inParsedCtxM ctx . const . tcAtomicPerms tp)

  145 -- | Parse a 'FunPerm' named by the first 'String' from the second 'String'
  146 parseFunPermString ::
  147   Fail.MonadFail m =>
  148   String                {- ^ object name                -} ->
  149   PermEnv               {- ^ permission environment     -} ->
  150   CruCtx args           {- ^ argument types             -} ->
  151   TypeRepr ret          {- ^ return type                -} ->
  152   String                {- ^ input text                 -} ->
  153   m (SomeFunPerm args ret)
  154 parseFunPermString nm env args ret =
  155   runParser nm env parseFunPerm (tcFunPerm args ret)
  156 
  157 -- | Parse a type context @x1:tp1, x2:tp2, ...@ named by the first 'String' from
  158 -- the second 'String' and return a 'ParsedCtx', which contains both the
  159 -- variable names @xi@ and their types @tpi@
  160 parseParsedCtxString ::
  161   Fail.MonadFail m =>
  162   String                {- ^ object name                -} ->
  163   PermEnv               {- ^ permission environment     -} ->
  164   String                {- ^ input text                 -} ->
  165   m (Some ParsedCtx)
  166 parseParsedCtxString nm env = runParser nm env parseCtx tcCtx
  167 
  168 -- | Parse a type context named by the first 'String' from the second 'String'
  169 parseCtxString ::
  170   Fail.MonadFail m =>
  171   String                {- ^ object name                -} ->
  172   PermEnv               {- ^ permission environment     -} ->
  173   String                {- ^ input text                 -} ->
  174   m (Some CruCtx)
  175 parseCtxString nm env str =
  176   fmapF parsedCtxCtx <$> parseParsedCtxString nm env str

  178 -- | Parse a type named by the first 'String' from the second 'String'

  183   String                {- ^ input text                 -} ->

  187 -- | Parse an expression of a given type from a 'String'
  188 parseExprInCtxString ::
  189   Fail.MonadFail m =>
  190   PermEnv -> TypeRepr a -> ParsedCtx ctx -> String -> m (Mb ctx (PermExpr a))
  191 parseExprInCtxString env tp ctx =
  192   runParser (permPrettyString emptyPPInfo tp) env parseExpr
  193     (inParsedCtxM ctx . const . tcExpr tp)

  195 -- | Parse a 'FunPerm' named by the first 'String' from the second 'String'.
  196 -- The 'FunPerm' can either be standard Heapster syntax, which begins with an
  197 -- open parenthesis (after optional whitespace), or it could be given in Rust
  198 -- syntax, which begins with an angle bracket. The @w@ argument gives the bit
  199 -- width of pointers in the current architecture.
  200 parseFunPermStringMaybeRust ::
  201   (1 <= w, KnownNat w, Fail.MonadFail m) =>
  202   String                {- ^ object name                -} ->
  203   prx w                 {- ^ pointer bit-width proxy    -} ->
  204   PermEnv               {- ^ permission environment     -} ->
  205   CruCtx args           {- ^ argument types             -} ->
  206   TypeRepr ret          {- ^ return type                -} ->
  207   String                {- ^ input text                 -} ->
  208   m (SomeFunPerm args ret)
  209 parseFunPermStringMaybeRust nm w env args ret str =
  210   case find (\c -> c == '<' || c == '(') str of
  211     Just '<' -> parseFunPermFromRust env w args ret str
  212     _ -> parseFunPermString nm env args ret str
