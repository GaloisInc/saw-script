   43 -- | Verbosity is currently a linear setting (vs a mask or tree).  Any given
   44 -- level includes the outputs of all lower levels.
   45 data Verbosity
   46   = Silent
   47   | OnlyCounterExamples
   48   | Error
   49   | Warn
   50   | Info
   51   | Debug
   52     deriving (Show,Eq,Ord)

   81 printOutWith :: Verbosity -> Verbosity -> String -> IO ()
   82 printOutWith setting level msg
   83     | setting >= level = do
   84       t <- formatTime defaultTimeLocale "%T.%3q" <$> getCurrentTime
   85       putStr $ "[" ++ t ++ "] " ++ msg
   86     | otherwise        = return ()
   87 
   88 printOutLn :: Options -> Verbosity -> String -> IO ()
   89 printOutLn o v s = printOutFn o v (s ++ "\n")

   91 -- | Perform some additional post-processing on an 'Options' value based on
   92 -- whether certain environment variables are defined.
   93 --
   94 -- XXX: this does not belong in Options.hs; it is a top-level
   95 -- operation that's really part of main. But it's shared between saw
   96 -- and saw-remote-api so it has to live somewhere shared, and for the
   97 -- time being there isn't anywhere else.
   98 processEnv :: Options -> IO Options
   99 processEnv opts = do
  100   curEnv <- getEnvironment
  101   opts' <- addJavaBinDirInducedOpts opts
  102   return $ foldr addSawOpt opts' curEnv
  103   where
  104     -- If a Java executable's path is specified (either by way of
  105     -- --java-bin-dirs or PATH, see the Haddocks for findJavaIn), then use that
  106     -- to detect the path to Java's standard rt.jar file and add it to the
  107     -- jarList on Java 8 or earlier. (Later versions of Java do not use
  108     -- rt.jarâ€”see Note [Loading classes from JIMAGE files] in
  109     -- Lang.JVM.Codebase in crucible-jvm.)
  110     -- If Java's path is not specified, return the Options unchanged.
  111     addJavaBinDirInducedOpts :: Options -> IO Options
  112     addJavaBinDirInducedOpts os@Options{javaBinDirs} = do
  113       mbJavaPath <- findJavaIn javaBinDirs
  114       case mbJavaPath of
  115         Nothing       -> pure os
  116         Just javaPath -> do
  117           javaMajorVersion <- findJavaMajorVersion javaPath
  118           if javaMajorVersion >= 9
  119              then pure os
  120              else addRTJar javaPath os
  121 
  122     -- rt.jar lives in a standard location relative to @java.home@. At least,
  123     -- this is the case on every operating system I've tested.
  124     addRTJar :: FilePath -> Options -> IO Options
  125     addRTJar javaPath os = do
  126       mbJavaHome <- findJavaProperty javaPath "java.home"
  127       case mbJavaHome of
  128         Nothing -> fail $ "Could not find where rt.jar lives for " ++ javaPath
  129         Just javaHome ->
  130           let rtJarPath = javaHome </> "lib" </> "rt.jar" in
  131           pure $ os{ jarList = rtJarPath : jarList os }
  132 
  133     addSawOpt :: (String, String) -> Options -> Options
  134     addSawOpt ("SAW_IMPORT_PATH", p) os =
  135       os { importPath = importPath os ++ splitSearchPath p }
  136     addSawOpt ("SAW_JDK_JAR", p) os = os { jarList = p : jarList os }
  137     addSawOpt ("CLASSPATH", p) os = os { jarList = jarList os ++ jars
  138                                        , classPath = classPath os ++ dirs
  139                                        }
  140                                     where
  141                                       es = splitSearchPath p
  142                                       (jars, dirs) = partition (".jar" `isExtensionOf`) es
  143     addSawOpt _ os = os
