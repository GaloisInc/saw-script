   17 type Logo = [String]
   18 
   19 logo :: Bool -> Logo
   20 logo useColor =
   21      [ sgr [SetColor Foreground Dull  White] ++ l | l <- ws ]
   22   ++ [ sgr [SetColor Foreground Vivid Blue ] ++ l | l <- vs ]
   23   ++ [ sgr [SetColor Foreground Dull  Blue ] ++ l | l <- ds ]
   24   ++ [ sgr [Reset] ]
   25   where
   26   sgr | useColor  = setSGRCode
   27       | otherwise = const []
   28   ver = sgr [SetColor Foreground Dull White]
   29         ++ replicate (lineLen - 20 - length versionText) ' '
   30         ++ versionText
   31   ls = (if useColor then logo2 else logo1) ver
   32   line      = case ls of
   33                 line':_ -> line'
   34                 [] -> panic "logo" ["empty lines"]
   35   lineLen   = length line
   36   slen      = length ls `div` 3
   37   (ws,rest) = splitAt slen ls
   38   (vs,ds)   = splitAt slen rest
   39 
   40 logo1 :: String -> [String]
   41 logo1 ver =
   42     [ "   ___  __ _ _ _ _"
   43     , "  / __|/ _\' | | | |"
   44     , "  \\__ \\ (_| | | | |"
   45     , "  |___/\\__,_|\\_,_/ " ++ ver
   46     ]
   47 
   48 logo2 :: String -> [String]
   49 logo2 ver =
   50     [ " ┏━━━┓━━━┓━┓━┓━┓"
   51     , " ┃ ━━┓ ╻ ┃ ┃ ┃ ┃"
   52     , " ┣━━ ┃ ╻ ┃┓ ╻ ┏┛"
   53     , " ┗━━━┛━┛━┛┗━┛━┛ " ++ ver
   54     ]
   55 
   56 displayLogo :: Bool -> IO ()
   57 displayLogo useColor = mapM_ putStrLn (logo useColor)
