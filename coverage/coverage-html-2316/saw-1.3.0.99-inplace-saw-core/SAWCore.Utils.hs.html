   27 sumBy :: (Foldable t, Num b) => (a -> b) -> t a -> b
   28 sumBy f = foldl' fn 0
   29   where fn e v = e + f v

   31 internalError :: String -> a
   32 internalError msg = error $ "internal: " ++ msg
