   32 -- | Combines LLVM Modules into a single, composite Module.  This is akin to
   33 -- linking, but just from the perspective of what is needed for SAW/Cryptol
   34 -- analysis.
   35 --
   36 -- Note that there is a Semigroup instance for 'Module', which is necessary for
   37 -- the llvm_pretty implementation of the @LLVM@ Monad containing a @WriterT
   38 -- Module@, but it should *only* be used for building a 'Module' form elements
   39 -- that do *not* have unresolved cross-Module symbols or other issues.  The
   40 -- implementation here provides proper combination of independent Modules,
   41 -- allowing symbol references in one module to be resolved by definitions in the
   42 -- other, etc.
   43 llvmModuleCombine :: Module -> Module -> Module
   44 llvmModuleCombine a b =
   45   let defs = a ^. modDefinesLens
   46       decls = a ^. modDeclaresLens
   47       newDefs = b ^. modDefinesLens
   48       newDecls = b ^. modDeclaresLens
   49       rmvDefined = flip (foldr removeDefined)
   50       newDeclsLessOldDefs = rmvDefined defs newDecls
   51       oldDeclsLessNewDefs = rmvDefined newDefs decls
   52       joinName n = Just $ (<> ("+" <> (fromMaybe "..." (modSourceName b)))) $ fromMaybe "..." n
   53   in a
   54      & modSourceNameLens %~ joinName
   55      & modDeclaresLens .~ (oldDeclsLessNewDefs <> newDeclsLessOldDefs)
   56      & modDefinesLens %~ deConflict (b ^. modDefinesLens)
   57      & modTypesLens <>~ b ^. modTypesLens
   58      & modNamedMdLens <>~ b ^. modNamedMdLens
   59      & modComdatLens <>~ b ^. modComdatLens
   60      & modGlobalsLens <>~ b ^. modGlobalsLens
   61      & modInlineAsmLens <>~ b ^. modInlineAsmLens
   62      & modAliasesLens <>~ b ^. modAliasesLens
   63   -- TODO Globals any should override Linkage external for the same name
   64   -- TODO extend modUnnamedMd and propagate those changes
   65   -- TODO verify modTriple and modDataLayout are the same?
   66

   68 -- | A Define takes precedence over a Declare.  When combining modules, module A
   69 -- may Declare a function that is handled by a Define in module B, so get rid of
   70 -- the Declare when putting A and B together.
   71 
   72 removeDefined :: Define -> [Declare] -> [Declare]
   73 removeDefined def = filter ((def ^. defNameLens /=) . view decNameLens)
   74

   76 -- | Module A may have a Define with Linkage=Private|LinkerPrivate|LinkerPrivateWeak|LinkerPRivateWeakDefAuto|Internal|Weak|WeakODR? Visibility=HiddenVisibility|ProtectedVisibility?
   77 
   78 deConflict :: [Define] -> [Define] -> [Define]
   79 deConflict new curr = uncurry (<>) $ foldl deConflictDef (curr, new) new
   80   where
   81     deConflictDef (ads, bds) bd =
   82       case find (((==) `on` defName) bd) ads of
   83         Nothing -> (ads, bds)
   84         Just ad -> handle ads bds ad bd
   85     handle ads bds ad bd =
   86       case bd ^. defLinkageLens of
   87         Just Private -> (ads, renameDef bd (view defNameLens <$> ads) bds)
   88         Just LinkerPrivate -> (ads, renameDef bd (view defNameLens <$> ads) bds)
   89         Just LinkerPrivateWeak ->
   90           (ads, renameDef bd (view defNameLens <$> ads) bds) -- ??
   91         Just LinkerPrivateWeakDefAuto ->
   92           (ads, renameDef bd (view defNameLens <$> ads) bds) -- ??
   93         Just Internal -> (ads, renameDef bd (view defNameLens <$> ads) bds)
   94         Just AvailableExternally ->
   95           -- Never happen: not allowed on defines.  Ignore
   96           (ads, bds)
   97         Just Linkonce -> (mergeDef ad bd ads, removeDef bd bds)
   98         Just Weak -> (mergeDef ad bd ads, removeDef bd bds)
   99         Just Common -> (mergeDef ad bd ads, removeDef bd bds)
  100         Just ExternWeak -> (mergeDef ad bd ads, removeDef bd bds)
  101         Just LinkonceODR -> (mergeDef ad bd ads, removeDef bd bds)
  102         Just WeakODR -> (mergeDef ad bd ads, removeDef bd bds)
  103         Just Appending -> (appendDef ad bd ads, removeDef bd bds)
  104         Just External ->
  105           -- This should never happen: it is truly a symbol conflict.  A
  106           -- linker would reject this, but here we will just preserve the
  107           -- original.
  108           (ads, removeDef bd bds)
  109         Just DLLImport -> (ads, removeDef bd bds) -- ??
  110         Just DLLExport -> (ads, removeDef bd bds) -- ??
  111         Nothing ->
  112           -- No linkage specified.  The default is 'External', with associated
  113           -- considerations as documented for that case above.
  114           (ads, removeDef bd bds)
  115 
  116 -- Note: Used for Linkonce, Weak, Common, ExternWeak, LinkonceODR, WeakODR. LLVM
  117 -- docs say "merged", but also indicates that maybe there is a replacement
  118 -- instead?  For now, treat "merged" as appending.
  119 mergeDef :: Define -> Define -> [Define] -> [Define]
  120 mergeDef = appendDef
  121 
  122 appendDef :: Define -> Define -> [Define] -> [Define]
  123 appendDef d1 d2 =
  124   let appenD = d1 & defBodyLens <>~ d2 ^. defBodyLens
  125   in (appenD :) . filter (((/=) `on` defName) d1)
  126 
  127 removeDef :: Define -> [Define] -> [Define]
  128 removeDef d = filter (((/=) `on` defName) d)

  131 -- Renames the Defined symbol to a new name using a discriminator to avoid a
  132 -- conflict.  Only valid for renaming Private/Internal Defines such that changing
  133 -- any reference to the original Symbol to the new Symbol in the provided set of
  134 -- Defines is sufficient to change all references.  Note therefore this excludes:
  135 -- renaming of global variables, changing a GlobalAlias.
  136 
  137 renameDef :: Define -> [Symbol] -> [Define] -> [Define]
  138 renameDef toRename known inDefs =
  139   -- KWQ TODO: needs ot change GlobalAlias aliasName?
  140   --
  141   let getNewName nm n =
  142         -- Note: adds a "discriminator" to the name in a way that is valid for
  143         -- both C functions and C++ mangled names (see
  144         -- https://itanium-cxx-abi.github.io/cxx-abi/abi.html#mangling-scope).
  145         let nn = if n < 10
  146                  then nm <> "_" <> show n
  147                  else nm <> "__" <> show n <> "_"
  148         in case find ((fromString nn ==) . defName) inDefs of
  149              Just _ -> getNewName nm $ succ n
  150              Nothing ->
  151                if (fromString nn) `elem` known
  152                then getNewName nm $ succ n
  153                else nn
  154       (Symbol oldname) = defName toRename
  155       newName = Symbol $ getNewName oldname (1 :: Integer)
  156   in changeSym (defName toRename) newName inDefs
  157

  159 changeSym :: Symbol -> Symbol -> [Define] -> [Define]
  160 changeSym old new defs = fmap chngDef defs
  161   where
  162     chngDef d =
  163       d & defNameLens %~ chngSym
  164       & defBodyLens %~ fmap chngBlock
  165     chngBlock = bbStmtsLens %~ fmap chngStmt
  166     chngStmt = \case
  167       Result i inst drs vmds ->
  168         Result i (chngInstr inst) drs $ (fmap chngValMd <$> vmds)
  169       Effect inst drs vmds ->
  170         Effect (chngInstr inst) drs $ (fmap chngValMd <$> vmds)
  171     chngInstr = \case
  172       Ret tv -> Ret $ chngTyVal tv
  173       RetVoid -> RetVoid
  174       Arith o tv v -> Arith o (chngTyVal tv) (chngVal v)
  175       UnaryArith o tv -> UnaryArith o $ chngTyVal tv
  176       Bit o tv v -> Bit o (chngTyVal tv) (chngVal v)
  177       Conv o tv t -> Conv o (chngTyVal tv) t
  178       Call b t v tvs -> Call b t (chngVal v) (chngTyVal <$> tvs)
  179       CallBr t v tvs l ls -> CallBr t (chngVal v) (chngTyVal <$> tvs) l ls
  180       Alloca t mbtv i -> Alloca t (chngTyVal <$> mbtv) i
  181       Load t tv o a -> Load t (chngTyVal tv) o a
  182       Store tv1 tv2 o a -> Store (chngTyVal tv1) (chngTyVal tv2) o a
  183       Fence s o -> Fence s o
  184       CmpXchg b1 b2 tv1 tv2 tv3 s o1 o2 ->
  185         CmpXchg b1 b2 (chngTyVal tv1) (chngTyVal tv2) (chngTyVal tv3) s o1 o2
  186       AtomicRW b o tv1 tv2 s a ->
  187         AtomicRW b o (chngTyVal tv1) (chngTyVal tv2) s a
  188       ICmp b o tv v -> ICmp b o (chngTyVal tv) (chngVal v)
  189       FCmp o tv v -> FCmp o (chngTyVal tv) (chngVal v)
  190       Phi t vs -> Phi t ((\(v,l) -> (chngVal v, l)) <$> vs)
  191       GEP as t tv tvs -> GEP as t (chngTyVal tv) (chngTyVal <$> tvs)
  192       Select tv1 tv2 v -> Select (chngTyVal tv1) (chngTyVal tv2) (chngVal v)
  193       ExtractValue tv is -> ExtractValue (chngTyVal tv) is
  194       InsertValue tv1 tv2 is -> InsertValue (chngTyVal tv1) (chngTyVal tv2) is
  195       ExtractElt tv v -> ExtractElt (chngTyVal tv) (chngVal v)
  196       InsertElt tv1 tv2 v -> InsertElt (chngTyVal tv1) (chngTyVal tv2) (chngVal v)
  197       ShuffleVector tv1 v tv2 ->
  198         ShuffleVector (chngTyVal tv1) (chngVal v) (chngTyVal tv2)
  199       Jump t -> Jump t
  200       Br tv l1 l2 -> Br (chngTyVal tv) l1 l2
  201       Invoke t v tvs l1 l2 -> Invoke t (chngVal v) (chngTyVal <$> tvs) l1 l2
  202       Comment s -> Comment s
  203       Unreachable -> Unreachable
  204       Unwind -> Unwind
  205       VaArg tv t -> VaArg (chngTyVal tv) t
  206       IndirectBr tv lbs -> IndirectBr (chngTyVal tv) lbs
  207       Switch tv l tgts -> Switch (chngTyVal tv) l tgts
  208       LandingPad t mbtv b cls ->
  209         LandingPad t (chngTyVal <$> mbtv) b (chngClause <$> cls)
  210       Resume tv -> Resume $ chngTyVal tv
  211       Freeze tv -> Freeze $ chngTyVal tv
  212     chngClause = \case
  213       Catch tv -> Catch $ chngTyVal tv
  214       Filter tv -> Filter $ chngTyVal tv
  215     chngVal = \case
  216       ValSymbol s -> ValSymbol $ chngSym s
  217       ValArray t vs -> ValArray t (chngVal <$> vs)
  218       ValVector t vs -> ValVector t (chngVal <$> vs)
  219       ValStruct tvs -> ValStruct (chngTyVal <$> tvs)
  220       ValPackedStruct tvs -> ValPackedStruct (chngTyVal <$> tvs)
  221       ValMd vmd -> ValMd $ chngValMd vmd
  222       o -> o
  223     chngTyVal = typedValueLens %~ chngVal
  224     chngValMd = \case
  225       ValMdValue tv -> ValMdValue $ chngTyVal tv
  226       ValMdNode mvmds -> ValMdNode $ (fmap chngValMd <$> mvmds)
  227       o -> o
  228     chngSym s = bool s new $ old == s
