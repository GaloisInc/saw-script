   22 -- | Translate a list of named terms with their types to a Coq definitions
   23 translateTypedTermMap ::
   24   TermTranslation.TermTranslationMonad m => [(Name,Term,Term)] -> m [Coq.Decl]
   25 translateTypedTermMap defs = forM defs translateAndRegisterEntry
   26   where
   27     translateAndRegisterEntry (name, t, tp) = do
   28       let nameStr = unpackIdent (nameIdent name)
   29       decl <-
   30         do t_trans <- TermTranslation.translateTerm t
   31            tp_trans <- TermTranslation.translateTerm tp
   32            return $ TermTranslation.mkDefinition nameStr t_trans tp_trans
   33       modify $ over TermTranslation.globalDeclarations (nameStr :)
   34       return decl
   35 
   36 -- | Translates a Cryptol Module into a list of Coq declarations.  This is done
   37 -- by going through the term map corresponding to the module, translating all
   38 -- terms, and accumulating the translated declarations of all top-level
   39 -- declarations encountered.
   40 translateCryptolModule ::
   41   SharedContext -> Env ->
   42   TranslationConfiguration ->
   43   -- | List of already translated global declarations
   44   [String] ->
   45   CryptolModule ->
   46   IO (Either (TranslationError Term) [Coq.Decl])
   47 translateCryptolModule sc env configuration globalDecls (CryptolModule _ tm) =
   48   do defs <-
   49        forM (Map.assocs tm) $ \(nm, t) ->
   50        do tp <- ttTypeAsTerm sc env t
   51           return (nm, ttTerm t, tp)
   52      return
   53        (reverse . view TermTranslation.topLevelDeclarations . snd <$>
   54         TermTranslation.runTermTranslationMonad
   55         configuration
   56         Nothing -- TODO: this should be Just no?
   57         globalDecls
   58         []
   59         (translateTypedTermMap defs))
