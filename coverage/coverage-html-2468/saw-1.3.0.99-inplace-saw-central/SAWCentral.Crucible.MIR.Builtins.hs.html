  184 -- | The workhorse for 'mir_alloc', 'mir_alloc_mut', 'mir_alloc_raw_ptr_const',
  185 -- and 'mir_alloc_raw_ptr_mut'.
  186 mir_alloc_internal :: MirPointerKind -> Mir.Mutability -> Mir.Ty -> CrucibleSetup MIR SetupValue
  187 mir_alloc_internal pkind mut mty = do
  188   st <- get
  189   let mcc = st ^. Setup.csCrucibleContext
  190       col = mcc ^. mccRustModule ^. Mir.rmCS ^. Mir.collection
  191 
  192   -- We disallow creating slice references (e.g., &[u8] or &str) using
  193   -- mir_alloc, as it is unclear what length to give the resulting slice
  194   -- value.
  195   case mty of
  196     Mir.TySlice _ ->
  197       fail $ unlines
  198         [ "mir_alloc and mir_alloc_mut cannot be used to create slice references."
  199         , "Use the mir_slice_value or mir_slice_range_value functions to take a"
  200         , "slice of an array reference instead."
  201         ]
  202     Mir.TyStr ->
  203       fail $ unlines
  204         [ "mir_alloc and mir_alloc_mut cannot be used to create str references."
  205         , "Use the mir_str_slice_value or mir_str_slice_range_value functions"
  206         , "to take a slice of a `u8` array reference instead."
  207         ]
  208     _ ->
  209       pure ()
  210 
  211   loc <- getW4Position "mir_alloc"
  212   Some tpr <- pure $ Mir.tyToRepr col mty
  213   n <- Setup.csVarCounter <<%= MS.nextAllocIndex
  214   tags <- view Setup.croTags
  215   let md = MS.ConditionMetadata
  216           { MS.conditionLoc = loc
  217           , MS.conditionTags = tags
  218           , MS.conditionType = "fresh allocation"
  219           , MS.conditionContext = ""
  220           }
  221   Setup.currentState . MS.csAllocs . at n ?=
  222     Some (MirAllocSpec { _maConditionMetadata = md
  223                       , _maType = tpr
  224                       , _maPtrKind = pkind
  225                       , _maMutbl = mut
  226                       , _maMirType = mty
  227                       , _maLen = 1
  228                       })
  229   return (MS.SetupVar n)

  267 mir_execute_func :: [SetupValue] -> MIRSetupM ()
  268 mir_execute_func args =
  269   MIRSetupM $
  270   do st <- get
  271      let cc = st ^. Setup.csCrucibleContext
  272      let mspec = st ^. Setup.csMethodSpec
  273      let env = MS.csAllocations mspec
  274      let nameEnv = MS.csTypeNames mspec
  275      let argTys = mspec ^. MS.csArgs
  276      let
  277        checkArg i expectedTy val =
  278          do valTy <- typeOfSetupValue cc env nameEnv val
  279             unless (checkCompatibleTys expectedTy valTy) $
  280               X.throwM (MIRArgTypeMismatch i expectedTy valTy)
  281      let
  282        checkArgs _ [] [] = pure ()
  283        checkArgs i [] vals =
  284          X.throwM (MIRArgNumberWrong i (i + length vals))
  285        checkArgs i tys [] =
  286          X.throwM (MIRArgNumberWrong (i + length tys) i)
  287        checkArgs i (ty : tys) (val : vals) =
  288          do checkArg i ty val
  289             checkArgs (i + 1) tys vals
  290      checkArgs 0 argTys args
  291      Setup.crucible_execute_func args

  355 -- | See 'mir_fresh_expanded_val'.
  356 --
  357 -- This is the recursively-called worker function.
  358 constructExpandedSetupValue ::
  359   MIRCrucibleContext ->
  360   SharedContext ->
  361   Text ->
  362   TypeShape tp ->
  363   CrucibleSetup MIR SetupValue
  364 constructExpandedSetupValue cc sc = go
  365   where
  366     col = cc ^. mccRustModule . Mir.rmCS . Mir.collection
  367 
  368     go :: forall tp.
  369           Text ->
  370           TypeShape tp ->
  371           CrucibleSetup MIR SetupValue
  372     go pfx shp =
  373       case shp of
  374         UnitShape _ ->
  375           pure $ MS.SetupTuple () []
  376         PrimShape ty _ -> do
  377           fv <- freshPrimVariable pfx ty
  378           pure $ MS.SetupTerm fv
  379         TupleShape _ _ fldShps -> do
  380           flds <- goFlds pfx fldShps
  381           pure $ MS.SetupTuple () flds
  382         ArrayShape ty elemTy elemShp ->
  383           case ty of
  384             Mir.TyArray _ n -> do
  385               elems <-
  386                 traverse
  387                   (\i -> go (pfx <> "_" <> Text.pack (show i)) elemShp)
  388                   [0..n-1]
  389               pure $ MS.SetupArray elemTy elems
  390             _ ->
  391               panic "constructExpandedSetupValue" [
  392                   "ArrayShape with non-TyArray type: " ,
  393                   "   " <> Text.pack (show (PP.pretty ty))
  394               ]
  395         StructShape ty _ fldShps ->
  396           case ty of
  397             Mir.TyAdt adtName _ _ ->
  398               case col ^. Mir.adts . at adtName of
  399                 Just adt@(Mir.Adt _ kind _ _ _ _ _) ->
  400                   case kind of
  401                     Mir.Struct -> do
  402                       flds <- goFlds pfx fldShps
  403                       pure $ MS.SetupStruct adt flds
  404                     _ ->
  405                       panic "constructExpandedSetupValue" [
  406                           "Expected struct, encountered " <> Text.pack (show kind)
  407                       ]
  408                 Nothing ->
  409                   adt_not_found_panic "StructShape" adtName
  410             _ ->
  411               non_adt_type_panic "StructShape" ty
  412         EnumShape ty _ variantShps _ discrShp ->
  413           case ty of
  414             Mir.TyAdt adtName _ _ ->
  415               case col ^. Mir.adts . at adtName of
  416                 Just adt@(Mir.Adt _ kind _ _ _ _ _) ->
  417                   case kind of
  418                     Mir.Enum _ ->
  419                       MS.SetupEnum <$> goEnum pfx adt discrShp variantShps
  420                     Mir.Struct ->
  421                       panic "constructExpandedSetupValue" [
  422                           "Expected enum, encountered struct"
  423                       ]
  424                     Mir.Union ->
  425                       panic "constructExpandedSetupValue" [
  426                           "Expected enum, encountered union"
  427                       ]
  428                 Nothing ->
  429                   adt_not_found_panic "EnumShape" adtName
  430             _ ->
  431               non_adt_type_panic "EnumShape" ty
  432         TransparentShape ty shp' ->
  433           case ty of
  434             Mir.TyAdt adtName _ _ -> do
  435               case col ^. Mir.adts . at adtName of
  436                 Just adt@(Mir.Adt adtNm kind variants _ _ _ _) ->
  437                   case kind of
  438                     Mir.Struct -> do
  439                       val <- go pfx shp'
  440                       pure $ MS.SetupStruct adt [val]
  441                     Mir.Enum{}
  442                       -- `repr(transparent)` enum values use MirSetupEnumVariant
  443                       -- rather than MirSetupEnumSymbolic. See the Haddocks for
  444                       -- MirSetupEnumSymbolic for an explanation.
  445                       |  [variant] <- variants
  446                       -> do val <- go pfx shp'
  447                             pure $ MS.SetupEnum
  448                                  $ MirSetupEnumVariant adt variant 0 [val]
  449 
  450                       |  otherwise
  451                       -> panic "constructExpandedSetupValue" [
  452                              "`repr(transparent)` enum that doesn't have exactly one variant",
  453                              "Enum: " <> Text.pack (show adtNm),
  454                              "Number of variants: " <> Text.pack (show (length variants))
  455                          ]
  456                     Mir.Union ->
  457                       panic "constructExpandedSetupValue" [
  458                           "Unexpected `repr(transparent)` union: " <>
  459                               Text.pack (show adtName)
  460                       ]
  461                 Nothing ->
  462                   adt_not_found_panic "TransparentShape" adtName
  463             _ ->
  464               non_adt_type_panic "TransparentShape" ty
  465         RefShape ty _ _ _ ->
  466           X.throwM $ MIRFreshExpandedValueUnsupportedType ty
  467         SliceShape ty _ _ _ ->
  468           X.throwM $ MIRFreshExpandedValueUnsupportedType ty
  469         FnPtrShape ty _ _ ->
  470           X.throwM $ MIRFreshExpandedValueUnsupportedType ty

  472     -- Create a fresh symbolic enum value, as described in
  473     -- Note [Symbolic enums] in SAWCentral.Crucible.MIR.Setup.Value.
  474     goEnum ::
  475       forall discrShp variantCtx.
  476       Text ->
  477       Mir.Adt ->
  478       TypeShape discrShp ->
  479       Ctx.Assignment VariantShape variantCtx ->
  480       CrucibleSetup MIR MirSetupEnum
  481     goEnum pfx adt@(Mir.Adt _ _ variants _ _ _ _) discrShp variantShps =
  482       mccWithBackend cc $ \bak ->
  483       do -- First, create a symbolic discriminant value.
  484          IsBVShape discrTy discrW <- pure $ testDiscriminantIsBV discrShp
  485          let discrPfx = pfx <> "_discr"
  486          discrVar <- freshPrimVariable discrPfx discrTy
  487          let discrVal = MS.SetupTerm discrVar
  488 
  489          -- Next, add Crucible assumptions that constraint the discriminant
  490          -- to be equal to one of the possible variants' discriminant values.
  491          -- This assumption will be of the form:
  492          --
  493          --   (discr == 0) \/ (discr == 1) \/ ...
  494          --
  495          -- It's tempting to simplify this assumption to just
  496          -- `discr < num_variants`, but this will not work in the event that
  497          -- the enum uses explicit discriminants for some variants, e.g.,
  498          --
  499          --   enum E {
  500          --       E0 = 42,
  501          --       E1,
  502          --   }
  503          discrWNat <- liftIO $ scNat sc $ natValue discrW
  504          possibleDiscrTerms <- liftIO $
  505            traverse (\discr -> do
  506                       discrNat  <- scNat sc $ fromInteger discr
  507                       scBvNat sc discrWNat discrNat)
  508                     (map getEnumVariantDiscr variants)
  509          scFalse <- liftIO $ scBool sc False
  510          possibleDiscrPredTerm <- liftIO $
  511            foldM
  512              (\z possibleDiscrTerm -> do
  513                p <- scBvEq sc discrWNat (ttTerm discrVar) possibleDiscrTerm
  514                scOr sc p z)
  515              scFalse
  516              possibleDiscrTerms
  517          possibleDiscrPred <- liftIO $ resolveSAWPred cc possibleDiscrPredTerm
  518          loc <- SS.toW4Loc "mir_fresh_expanded_value" <$> lift (lift getPosition)
  519          liftIO $ Crucible.addAssumption bak $
  520            Crucible.GenericAssumption
  521              loc "Symbolic enum discriminant constraints" possibleDiscrPred
  522 
  523          -- Finally, create symbolic fields for each of the possible variants.
  524          let variantAssns :: [Some (Ctx.Assignment FieldShape)]
  525              variantAssns =
  526                FC.toListFC
  527                  (\(VariantShape fldShps) -> Some fldShps)
  528                  variantShps
  529          variantVals <-
  530            zipWithM
  531              (\variant (Some fldShps) ->
  532                let variantPfx = pfx <> "_" <> getEnumVariantShortName variant in
  533                goFlds variantPfx fldShps)
  534              variants
  535              variantAssns
  536 
  537          pure $ MirSetupEnumSymbolic adt discrVal variantVals

  539     goFlds :: forall ctx.
  540               Text ->
  541               Ctx.Assignment FieldShape ctx ->
  542               CrucibleSetup MIR [SetupValue]
  543     goFlds pfx fldShps = sequenceA $
  544       FCI.itoListFC
  545         (\idx fldShp ->
  546           let pfx' = pfx <> "_" <> Text.pack (show idx) in
  547           case fldShp of
  548             ReqField shp' -> go pfx' shp'
  549             OptField shp' -> go pfx' shp')
  550         fldShps

  727 mir_verify ::
  728   Mir.RustModule ->
  729   Text {- ^ method name -} ->
  730   [Lemma] {- ^ overrides -} ->
  731   Bool {- ^ path sat checking -} ->
  732   MIRSetupM () ->
  733   ProofScript () ->
  734   TopLevel Lemma
  735 mir_verify rm nm lemmas checkSat setup tactic =
  736   do start <- io getCurrentTime
  737      opts <- getOptions
  738 
  739      -- set up the metadata map for tracking proof obligation metadata
  740      mdMap <- io $ newIORef mempty
  741 
  742      cc <- setupCrucibleContext rm
  743      SomeOnlineBackend bak <- pure (cc^.mccBackend)
  744      let sym = cc^.mccSym
  745      let globals0 = cc^.mccSymGlobalState
  746 
  747      sosp <- rwSingleOverrideSpecialCase <$> getTopLevelRW
  748      let ?singleOverrideSpecialCase = sosp
  749 
  750      pos <- getPosition
  751      let loc = SS.toW4Loc "_SAW_mir_verify" pos
  752 
  753      profFile <- rwProfilingFile <$> getTopLevelRW
  754      (writeFinalProfile, pfs) <- io $ setupProfiling sym "mir_verify" profFile
  755 
  756      fn <- findFn rm nm
  757      let st0 = initialCrucibleSetupState cc fn loc
  758 
  759      -- execute commands of the method spec
  760      io $ W4.setCurrentProgramLoc sym loc
  761      methodSpec <- view Setup.csMethodSpec <$>
  762                      execStateT
  763                        (runReaderT (runMIRSetupM setup) Setup.makeCrucibleSetupRO)
  764                      st0
  765 
  766      printOutLnTop Info $
  767        unwords ["Verifying", show (methodSpec ^. MS.csMethod), "..."]
  768 
  769      -- construct the initial state for verifications
  770      (args, assumes, env, globals1) <- io $ verifyPrestate cc methodSpec globals0
  771 
  772      -- save initial path conditions
  773      frameIdent <- io $ Crucible.pushAssumptionFrame bak
  774 
  775      -- check for contradictory preconditions
  776      when (detectVacuity opts) $
  777        Vacuity.checkAssumptionsForContradictions
  778          sym
  779          methodSpec
  780          tactic
  781          assumes
  782 
  783      -- run the symbolic execution
  784      printOutLnTop Info $
  785        unwords ["Simulating", show (methodSpec ^. MS.csMethod), "..."]
  786      top_loc <- SS.toW4Loc "mir_verify" <$> getPosition
  787      (ret, globals2) <-
  788        io $ verifySimulate opts cc pfs methodSpec args assumes top_loc lemmas globals1 checkSat mdMap
  789 
  790      -- collect the proof obligations
  791      asserts <- verifyPoststate cc
  792                     methodSpec env globals2 ret mdMap
  793 
  794      -- restore previous assumption state
  795      _ <- io $ Crucible.popAssumptionFrame bak frameIdent
  796 
  797      -- attempt to verify the proof obligations
  798      printOutLnTop Info $
  799        unwords ["Checking proof obligations", show (methodSpec ^. MS.csMethod), "..."]
  800      (stats,vcstats) <- verifyObligations cc methodSpec tactic assumes asserts
  801      io $ writeFinalProfile
  802 
  803      let lemmaSet = Set.fromList (map (view MS.psSpecIdent) lemmas)
  804      end <- io getCurrentTime
  805      let diff = diffUTCTime end start
  806      ps <- io (MS.mkProvedSpec MS.SpecProved methodSpec stats vcstats lemmaSet diff)
  807      returnMIRProof ps

  813 -- | Construct a specific enum variant. This does a light amount of validity
  814 -- checking, which is the only reason that this function is monadic.
  815 mir_enum_value ::
  816   X.MonadThrow m =>
  817   Mir.Adt ->
  818   Text ->
  819   [MS.SetupValue MIR] ->
  820   m (MS.SetupValue MIR)
  821 mir_enum_value adt variantNm vs =
  822   case adt of
  823     Mir.Adt adtNm (Mir.Enum _) variants _ _ _ _ -> do
  824       (variantIdx, variant) <-
  825         case FWI.ifind (\_ v -> variantDefIdMatches v) variants of
  826           Just iv ->
  827             pure iv
  828           Nothing ->
  829             X.throwM $ MIREnumValueVariantNotFound adtNm variantNm
  830       pure $ MS.SetupEnum $ MirSetupEnumVariant adt variant variantIdx vs
  831     Mir.Adt adtNm Mir.Struct _ _ _ _ _ ->
  832       X.throwM $ MIREnumValueNonEnum adtNm "struct"
  833     Mir.Adt adtNm Mir.Union _ _ _ _ _ ->
  834       X.throwM $ MIREnumValueNonEnum adtNm "union"
  835   where
  836     -- Check if the user-supplied String argument matches the name of the given
  837     -- variant's DefId. For instance, the variant DefId might be named
  838     -- @core::option[0]::Option[0]::Some[0]@, but the user will simply write
  839     -- @Some@, so we must strip off the other parts of the DefId before checking
  840     -- if the two are the same.
  841     variantDefIdMatches :: Mir.Variant -> Bool
  842     variantDefIdMatches variant =
  843       getEnumVariantShortName variant == variantNm

 1007 resolveArguments ::
 1008   MIRCrucibleContext ->
 1009   MethodSpec ->
 1010   Map MS.AllocIndex (Some (MirPointer Sym)) ->
 1011   IO [(Mir.Ty, MIRVal)]
 1012 resolveArguments cc mspec env = mapM resolveArg [0..(nArgs-1)]
 1013   where
 1014     nArgs = toInteger (length (mspec ^. MS.csArgs))
 1015     tyenv = MS.csAllocations mspec
 1016     nameEnv = mspec ^. MS.csPreState . MS.csVarTypeNames

 1019     checkArgTy i mt mt' =
 1020       unless (checkCompatibleTys mt mt') $
 1021            fail $ unlines [ "Type mismatch in argument " ++ show i ++ " when verifying " ++ show nm
 1022                           , "Argument is declared with type: " ++ show mt
 1023                           , "but provided argument has incompatible type: " ++ show mt'
 1024                           , "Note: this may be because the signature of your " ++
 1025                             "function changed during compilation."
 1026                           ]
 1027     resolveArg i =
 1028       case Map.lookup i (mspec ^. MS.csArgBindings) of
 1029         Just (mt, sv) -> do
 1030           mt' <- typeOfSetupValue cc tyenv nameEnv sv
 1031           checkArgTy i mt mt'

 1033           return (mt, v)

 1049 -- | Sets up globals (ghost variable), and collects boolean terms
 1050 -- that should be assumed to be true.
 1051 setupPrestateConditions ::
 1052   MethodSpec ->
 1053   MIRCrucibleContext ->
 1054   Map MS.AllocIndex (Some (MirPointer Sym)) ->
 1055   Crucible.SymGlobalState Sym ->
 1056   [SetupCondition] ->
 1057   IO ( Crucible.SymGlobalState Sym, [Crucible.LabeledPred Term AssumptionReason]
 1058      )
 1059 setupPrestateConditions mspec cc env = aux []
 1060   where
 1061     tyenv   = MS.csAllocations mspec
 1062     nameEnv = mspec ^. MS.csPreState . MS.csVarTypeNames
 1063 
 1064     aux acc globals [] = return (globals, acc)
 1065 
 1066     aux acc globals (MS.SetupCond_Equal loc val1 val2 : xs) =
 1067       do val1' <- resolveSetupVal cc env tyenv nameEnv val1
 1068          val2' <- resolveSetupVal cc env tyenv nameEnv val2
 1069          t     <- assertEqualVals cc val1' val2'
 1070          let lp = Crucible.LabeledPred t (loc, "equality precondition")
 1071          aux (lp:acc) globals xs
 1072 
 1073     aux acc globals (MS.SetupCond_Pred loc tm : xs) =
 1074       let lp = Crucible.LabeledPred (ttTerm tm) (loc, "precondition") in
 1075       aux (lp:acc) globals xs

 1077     aux acc globals (MS.SetupCond_Ghost _md var val : xs) =
 1078       case val of
 1079         TypedTerm (TypedTermSchema sch) tm ->
 1080           aux acc (Crucible.insertGlobal var (sch,tm) globals) xs

 1083             [ "Setup term for global variable expected to have Cryptol schema type, but got"

 1087 verifyObligations ::
 1088   MIRCrucibleContext ->
 1089   MethodSpec ->
 1090   ProofScript () ->
 1091   [Crucible.LabeledPred Term AssumptionReason] ->
 1092   [(String, MS.ConditionMetadata, Term)] ->
 1093   TopLevel (SolverStats, [MS.VCStats])
 1094 verifyObligations cc mspec tactic assumes asserts =
 1095   do let sym = cc^.mccSym
 1096      st <- io $ sawCoreState sym
 1097      let sc = saw_ctx st
 1098      assume <- io $ scAndList sc (toListOf (folded . Crucible.labeledPred) assumes)
 1099      let nm = show $ mspec ^. MS.csMethod
 1100      outs <- forM (zip [(0::Int)..] asserts) $ \(n, (msg, md, assert)) -> do
 1101        goal   <- io $ scImplies sc assume assert
 1102        goal'  <- io $ boolToProp sc [] goal -- TODO, generalize over inputs
 1103        let ploc = MS.conditionLoc md
 1104        let gloc = (unwords [show (W4.plSourceLoc ploc)
 1105                           ,"in"
 1106                           , show (W4.plFunction ploc)]) ++
 1107                   (if Prelude.null (MS.conditionContext md) then [] else
 1108                      "\n" ++ MS.conditionContext md)
 1109        let goalname = concat [nm, " (", takeWhile (/= '\n') msg, ")"]
 1110        let proofgoal = ProofGoal
 1111                        { goalNum  = n
 1112                        , goalType = MS.conditionType md
 1113                        , goalName = nm
 1114                        , goalLoc  = gloc
 1115                        , goalDesc = msg
 1116                        , goalSequent = propToSequent goal'
 1117                        , goalTags = MS.conditionTags md
 1118                        }
 1119        res <- runProofScript tactic goal' proofgoal (Just ploc)
 1120                 (Text.unwords
 1121                  ["MIR verification condition:", Text.pack (show n), Text.pack goalname])
 1122                 False -- do not record in the theorem database
 1123                 False -- TODO, useSequentGoals...
 1124        case res of
 1125          ValidProof stats thm ->
 1126            return (stats, MS.VCStats md stats (thmSummary thm) (thmNonce thm) (thmDepends thm) (thmElapsedTime thm))
 1127          InvalidProof stats vals _pst -> do
 1128            printOutLnTop Info $ unwords ["Subgoal failed:", nm, msg]
 1129            printOutLnTop Info (show stats)
 1130            printOutLnTop OnlyCounterExamples "----------Counterexample----------"
 1131            opts <- rwPPOpts <$> getTopLevelRW
 1132            let showEC ec = Text.unpack (toShortName (ecName ec))
 1133            let showAssignment (name, val) = "  " ++ showEC name ++ ": " ++ show (ppFirstOrderValue opts val)
 1134            mapM_ (printOutLnTop OnlyCounterExamples . showAssignment) vals
 1135            io $ fail "Proof failed." -- Mirroring behavior of llvm_verify
 1136          UnfinishedProof pst ->
 1137            io $ fail $ "Proof failed " ++ show (length (psGoals pst)) ++ " goals remaining."
 1138 
 1139      printOutLnTop Info $ unwords ["Proof succeeded!", nm]
 1140 
 1141      let stats = mconcat (map fst outs)
 1142      let vcstats = map snd outs
 1143      return (stats, vcstats)

 1145 verifyPoststate ::
 1146   MIRCrucibleContext                        {- ^ crucible context        -} ->
 1147   MethodSpec                                {- ^ specification           -} ->
 1148   Map MS.AllocIndex (Some (MirPointer Sym)) {- ^ allocation substitution -} ->
 1149   Crucible.SymGlobalState Sym               {- ^ global variables        -} ->
 1150   Maybe MIRVal                              {- ^ optional return value   -} ->
 1151   IORef MetadataMap                         {- ^ metadata map            -} ->
 1152   TopLevel [(String, MS.ConditionMetadata, Term)] {- ^ generated labels and verification conditions -}
 1153 verifyPoststate cc mspec env0 globals ret mdMap =
 1154   mccWithBackend cc $ \bak ->
 1155   do opts <- getOptions
 1156      sc <- getSharedContext
 1157      poststateLoc <- SS.toW4Loc "_SAW_MIR_verifyPoststate" <$> getPosition
 1158      io $ W4.setCurrentProgramLoc sym poststateLoc
 1159 
 1160      -- This discards all the obligations generated during
 1161      -- symbolic execution itself, i.e., which are not directly
 1162      -- generated from specification postconditions. This
 1163      -- is, in general, unsound.
 1164      skipSafetyProofs <- gets rwSkipSafetyProofs
 1165      when skipSafetyProofs (io (Crucible.clearProofObligations bak))
 1166 
 1167      let ecs0 = IntMap.fromList
 1168            [ (ecVarIndex ec, ec)
 1169            | tt <- mspec ^. MS.csPreState . MS.csFreshVars
 1170            , let ec = tecExt tt ]

 1601 {-
 1602 Note [Translating MIR statics in SAW]
 1603 ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
 1604 Translating top-level static values in the SAW MIR backend requires some care.
 1605 This is because we want to treat immutable and mutable static differently:
 1606 
 1607 * Immutable statics should be implicitly initialized in each specification
 1608   without any action required on the user's end.
 1609 
 1610 * Mutable statics should /not/ be initialized in a specification unless a user
 1611   explicitly declares it with `mir_points_to (mir_static ...) ...`. This is
 1612   because a mutable static's value can change over the course of a program, so
 1613   we require users to be precise about what the value should be before and after
 1614   invoking a function.
 1615 
 1616 This poses a challenge when translating static values in SAW. It is tempting to
 1617 only translate the immutable statics and not the mutable statics (similarly to
 1618 how the SAW LLVM backend handles translation), but this will not work in
 1619 general. Consider this program:
 1620 
 1621   static mut S1: u32 = 1;
 1622   static S2: u32 = unsafe { S1 };
 1623 
 1624   pub fn f() {
 1625       ... S2 ...
 1626   }
 1627 
 1628 The initial value of S2 (an immutable static) depends on having first translated
 1629 the initial value of S1 (a mutable static). If we do not translate mutable
 1630 statics in SAW, the translation of S1 will not succeed.
 1631 
 1632 We solve this problem by doing the following:
 1633 
 1634 1. Use Crucible to translate all top-level static values (both immutable and
 1635    mutable). This produces a SymGlobalState that contains every static's
 1636    initializer value.
 1637 
 1638 2. Consult the updated SymGlobalState to produce the following:
 1639 
 1640    (a) A subset of the SymGlobalState that only contains the initializer values
 1641        for the immutable statics. This will be used later to initialize the
 1642        Crucible state when translating the function being verified with
 1643        mir_verify.
 1644 
 1645    (b) A MirStaticInitializerMap mapping all static variables (both immutable
 1646        and mutable) to their initializer values. This will be used later to
 1647        power the mir_static_initializer function.
 1648 
 1649 This is a bit clunky, but it's unclear how to do better without significantly
 1650 changing how crucible-mir translates top-level static values.
 1651 -}
