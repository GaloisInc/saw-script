  287 data Value
  288   = VBool Bool
  289   | VString Text
  290   | VInteger Integer
  291   | VArray [Value]
  292   | VTuple [Value]
  293   | VRecord (Map SS.Name Value)
  294   | VLambda LocalEnv SS.Pattern SS.Expr
  295     -- | Function-shaped value that's a Haskell-level function. This
  296     --   is how builtins appear.
  297     --
  298     -- XXX: Calling this "VBuiltin" was optimistic. It actually covers
  299     -- everything function-shaped that isn't a SAWScript-level lambda,
  300     -- which includes not just builtins but also the closures used to
  301     -- implement stack traces and possibly other messes, all of which
  302     -- should be removed.
  303   | VBuiltin (Value -> TopLevel Value)
  304   | VTerm TypedTerm
  305   | VType Cryptol.Schema
  306     -- | Returned value in unspecified monad
  307   | VReturn Value
  308     -- | Not-yet-executed do-block in unspecified monad
  309     --
  310     --   The string is a hack hook for the current implementation of
  311     --   stack traces. See the commit message that added it for further
  312     --   information. XXX: to be removed along with the stack trace code
  313   | VDo SS.Pos (Maybe String) LocalEnv [SS.Stmt]
  314     -- | Single monadic bind in unspecified monad.
  315     --   This exists only to support the "for" builtin; see notes there
  316     --   for why this is so. XXX: remove it once that's no longer needed
  317   | VBindOnce Value Value
  318   | VTopLevel (TopLevel Value)
  319   | VProofScript (ProofScript Value)
  320   | VSimpset SAWSimpset
  321   | VRefnset SAWRefnset
  322   | VTheorem Theorem
  323   | VBisimTheorem BisimTheorem
  324   -----
  325   | VLLVMCrucibleSetup !(LLVMCrucibleSetupM Value)
  326   | VLLVMCrucibleMethodSpec (CMSLLVM.SomeLLVM CMS.ProvedSpec)
  327   | VLLVMCrucibleSetupValue (CMSLLVM.AllLLVM CMS.SetupValue)
  328   -----
  329   | VJVMSetup !(JVMSetupM Value)
  330   | VJVMMethodSpec !(CMS.ProvedSpec CJ.JVM)
  331   | VJVMSetupValue !(CMS.SetupValue CJ.JVM)
  332   -----
  333   | VMIRSetup !(MIRSetupM Value)
  334   | VMIRMethodSpec !(CMS.ProvedSpec MIR)
  335   | VMIRSetupValue !(CMS.SetupValue MIR)
  336   -----
  337   | VLLVMModuleSkeleton ModuleSkeleton
  338   | VLLVMFunctionSkeleton FunctionSkeleton
  339   | VLLVMSkeletonState SkeletonState
  340   | VLLVMFunctionProfile FunctionProfile
  341   -----
  342   | VJavaType JavaType
  343   | VLLVMType LLVM.Type
  344   | VMIRType MIR.Ty
  345   | VCryptolModule CryptolModule
  346   | VJavaClass JSS.Class
  347   | VLLVMModule (Some CMSLLVM.LLVMModule)
  348   | VMIRModule RustModule
  349   | VMIRAdt MIR.Adt
  350   | VHeapsterEnv HeapsterEnv
  351   | VSatResult SatResult
  352   | VProofResult ProofResult
  353   | VUninterp Uninterp
  354   | VAIG AIGNetwork
  355   | VCFG SAW_CFG
  356   | VGhostVar CMS.GhostGlobal
  357   | VYosysModule YosysIR
  358   | VYosysImport YosysImport
  359   | VYosysSequential YosysSequential
  360   | VYosysTheorem YosysTheorem

  402 data SatResult
  403   = Unsat SolverStats
  404   | Sat SolverStats [(ExtCns Term, FirstOrderValue)]
  405   | SatUnknown
  406     deriving (Show)

  408 showsProofResult :: PPS.Opts -> ProofResult -> ShowS
  409 showsProofResult opts r =
  410   case r of
  411     ValidProof _ _ -> showString "Valid"
  412     InvalidProof _ ts _ -> showString "Invalid: [" . showMulti "" ts
  413     UnfinishedProof st  -> showString "Unfinished: " . shows (length (psGoals st)) . showString " goals remaining"
  414   where
  415     showVal t = shows (ppFirstOrderValue opts t)
  416     showEqn (x, t) = showEC x . showString " = " . showVal t
  417     showEC ec = showString (Text.unpack (toShortName (ecName ec)))
  418 
  419     showMulti _ [] = showString "]"
  420     showMulti s (eqn : eqns) = showString s . showEqn eqn . showMulti ", " eqns
  421 
  422 showsSatResult :: PPS.Opts -> SatResult -> ShowS
  423 showsSatResult opts r =
  424   case r of
  425     Unsat _ -> showString "Unsat"
  426     Sat _ ts -> showString "Sat: [" . showMulti "" ts
  427     SatUnknown  -> showString "Unknown"
  428   where
  429     showVal t = shows (ppFirstOrderValue opts t)
  430     showEC ec = showString (Text.unpack (toShortName (ecName ec)))
  431     showEqn (x, t) = showEC x . showString " = " . showVal t
  432     showMulti _ [] = showString "]"
  433     showMulti s (eqn : eqns) = showString s . showEqn eqn . showMulti ", " eqns
  434 
  435 showSimpset :: PPS.Opts -> Simpset a -> String
  436 showSimpset opts ss =
  437   unlines ("Rewrite Rules" : "=============" : map (show . ppRule) (listRules ss))
  438   where
  439     ppRule r =
  440       PP.pretty '*' PP.<+>
  441       (PP.nest 2 $ PP.fillSep
  442        [ ppTerm (lhsRewriteRule r)
  443        , PP.pretty '=' PP.<+> ppTerm (rhsRewriteRule r) ])
  444     ppTerm t = SAWCorePP.ppTerm opts t
  445 
  446 -- | Pretty-print a 'Refnset' to a 'String'
  447 showRefnset :: PPS.Opts -> MRSolver.Refnset a -> String
  448 showRefnset opts ss =
  449   unlines ("Refinements" : "=============" : map (show . ppFunAssump)
  450                                                  (MRSolver.listFunAssumps ss))
  451   where
  452     ppFunAssump (MRSolver.FunAssump ctx f args rhs _) =
  453       PP.pretty '*' PP.<+>
  454       (PP.nest 2 $ PP.fillSep
  455        [ ppTermAppInCtx opts ctx (funNameTerm f) args
  456        , PP.pretty ("|=" :: String) PP.<+> ppFunAssumpRHS ctx rhs ])
  457     ppFunAssumpRHS ctx (OpaqueFunAssump f args) =
  458       ppTermAppInCtx opts ctx (funNameTerm f) args
  459     ppFunAssumpRHS ctx (RewriteFunAssump rhs) =
  460       SAWCorePP.ppTermInCtx opts (map fst $ mrVarCtxInnerToOuter ctx) rhs

  462 -- XXX the precedence in here needs to be cleaned up
  463 showsPrecValue :: PPS.Opts -> DisplayNameEnv -> Int -> Value -> ShowS
  464 showsPrecValue opts nenv p v =
  465   case v of
  466     VBool True -> showString "true"
  467     VBool False -> showString "false"
  468     VString s -> shows s
  469     VInteger n -> shows n
  470     VArray vs -> PPS.showBrackets $ PPS.showCommaSep $ map (showsPrecValue opts nenv 0) vs
  471     VTuple vs -> showParen True $ PPS.showCommaSep $ map (showsPrecValue opts nenv 0) vs
  472     VRecord m ->
  473       PPS.showBraces $ PPS.showCommaSep $ map showFld (M.toList m)
  474         where
  475           showFld (n, fv) =
  476             showString (Text.unpack n) . showString "=" . showsPrecValue opts nenv 0 fv
  477 
  478     VLambda _env pat e ->
  479       let pat' = PP.pretty pat
  480           e' = PP.pretty e
  481       in
  482       shows $ PP.sep ["\\", pat', "->", e']
  483 
  484     VBuiltin {} -> showString "<<builtin>>"
  485     VTerm t -> showString (SAWCorePP.showTermWithNames opts nenv (ttTerm t))
  486     VType sig -> showString (pretty sig)
  487     VReturn v' -> showString "return " . showsPrecValue opts nenv (p + 1) v'
  488     VDo pos _name _env stmts ->
  489       let e = SS.Block pos stmts in
  490       shows (PP.pretty e)
  491     VBindOnce v1 v2 ->
  492       let v1' = showsPrecValue opts nenv 0 v1
  493           v2' = showsPrecValue opts nenv 0 v2
  494       in
  495       v1' . showString " >>= " . v2'
  496     VTopLevel {} -> showString "<<TopLevel>>"
  497     VSimpset ss -> showString (showSimpset opts ss)
  498     VRefnset ss -> showString (showRefnset opts ss)
  499     VProofScript {} -> showString "<<proof script>>"
  500     VTheorem thm ->
  501       showString "Theorem " .
  502       showParen True (showString (prettyProp opts nenv (thmProp thm)))
  503     VBisimTheorem _ -> showString "<<Bisimulation theorem>>"
  504     VLLVMCrucibleSetup{} -> showString "<<Crucible Setup>>"
  505     VLLVMCrucibleSetupValue{} -> showString "<<Crucible SetupValue>>"
  506     VLLVMCrucibleMethodSpec{} -> showString "<<Crucible MethodSpec>>"
  507     VLLVMModuleSkeleton s -> shows s
  508     VLLVMFunctionSkeleton s -> shows s
  509     VLLVMSkeletonState _ -> showString "<<Skeleton state>>"

  701 instance MonadIO TopLevel where
  702   liftIO = io

  704 io :: IO a -> TopLevel a
  705 io f = (TopLevel_ (liftIO f))
  706        `catches`
  707        [ Handler (\(ex :: X86Unsupported) -> handleX86Unsupported ex)
  708        , Handler (\(ex :: X86Error) -> handleX86Error ex)
  709        , Handler handleTopLevel
  710        , Handler handleIO
  711        ]
  712   where
  713     rethrow :: X.Exception ex => ex -> TopLevel a
  714     rethrow ex =
  715       do stk <- getStackTrace
  716          throwM (SS.TraceException stk (X.SomeException ex))
  717 
  718     handleTopLevel :: SS.TopLevelException -> TopLevel a
  719     handleTopLevel e = rethrow e
  720 
  721     handleIO :: X.IOException -> TopLevel a
  722     handleIO e
  723       | IOError.isUserError e =
  724           do pos <- getPosition
  725              rethrow (SS.TopLevelException pos (dropEnd 1 . drop 12 $ show e))
  726       | otherwise = rethrow e
  727 
  728     handleX86Unsupported (X86Unsupported path s) =
  729       do let pos = SS.FileOnlyPos path
  730          rethrow (SS.TopLevelException pos ("Unsupported x86 feature: " ++ s))
  731 
  732     handleX86Error (X86Error path optfunc s) =
  733       do let pos = case optfunc of
  734                Nothing -> SS.FileOnlyPos path
  735                Just func -> SS.FileAndFunctionPos path (Text.unpack func)
  736          rethrow (SS.TopLevelException pos ("Error in x86 code: " ++ Text.unpack s))
  737 
  738 throwTopLevel :: String -> TopLevel a
  739 throwTopLevel msg = do
  740   pos <- getPosition
  741   stk <- getStackTrace
  742   throwM (SS.TraceException stk (X.SomeException (SS.TopLevelException pos msg)))

  779 getJavaCodebase :: TopLevel JSS.Codebase
  780 getJavaCodebase =
  781   do
  782     status <- TopLevel_ (gets rwJavaCodebase)
  783     case status of
  784       JavaInitialized s -> pure s
  785       JavaUnitialized   ->
  786         do
  787           opts <- getOptions
  788           mb   <- liftIO (X.try (initJava opts))
  789           case mb of
  790             Right jcb ->
  791               TopLevel_ (modify (\s -> s { rwJavaCodebase = JavaInitialized jcb })) >>
  792               pure jcb
  793             Left err ->
  794               fail $ unlines $
  795                 "Error: failed to initialize Java." :
  796                 [ "  " ++ l | l <- lines (X.displayException (err :: X.SomeException)) ]
  797   where
  798   -- If a Java executable's path is specified (either by way of
  799   -- --java-bin-dirs or PATH, see the Haddocks for findJavaIn), then use that
  800   -- to detect the path to Java's standard rt.jar file and add it to the
  801   -- jarList on Java 8 or earlier. (Later versions of Java do not use
  802   -- rt.jar—see Note [Loading classes from JIMAGE files] in
  803   -- Lang.JVM.Codebase in crucible-jvm.)
  804   -- If Java's path is not specified, return the Options unchanged.
  805   initJava opts =
  806     do
  807       mbJavaPath <- CJ.findJavaIn (Opt.javaBinDirs opts)
  808       javaPath <-
  809         case mbJavaPath of
  810           Nothing   -> fail "Failed to find a `java` executable."
  811           Just path -> pure path
  812       version <- CJ.findJavaMajorVersion javaPath
  813       jars <-
  814         -- rt.jar lives in a standard location relative to @java.home@. At least,
  815         -- this is the case on every operating system I've tested.
  816         if version < 9 then
  817           do
  818             mbJavaHome <- CJ.findJavaProperty javaPath "java.home"
  819             case mbJavaHome of
  820               Nothing ->
  821                 fail ("Could not find where rt.jar lives for " ++ javaPath)
  822               Just javaHome ->
  823                 let jar = javaHome </> "lib" </> "rt.jar"
  824                 in pure (jar : Opt.jarList opts)
  825         else pure (Opt.jarList opts)
  826       JSS.loadCodebase jars (Opt.classPath opts) (Opt.javaBinDirs opts)

  926 extendEnv ::
  927   SharedContext ->
  928   SS.LName -> SS.Schema -> Maybe String -> Value -> TopLevelRW -> IO TopLevelRW
  929 extendEnv sc x ty md v rw =
  930   do ce' <-
  931        case v of
  932          VTerm t ->
  933            pure $ CEnv.bindTypedTerm (ident, t) ce
  934          VType s ->
  935            pure $ CEnv.bindType (ident, s) ce
  936          VInteger n ->
  937            pure $ CEnv.bindInteger (ident, n) ce
  938          VCryptolModule m ->
  939            pure $ CEnv.bindCryptolModule (modname, m) ce
  940          VString s ->
  941            do tt <- typedTermOfString sc (Text.unpack s)
  942               pure $ CEnv.bindTypedTerm (ident, tt) ce
  943          VBool b ->
  944            do tt <- typedTermOfBool sc b
  945               pure $ CEnv.bindTypedTerm (ident, tt) ce
  946          _ ->
  947            pure ce
  948      pure $
  949       rw { rwValueInfo  = M.insert name (SS.Current, ty, v) (rwValueInfo rw)
  950          , rwDocs    = maybeInsert (SS.getVal name) md (rwDocs rw)
  951          , rwCryptol = ce'
  952          }
  953   where
  954     name = x
  955     -- XXX why is this using getOrig?
  956     ident = T.mkIdent (SS.getOrig x)
  957     modname = T.packModName [SS.getOrig x]
  958     ce = rwCryptol rw
  959 
  960 typedTermOfString :: SharedContext -> String -> IO TypedTerm
  961 typedTermOfString sc str =
  962   do let schema = Cryptol.tMono (Cryptol.tString (length str))
  963      bvNat <- scGlobalDef sc "Prelude.bvNat"
  964      bvNat8 <- scApply sc bvNat =<< scNat sc 8
  965      byteT <- scBitvector sc 8
  966      let scChar c = scApply sc bvNat8 =<< scNat sc (fromIntegral (fromEnum c))
  967      ts <- traverse scChar str
  968      trm <- scVector sc byteT ts
  969      pure (TypedTerm (TypedTermSchema schema) trm)
  970 
  971 typedTermOfBool :: SharedContext -> Bool -> IO TypedTerm
  972 typedTermOfBool sc b =
  973   do let schema = Cryptol.tMono Cryptol.tBit
  974      trm <- scBool sc b
  975      pure (TypedTerm (TypedTermSchema schema) trm)

 1054 -- TODO: remove the "reason" parameter and compute it from the
 1055 --       initial proof goal instead
 1056 runProofScript ::
 1057   ProofScript a ->
 1058   Prop ->
 1059   ProofGoal ->
 1060   Maybe ProgramLoc ->
 1061   Text ->
 1062   Bool {- ^ record the theorem in the database? -} ->
 1063   Bool {- ^ do we need to normalize the sequent goal? -} ->
 1064   TopLevel ProofResult
 1065 runProofScript (ProofScript m) concl gl ploc rsn recordThm useSequentGoals =
 1066   do pos <- getPosition
 1067      ps <- io (startProof gl pos ploc rsn)
 1068      (r,pstate) <- runStateT (runExceptT m) ps
 1069      case r of
 1070        Left (stats,cex) -> return (SAWCentral.Proof.InvalidProof stats cex pstate)
 1071        Right _ ->
 1072          do sc <- getSharedContext
 1073             db <- getTheoremDB
 1074             what4PushMuxOps <- gets rwWhat4PushMuxOps
 1075             (thmResult, db') <- io (finishProof sc db concl pstate recordThm useSequentGoals what4PushMuxOps)
 1076             putTheoremDB db'
 1077             pure thmResult
