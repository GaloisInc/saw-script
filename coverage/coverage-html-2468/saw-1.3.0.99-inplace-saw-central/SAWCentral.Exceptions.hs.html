   20 newtype TypeErrors = TypeErrors [(Pos, String)]
   21 
   22 instance Show TypeErrors where
   23   show (TypeErrors []) = "Unspecified type error"
   24   show (TypeErrors [(pos, msg)]) = show pos ++ ": " ++ msg
   25   show (TypeErrors errs) = "Type errors:\n" ++ showErrs errs
   26     where showErrs = unlines . map showErr
   27           showErr (pos, msg) = "  " ++ show pos ++ ": " ++ msg
   28 
   29 instance Exception TypeErrors where
   30 
   31 failTypecheck :: [(Pos, String)] -> a
   32 failTypecheck = throw . TypeErrors
   33 
   34 data TopLevelException
   35   = TopLevelException Pos String
   36   | JavaException Pos String
   37   | CrucibleSetupException ProgramLoc String
   38   | OverrideMatcherException ProgramLoc String
   39   | LLVMMethodSpecException ProgramLoc String
   40   | forall e. Exception e => SomeTopLevelException e
   41 
   42 instance Show TopLevelException where
   43   show (TopLevelException _ msg) = msg
   44   show (JavaException _ msg) = msg
   45   show (CrucibleSetupException _ msg) = msg
   46   show (OverrideMatcherException _ msg) = msg
   47   show (LLVMMethodSpecException _ msg) = msg
   48   show (SomeTopLevelException e) = show e
   49 
   50 -- | To use a custom structured exception type that works with the
   51 -- saw-script REPL's exception handlers and stack tracing, define
   52 -- 'toException' as 'topLevelExceptionToException' in the custom
   53 -- exception type's 'Exception' class instance.
   54 topLevelExceptionToException :: Exception e => e -> SomeException
   55 topLevelExceptionToException = toException . SomeTopLevelException
   56 
   57 -- | To use a custom structured exception type that works with the
   58 -- saw-script REPL's exception handlers and stack tracing, define
   59 -- 'fromException' as 'topLevelExceptionFromException' in the custom
   60 -- exception type's 'Exception' class instance.
   61 topLevelExceptionFromException :: Exception e => SomeException -> Maybe e
   62 topLevelExceptionFromException x =
   63   do SomeTopLevelException a <- fromException x
   64      cast a
   65 
   66 instance Exception TopLevelException
   67 
   68 data TraceException = TraceException [String] SomeException
   69 
   70 instance Show TraceException where
   71   show (TraceException msg ex) =
   72     unlines (["Stack trace:"] ++ msg ++ [displayException ex])
   73 
   74 instance Exception TraceException
   75

   77 -- | Show this when we fail to setup Java properly.
   78 disableJavaWarning :: String -> SomeException -> String
   79 disableJavaWarning reason e =
   80   unlines $ [
   81     "WARNING: Disabling Java support.",
   82     "  Reason: " ++ reason,
   83     "  Output:"
   84   ] ++
   85   [ "    " ++ l
   86     | l <- lines (displayException e)
   87   ]
