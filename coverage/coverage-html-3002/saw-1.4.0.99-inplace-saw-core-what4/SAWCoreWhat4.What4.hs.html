  285 swBvWidth :: SWord sym -> Int
  286 swBvWidth x
  287   | w <= toInteger (maxBound :: Int) = fromInteger w
  288   | otherwise = croak
  289  where w = SW.bvWidth x
  290        croak = panic "swBvWidth" [
  291            "Bitvector too long: width " <> Text.pack (show w) <>
  292                ", value " <> Text.pack (show x)

  337 -- primitive intToNat : Integer -> Nat;
  338 -- intToNat x == max 0 x
  339 intToNatOp :: forall sym. Sym sym => sym -> SPrim sym
  340 intToNatOp sym =
  341   Prims.intFun $ \i ->
  342   Prims.Prim $
  343     case W.asInteger i of
  344       Just i'
  345         | 0 <= i'   -> pure (VNat (fromInteger i'))
  346         | otherwise -> pure (VNat 0)
  347       Nothing ->
  348         do z <- W.intLit sym 0
  349            pneg <- W.intLt sym i z
  350            i' <- W.intIte sym pneg z i
  351            pure (VIntToNat (VInt i'))

  387 -- | Shift left, shifting in copies of the given bit
  388 bvShl :: IsExprBuilder sym => sym -> Pred sym -> SWord sym -> SWord sym -> IO (SWord sym)
  389 bvShl sym z w i =
  390   W.iteM SW.bvIte sym z
  391     (do w' <- SW.bvNot sym w
  392         SW.bvNot sym =<< SW.bvShl sym w' i)
  393     (SW.bvShl sym w i)
  394 
  395 -- | Shift right, shifting in copies of the given bit
  396 bvShr :: IsExprBuilder sym => sym -> Pred sym -> SWord sym -> SWord sym -> IO (SWord sym)
  397 bvShr sym z w i =
  398   W.iteM SW.bvIte sym z
  399     (do w' <- SW.bvNot sym w
  400         SW.bvNot sym =<< SW.bvLshr sym w' i)
  401     (SW.bvLshr sym w i)
  402 
  403 liftShift :: IsExprBuilder sym =>
  404   sym ->
  405   (SWord sym -> SWord sym -> IO (SWord sym)) ->
  406   SWord sym -> Integer -> IO (SWord sym)
  407 liftShift sym f w i =
  408   f w =<< SW.bvLit sym (SW.bvWidth w) (i `min` SW.bvWidth w)
  409 
  410 liftRotate :: IsExprBuilder sym =>
  411   sym ->
  412   (SWord sym -> SWord sym -> IO (SWord sym)) ->
  413   SWord sym -> Integer -> IO (SWord sym)
  414 liftRotate sym f w i =
  415   f w =<< SW.bvLit sym (SW.bvWidth w) (i `mod` SW.bvWidth w)

  418 -- | op : (n : Nat) -> Vec n Bool -> Nat -> Vec n Bool
  419 bvShiftOp :: Sym sym => sym ->
  420              (SWord sym -> SWord sym -> IO (SWord sym)) ->
  421              (SWord sym -> Integer   -> IO (SWord sym)) -> SPrim sym
  422 bvShiftOp sym bvOp natOp =
  423   Prims.constFun  $                  -- additional argument? the size?
  424   wordFun sym $ \x ->                -- word to shift
  425   Prims.strictFun $ \y ->            -- amount to shift as a nat
  426   Prims.Prim $
  427     case y of
  428       VNat i   -> VWord <$> natOp x j
  429         where j = toInteger i `min` SW.bvWidth x
  430       VBVToNat _ v -> VWord <$> (bvOp x =<< toWord sym v)
  431       _        -> error $ unwords ["SAWCoreWhat4.What4.bvShiftOp", show y]

  448 bvForall :: W.IsSymExprBuilder sym =>
  449   sym -> Natural -> (SWord sym -> IO (Pred sym)) -> IO (Pred sym)
  450 bvForall sym n f =
  451   do let indexSymbol = W.safeSymbol "i"
  452      case mkNatRepr n of
  453         Some w
  454           | Just LeqProof <- testLeq (knownNat @1) w ->
  455             withKnownNat w $ do
  456               i <- W.freshBoundVar sym indexSymbol $ W.BaseBVRepr w
  457               body <- f . DBV $ W.varExpr sym i
  458               W.forallPred sym i body
  459           | otherwise -> f ZBV

  475 -- NB: What4's division operation provides SMT-LIB's Euclidean division, which
  476 -- doesn't match the round-to-neg-infinity semantics of Cryptol, so we have to
  477 -- do some work to get the desired semantics.
  478 intDiv ::
  479   IsExprBuilder sym =>
  480   sym ->
  481   SymExpr sym BaseIntegerType ->
  482   SymExpr sym BaseIntegerType ->
  483   IO (SymExpr sym BaseIntegerType)
  484 intDiv sym x y = do
  485   neg <- W.intLt sym y =<< W.intLit sym 0
  486   case W.asConstantPred neg of
  487     Just False -> W.intDiv sym x y
  488     Just True  ->
  489        do xneg <- W.intNeg sym x
  490           yneg <- W.intNeg sym y
  491           W.intDiv sym xneg yneg
  492     Nothing ->
  493        do xneg <- W.intNeg sym x
  494           yneg <- W.intNeg sym y
  495           zneg <- W.intDiv sym xneg yneg
  496           z    <- W.intDiv sym x y
  497           W.intIte sym neg zneg z

  591 streamGet :: forall sym. Sym sym => sym -> SValue sym -> SValue sym -> IO (SValue sym)
  592 streamGet sym xs ix =
  593     case ix of
  594       VNat n -> lookupSStream xs n
  595       VBVToNat _ w ->
  596         do ilv <- toWord sym w
  597            selectV sym (lazyMux @sym (muxBVal sym)) ((2 ^ SW.bvWidth ilv) - 1) (lookupSStream xs) ilv
  598       v -> panic "streamGetOp" ["Expected Nat value, found: " <> Text.pack (show v)]

  600 lookupSStream :: SValue sym -> Natural -> IO (SValue sym)
  601 lookupSStream (VExtra (SStream f r)) n = do
  602    m <- readIORef r
  603    case Map.lookup n m of
  604      Just v  -> return v
  605      Nothing -> do v <- f n
  606                    writeIORef r (Map.insert n v m)
  607                    return v

  637 lazyMux :: (IsExpr (SymExpr sym)) =>
  638   (SBool sym  -> a -> a -> IO a) -> (SBool sym -> IO a -> IO a -> IO a)
  639 lazyMux muxFn c tm fm =

  648 -- @selectV sym merger maxValue valueFn vx@ treats @vx@ as an index, represented
  649 -- as a big-endian list of bits. It does a binary lookup, using @merger@ as an
  650 -- if-then-else operator. If the index is greater than @maxValue@, then it
  651 -- returns @valueFn maxValue@.
  652 selectV :: forall sym b.
  653   Sym sym =>
  654   sym ->
  655   (SBool sym -> IO b -> IO b -> IO b) -> Natural -> (Natural -> IO b) -> SWord sym -> IO b
  656 selectV sym merger maxValue valueFn vx =
  657   case SW.bvAsUnsignedInteger vx of
  658     Just i  -> valueFn (fromInteger i :: Natural)
  659     Nothing -> impl (swBvWidth vx) 0
  660   where
  661     impl :: Int -> Natural -> IO b
  662     impl _ x | x > maxValue || x < 0 = valueFn maxValue
  663     impl 0 y = valueFn y
  664     impl i y = do
  665       -- NB: `i` counts down in each iteration, so we use bvAtLE (a
  666       -- little-endian indexing function) to ensure that the bits are processed
  667       -- in big-endian order. Alternatively, we could have `i` count up and use
  668       -- bvAtBE (a big-endian indexing function), but we use bvAtLE as it is
  669       -- slightly cheaper to compute.
  670       p <- SW.bvAtLE sym vx (toInteger j)
  671       merger p (impl j (y `setBit` j)) (impl j y) where j = i - 1

  881 w4SolveBasic ::
  882   forall sym. IsSymExprBuilder sym =>
  883   sym ->
  884   SharedContext ->
  885   Map Ident (SPrim sym) {- ^ additional primitives -} ->
  886   Map VarIndex (SValue sym) {- ^ bindings for free variables -} ->
  887   IORef (SymFnCache sym) {- ^ cache for uninterpreted function symbols -} ->
  888   Set VarIndex {- ^ 'unints' Constants in this list are kept uninterpreted -} ->
  889   Term {- ^ term to simulate -} ->
  890   IO (SValue sym)
  891 w4SolveBasic sym sc addlPrims varMap ref unintSet t =
  892   do m <- scGetModuleMap sc
  893      let variable (VarName ix x) ty
  894             | Just v <- Map.lookup ix varMap = return v
  895             | otherwise = parseUninterpreted sym ref (mkUnintApp (Text.unpack x ++ "_" ++ show ix)) ty
  896      let uninterpreted nm ty
  897            | Set.member (nameIndex nm) unintSet =
  898              let vn = VarName (nameIndex nm) (toShortName (nameInfo nm))
  899              in Just (variable vn ty)
  900            | otherwise                          = Nothing
  901      let primHandler = Sim.defaultPrimHandler
  902      let mux = Prims.lazyMuxValue (prims sym)
  903      cfg <-
  904        Sim.evalGlobal m (constMap sym `Map.union` addlPrims)
  905        variable uninterpreted (recursor sym) primHandler mux
  906      Sim.evalSharedTerm cfg t

  916 type SymFnCache sym = Map W.SolverSymbol (MapF (Assignment BaseTypeRepr) (SymFnWrapper sym))

  918 lookupSymFn ::
  919   W.SolverSymbol -> Assignment BaseTypeRepr args -> BaseTypeRepr ty ->
  920   SymFnCache sym -> Maybe (W.SymFn sym args ty)
  921 lookupSymFn s args ty cache =
  922   do m <- Map.lookup s cache
  923      SymFnWrapper fn <- MapF.lookup (Ctx.extend args ty) m
  924      return fn

  926 insertSymFn ::
  927   W.SolverSymbol -> Assignment BaseTypeRepr args -> BaseTypeRepr ty ->
  928   W.SymFn sym args ty -> SymFnCache sym -> SymFnCache sym
  929 insertSymFn s args ty fn = Map.alter upd s
  930   where
  931     upd Nothing = Just (MapF.singleton (Ctx.extend args ty) (SymFnWrapper fn))
  932     upd (Just m) = Just (MapF.insert (Ctx.extend args ty) (SymFnWrapper fn) m)

  934 mkSymFn ::
  935   forall sym args ret. (IsSymExprBuilder sym) =>
  936   sym -> IORef (SymFnCache sym) ->
  937   String -> Assignment BaseTypeRepr args -> BaseTypeRepr ret ->
  938   IO (W.SymFn sym args ret)
  939 mkSymFn sym ref nm args ret =
  940   do let s = W.safeSymbol nm
  941      cache <- readIORef ref
  942      case lookupSymFn s args ret cache of
  943        Just fn -> return fn
  944        Nothing ->
  945          do fn <- W.freshTotalUninterpFn sym s args ret
  946             writeIORef ref (insertSymFn s args ret fn cache)
  947             return fn

  958 base type that is mentioned in the result of the original uninterpreted
  959 function.  If we need more than one value of a type, then we return an
  960 *array* of values, and the actual results of the original function are

  964 A call `f x` will be translated like this:

  966   f_bv8:  [16] -> Array [2] [8]   // Uninterpreted
  967   f_bool: [16] -> Bool            // Uninterpreted
  968   w8s   = f_bv8 x                 // This has some auto-generated name
  969   bools = f_bool x                // This has some auto-generated name
  970   
  971 Value for `f x`:
  972   ([ w8s ! 0, w8s ! 1, w8s ! 2, w8s ! 3 ], bools)

  987 -- | Generate a call to an uninterpreted function.
  988 parseUninterpreted ::
  989   forall sym.
  990   (IsSymExprBuilder sym) =>
  991   sym                         {- ^ How to make symbolic expressions -} ->
  992   IORef (SymFnCache sym)      {- ^ Cache of known uninterpreted functions -} ->
  993   UnintApp (SymExpr sym)      {- ^ Name of function and its arguments -} ->
  994   TValue (What4 sym)          {- ^ Type of the function's result -} ->
  995   IO (SValue sym)             {- ^ The symbolic value for the call to the uninterpreted function -}
  996 parseUninterpreted sym ref app@(UnintApp nm args tys) ty =
  997   do
  998     un <- MapF.traverseWithKey mkUnint (countUninterpreted 1 MapF.empty ty)
  999     evalStateT (parseUninterpreted' sym ref app ty) un
 1000   where
 1001   mkUnint :: BaseTypeRepr tc -> UnintCount tc -> IO (Arr sym tc)
 1002   mkUnint ret (UnintCount n)
 1003     | let lim = n - 1
 1004     , Just (Some w) <- someNat (width lim) =
 1005       case isZeroOrGT1 w of
 1006         Left Refl ->
 1007           do
 1008             fn <- mkSymFn sym ref (nm ++ suff ret) tys ret
 1009             el <- W.applySymFn sym fn args
 1010             pure (Arr [el])
 1011         Right LeqProof ->
 1012           do
 1013             fn   <- mkSymFn sym ref (nm ++ suff ret) tys (BaseArrayRepr (Ctx.singleton (BaseBVRepr w)) ret)
 1014             arr  <- W.applySymFn sym fn args
 1015             els <- forM [ 0 .. lim ] $ \i ->
 1016               do
 1017                 ind <- W.bvLit sym w (BV.mkBV w (fromIntegral i))
 1018                 W.arrayLookup sym arr (Ctx.singleton ind)
 1019             pure (Arr els)
 1020     | otherwise = panic "parseUninterpreted" ["`someNat` returned `Nothing` for `Natural`"]

 1022   -- | Compute the number of bits required to represent the given integer.
 1023   width :: Natural -> Natural
 1024   width x = go 0 x
 1025     where
 1026       go s 0 = s
 1027       go s n = let s' = s + 1 in s' `seq` go s' (n `shiftR` 1)

 1029   -- Type suffixes for uninterpreted functions of each base type.
 1030   suff :: BaseTypeRepr a -> String
 1031   suff t =
 1032     case t of
 1033       BaseBoolRepr -> "_bool"
 1034       BaseIntegerRepr -> "_int"
 1035       BaseRealRepr -> "_real"
 1036       BaseComplexRepr -> "_complex"
 1037       BaseBVRepr w -> "_bv" ++ show w
 1038       BaseFloatRepr (FloatingPointPrecisionRepr x y) -> "_float_" ++ show x ++ "_" ++ show y
 1039       BaseStringRepr w -> "_str" ++ strw
 1040         where strw = case w of
 1041                        Char8Repr -> "8"
 1042                        Char16Repr -> "16"
 1043                        UnicodeRepr -> "U"
 1044       BaseArrayRepr i e -> "_fun" ++ suff_asgn i ++ "_to" ++ suff e
 1045       BaseStructRepr flds -> "_struct" ++ suff_asgn flds ++ "_end"

 1047   suff_asgn i = intercalate "_and" (V.toList (Ctx.toVector i suff))

 1051 -- | Count how many uninterpreted symbols we need to represent a value
 1052 -- of the given type.  Note that this function should match exactly what
 1053 -- is needed by `parseUninterpreted'`.
 1054 countUninterpreted ::
 1055   IsSymExprBuilder sym =>
 1056   Natural -> MapF BaseTypeRepr UnintCount -> TValue (What4 sym) -> MapF BaseTypeRepr UnintCount
 1057 countUninterpreted scale count ty =
 1058   case ty of
 1059     VPiType {}      -> count 
 1060     VBoolType       -> add BaseBoolRepr
 1061     VIntType        -> add BaseIntegerRepr
 1062     VIntModType {}  -> add BaseIntegerRepr
 1063     VVecType n VBoolType ->
 1064       case somePosNat n of
 1065         Just (Some (PosNat w)) -> add (BaseBVRepr w)
 1066         _                      -> count
 1067 
 1068     VVecType n et -> if n == 0 then count else countUninterpreted (n * scale) count et
 1069       
 1070     VArrayType ity ety
 1071       | Just (Some idx_repr) <- valueAsBaseType ity
 1072       , Just (Some elm_repr) <- valueAsBaseType ety
 1073       -> add (BaseArrayRepr (Ctx.Empty Ctx.:> idx_repr) elm_repr)
 1074 
 1075     VUnitType -> count
 1076 
 1077     VPairType ty1 ty2 -> countUninterpreted scale (countUninterpreted scale count ty1) ty2
 1078 
 1079     VRecordType elem_tps ->
 1080       foldl (\ct (_,t) -> countUninterpreted scale ct t) count elem_tps
 1081 
 1082     _ -> count
 1083   where
 1084   add :: BaseTypeRepr tc -> MapF BaseTypeRepr UnintCount
 1085   add bt =
 1086     MapF.insertWith
 1087       (\(UnintCount x) (UnintCount y) -> UnintCount (x + y))
 1088       bt (UnintCount scale) count

 1180 -- | Flatten an 'SValue' to a sequence of components, each of which is
 1181 -- a symbolic value of a base type (e.g. word or boolean), and add
 1182 -- them to the list of arguments of the given 'UnintApp'. If the
 1183 -- 'SValue' contains any values built from data constructors, then
 1184 -- encode them as suffixes on the function name of the 'UnintApp'.
 1185 applyUnintApp ::
 1186   forall sym.
 1187   (W.IsExprBuilder sym) =>
 1188   sym ->
 1189   UnintApp (SymExpr sym) ->
 1190   SValue sym ->
 1191   IO (UnintApp (SymExpr sym))
 1192 applyUnintApp sym app0 v =
 1193   case v of
 1194     VUnit                     -> return app0
 1195     VPair x y                 -> do app1 <- applyUnintApp sym app0 =<< force x
 1196                                     app2 <- applyUnintApp sym app1 =<< force y
 1197                                     return app2
 1198     VRecordValue elems        -> foldM (applyUnintApp sym) app0 =<< traverse (force . snd) elems
 1199     VVector xv                -> foldM (applyUnintApp sym) app0 =<< traverse force xv

 1206     VWord (DBV sw)            -> return (extendUnintApp app0 sw (W.exprType sw))
 1207     VArray (SArray sa)        -> return (extendUnintApp app0 sa (W.exprType sa))
 1208     VWord ZBV                 -> return app0

 1210                                    where app' = suffixUnintApp ("_" ++ (Text.unpack (toShortName (nameInfo i)))) app0

 1213                                    where app' = suffixUnintApp ("_" ++ show w) app0
 1214     TValue (suffixTValue -> Just s)
 1215                               -> return (suffixUnintApp s app0)
 1216     VFun {} ->
 1217       fail $
 1218       "Cannot create uninterpreted higher-order function " ++
 1219       show (stringOfUnintApp app0)

 1277 -- | Given a list of variable names with first-order types, descend
 1278 -- into the structure of those types (as needed) and construct
 1279 -- corresponding What4 bound variables so we can bind them using a
 1280 -- forall quantifier.
 1281 -- At the same time construct @SValue@s containing those variables
 1282 -- suitable for passing to the term evaluator as substitutions for the
 1283 -- given variables.
 1284 -- The length of the @SValue@ list returned will match the list of the
 1285 -- input variable list, but the list of What4 @BoundVar@s might not.
 1286 --
 1287 -- This procedure is capable of handling most first-order types,
 1288 -- except that Array types must have base types as index and result
 1289 -- types rather than more general first-order types. (TODO? should we
 1290 -- actually restrict the @FirstOrderType@ in the same way?)
 1291 boundFOTs :: forall sym.
 1292   IsSymExprBuilder sym =>
 1293   sym ->
 1294   [(VarName, FirstOrderType)] ->
 1295   IO ([SValue sym], [Some (BoundVar sym)])
 1296 boundFOTs sym vars =
 1297   do (svals,(bndvars,_)) <- runStateT (mapM (uncurry handleVar) vars) ([], 0)
 1298      return (svals, bndvars)
 1299 
 1300  where
 1301    freshBnd :: VarName -> W.BaseTypeRepr tp -> StateT ([Some (BoundVar sym)],Integer) IO (SymExpr sym tp)
 1302    freshBnd vn tpr =
 1303      do (vs,n) <- get
 1304         let nm = Text.unpack (vnName vn) ++ "." ++ show n
 1305         bvar <- lift (W.freshBoundVar sym (W.safeSymbol nm) tpr)
 1306         put (Some bvar : vs, n+1)
 1307         return (W.varExpr sym bvar)
 1308 
 1309    handleVar :: VarName -> FirstOrderType -> StateT ([Some (BoundVar sym)], Integer) IO (SValue sym)
 1310    handleVar x fot =
 1311      case fot of
 1312        FOTBit -> VBool <$> freshBnd x BaseBoolRepr
 1313        FOTInt -> VInt  <$> freshBnd x BaseIntegerRepr
 1314        FOTIntMod m -> VIntMod m <$> freshBnd x BaseIntegerRepr
 1315 
 1316        FOTVec n FOTBit ->
 1317          case somePosNat n of
 1318            Nothing -> -- n == 0
 1319              return (VWord ZBV)
 1320            Just (Some (PosNat nr)) ->
 1321              VWord . DBV <$> freshBnd x (BaseBVRepr nr)
 1322 
 1323        FOTVec n tp -> -- NB, not Bit
 1324          do vs  <- V.replicateM (fromIntegral n) (handleVar x tp)
 1325             vs' <- traverse (return . ready) vs
 1326             return (VVector vs')
 1327 
 1328        FOTRec tm ->
 1329          do vs  <- traverse (handleVar x) tm
 1330             vs' <- traverse (return . ready) vs
 1331             return (vRecord vs')
 1332 
 1333        FOTTuple ts ->
 1334          do vs  <- traverse (handleVar x) ts
 1335             vs' <- traverse (return . ready) vs
 1336             return (vTuple vs')
 1337 
 1338        FOTArray idx res
 1339          | Just (Some idx_repr) <- fotToBaseType idx
 1340          , Just (Some res_repr) <- fotToBaseType res
 1341 
 1342          -> VArray . SArray <$> freshBnd x (BaseArrayRepr (Ctx.Empty Ctx.:> idx_repr) res_repr)
 1343 
 1344          | otherwise -> fail ("boundFOTs: cannot handle " ++ show fot)

 1674   IO ([String], ([Maybe (Labeler (B.ExprBuilder n st fs))], SBool (B.ExprBuilder n st fs)))
 1675 w4Eval sym st sc ps uintSet t =
 1676   do (argNames, _, bvs, bval) <- w4EvalAny sym st sc ps uintSet t
 1677      case bval of
 1678        VBool b -> return (argNames, (bvs, b))
 1679        _ -> fail $ "w4Eval: non-boolean result type. " ++ show bval

 1681 -- | Simplify a saw-core term by evaluating it through the saw backend
 1682 -- of what4.
 1683 w4EvalBasic ::
 1684   forall n st fs.
 1685   B.ExprBuilder n st fs ->
 1686   SAWCoreState n ->
 1687   SharedContext ->
 1688   ModuleMap ->
 1689   Map Ident (SPrim (B.ExprBuilder n st fs)) {- ^ additional primitives -} ->
 1690   IntMap (SValue (B.ExprBuilder n st fs)) {- ^ bindings for free variables -} ->
 1691   IORef (SymFnCache (B.ExprBuilder n st fs)) {- ^ cache for uninterpreted function symbols -} ->
 1692   Set VarIndex {- ^ 'unints' Constants in this list are kept uninterpreted -} ->
 1693   Term {- ^ term to simulate -} ->
 1694   IO (SValue (B.ExprBuilder n st fs))
 1695 w4EvalBasic sym st sc m addlPrims varCons ref unintSet t =
 1696   do let variable tf (VarName ix nm) ty
 1697            | Just v <- IntMap.lookup ix varCons = pure v
 1698            | otherwise =
 1699            do trm <- ArgTermConst <$> scTermF sc tf
 1700               parseUninterpretedSAW sym st sc ref trm
 1701                  (mkUnintApp (Text.unpack nm ++ "_" ++ show ix)) ty
 1702      let variable' tp vn ty = variable (Variable vn tp) vn ty
 1703      let uninterpreted nm ty
 1704            | Set.member (nameIndex nm) unintSet =
 1705              let vn = VarName (nameIndex nm) (toShortName (nameInfo nm))
 1706              in Just (variable (Constant nm) vn ty)
 1707            | otherwise                          = Nothing
 1708      let primHandler = Sim.defaultPrimHandler
 1709      let mux = Prims.lazyMuxValue (prims sym)
 1710      cfg <-
 1711        Sim.evalGlobal' m (constMap sym `Map.union` addlPrims)
 1712        variable' uninterpreted (recursor sym) primHandler mux
 1713      Sim.evalSharedTerm cfg t

 1715 -- | Evaluate a saw-core term to a What4 value for the purposes of
 1716 --   using it as an input for symbolic simulation.  This will evaluate
 1717 --   primitives, but will cancel evaluation and return the associated
 1718 --   'NameInfo' if it encounters a constant value with a 'Name'
 1719 --   that is not accepted by the filter.
 1720 w4SimulatorEval ::
 1721   forall n st fs.
 1722   B.ExprBuilder n st fs ->
 1723   SAWCoreState n ->
 1724   SharedContext ->
 1725   ModuleMap ->
 1726   Map Ident (SPrim (B.ExprBuilder n st fs)) {- ^ additional primitives -} ->
 1727   IORef (SymFnCache (B.ExprBuilder n st fs)) {- ^ cache for uninterpreted function symbols -} ->
 1728   (Name -> TValue (What4 (B.ExprBuilder n st fs)) -> Bool)
 1729     {- ^ Filter for constant values.  True means unfold, false means halt evaluation. -} ->
 1730   Term {- ^ term to simulate -} ->
 1731   IO (Either NameInfo (SValue (B.ExprBuilder n st fs)))
 1732 w4SimulatorEval sym st sc m addlPrims ref constantFilter t =
 1733   do let variable tf (VarName ix nm) ty =
 1734            do trm <- ArgTermConst <$> scTermF sc tf
 1735               parseUninterpretedSAW sym st sc ref trm (mkUnintApp (Text.unpack nm ++ "_" ++ show ix)) ty
 1736      let uninterpreted nm ty =
 1737           if constantFilter nm ty then Nothing else Just (X.throwIO (NeutralTermEx (nameInfo nm)))
 1738      let variable' tp vn ty = variable (Variable vn tp) vn ty
 1739      let primHandler = Sim.defaultPrimHandler
 1740      let mux = Prims.lazyMuxValue (prims sym)
 1741      res <- X.try $ do
 1742               cfg <-
 1743                 Sim.evalGlobal' m (constMap sym `Map.union` addlPrims)
 1744                 variable' uninterpreted (recursor sym) primHandler mux
 1745               Sim.evalSharedTerm cfg t
 1746      case res of
 1747        Left (NeutralTermEx nmi) -> pure (Left nmi)
 1748        Right x -> pure (Right x)

 1751 instance X.Exception NeutralTermException

 1753 -- | Given a constant nm of (saw-core) type ty, construct an
 1754 -- uninterpreted constant with that type. The 'Term' argument should
 1755 -- be an open term, which should have the designated return type when
 1756 -- the local variables have the corresponding types from the
 1757 -- 'Assignment'.
 1758 parseUninterpretedSAW ::
 1759   forall n st fs.
 1760   B.ExprBuilder n st fs ->
 1761   SAWCoreState n ->
 1762   SharedContext ->
 1763   IORef (SymFnCache (B.ExprBuilder n st fs)) ->
 1764   ArgTerm {- ^ representation of function applied to arguments -} ->
 1765   UnintApp (SymExpr (B.ExprBuilder n st fs)) ->
 1766   TValue (What4 (B.ExprBuilder n st fs)) {- ^ return type -} ->
 1767   IO (SValue (B.ExprBuilder n st fs))
 1768 parseUninterpretedSAW sym st sc ref trm app ty =
 1769   case ty of
 1770     VPiType t1 body
 1771       -> pure $ VFun $ \x ->
 1772            do x' <- force x
 1773               app' <- applyUnintApp sym app x'
 1774               arg <- mkArgTerm sc t1 x'
 1775               let trm' = ArgTermApply trm arg
 1776               t2 <- applyPiBody body (ready x')
 1777               parseUninterpretedSAW sym st sc ref trm' app' t2
 1778 
 1779     VBoolType
 1780       -> VBool <$> mkUninterpretedSAW sym st sc ref trm app BaseBoolRepr
 1781 
 1782     VIntType
 1783       -> VInt  <$> mkUninterpretedSAW sym st sc ref trm app BaseIntegerRepr
 1784 
 1785     VIntModType n
 1786       -> VIntMod n <$> mkUninterpretedSAW sym st sc ref (ArgTermFromIntMod n trm) app BaseIntegerRepr
 1787 
 1788     -- 0 width bitvector is a constant
 1789     VVecType 0 VBoolType
 1790       -> return $ VWord ZBV
 1791 
 1792     VVecType n VBoolType
 1793       | Just (Some (PosNat w)) <- somePosNat n
 1794       -> (VWord . DBV) <$> mkUninterpretedSAW sym st sc ref trm app (BaseBVRepr w)
 1795 
 1796     VVecType n ety | n >= 0
 1797       ->  do ety' <- termOfTValue sc ety
 1798              let mkElem i =
 1799                    do let trm' = ArgTermAt n ety' trm i
 1800                       let app' = suffixUnintApp ("_a" ++ show i) app
 1801                       parseUninterpretedSAW sym st sc ref trm' app' ety
 1802              xs <- traverse mkElem (genericTake n [0 ..])
 1803              return (VVector (V.fromList (map ready xs)))
 1804 
 1805     VArrayType ity ety
 1806       | Just (Some idx_repr) <- valueAsBaseType ity
 1807       , Just (Some elm_repr) <- valueAsBaseType ety
 1808       -> (VArray . SArray) <$>
 1809           mkUninterpretedSAW sym st sc ref trm app (BaseArrayRepr (Ctx.Empty Ctx.:> idx_repr) elm_repr)
 1810 
 1811     VUnitType
 1812       -> return VUnit
 1813 
 1814     VPairType ty1 ty2
 1815       -> do let trm1 = ArgTermPairLeft trm
 1816             let trm2 = ArgTermPairRight trm
 1817             x1 <- parseUninterpretedSAW sym st sc ref trm1 (suffixUnintApp "_L" app) ty1
 1818             x2 <- parseUninterpretedSAW sym st sc ref trm2 (suffixUnintApp "_R" app) ty2
 1819             return (VPair (ready x1) (ready x2))
 1820 
 1821     _ -> fail $ "could not create uninterpreted symbol of type " ++ show ty
 1822 
 1823 mkUninterpretedSAW ::
 1824   forall n st fs t.
 1825   B.ExprBuilder n st fs ->
 1826   SAWCoreState n ->
 1827   SharedContext ->
 1828   IORef (SymFnCache (B.ExprBuilder n st fs)) ->
 1829   ArgTerm ->
 1830   UnintApp (SymExpr (B.ExprBuilder n st fs)) ->
 1831   BaseTypeRepr t ->
 1832   IO (SymExpr (B.ExprBuilder n st fs) t)
 1833 mkUninterpretedSAW sym st sc ref trm (UnintApp nm args tys) ret
 1834   | Just Refl <- testEquality Ctx.empty tys =
 1835     bindSAWTerm sym st ret =<< reconstructArgTerm trm sc []
 1836   | otherwise =
 1837   do fn <- mkSymFn sym ref nm tys ret
 1838      sawRegisterSymFunInterp st fn (reconstructArgTerm trm)
 1839      W.applySymFn sym fn args

 1841 -- | An 'ArgTerm' is a description of how to reassemble a saw-core
 1842 -- term from a list of the atomic components it is composed of.
 1843 data ArgTerm
 1844   = ArgTermVar
 1845   | ArgTermBVZero -- ^ scBvNat 0 0
 1846   | ArgTermToIntMod Natural ArgTerm -- ^ toIntMod n x
 1847   | ArgTermFromIntMod Natural ArgTerm -- ^ fromIntMod n x
 1848   | ArgTermVector Term [ArgTerm] -- ^ element type, elements
 1849   | ArgTermUnit
 1850   | ArgTermPair ArgTerm ArgTerm
 1851   | ArgTermRecord [(FieldName, ArgTerm)]
 1852   | ArgTermConst Term
 1853   | ArgTermApply ArgTerm ArgTerm
 1854   | ArgTermAt Natural Term ArgTerm Natural
 1855     -- ^ length, element type, list, index
 1856   | ArgTermPairLeft ArgTerm
 1857   | ArgTermPairRight ArgTerm
 1858   | ArgTermBVToNat Natural ArgTerm
 1859 
 1860 -- | Reassemble a saw-core term from an 'ArgTerm' and a list of parts.
 1861 -- The length of the list should be equal to the number of
 1862 -- 'ArgTermVar' constructors in the 'ArgTerm'.
 1863 reconstructArgTerm :: ArgTerm -> SharedContext -> [Term] -> IO Term
 1864 reconstructArgTerm atrm sc ts =
 1865   do (t, unused) <- parse atrm ts
 1866      unless (null unused) $ fail "reconstructArgTerm: too many function arguments"
 1867      return t
 1868   where
 1869     parse :: ArgTerm -> [Term] -> IO (Term, [Term])
 1870     parse at ts0 =
 1871       case at of
 1872         ArgTermVar ->
 1873           case ts0 of
 1874             x : ts1 -> return (x, ts1)
 1875             [] -> fail "reconstructArgTerm: too few function arguments"
 1876         ArgTermBVZero ->
 1877           do z <- scNat sc 0
 1878              x <- scBvNat sc z z
 1879              return (x, ts0)
 1880         ArgTermToIntMod n at1 ->
 1881           do n' <- scNat sc n
 1882              (x1, ts1) <- parse at1 ts0
 1883              x <- scToIntMod sc n' x1
 1884              pure (x, ts1)
 1885         ArgTermFromIntMod n at1 ->
 1886           do n' <- scNat sc n
 1887              (x1, ts1) <- parse at1 ts0
 1888              x <- scFromIntMod sc n' x1
 1889              pure (x, ts1)
 1890         ArgTermVector ty ats ->
 1891           do (xs, ts1) <- parseList ats ts0
 1892              x <- scVectorReduced sc ty xs
 1893              return (x, ts1)
 1894         ArgTermUnit ->
 1895           do x <- scUnitValue sc
 1896              return (x, ts0)
 1897         ArgTermPair at1 at2 ->
 1898           do (x1, ts1) <- parse at1 ts0
 1899              (x2, ts2) <- parse at2 ts1
 1900              x <- scPairValue sc x1 x2
 1901              return (x, ts2)
 1902         ArgTermRecord flds ->
 1903           do let (tags, ats) = unzip flds
 1904              (xs, ts1) <- parseList ats ts0
 1905              x <- scRecord sc (Map.fromList (zip tags xs))
 1906              return (x, ts1)
 1907         ArgTermConst x ->
 1908           do return (x, ts0)
 1909         ArgTermApply at1 at2 ->
 1910           do (x1, ts1) <- parse at1 ts0
 1911              (x2, ts2) <- parse at2 ts1
 1912              x <- scApply sc x1 x2
 1913              return (x, ts2)
 1914         ArgTermAt n ty at1 i ->
 1915           do n' <- scNat sc n
 1916              (x1, ts1) <- parse at1 ts0
 1917              i' <- scNat sc i
 1918              x <- scAt sc n' ty x1 i'
 1919              return (x, ts1)
 1920         ArgTermPairLeft at1 ->
 1921           do (x1, ts1) <- parse at1 ts0
 1922              x <- scPairLeft sc x1
 1923              return (x, ts1)
 1924         ArgTermPairRight at1 ->
 1925           do (x1, ts1) <- parse at1 ts0
 1926              x <- scPairRight sc x1
 1927              return (x, ts1)
 1928         ArgTermBVToNat w at1 ->
 1929           do (x1, ts1) <- parse at1 ts0
 1930              x <- scBvToNat sc w x1
 1931              pure (x, ts1)
 1932 
 1933     parseList :: [ArgTerm] -> [Term] -> IO ([Term], [Term])
 1934     parseList [] ts0 = return ([], ts0)
 1935     parseList (at : ats) ts0 =
 1936       do (x, ts1) <- parse at ts0
 1937          (xs, ts2) <- parseList ats ts1
 1938          return (x : xs, ts2)

 1940 -- | Given a type and value encoded as 'SValue's, construct an
 1941 -- 'ArgTerm' that builds a term of that type from local variables with
 1942 -- base types. The number of 'ArgTermVar' constructors should match
 1943 -- the number of arguments appended by 'applyUnintApp'.
 1944 mkArgTerm :: SharedContext -> TValue (What4 sym) -> SValue sym -> IO ArgTerm
 1945 mkArgTerm sc ty val =
 1946   case (ty, val) of
 1947     (VBoolType, VBool _) -> return ArgTermVar
 1948     (VIntType, VInt _)   -> return ArgTermVar
 1949     (_, VWord ZBV)       -> return ArgTermBVZero     -- 0-width bitvector is a constant
 1950     (_, VWord (DBV _))   -> return ArgTermVar
 1951     (_, VArray{})        -> return ArgTermVar
 1952     (VUnitType, VUnit)   -> return ArgTermUnit
 1953     (VIntModType n, VIntMod _ _) -> pure (ArgTermToIntMod n ArgTermVar)
 1954 
 1955     (VVecType _ ety, VVector vv) ->
 1956       do vs <- traverse force (V.toList vv)
 1957          xs <- traverse (mkArgTerm sc ety) vs
 1958          ety' <- termOfTValue sc ety
 1959          return (ArgTermVector ety' xs)
 1960 
 1961     (VPairType ty1 ty2, VPair v1 v2) ->
 1962       do x1 <- mkArgTerm sc ty1 =<< force v1
 1963          x2 <- mkArgTerm sc ty2 =<< force v2
 1964          return (ArgTermPair x1 x2)
 1965 
 1966     (VRecordType tys, VRecordValue flds) | map fst tys == map fst flds ->
 1967       do let tags = map fst tys
 1968          vs <- traverse (force . snd) flds
 1969          xs <- sequence [ mkArgTerm sc t v | (t, v) <- zip (map snd tys) vs ]
 1970          return (ArgTermRecord (zip tags xs))
 1971 
 1972     (_, VCtorApp i _ ps vv) ->
 1973       do mm <- scGetModuleMap sc
 1974          ctor <-
 1975            case lookupVarIndexInMap (nameIndex i) mm of
 1976              Just (ResolvedCtor ctor) -> pure ctor
 1977              _ -> panic "mkArgTerm" ["Constructor not found: " <> toAbsoluteName (nameInfo i)]
 1978          ps' <- traverse (termOfSValue sc <=< force) ps
 1979          vv' <- traverse (termOfSValue sc <=< force) vv
 1980          x   <- scConstApply sc (ctorName ctor) (ps' ++ vv')
 1981          return (ArgTermConst x)
 1982 
 1983     (_, TValue tval) ->
 1984       do x <- termOfTValue sc tval
 1985          pure (ArgTermConst x)
 1986 
 1987     (_, VNat n) ->
 1988       do x <- scNat sc n
 1989          pure (ArgTermConst x)
 1990 
 1991     (_, VBVToNat w v) ->
 1992       do let w' = fromIntegral w -- FIXME: make w :: Natural to avoid fromIntegral
 1993          x <- mkArgTerm sc (VVecType w' VBoolType) v
 1994          pure (ArgTermBVToNat w' x)
 1995 
 1996     _ -> fail $ "could not create uninterpreted function argument of type " ++ show ty
 1997 
 1998 termOfTValue :: SharedContext -> TValue (What4 sym) -> IO Term
 1999 termOfTValue sc val =
 2000   case val of
 2001     VBoolType -> scBoolType sc
 2002     VIntType -> scIntegerType sc
 2003     VUnitType -> scUnitType sc
 2004     VVecType n a ->
 2005       do n' <- scNat sc n
 2006          a' <- termOfTValue sc a
 2007          scVecType sc n' a'
 2008     VPairType a b
 2009       -> do a' <- termOfTValue sc a
 2010             b' <- termOfTValue sc b
 2011             scPairType sc a' b'
 2012     VRecordType flds
 2013       -> do flds' <- traverse (traverse (termOfTValue sc)) flds
 2014             scRecordType sc flds'
 2015     _ -> fail $ "termOfTValue: " ++ show val
 2016 
 2017 termOfSValue :: SharedContext -> SValue sym -> IO Term
 2018 termOfSValue sc val =

 2020     VUnit -> scUnitValue sc
 2021     VNat n
 2022       -> scNat sc n
 2023     TValue tv -> termOfTValue sc tv
 2024     _ -> fail $ "termOfSValue: " ++ show val
