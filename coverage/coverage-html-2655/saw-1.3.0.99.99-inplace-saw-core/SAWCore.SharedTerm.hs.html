  353 -- | A TermFMap is a data structure used for hash-consing of terms.
  354 data TermFMap a
  355   = TermFMap
  356   { appMapTFM :: !(IntMap (IntMap a))
  357   , hashMapTFM :: !(HashMap (TermF Term) a)
  358   }
  359 
  360 emptyTFM :: TermFMap a
  361 emptyTFM = TermFMap mempty mempty
  362 
  363 lookupTFM :: TermF Term -> TermFMap a -> Maybe a
  364 lookupTFM tf tfm =
  365   case tf of
  366     App (STApp{ stAppIndex = i }) (STApp{ stAppIndex = j}) ->
  367       IntMap.lookup i (appMapTFM tfm) >>= IntMap.lookup j
  368     _ -> HMap.lookup tf (hashMapTFM tfm)
  369 
  370 insertTFM :: TermF Term -> a -> TermFMap a -> TermFMap a
  371 insertTFM tf x tfm =
  372   case tf of
  373     App (STApp{ stAppIndex = i }) (STApp{ stAppIndex = j}) ->
  374       let f Nothing = Just (IntMap.singleton j x)
  375           f (Just m) = Just (IntMap.insert j x m)
  376       in tfm { appMapTFM = IntMap.alter f i (appMapTFM tfm) }
  377     _ -> tfm { hashMapTFM = HMap.insert tf x (hashMapTFM tfm) }

  380 -- SharedContext: a high-level interface for building Terms.
  381 
  382 -- | 'SharedContext' is an abstract datatype representing all the
  383 -- information necessary to resolve names and to construct,
  384 -- type-check, normalize, and evaluate SAWCore 'Term's.
  385 -- A 'SharedContext' contains mutable references so that it can be
  386 -- extended at run-time with new names and declarations.
  387 
  388 -- Invariant: scGlobalEnv is a cache with one entry for every global
  389 -- declaration in 'scModuleMap' whose name is a 'ModuleIdentifier'.
  390 -- Each map entry points to a 'Constant' term with the same 'Ident'.
  391 -- It exists only to save one map lookup when building terms: Without
  392 -- it we would first have to look up the Ident by URI in scURIEnv, and
  393 -- then do another lookup for hash-consing the Constant term.
  394 data SharedContext = SharedContext
  395   { scModuleMap      :: IORef ModuleMap
  396   , scTermF          :: TermF Term -> IO Term
  397   , scDisplayNameEnv :: IORef DisplayNameEnv
  398   , scURIEnv         :: IORef (Map URI VarIndex)
  399   , scGlobalEnv      :: IORef (HashMap Ident Term)
  400   , scNextVarIndex   :: IORef VarIndex
  401   }
  402 
  403 data SharedContextCheckpoint =
  404   SCC
  405   { sccModuleMap :: ModuleMap
  406   , sccNamingEnv :: DisplayNameEnv
  407   , sccURIEnv    :: Map URI VarIndex
  408   , sccGlobalEnv :: HashMap Ident Term
  409   }
  410 
  411 checkpointSharedContext :: SharedContext -> IO SharedContextCheckpoint
  412 checkpointSharedContext sc =
  413   do mmap <- readIORef (scModuleMap sc)
  414      nenv <- readIORef (scDisplayNameEnv sc)
  415      uenv <- readIORef (scURIEnv sc)
  416      genv <- readIORef (scGlobalEnv sc)
  417      return SCC
  418             { sccModuleMap = mmap
  419             , sccNamingEnv = nenv
  420             , sccURIEnv = uenv
  421             , sccGlobalEnv = genv
  422             }
  423 
  424 restoreSharedContext :: SharedContextCheckpoint -> SharedContext -> IO SharedContext
  425 restoreSharedContext scc sc =
  426   do writeIORef (scModuleMap sc) (sccModuleMap scc)
  427      writeIORef (scDisplayNameEnv sc) (sccNamingEnv scc)
  428      writeIORef (scURIEnv sc) (sccURIEnv scc)
  429      writeIORef (scGlobalEnv sc) (sccGlobalEnv scc)
  430      return sc

  457 -- | Internal function to get the next available 'VarIndex'. Not exported.
  458 scFreshVarIndex :: SharedContext -> IO VarIndex
  459 scFreshVarIndex sc = atomicModifyIORef' (scNextVarIndex sc) (\i -> (i + 1, i))

  461 -- | Internal function to register a name with a caller-provided
  462 -- 'VarIndex'. Not exported.
  463 scRegisterNameWithIndex :: SharedContext -> VarIndex -> NameInfo -> IO ()
  464 scRegisterNameWithIndex sc i nmi =
  465   do uris <- readIORef (scURIEnv sc)
  466      let uri = nameURI nmi
  467      when (Map.member uri uris) $ throwIO (DuplicateNameException uri)
  468      writeIORef (scURIEnv sc) (Map.insert uri i uris)
  469      modifyIORef' (scDisplayNameEnv sc) $ extendDisplayNameEnv i (nameAliases nmi)

  517 -- | Returns shared term associated with ident.

  519 scGlobalDef :: SharedContext -> Ident -> IO Term
  520 scGlobalDef sc ident =
  521   do m <- readIORef (scGlobalEnv sc)

  523        Nothing -> fail ("Could not find global: " ++ show ident)

  526 -- | Internal function to register an 'Ident' with a 'Term' (which
  527 -- must be a 'Constant' term with the same 'Ident') in the
  528 -- 'scGlobalEnv' map of the 'SharedContext'. Not exported.
  529 scRegisterGlobal :: SharedContext -> Ident -> Term -> IO ()
  530 scRegisterGlobal sc ident t =
  531   do dup <- atomicModifyIORef' (scGlobalEnv sc) f
  532      when dup $ fail ("Global identifier already registered: " ++ show ident)
  533   where
  534     f m =
  535       case HMap.lookup ident m of
  536         Just _ -> (m, True)
  537         Nothing -> (HMap.insert ident t m, False)

  539 -- | Find a variant of an identifier that is not already being used as a global,

  543   readIORef (scGlobalEnv sc) >>= \gmap ->
  544   return $ fromJust $ find (\i -> not $ HMap.member i gmap) $
  545   ident : map (mkIdent (identModule ident) .
  546                Text.append (identBaseName ident) .
  547                Text.pack . show) [(0::Integer) ..]

  588 -- | Internal function to insert a definition into the 'ModuleMap' of
  589 -- the 'SharedContext'. Throws 'DuplicateNameException' if the name is
  590 -- already registered.
  591 scInsDefInMap :: SharedContext -> Def -> IO ()
  592 scInsDefInMap sc d =
  593   do e <- atomicModifyIORef' (scModuleMap sc) $ \mm ->
  594        case insDefInMap d mm of
  595          Left i -> (mm, Just (DuplicateNameException (moduleIdentToURI i)))
  596          Right mm' -> (mm', Nothing)
  597      maybe (pure ()) throwIO e

  599 -- | Internal function to extend the SAWCore global environment with a
  600 -- new constant, which may or may not have a definition. Not exported.
  601 -- Assumes that the type and body (if present) are closed terms, and
  602 -- that the body has the given type.
  603 scDeclareDef ::
  604   SharedContext -> Name -> DefQualifier -> Term -> Maybe Term -> IO Term
  605 scDeclareDef sc nm q ty body =
  606   do scInsDefInMap sc $
  607        Def
  608        { defName = nm
  609        , defQualifier = q
  610        , defType = ty
  611        , defBody = body
  612        }
  613      t <- scConst sc nm
  614      -- Register constant in scGlobalEnv if it has an Ident name
  615      case nameInfo nm of
  616        ModuleIdentifier ident -> scRegisterGlobal sc ident t
  617        ImportedName{} -> pure ()
  618      pure t

  820 -- | Build a 'Ctor' from a 'CtorArgStruct' and a list of the other constructor
  821 -- names of the 'DataType'. Note that we cannot look up the latter information,
  822 -- as 'scBuildCtor' is called during construction of the 'DataType'.
  823 scBuildCtor ::
  824   SharedContext ->
  825   Name {- ^ data type -} ->
  826   Ident {- ^ constructor name -} ->
  827   CtorArgStruct {- ^ constructor formal arguments -} ->
  828   IO Ctor
  829 scBuildCtor sc d c arg_struct =
  830   do
  831     -- Step 0: allocate a fresh unique variable index for this constructor
  832     -- and register its name in the naming environment
  833     cname <- scRegisterName sc (ModuleIdentifier c)
  834 
  835     -- Step 1: build the types for the constructor and the type required
  836     -- of its eliminator functions
  837     tp <- ctxCtorType sc d arg_struct
  838     elim_tp_fun <- mkCtorElimTypeFun sc d cname arg_struct
  839 
  840     -- Step 2: build free variables for rec, elim and the
  841     -- constructor argument variables
  842     let num_args =
  843           case arg_struct of
  844             CtorArgStruct {..} -> length ctorArgs
  845 
  846     vars <- reverse <$> mapM (scLocalVar sc) (take num_args [0 ..])
  847     elim_var <- scLocalVar sc num_args
  848     rec_var  <- scLocalVar sc (num_args+1)
  849 
  850     -- Step 3: pass these variables to ctxReduceRecursor to build the
  851     -- ctorIotaTemplate field
  852     iota_red <- ctxReduceRecursor sc rec_var elim_var vars arg_struct
  853 
  854     -- Step 4: build the API function that shuffles the terms around in the
  855     -- correct way.
  856     let iota_fun rec cs_fs args =
  857           do let elim = case Map.lookup (nameIndex cname) cs_fs of
  858                    Just e -> e
  859                    Nothing ->
  860                      panic "ctorIotaReduction" [
  861                          "no eliminator for constructor " <> Text.pack (show c)
  862                      ]
  863              instantiateVarList sc 0 (reverse ([rec,elim]++args)) iota_red
  864 
  865     -- Finally, return the required Ctor record
  866     return $ Ctor
  867       { ctorName = cname
  868       , ctorArgStruct = arg_struct
  869       , ctorDataType = d
  870       , ctorType = tp
  871       , ctorElimTypeFun = \ps p_ret -> elim_tp_fun ps p_ret
  872       , ctorIotaTemplate  = iota_red
  873       , ctorIotaReduction = iota_fun
  874       }

  876 -- | Compute the type of an eliminator function for a constructor from the name
  877 -- of its datatype, its name, and its 'CtorArgStruct'. This type has, as free
  878 -- variables, both the parameters of the datatype and a "motive" function from
  879 -- indices of the datatype to a return type. It is of the form
  880 --
  881 -- > (x1::arg1) -> maybe (rec1::rec_tp1) -> .. ->
  882 -- > (xn::argn) -> maybe (recn::rec_tpn) ->
  883 -- >   p_ret ix_1 .. ix_k (ctor params x1 .. xn)
  884 --
  885 -- where the ixs are the type indices of the return type for the constructor,
  886 -- the (xi::argi) are the arguments of the constructor, and the @maybe@s
  887 -- indicate additional arguments that are present only for arguments of
  888 -- recursive type, that is, where @argi@ has the form
  889 --
  890 -- > (z1::Z1) -> .. -> (zm::Zm) -> d params t1 .. tk
  891 --
  892 -- In this case, @rec_tpi@ has the form
  893 --
  894 -- > (z1::Z1) -> .. -> (zm::Zm) -> p_ret t1 .. tk (f z1 .. zm)
  895 --
  896 -- Note that the output type cannot be expressed in the type of this function,
  897 -- since it depends on fields of the 'CtorArgStruct', so, instead, the result is
  898 -- just casted to whatever type the caller specifies.
  899 ctxCtorElimType ::
  900   SharedContext ->
  901   Name {- ^ data type name -} ->
  902   Name {- ^ constructor name -} ->
  903   Term {- ^ motive function -} ->
  904   CtorArgStruct ->
  905   IO Term
  906 ctxCtorElimType sc d c p_ret (CtorArgStruct{..}) =
  907   do params <- traverse (scVariable sc) ctorParams
  908      d_params <- scConstApply sc d params
  909      let helper :: [Term] -> [(Name, CtorArg)] -> IO Term
  910          helper prevs ((nm, ConstArg tp) : args) =
  911            -- For a constant argument type, just abstract it and continue
  912            do arg <- scVariable sc (EC nm tp)
  913               rest <- helper (prevs ++ [arg]) args
  914               scGeneralizeTerms sc [arg] rest
  915          helper prevs ((nm, RecursiveArg zs ts) : args) =
  916            -- For a recursive argument type of the form
  917            --
  918            -- (z1::Z1) -> .. -> (zm::Zm) -> d params t1 .. tk
  919            --
  920            -- form the type abstraction
  921            --
  922            -- (arg:: (z1::Z1) -> .. -> (zm::Zm) -> d params t1 .. tk) ->
  923            -- (ih :: (z1::Z1) -> .. -> (zm::Zm) -> p_ret t1 .. tk (arg z1 .. zm)) ->
  924            -- rest
  925            --
  926            -- where rest is the result of a recursive call
  927            do d_params_ts <- scApplyAll sc d_params ts
  928               -- Build the type of the argument arg
  929               arg_tp <- scGeneralizeExts sc zs d_params_ts
  930               arg <- scVariable sc (EC nm arg_tp)
  931               -- Build the type of ih
  932               pret_ts <- scApplyAll sc p_ret ts
  933               z_vars <- traverse (scVariable sc) zs
  934               arg_zs <- scApplyAll sc arg z_vars
  935               ih_ret <- scApply sc pret_ts arg_zs
  936               ih_tp <- scGeneralizeExts sc zs ih_ret
  937               -- Finally, build the pi-abstraction for arg and ih around the rest
  938               rest <- helper (prevs ++ [arg]) args
  939               scGeneralizeTerms sc [arg] =<< scFun sc ih_tp rest
  940          helper prevs [] =
  941            -- If we are finished with our arguments, construct the final result type
  942            -- (p_ret ret_ixs (c params prevs))
  943            do p_ret_ixs <- scApplyAll sc p_ret ctorIndices
  944               appliedCtor <- scConstApply sc c (params ++ prevs)
  945               scApply sc p_ret_ixs appliedCtor
  946      helper [] ctorArgs

  948 -- | Build a function that substitutes parameters and a @p_ret@ return type
  949 -- function into the type of an eliminator, as returned by 'ctxCtorElimType',
  950 -- for the given constructor. We return the substitution function in the monad
  951 -- so that we only call 'ctxCtorElimType' once but can call the function many
  952 -- times, in order to amortize the overhead of 'ctxCtorElimType'.
  953 mkCtorElimTypeFun ::
  954   SharedContext ->
  955   Name {- ^ data type name -} ->
  956   Name {- ^ constructor name -} ->
  957   CtorArgStruct ->
  958   IO ([Term] -> Term -> IO Term)
  959 mkCtorElimTypeFun sc d c argStruct =
  960   do -- Use de Bruijn variable for p_ret so we can instantiate it later
  961      -- NOTE: This is kind of gross, because the p_ret in the callback
  962      -- argument below does not always have the same type (it is as
  963      -- computed by ctxPRetTp, but the return sort may vary)
  964      p_ret_var <- scLocalVar sc 0
  965      ctxElimType <- ctxCtorElimType sc d c p_ret_var argStruct
  966      let vs = map ecVarIndex (ctorParams argStruct)
  967      return $ \params p_ret ->
  968        do t <- instantiateVarList sc 0 [p_ret] ctxElimType
  969           let subst = IntMap.fromList (zip vs params)
  970           scWhnf sc =<< scInstantiateExt sc subst t

  979 -- | Reduce an application of a recursor to a particular constructor.
  980 -- This is known in the Coq literature as an iota reduction. More specifically,
  981 -- the call
  982 --
  983 -- > ctxReduceRecursor rec f_c [x1, .., xk]
  984 --
  985 -- reduces the term @(RecursorApp rec ixs (CtorApp c ps xs))@ to
  986 --
  987 -- > f_c x1 (maybe rec_tm_1) .. xk (maybe rec_tm_k)
  988 --
  989 -- where @f_c@ is the eliminator function associated to the constructor @c@ by the
  990 -- recursor value @rec@.  Here @maybe rec_tm_i@ indicates an optional recursive call
  991 -- of the recursor on one of the @xi@. These recursive calls only exist for those
  992 -- arguments @xi@ that are recursive arguments, i.e., that are specified with
  993 -- 'RecursiveArg', and are omitted for non-recursive arguments specified by 'ConstArg'.
  994 --
  995 -- Specifically, for a @'RecursiveArg' zs ixs@ argument @xi@, which has type
  996 -- @\(z1::Z1) -> .. -> d p1 .. pn ix1 .. ixp@, we build the recursive call
  997 --
  998 -- > \(z1::[xs/args]Z1) -> .. ->
  999 -- >   RecursorApp rec [xs/args]ixs (xi z1 ... zn)
 1000 --
 1001 -- where @[xs/args]@ substitutes the concrete values for the earlier
 1002 -- constructor arguments @xs@ for the remaining free variables.
 1003 
 1004 ctxReduceRecursor ::
 1005   SharedContext ->
 1006   Term {- ^ abstracted recursor -} ->
 1007   Term {- ^ constructor eliminator function -} ->
 1008   [Term] {- ^ constructor arguments -} ->
 1009   CtorArgStruct {- ^ constructor formal argument descriptor -} ->
 1010   IO Term
 1011 ctxReduceRecursor sc rec elimf c_args CtorArgStruct{..} =
 1012   case zipSameLength c_args ctorArgs of
 1013      Just argsCtx_ctorArgs ->
 1014        ctxReduceRecursor_ sc rec elimf argsCtx_ctorArgs
 1015      Nothing ->
 1016        error "ctxReduceRecursorRaw: wrong number of constructor arguments!"

 1019 -- | This operation does the real work of building the iota reduction
 1020 -- for @ctxReduceRecursor@.
 1021 ctxReduceRecursor_ ::
 1022   SharedContext ->
 1023   Term     {- ^ recursor value eliminatiting data type d -}->
 1024   Term     {- ^ eliminator function for the constructor -} ->
 1025   [(Term, (Name, CtorArg))] {- ^ constructor actual arguments plus argument descriptions -} ->
 1026   IO Term
 1027 ctxReduceRecursor_ sc rec fi args0_argCtx =
 1028   do args <- mk_args IntMap.empty args0_argCtx
 1029      scWhnf sc =<< scApplyAll sc fi args

 1032     mk_args :: IntMap Term ->  -- already processed parameters/arguments
 1033                [(Term, (Name, CtorArg))] ->
 1034                  -- remaining actual arguments to process, with
 1035                  -- telescope for typing the actual arguments
 1036                IO [Term]
 1037     -- no more arguments to process
 1038     mk_args _ [] = return []
 1039 
 1040     -- process an argument that is not a recursive call
 1041     mk_args pre_xs ((x, (nm, ConstArg _)) : xs_args) =
 1042       do tl <- mk_args (IntMap.insert (nameIndex nm) x pre_xs) xs_args
 1043          pure (x : tl)
 1044 
 1045     -- process an argument that is a recursive call
 1046     mk_args pre_xs ((x, (nm, RecursiveArg zs ixs)) : xs_args) =
 1047       do zs'  <- traverse (traverse (scInstantiateExt sc pre_xs)) zs
 1048          ixs' <- traverse (scInstantiateExt sc pre_xs) ixs
 1049          recx <- mk_rec_arg zs' ixs' x

 1053     -- Build an individual recursive call, given the parameters, the bindings
 1054     -- for the RecursiveArg, and the argument we are going to recurse on
 1055     mk_rec_arg ::
 1056       [ExtCns Term] ->                -- telescope describing the zs
 1057       [Term] ->                        -- actual values for the indices, shifted under zs

 1061       -- eta expand over the zs and apply the RecursorApp form
 1062       do zs <- traverse (scVariable sc) zs_ctx
 1063          x_zs <- scApplyAll sc x zs
 1064          body <- scRecursorApp sc rec ixs x_zs
 1065          scAbstractExts sc zs_ctx body

 1181 -- | Reduces beta-redexes, tuple/record selectors, recursor applications, and
 1182 -- definitions at the top level of a term, and evaluates all arguments to type
 1183 -- constructors (including function, record, and tuple types).
 1184 --
 1185 -- NOTE: this notion of weak head normal form differs from the standard type
 1186 -- theory definition, in that it normalizes the arguments of type-forming
 1187 -- constructs like pi types, pair types, etc. The idea is that these constructs
 1188 -- are being treated as strict constructors in the Haskell sense.
 1189 scWhnf :: SharedContext -> Term -> IO Term

 1191   do cache <- newCacheIntMap

 1193   where
 1194     memo :: (?cache :: Cache IO TermIndex Term) => Term -> IO Term

 1197         Unshared _ -> go [] t

 1203     go xs                     (asRecordSelector -> Just (t, n)) = go (ElimProj n : xs) t
 1204     go xs                     (asPairSelector -> Just (t, i))   = go (ElimPair i : xs) t
 1205     go (ElimApp x : xs)       (asLambda -> Just (_, _, body))   = betaReduce xs [x] body
 1206     go (ElimPair i : xs)      (asPairValue -> Just (a, b))      = go xs (if i then b else a)
 1207     go (ElimProj fld : xs)    (asRecordValue -> Just elems)     = case Map.lookup fld elems of
 1208                                                                     Just t -> go xs t
 1209                                                                     Nothing ->
 1210                                                                       error "scWhnf: field missing in record"

 1212                               (asNat -> Just n)                 = scReduceNatRecursor sc rec crec n >>= go xs

 1214                                 Just (r, crec, ixs, arg))       = go (ElimRecursor r crec ixs : xs) arg

 1216                                                                      t' <- scPairValue sc a b'

 1219                                                                      b' <- memo b
 1220                                                                      t' <- scPairType sc a' b'
 1221                                                                      foldM reapply t' xs
 1222     go xs                     (asRecordType -> Just elems)      = do elems' <- mapM (\(i,t) -> (i,) <$> memo t)
 1223                                                                                     (Map.assocs elems)
 1224                                                                      t' <- scRecordType sc elems'
 1225                                                                      foldM reapply t' xs
 1226     go xs                     (asPi -> Just (x,aty,rty))        = do aty' <- memo aty
 1227                                                                      rty' <- memo rty
 1228                                                                      t' <- scPi sc x aty' rty'
 1229                                                                      foldM reapply t' xs
 1230     go xs                     t@(asConstant -> Just nm)         = do r <- resolveConstant nm
 1231                                                                      case r of
 1232                                                                        ResolvedDef d ->
 1233                                                                          case defBody d of
 1234                                                                            Just body -> go xs body
 1235                                                                            Nothing -> foldM reapply t xs
 1236                                                                        ResolvedCtor ctor ->
 1237                                                                          case asArgsRec xs of
 1238                                                                            Nothing -> foldM reapply t xs
 1239                                                                            Just (rec, crec, args, xs') ->
 1240                                                                              do let args' = drop (ctorNumParams ctor) args

 1287        whnf _c t@(Unshared _) = unwrapTermF <$> eval sc t

 1523 -- | The second argument is a function that takes the number of
 1524 -- enclosing lambdas and the de Bruijn index of the variable,
 1525 -- returning the new term to replace it with.
 1526 instantiateLocalVars ::
 1527   SharedContext ->
 1528   (DeBruijnIndex -> DeBruijnIndex -> IO Term) ->
 1529   DeBruijnIndex -> Term -> IO Term
 1530 instantiateLocalVars sc f initialLevel t0 =
 1531   do cache <- newCache
 1532      let ?cache = cache in go initialLevel t0
 1533   where
 1534     go :: (?cache :: Cache IO (TermIndex, DeBruijnIndex) Term) =>
 1535           DeBruijnIndex -> Term -> IO Term
 1536     go l t =
 1537       case t of
 1538         Unshared tf -> go' l tf
 1539         STApp{ stAppIndex = tidx, stAppTermF = tf }
 1540           | termIsClosed t -> return t -- closed terms map to themselves

 1620 -- this as:
 1621 --
 1622 -- > instantiateVarList 0 [x,y,z] t ==
 1623 -- >    instantiateVar 0 x (instantiateVar 1 (incVars 0 1 y)
 1624 -- >                       (instantiateVar 2 (incVars 0 2 z) t))

 1626                    -> DeBruijnIndex -> [Term] -> Term -> IO Term
