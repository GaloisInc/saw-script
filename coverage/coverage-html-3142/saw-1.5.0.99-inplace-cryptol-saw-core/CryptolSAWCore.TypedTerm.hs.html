   66 -- Typed terms -----------------------------------------------------------------
   67 
   68 -- | Within SAWScript, we represent an object language term as a
   69 -- SAWCore shared term paired with a Cryptol type schema. The Cryptol
   70 -- type is used for type inference/checking of inline Cryptol
   71 -- expressions.
   72 
   73 data TypedTerm =
   74   TypedTerm
   75   { ttType :: TypedTermType
   76   , ttTerm :: Term
   77   }
   78   deriving Show
   79 
   80 
   81 -- | The different notions of Cryptol types that a SAWCore term might
   82 --   have. Kinds, types, and values are stratified in Cryptol, but not
   83 --   in SAWCore because SAWCore is dependently typed.
   84 --
   85 data TypedTermType
   86   = TypedTermSchema C.Schema
   87   | TypedTermKind   C.Kind
   88   | TypedTermOther  Term
   89  deriving Show
   90

   92 -- It is not ideal to have two copies of these but it's not that
   93 -- helpful to try to share. In the long run the pure ones should
   94 -- probably go away as they produce worse output.

   96 -- | Print a `TypedTerm` to a `PP.Doc`.
   97 prettyTypedTerm :: SharedContext -> TypedTerm -> IO (PP.Doc ann)
   98 prettyTypedTerm sc (TypedTerm tp tm) = do
   99   tm' <- prettyTerm sc tm
  100   tp' <- prettyTypedTermType sc tp
  101   pure $ PP.unAnnotate tm' <+> ":" <+> tp'
  102 
  103 -- | Print a `TypedTermType` to a `PP.Doc`.
  104 prettyTypedTermType :: SharedContext -> TypedTermType -> IO (PP.Doc ann)
  105 prettyTypedTermType _sc (TypedTermSchema sch) =
  106   pure $ CryPP.pretty sch
  107 prettyTypedTermType _sc (TypedTermKind k) =
  108   pure $ CryPP.pretty k
  109 prettyTypedTermType sc (TypedTermOther tp) = do
  110   tp' <- prettyTerm sc tp
  111   pure $ PP.unAnnotate tp'
  112 
  113 -- | Print a `TypedTerm` to a `PP.Doc`, without needing the
  114 --   `SharedContext` or `IO`.
  115 --
  116 --   This produces inferior output at the SAWCore level. See `prettyTermPure`.
  117 --
  118 prettyTypedTermPure :: TypedTerm -> PP.Doc ann
  119 prettyTypedTermPure (TypedTerm tp tm) =
  120   PP.unAnnotate (prettyTermPure PPS.defaultOpts tm)
  121   <+> ":" <+>
  122   prettyTypedTermTypePure tp
  123

  129 prettyTypedTermTypePure :: TypedTermType -> PP.Doc ann

  175 -- | Convert the 'ttType' field of a 'TypedTerm' to a SAWCore term
  176 ttTypeAsTerm :: SharedContext -> CryptolEnv -> TypedTerm -> IO Term
  177 ttTypeAsTerm sc env (TypedTerm (TypedTermSchema schema) _) =
  178   translateSchema sc env schema
  179 ttTypeAsTerm sc _ (TypedTerm (TypedTermKind k) _) = importKind sc k
  180 ttTypeAsTerm _ _ (TypedTerm (TypedTermOther tp) _) = return tp
  181 
  182 ttTermLens :: Functor f => (Term -> f Term) -> TypedTerm -> f TypedTerm
  183 ttTermLens f tt = tt `seq` fmap (\x -> tt{ttTerm = x}) (f (ttTerm tt))
  184 
  185 -- | Check if a `TypedTermType` is monomorphic and if so return its type.
  186 ttIsMono :: TypedTermType -> Maybe C.Type
  187 ttIsMono ttp =
  188   case ttp of
  189     TypedTermSchema sch -> C.isMono sch
  190     _ -> Nothing
  191 
  192 -- | Cons a `TypedTerm` from a `Term` by attempting to lift its
  193 --   SAWCore type back to Cryptol. See `scCryptolType`.
  194 mkTypedTerm :: SharedContext -> Term -> IO TypedTerm
  195 mkTypedTerm sc trm = do
  196   ty <- scTypeOf sc trm
  197   ct <- scCryptolType sc ty
  198   let ttt = case ct of
  199         Nothing        -> TypedTermOther ty
  200         Just (Left k)  -> TypedTermKind k
  201         Just (Right t) -> TypedTermSchema (C.tMono t)
  202   return (TypedTerm ttt trm)
  203 
  204 -- | Apply a 'TypedTerm' to a list of arguments. This operation fails
  205 -- if the first 'TypedTerm' does not have a function type of
  206 -- sufficient arity, or if the types of the arguments do not match
  207 -- the type of the function.
  208 applyTypedTerms :: SharedContext -> TypedTerm -> [TypedTerm] -> IO TypedTerm
  209 applyTypedTerms sc (TypedTerm _ fn) args =
  210   do trm <- foldM (scApply sc) fn (map ttTerm args)
  211      ty <- scTypeOf sc trm
  212      -- NB, scCryptolType can behave in strange ways due to the non-injectivity
  213      -- of the mapping from Cryptol to SAWCore types. Perhaps we would be better
  214      -- to combine the incoming type information instead of applying and then
  215      -- reconstructing here.
  216      ct <- scCryptolType sc ty
  217      let ttt = case ct of
  218            Nothing        -> TypedTermOther ty
  219            Just (Left k)  -> TypedTermKind k
  220            Just (Right t) -> TypedTermSchema (C.tMono t)
  221      return (TypedTerm ttt trm)

  223 -- First order types and values ------------------------------------------------
  224 
  225 -- | Convert a `FirstOrderType` to a Cryptol `C.Type`.
  226 cryptolTypeOfFirstOrderType :: FirstOrderType -> C.Type
  227 cryptolTypeOfFirstOrderType fot =
  228   case fot of
  229     FOTBit -> C.tBit
  230     FOTInt -> C.tInteger
  231     FOTIntMod n -> C.tIntMod (C.tNum n)
  232     FOTVec n t -> C.tSeq (C.tNum n) (cryptolTypeOfFirstOrderType t)
  233     -- NB, special case, don't produce 1-tuples
  234     FOTTuple [x] -> cryptolTypeOfFirstOrderType x
  235     FOTTuple ts -> C.tTuple (map cryptolTypeOfFirstOrderType ts)
  236     FOTArray a b ->
  237       C.tArray
  238       (cryptolTypeOfFirstOrderType a)
  239       (cryptolTypeOfFirstOrderType b)
  240     FOTRec m ->
  241       C.tRec $
  242       C.recordFromFields $
  243       [ (C.mkIdent l, cryptolTypeOfFirstOrderType t)
  244       | (l, t) <- Map.assocs m ]
  245 
  246 -- | Convert a `FirstOrderValue` to a `TypedTerm`.
  247 typedTermOfFirstOrderValue :: SharedContext -> FirstOrderValue -> IO TypedTerm
  248 typedTermOfFirstOrderValue sc fov =
  249   do let fot = firstOrderTypeOf fov
  250      let cty = cryptolTypeOfFirstOrderType fot
  251      t <- scFirstOrderValue sc fov
  252      pure $ TypedTerm (TypedTermSchema (C.tMono cty)) t

  256 data TypedVariable =
  257   TypedVariable
  258   { tvCType :: C.Type
  259   , tvName :: VarName
  260   , tvType :: Term
  261   }
  262   deriving Show
  263 
  264 -- | Recognize 'TypedTerm's that are variables.
  265 asTypedVariable :: TypedTerm -> Maybe TypedVariable
  266 asTypedVariable (TypedTerm tp t) =
  267   do cty <- ttIsMono tp
  268      (x, ty) <- asVariable t
  269      pure $ TypedVariable cty x ty
  270 
  271 -- | Make a 'TypedTerm' from a 'TypedVariable'.
  272 typedTermOfVariable :: SharedContext -> TypedVariable -> IO TypedTerm
  273 typedTermOfVariable sc (TypedVariable cty vn ty) =
  274   TypedTerm (TypedTermSchema (C.tMono cty)) <$> scVariable sc vn ty
  275 
  276 abstractTypedVars :: SharedContext -> [TypedVariable] -> TypedTerm -> IO TypedTerm
  277 abstractTypedVars sc tvars (TypedTerm (TypedTermSchema (C.Forall params props ty)) trm) =
  278   do let tys = map tvCType tvars
  279      let vars = map (\(TypedVariable _ vn tp) -> (vn, tp)) tvars
  280      let ty' = foldr C.tFun ty tys
  281      trm' <- scLambdaList sc vars trm
  282      pure $ TypedTerm (TypedTermSchema (C.Forall params props ty')) trm'
  283 abstractTypedVars sc tvars (TypedTerm (TypedTermKind k) trm) =
  284   do let vars = map (\(TypedVariable _ vn tp) -> (vn, tp)) tvars
  285      trm' <- scLambdaList sc vars trm
  286      pure $ TypedTerm (TypedTermKind k) trm'
  287 abstractTypedVars sc tvars (TypedTerm (TypedTermOther _tp) trm) =
  288   do let vars = map (\(TypedVariable _ vn tp) -> (vn, tp)) tvars
  289      trm' <- scLambdaList sc vars trm
  290      tp'  <- scTypeOf sc trm'
  291      pure $ TypedTerm (TypedTermOther tp') trm'

  293 -- Typed modules ---------------------------------------------------------------
  294 
  295 -- | In SAWScript, we can refer to a Cryptol module as a first class
  296 -- value. These are represented simply as maps from names to typed
  297 -- terms.
  298 
  299 data CryptolModule =
  300   CryptolModule
  301     (Map C.Name C.TySyn)    -- ^ type synonyms
  302     (Map C.Name TypedTerm)  -- ^ symbols (mapping to SAWCore things).
  303 
  304 -- | Print a `CryptolModule` (module handle).
  305 prettyCryptolModule :: CryptolModule -> PP.Doc ann
  306 prettyCryptolModule (CryptolModule sm tm) =
  307   let prettyTSyn (C.TySyn name params _props rhs _doc) =
  308         let name' = CryPP.pretty (nameIdent name)
  309             params' = map CryPP.pretty params
  310             rhs' = CryPP.pretty rhs
  311         in
  312         PP.indent 4 $ PP.hsep (name' : params') <+> "=" <+> rhs'
  313 
  314       prettyBinding (name, TypedTerm (TypedTermSchema schema) _) =
  315         let name' = CryPP.pretty (nameIdent name)
  316             schema' = CryPP.pretty schema
  317         in
  318         [PP.indent 4 $ name' <+> ":" <+> schema']
  319       prettyBinding _ =
  320         []
  321       synonyms =
  322         if Map.null sm then []
  323         else
  324             "Type Synonyms" : "=============" : map prettyTSyn (Map.elems sm) ++ [""]
  325       symbols =
  326         "Symbols" : "=======" : concatMap prettyBinding (Map.assocs tm)
  327   in
  328   PP.vsep $ synonyms ++ symbols
