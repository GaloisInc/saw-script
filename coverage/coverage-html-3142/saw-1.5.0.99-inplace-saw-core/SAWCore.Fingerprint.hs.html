   53 data FPSt = FPSt
   54   { fpTermMemo    :: !(IntMap ByteString)
   55     -- ^ TermIndex -> hash
   56   , fpDefMemo     :: !(IntMap ByteString)
   57     -- ^ VarIndex of global Name -> hash of its definition
   58   , fpDefVisiting :: !IntSet
   59     -- ^ VarIndexes currently being traversed (cycle detection)
   60   , fpModuleMap   :: !ModuleMap
   61     -- ^ The module map for looking up definitions
   62   , fpUninterp    :: !(Set VarIndex)
   63     -- ^ Uninterpreted constants (only hash name + type, not body)
   64   }

   66 insertTerm :: MonadState FPSt m => Term -> ByteString -> m ByteString
   67 insertTerm t h = do
   68   modify' $
   69     \s -> s { fpTermMemo = IntMap.insert (termIndex t) h $ fpTermMemo s }
   70   pure h
   71 
   72 lookupTerm :: MonadState FPSt m => Term -> m (Maybe ByteString)
   73 lookupTerm t = IntMap.lookup (termIndex t) <$> gets fpTermMemo
   74 
   75 insertDef :: MonadState FPSt m => Name -> ByteString -> m ByteString
   76 insertDef nm h = do
   77   modify' $
   78     \s -> s { fpDefMemo = IntMap.insert (nameIndex nm) h $ fpDefMemo s }
   79   pure h
   80 
   81 lookupDef :: MonadState FPSt m => Name -> m (Bool, Maybe ByteString)
   82 lookupDef nm = do
   83   visiting <- IntSet.member (nameIndex nm) <$> gets fpDefVisiting
   84   def      <- IntMap.lookup (nameIndex nm) <$> gets fpDefMemo
   85   pure (visiting, def)
   86 
   87 visit :: MonadState FPSt m => Name -> m ()
   88 visit nm = modify' $
   89   \s -> s { fpDefVisiting = IntSet.insert (nameIndex nm) $ fpDefVisiting s }
   90 
   91 unvisit :: MonadState FPSt m => Name -> m ()
   92 unvisit nm = modify' $
   93   \s -> s { fpDefVisiting = IntSet.delete (nameIndex nm) $ fpDefVisiting s }
   94 
   95 lookupName :: MonadState FPSt m => Name -> m (Bool, Maybe ResolvedName)
   96 lookupName nm = do
   97    isUninterp <- Set.member (nameIndex nm) <$> gets fpUninterp
   98    resolved   <- lookupVarIndexInMap (nameIndex nm) <$> gets fpModuleMap
   99    pure (isUninterp, resolved)

  101 sha256 :: [ByteString] -> ByteString
  102 sha256 = SHA256.finalize . foldl' SHA256.update SHA256.init

  104 byte :: Word8 -> ByteString
  105 byte = BS.singleton

  107 -- | Basic serialization of small morsels.
  108 class Bytes a where
  109   bytes :: a -> ByteString
  110 instance Bytes Int where
  111   -- Minor optimization for the Int case.
  112   bytes n = BS.pack
  113     [ fromIntegral (n `shiftR` 56), fromIntegral (n `shiftR` 48)
  114     , fromIntegral (n `shiftR` 40), fromIntegral (n `shiftR` 32)
  115     , fromIntegral (n `shiftR` 24), fromIntegral (n `shiftR` 16)
  116     , fromIntegral (n `shiftR` 8),  fromIntegral n
  117     ]
  118 instance Bytes Natural where
  119   bytes = BSL.toStrict . encode
  120 instance Bytes Text where
  121   bytes txt = bytes (BS.length bs) <> bs
  122     where
  123       bs = encodeUtf8 txt
  124 instance Bytes NameInfo where
  125   bytes = \case
  126     ModuleIdentifier i -> byte 0x1 <> bytes (QN.ppQualName $ moduleIdentToQualName i)
  127     ImportedName qn _  -> byte 0x2 <> bytes (QN.ppQualName qn)
  128 instance Bytes Sort where
  129   bytes = \case
  130     PropSort   -> byte 0x1
  131     TypeSort n -> byte 0x2 <> bytes n
  132 instance Bytes SortFlags where
  133   bytes sf = b (flagInhabited sf) <> b (flagQuantType sf)
  134     where b = byte . fromIntegral . fromEnum
  135 instance Bytes FirstOrderType where
  136   bytes = \case
  137     FOTBit       -> byte 0x1
  138     FOTInt       -> byte 0x2
  139     FOTIntMod n  -> byte 0x3 <> bytes n
  140     FOTVec n t   -> byte 0x4 <> bytes n <> bytes t
  141     FOTArray k v -> byte 0x5 <> bytes k <> bytes v
  142     FOTTuple ts  -> byte 0x6 <> bytes (length ts)
  143                              <> mconcat (bytes <$> ts)
  144     FOTRec m     -> byte 0x7 <> bytes (Map.size m)
  145                              <> mconcat [ bytes k <> byte 0 <> bytes v
  146                                         | (k, v) <- Map.toAscList m ]
  147 
  148 fpTerms :: MonadState FPSt m => VarCtx -> [Term] -> m [ByteString]
  149 fpTerms ctx ts = (bytes (length ts) :) <$> mapM (fpTerm ctx) ts
  150 
  151 fpTerm :: MonadState FPSt m => VarCtx -> Term -> m ByteString
  152 fpTerm ctx t = lookupTerm t >>= \case
  153   Just h  -> pure h
  154   Nothing -> fpTermF ctx (unwrapTermF t) >>= insertTerm t
  155 
  156 fpTermF :: MonadState FPSt m => VarCtx -> TermF Term -> m ByteString
  157 fpTermF ctx = \case
  158 
  159   App e1 e2 -> do
  160     h1 <- fpTerm ctx e1
  161     h2 <- fpTerm ctx e2
  162     pure $ sha256 [byte 0x1, h1, h2]
  163 
  164   Lambda x ty body -> do
  165     hty   <- fpTerm ctx ty
  166     hbody <- fpTerm (consVarCtx x ctx) body
  167     pure $ sha256 [byte 0x2, hty, hbody]
  168 
  169   Pi x ty body -> do
  170     hty   <- fpTerm ctx ty
  171     hbody <- fpTerm (consVarCtx x ctx) body
  172     pure $ sha256 [byte 0x3, hty, hbody]
  173 
  174   Constant nm -> do
  175     hdef <- fpDef nm
  176     pure $ sha256 [byte 0x4, hdef]
  177 
  178   Variable vn ty
  179     | Just dbi <- lookupVarCtx vn ctx ->      -- Bound var: use index.
  180       pure $ sha256 [byte 0x5, bytes dbi]
  181     | otherwise -> do                         -- Free var: use name.
  182       hty <- fpTerm ctx ty
  183       pure $ sha256 [byte 0x6, bytes $ vnName vn, hty]
  184 
  185   FTermF (Sort s flags) ->
  186     pure $ sha256 [byte 0x7, bytes s, bytes flags]
  187 
  188   FTermF (StringLit txt) ->
  189     pure $ sha256 [byte 0x8, bytes txt]
  190 
  191   FTermF (ArrayValue elemTy elems) -> do
  192     hty <- fpTerm ctx elemTy
  193     hs  <- fpTerms ctx $ V.toList elems
  194     pure $ sha256 $ [byte 0x9, hty] <> hs
  195 
  196   FTermF (Recursor cr) -> do
  197     let dtNi    = bytes $ nameInfo $ recursorDataType cr
  198         ctorNis = bytes . nameInfo <$> recursorCtorOrder cr
  199     dtH <- fpDef $ recursorDataType cr
  200     pure $ sha256 $ [ byte 0xa
  201                     , dtNi
  202                     , bytes $ recursorSort cr
  203                     , bytes $ recursorNumParams cr
  204                     , bytes $ recursorNumIxs cr
  205                     , bytes $ length $ recursorCtorOrder cr
  206                     ] <> ctorNis <> [dtH]
  207 
  208 fpDef :: MonadState FPSt m => Name -> m ByteString
  209 fpDef nm = do
  210   lookupDef nm >>= \case
  211     (_,    Just h)  -> pure h
  212     (True, Nothing) -> pure ni
  213     _               -> do
  214       visit nm
  215 
  216       h <- lookupName nm >>= \case
  217         (True, Just (ResolvedDef d)) -> do -- Uninterpreted.
  218           hty <- fpTerm emptyVarCtx $ defType d
  219           pure $ sha256 [byte 0x11, ni, hty]
  220         (_, Just (ResolvedDef d))
  221           | Just body <- defBody d -> do
  222             hty   <- fpTerm emptyVarCtx $ defType d
  223             hbody <- fpTerm emptyVarCtx body
  224             pure $ sha256 [byte 0x13, ni, hty, hbody]
  225         (_, Just (ResolvedDef d)) -> do
  226           hty   <- fpTerm emptyVarCtx $ defType d
  227           pure $ sha256 [byte 0x12, ni, hty]
  228         (_, Just (ResolvedDataType dt)) -> do
  229           hty    <- fpTerm emptyVarCtx $ dtType dt
  230           hctors <- fpTerms emptyVarCtx $ (ctorType <$> dtCtors dt)
  231           pure $ sha256 $ [byte 0x14, ni, hty, bytes $ dtSort dt] <> hctors
  232         (_, Just (ResolvedCtor c)) -> do
  233           hty <- fpTerm emptyVarCtx $ ctorType c
  234           pure $ sha256 [byte 0x15, ni, hty]
  235         (_, Nothing) -> pure $ sha256 [byte 0x16, ni]
  236 
  237       unvisit nm
  238       insertDef nm h
  239   where
  240     ni :: ByteString
  241     ni = bytes $ nameInfo nm
  242 
  243 fpAssert :: MonadState FPSt m => VarCtx -> SATAssert -> m ByteString
  244 fpAssert ctx = \case
  245   BoolAssert tm -> do
  246     h <- fpTerm ctx tm
  247     pure $ sha256 [byte 0x21, h]
  248   UniversalAssert vars hyps concl -> do
  249     let varTys = bytes . snd <$> vars
  250         ctx'   = foldl' (flip consVarCtx) ctx $ fst <$> vars
  251     hhyps  <- fpTerms ctx' hyps
  252     hconcl <- fpTerm ctx' concl
  253     pure $ sha256 $ [byte 0x22, bytes $ length vars]
  254                  <> varTys
  255                  <> hhyps
  256                  <> [hconcl]

  258 -- | Compute a SHA-256 fingerprint of a SATQuery, intended for use as a
  259 --   solver cache key. The result should be:
  260 --   * Stable across runs (has no VarIndex/TermIndex dependence)
  261 --   * Alpha-equivalence-respecting (uses de Bruijn indices for bound variables)
  262 --   * Definition-sensitive (transitively hashes constant bodies)
  263 fingerprintSATQuery :: MonadIO m => SharedContext -> SATQuery -> m ByteString
  264 fingerprintSATQuery sc satq = do
  265   mm <- liftIO $ scGetModuleMap sc
  266   let st0 = FPSt IntMap.empty IntMap.empty IntSet.empty mm $ satUninterp satq
  267   evalStateT fpSATQuery st0
  268   where
  269     fpSATQuery :: MonadState FPSt m => m ByteString
  270     fpSATQuery = do
  271       hasserts <- mapM (fpAssert queryCtx) $ satAsserts satq
  272       pure $ sha256 $ [bytes $ length vars]
  273                    <> varTys
  274                    <> sort hasserts
  275 
  276     -- Build a VarCtx from satVariables so free variables in assertion
  277     -- terms get de Bruijn indices instead of being hashed by name.
  278     queryCtx :: VarCtx
  279     queryCtx = foldl' (flip consVarCtx) emptyVarCtx $ fst <$> vars
  280 
  281     vars :: [(VarName, FirstOrderType)]
  282     vars = Map.toAscList $ satVariables satq
  283 
  284     varTys :: [ByteString]
  285     varTys = bytes . snd <$> vars
