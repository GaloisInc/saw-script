   48 -- | Read-only bits of state.
   49 data FPCfg = FPCfg
   50   { fpModuleMap :: !ModuleMap
   51     -- ^ The module map for looking up definitions.
   52   , fpUninterp  :: !(Set VarIndex)
   53     -- ^ Uninterpreted constants (only serialize name + type, not body).
   54   }
   55 
   56 -- | Mutable state tracked while serializing.
   57 data FPSt = FPSt
   58   { fpTermMemo    :: !(IntMap Int)
   59     -- ^ TermIndex -> sequence number (assigned on first serialization).
   60   , fpDefMemo     :: !(IntMap Int)
   61     -- ^ VarIndex of global Name -> sequence number.
   62   , fpDefVisiting :: !IntSet
   63     -- ^ VarIndexes currently being traversed.
   64   , fpNextRef     :: !Int
   65     -- ^ Next available sequence number.
   66   }
   67 
   68 freshRef :: MonadState FPSt m => m Int
   69 freshRef = state $ \s ->
   70   let n = fpNextRef s in (n, s { fpNextRef = n + 1 })
   71 
   72 rememberTerm :: MonadState FPSt m => Term -> Int -> m ()
   73 rememberTerm t n = modify' $
   74   \s -> s { fpTermMemo = IntMap.insert (termIndex t) n $ fpTermMemo s }
   75 
   76 lookupTermRef :: MonadState FPSt m => Term -> m (Maybe Int)
   77 lookupTermRef t = IntMap.lookup (termIndex t) <$> gets fpTermMemo
   78 
   79 rememberDef :: MonadState FPSt m => Name -> Int -> m ()
   80 rememberDef nm n = modify' $
   81   \s -> s { fpDefMemo = IntMap.insert (nameIndex nm) n $ fpDefMemo s }
   82 
   83 lookupDefRef :: MonadState FPSt m => Name -> m (Bool, Maybe Int)
   84 lookupDefRef nm = do
   85   visiting <- IntSet.member (nameIndex nm) <$> gets fpDefVisiting
   86   def      <- IntMap.lookup (nameIndex nm) <$> gets fpDefMemo
   87   pure (visiting, def)
   88 
   89 visit :: MonadState FPSt m => Name -> m ()
   90 visit nm = modify' $
   91   \s -> s { fpDefVisiting = IntSet.insert (nameIndex nm) $ fpDefVisiting s }
   92 
   93 unvisit :: MonadState FPSt m => Name -> m ()
   94 unvisit nm = modify' $
   95   \s -> s { fpDefVisiting = IntSet.delete (nameIndex nm) $ fpDefVisiting s }
   96 
   97 lookupName :: MonadReader FPCfg m => Name -> m (Bool, Maybe ResolvedName)
   98 lookupName nm = do
   99   cfg <- ask
  100   pure ( Set.member (nameIndex nm) $ fpUninterp cfg
  101        , lookupVarIndexInMap (nameIndex nm) $ fpModuleMap cfg
  102        )

  104 byte :: Word8 -> BS.Builder
  105 byte = BS.word8
  106 
  107 -- | Markers used to preserve structural sharing without hashing.
  108 tagRef, tagDef, tagCycle :: Word8
  109 tagRef   = 0xF1  -- Reference to a previously serialized term or definition.
  110 tagDef   = 0xF2  -- Introduces a new sequence number followed by its body.
  111 tagCycle = 0xF3  -- Placeholder emitted when a definition cycle is detected.
  112 
  113 -- | Basic serialization of small morsels.
  114 class Bytes a where
  115   bytes :: a -> BS.Builder
  116 instance Bytes Int where
  117   bytes = BS.int64BE . fromIntegral
  118 instance Bytes Natural where
  119   bytes = BS.lazyByteString . encode
  120 instance Bytes Text where
  121   bytes txt = bytes (BS.length bs) <> BS.byteString bs
  122     where
  123       bs = encodeUtf8 txt
  124 instance Bytes NameInfo where
  125   bytes = \case
  126     ModuleIdentifier i -> byte 0x1 <> bytes (QN.ppQualName $ moduleIdentToQualName i)
  127     ImportedName qn _  -> byte 0x2 <> bytes (QN.ppQualName qn)
  128 instance Bytes Sort where
  129   bytes = \case
  130     PropSort   -> byte 0x1
  131     TypeSort n -> byte 0x2 <> bytes n
  132 instance Bytes SortFlags where
  133   bytes sf = b (flagInhabited sf) <> b (flagQuantType sf)
  134     where b = byte . fromIntegral . fromEnum
  135 instance Bytes FirstOrderType where
  136   bytes = \case
  137     FOTBit       -> byte 0x1
  138     FOTInt       -> byte 0x2
  139     FOTIntMod n  -> byte 0x3 <> bytes n
  140     FOTVec n t   -> byte 0x4 <> bytes n <> bytes t
  141     FOTArray k v -> byte 0x5 <> bytes k <> bytes v
  142     FOTTuple ts  -> byte 0x6 <> bytes (length ts)
  143                              <> foldMap bytes ts
  144     FOTRec m     -> byte 0x7 <> bytes (Map.size m)
  145                              <> foldMap (\(k, v) -> bytes k <> byte 0 <> bytes v)
  146                                         (Map.toAscList m)
  147 
  148 fpTerms :: (MonadState FPSt m, MonadReader FPCfg m) => VarCtx -> [Term] -> m BS.Builder
  149 fpTerms ctx ts = do
  150   bs <- mapM (fpTerm ctx) ts
  151   pure $ bytes (length ts) <> mconcat bs
  152 
  153 fpTerm :: (MonadState FPSt m, MonadReader FPCfg m) => VarCtx -> Term -> m BS.Builder
  154 fpTerm ctx t = lookupTermRef t >>= \case
  155   Just n  -> pure $ byte tagRef <> bytes n
  156   Nothing -> do
  157     n <- freshRef
  158     rememberTerm t n
  159     body <- fpTermF ctx (unwrapTermF t)
  160     pure $ byte tagDef <> bytes n <> body
  161 
  162 fpTermF :: (MonadState FPSt m, MonadReader FPCfg m) => VarCtx -> TermF Term -> m BS.Builder
  163 fpTermF ctx = \case
  164 
  165   App e1 e2 -> do
  166     b1 <- fpTerm ctx e1
  167     b2 <- fpTerm ctx e2
  168     pure $ byte 0x1 <> b1 <> b2
  169 
  170   Lambda x ty body -> do
  171     bty   <- fpTerm ctx ty
  172     bbody <- fpTerm (consVarCtx x ctx) body
  173     pure $ byte 0x2 <> bty <> bbody
  174 
  175   Pi x ty body -> do
  176     bty   <- fpTerm ctx ty
  177     bbody <- fpTerm (consVarCtx x ctx) body
  178     pure $ byte 0x3 <> bty <> bbody
  179 
  180   Constant nm -> do
  181     bdef <- fpDef nm
  182     pure $ byte 0x4 <> bdef
  183 
  184   Variable vn ty
  185     | Just dbi <- lookupVarCtx vn ctx ->      -- Bound var: use index.
  186       pure $ byte 0x5 <> bytes dbi
  187     | otherwise -> do                         -- Free var: use name.
  188       bty <- fpTerm ctx ty
  189       pure $ byte 0x6 <> bytes (vnName vn) <> bty
  190 
  191   FTermF (Sort s flags) ->
  192     pure $ byte 0x7 <> bytes s <> bytes flags
  193 
  194   FTermF (StringLit txt) ->
  195     pure $ byte 0x8 <> bytes txt
  196 
  197   FTermF (ArrayValue elemTy elems) -> do
  198     bty <- fpTerm ctx elemTy
  199     bs  <- fpTerms ctx (V.toList elems)
  200     pure $ byte 0x9 <> bty <> bs
  201 
  202   FTermF (Recursor cr) -> do
  203     let dtNi    = bytes $ nameInfo $ recursorDataType cr
  204         ctorNis = foldMap (bytes . nameInfo) $ recursorCtorOrder cr
  205     dtB <- fpDef (recursorDataType cr)
  206     pure $ byte 0xa
  207         <> dtNi
  208         <> bytes (recursorSort cr)
  209         <> bytes (recursorNumParams cr)
  210         <> bytes (recursorNumIxs cr)
  211         <> bytes (length $ recursorCtorOrder cr)
  212         <> ctorNis
  213         <> dtB
  214 
  215 fpDef :: (MonadState FPSt m, MonadReader FPCfg m) => Name -> m BS.Builder
  216 fpDef nm = lookupDefRef nm >>= \case
  217   (_,    Just n)  -> pure $ byte tagRef <> bytes n
  218   (True, Nothing) -> pure $ byte tagCycle <> ni
  219   (False, Nothing) -> do
  220     visit nm
  221     body <- lookupName nm >>= \case
  222       (True, Just (ResolvedDef d)) -> do -- Uninterpreted.
  223         bty <- fpTerm emptyVarCtx $ defType d
  224         pure $ byte 0x11 <> ni <> bty
  225       (_, Just (ResolvedDef d))
  226         | Just b <- defBody d -> do
  227           bty   <- fpTerm emptyVarCtx $ defType d
  228           bbody <- fpTerm emptyVarCtx b
  229           pure $ byte 0x13 <> ni <> bty <> bbody
  230       (_, Just (ResolvedDef d)) -> do
  231         bty <- fpTerm emptyVarCtx $ defType d
  232         pure $ byte 0x12 <> ni <> bty
  233       (_, Just (ResolvedDataType dt)) -> do
  234         bty    <- fpTerm emptyVarCtx $ dtType dt
  235         bctors <- fpTerms emptyVarCtx $ ctorType <$> dtCtors dt
  236         pure $ byte 0x14 <> ni <> bty <> bytes (dtSort dt) <> bctors
  237       (_, Just (ResolvedCtor c)) -> do
  238         bty <- fpTerm emptyVarCtx $ ctorType c
  239         pure $ byte 0x15 <> ni <> bty
  240       (_, Nothing) -> pure $ byte 0x16 <> ni
  241     unvisit nm
  242     n <- freshRef
  243     rememberDef nm n
  244     pure $ byte tagDef <> bytes n <> body
  245   where
  246     ni :: BS.Builder
  247     ni = bytes $ nameInfo nm

  249 fpAssert :: (MonadState FPSt m, MonadReader FPCfg m) => VarCtx -> SATAssert -> m BS.Builder
  250 fpAssert ctx = \case
  251   BoolAssert tm -> do
  252     b <- fpTerm ctx tm
  253     pure $ byte 0x21 <> b
  254   UniversalAssert vars hyps concl -> do
  255     let varTys = foldMap (bytes . snd) vars
  256         ctx'   = foldl' (\c (v, _) -> consVarCtx v c) ctx vars
  257     bhyps  <- fpTerms ctx' hyps
  258     bconcl <- fpTerm ctx' concl
  259     pure $ byte 0x22
  260         <> bytes (length vars)
  261         <> varTys
  262         <> bhyps
  263         <> bconcl

  265 -- | Produce a lazy-ByteString fingerprint of a SATQuery, intended as input to
  266 --   a hash function to form a solver cache key. This serialization should be:
  267 --   * Stable across runs (has no VarIndex/TermIndex dependence).
  268 --   * Alpha-equivalence-respecting (uses de Bruijn indices for bound variables).
  269 --   * Definition-sensitive (transitively serializes constant bodies).
  270 --   * Relatively fast.
  271 fingerprintSATQuery :: ModuleMap -> SATQuery -> ByteString
  272 fingerprintSATQuery mm satq =
  273   BS.toLazyByteString $ evalState (runReaderT satq' cfg) st0
  274   where
  275     cfg :: FPCfg
  276     cfg = FPCfg mm $ satUninterp satq
  277 
  278     st0 :: FPSt
  279     st0 = FPSt mempty mempty mempty 0
  280 
  281     vars :: [(VarName, FirstOrderType)]
  282     vars = Map.toAscList $ satVariables satq
  283 
  284     varTys :: BS.Builder
  285     varTys = foldMap (bytes . snd) vars
  286 
  287     queryCtx :: VarCtx
  288     queryCtx = foldl' (\c (v, _) -> consVarCtx v c) emptyVarCtx vars
  289 
  290     -- Note: asserts are not sorted here, so re-ordering the same
  291     -- asserts will produce a different fingerprint.
  292     satq' :: (MonadState FPSt m, MonadReader FPCfg m) => m BS.Builder
  293     satq' = do
  294       asserts <- mapM (fpAssert queryCtx) $ satAsserts satq
  295       pure $ bytes (length vars) <> varTys <> mconcat asserts
