  105 mutIso :: Iso' M.Mutability Bool
  106 mutIso =
  107   iso
  108     (\case
  109       M.Mut -> True
  110       M.Immut -> False)
  111     (\case
  112       True -> M.Mut
  113       False -> M.Immut)

  120 initialDefCrucibleMethodSpecIR ::
  121   CollectionState ->
  122   M.Fn ->
  123   ProgramLoc ->
  124   MS.CrucibleMethodSpecIR MIR
  125 initialDefCrucibleMethodSpecIR rm fn loc =
  126   let fname = fn ^. M.fname
  127       fsig = fn ^. M.fsig
  128       argTys = fsig ^. M.fsarg_tys
  129       retTy = case fsig ^. M.fsreturn_ty of
  130                 M.TyTuple [] -> Nothing
  131                 ty -> Just ty in
  132   MS.makeCrucibleMethodSpecIR fname argTys retTy loc rm
  133 
  134 initialCrucibleSetupState ::
  135   MIRCrucibleContext ->
  136   M.Fn ->
  137   ProgramLoc ->
  138   Setup.CrucibleSetupState MIR
  139 initialCrucibleSetupState cc fn loc =
  140   Setup.makeCrucibleSetupState () cc $
  141     initialDefCrucibleMethodSpecIR
  142       (cc ^. mccRustModule ^. rmCS)
  143       fn
  144       loc
