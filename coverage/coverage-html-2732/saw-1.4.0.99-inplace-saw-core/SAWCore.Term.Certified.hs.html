   65 --------------------------------------------------------------------------------
   66 -- * Certified typed terms
   67 
   68 -- | An abstract datatype pairing a well-formed 'Raw.Term' with its type.
   69 data Term =
   70   Term
   71   Raw.Term -- ^ value
   72   Raw.Term -- ^ type
   73   (IntMap Raw.Term) -- ^ typing context
   74 
   75 -- | The raw term of a 'Term'.
   76 rawTerm :: Term -> Raw.Term
   77 rawTerm (Term trm _ _) = trm
   78 
   79 -- | The type of a 'Term' as a raw term.
   80 rawType :: Term -> Raw.Term
   81 rawType (Term _ typ _) = typ
   82 
   83 -- | The typing context of a 'Term', keyed by the 'VarIndex' of each
   84 -- 'VarName' in the term.
   85 rawCtx :: Term -> IntMap Raw.Term
   86 rawCtx (Term _ _ scx) = scx
   87 
   88 -- | Reduce the given 'Raw.Term' to WHNF, using all reductions allowed by
   89 -- the SAWCore type system.
   90 scTypeCheckWHNF :: SharedContext -> Raw.Term -> IO Raw.Term
   91 scTypeCheckWHNF sc t =
   92   do (_, t') <- rewriteSharedTerm sc (addConvs natConversions emptySimpset :: Simpset ()) t
   93      Raw.scWhnf sc t'
   94 
   95 -- | Check if two terms are "convertible for type-checking", meaning that they
   96 -- are convertible up to 'natConversions'.
   97 scTypeConvertible :: SharedContext -> Raw.Term -> Raw.Term -> IO Bool
   98 scTypeConvertible sc t1 t2 = Raw.scConvertibleEval sc scTypeCheckWHNF True t1 t2
   99 
  100 -- | Check whether one type is a subtype of another: Either they are
  101 -- convertible, or they are both Pi types with convertible argument
  102 -- types and result sorts @s1@ and @s2@ with @s1 <= s2@.
  103 scSubtype :: SharedContext -> Raw.Term -> Raw.Term -> IO Bool
  104 scSubtype sc t1 t2
  105   | alphaEquiv t1 t2 = pure True
  106   | otherwise =
  107     do t1' <- Raw.scWhnf sc t1
  108        t2' <- Raw.scWhnf sc t2
  109        case (t1', t2') of
  110          (asSort -> Just s1, asSort -> Just s2) ->
  111            pure (s1 <= s2)
  112          (unwrapTermF -> Pi x1 a1 b1, unwrapTermF -> Pi x2 a2 b2)
  113            | x1 == x2 ->
  114              (&&) <$> scTypeConvertible sc a1 a2 <*> scSubtype sc b1 b2
  115            | otherwise ->
  116              do conv1 <- scTypeConvertible sc a1 a2
  117                 var1 <- Raw.scVariable sc x1 a1
  118                 b2' <- Raw.scInstantiateExt sc (IntMap.singleton (vnIndex x2) var1) b2
  119                 conv2 <- scSubtype sc b1 b2'
  120                 pure (conv1 && conv2)
  121          _ ->
  122            scTypeConvertible sc t1' t2'

  127 -- | Compute the type of a 'Term'.
  128 scTypeOf :: SharedContext -> Term -> IO Term
  129 scTypeOf sc (Term _tm tp scx) =
  130   do tp_tp <- Raw.scTypeOf' sc scx tp
  131      -- Shrink typing context if possible
  132      let scx' = pruneContext (freeVars tp_tp) scx
  133      pure (Term tp tp_tp scx')
  134 
  135 -- | Reduce a 'Cterm' to WHNF (see also 'scTypeCheckWHNF').
  136 scWHNF :: SharedContext -> Term -> IO Term
  137 scWHNF sc (Term tm tp scx) =
  138   do tm' <- scTypeCheckWHNF sc tm
  139      pure (Term tm' tp scx)
  140 
  141 scGlobal :: SharedContext -> Ident -> IO Term
  142 scGlobal sc ident =
  143   do tm <- Raw.scGlobalDef sc ident
  144      tp <- Raw.scTypeOfIdent sc ident
  145      pure (Term tm tp IntMap.empty)

  148 -- * Building certified terms
  149 
  150 -- possible errors: not a pi type, bad argument type, context mismatch
  151 scApply :: SharedContext -> Term -> Term -> IO Term
  152 scApply sc f arg =
  153   do tm <- Raw.scApply sc (rawTerm f) (rawTerm arg)
  154      (vnIndex -> i, t1, t2) <- ensurePi sc (rawType f)
  155      ok <- scSubtype sc (rawType arg) t1
  156      unless ok $ fail $ unlines $
  157        ["Not a subtype", "expesced: " ++ showTerm t1, "got: " ++ showTerm (rawType arg)]
  158      tp <- Raw.scInstantiateExt sc (IntMap.singleton i (rawTerm arg)) t2
  159      scx <- unifyContexts "scApply" (rawCtx f) (rawCtx arg)
  160      pure (Term tm tp scx)
  161 
  162 -- possible errors: not a type, context mismatch, variable free in context
  163 scLambda :: SharedContext -> VarName -> Term -> Term -> IO Term
  164 scLambda sc x t body =
  165   do _s <- ensureSort sc (rawType t)
  166      tm <- Raw.scLambda sc x (rawTerm t) (rawTerm body)
  167      ensureNotFreeInContext x body
  168      _ <- unifyContexts "scLambda" (IntMap.singleton (vnIndex x) (rawTerm t)) (rawCtx body)
  169      scx <- unifyContexts "scLambda" (rawCtx t) (IntMap.delete (vnIndex x) (rawCtx body))
  170      tp <- Raw.scPi sc x (rawTerm t) (rawType body)
  171      pure (Term tm tp scx)
  172 
  173 -- possible errors: not a type, context mismatch, variable free in context
  174 scPi :: SharedContext -> VarName -> Term -> Term -> IO Term
  175 scPi sc x t body =
  176   do tm <- Raw.scPi sc x (rawTerm t) (rawTerm body)
  177      ensureNotFreeInContext x body
  178      _ <- unifyContexts "scTypedPi" (IntMap.singleton (vnIndex x) (rawTerm t)) (rawCtx body)
  179      scx <- unifyContexts "scTypedPi" (rawCtx t) (IntMap.delete (vnIndex x) (rawCtx body))
  180      s1 <- ensureSort sc (rawType t)
  181      s2 <- ensureSort sc (rawType body)
  182      tp <- Raw.scSort sc (piSort s1 s2)
  183      pure (Term tm tp scx)
  184 
  185 -- possible errors: not a type, context mismatch
  186 scFun :: SharedContext -> Term -> Term -> IO Term
  187 scFun sc a b =
  188   do tm <- Raw.scFun sc (rawTerm a) (rawTerm b)
  189      sa <- ensureSort sc (rawType a)
  190      sb <- ensureSort sc (rawType b)
  191      tp <- Raw.scSort sc (piSort sa sb)
  192      scx <- unifyContexts "scFun" (rawCtx a) (rawCtx b)
  193      pure (Term tm tp scx)

  232 -- possible errors: not a type
  233 scPairType :: SharedContext -> Term -> Term -> IO Term
  234 scPairType sc x y =
  235   do tm <- Raw.scPairType sc (rawTerm x) (rawTerm y)
  236      sx <- ensureSort sc (rawType x)
  237      sy <- ensureSort sc (rawType y)
  238      tp <- Raw.scSort sc (max sx sy)
  239      scx <- unifyContexts "scPairType" (rawCtx x) (rawCtx y)

  245   do tm <- Raw.scPairLeft sc (rawTerm x)
  246      tp <- fst <$> ensurePairType sc (rawType x)
  247      let scx = rawCtx x
  248      pure (Term tm tp scx)

  254      tp <- snd <$> ensurePairType sc (rawType x)
  255      let scx = rawCtx x

  258 -- possible errors: not a datatype, bad elimination sort
  259 scRecursor :: SharedContext -> Name -> Sort -> IO Term
  260 scRecursor sc nm s =
  261   do mm <- Raw.scGetModuleMap sc
  262      case lookupVarIndexInMap (nameIndex nm) mm of
  263        Just (ResolvedDataType dt) ->
  264          do unless (Raw.allowedElimSort dt s) $ fail "Disallowed propositional elimination"
  265             let d = dtName dt
  266             let nparams = length (dtParams dt)
  267             let nixs = length (dtIndices dt)
  268             let ctorOrder = map ctorName (dtCtors dt)
  269             let crec = CompiledRecursor d s nparams nixs ctorOrder
  270             tm <- Raw.scFlatTermF sc (Recursor crec)
  271             tp <- Raw.scRecursorType sc dt s
  272             pure (Term tm tp IntMap.empty)
  273        _ ->
  274          fail "datatype not found"
  275 
  276 -- possible errors: field not a type, context mismatch
  277 scRecordType :: SharedContext -> [(FieldName, Term)] -> IO Term
  278 scRecordType sc fields =
  279   do tm <- Raw.scRecordType sc (map (fmap rawTerm) fields)
  280      sorts <- traverse (ensureSort sc . rawType . snd) fields
  281      tp <- Raw.scSort sc (foldl max (mkSort 0) sorts)
  282      scx <- unifyContextList "scRecordType" (map (rawCtx . snd) fields)
  283      pure (Term tm tp scx)
  284 
  285 -- possible errors: duplicate field name
  286 scRecordValue :: SharedContext -> [(FieldName, Term)] -> IO Term
  287 scRecordValue sc fields =
  288   do tm <- Raw.scFlatTermF sc $ RecordValue (map (fmap rawTerm) fields)
  289      tp <- Raw.scRecordType sc (map (fmap rawType) fields)
  290      scx <- foldM (unifyContexts "scRecordValue") IntMap.empty (map (rawCtx . snd) fields)
  291      pure (Term tm tp scx)
  292 
  293 -- possible errors: not a record type, field name not found
  294 scRecordProj :: SharedContext -> Term -> FieldName -> IO Term
  295 scRecordProj sc t fname =
  296   do tm <- Raw.scRecordSelect sc (rawTerm t) fname
  297      let scx = rawCtx t
  298      tps <- ensureRecordType sc (rawType t)
  299      case Map.lookup fname tps of
  300        Nothing -> fail "scRecordProj: field name not found"
  301        Just tp -> pure (Term tm tp scx)

  303 -- no possible errors

  307 -- | A variant of 'scSort' that also takes a 'SortFlags' argument.

  342 -- | Prune a typing context by dropping indices unreachable from the
  343 -- given 'IntSet'.
  344 pruneContext :: IntSet -> IntMap Raw.Term -> IntMap Raw.Term
  345 pruneContext vs0 ctx = IntMap.restrictKeys ctx (reachable mempty vs0)
  346   where
  347     fvs = fmap freeVars ctx
  348     reachable old new
  349       | IntSet.null new = old
  350       | otherwise = reachable old' new'
  351           where old' = old <> new
  352                 new' = IntSet.difference (Fold.fold (IntMap.restrictKeys fvs new)) old'
  353 
  354 -- | Two typing contexts are unifiable if they agree perfescly on all
  355 -- entries where they overlap.
  356 unifyContexts :: String -> IntMap Raw.Term -> IntMap Raw.Term -> IO (IntMap Raw.Term)
  357 unifyContexts msg ctx1 ctx2 =
  358   do let check t1 t2 =
  359            unless (t1 == t2) $
  360            fail $ unlines ["unifyContexts: context mismatch", msg,
  361                            "t1: " ++ showTerm t1,
  362                            "t2: " ++ showTerm t2]
  363      sequence_ (IntMap.intersectionWith check ctx1 ctx2)
  364      pure (IntMap.union ctx1 ctx2)
  365 
  366 unifyContextList :: String -> [IntMap Raw.Term] -> IO (IntMap Raw.Term)
  367 unifyContextList msg = foldM (unifyContexts msg) IntMap.empty

  369 ensureRecognizer :: String -> SharedContext -> (Raw.Term -> Maybe a) -> Raw.Term -> IO a
  370 ensureRecognizer s sc f trm =
  371   case f trm of
  372     Just a -> pure a
  373     Nothing ->
  374       do trm' <- scTypeCheckWHNF sc trm
  375          case f trm' of
  376            Just a -> pure a
  377            Nothing ->
  378              fail $ "ensureRecognizer: Expected " ++ s ++ ", found: " ++ showTerm trm'
  379 
  380 ensureSort :: SharedContext -> Raw.Term -> IO Sort
  381 ensureSort sc tp = ensureRecognizer "Sort" sc asSort tp
  382 
  383 ensurePi :: SharedContext -> Raw.Term -> IO (VarName, Raw.Term, Raw.Term)
  384 ensurePi sc tp = ensureRecognizer "Pi" sc asPi tp
  385 
  386 ensurePairType :: SharedContext -> Raw.Term -> IO (Raw.Term, Raw.Term)
  387 ensurePairType sc tp = ensureRecognizer "PairType" sc asPairType tp
  388 
  389 ensureRecordType :: SharedContext -> Raw.Term -> IO (Map FieldName Raw.Term)
  390 ensureRecordType sc tp = ensureRecognizer "RecordType" sc asRecordType tp

  392 piSort :: Sort -> Sort -> Sort
  393 piSort s1 s2 = if s2 == propSort then propSort else max s1 s2

  395 -- | Check whether the given 'VarName' occurs free in the type of
  396 -- another variable in the context of the given 'Term', and fail if it
  397 -- does.
  398 ensureNotFreeInContext :: VarName -> Term -> IO ()
  399 ensureNotFreeInContext x body =
  400   when (any (IntSet.member (vnIndex x) . freeVars) (rawCtx body)) $
  401     fail $ "Variable occurs free in context: " ++ show (vnName x)
