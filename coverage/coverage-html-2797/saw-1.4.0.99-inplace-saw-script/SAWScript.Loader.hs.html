   44 -- | Type shorthand for an operation that can return warnings and/or
   45 --   errors. On error it returns Left and a list of messages, at least
   46 --   one of which is an error and therefore fatal. On success, it
   47 --   returns a (possibly empty) list of messages, which are all
   48 --   warnings and not fatal, and a result of type a.
   49 type WithMsgs a = Either [Text] ([Text], a)
   50 
   51 -- | The messages produced by the typechecker are pairs of position and
   52 --   String. Convert to Text. XXX: the error infrastructure is supposed
   53 --   to be what knows how to print source positions.
   54 convertTypeMsg :: (Pos, String) -> Text
   55 convertTypeMsg (pos, msg) =
   56     let pos' = Text.pack (show pos)
   57         msg' = Text.pack msg
   58     in
   59     pos' <> ": " <> msg'
   60 
   61 -- | Read some SAWScript text, using the selected parser entry point.
   62 --
   63 --   Returns only in Either; caller is responsible for handling the
   64 --   resulting warnings and errors.
   65 --
   66 --   The FilePath argument is the name of the file the text comes
   67 --   from. If the text didn't come from a file, it should have a
   68 --   suitable placeholder string; that's what ends up in the source
   69 --   positions in the results.
   70 --
   71 readAny :: FilePath -> Text.Text -> ([Token Pos] -> Either ParseError a) -> WithMsgs a
   72 readAny fileName str parser =
   73     -- XXX printing of positions properly belongs to the
   74     -- message-printing infrastructure.
   75     let positionMsg pos msg =
   76           Text.pack (show pos) <> ": " <> msg
   77     in
   78     case lexSAW fileName str of
   79         Left (_verbosity, pos, msg) ->
   80             Left [positionMsg pos msg]
   81         Right (tokens, optmsg) ->
   82             let msgs = case optmsg of
   83                   Nothing -> []
   84                   Just (Options.Error, _, _) ->
   85                       panic "readAny" ["Lexer returned an error in the warning slot"]
   86                   Just (_verbosity, pos, msg) ->
   87                       [positionMsg pos msg]
   88             in
   89             case parser tokens of
   90                 Left err ->
   91                     let err' = Text.pack (show err) in
   92                     Left (msgs ++ [err'])
   93                 Right tree ->
   94                     -- Note: there's no such things as warnings in a
   95                     -- Happy parser, so there are no more messages to
   96                     -- add in this branch.
   97                     Right (msgs, tree)

   99 -- | Use the readAny result to panic if any messages were generated.
  100 panicOnMsgs :: Text -> WithMsgs a -> a
  101 panicOnMsgs whoAmI result = case result of
  102    Left errs ->
  103        panic whoAmI ("Unexpected errors:" : errs)
  104    Right ([], tree) ->
  105        tree
  106    Right (warns, _) ->
  107        panic whoAmI ("Unexpected warnings:" : warns)
  108 
  109 -- | Handle the readAny result in IO.
  110 --
  111 --   Also return in Either, to be consistent with the current interface
  112 --   into this file. Prints any warnings, and return just errors or a
  113 --   result.
  114 --
  115 dispatchMsgs :: Options -> WithMsgs a -> IO (Either [Text] a)
  116 dispatchMsgs opts result =
  117     let printMsg vrb msg =
  118           Options.printOutLn opts vrb (Text.unpack msg)
  119     in
  120     case result of
  121         Left errs ->
  122             pure $ Left errs
  123         Right (msgs, tree) -> do
  124             mapM_ (printMsg Options.Warn) msgs
  125             pure $ Right tree

  127 -- | Read a type schema from a string. This is used to digest the type
  128 --   signatures for builtins, and the expansions for builtin typedefs.
  129 --
  130 -- Pure function that panics on any error or warning. This is
  131 -- appropriate since we use it only to handle the builtins; any
  132 -- glitches there are properly panics.
  133 --
  134 -- The first argument (fileName) is a string to pass as the
  135 -- filename for the lexer, which (complete with line and column
  136 -- numbering of dubious value) will go into the positions of the
  137 -- elements of the resulting type.
  138 --
  139 -- The fake file name names the builtin object involved, so we'll also
  140 -- use it as the panic location string if we need to panic.
  141 --
  142 -- FUTURE: we should figure out how to generate more meaningful
  143 -- positions (like "third argument of concat") but this at least
  144 -- allows telling the user which builtin the type came from.
  145 --
  146 readSchemaPure ::
  147     FilePath ->
  148     PrimitiveLifecycle ->
  149     Map Name (PrimitiveLifecycle, NamedType) ->
  150     Text ->
  151     Schema
  152 readSchemaPure fakeFileName lc tyenv str = do
  153     let result = readAny fakeFileName str parseSchema
  154     let result' = case result of
  155           Left errs -> Left errs
  156           Right (msgs, schema) ->
  157               let (errs_or_results, warns) = checkSchema lc tyenv schema
  158                   warns' = map convertTypeMsg warns
  159               in
  160               case errs_or_results of
  161                   Left errs -> Left (msgs ++ warns' ++ map convertTypeMsg errs)
  162                   Right () -> Right (msgs ++ warns', schema)
  163     panicOnMsgs (Text.pack fakeFileName) result'

  165 -- | Read a schema pattern from a string. This is used by the
  166 --   :search REPL command.
  167 --
  168 --   Also runs the typechecker to check the pattern.
  169 --
  170 readSchemaPattern ::
  171     FilePath -> Environ -> RebindableEnv -> Set PrimitiveLifecycle -> Text ->
  172     IO (Either [Text] SchemaPattern)
  173 readSchemaPattern fileName environ rbenv avail str = do
  174   -- XXX: this preserves the original behavior of ignoring the
  175   -- verbosity setting. We could expect the caller to pass in the
  176   -- options value to get the verbosity setting, and that's really
  177   -- the right thing to do; except that current plans are to get
  178   -- rid of that verbosity setting in the near future anyway.
  179   let opts = Options.defaultOptions
  180 
  181   let result = readAny fileName str parseSchemaPattern
  182   let result' = case result of
  183         Left errs -> Left errs
  184         Right (msgs, pat) ->
  185           let Environ varenv tyenv _cryenv = environ in
  186 
  187           -- XXX it should not be necessary to do this munging
  188           let squash (pos, lc, ty, _val, _doc) = (pos, lc, ReadOnlyVar, ty)
  189               varenv' = Map.map squash $ ScopedMap.flatten varenv
  190               tyenv' = ScopedMap.flatten tyenv
  191               rbsquash (pos, ty, _val) = (pos, Current, RebindableVar, ty)
  192               rbenv' = Map.map rbsquash rbenv
  193               varenv'' = Map.union varenv' rbenv'
  194           in
  195           let (errs_or_results, warns) = checkSchemaPattern avail varenv'' tyenv' pat
  196               warns' = map convertTypeMsg warns
  197           in
  198           case errs_or_results of
  199               Left errs -> Left (msgs ++ warns' ++ map convertTypeMsg errs)
  200               Right results -> Right (msgs ++ warns', results)
  201   dispatchMsgs opts result'

  203 -- | Read an expression from a string. This is used by the
  204 --   :type REPL command.
  205 readExpression :: FilePath -> Environ -> RebindableEnv -> Set PrimitiveLifecycle -> Text -> IO (Either [Text] (Schema, Expr))
  206 readExpression fileName environ rbenv avail str = do
  207   -- XXX as above
  208   let opts = Options.defaultOptions
  209 
  210   let result = readAny fileName str parseExpression
  211   let result' = case result of
  212         Left errs -> Left errs
  213         Right (msgs, expr) ->
  214            let Environ varenv tyenv _cryenvs = environ in
  215 
  216            -- XXX it should not be necessary to do this munging
  217            let squash (defpos, lc, ty, _val, _doc) = (defpos, lc, ReadOnlyVar, ty)
  218                varenv' = Map.map squash $ ScopedMap.flatten varenv
  219                tyenv' = ScopedMap.flatten tyenv
  220                rbsquash (defpos, ty, _val) = (defpos, Current, RebindableVar, ty)
  221                rbenv' = Map.map rbsquash rbenv
  222                varenv'' = Map.union varenv' rbenv'
  223            in
  224            -- XXX: also it shouldn't be necessary to do this wrappery
  225            let pos = getPos expr
  226                decl = Decl pos (PWild pos Nothing) Nothing expr
  227            in
  228            let (errs_or_results, warns) = checkDecl avail varenv'' tyenv' decl
  229                warns' = map convertTypeMsg warns
  230            in
  231            case errs_or_results of
  232                Left errs -> Left (msgs ++ warns' ++ map convertTypeMsg errs)
  233                Right decl' ->
  234                    let expr' = dDef decl'
  235                        schema = case dType decl' of
  236                          Just sch -> sch
  237                          Nothing ->
  238                              -- If the typechecker didn't insert a type, it's bust,
  239                              -- so panic. Not much point in printing the expression
  240                              -- or position in panic, since it's what the user just
  241                              -- typed.
  242                              panic "readExpressionChecked" [
  243                                  "Typechecker failed to produce a type"
  244                              ]
  245                    in
  246                    Right (msgs ++ warns', (schema, expr'))
  247   dispatchMsgs opts result'

  259 -- | Load the 'Stmt's in a @.saw@ file.
  260 --   Doesn't run the typechecker (yet).
  261 loadFileUnchecked :: Options -> FilePath -> IO (Either [Text] [Stmt])
  262 loadFileUnchecked opts fname = do
  263   Options.printOutLn opts Options.Info $ "Loading file " ++ show fname
  264   ftext <- TextIO.readFile fname
  265 
  266   let result = readAny fname ftext parseModule
  267   dispatchMsgs opts result

  269 -- | Find a file, potentially looking in a list of multiple search paths (as
  270 -- specified via the @SAW_IMPORT_PATH@ environment variable or
  271 -- @-i@/@--import-path@ command-line options). If the file was successfully
  272 -- found, load it. If not, fail by returning `Left`.
  273 --
  274 -- Doesn't run the typechecker (yet).
  275 findAndLoadFileUnchecked :: Options -> FilePath -> IO (Either [Text] [Stmt])
  276 findAndLoadFileUnchecked opts fp = do
  277   let paths = Options.importPath opts
  278   mfname <- findFile paths fp
  279   case mfname of
  280     Nothing -> do
  281         let msgs = [
  282                 "Couldn't find file: " <> Text.pack fp,
  283                 "  Searched in directories:"
  284              ] ++ map (\p -> "    " <> Text.pack p) paths
  285         return $ Left msgs
  286     Just fname ->
  287       -- NB: Normalise the path name. The default SAW_IMPORT_PATH contains ".",
  288       -- and the behavior of filepath's 'normalise' function is to prepend a
  289       -- search path to the front of the file path that is found, which can
  290       -- cause paths like "./foo.saw" to be returned. This looks ugly in error
  291       -- messages, where we would rather display "foo.saw" instead.
  292       loadFileUnchecked opts (normalise fname)
