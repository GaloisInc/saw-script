   37 -- | Exception thrown by fatal errors.
   38 --
   39 -- The boolean argument is called @needTrace@; if it is true, a stack
   40 -- trace was requested but has not been printed. This is temporary
   41 -- scaffolding to work around things the `IO` instance of `SAWConsole`
   42 -- can't do properly. See `SAWCentral.Value`.
   43 --
   44 -- The `Show` instance is required to be an `Exception`; however,
   45 -- ideally, the `Show` text should never be seen. (It contains a real
   46 -- message rather than being "" to avoid confusion in the event it
   47 -- _does_ get printed at some point.)
   48 --
   49 -- This exception is thrown after an error message has already been
   50 -- posted (or more than one) and no further printing is needed. To
   51 -- accomplish this, it's caught at strategic points:
   52 --    - in the REPL, it rolls back the operation that was in progress
   53 --      and continues;
   54 --    - when executing SAWScript straight from a file, it just exits;
   55 --    - in the remote API it results in a failed action.
   56 --
   57 data Fatal = Fatal Bool
   58 instance Show Fatal where show (Fatal _) = "Encountered an error."
   59 instance Exception Fatal

   64 -- | Fetch the version info from `GitRev`.
   65 --   XXX: this is pasted from PanicSupport and should be shared.
   66 {-# Noinline unimplementedRevision #-}
   67 unimplementedRevision :: (Text, Text)
   68 unimplementedRevision =
   69     if GitRev.foundGit then
   70         let hash = fromMaybe "(unknown revision)" GitRev.hash 
   71             branch = fromMaybe "(unknown branch)" GitRev.branch
   72         in
   73         (Text.pack hash, Text.pack branch)
   74     else
   75         ("(VCS-less build)", "(VCS-less build)")
   76 
   77 -- | Generate the message to be used for "unimplemented", which is
   78 --   intentionally like what `panic` produces.
   79 unimplementedMessage :: Text -> Text -> [Text]
   80 unimplementedMessage loc msg =
   81     let (revision, branch) = unimplementedRevision in
   82     [
   83         "You have encountered a limitation in SAW.",
   84         "If you need this functionality, please create an issue at",
   85         "https://github.com/GaloisInc/saw-script/issues",
   86         "",
   87         "%< ---------------------------------------------------",
   88         "Revision:  " <> revision,
   89         "Branch:    " <> branch,
   90         "Location:  " <> loc,
   91         "Message:   " <> msg,
   92         "%< ---------------------------------------------------"
   93     ]

   96 -- | Backend for `errN` and `errDN` in `SAWConsole` in terms of a
   97 --   function `say` for issuing a single message. Note: the
   98 --   `SAWConsole` implementation is still responsible for bailing
   99 --   out or recording that there was an error, respectively.
  100 errN :: SayFunc m -> Text -> m ()
  101 errN say msg =
  102     say $ "Error: " <> msg
  103 
  104 -- | Backend for `errP`, `errDP`, `errX`, and `errDX` in `SAWConsole`
  105 --   in terms of a function `say` for issuing a single message. Note:
  106 --   the `SAWConsole` implementation is still responsible for bailing
  107 --   out or recording that there was an error, as appropriate.
  108 errP :: IsPosition pos => SayFunc m -> pos -> Text -> m ()
  109 errP say pos msg = do
  110     let pos' = ppPosition pos
  111     say $ pos' <> ": Error: " <> msg

  142 -- | Backend for `errN'` and `errDN'` in `SAWConsole` in terms of a
  143 --   function `say` for issuing a single message. Note: the
  144 --   `SAWConsole` implementation is still responsible for bailing out
  145 --   or recording that there was an error, respectively.
  146 --
  147 --   XXX: this will do to start with but ideally it should
  148 --      - account for the width of the "Error: " prefix when rendering
  149 --      - indent second and subsequent lines so long messages are visually
  150 --        distinct
  151 --
  152 --   Alternatively it's possible that we should just have a SayFunc'
  153 --   type that takes a Doc and leave all that to the implementation,
  154 --   which might want to, for example, accumulate all errors as Doc
  155 --   rather then Text.
  156 --
  157 errN' :: SayFunc m -> PPS.Opts -> PPS.Doc -> m ()
  158 errN' say opts msg = errN say $ PPS.renderText opts msg
