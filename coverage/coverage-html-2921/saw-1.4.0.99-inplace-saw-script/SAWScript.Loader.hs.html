   50 -- | Type shorthand for an operation that can return warnings and/or
   51 --   errors. On error it returns Left and a list of messages, at least
   52 --   one of which is an error and therefore fatal. On success, it
   53 --   returns a (possibly empty) list of messages, which are all
   54 --   warnings and not fatal, and a result of type a.
   55 --
   56 --   The position is a Maybe because errors from the parser come back
   57 --   with a position already stuffed into the message text, and we would
   58 --   like to not print another vacuous one along with it.
   59 type WithMsgs a = Either [(Maybe Pos, Text)] ([(Maybe Pos, Text)], a)
   60 
   61 -- | Type shorthand for an include path.
   62 --
   63 --   The LHS is the directory the current file was included from
   64 --   (starts as ".") and the RHS is the user's search path.
   65 type IncludePath = (FilePath, [FilePath])

   67 -- | Type shorthand for the set of includes already seen.
   68 --
   69 --   We pass it around as an IORef because it needs to accumulate
   70 --   as we go.
   71 type SeenSet = IORef (Set FilePath)

   73 -- | Construct an empty SeenSet.
   74 emptySeenSet :: IO SeenSet
   75 emptySeenSet = IORef.newIORef Set.empty

   77 -- | Check for membership in a SeenSet.
   78 seenSetMember :: FilePath -> SeenSet -> IO Bool
   79 seenSetMember x ref = do
   80     set <- IORef.readIORef ref
   81     pure $ Set.member x set
   82 
   83 -- | Insert into a SeenSet.
   84 seenSetInsert :: FilePath -> SeenSet -> IO ()
   85 seenSetInsert x ref = do
   86     set <- IORef.readIORef ref
   87     IORef.writeIORef ref (Set.insert x set)
   88     pure ()

   92 --   See the notes in the interpreter for why we do this.

   94 --   FUTURE: if we keep this thing, it would be nice to attach the
   95 --   source position of the include to the push and pop statements.

   97 wrapDir :: FilePath -> IO [Stmt] -> IO [Stmt]
   98 wrapDir dir m = do
   99     stmts <- m
  100     -- XXX this is here to avoid reindenting, remove later
  101     pure $
  102                   let pos = PosInternal "pushd/popd derived from include" in
  103                   [StmtPushdir pos dir] ++ stmts ++ [StmtPopdir pos]

  105 -- | Read some SAWScript text, using the selected parser entry point.
  106 --
  107 --   Returns only in Either; caller is responsible for handling the
  108 --   resulting warnings and errors.
  109 --
  110 --   The FilePath argument is the name of the file the text comes
  111 --   from. If the text didn't come from a file, it should have a
  112 --   suitable placeholder string; that's what ends up in the source
  113 --   positions in the results.
  114 --
  115 readAny :: FilePath -> Text -> ([Token Pos] -> Either ParseError a) -> WithMsgs a
  116 readAny fileName str parser =
  117     case lexSAW fileName str of
  118         Left (_verbosity, pos, msg) ->
  119             Left [(Just pos, msg)]
  120         Right (tokens, optmsg) ->
  121             let msgs = case optmsg of
  122                   Nothing -> []
  123                   Just (Options.Error, _, _) ->
  124                       panic "readAny" ["Lexer returned an error in the warning slot"]
  125                   Just (_verbosity, pos, msg) ->
  126                       [(Just pos, msg)]
  127             in
  128             case parser tokens of
  129                 Left err ->
  130                     let err' = Text.pack (show err) in
  131                     Left (msgs ++ [(Nothing, err')])
  132                 Right tree ->
  133                     -- Note: there's no such things as warnings in a
  134                     -- Happy parser, so there are no more messages to
  135                     -- add in this branch.
  136                     Right (msgs, tree)
  137 
  138 -- | Use the readAny result to panic if any messages were generated.
  139 panicOnMsgs :: Text -> WithMsgs a -> a
  140 panicOnMsgs whoAmI result =
  141   -- Properly, printing the positions with the errors should be done
  142   -- by the error infrastructure. However, the error infrastructure
  143   -- necessarily needs to run in `IO`, and we need to _not_ run in
  144   -- `IO` here because we're part of initializing the builtins table.
  145   -- Do it by hand instead. I don't think it's worth adding extra
  146   -- stuff to SAWConsole just to avoid needing this code.
  147   let pp (mpos, msg) = case mpos of
  148         Nothing -> msg
  149         Just pos -> PosSupport.ppPosition pos <> ": " <> msg
  150   in
  151   case result of
  152    Left errs ->
  153        panic whoAmI ("Unexpected errors:" : map pp errs)
  154    Right ([], tree) ->
  155        tree
  156    Right (warns, _) ->
  157        panic whoAmI ("Unexpected warnings:" : map pp warns)
  158 
  159 -- | Like `panicOnMsgs` but for typechecker results. XXX: the
  160 --   typechecker should issue its own messages; if not, it at least
  161 --   shouldn't be arbitrarily different.
  162 panicOnMsgs' :: Text -> (Either [(Pos, String)] a, [(Pos, String)]) -> a
  163 panicOnMsgs' whoAmI (errs_or_results, warns) =
  164     let pp (pos, msg) =
  165           PosSupport.ppPosition pos <> ": " <> Text.pack msg
  166     in
  167     case warns of
  168         [] -> case errs_or_results of
  169             Left errs -> panic whoAmI ("Unexpected errors: " : map pp errs)
  170             Right result -> result
  171         _ -> panic whoAmI ("Unexpected warnings: " : map pp warns)

  173 -- | Handle the readAny result in IO.
  174 --
  175 --   Add HasCallStack because if the panic happens we'll want to know
  176 --   where we came from. XXX: figure out how to get rid of the panic
  177 --   and remove HasCallStack again.
  178 dispatchMsgs :: HasCallStack => WithMsgs a -> IO a
  179 dispatchMsgs result =
  180     case result of
  181         Left errs -> do
  182             -- FUTURE: might be possible to remove this glop with a
  183             -- sufficiently clever IsPosition instance for Maybe a.
  184             let pp (mpos, msg) = case mpos of
  185                   Nothing -> Cons.errDN msg
  186                   Just pos -> Cons.errDP pos msg
  187             mapM_ pp errs
  188             Cons.checkFail
  189             panic "dispatchMsgs" ["checkFail didn't fail"]
  190         Right (msgs, tree) -> do
  191             let pp (mpos, msg) = case mpos of
  192                   Nothing -> Cons.warnN msg
  193                   Just pos -> Cons.warnP pos msg
  194             mapM_ pp msgs
  195             pure tree
  196 
  197 -- | Handle a typechecker result in IO.
  198 --
  199 --   Add HasCallStack because if the panic happens we'll want to know
  200 --   where we came from. XXX: figure out how to get rid of the panic
  201 --   and remove HasCallStack again.
  202 dispatchMsgs' :: HasCallStack => (Either [(Pos, String)] a, [(Pos, String)]) -> IO a
  203 dispatchMsgs' (errs_or_result, warns) = do
  204     mapM_ (\(pos, msg) -> Cons.warnP pos $ Text.pack msg) warns
  205     case errs_or_result of
  206         Left errs -> do
  207             mapM_ (\(pos, msg) -> Cons.errDP pos $ Text.pack msg) errs
  208             Cons.checkFail
  209             panic "dispatchMsgs'" ["checkFail didn't fail"]
  210         Right tree ->
  211             pure tree

  213 -- | Call `readAny` then `panicOnMsgs`.
  214 readAnyPure :: FilePath -> Text -> ([Token Pos] -> Either ParseError a) -> Text -> a
  215 readAnyPure fileName str parser whoAmI =
  216     panicOnMsgs whoAmI $ readAny fileName str parser

  218 -- | Call `readAny` then `dispatchMsgs`.
  219 readAnyIO :: FilePath -> Text -> ([Token Pos] -> Either ParseError a) -> IO a
  220 readAnyIO fileName str parser =
  221     dispatchMsgs $ readAny fileName str parser

  223 -- | Run the readAny result through the `Include` module to resolve
  224 --   @include@ statements.
  225 --
  226 --   If we fail after collecting warnings, print the warnings before
  227 --   returning the failure.
  228 --
  229 resolveIncludes ::
  230     Int -> SeenSet -> IncludePath -> Options -> Inc.Processor a -> a -> IO a
  231 resolveIncludes depth seen incpath opts process tree =
  232     process (includeFile depth seen incpath opts) tree

  234 -- | Read a type schema from a string. This is used to digest the type
  235 --   signatures for builtins, and the expansions for builtin typedefs.
  236 --
  237 -- Pure function that panics on any error or warning. This is
  238 -- appropriate since we use it only to handle the builtins; any
  239 -- glitches there are properly panics.
  240 --
  241 -- The first argument (fileName) is a string to pass as the
  242 -- filename for the lexer, which (complete with line and column
  243 -- numbering of dubious value) will go into the positions of the
  244 -- elements of the resulting type.
  245 --
  246 -- The fake file name names the builtin object involved, so we'll also
  247 -- use it as the panic location string if we need to panic.
  248 --
  249 -- FUTURE: we should figure out how to generate more meaningful
  250 -- positions (like "third argument of concat") but this at least
  251 -- allows telling the user which builtin the type came from.
  252 --
  253 -- Note: there is no way to reach @include@ statements from type
  254 -- schemas, so no need to process includes in what we read.
  255 --
  256 readSchemaPure ::
  257     FilePath ->
  258     PrimitiveLifecycle ->
  259     ScopedMap Name (PrimitiveLifecycle, NamedType) ->
  260     Text ->
  261     Schema
  262 readSchemaPure fakeFileName lc tyenv str =
  263     let schema = readAnyPure fakeFileName str parseSchema (Text.pack fakeFileName) in
  264     panicOnMsgs' (Text.pack fakeFileName) $ checkSchema lc tyenv schema

  266 -- | Read a schema pattern from a string. This is used by the
  267 --   :search REPL command.
  268 --
  269 --   Also runs the typechecker to check the pattern.
  270 --
  271 -- Note: there is no way to reach @include@ statements from schema
  272 -- patterns, so no need to process includes in what we read.
  273 --
  274 readSchemaPattern ::
  275     Options ->
  276     FilePath -> Environ -> RebindableEnv -> Set PrimitiveLifecycle -> Text ->
  277     IO SchemaPattern
  278 readSchemaPattern _opts fileName environ rbenv avail str = do
  279   pat <- readAnyIO fileName str parseSchemaPattern
  280   -- XXX this is here to avoid reindenting, remove later
  281   do
  282           let Environ varenv tyenv _cryenv = environ
  283 
  284           -- XXX it should not be necessary to do this munging
  285           --
  286           -- Note that we need to flatten the scopes we have here
  287           -- (outside the typechecker) in order to have the union with
  288           -- the rebindable env to work right.
  289           let squash (pos, lc, ty, _val, _doc) = (pos, lc, ReadOnlyVar, ty)
  290               varenv' = Map.map squash $ ScopedMap.flatten varenv
  291               rbsquash (pos, ty, _val) = (pos, Current, RebindableVar, ty)
  292               rbenv' = Map.map rbsquash rbenv
  293               varenv'' = Map.union varenv' rbenv'
  294               varenv''' = ScopedMap.seed varenv''
  295 
  296           dispatchMsgs' $ checkSchemaPattern avail varenv''' tyenv pat

  298 -- | Read an expression from a string. This is used by the
  299 --   :type REPL command.
  300 --
  301 -- It is possible for someone to provide an expression that includes
  302 -- an @include@ in a do-block, so just in case call into `Include`
  303 -- to resolve any that appear.
  304 --
  305 readExpression ::
  306     Options ->
  307     FilePath -> Environ -> RebindableEnv -> Set PrimitiveLifecycle -> Text ->
  308     IO (Schema, Expr)
  309 readExpression opts fileName environ rbenv avail str = do
  310   seen <- emptySeenSet
  311   let incpath = (".", Options.importPath opts)
  312 
  313   expr0 <- readAnyIO fileName str parseExpression
  314   expr <- resolveIncludes 0{-depth-} seen incpath opts Inc.processExpr expr0
  315   -- XXX this is here to avoid reindenting, remove later
  316   do
  317            let Environ varenv tyenv _cryenvs = environ
  318 
  319            -- XXX it should not be necessary to do this munging
  320            let squash (defpos, lc, ty, _val, _doc) = (defpos, lc, ReadOnlyVar, ty)
  321                varenv' = Map.map squash $ ScopedMap.flatten varenv
  322                rbsquash (defpos, ty, _val) = (defpos, Current, RebindableVar, ty)
  323                rbenv' = Map.map rbsquash rbenv
  324                varenv'' = Map.union varenv' rbenv'
  325                varenv''' = ScopedMap.seed varenv''
  326 
  327            -- XXX: also it shouldn't be necessary to do this wrappery
  328            let pos = Pos.getPos expr
  329                decl = Decl pos (PWild pos Nothing) Nothing expr
  330 
  331            decl' <- dispatchMsgs' $ checkDecl avail varenv''' tyenv decl
  332 
  333            -- XXX this is here to avoid reindenting, remove later
  334            pure $
  335                    let expr' = dDef decl'
  336                        schema = case dType decl' of
  337                          Just sch -> sch
  338                          Nothing ->
  339                              -- If the typechecker didn't insert a type, it's bust,
  340                              -- so panic. Not much point in printing the expression
  341                              -- or position in panic, since it's what the user just
  342                              -- typed.
  343                              panic "readExpressionChecked" [
  344                                  "Typechecker failed to produce a type"
  345                              ]
  346                    in
  347                    (schema, expr')

  362 -- | Find a file, potentially looking in a list of multiple search paths (as
  363 -- specified via the @SAW_IMPORT_PATH@ environment variable or
  364 -- @-i@/@--import-path@ command-line options). If the file was successfully
  365 -- found, return the full path. If not, fail by returning `Left`.
  366 --
  367 locateFile :: IncludePath -> FilePath -> IO FilePath
  368 locateFile (current, rawdirs) file = do
  369   -- Replace "." in the search path with the directory the current
  370   -- file came from, which might or might not be ".".
  371   let dirs = map adjust rawdirs
  372       adjust "." = current
  373       adjust dir = dir
  374 
  375   mfname <- findFile dirs file
  376   case mfname of
  377     Nothing -> do
  378         -- XXX make this a pp-doc instead
  379         let msg = Text.unlines ([
  380                       "Couldn't find file: " <> Text.pack file,
  381                       "  Searched in directories:"
  382                   ] ++ map (\p -> "    " <> Text.pack p) dirs)
  383         -- XXX pass in the include position
  384         Cons.errN msg
  385     Just fname ->
  386       -- NB: Normalise the path name. The default SAW_IMPORT_PATH contains ".",
  387       -- and the behavior of filepath's 'normalise' function is to prepend a
  388       -- search path to the front of the file path that is found, which can
  389       -- cause paths like "./foo.saw" to be returned. This looks ugly in error
  390       -- messages, where we would rather display "foo.saw" instead.
  391       return $ normalise fname
  392 
  393 -- | Load the 'Stmt's in a @.saw@ file.
  394 --   Doesn't run the typechecker (yet).
  395 includeFile ::
  396     Int -> SeenSet -> IncludePath -> Options -> FilePath -> Bool -> IO [Stmt]
  397 includeFile depth seen incpath opts fname once = do
  398   fname' <- locateFile incpath fname
  399   -- XXX this is here to avoid reindenting, remove later
  400   do
  401       alreadySeen <- seenSetMember fname' seen 
  402       if depth > 128 then
  403           -- XXX pass in the include position as well
  404           Cons.errX $ "Maximum include depth exceeded"
  405       else if once && alreadySeen then do
  406           -- include_once and we've already seen this file
  407           Cons.noteN $ "Skipping already-included file \"" <>
  408                        Text.pack fname' <> "\""
  409           pure []
  410       else do
  411           seenSetInsert fname' seen
  412           let (_current, dirs) = incpath
  413               current' = takeDirectory fname'
  414               incpath' = (current', dirs)
  415 
  416           Cons.noteN $ "Loading file \"" <> Text.pack fname' <> "\""
  417           ftext <- TextIO.readFile fname'
  418 
  419           stmts <- wrapDir current' $ readAnyIO fname ftext parseModule
  420           resolveIncludes (depth + 1) seen incpath' opts Inc.processStmts stmts

  422 -- | Find a file, potentially looking in a list of multiple search paths (as
  423 -- specified via the @SAW_IMPORT_PATH@ environment variable or
  424 -- @-i@/@--import-path@ command-line options). If the file was successfully
  425 -- found, load it. If not, fail by returning `Left`.
  426 --
  427 -- Doesn't run the typechecker (yet).
  428 findAndLoadFileUnchecked :: Options -> FilePath -> IO [Stmt]
  429 findAndLoadFileUnchecked opts fp = do
  430   let depth = 0
  431   seen <- emptySeenSet
  432   let incpath = (".", Options.importPath opts)
  433       once = False
  434   includeFile depth seen incpath opts fp once
