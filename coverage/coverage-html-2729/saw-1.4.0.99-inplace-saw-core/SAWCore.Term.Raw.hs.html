   45 -- Term Datatype ---------------------------------------------------------------
   46 
   47 type TermIndex = Int -- Word64
   48 
   49 -- | For more information on the semantics of 'Term's, see the
   50 -- [manual](https://saw.galois.com/manual.html). 'Term' and 'TermF' are split
   51 -- into two structures to facilitate mutual structural recursion (sometimes
   52 -- referred to as the ["knot-tying"](https://wiki.haskell.org/Tying_the_Knot)
   53 -- pattern, sometimes referred to in terms of ["recursion
   54 -- schemes"](https://blog.sumtypeofway.com/posts/introduction-to-recursion-schemes.html))
   55 -- and term object reuse via hash-consing.
   56 data Term
   57   = STApp
   58     -- ^ This constructor \"wraps\" a 'TermF' 'Term', assigning it a
   59     -- guaranteed-unique integer identifier and caching its likely-unique hash.
   60     -- Most 'Term's are constructed via 'STApp'. When a fresh 'TermF' is evinced
   61     -- in the course of a SAW invocation and needs to be lifted into a 'Term',
   62     -- we can see if we've already created a 'Term' wrapper for an identical
   63     -- 'TermF', and reuse it if so. The implementation of hash-consed 'Term'
   64     -- construction exists in 'SAWCore.SharedTerm', in particular in the
   65     -- 'SAWCore.SharedTerm.scTermF' field of the
   66     -- t'SAWCore.SharedTerm.SharedContext' object.
   67      { stAppIndex    :: {-# UNPACK #-} !TermIndex
   68        -- ^ The UID associated with a 'Term'. It is guaranteed unique across a
   69        -- universe of properly-constructed 'Term's within a single SAW
   70        -- invocation.
   71      , stAppHash     :: {-# UNPACK #-} !Int
   72        -- ^ The hash, according to 'hash', of the 'stAppTermF' field associated
   73        -- with this 'Term'. This should be as unique as a hash can be, but is
   74        -- not guaranteed unique as 'stAppIndex' is.
   75      , stAppFreeVars :: !IntSet
   76        -- ^ A set containing the 'VarIndex' of each of the free named
   77        -- variables from 'Variable' constructors in the term.
   78      , stAppTermF    :: !(TermF Term)
   79        -- ^ The underlying 'TermF' that this 'Term' wraps. This field "ties the
   80        -- knot" of the 'Term'/'TermF' recursion scheme.
   81      }
   82   | Unshared !(TermF Term)
   83     -- ^ Used for constructing 'Term's that don't need to be shared/reused.
   84   deriving (Show, Typeable)
   85 
   86 instance Hashable Term where
   87   -- The hash of an 'STApp' depends on its not-necessarily-unique
   88   -- 'stAppHash' instead of its necessarily-unique 'stAppIndex'.
   89   -- The reason is that per #1830 (PR) and #1831 (issue), we want to
   90   -- to derive references to terms based solely on their shape.
   91   -- Indices have nothing to do with a term's shape - they're assigned
   92   -- sequentially when building terms, according to the (arbitrary)
   93   -- order in which a term is built.
   94   -- As for uniqueness, though hashing a term based on its subterms'
   95   -- hashes introduces less randomness/freshness, it maintains plenty,
   96   -- and provides benefits as described above.
   97   -- No code should ever rely on total uniqueness of hashes, and terms
   98   -- are no exception.
   99   --
  100   -- Note: Nevertheless, we do take some minor liberties with the
  101   -- contract of 'hashWithSalt'. The contract states that if two
  102   -- values are equal according to '(==)', then they must have the
  103   -- same hash.
  104   -- For terms constructed by/within SAW, this will hold, because
  105   -- SAW's handling of index generation and assignment ensures that
  106   -- equality of indices implies equality of terms and term hashes
  107   -- (see 'SAWCore.SharedTerm.getTerm').
  108   -- However, if terms are constructed outside this standard procedure
  109   -- or in a way that does not respect index uniqueness rules,
  110   -- 'hashWithSalt''s contract could be violated.
  111   hash STApp{ stAppHash = h } = h
  112   hash (Unshared t) = hash t
  113   hashWithSalt salt = hashWithSalt salt . hash
  114 
  115 instance Eq Term where
  116   (==) = equalTerm
  117 
  118 equalTerm :: Term -> Term -> Bool
  119 equalTerm (Unshared tf1) (Unshared tf2) = tf1 == tf2
  120 equalTerm Unshared{} STApp{} = False
  121 equalTerm STApp{} Unshared{} = False
  122 equalTerm (STApp{stAppIndex = i1, stAppHash = h1, stAppTermF = tf1})
  123           (STApp{stAppIndex = i2, stAppHash = h2, stAppTermF = tf2}) =
  124   i1 == i2 || (h1 == h2 && tf1 == tf2)
  125   -- The hash check (^) is merely an optimization that enables us to
  126   -- quickly return 'False' in most cases. Since we're assuming the
  127   -- contract of 'hashWithSalt' holds, then we know @tf1 == tf2@
  128   -- implies @h1 == h2@. Thus we could safely remove @h1 == h2@ without
  129   -- changing the behavior of this function, but keeping it in enables
  130   -- us to utilize the fact that we save 'STApp' hashes to get away
  131   -- with not traversing the 'stAppTermF' fields in most cases of
  132   -- inequality.
  133 
  134 -- | Return 'True' iff the given terms are equal modulo alpha equivalence (i.e.
  135 -- 'VarName's in 'Lambda' and 'Pi' expressions) and sharing (i.e. 'STApp' vs.
  136 -- 'Unshared' expressions).
  137 alphaEquiv :: Term -> Term -> Bool
  138 alphaEquiv = term IntMap.empty
  139   where
  140     term :: IntMap VarIndex -> Term -> Term -> Bool
  141     term vm (Unshared tf1) (Unshared tf2) = termf vm tf1 tf2
  142     term vm (Unshared tf1) (STApp{stAppTermF = tf2}) = termf vm tf1 tf2
  143     term vm (STApp{stAppTermF = tf1}) (Unshared tf2) = termf vm tf1 tf2
  144     term vm
  145       (STApp{stAppIndex = i1, stAppTermF = tf1, stAppFreeVars = vs1})
  146       (STApp{stAppIndex = i2, stAppTermF = tf2}) =
  147       (IntSet.disjoint vs1 (IntMap.keysSet vm) && i1 == i2) || termf vm tf1 tf2
  148 
  149     termf :: IntMap VarIndex -> TermF Term -> TermF Term -> Bool
  150     termf vm (FTermF ftf1) (FTermF ftf2) = ftermf vm ftf1 ftf2
  151     termf vm (App t1 u1) (App t2 u2) = term vm t1 t2 && term vm u1 u2
  152     termf vm (Lambda (vnIndex -> i1) t1 u1) (Lambda (vnIndex -> i2) t2 u2) =
  153       let vm' = if i1 == i2 then vm else IntMap.insert i1 i2 vm
  154       in term vm t1 t2 && term vm' u1 u2
  155     termf vm (Pi (vnIndex -> i1) t1 u1) (Pi (vnIndex -> i2) t2 u2) =
  156       let vm' = if i1 == i2 then vm else IntMap.insert i1 i2 vm
  157       in term vm t1 t2 && term vm' u1 u2
  158     termf _vm (Constant x1) (Constant x2) = x1 == x2
  159     termf vm (Variable x1 _t1) (Variable x2 _t2) =
  160       case IntMap.lookup (vnIndex x1) vm of
  161         Just i -> vnIndex x2 == i
  162         Nothing -> x1 == x2
  163     termf _ FTermF{}   _ = False
  164     termf _ App{}      _ = False
  165     termf _ Lambda{}   _ = False
  166     termf _ Pi{}       _ = False
  167     termf _ Constant{} _ = False
  168     termf _ Variable{} _ = False
  169 
  170     ftermf :: IntMap Int -> FlatTermF Term -> FlatTermF Term -> Bool
  171     ftermf vm ftf1 ftf2 =
  172       case zipWithFlatTermF (term vm) ftf1 ftf2 of
  173         Nothing -> False
  174         Just ftf3 -> Foldable.and ftf3
  175 
  176 unwrapTermF :: Term -> TermF Term
  177 unwrapTermF STApp{stAppTermF = tf} = tf
  178 unwrapTermF (Unshared tf) = tf
  179 
  180 instance Ord Term where
  181   compare (STApp{stAppIndex = i}) (STApp{stAppIndex = j}) | i == j = EQ
  182   compare STApp{} Unshared{} = LT -- matches what we'd get from derived Ord instance
  183   compare Unshared{} STApp{} = GT
  184   compare x y = compare (unwrapTermF x) (unwrapTermF y)

  186 -- TermNet Patterns ------------------------------------------------------------
  187 
  188 instance Net.Pattern Term where
  189   toPat = termToPat
  190 
  191 termToPat :: Term -> Net.Pat
  192 termToPat t =
  193     case unwrapTermF t of
  194       Constant nm               -> Net.Atom (toShortName (nameInfo nm))
  195       App t1 t2                 -> Net.App (termToPat t1) (termToPat t2)
  196       FTermF (Sort s _)         -> Net.Atom (Text.pack ('*' : show s))
  197       FTermF (NatLit _)         -> Net.Var
  198       _                         -> Net.Var

  200 -- Free Variables --------------------------------------------------------------
  201 
  202 -- | Return an 'IntSet' containing the 'VarIndex' of all free
  203 -- variables in the 'Term'.
  204 freeVars :: Term -> IntSet
  205 freeVars STApp{ stAppFreeVars = s } = s
  206 freeVars (Unshared tf) = freesTermF (fmap freeVars tf)
  207 
  208 -- | Test whether a 'Term' is closed, i.e., it has no free variables.
  209 closedTerm :: Term -> Bool
  210 closedTerm t = IntSet.null (freeVars t)
