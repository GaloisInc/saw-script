   43 llvmDisCmdHelp =
   44   "show LLVM disassembly for LLVMModule plus: symbol, file:line, or !N metadata"

   46 llvmDisCmd :: Text -> Text -> REPL ()
   47 llvmDisCmd modref ref
   48 
   49   | Text.null modref || Text.null ref =
   50     liftIO $ putStrLn usage
   51 
   52   | Just ('!', linestr) <- Text.uncons ref
   53   , Just mdId <- readMaybe (drop 1 $ Text.unpack linestr) :: Maybe Int =
   54       getModule modref >>= \case
   55       Right llvmMod ->
   56         case find ((mdId ==) . umIndex) $ modUnnamedMd llvmMod of
   57           Just um -> disp $ ppLLVM llvmVlatest $ ppUnnamedMd um
   58           Nothing -> err $ "Could not find metadata with ID " <> show mdId
   59       Left e -> err e
   60 
   61   | (file, linestr) <- Text.span (/= ':') ref
   62   , not (Text.null file)
   63   , Just line <- readMaybe (drop 1 $ Text.unpack linestr) :: Maybe Integer =
   64       getModule modref >>= \case
   65       Right llvmMod -> disp
   66                        $ ppLLVM llvmVlatest
   67                        $ ppModuleAtLine (Text.unpack file) line llvmMod
   68       Left e -> err e
   69 
   70   | otherwise =
   71     getModule modref >>= \case
   72       Right llvmMod ->
   73         do let dname = fromString $ Text.unpack ref
   74            case find ((dname ==) . defName) $ modDefines llvmMod of
   75              Just d -> disp $ ppLLVM llvmVlatest $ ppDefine d
   76              Nothing -> err $ Text.unpack ref <> " is not a function in this module"
   77       Left e -> err e
   78 
   79   where
   80     usage = unlines
   81             [ "The :llvmdis command requires an LLVMModule\
   82               \ followed by a symbol, a file:line, or !N argument."
   83               , ""
   84               , "For example:"
   85               , "   sawscript> bc <- llvm_load_module \"intTests/testmulti/foo.bc\""
   86               , "   sawscript> :llvmdis bc foo"
   87               , "   ... shows the foo function in LLVM textual format"
   88               , "   sawscript> :llvmdis bc foo.c:2"
   89               , "   ... shows just line 2 of the foo.bc file in LLVM textual format"
   90               , "   sawscript> :llvmdis bc !10"
   91               , "   ... shows the metadata at index 10"
   92             ]
   93     err msg = liftIO $ putStrLn $ "Error: " <> msg <> "\n\n  " <> usage <> "\n"
   94     disp = liftIO . putStrLn . PPPJ.render

   97 getModule :: SAW_AST.Name -> REPL (Either String Module)
   98 getModule nm = do
   99       rw <- getTopLevelRW
  100 
  101       let environ = rwEnviron rw
  102 
  103       let varEnv = eVarEnv environ
  104       case ScopedMap.lookup nm varEnv of
  105         Just (_pos, _life, _schema, val, _help) ->
  106           do case val of
  107                VLLVMModule someCMSMod -> return $ Right
  108                                          $ viewSome CMSLLVM.modAST someCMSMod
  109                _ -> return $ Left "Error: first argument must be an LLVMModule"
  110         Nothing -> return $ Left $ "Could not find " <> show nm
