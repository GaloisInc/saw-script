    8 module SAWScript.REPL (run, runFromFile) where
    9 
   10 import SAWScript.Options (Options(..))
   11 import SAWScript.REPL.Logo (displayLogo)
   12 import SAWScript.REPL.Haskeline (repl)
   13 
   14 -- | Main function
   15 run :: Options -> IO ()
   16 run opts = do
   17   displayLogo $ useColor opts
   18   repl Nothing opts (return ())
   19 
   20 -- | Alternate main function for batch mode
   21 runFromFile :: FilePath -> Options -> IO ()
   22 runFromFile file opts = do
   23   repl (Just file) opts (return ())
