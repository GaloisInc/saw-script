   69 -- | The type of state kept in the 'CrucibleSetup' monad
   70 data CrucibleSetupState ext =
   71   CrucibleSetupState
   72   { _csVarCounter      :: !MS.AllocIndex
   73   , _csPrePost         :: !MS.PrePost
   74   , _csResolvedState   :: MS.ResolvedState ext
   75   , _csMethodSpec      :: MS.CrucibleMethodSpecIR ext
   76   , _csCrucibleContext :: MS.CrucibleContext ext
   77   }

   82 makeCrucibleSetupRO =
   83   CrucibleSetupRO

   86 makeCrucibleSetupState ::
   87   MS.ResolvedState ext ->
   88   MS.CrucibleContext ext ->
   89   MS.CrucibleMethodSpecIR ext ->
   90   CrucibleSetupState ext
   91 makeCrucibleSetupState rs cc mspec =
   92   CrucibleSetupState
   93     { _csVarCounter      = MS.AllocIndex 0
   94     , _csPrePost         = MS.PreState
   95     , _csResolvedState   = rs
   96     , _csMethodSpec      = mspec
   97     , _csCrucibleContext = cc
   98     }

  109 currentState :: Lens' (CrucibleSetupState ext) (MS.StateSpec ext)
  110 currentState f x = case x^. csPrePost of
  111   MS.PreState  -> csMethodSpec (MS.csPreState f) x
  112   MS.PostState -> csMethodSpec (MS.csPostState f) x
  113 
  114 addPointsTo :: Monad m => MS.PointsTo ext -> CrucibleSetupT ext m ()
  115 addPointsTo pt = currentState . MS.csPointsTos %= (pt : )
  116 
  117 addAllocGlobal :: Monad m => MS.AllocGlobal ext -> CrucibleSetupT ext m ()
  118 addAllocGlobal ag = csMethodSpec . MS.csGlobalAllocs %= (ag : )
  119 
  120 addCondition :: Monad m => MS.SetupCondition ext -> CrucibleSetupT ext m ()
  121 addCondition cond = currentState . MS.csConditions %= (cond : )

  123 -- | Allocate a fresh variable in the form of a 'TypedExtCns' and
  124 -- record this allocation in the setup state.
  125 freshTypedExtCns ::
  126   MonadIO m =>
  127   SharedContext {- ^ shared context -} ->
  128   Text          {- ^ variable name  -} ->
  129   Cryptol.Type  {- ^ variable type  -} ->
  130   CrucibleSetupT arch m TypedExtCns
  131 freshTypedExtCns sc name cty =
  132   do ty <- liftIO $ Cryptol.importType sc Cryptol.emptyEnv cty
  133      ec <- liftIO $ scFreshEC sc name ty
  134      let tt = TypedExtCns cty ec
  135      currentState . MS.csFreshVars %= cons tt
  136      return tt
  137 
  138 -- | Allocate a fresh variable in the form of a 'TypedTerm' and record
  139 -- this allocation in the setup state.
  140 freshVariable ::
  141   MonadIO m =>
  142   SharedContext {- ^ shared context -} ->
  143   Text          {- ^ variable name  -} ->
  144   Cryptol.Type  {- ^ variable type  -} ->
  145   CrucibleSetupT arch m TypedTerm
  146 freshVariable sc name cty =
  147   do tec <- freshTypedExtCns sc name cty
  148      liftIO $ typedTermOfExtCns sc tec
  149 
  150 setupWithTag :: String -> CrucibleSetupT arch m a -> CrucibleSetupT arch m a
  151 setupWithTag tag m =
  152   withReaderT (croTags %~ Set.insert tag) m
