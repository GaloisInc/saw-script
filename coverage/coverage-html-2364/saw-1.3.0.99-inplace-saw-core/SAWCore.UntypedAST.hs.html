   65 data UTerm
   66   = Name (PosPair Text)
   67   | Sort Pos Sort SortFlags
   68   | App UTerm UTerm
   69   | Lambda Pos UTermCtx UTerm
   70   | Pi Pos UTermCtx UTerm
   71   | Recursor (Maybe ModuleName) (PosPair Text)
   72   | UnitValue Pos
   73   | UnitType Pos
   74     -- | New-style records
   75   | RecordValue Pos [(PosPair FieldName, UTerm)]
   76   | RecordType Pos [(PosPair FieldName, UTerm)]
   77   | RecordProj UTerm FieldName
   78     -- | Simple pairs
   79   | PairValue Pos UTerm UTerm
   80   | PairType Pos UTerm UTerm
   81   | PairLeft UTerm
   82   | PairRight UTerm
   83     -- | Identifies a type constraint on the term, i.e., a type ascription
   84   | TypeConstraint UTerm Pos UTerm
   85   | NatLit Pos Natural
   86   | StringLit Pos Text
   87     -- | Vector literal.
   88   | VecLit Pos [UTerm]
   89     -- | Bitvector literal.
   90   | BVLit Pos [Bool]
   91   | BadTerm Pos
   92  deriving (Show, TH.Lift)
   93 
   94 -- | A pattern used for matching a variable.
   95 data UTermVar
   96   = TermVar (PosPair LocalName)
   97   | UnusedVar Pos
   98   deriving (Eq, Ord, Show, TH.Lift)
   99 
  100 -- | Return the 'String' name associated with a 'UTermVar'
  101 termVarString :: UTermVar -> String
  102 termVarString (TermVar (PosPair _ str)) = Text.unpack str
  103 termVarString (UnusedVar _) = "_"
  104 
  105 -- | Return the 'LocalName' associated with a 'UTermVar'
  106 termVarLocalName :: UTermVar -> LocalName
  107 termVarLocalName (TermVar (PosPair _ str)) = str
  108 termVarLocalName (UnusedVar _) = "_"
  109 
  110 -- | A context of 0 or more variable bindings, with types
  111 type UTermCtx = [(UTermVar, UTerm)]
  112 
  113 instance Positioned UTerm where
  114   pos t =
  115     case t of
  116       Name i               -> pos i
  117       Sort p _ _           -> p
  118       Lambda p _ _         -> p
  119       App x _              -> pos x
  120       Pi p _ _             -> p
  121       Recursor _ i         -> pos i
  122       UnitValue p          -> p
  123       UnitType p           -> p
  124       RecordValue p _      -> p
  125       RecordType p _       -> p
  126       RecordProj x _       -> pos x
  127       PairValue p _ _      -> p
  128       PairType p _ _       -> p
  129       PairLeft x           -> pos x
  130       PairRight x          -> pos x
  131       TypeConstraint _ p _ -> p
  132       NatLit p _           -> p
  133       StringLit p _        -> p
  134       VecLit p _           -> p
  135       BVLit p _            -> p
  136       BadTerm p            -> p
  137 
  138 instance Positioned UTermVar where
  139   pos (TermVar i) = pos i
  140   pos (UnusedVar p) = p
  141 
  142 badTerm :: Pos -> UTerm
  143 badTerm = BadTerm

  145 -- | A constructor declaration of the form @c (x1 :: tp1) .. (xn :: tpn) :: tp@
  146 data CtorDecl = Ctor (PosPair Text) UTermCtx UTerm
  147   deriving (Show, TH.Lift)
  148 
  149 -- | A top-level declaration in a saw-core file
  150 data Decl
  151    = TypeDecl DefQualifier (PosPair Text) UTerm
  152      -- ^ A declaration of something having a type, where the declaration
  153      -- qualifier states what flavor of thing it is
  154    | DataDecl (PosPair Text) UTermCtx UTerm [CtorDecl]
  155      -- ^ A declaration of an inductive data types, with a name, a parameter
  156      -- context, a return type, and a list of constructor declarations
  157    | TermDef (PosPair Text) [UTermVar] UTerm
  158      -- ^ A declaration of a term having a definition, with variables
  159    | TypedDef (PosPair Text) [(UTermVar, UTerm)] UTerm UTerm
  160      -- ^ A definition of something with a specific type, with parameters
  161    | InjectCodeDecl Text Text
  162      -- ^ Some raw text to inject into a translation
  163   deriving (Show, TH.Lift)
  164 
  165 -- | A set of constraints on what 'String' names to import from a module
  166 data ImportConstraint
  167   = SpecificImports [String]
  168     -- ^ Only import the given names
  169   | HidingImports [String]
  170     -- ^ Import all but the given names
  171  deriving (Eq, Ord, Show, TH.Lift)
  172 
  173 -- | An import declaration
  174 data Import = Import { importModName :: PosPair ModuleName
  175                        -- ^ The name of the module to import
  176                      , importConstraints :: Maybe ImportConstraint
  177                        -- ^ The constraints on what to import
  178                      }
  179             deriving (Show, TH.Lift)
  180 
  181 -- | Test whether a 'String' name satisfies the constraints of an 'Import'
  182 nameSatsConstraint :: Maybe ImportConstraint -> String -> Bool
  183 nameSatsConstraint Nothing _ = True
  184 nameSatsConstraint (Just (SpecificImports ns)) n = elem n ns
  185 nameSatsConstraint (Just (HidingImports ns)) n = notElem n ns

  188 -- | A module declaration gives:
  189 -- * A name for the module;
  190 -- * A list of imports; AND
  191 -- * A list of top-level declarations
  192 data Module = Module (PosPair ModuleName) [Import] [Decl]
  193   deriving (Show, TH.Lift)
  194 
  195 moduleName :: Module -> ModuleName
  196 moduleName (Module (PosPair _ mnm) _ _) = mnm
  197 
  198 -- | Get a list of all names (i.e., definitions, axioms, or primitives) declared
  199 -- in a module, along with their types and qualifiers
  200 moduleTypedDecls :: Module -> [(Text, UTerm)]
  201 moduleTypedDecls (Module _ _ decls) = concatMap helper decls where
  202   helper :: Decl -> [(Text, UTerm)]
  203   helper (TypeDecl _ (PosPair _ nm) tm) = [(nm,tm)]
  204   helper _ = []
  205 
  206 -- | Get a list of all datatypes declared in a module
  207 moduleDataDecls :: Module -> [(Text, UTermCtx, UTerm, [CtorDecl])]
  208 moduleDataDecls (Module _ _ decls) = concatMap helper decls where
  209   helper :: Decl -> [(Text, UTermCtx, UTerm, [CtorDecl])]
  210   helper (DataDecl (PosPair _ nm) params tp ctors) = [(nm, params, tp, ctors)]
  211   helper _ = []
  212 
  213 moduleTypedDataDecls :: Module -> [(Text, UTerm)]
  214 moduleTypedDataDecls =
  215   map (\(nm,p_ctx,tp,_) ->
  216         (nm, Pi (pos tp) p_ctx tp)) . moduleDataDecls
  217 
  218 -- | Get a list of all constructors declared in a module, along with the context
  219 -- of parameters for each one
  220 moduleCtorDecls :: Module -> [(UTermCtx, CtorDecl)]
  221 moduleCtorDecls =
  222   concatMap (\(_,p_ctx,_,ctors) -> map (p_ctx,) ctors) . moduleDataDecls
  223 
  224 -- | Get a list of the names and types of all the constructors in a module
  225 moduleTypedCtorDecls :: Module -> [(Text, UTerm)]
  226 moduleTypedCtorDecls =
  227   concatMap (\(_,p_ctx,_,ctors) ->
  228               map (\(Ctor (PosPair _ nm) ctx tp) ->
  229                     (nm, Pi (pos tp) (p_ctx ++ ctx) tp)) ctors)
  230   . moduleDataDecls
  231 
  232 asPiList :: UTerm -> (UTermCtx, UTerm)
  233 asPiList (Pi _ ctx1 body1) =
  234   let (ctx2,body2) = asPiList body1 in
  235   (ctx1 ++ ctx2, body2)
  236 asPiList t = ([], t)
  237 
  238 asApp :: UTerm -> (UTerm, [UTerm])
  239 asApp = go []
  240   where go l (App t u)   = go (u:l) t
  241         go l t = (t,l)
  242 
  243 -- | Build a tuple value @(x1, .., xn)@.
  244 mkTupleValue :: Pos -> [UTerm] -> UTerm
  245 mkTupleValue p [] = UnitValue p
  246 mkTupleValue _ [x] = x
  247 mkTupleValue p (x:xs) = PairValue (pos x) x (mkTupleValue p xs)
  248 
  249 -- | Build a tuple type @#(x1, .., xn)@.
  250 mkTupleType :: Pos -> [UTerm] -> UTerm
  251 mkTupleType p [] = UnitType p
  252 mkTupleType _ [x] = x
  253 mkTupleType p (x:xs) = PairType (pos x) x (mkTupleType p xs)
  254 
  255 -- | Build a projection @t.i@ of a tuple. NOTE: This function does not
  256 -- work to access the last component in a tuple, since it always
  257 -- generates a @PairLeft@.
  258 mkTupleSelector :: UTerm -> Natural -> UTerm
  259 mkTupleSelector t i
  260   | i == 1    = PairLeft t
  261   | i > 1     = mkTupleSelector (PairRight t) (i - 1)
  262   | otherwise = error "mkTupleSelector: non-positive index"
