  189 -- | The initial override matching state starts with an empty substitution

  192   sym                           {- ^ simulator                      -} ->

  237 {-
  238 Note [oeConditionalPred]
  239 ~~~~~~~~~~~~~~~~~~~~~~~~
  240 oeConditionalPred is a predicate that is used to construct an implication for
  241 any assumption or assertion used in a specification. That is, oeConditionalPred
  242 can be thought of as the path condition for the part of the specification where
  243 the assumption/assertion is created. By default, the oeConditionalPred is
  244 simply `True` (see `initialEnv`), so when an assertion is created, e.g.,
  245 
  246   llvm_assert {{ x == 2*y }};
  247 
  248 Then the overall assertion would be `True ==> (x == 2*y)`. An implication with
  249 `True` as the premise is not very interesting, of course, but other parts of
  250 the program may add additional premises (see the `withConditionalPred`
  251 function).
  252 
  253 Currently, the only place in SAW where non-trivial `oeConditionalPred`s are
  254 added is when matching against an `llvm_conditional_points_to` statement. For
  255 instance, consider this spec (taken from #1945):
  256 
  257   let test_spec = do {
  258     p <- llvm_alloc (llvm_int 8);
  259     x <- llvm_fresh_var "x" (llvm_int 8);
  260     llvm_points_to p (llvm_term x);
  261 
  262     llvm_execute_func [p];
  263 
  264     llvm_conditional_points_to {{ x == 1 }} p (llvm_term {{ 1 : [8] }});
  265   };
  266 
  267 The `llvm_conditional_points_to` statement in the postcondition generates an
  268 assertion that checks `x` (the value that `p` points to) against `1 : [8]`. But
  269 we only want to check this under the assumption that `x` already equals `1` due
  270 to the `x == 1` part of the `llvm_conditional_points_to` statement. To do this,
  271 the implementation of `llvm_conditional_points_to` will add `x == 1` to the
  272 oeConditionalPred. This way, the assertion that gets generated will be:
  273 
  274   (x == 1 /\ True) ==> (x == 1)
  275 
  276 Here, leaving out the (x == 1) premise would be catastrophic, as that would
  277 result in the far more general assertion `True ==> (x == 1)`. (This was
  278 ultimately the cause of #1945.)
  279 -}

  284 data OverrideFailureReason ext
  285   = AmbiguousPointsTos [MS.PointsTo ext]
  286   | AmbiguousVars [TypedExtCns]
  287   | BadTermMatch Term Term -- ^ simulated and specified terms did not match
  288   | BadPointerCast -- ^ Pointer required to process points-to
  289   | BadReturnSpecification (Some Crucible.TypeRepr)
  290     -- ^ type mismatch in return specification
  291   | NonlinearPatternNotSupported
  292   | BadEqualityComparison -- ^ Comparison on an undef value
  293   | BadPointerLoad (Either (MS.PointsTo ext) (PP.Doc Void)) (PP.Doc Void)
  294     -- ^ @loadRaw@ failed due to type error
  295   | StructuralMismatch (PP.Doc Void) (PP.Doc Void) (Maybe (ExtType ext)) (ExtType ext)
  296     -- ^
  297     -- * pretty-printed simulated value
  298     -- * pretty-printed specified value
  299     -- * type of specified value
  300     -- * type of simulated value
  301 
  302 instance ( PP.Pretty (ExtType ext)
  303          , PP.Pretty (MS.PointsTo ext)
  304          ) => PP.Pretty (OverrideFailureReason ext) where
  305   pretty = ppOverrideFailureReason
  306 
  307 instance ( PP.Pretty (ExtType ext)
  308          , PP.Pretty (MS.PointsTo ext)
  309          ) => Show (OverrideFailureReason ext) where
  310   show = show . PP.pretty
  311 
  312 ppOverrideFailureReason ::
  313   ( PP.Pretty (ExtType ext)
  314   , PP.Pretty (MS.PointsTo ext)
  315   ) => OverrideFailureReason ext -> PP.Doc ann
  316 ppOverrideFailureReason rsn = case rsn of
  317   AmbiguousPointsTos pts ->
  318     PP.vcat
  319     [ PP.pretty "LHS of points-to assertion(s) not reachable via points-tos from inputs/outputs:"
  320     , PP.indent 2 $ PP.vcat (map PP.pretty pts)
  321     ]
  322   AmbiguousVars vs ->
  323     PP.vcat
  324     [ PP.pretty "Fresh variable(s) not reachable via points-tos from function inputs/outputs:"
  325     , PP.indent 2 $ PP.vcat (map MS.ppTypedExtCns vs)
  326     ]
  327   BadTermMatch x y ->
  328     PP.vcat
  329     [ PP.pretty "terms do not match"
  330     , PP.indent 2 (PP.unAnnotate (ppTerm PPS.defaultOpts x))
  331     , PP.indent 2 (PP.unAnnotate (ppTerm PPS.defaultOpts y))
  332     ]
  333   BadPointerCast ->
  334     PP.pretty "bad pointer cast"
  335   BadReturnSpecification ty ->
  336     PP.vcat
  337     [ PP.pretty "Spec had no return value, but the function returns a value of type:"
  338     , PP.viaShow ty
  339     ]
  340   NonlinearPatternNotSupported ->
  341     PP.pretty "nonlinear pattern not supported"
  342   BadEqualityComparison ->
  343     PP.pretty "value containing `undef` compared for equality"
  344   BadPointerLoad pointsTo msg ->
  345     PP.vcat
  346     [ PP.pretty "error when loading through pointer that" PP.<+>
  347       PP.pretty "appeared in the override's points-to precondition(s):"
  348     , PP.pretty "Precondition:"
  349     , PP.indent 2 (either PP.pretty PP.unAnnotate pointsTo)
  350     , PP.pretty "Failure reason: "
  351     , PP.indent 2 (PP.unAnnotate msg) -- this can be long
  352     ]
  353   StructuralMismatch simVal setupVal setupValTy ty ->
  354     PP.vcat
  355     [ PP.pretty "could not match specified value with actual value:"
  356     , PP.vcat (map (PP.indent 2) $
  357               [ PP.pretty "actual (simulator) value:" PP.<+> PP.unAnnotate simVal
  358               , PP.pretty "specified value:         " PP.<+> PP.unAnnotate setupVal
  359               , PP.pretty "type of actual value:   " PP.<+> PP.pretty ty
  360               ] ++ let msg ty_ =
  361                          [PP.pretty "type of specified value:"
  362                           PP.<+> PP.pretty ty_]
  363                    in maybe [] msg setupValTy)
  364     ]

  401 -- | The 'OverrideMatcher' type provides the operations that are needed
  402 -- to match a specification's arguments with the arguments provided by
  403 -- the Crucible simulation in order to compute the variable substitution
  404 -- and side-conditions needed to proceed.
  405 newtype OverrideMatcher' sym ext rorw m a =
  406   OM (ReaderT (OverrideEnv' sym) (StateT (OverrideState' sym ext) (ExceptT (OverrideFailure ext) m)) a)
  407   deriving (Applicative, Functor, Generic, Generic1, Monad, MonadIO, MonadThrow)
  408 
  409 type OverrideMatcher ext rorw a = OverrideMatcher' Sym ext rorw IO a
  410 
  411 instance Wrapped (OverrideMatcher' sym ext rorw m a) where
  412 
  413 deriving instance Monad m => MonadReader (OverrideEnv' sym) (OverrideMatcher' sym ext rorw m)
  414 deriving instance Monad m => MonadState (OverrideState' sym ext) (OverrideMatcher' sym ext rorw m)
  415 deriving instance Monad m => MonadError (OverrideFailure ext) (OverrideMatcher' sym ext rorw m)
  416 
  417 instance MonadTrans (OverrideMatcher' sym ext rorw) where
  418     lift f = OM $ lift $ lift $ lift f

  420 throwOverrideMatcher :: Monad m => String -> OverrideMatcher' sym ext rorw m a

  428 -- | "Run" function for OverrideMatcher. The final result and state
  429 -- are returned. The state will contain the updated globals and substitutions
  430 runOverrideMatcher ::
  431    (Monad m, W4.IsExprBuilder sym) =>
  432    sym                         {- ^ simulator                       -} ->
  433    Crucible.SymGlobalState sym {- ^ initial global variables        -} ->
  434    Map AllocIndex (Pointer' ext sym) {- ^ initial allocation substitution -} ->
  435    Map VarIndex Term           {- ^ initial term substitution       -} ->
  436    Set VarIndex                {- ^ initial free variables          -} ->
  437    W4.ProgramLoc               {- ^ override location information   -} ->
  438    OverrideMatcher' sym ext md m a {- ^ matching action                 -} ->
  439    m (Either (OverrideFailure ext) (a, OverrideState' sym ext))
  440 runOverrideMatcher sym g a t free loc (OM m) =
  441   runExceptT (runStateT (runReaderT m (initialEnv sym)) (initialState sym g a t free loc))

  443 addTermEq ::
  444   Term {- ^ term equality -} ->
  445   ConditionMetadata ->
  446   Crucible.SimError {- ^ reason   -} ->
  447   OverrideMatcher ext rorw ()
  448 addTermEq t md r = do
  449   env <- ask
  450   let cond = env ^. oeConditionalPred
  451   OM (termEqs %= cons (cond, t, md, r))

  453 addAssert ::
  454   (MonadIO m, W4.IsExprBuilder sym) =>
  455   W4.Pred sym       {- ^ property -} ->
  456   ConditionMetadata ->
  457   Crucible.SimError {- ^ reason   -} ->
  458   OverrideMatcher' sym ext rorw m ()
  459 addAssert p md r = do
  460   sym <- getSymInterface
  461   env <- ask
  462   p' <- liftIO $ W4.impliesPred sym (env ^. oeConditionalPred) p
  463   OM (osAsserts %= cons (md,W4.LabeledPred p' r))

  465 addAssume ::
  466   (MonadIO m, W4.IsExprBuilder sym) =>
  467   W4.Pred sym       {- ^ property -} ->
  468   ConditionMetadata ->
  469   OverrideMatcher' sym ext rorw m ()
  470 addAssume p md = do
  471   sym <- getSymInterface
  472   env <- ask
  473   p' <- liftIO $ W4.impliesPred sym (env ^. oeConditionalPred) p
  474   OM (osAssumes %= cons (md,p'))

  476 -- | Add an additional premise to the path condition when executing an
  477 -- `OverrideMatcher'` subcomputation. See @Note [oeConditionalPred]@ for an
  478 -- explanation of where this is used.
  479 withConditionalPred ::
  480   (MonadIO m, W4.IsExprBuilder sym) =>
  481   W4.Pred sym {- ^ The additional premise -} ->
  482   OverrideMatcher' sym ext rorw m a {- ^ The subcomputation -} ->
  483   OverrideMatcher' sym ext rorw m a
  484 withConditionalPred premise om = do
  485   sym <- getSymInterface
  486   env <- ask
  487   premise' <- liftIO $ W4.andPred sym premise (env ^. oeConditionalPred)
  488   let env' = env & oeConditionalPred .~ premise'
  489   local (const env') om

  491 readGlobal ::
  492   Monad m =>
  493   Crucible.GlobalVar tp ->
  494   OverrideMatcher' sym ext rorw m (Crucible.RegValue sym tp)
  495 readGlobal k =
  496   do mb <- OM (uses overrideGlobals (Crucible.lookupGlobal k))
  497      case mb of
  498        Nothing -> throwOverrideMatcher ("No such global: " ++ show k)
  499        Just v  -> return v

  501 writeGlobal ::
  502   Monad m =>
  503   Crucible.GlobalVar    tp ->
  504   Crucible.RegValue sym tp ->
  505   OverrideMatcher' sym ext RW m ()
  506 writeGlobal k v = OM (overrideGlobals %= Crucible.insertGlobal k v)

  520 -- | Verify that all of the fresh variables for the given
  521 -- state spec have been "learned". If not, throws
  522 -- 'AmbiguousVars' exception.
  523 enforceCompleteSubstitution ::
  524   W4.ProgramLoc ->
  525   MS.StateSpec ext ->
  526   OverrideMatcher ext w ()
  527 enforceCompleteSubstitution loc ss =
  528 
  529   do sub <- OM (use termSub)
  530 
  531      let -- predicate matches terms that are not covered by the computed
  532          -- term substitution
  533          isMissing tt = ecVarIndex (tecExt tt) `Map.notMember` sub
  534 
  535          -- list of all terms not covered by substitution
  536          missing = filter isMissing (view MS.csFreshVars ss)
  537 
  538      unless (null missing) (failure loc (AmbiguousVars missing))
  539 
  540 -- | Allocate fresh variables for all of the "fresh" vars
  541 -- used in this phase and add them to the term substitution.
  542 refreshTerms ::
  543   SharedContext    {- ^ shared context -} ->
  544   MS.StateSpec ext {- ^ current phase spec -} ->
  545   OverrideMatcher ext w ()
  546 refreshTerms sc ss =
  547   do extension <- Map.fromList <$> traverse freshenTerm (view MS.csFreshVars ss)
  548      OM (termSub %= Map.union extension)
  549   where
  550     freshenTerm (TypedExtCns _cty ec) =
  551       do ec' <- liftIO $ scFreshEC sc (toShortName (ecName ec)) (ecType ec)
  552          new <- liftIO $ scExtCns sc ec'
  553          return (ecVarIndex ec, new)

  568 -- | Partition into three groups:
  569 --   * Preconditions concretely succeed
  570 --   * Preconditions concretely fail
  571 --   * Preconditions are symbolic
  572 partitionOWPsConcrete :: forall ext.
  573   Sym ->
  574   [OverrideWithPreconditions ext] ->
  575   IO ([OverrideWithPreconditions ext], [OverrideWithPreconditions ext], [OverrideWithPreconditions ext])
  576 partitionOWPsConcrete sym =
  577   let trav = owpPreconditions . each . _2 . W4.labeledPred
  578   in W4.partitionByPredsM (Just sym) $
  579        foldlMOf trav (W4.andPred sym) (W4.truePred sym)

  581 -- | Like 'W4.partitionByPreds', but partitions on solver responses, not just
  582 --   concretized values.
  583 partitionBySymbolicPreds ::
  584   (OnlineSolver solver, Foldable t) =>
  585   Backend solver {- ^ solver connection -} ->
  586   (a -> W4.Pred Sym) {- ^ how to extract predicates -} ->
  587   t a ->
  588   IO (Map Crucible.BranchResult [a])
  589 partitionBySymbolicPreds sym getPred =
  590   let step mp a =
  591         Crucible.considerSatisfiability sym Nothing (getPred a) <&> \k ->
  592           Map.insertWith (++) k [a] mp
  593   in foldM step Map.empty

  595 -- | Find individual preconditions that are symbolically false
  596 --
  597 -- We should probably be using unsat cores for this.
  598 findFalsePreconditions ::
  599   OnlineSolver solver =>
  600   Backend solver ->
  601   OverrideWithPreconditions ext ->
  602   IO [(MS.ConditionMetadata, LabeledPred Sym)]
  603 findFalsePreconditions bak owp =
  604   fromMaybe [] . Map.lookup (Crucible.NoBranch False) <$>
  605     partitionBySymbolicPreds bak (view (_2 . W4.labeledPred)) (owp ^. owpPreconditions)

  607 -- | Is this group of predicates collectively unsatisfiable?
  608 unsatPreconditions ::
  609   OnlineSolver solver =>
  610   Backend solver {- ^ solver connection -} ->
  611   Fold s (W4.Pred Sym) {- ^ how to extract predicates -} ->
  612   s {- ^ a container full of predicates -}->
  613   IO Bool
  614 unsatPreconditions bak container getPreds = do
  615   let sym = Crucible.backendGetSym bak
  616   conj <- W4.andAllOf sym container getPreds
  617   Crucible.considerSatisfiability bak Nothing conj >>=
  618     \case
  619       Crucible.NoBranch False -> pure True
  620       _ -> pure False

  622 -- | Print a message about failure of an override's preconditions
  623 ppFailure ::
  624   (PP.Pretty (ExtType ext), PP.Pretty (MethodId ext)) =>
  625   OverrideWithPreconditions ext ->
  626   [LabeledPred Sym] ->
  627   PP.Doc ann
  628 ppFailure owp false =
  629   PP.vcat
  630   [ MS.ppMethodSpec (owp ^. owpMethodSpec)
  631     -- TODO: remove viaShow when crucible switches to prettyprinter
  632   , bullets '*' (map (PP.viaShow . Crucible.ppSimError)
  633                   (false ^.. traverse . W4.labeledPredMsg))
  634   ]

  636 -- | Print a message about concrete failure of an override's preconditions
  637 --
  638 -- Assumes that the override it's being passed does have concretely failing
  639 -- preconditions. Otherwise, the error won't make much sense.
  640 ppConcreteFailure ::
  641   (PP.Pretty (ExtType ext), PP.Pretty (MethodId ext)) =>
  642   OverrideWithPreconditions ext ->
  643   PP.Doc ann
  644 ppConcreteFailure owp =
  645   let (_, false, _) =
  646         W4.partitionLabeledPreds (Proxy :: Proxy Sym) (map snd (owp ^. owpPreconditions))
  647   in ppFailure owp false

  659 learnGhost ::
  660   SharedContext                                          ->
  661   MS.ConditionMetadata                                   ->
  662   PrePost                                                ->
  663   MS.GhostGlobal                                            ->
  664   TypedTerm                                              ->
  665   OverrideMatcher ext md ()
  666 learnGhost sc md prepost var (TypedTerm (TypedTermSchema schEx) tmEx) =
  667   do (sch,tm) <- readGlobal var
  668      when (sch /= schEx) $ fail $ unlines $
  669        [ "Ghost variable had the wrong type:"
  670        , "- Expected: " ++ show (Cryptol.pp schEx)
  671        , "- Actual:   " ++ show (Cryptol.pp sch)
  672        ]
  673      instantiateExtMatchTerm sc md prepost tm tmEx
  674 learnGhost _sc _md _prepost _var (TypedTerm tp _)
  675   = fail $ unlines
  676       [ "Ghost variable expected value has improper type"
  677       , "expected Cryptol schema type, but got"
  678       , show (MS.ppTypedTermType tp)
  679       ]

  681 executeGhost ::
  682   SharedContext ->
  683   MS.ConditionMetadata ->
  684   MS.GhostGlobal ->
  685   TypedTerm ->
  686   OverrideMatcher ext RW ()
  687 executeGhost sc _md var (TypedTerm (TypedTermSchema sch) tm) =
  688   do s <- OM (use termSub)

  690      writeGlobal var (sch,tm')

  694     , "expected Cryptol schema type, but got"
  695     , show (MS.ppTypedTermType tp)
  696     ]

  698 -- | NOTE: The two 'Term' arguments must have the same type.
  699 instantiateExtMatchTerm ::
  700   SharedContext   {- ^ context for constructing SAW terms    -} ->
  701   MS.ConditionMetadata ->
  702   PrePost                                                       ->
  703   Term            {- ^ exported concrete term                -} ->
  704   Term            {- ^ expected specification term           -} ->
  705   OverrideMatcher ext md ()
  706 instantiateExtMatchTerm sc md prepost actual expected = do
  707   sub <- OM (use termSub)
  708   matchTerm sc md prepost actual =<< liftIO (scInstantiateExt sc sub expected)
  709 
  710 matchTerm ::
  711   SharedContext   {- ^ context for constructing SAW terms    -} ->
  712   MS.ConditionMetadata ->
  713   PrePost                                                       ->
  714   Term            {- ^ exported concrete term                -} ->
  715   Term            {- ^ expected specification term           -} ->
  716   OverrideMatcher ext md ()
  717 
  718 matchTerm _ _ _ real expect | real == expect = return ()
  719 matchTerm sc md prepost real expect =
  720   do let loc = MS.conditionLoc md
  721      free <- OM (use osFree)
  722      case unwrapTermF expect of
  723        FTermF (ExtCns ec)
  724          | Set.member (ecVarIndex ec) free ->
  725          do assignTerm sc md prepost (ecVarIndex ec) real
  726 
  727        _ ->
  728          do t <- liftIO $ scEq sc real expect
  729             -- XXX get the user's ppOpts setting from somewhere
  730             let ppOpts = PPS.defaultOpts
  731             -- clamp the print depth to 20
  732             let ppOpts' = PPS.limitMaxDepth ppOpts 20
  733                 expect' = scPrettyTerm ppOpts' expect
  734                 real' = scPrettyTerm ppOpts' real
  735             let msg = unlines $
  736                   [ "Literal equality " ++ MS.stateCond prepost
  737                   , "Expected term: "
  738                   , expect'
  739                   , "Actual term:"
  740                   , real'
  741                   ]
  742             addTermEq t md $ Crucible.SimError loc $ Crucible.AssertFailureSimError msg ""
  743 
  744 assignTerm ::
  745   SharedContext      {- ^ context for constructing SAW terms    -} ->
  746   MS.ConditionMetadata ->
  747   PrePost                                                          ->
  748   VarIndex {- ^ external constant index -} ->
  749   Term     {- ^ value                   -} ->
  750   OverrideMatcher ext md ()
  751 
  752 assignTerm sc md prepost var val =
  753   do mb <- OM (use (termSub . at var))
  754      case mb of
  755        Nothing -> OM (termSub . at var ?= val)
  756        Just old ->
  757          matchTerm sc md prepost val old
