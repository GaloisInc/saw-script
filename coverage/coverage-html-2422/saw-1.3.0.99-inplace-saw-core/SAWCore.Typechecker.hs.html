   71 -- | Infer the type of an untyped term and complete it to a 'Term', all in the
   72 -- empty typing context
   73 inferCompleteTerm :: SharedContext -> Maybe ModuleName -> Un.UTerm ->
   74                      IO (Either PPS.Doc Term)
   75 inferCompleteTerm sc mnm t = inferCompleteTermCtx sc mnm [] t
   76 
   77 -- | Infer the type of an untyped term and complete it to a 'Term' in a given
   78 -- typing context
   79 inferCompleteTermCtx ::
   80   SharedContext -> Maybe ModuleName -> [(LocalName, Term)] ->
   81   Un.UTerm -> IO (Either PPS.Doc Term)
   82 inferCompleteTermCtx sc mnm ctx t =
   83   do res <- runTCM (typeInferComplete t) sc mnm ctx
   84      case res of
   85        -- TODO: avoid intermediate 'String's from 'prettyTCError'
   86        Left err -> return $ Left $ vsep $ map pretty $ prettyTCError err
   87        Right t' -> return $ Right $ typedVal t'

   89 -- | Look up the current module name, raising an error if it is not set
   90 getModuleName :: TCM ModuleName
   91 getModuleName =
   92   do maybe_modname <- askModName
   93      case maybe_modname of
   94        Just mnm -> return mnm
   95        Nothing ->
   96          panic "getModuleName" ["Current module name not set during typechecking"]

   98 -- | Build a multi-arity application of 'SCTypedTerm's
   99 inferApplyAll :: SCTypedTerm -> [SCTypedTerm] -> TCM SCTypedTerm
  100 inferApplyAll t [] = return t
  101 inferApplyAll t (arg:args) =
  102   do app1 <- typeInferComplete (App t arg)
  103      inferApplyAll app1 args
  104 
  105 -- | Resolve a name in the current module and apply it to some arguments
  106 inferResolveNameApp :: Text -> [SCTypedTerm] -> TCM SCTypedTerm
  107 inferResolveNameApp n args =
  108   do ctx <- askCtx
  109      mnm <- getModuleName
  110      mm <- liftTCM scGetModuleMap
  111      let ident = mkIdent mnm n
  112      case (findIndex ((== n) . fst) ctx, resolveNameInMap mm ident) of
  113        (Just i, _) ->
  114          do t <- typeInferComplete (LocalVar i :: TermF SCTypedTerm)
  115             inferApplyAll t args
  116        (_, Just (ResolvedCtor ctor)) ->
  117          do c <- traverse typeInferComplete (ctorExtCns ctor)
  118             t <- typeInferComplete (Constant c)
  119             inferApplyAll t args
  120        (_, Just (ResolvedDataType dt)) ->
  121          do let (params, ixs) = splitAt (length $ dtParams dt) args
  122             d <- traverse typeInferComplete (dtExtCns dt)
  123             -- NOTE: typeInferComplete will check that we have the correct number
  124             -- of indices
  125             typeInferComplete (DataTypeApp d params ixs)
  126        (_, Just (ResolvedDef d)) ->
  127          do t <- liftTCM scDefTerm d
  128             f <- SCTypedTerm t <$> liftTCM scTypeOf t
  129             inferApplyAll f args
  130        (Nothing, Nothing) ->
  131          throwTCError $ UnboundName n

  133 -- | Match an untyped term as a name applied to 0 or more arguments
  134 matchAppliedName :: Un.UTerm -> Maybe (Text, [Un.UTerm])
  135 matchAppliedName (Un.Name (PosPair _ n)) = Just (n, [])
  136 matchAppliedName (Un.App f arg) =
  137   do (n, args) <- matchAppliedName f
  138      return (n, args++[arg])
  139 matchAppliedName _ = Nothing
  140 
  141 -- | Match an untyped term as a recursor applied to 0 or more arguments
  142 matchAppliedRecursor :: Un.UTerm -> Maybe (Text, [Un.UTerm])
  143 matchAppliedRecursor (Un.Recursor (PosPair _ n)) = Just (n, [])
  144 matchAppliedRecursor (Un.App f arg) =
  145   do (n, args) <- matchAppliedRecursor f
  146      return (n, args++[arg])
  147 matchAppliedRecursor _ = Nothing

  158 -- The type inference engine for untyped terms, which mostly just dispatches to
  159 -- the type inference engine for (FTermF SCTypedTerm) defined in SCTypeCheck.hs
  160 instance TypeInfer Un.UTerm where
  161   typeInfer t = typedVal <$> typeInferComplete t
  162 
  163   typeInferComplete t =
  164     do typeInferDebug ("typechecking term: " ++ show t)
  165        res <- atPos (pos t) $ typeInferCompleteTerm t
  166        typeInferDebug ("completed typechecking term: " ++ show t ++ "\n"
  167                        ++ "type = " ++ show (typedType res))
  168        return res
  169 
  170 -- | Main workhorse function for type inference on untyped terms
  171 typeInferCompleteTerm :: Un.UTerm -> TCM SCTypedTerm
  172 
  173 -- Names
  174 typeInferCompleteTerm (matchAppliedName -> Just (n, args)) =
  175   mapM typeInferComplete args >>= inferResolveNameApp n
  176 typeInferCompleteTerm (Un.Name (PosPair _ n)) =
  177   -- NOTE: this is actually covered by the previous case, but we put it here
  178   -- so GHC doesn't complain about coverage
  179   inferResolveNameApp n []
  180 
  181 -- Sorts
  182 typeInferCompleteTerm (Un.Sort _ srt h) =
  183   typeInferComplete (Sort srt h :: FlatTermF SCTypedTerm)
  184 
  185 -- Recursors (must come before applications)
  186 typeInferCompleteTerm (matchAppliedRecursor -> Just (str, args)) =
  187   do mnm <- getModuleName
  188      mm <- liftTCM scGetModuleMap
  189      let dt_ident = mkIdent mnm str
  190      dt <- case findDataTypeInMap dt_ident mm of
  191        Just d -> return d
  192        Nothing -> throwTCError $ NoSuchDataType (ModuleIdentifier dt_ident)
  193      typed_args <- mapM typeInferComplete args
  194      case typed_args of
  195        (splitAt (length $ dtParams dt) ->
  196         (params,
  197          motive :
  198          (splitAt (length $ dtCtors dt) ->
  199           (elims,
  200            (splitAt (length $ dtIndices dt) ->
  201             (ixs, arg : rem_args)))))) ->
  202          do crec    <- compileRecursor dt params motive elims
  203             r       <- typeInferComplete (Recursor crec)
  204             typed_r <- typeInferComplete (RecursorApp r ixs arg)
  205             inferApplyAll typed_r rem_args
  206 
  207        _ -> throwTCError $ NotFullyAppliedRec (dtExtCns dt)
  208 
  209 typeInferCompleteTerm (Un.Recursor _) =
  210   error "typeInferComplete: found a bare Recursor, which should never happen!"
  211 
  212 -- Applications, lambdas, and pis
  213 typeInferCompleteTerm (Un.App f arg) =
  214   (App <$> typeInferComplete f <*> typeInferComplete arg)
  215   >>= typeInferComplete
  216 typeInferCompleteTerm (Un.Lambda _ [] t) = typeInferComplete t
  217 typeInferCompleteTerm (Un.Lambda p ((Un.termVarLocalName -> x, tp) : ctx) t) =
  218   do tp_trm <- typeInferComplete tp
  219      -- NOTE: we need the type of x to be normalized when we add it to the
  220      -- context in withVar, but we do not want to normalize this type in the
  221      -- output, as the contract for typeInferComplete only normalizes the type,
  222      -- so we use the unnormalized tp_trm in the return
  223      tp_whnf <- typeCheckWHNF $ typedVal tp_trm
  224      body <- withVar x tp_whnf $
  225        typeInferComplete $ Un.Lambda p ctx t
  226      typeInferComplete (Lambda x tp_trm body)
  227 typeInferCompleteTerm (Un.Pi _ [] t) = typeInferComplete t
  228 typeInferCompleteTerm (Un.Pi p ((Un.termVarLocalName -> x, tp) : ctx) t) =
  229   do tp_trm <- typeInferComplete tp
  230      -- NOTE: we need the type of x to be normalized when we add it to the
  231      -- context in withVar, but we do not want to normalize this type in the
  232      -- output, as the contract for typeInferComplete only normalizes the type,
  233      -- so we use the unnormalized tp_trm in the return
  234      tp_whnf <- typeCheckWHNF $ typedVal tp_trm
  235      body <- withVar x tp_whnf $
  236        typeInferComplete $ Un.Pi p ctx t
  237      typeInferComplete (Pi x tp_trm body)
  238 
  239 -- Non-dependent records
  240 typeInferCompleteTerm (Un.RecordValue _ elems) =
  241   do typed_elems <-
  242        mapM (\(PosPair _ fld, t) -> (fld,) <$> typeInferComplete t) elems
  243      typeInferComplete (RecordValue typed_elems)
  244 typeInferCompleteTerm (Un.RecordType _ elems) =
  245   do typed_elems <-
  246        mapM (\(PosPair _ fld, t) -> (fld,) <$> typeInferComplete t) elems
  247      typeInferComplete (RecordType typed_elems)
  248 typeInferCompleteTerm (Un.RecordProj t prj) =
  249   (RecordProj <$> typeInferComplete t <*> return prj) >>= typeInferComplete
  250 
  251 -- Unit
  252 typeInferCompleteTerm (Un.UnitValue _) =
  253   typeInferComplete (UnitValue :: FlatTermF SCTypedTerm)
  254 typeInferCompleteTerm (Un.UnitType _) =
  255   typeInferComplete (UnitType :: FlatTermF SCTypedTerm)
  256 
  257 -- Simple pairs
  258 typeInferCompleteTerm (Un.PairValue _ t1 t2) =
  259   (PairValue <$> typeInferComplete t1 <*> typeInferComplete t2)
  260   >>= typeInferComplete
  261 typeInferCompleteTerm (Un.PairType _ t1 t2) =
  262   (PairType <$> typeInferComplete t1 <*> typeInferComplete t2)
  263   >>= typeInferComplete
  264 typeInferCompleteTerm (Un.PairLeft t) =
  265   (PairLeft <$> typeInferComplete t) >>= typeInferComplete
  266 typeInferCompleteTerm (Un.PairRight t) =
  267   (PairRight <$> typeInferComplete t) >>= typeInferComplete
  268 
  269 -- Type ascriptions
  270 typeInferCompleteTerm (Un.TypeConstraint t _ tp) =
  271   do typed_t <- typeInferComplete t
  272      typed_tp <- typeInferComplete tp
  273      _ <- ensureSort (typedType typed_tp)
  274      checkSubtype typed_t (typedVal typed_tp)
  275      return typed_t
  276 
  277 -- Literals
  278 typeInferCompleteTerm (Un.NatLit _ i) =
  279   typeInferComplete (NatLit i :: FlatTermF SCTypedTerm)
  280 typeInferCompleteTerm (Un.StringLit _ str) =
  281   typeInferComplete (StringLit str :: FlatTermF SCTypedTerm)
  282 typeInferCompleteTerm (Un.VecLit _ []) = throwTCError EmptyVectorLit
  283 typeInferCompleteTerm (Un.VecLit _ ts) =
  284   do typed_ts <- mapM typeInferComplete ts
  285      tp <- case typed_ts of
  286        (t1:_) -> return $ typedType t1
  287        [] -> throwTCError $ EmptyVectorLit
  288      type_of_tp <- typeInfer tp
  289      typeInferComplete (ArrayValue (SCTypedTerm tp type_of_tp) $
  290                         V.fromList typed_ts)
  291 typeInferCompleteTerm (Un.BVLit _ []) = throwTCError EmptyVectorLit
  292 typeInferCompleteTerm (Un.BVLit _ bits) =
  293   do tp <- liftTCM scBoolType
  294      bit_tms <- mapM (liftTCM scBool) bits
  295      typeInferComplete $ ArrayValue tp $ V.fromList bit_tms
  296 
  297 typeInferCompleteTerm (Un.BadTerm _) =
  298   -- Should be unreachable, since BadTerms represent parse errors, that should
  299   -- already have been signaled before type inference
  300   panic "typeInferCompleteTerm" ["Type inference encountered a BadTerm"]

  312 -- | Type-check a list of declarations and insert them into the current module
  313 processDecls :: [Un.Decl] -> TCM ()
  314 processDecls [] = return ()
  315 
  316 processDecls (Un.InjectCodeDecl ns txt : rest) =
  317   do mnm <- getModuleName
  318      liftTCM scInjectCode mnm ns txt
  319      processDecls rest
  320 
  321 processDecls (Un.TypedDef nm params rty body : rest) =
  322   processDecls (Un.TypeDecl NoQualifier nm (Un.Pi (pos nm) params rty) :
  323                 Un.TermDef nm (map fst params) body : rest)
  324 
  325 processDecls (Un.TypeDecl NoQualifier (PosPair p nm) tp :
  326               Un.TermDef (PosPair _ ((== nm) -> True)) vars body : rest) =
  327   -- Type-checking for definitions
  328   (atPos p $
  329    do
  330      -- Step 1: type-check the type annotation, and make sure it is a type
  331      typed_tp <- typeInferComplete tp
  332      void $ ensureSort $ typedType typed_tp
  333      let def_tp = typedVal typed_tp
  334      def_tp_whnf <- liftTCM scTypeCheckWHNF def_tp
  335 
  336      -- Step 2: assign types to the bound variables of the definition, by
  337      -- peeling off the pi-abstraction variables in the type annotation. Any
  338      -- remaining body of the pi-type is the required type for the def body.
  339      (ctx, req_body_tp) <-
  340        case matchPiWithNames (map Un.termVarLocalName vars) def_tp_whnf of
  341          Just x -> return x
  342          Nothing ->
  343              throwTCError $
  344              DeclError nm ("More variables " ++ show vars ++
  345                            " than length of function type:\n" ++
  346                            showTerm (typedVal typed_tp))
  347 
  348      -- Step 3: type-check the body of the definition in the context of its
  349      -- variables, and build a function that takes in those variables
  350      def_tm <-
  351        withCtx ctx $
  352        do typed_body <- typeInferComplete body
  353           checkSubtype typed_body req_body_tp
  354           liftTCM scLambdaList ctx (typedVal typed_body)
  355 
  356      -- Step 4: add the definition to the current module
  357      mnm <- getModuleName
  358      let ident = mkIdent mnm nm
  359      liftTCM scInsertDef ident def_tp def_tm) >>
  360   processDecls rest
  361 
  362 processDecls (Un.TypeDecl NoQualifier (PosPair p nm) _ : _) =
  363   atPos p $ throwTCError $ DeclError nm "Definition without defining equation"
  364 
  365 processDecls (Un.TypeDecl _ (PosPair p nm) _ :
  366               Un.TermDef (PosPair _ ((== nm) -> True)) _ _ : _) =
  367   atPos p $ throwTCError $ DeclError nm "Primitive or axiom with definition"
  368 
  369 processDecls (Un.TypeDecl q (PosPair p nm) tp : rest) =
  370   (atPos p $
  371    do typed_tp <- typeInferComplete tp
  372       void $ ensureSort $ typedType typed_tp
  373       mnm <- getModuleName
  374       let ident = mkIdent mnm nm
  375       let def_tp = typedVal typed_tp
  376       liftTCM scDeclarePrim ident q def_tp) >>
  377   processDecls rest
  378 
  379 processDecls (Un.TermDef (PosPair p nm) _ _ : _) =
  380   atPos p $ throwTCError $ DeclError nm "Dangling definition without a type"
  381 
  382 processDecls (Un.DataDecl (PosPair p nm) param_ctx dt_tp c_decls : rest) =
  383   -- This top line makes sure that we process the rest of the decls after the
  384   -- main body of the code below, which processes just the current data decl
  385   (>> processDecls rest) $ atPos p $
  386   -- Step 1: type-check the parameters
  387   typeInferCompleteInCtx param_ctx $ \params -> do
  388   let dtParams = map (\(x,tp,_) -> (x,tp)) params
  389   let param_sort = maxSort (map (\(_,_,s) -> s) params)
  390   let err :: String -> TCM a
  391       err msg = throwTCError $ DeclError nm msg
  392 
  393   -- Step 2: type-check the type given for d, and make sure it is of the form
  394   -- (i1:ix1) -> ... -> (in:ixn) -> Type s for some sort s. Then form the full
  395   -- type of d as (p1:param1) -> ... -> (i1:ix1) -> ... -> Type s
  396   (dt_ixs, dtSort) <-
  397     case Un.asPiList dt_tp of
  398       (ixs, Un.Sort _ s h) | h == noFlags ->
  399         return (ixs, s) -- NB, don't allow `isort`, etc.
  400       _ -> err "Wrong form for type of datatype"
  401   dt_ixs_typed <- typeInferCompleteCtx dt_ixs
  402   let dtIndices = map (\(x,tp,_) -> (x,tp)) dt_ixs_typed
  403       ixs_max_sort = maxSort (map (\(_,_,s) -> s) dt_ixs_typed)
  404 
  405   -- Step 3: do the necessary universe inclusion checking for any predicative
  406   -- (non-Prop) inductive type, which includes:
  407   --
  408   -- 1. All ix types must be of sort dtSort; AND
  409   -- 2. All param types must be of sort dtSort+1
  410   if dtSort /= propSort && param_sort > sortOf dtSort then
  411     err ("Universe level of parameters should be no greater" ++
  412          " than that of the datatype")
  413     else return ()
  414   if dtSort /= propSort && ixs_max_sort > dtSort then
  415     err ("Universe level of indices should be strictly contained" ++
  416          " in that of the datatype")
  417     else return ()
  418 
  419   -- Step 4: Add d as an empty datatype, so we can typecheck the constructors
  420   mnm <- getModuleName
  421   let dtName = mkIdent mnm nm
  422   pn <- liftTCM scBeginDataType dtName dtParams dtIndices dtSort
  423 
  424   -- Step 5: typecheck the constructors, and build Ctors for them
  425   typed_ctors <-
  426     mapM (\(Un.Ctor (PosPair p' c) ctx body) ->
  427            (c,) <$> typeInferComplete (Un.Pi p' ctx body)) c_decls
  428   ctors <-
  429     case ctxBindingsOfTerms dtParams of
  430       ExistsTp p_ctx ->
  431         case ctxBindingsOfTerms dtIndices of
  432           ExistsTp ix_ctx ->
  433             forM typed_ctors $ \(c, typed_tp) ->
  434             -- Check that the universe level of the type of each constructor
  435             (case asSort (typedType typed_tp) of
  436                 Just ctor_sort
  437                   | dtSort /= propSort && ctor_sort > dtSort ->
  438                     err ("Universe level of constructors should be strictly" ++
  439                          " contained in that of the datatype")
  440                 Just _ ->
  441                     return ()
  442                 Nothing ->
  443                     panic "processDecls" [
  444                         "Type of the type of constructor is not a sort!",
  445                         "Constructor type: " <> Text.pack (showTerm $ typedVal typed_tp),
  446                         "Type of that type: " <> Text.pack (showTerm $ typedType typed_tp)
  447                     ]
  448             ) >>
  449             let tp = typedVal typed_tp in
  450             case mkCtorArgStruct pn p_ctx ix_ctx tp of
  451               Just arg_struct ->
  452                 liftTCM scBuildCtor pn (mkIdent mnm c) arg_struct
  453               Nothing -> err ("Malformed type form constructor: " ++ show c)
  454 
  455   -- Step 6: complete the datatype with the given ctors
  456   liftTCM scCompleteDataType dtName ctors
  457

  459 -- | Typecheck a module and, on success, insert it into the current context
  460 tcInsertModule :: SharedContext -> Un.Module -> IO ()
  461 tcInsertModule sc (Un.Module (PosPair _ mnm) imports decls) = do
  462   -- First, insert an empty module for mnm
  463   scLoadModule sc $ emptyModule mnm
  464   -- Next, process all the imports
  465   forM_ imports $ \imp ->
  466     do let imn = val $ Un.importModName imp
  467        i_exists <- scModuleIsLoaded sc imn
  468        unless i_exists $ fail $ "Imported module not found: " ++ show imn
  469        scImportModule sc (Un.nameSatsConstraint (Un.importConstraints imp) . Text.unpack) imn mnm
  470   -- Finally, process all the decls
  471   decls_res <- runTCM (processDecls decls) sc (Just mnm) []
  472   case decls_res of
  473     Left err -> fail $ unlines $ prettyTCError err
  474     Right _ -> return ()

  481 -- | Pattern match a nested pi-abstraction, like 'asPiList', but only match as
  482 -- far as the supplied list of variables, and use them as the new names
  483 matchPiWithNames :: [LocalName] -> Term -> Maybe ([(LocalName, Term)], Term)
  484 matchPiWithNames [] tp = return ([], tp)
  485 matchPiWithNames (var:vars) (asPi -> Just (_, arg_tp, body_tp)) =
  486   do (ctx,body) <- matchPiWithNames vars body_tp
  487      return ((var,arg_tp):ctx,body)
  488 matchPiWithNames _ _ = Nothing
