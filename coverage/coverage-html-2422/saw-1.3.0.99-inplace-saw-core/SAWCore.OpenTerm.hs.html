   16 
   17 This module defines an interface to building SAW core terms in an incrementally
   18 type-checked way, meaning that type-checking is performed as the terms are
   19 built. The interface provides a convenient DSL for building terms in a pure way,
   20 where sub-terms can be freely composed and combined into super-terms without
   21 monadic sequencing or 'IO' computations; the 'IO' computation is only run at the
   22 top level when all the term-building is complete. Users of this interface can
   23 also build binding constructs like lambda- and pi-abstractions without worrying
   24 about deBruijn indices, lifting, and free variables. Instead, a key feature of
   25 this interface is that it uses higher-order abstract syntax for lambda- and
   26 pi-abstractions, meaning that the bodies of these term constructs are specified
   27 as Haskell functions that take in terms for the bound variables. The library
   28 takes care of all the deBruijn indices under the hood.
   29 
   30 To use the 'OpenTerm' API, the caller builds up 'OpenTerm's using a variety of
   31 combinators that mirror the SAW core 'Term' structure. As some useful examples
   32 of 'OpenTerm' operations, 'applyOpenTerm' applies one 'OpenTerm' to another,
   33 'globalOpenTerm' builds an 'OpenTerm' for a global identifier, and
   34 'lambdaOpenTerm' builds a lambda-abstraction. For instance, the SAW core term
   35 
   36 > \ (f : Bool -> Bool) (x : Bool) -> f x
   37 
   38 can be built with the 'OpenTerm' expression
   39 
   40 > let bool = globalOpenTerm "Prelude.Bool" in
   41 > lambdaOpenTerm "f" (arrowOpenTerm bool bool) $ \f ->
   42 > lambdaOpenTerm "x" (globalOpenTerm "Prelude.Bool") $ \x ->
   43 > applyOpenTerm f x
   44 
   45 Existing SAW core 'Term's can be used in 'OpenTerm' by applying 'closedOpenTerm'
   46 if the 'Term' is closed (meaning it has no free variables) or 'openOpenTerm' if
   47 it does, where the latter requires the context of free variables to be
   48 specified. At the top level, 'completeOpenTerm' then "completes" an 'OpenTerm'
   49 by running its underlying 'IO' computation to build and type-check the resulting
   50 SAW core 'Term'.
   51 -}

  137 -- | An open term is represented as a type-checking computation that computes a
  138 -- SAW core term and its type
  139 newtype OpenTerm = OpenTerm { unOpenTerm :: TCM SCTypedTerm }

  141 -- | \"Complete\" an 'OpenTerm' to a closed term or 'fail' on type-checking
  142 -- error
  143 completeOpenTerm :: SharedContext -> OpenTerm -> IO Term
  144 completeOpenTerm sc (OpenTerm termM) =
  145   either (fail . show) return =<<
  146   runTCM (typedVal <$> termM) sc Nothing []
  147 
  148 -- | \"Complete\" an 'OpenTerm' to a closed term and 'betaNormalize' the result
  149 completeNormOpenTerm :: SharedContext -> OpenTerm -> IO Term
  150 completeNormOpenTerm sc m =
  151   completeOpenTerm sc m >>= sawLetMinimize sc >>= betaNormalize sc
  152 
  153 -- | \"Complete\" an 'OpenTerm' to a closed term for its type
  154 completeOpenTermType :: SharedContext -> OpenTerm -> IO Term
  155 completeOpenTermType sc (OpenTerm termM) =
  156   either (fail . show) return =<<
  157   runTCM (typedType <$> termM) sc Nothing []

  159 -- | Embed a closed 'Term' into an 'OpenTerm'
  160 closedOpenTerm :: Term -> OpenTerm
  161 closedOpenTerm t = OpenTerm $ typeInferComplete t

  163 -- | Embed a 'Term' in the given typing context into an 'OpenTerm'
  164 openOpenTerm :: [(LocalName, Term)] -> Term -> OpenTerm
  165 openOpenTerm ctx t =
  166   -- Extend the local type-checking context, wherever this OpenTerm gets used,
  167   -- by appending ctx to the end, so that variables 0..length ctx-1 all get
  168   -- type-checked with ctx. If these are really the only free variables, then it
  169   -- won't matter what the rest of the ambient context is.
  170   --
  171   -- FIXME: we should check that the free variables of t are all < length ctx
  172   OpenTerm $ withCtx ctx $ typeInferComplete t

  174 -- | Build an 'OpenTerm' that 'fail's in the underlying monad when completed

  176 failOpenTerm str = OpenTerm $ fail str

  178 -- | Bind the result of a type-checking computation in building an 'OpenTerm'.
  179 -- NOTE: this operation should be considered \"unsafe\" because it can create
  180 -- malformed 'OpenTerm's if the result of the 'TCM' computation is used as part
  181 -- of the resulting 'OpenTerm'. For instance, @a@ should not be 'OpenTerm'.
  182 bindTCMOpenTerm :: TCM a -> (a -> OpenTerm) -> OpenTerm
  183 bindTCMOpenTerm m f = OpenTerm (m >>= unOpenTerm . f)

  185 -- | Bind the result of pretty-printing an 'OpenTerm' while building another
  186 bindPPOpenTerm :: OpenTerm -> (String -> OpenTerm) -> OpenTerm
  187 bindPPOpenTerm (OpenTerm m) f =
  188   OpenTerm $
  189   do ctx <- askCtx
  190      t <- typedVal <$> m
  191      -- XXX: this could use scPrettyTermInCtx (which builds in the call to
  192      -- PPS.render) except that it's slightly different under the covers
  193      -- (in its use of the "global" flag, and it isn't entirely clear what
  194      -- that actually does)
  195      unOpenTerm $ f $ PPS.render PPS.defaultOpts $
  196        ppTermInCtx PPS.defaultOpts (map fst ctx) t

  344 -- | Build an 'OpenTerm' for a constructor applied to its arguments
  345 ctorOpenTerm :: Ident -> [OpenTerm] -> OpenTerm
  346 ctorOpenTerm c all_args = OpenTerm $ do
  347   maybe_ctor <- liftTCM scFindCtor c
  348   ctor <- case maybe_ctor of
  349             Just ctor -> pure ctor
  350             Nothing -> throwTCError $ NoSuchCtor (ModuleIdentifier c)
  351   (params, args) <- splitAt (ctorNumParams ctor) <$> mapM unOpenTerm all_args
  352   c' <- traverse typeInferComplete (ctorExtCns ctor)
  353   typeInferComplete $ CtorApp c' params args
  354 
  355 -- | Build an 'OpenTerm' for a datatype applied to its arguments
  356 dataTypeOpenTerm :: Ident -> [OpenTerm] -> OpenTerm
  357 dataTypeOpenTerm d all_args = OpenTerm $ do
  358   maybe_dt <- liftTCM scFindDataType d
  359   dt <- case maybe_dt of
  360           Just dt -> pure dt
  361           Nothing -> throwTCError $ NoSuchDataType d
  362   (params, args) <- splitAt (dtNumParams dt) <$> mapM unOpenTerm all_args
  363   d' <- traverse typeInferComplete (dtPrimName dt)
  364   typeInferComplete $ DataTypeApp d' params args

  373 -- | Build an 'OpenTerm' for an 'Ident', which can either refer to a definition,
  374 -- a constructor, or a datatype
  375 identOpenTerm :: Ident -> OpenTerm
  376 identOpenTerm ident =
  377   OpenTerm $
  378   do maybe_ctor <- liftTCM scFindCtor ident
  379      maybe_dt <- liftTCM scFindDataType ident
  380 
  381      -- First, determine the variables we need to abstract over and the function
  382      -- for building an application of this identifier dependent on the class of
  383      -- identifier
  384      let (var_ctx, app_fun) =
  385            case (maybe_ctor, maybe_dt) of
  386              (Just ctor, _) -> (fst (asPiList (ctorType ctor)), scCtorApp)
  387              (_, Just dt) -> (dtParams dt ++ dtIndices dt, scDataTypeApp)
  388              (Nothing, Nothing) -> ([], scGlobalApply)
  389 
  390      -- Build the term \ (x1:tp1) ... (xn:tpn) -> ident x1 ... xn as follows:
  391      -- 1. Construct vars as the list x1 ... xn of terms, noting that xn has
  392      --    deBruijn index 0 and x1 has deBruijn index (length var_ctx) - 1;
  393      -- 2. Apply ident to those variables; and
  394      -- 3. Lambda-abstract the variables.
  395      vars <- reverse <$> mapM (liftTCM scLocalVar) [0 .. (length var_ctx) - 1]
  396      ident_app <- liftTCM app_fun ident vars
  397      lam <- liftTCM scLambdaList var_ctx ident_app
  398      typeInferComplete lam

  417 -- | Compute the output type of applying a function of a given type to an
  418 -- argument. That is, given @tp@ and @arg@, compute the type of applying any @f@
  419 -- of type @tp@ to @arg@.
  420 applyPiOpenTerm :: OpenTerm -> OpenTerm -> OpenTerm
  421 applyPiOpenTerm (OpenTerm m_f) (OpenTerm m_arg) =
  422   OpenTerm $
  423   do f <- m_f
  424      arg <- m_arg
  425      ret <- applyPiTyped (NotFuncTypeInApp f arg) (typedVal f) arg
  426      ctx <- askCtx
  427      ret_tp <- liftTCM scTypeOf' (map snd ctx) ret
  428      return (SCTypedTerm ret ret_tp)

  430 -- | Get the argument type of a function type, 'fail'ing if the input term is
  431 -- not a function type
  432 piArgOpenTerm :: OpenTerm -> OpenTerm
  433 piArgOpenTerm (OpenTerm m) =
  434   OpenTerm $ m >>= \case
  435   (unwrapTermF . typedVal -> Pi _ tp _) -> typeInferComplete tp
  436   t ->
  437     do ctx <- askCtx
  438        fail ("piArgOpenTerm: not a pi type: " ++
  439              scPrettyTermInCtx PPS.defaultOpts (map fst ctx) (typedVal t))

  441 -- | Build an 'OpenTerm' for the top variable in the current context, by
  442 -- building the 'TCM' computation which checks how much longer the context has
  443 -- gotten since the variable was created and uses this to compute its deBruijn
  444 -- index
  445 openTermTopVar :: TCM OpenTerm
  446 openTermTopVar =
  447   do outer_ctx <- askCtx
  448      return $ OpenTerm $ do
  449        inner_ctx <- askCtx
  450        typeInferComplete (LocalVar (length inner_ctx
  451                                     - length outer_ctx) :: TermF Term)

  453 -- | Build an open term inside a binder of a variable with the given name and
  454 -- type, where the binder is represented as a Haskell function on 'OpenTerm's
  455 bindOpenTerm :: LocalName -> SCTypedTerm -> (OpenTerm -> OpenTerm) ->
  456                 TCM SCTypedTerm
  457 bindOpenTerm x tp body_f =
  458   do tp_whnf <- typeCheckWHNF $ typedVal tp
  459      withVar x tp_whnf (openTermTopVar >>= (unOpenTerm . body_f))

  462 lambdaOpenTerm :: LocalName -> OpenTerm -> (OpenTerm -> OpenTerm) -> OpenTerm
  463 lambdaOpenTerm x (OpenTerm tpM) body_f = OpenTerm $
  464   do tp <- tpM
  465      body <- bindOpenTerm x tp body_f
  466      typeInferComplete $ Lambda x tp body

  534 -- | Build the type @Sigma a (\ (x:a) -> b)@ from variable name @x@, type @a@,
  535 -- and type-level function @b@
  536 sigmaTypeOpenTerm :: LocalName -> OpenTerm -> (OpenTerm -> OpenTerm) -> OpenTerm
  537 sigmaTypeOpenTerm x tp f =
  538   dataTypeOpenTerm "Prelude.Sigma" [tp, lambdaOpenTerm x tp f]

  540 -- | Build the type @Sigma a1 (\ (x1:a1) -> Sigma a2 (\ (x2:a2) -> ...))@
  541 sigmaTypeOpenTermMulti :: LocalName -> [OpenTerm] -> ([OpenTerm] -> OpenTerm) ->
  542                           OpenTerm
  543 sigmaTypeOpenTermMulti _ [] f = f []
  544 sigmaTypeOpenTermMulti x (tp:tps) f =
  545   sigmaTypeOpenTerm x tp $ \ t ->
  546   sigmaTypeOpenTermMulti x tps $ \ts -> f (t:ts)

  548 -- | Build the dependent pair @exists a (\ (x:a) -> b) x y@ whose type is given
  549 -- by 'sigmaTypeOpenTerm'
  550 sigmaOpenTerm :: LocalName -> OpenTerm -> (OpenTerm -> OpenTerm) ->
  551                  OpenTerm -> OpenTerm -> OpenTerm
  552 sigmaOpenTerm x tp tp_f trm_l trm_r =
  553   ctorOpenTerm "Prelude.exists" [tp, lambdaOpenTerm x tp tp_f, trm_l, trm_r]
  554 
  555 -- | Build the right-nested dependent pair @(x1, (x2, ...(xn, y)))@ whose type
  556 -- is given by 'sigmaTypeOpenTermMulti'
  557 sigmaOpenTermMulti :: LocalName -> [OpenTerm] -> ([OpenTerm] -> OpenTerm) ->
  558                       [OpenTerm] -> OpenTerm -> OpenTerm
  559 sigmaOpenTermMulti _ [] _ [] trm = trm
  560 sigmaOpenTermMulti x (tp:tps) tp_f (trm_l:trms_l) trm_r =
  561   sigmaOpenTerm x tp (\t -> sigmaTypeOpenTermMulti x tps (tp_f . (t:))) trm_l $
  562   sigmaOpenTermMulti x tps (tp_f . (trm_l:)) trms_l trm_r
  563 sigmaOpenTermMulti _ tps _ trms _ =
  564   panic "sigmaOpenTermMulti" [
  565      "The number of types and arguments disagree:",
  566      Text.pack (show $ length tps) <> " Remaining types",
  567      Text.pack (show $ length trms) <> " remaining terms",
  568      "(sorry, the values themselves are unresolved monadic computations)"
  569   ]

  571 -- | Take a nested dependent pair (of the type returned by
  572 -- 'sigmaTypeOpenTermMulti') and apply a function @f@ to all of its projections
  573 sigmaElimOpenTermMulti :: LocalName -> [OpenTerm] -> ([OpenTerm] -> OpenTerm) ->
  574                           OpenTerm -> ([OpenTerm] -> OpenTerm) -> OpenTerm
  575 sigmaElimOpenTermMulti _ [] _ t f_elim = f_elim [t]
  576 sigmaElimOpenTermMulti x (tp:tps) tp_f sig f_elim =
  577   let b_fun = lambdaOpenTerm x tp (\t -> sigmaTypeOpenTermMulti x tps (tp_f . (t:)))
  578       proj1 = applyGlobalOpenTerm "Prelude.Sigma_proj1" [tp, b_fun, sig]
  579       proj2 = applyGlobalOpenTerm "Prelude.Sigma_proj2" [tp, b_fun, sig] in
  580   sigmaElimOpenTermMulti x tps (tp_f . (proj1:)) proj2 (f_elim . (proj1:))

  868 -- | The monad for building 'OpenTerm's if you want to add in 'IO' actions. This

  894 -- | Build an open term inside a binder of a variable with the given name and
  895 -- type, where the binder is represented as a monadic Haskell function on
  896 -- 'OpenTerm's that also returns an auxiliary value. Returns the normalized type
  897 -- and the body, along with the auxiliary result returned by the body-generating
  898 -- function.
  899 bindOpenTermAuxM ::
  900   LocalName -> OpenTerm ->
  901   (OpenTerm -> OpenTermM (OpenTerm, a)) ->
  902   OpenTermM (SCTypedTerm, SCTypedTerm, a)
  903 bindOpenTermAuxM x (OpenTerm tpM) body_f =
  904   OpenTermM $
  905   do SCTypedTerm tp tp_tp <- tpM
  906      tp_whnf <- typeCheckWHNF tp
  907      (OpenTerm bodyM, a) <-
  908        withVar x tp_whnf (openTermTopVar >>= (unOpenTermM . body_f))
  909      body <- bodyM
  910      return (SCTypedTerm tp_whnf tp_tp, body, a)

  912 -- | Build a lambda abstraction in the 'OpenTermM' monad
  913 lambdaOpenTermM ::
  914   LocalName -> OpenTerm -> (OpenTerm -> OpenTermM OpenTerm) ->
  915   OpenTermM OpenTerm
  916 lambdaOpenTermM x tp body_f =
  917   fst <$> lambdaOpenTermAuxM x tp (body_f >=> (\t -> return (t, ())))

  920 piOpenTermM ::
  921   LocalName -> OpenTerm -> (OpenTerm -> OpenTermM OpenTerm) ->
  922   OpenTermM OpenTerm
  923 piOpenTermM x tp body_f =
  924   fst <$> piOpenTermAuxM x tp (body_f >=> (\t -> return (t, ())))

  926 -- | Build a lambda abstraction with an auxiliary return value in the
  927 -- 'OpenTermM' monad
  928 lambdaOpenTermAuxM ::
  929   LocalName -> OpenTerm ->
  930   (OpenTerm -> OpenTermM (OpenTerm, a)) ->
  931   OpenTermM (OpenTerm, a)
  932 lambdaOpenTermAuxM x tp body_f =

  947 -- Types that provide similar operations to 'OpenTerm'
  948 
  949 class OpenTermLike t where
  950   -- | Convert an 'OpenTerm' to a @t@
  951   openTermLike :: OpenTerm -> t
  952   -- | Get the type of a @t@
  953   typeOfTermLike :: t -> t
  954   -- | Build a @t@ from a 'FlatTermF'
  955   flatTermLike :: FlatTermF t -> t
  956   -- | Apply a @t@ to another @t@
  957   applyTermLike :: t -> t -> t
  958   -- | Build a lambda abstraction as a @t@
  959   lambdaTermLike :: LocalName -> t -> (t -> t) -> t
  960   -- | Build a pi abstraction as a @t@
  961   piTermLike :: LocalName -> t -> (t -> t) -> t
  962   -- | Build a @t@ for a constructor applied to its arguments
  963   ctorTermLike :: Ident -> [t] -> t
  964   -- | Build a @t@ for a datatype applied to its arguments
  965   dataTypeTermLike :: Ident -> [t] -> t
  966 
  967 instance OpenTermLike OpenTerm where
  968   openTermLike = id
  969   typeOfTermLike = openTermType
  970   flatTermLike = flatOpenTerm
  971   applyTermLike = applyOpenTerm
  972   lambdaTermLike = lambdaOpenTerm
  973   piTermLike = piOpenTerm
  974   ctorTermLike = ctorOpenTerm
  975   dataTypeTermLike = dataTypeOpenTerm
  976 
  977 -- Lift an OpenTermLike instance from t to functions from some type a to t,
  978 -- where the OpenTermLike methods pass the same input a argument to all subterms
  979 instance OpenTermLike t => OpenTermLike (a -> t) where
  980   openTermLike t = const $ openTermLike t
  981   typeOfTermLike t = \x -> typeOfTermLike (t x)
  982   flatTermLike ftf = \x -> flatTermLike (fmap ($ x) ftf)
  983   applyTermLike f arg = \x -> applyTermLike (f x) (arg x)
  984   lambdaTermLike nm tp bodyF =
  985     \x -> lambdaTermLike nm (tp x) (\y -> bodyF (const y) x)
  986   piTermLike nm tp bodyF =
  987     \x -> piTermLike nm (tp x) (\y -> bodyF (const y) x)
  988   ctorTermLike c args = \x -> ctorTermLike c (map ($ x) args)
  989   dataTypeTermLike d args = \x -> dataTypeTermLike d (map ($ x) args)
  990 
  991 -- This is the same as the function instance above
  992 instance OpenTermLike t => OpenTermLike (Reader r t) where
  993   openTermLike t = reader $ openTermLike t
  994   typeOfTermLike t = reader $ typeOfTermLike $ runReader t
  995   flatTermLike ftf = reader $ flatTermLike $ fmap runReader ftf
  996   applyTermLike f arg = reader $ applyTermLike (runReader f) (runReader arg)
  997   lambdaTermLike x tp body =
  998     reader $ lambdaTermLike x (runReader tp) (runReader . body . reader)

 1128 -- | A map from each deBruijn index to a count of its occurrences in a term
 1129 newtype VarOccs = VarOccs [Integer]
 1130 
 1131 -- | Make a 'VarOccs' with a single occurrence of a deBruijn index
 1132 varOccs1 :: DeBruijnIndex -> VarOccs
 1133 varOccs1 i = VarOccs (take i (repeat 0) ++ [1])
 1134 
 1135 -- | Move a 'VarOccs' out of a binder by returning the number of occurrences of
 1136 -- deBruijn index 0 along with the result of subtracting 1 from all other indices
 1137 unconsVarOccs :: VarOccs -> (Integer, VarOccs)
 1138 unconsVarOccs (VarOccs []) = (0, VarOccs [])
 1139 unconsVarOccs (VarOccs (cnt:occs)) = (cnt, VarOccs occs)
 1140 
 1141 -- | Multiply every index in a 'VarOccs' by a constant
 1142 multVarOccs :: Integer -> VarOccs -> VarOccs
 1143 multVarOccs i (VarOccs occs) = VarOccs $ map (* i) occs

 1145 -- | The infinite list of zeroes
 1146 zeroes :: [Integer]
 1147 zeroes = 0:zeroes

 1149 instance Semigroup VarOccs where
 1150   (VarOccs occs1) <> (VarOccs occs2)
 1151     | length occs1 < length occs2
 1152     = VarOccs (zipWith (+) (occs1 ++ zeroes) occs2)
 1153   (VarOccs occs1) <> (VarOccs occs2)

 1159 -- | 'listen' to the output of a writer computation and return that output but
 1160 -- drop it from the writer output of the computation
 1161 listenDrop :: MonadWriter w m => m a -> m (a, w)
 1162 listenDrop m = pass (listen m >>= \aw -> return (aw, const mempty))

 1164 -- | The monad for sawLet minimization
 1165 type SLMinM = StateT (IntMap (Term, VarOccs)) (WriterT VarOccs IO)
 1166 
 1167 -- | Find every subterm of the form @sawLet a b rhs (\ x -> body)@ and, whenever
 1168 -- @x@ occurs at most once in @body@, unfold the @sawLet@ by substituting @rhs@
 1169 -- into @body@
 1170 sawLetMinimize :: SharedContext -> Term -> IO Term
 1171 sawLetMinimize sc t_top =
 1172   fst <$> runWriterT (evalStateT (slMinTerm t_top) IntMap.empty) where
 1173   slMinTerm :: Term -> SLMinM Term
 1174   slMinTerm (Unshared tf) = slMinTermF tf
 1175   slMinTerm t@(STApp { stAppIndex = i }) =
 1176     do memo_table <- get
 1177        case IntMap.lookup i memo_table of
 1178          Just (t', occs) ->
 1179            -- NOTE: the fact that we explicitly tell occs here means that we are
 1180            -- going to double-count variable occurrences for multiple
 1181            -- occurrences of the same subterm. That is, a variable occurence
 1182            -- counts for each copy of a shared subterm.
 1183            tell occs >> return t'
 1184          Nothing ->
 1185            do (t', occs) <- listen $ slMinTermF (unwrapTermF t)
 1186               modify $ IntMap.insert i (t', occs)
 1187               return t'
 1188 
 1189   slMinTermF :: TermF Term -> SLMinM Term
 1190   slMinTermF tf@(App (asApplyAll ->
 1191                       (isGlobalDef "Prelude.sawLet" -> Just _, [_a, _b, rhs]))
 1192                  (asLambda -> Just (_, _, body))) =
 1193     do (body', (unconsVarOccs ->
 1194                 (x_cnt, body_occs))) <- listenDrop $ slMinTerm body
 1195        if x_cnt > 1 then slMinTermF' tf else
 1196          do (rhs', rhs_occs) <- listenDrop $ slMinTerm rhs
 1197             tell (multVarOccs x_cnt rhs_occs <> body_occs)
 1198             liftIO $ instantiateVar sc 0 rhs' body'
 1199   slMinTermF tf = slMinTermF' tf
 1200 
 1201   slMinTermF' :: TermF Term -> SLMinM Term
 1202   slMinTermF' (FTermF ftf) = slMinFTermF ftf
 1203   slMinTermF' (App f arg) =
 1204     do f' <- slMinTerm f
 1205        arg' <- slMinTerm arg
 1206        liftIO $ scTermF sc (App f' arg')
 1207   slMinTermF' (Lambda x tp body) =
 1208     do tp' <- slMinTerm tp
 1209        (body', body_occs) <- listenDrop $ slMinTerm body
 1210        tell $ snd $ unconsVarOccs body_occs
 1211        liftIO $ scTermF sc (Lambda x tp' body')
 1212   slMinTermF' (Pi x tp body) =
 1213     do tp' <- slMinTerm tp
 1214        (body', body_occs) <- listenDrop $ slMinTerm body
 1215        tell $ snd $ unconsVarOccs body_occs
 1216        liftIO $ scTermF sc (Pi x tp' body')
 1217   slMinTermF' tf@(LocalVar i) =
 1218     tell (varOccs1 i) >> liftIO (scTermF sc tf)
 1219   slMinTermF' tf@(Constant _) = liftIO (scTermF sc tf)
 1220 
 1221   slMinFTermF :: FlatTermF Term -> SLMinM Term
 1222   slMinFTermF ftf@(ExtCns _) = liftIO $ scFlatTermF sc ftf
 1223   slMinFTermF ftf = traverse slMinTerm ftf >>= liftIO . scFlatTermF sc
