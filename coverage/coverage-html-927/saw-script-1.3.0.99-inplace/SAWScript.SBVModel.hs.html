   19 -- Auxiliary Cryptol types
   20 type Major = Int
   21 type Minor = Int
   22 data Version = Version Major Minor deriving Show
   23 type TyCon = String
   24 type Name = String
   25 data Loc = Loc String Int{-line-} Int{-col-}
   26          | LocPrim String
   27          | LocDefault String
   28          | LocExpr
   29          | LocInternal
   30          | LocNone
   31          deriving Show
   32 data Scheme a = Scheme [String] [a] [Predicate a] (Type a) deriving Show
   33 
   34 schemeType :: Scheme a -> Type a
   35 schemeType (Scheme _ _ _ t) = t
   36 
   37 embedScheme :: Type a -> Scheme a
   38 embedScheme t = Scheme [] [] [] t
   39 data Predicate a
   40     = PredEq    (Type a) (Type a) (Maybe (Predicate a)) [Loc]
   41     | PredSubst a (Type a)                              [Loc]
   42     | PredGrEq  (Type a) (Type a)                       [Loc]
   43     | PredFin (Type a)                                  [Loc]
   44     | PredImplied (Predicate a)
   45     deriving Show
   46 data Type a
   47     = TVar a
   48     | TInt Integer
   49     | TApp TyCon [Type a]
   50     | TRecord [(Name, Scheme a)]
   51     deriving Show
   52 type IRTyVar  = TyVar Int
   53 data TyVar a
   54   = TyVar { tyvarName :: a, tyvarKind :: Kind
   55           , tyvarIgnorable :: Bool, tyvarDefaultable :: Bool }
   56    deriving Show
   57 type IRType   = Type IRTyVar
   58 data Kind
   59     = KStar
   60     | KSize
   61     | KShape
   62     deriving Show
   63 -- SBV programs
   64 type Size      = Integer
   65 type GroundVal = Integer
   66 type NodeId    = Int
   67 data SBV = SBV !Size !(Either GroundVal NodeId) deriving Show
   68 data SBVCommand = Decl   CtrlPath !SBV (Maybe SBVExpr)
   69                 | Output !SBV
   70                 deriving Show
   71 data Operator = BVAdd | BVSub | BVMul | BVDiv Loc | BVMod Loc | BVPow
   72               | BVIte
   73               | BVShl | BVShr | BVRol | BVRor
   74               | BVExt !Integer !Integer  -- hi lo
   75               | BVAnd | BVOr | BVXor | BVNot
   76               | BVEq  | BVGeq | BVLeq | BVGt | BVLt
   77               | BVApp   -- append
   78               -- following is not essential, but simplifies lookups
   79               | BVLkUp !Integer !Integer -- index size, result size
   80               -- Uninterpreted constants
   81               | BVUnint Loc [(String, String)] (String, IRType) -- loc, code-gen info, name/type
   82               deriving Show
   83 data SBVExpr = SBVAtom !SBV
   84              | SBVApp  !Operator ![SBV]
   85              deriving Show
   86 type CtrlPath = [Either SBV SBV]
   87 type VC = ( CtrlPath    -- path to the VC
   88           , SBVExpr     -- stipulated trap; need to prove this expr is not true
   89           , String)     -- associated error message
   90 data SBVPgm = SBVPgm (Version, IRType, [SBVCommand], [VC]
   91                      , [String] -- warnings
   92                      , [((String, Loc), IRType, [(String, String)])] -- uninterpeted functions and code
   93                      )
   94             deriving Show
   95 -- make it an instance of Binary
   96 instance Binary Loc where
   97   put (Loc s i j)    = putWord8 0 >> put s >> put i >> put j
   98   put (LocPrim s)    = putWord8 1 >> put s
   99   put (LocDefault s) = putWord8 2 >> put s
  100   put LocExpr        = putWord8 3
  101   put LocInternal    = putWord8 4
  102   put LocNone        = putWord8 5
  103   get = do tag <- getWord8
  104            case tag of
  105              0 -> liftM3 Loc get get get
  106              1 -> liftM LocPrim get
  107              2 -> liftM LocDefault get
  108              3 -> return LocExpr
  109              4 -> return LocInternal
  110              5 -> return LocNone
  111              n -> error $ "Binary.Loc: " ++ show n
  112 instance Binary Version where
  113    put (Version ma mi) = put ma >> put mi
  114    get = liftM2 Version get get
  115 instance Binary SBVPgm where
  116   put (SBVPgm cmds) = put cmds
  117   get = liftM SBVPgm get
  118 
  119 -- only monomorphic types supported below
  120 sortUnder :: (Ord b) => (a -> b) -> [a] -> [a]
  121 sortUnder f = sortBy (\a b -> compare (f a) (f b))
  122 
  123 instance Binary IRType where
  124   put (TInt i)      = putWord8 0 >> put i
  125   put (TApp t ts)   = putWord8 1 >> put t >> put ts
  126   put (TRecord nss) = putWord8 2 >> put [(n, schemeType s) | (n, s) <- sortUnder fst nss]
  127   put TVar{} = error $ "internal: put not defined on TVar"
  128   get = do tag <- getWord8
  129            case tag of
  130              0 -> liftM TInt get
  131              1 -> liftM2 TApp get get
  132              2 -> do nts <- get
  133                      return (TRecord [(n, embedScheme t) | (n, t) <- nts])
  134              n -> error $ "Binary.IRType: " ++ show n
  135 instance Binary SBV where
  136   put (SBV i esn) = put i >> put esn
  137   get             = liftM2 SBV get get
  138 instance Binary SBVCommand where
  139   put (Decl path v mbe) = putWord8 0 >> put path >> put v >> put mbe
  140   put (Output v)        = putWord8 1 >> put v
  141   get = do tag <- getWord8
  142            case tag of
  143              0 -> liftM3 Decl get get get
  144              1 -> liftM  Output get
  145              n -> error $ "Binary.SBVCommand: " ++ show n
  146 instance Binary SBVExpr where
  147   put (SBVAtom s)   = putWord8 0 >> put s
  148   put (SBVApp o es) = putWord8 1 >> put o >> put es
  149   get = do tag <- getWord8
  150            case tag of
  151              0 -> liftM  SBVAtom get
  152              1 -> liftM2 SBVApp  get get
  153              n -> error $ "Binary.SBVExpr: " ++ show n
  154 instance Binary Operator where
  155    put BVAdd = putWord8 0
  156    put BVSub = putWord8 1
  157    put BVMul = putWord8 2
  158    put (BVDiv l) = putWord8 3 >> put l
  159    put (BVMod l) = putWord8 4 >> put l
  160    put BVPow = putWord8 5
  161    put BVIte = putWord8 6
  162    put BVShl = putWord8 7
  163    put BVShr = putWord8 8
  164    put BVRol = putWord8 9
  165    put BVRor = putWord8 10
  166    put (BVExt hi lo) = putWord8 11 >> put hi >> put lo
  167    put BVAnd = putWord8 12
  168    put BVOr  = putWord8 13
  169    put BVXor = putWord8 14
  170    put BVNot = putWord8 15
  171    put BVEq  = putWord8 16
  172    put BVGeq = putWord8 17
  173    put BVLeq = putWord8 18
  174    put BVGt  = putWord8 19
  175    put BVLt  = putWord8 20
  176    put BVApp = putWord8 21
  177    put (BVLkUp i r) = putWord8 22 >> put i >> put r
  178    put (BVUnint l cgs nt) = putWord8 23 >> put l >> put cgs >> put nt
  179    get = do tag <- getWord8
  180             case tag of
  181               0 -> return BVAdd
  182               1 -> return BVSub
  183               2 -> return BVMul
  184               3 -> liftM BVDiv get
  185               4 -> liftM BVMod get
  186               5 -> return BVPow
  187               6 -> return BVIte
  188               7 -> return BVShl
  189               8 -> return BVShr
  190               9 -> return BVRol
  191               10 -> return BVRor
  192               11 -> liftM2 BVExt get get
  193               12 -> return BVAnd
  194               13 -> return BVOr
  195               14 -> return BVXor
  196               15 -> return BVNot
  197               16 -> return BVEq
  198               17 -> return BVGeq
  199               18 -> return BVLeq
  200               19 -> return BVGt
  201               20 -> return BVLt
  202               21 -> return BVApp
  203               22 -> liftM2 BVLkUp get get
  204               23 -> liftM3 BVUnint get get get
  205               n  -> error $ "Binary.Operator: " ++ show n
  206 loadSBV :: FilePath -> IO SBVPgm
  207 loadSBV f = do
  208         h <- openBinaryFile f ReadMode
  209         pgm <- B.hGetContents h
  210         B.length pgm `seq` hClose h   -- oh, the horror..
  211         return (decode pgm)
