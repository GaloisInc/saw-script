   89 isVUnit :: Value -> Bool
   90 isVUnit (VTuple []) = True
   91 isVUnit _ = False

   93 indexValue :: Value -> Value -> Value
   94 indexValue (VArray vs) (VInteger x)
   95     | i < length vs = vs !! i
   96     | otherwise = error "array index out of bounds"
   97     where i = fromInteger x
   98 indexValue _ _ = error "indexValue"
   99 
  100 lookupValue :: Value -> Text -> Value
  101 lookupValue (VRecord vm) name =
  102     case M.lookup name vm of
  103       Nothing -> error $ "no such record field: " ++ unpack name
  104       Just x -> x
  105 lookupValue _ _ = error "lookupValue"
  106 
  107 tupleLookupValue :: Value -> Integer -> Value
  108 tupleLookupValue (VTuple vs) i
  109   | 0 <= i && fromIntegral i < length vs = vs !! fromIntegral i
  110   | otherwise = error $ "no such tuple index: " ++ show i
  111 tupleLookupValue _ _ = error "tupleLookupValue"

  113 -- NB, the precise locations of VTopLevel constructors in this
  114 --  operator are rather delicate, which is why we write it out
  115 --  here explicitly.
  116 toplevelCallCC :: Value
  117 toplevelCallCC = VLambda body
  118   where
  119    body (VLambda m) =
  120      callCC $ \(k :: Value -> TopLevel Value) ->
  121            m (VLambda (\v -> return (VTopLevel (k ((VTopLevel (return v)))))))
  122    body _ = error "toplevelCallCC : expected lambda"
  123 
  124 toplevelSubshell :: Value
  125 toplevelSubshell = VLambda $ \_ ->
  126   do m <- roSubshell <$> ask
  127      env <- getLocalEnv
  128      return (VTopLevel (toValue <$> withLocalEnv env m))
  129 
  130 proofScriptSubshell :: Value
  131 proofScriptSubshell = VLambda $ \_ ->
  132   do m <- roProofSubshell <$> ask
  133      env <- getLocalEnv
  134      return (VProofScript (toValue <$> withLocalEnvProof env m))

  136 thenValue :: SS.Pos -> Value -> Value -> Value
  137 thenValue pos v1 v2 = VBind pos v1 (VLambda (const (return v2)))
  138 
  139 bindValue :: SS.Pos -> Value -> Value -> TopLevel Value
  140 bindValue pos v1 v2 = return (VBind pos v1 v2)
  141 
  142 forValue :: [Value] -> Value -> TopLevel Value
  143 forValue [] _ = return $ VReturn (VArray [])
  144 forValue (x : xs) f =
  145   do m1 <- applyValue f x
  146      m2 <- forValue xs f
  147      bindValue (SS.PosInternal "<for>") m1 (VLambda $ \v1 ->
  148        bindValue (SS.PosInternal "<for>") m2 (VLambda $ \v2 ->
  149          return $ VReturn (VArray (v1 : fromValue v2))))

  157 -- | A version of 'Control.Exception.bracket' specialized to 'TopLevel'. We
  158 -- can't use 'Control.Monad.Catch.bracket' because it requires 'TopLevel' to
  159 -- implement 'Control.Monad.Catch.MonadMask', which it can't do.
  160 bracketTopLevel :: TopLevel a -> (a -> TopLevel b) -> (a -> TopLevel c) -> TopLevel c
  161 bracketTopLevel acquire release action =
  162   do  resource <- acquire
  163       try (action resource) >>= \case
  164         Left (bad :: X.SomeException) -> release resource >> throwM bad
  165         Right good -> release resource >> pure good

  167 combineRW :: TopLevelCheckpoint -> TopLevelRW -> IO TopLevelRW
  168 combineRW (TopLevelCheckpoint chk scc) rw =
  169   do cenv' <- CEnv.combineCryptolEnv (rwCryptol chk) (rwCryptol rw)
  170      sc' <- restoreSharedContext scc (rwSharedContext rw)
  171      return chk{ rwCryptol = cenv'
  172                , rwSharedContext = sc'
  173                }

  175 -- | Represents the mutable state of the TopLevel monad
  176 --   that can later be restored.
  177 data TopLevelCheckpoint =
  178   TopLevelCheckpoint
  179     TopLevelRW
  180     SharedContextCheckpoint
  181 
  182 makeCheckpoint :: TopLevelRW -> IO TopLevelCheckpoint
  183 makeCheckpoint rw =
  184   do scc <- checkpointSharedContext (rwSharedContext rw)
  185      return (TopLevelCheckpoint rw scc)
  186 
  187 restoreCheckpoint :: TopLevelCheckpoint -> TopLevel ()
  188 restoreCheckpoint chk =
  189   do rw <- getTopLevelRW
  190      rw' <- io (combineRW chk rw)
  191      putTopLevelRW rw'

  193 -- | Capture the current state of the TopLevel monad
  194 --   and return an action that, if invoked, resets
  195 --   the state back to that point.
  196 checkpoint :: TopLevel (() -> TopLevel ())
  197 checkpoint = TopLevel_ $
  198   do chk <- liftIO . makeCheckpoint =<< get
  199      return $ \_ ->
  200        do printOutLnTop Info "Restoring state from checkpoint"
  201           restoreCheckpoint chk
  202 
  203 -- | Capture the current proof state and return an
  204 --   action that, if invoked, resets the state back to that point.
  205 proof_checkpoint :: ProofScript (() -> ProofScript ())
  206 proof_checkpoint =
  207   do ps <- get
  208      return $ \_ ->
  209        do scriptTopLevel (printOutLnTop Info "Restoring proof state from checkpoint")
  210           put ps

  212 withLocalEnv :: LocalEnv -> TopLevel a -> TopLevel a
  213 withLocalEnv env m = do
  214   prevEnv <- gets rwLocalEnv
  215   modify (\rw -> rw { rwLocalEnv = env })
  216   result <- m
  217   modify (\rw -> rw { rwLocalEnv = prevEnv })
  218   return result
  219 
  220 withLocalEnvProof :: LocalEnv -> ProofScript a -> ProofScript a
  221 withLocalEnvProof env (ProofScript m) = do
  222   ProofScript (underExceptT (underStateT (withLocalEnv env)) m)
  223 
  224 withOptions :: (Options -> Options) -> TopLevel a -> TopLevel a
  225 withOptions f (TopLevel_ m) =
  226   TopLevel_ (local (\x -> x {roOptions = f (roOptions x)}) m)

  228 -- | Implement stack tracing by adding error handlers that rethrow
  229 -- user errors, prepended with the given string.
  230 withStackTraceFrame :: String -> Value -> Value
  231 withStackTraceFrame str val =
  232   let doTopLevel :: TopLevel a -> TopLevel a
  233       doTopLevel action = do
  234         trace <- gets rwStackTrace
  235         modify (\rw -> rw { rwStackTrace = str : trace })
  236         result <- action
  237         modify (\rw -> rw { rwStackTrace = trace })
  238         return result
  239       doProofScript :: ProofScript a -> ProofScript a
  240       doProofScript (ProofScript m) =
  241         let m' =
  242               underExceptT (underStateT doTopLevel) m
  243         in
  244         ProofScript m'
  245       doLLVM :: LLVMCrucibleSetupM a -> LLVMCrucibleSetupM a
  246       doLLVM (LLVMCrucibleSetupM m) =
  247         LLVMCrucibleSetupM (underReaderT (underStateT doTopLevel) m)
  248       doJVM :: JVMSetupM a -> JVMSetupM a
  249       doJVM (JVMSetupM m) =
  250         JVMSetupM (underReaderT (underStateT doTopLevel) m)
  251       doMIR :: MIRSetupM a -> MIRSetupM a
  252       doMIR (MIRSetupM m) =
  253         MIRSetupM (underReaderT (underStateT doTopLevel) m)
  254   in
  255   case val of
  256     VLambda f ->
  257       let wrap x =
  258             withStackTraceFrame str `fmap` doTopLevel (f x)
  259       in
  260       VLambda wrap
  261     VTopLevel m ->
  262       let m' =
  263             withStackTraceFrame str `fmap` doTopLevel m
  264       in
  265       VTopLevel m'
  266     VProofScript m ->
  267       let m' =
  268             withStackTraceFrame str `fmap` doProofScript m
  269       in
  270       VProofScript m'
  271     VBind pos v1 v2 ->
  272       let v1' = withStackTraceFrame str v1
  273           v2' = withStackTraceFrame str v2
  274       in
  275       VBind pos v1' v2'
  276     VLLVMCrucibleSetup m ->
  277       let m' =
  278             withStackTraceFrame str `fmap` doLLVM m
  279       in
  280       VLLVMCrucibleSetup m'
  281     VJVMSetup m ->
  282       let m' =
  283             withStackTraceFrame str `fmap` doJVM m
  284       in
  285       VJVMSetup m'
  286     VMIRSetup m ->
  287       let m' =
  288             withStackTraceFrame str `fmap` doMIR m
  289       in
  290       VMIRSetup m'
  291     _ ->
  292       val
