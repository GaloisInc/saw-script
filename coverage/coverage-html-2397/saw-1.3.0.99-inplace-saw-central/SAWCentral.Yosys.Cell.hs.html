   34 -- | A SAWCore bitvector term along with its width and whether it should be interpreted as signed.
   35 data CellTerm = CellTerm
   36   { cellTermTerm :: SC.Term
   37   , cellTermWidth :: Natural
   38   , cellTermSigned :: Bool
   39   }
   40 
   41 cellTermNat :: forall m. MonadIO m => SC.SharedContext -> CellTerm -> m SC.Term
   42 cellTermNat sc (CellTerm { cellTermTerm = t, cellTermWidth = w }) = liftIO $ SC.scBvToNat sc w t
   43 
   44 -- | Apply the appropriate (possibly signed) extension or truncation to make the given bitvector
   45 -- match the given width.
   46 extTrunc :: forall m. MonadIO m => SC.SharedContext -> Natural -> CellTerm -> m CellTerm
   47 extTrunc sc width (CellTerm { cellTermTerm = t, cellTermWidth = bvw, cellTermSigned = signed}) = do
   48   wterm <- liftIO $ SC.scNat sc width
   49   bvwterm <- liftIO $ SC.scNat sc bvw
   50   res <- if
   51     | bvw > width -> do
   52         diffterm <- liftIO . SC.scNat sc $ bvw - width
   53         liftIO $ SC.scBvTrunc sc diffterm wterm t
   54     | width > bvw && signed -> do
   55         bvwpredterm <- liftIO . SC.scNat sc $ bvw - 1
   56         diffterm <- liftIO . SC.scNat sc $ width - bvw
   57         liftIO $ SC.scBvSExt sc diffterm bvwpredterm t
   58     | width > bvw && not signed -> do
   59         diffterm <- liftIO . SC.scNat sc $ width - bvw
   60         liftIO $ SC.scBvUExt sc diffterm bvwterm t
   61     | otherwise -> pure t
   62   pure $ CellTerm res width signed
   63 
   64 -- | Given two bitvectors, extend the narrower bitvector to match the wider.
   65 extMax :: forall m. MonadIO m => SC.SharedContext -> CellTerm -> CellTerm -> m (CellTerm, CellTerm)
   66 extMax sc c1 c2 = do
   67   let
   68     w1 = cellTermWidth c1
   69     w2 = cellTermWidth c2
   70     w = max w1 w2
   71   res1 <- extTrunc sc w c1
   72   res2 <- extTrunc sc w c2
   73   pure (res1, res2)
   74 
   75 liftUnary :: forall m.
   76   MonadIO m =>
   77   SC.SharedContext ->
   78   (SC.Term -> SC.Term -> IO SC.Term) -> -- (w : Nat) -> [w] -> [w]
   79   CellTerm -> m CellTerm
   80 liftUnary sc f c@(CellTerm { cellTermTerm = t }) = do
   81   wt <- liftIO . SC.scNat sc $ cellTermWidth c
   82   res <- liftIO $ f wt t
   83   pure $ c { cellTermTerm = res }
   84 
   85 liftBinary :: forall m.
   86   MonadIO m =>
   87   SC.SharedContext ->
   88   (SC.Term -> SC.Term -> SC.Term -> IO SC.Term) -> -- (w : Nat) -> [w] -> [w] -> [w]
   89   CellTerm -> CellTerm -> m CellTerm
   90 liftBinary sc f c1@(CellTerm { cellTermTerm = t1 }) (CellTerm { cellTermTerm = t2 }) = do
   91   wt <- liftIO . SC.scNat sc $ cellTermWidth c1
   92   res <- liftIO $ f wt t1 t2
   93   pure $ c1 { cellTermTerm = res }
   94 
   95 liftBinaryCmp :: forall m.
   96   MonadIO m =>
   97   SC.SharedContext ->
   98   (SC.Term -> SC.Term -> SC.Term -> IO SC.Term) -> -- (w : Nat) -> [w] -> [w] -> Bool
   99   CellTerm -> CellTerm -> m SC.Term
  100 liftBinaryCmp sc f c1@(CellTerm { cellTermTerm = t1 }) (CellTerm { cellTermTerm = t2 }) = do
  101   wt <- liftIO . SC.scNat sc $ cellTermWidth c1
  102   liftIO $ f wt t1 t2

  104 -- | Given a primitive Yosys cell and a map of terms for its arguments, construct a record term representing the output.
  105 -- If the provided cell is not a primitive, return Nothing.
  106 primCellToTerm ::
  107   forall m b.
  108   MonadIO m =>
  109   SC.SharedContext ->
  110   Cell [b] {- ^ Cell type -} ->
  111   Map Text SC.Term {- ^ Mapping of input names to input terms -} ->
  112   m (Maybe SC.Term)
  113 primCellToTerm sc c args = do
  114   mm <- primCellToMap sc c args
  115   mt <- traverse (cryptolRecord sc) mm
  116   traverse (validateTerm sc typeCheckMsg) mt
  117   where
  118     typeCheckMsg :: Text
  119     typeCheckMsg = mconcat
  120       [ "translating a cell with type \""
  121       , Text.pack $ show $ c ^. cellType
  122       , "\""
  123       ]
  124 
  125 primCellToMap ::
  126   forall m b.
  127   MonadIO m =>
  128   SC.SharedContext ->
  129   Cell [b] {- ^ Cell type -} ->
  130   Map Text SC.Term {- ^ Mapping of input names to input terms -} ->
  131   m (Maybe (Map Text SC.Term))
  132 primCellToMap sc c args = case c ^. cellType of
  133   CellTypeNot -> bvUnaryOp . liftUnary sc $ SC.scBvNot sc
  134   CellTypePos -> do
  135     res <- input "A"
  136     output res
  137   CellTypeNeg -> bvUnaryOp . liftUnary sc $ SC.scBvNeg sc
  138   CellTypeAnd -> bvBinaryOp . liftBinary sc $ SC.scBvAnd sc
  139   CellTypeOr -> bvBinaryOp . liftBinary sc $ SC.scBvOr sc
  140   CellTypeXor -> bvBinaryOp . liftBinary sc $ SC.scBvXor sc
  141   CellTypeXnor -> bvBinaryOp . liftBinary sc $ \w x y -> do
  142     r <- SC.scBvXor sc w x y
  143     SC.scBvNot sc w r
  144   CellTypeReduceAnd -> bvReduce True =<< do
  145     liftIO . SC.scLookupDef sc $ SC.mkIdent SC.preludeName "and"
  146   CellTypeReduceOr -> bvReduce False =<< do
  147     liftIO . SC.scLookupDef sc $ SC.mkIdent SC.preludeName "or"
  148   CellTypeReduceXor -> bvReduce False =<< do
  149     liftIO . SC.scLookupDef sc $ SC.mkIdent SC.preludeName "xor"
  150   CellTypeReduceXnor -> bvReduce True =<< do
  151     boolTy <- liftIO $ SC.scBoolType sc
  152     xEC <- liftIO $ SC.scFreshEC sc "x" boolTy
  153     x <- liftIO $ SC.scExtCns sc xEC
  154     yEC <- liftIO $ SC.scFreshEC sc "y" boolTy
  155     y <- liftIO $ SC.scExtCns sc yEC
  156     r <- liftIO $ SC.scXor sc x y
  157     res <- liftIO $ SC.scNot sc r
  158     liftIO $ SC.scAbstractExts sc [xEC, yEC] res
  159   CellTypeReduceBool -> bvReduce False =<< do
  160     liftIO . SC.scLookupDef sc $ SC.mkIdent SC.preludeName "or"
  161   CellTypeShl -> do
  162     ta <- fmap cellTermTerm $ input "A"
  163     nb <- cellTermNat sc =<< input "B"
  164     w <- outputWidth
  165     res <- liftIO $ SC.scBvShl sc w ta nb
  166     output (CellTerm res (connWidthNat "A") (connSigned "A"))
  167   CellTypeShr -> do
  168     ta <- fmap cellTermTerm $ input "A"
  169     nb <- cellTermNat sc =<< input "B"
  170     w <- outputWidth
  171     res <- liftIO $ SC.scBvShr sc w ta nb
  172     output (CellTerm res (connWidthNat "A") (connSigned "A"))
  173   CellTypeSshl -> do
  174     ta <- fmap cellTermTerm $ input "A"
  175     nb <- cellTermNat sc =<< input "B"
  176     w <- outputWidth
  177     res <- liftIO $ SC.scBvShl sc w ta nb
  178     output (CellTerm res (connWidthNat "A") (connSigned "A"))
  179   CellTypeSshr -> do
  180     ta <- fmap cellTermTerm $ input "A"
  181     nb <- cellTermNat sc =<< input "B"
  182     w <- outputWidth
  183     res <- liftIO $ SC.scBvSShr sc w ta nb
  184     output (CellTerm res (connWidthNat "A") (connSigned "A"))
  185   -- "$shift" -> _
  186   CellTypeShiftx -> do
  187     let w = max (connWidthNat "A") (connWidthNat "B")
  188     wt <- liftIO $ SC.scNat sc w
  189     CellTerm ta _ _ <- extTrunc sc w =<< input "A"
  190     CellTerm tb _ _ <- extTrunc sc w =<< input "B"
  191     zero <- liftIO $ SC.scBvConst sc w 0
  192     tbn <- liftIO $ SC.scBvToNat sc w tb
  193     tbneg <- liftIO $ SC.scBvNeg sc wt tb
  194     tbnegn <- liftIO $ SC.scBvToNat sc w tbneg
  195     cond <- liftIO $ SC.scBvSGe sc wt tb zero
  196     tcase <- liftIO $ SC.scBvShr sc wt ta tbn
  197     ecase <- liftIO $ SC.scBvShl sc wt ta tbnegn
  198     ty <- liftIO . SC.scBitvector sc $ connWidthNat "A"
  199     res <- if connSigned "B"
  200       then liftIO $ SC.scIte sc ty cond tcase ecase
  201       else pure tcase
  202     output (CellTerm res (connWidthNat "A") (connSigned "A"))
  203   CellTypeLt -> bvBinaryCmp . liftBinaryCmp sc $ SC.scBvULt sc
  204   CellTypeLe -> bvBinaryCmp . liftBinaryCmp sc $ SC.scBvULe sc
  205   CellTypeGt -> bvBinaryCmp . liftBinaryCmp sc $ SC.scBvUGt sc
  206   CellTypeGe -> bvBinaryCmp . liftBinaryCmp sc $ SC.scBvUGe sc
  207   CellTypeEq -> bvBinaryCmp . liftBinaryCmp sc $ SC.scBvEq sc
  208   CellTypeNe -> bvBinaryCmp . liftBinaryCmp sc $ \w x y -> do
  209     r <- SC.scBvEq sc w x y
  210     SC.scNot sc r
  211   CellTypeEqx -> bvBinaryCmp . liftBinaryCmp sc $ SC.scBvEq sc
  212   CellTypeNex -> bvBinaryCmp . liftBinaryCmp sc $ \w x y -> do
  213     r <- SC.scBvEq sc w x y
  214     SC.scNot sc r
  215   CellTypeAdd -> bvBinaryOp . liftBinary sc $ SC.scBvAdd sc
  216   CellTypeSub -> bvBinaryOp . liftBinary sc $ SC.scBvSub sc
  217   CellTypeMul -> bvBinaryOp . liftBinary sc $ SC.scBvMul sc
  218   CellTypeDiv -> bvBinaryOp . liftBinary sc $ SC.scBvUDiv sc
  219   CellTypeMod -> bvBinaryOp . liftBinary sc $ SC.scBvURem sc
  220   -- "$modfloor" -> _
  221   CellTypeLogicNot -> do
  222     w <- connWidth "A"
  223     ta <- cellTermTerm <$> input "A"
  224     anz <- liftIO $ SC.scBvNonzero sc w ta
  225     res <- liftIO $ SC.scNot sc anz
  226     outputBit res
  227   CellTypeLogicAnd -> bvBinaryCmp . liftBinaryCmp sc $ \w x y -> do
  228     xnz <- liftIO $ SC.scBvNonzero sc w x
  229     ynz <- liftIO $ SC.scBvNonzero sc w y
  230     liftIO $ SC.scAnd sc xnz ynz
  231   CellTypeLogicOr -> bvBinaryCmp . liftBinaryCmp sc $ \w x y -> do
  232     xnz <- liftIO $ SC.scBvNonzero sc w x
  233     ynz <- liftIO $ SC.scBvNonzero sc w y
  234     liftIO $ SC.scOr sc xnz ynz
  235   CellTypeMux -> do
  236     ta <- cellTermTerm <$> input "A"
  237     tb <- cellTermTerm <$> input "B"
  238     ts <- cellTermTerm <$> input "S"
  239     swidth <- connWidth "S"
  240     snz <- liftIO $ SC.scBvNonzero sc swidth ts
  241     let width = connWidthNat "Y"
  242     ty <- liftIO $ SC.scBitvector sc width
  243     res <- liftIO $ SC.scIte sc ty snz tb ta
  244     output $ CellTerm res (connWidthNat "A") (connSigned "A")
  245   CellTypePmux -> do
  246     ta <- cellTermTerm <$> input "A"
  247     tb <- cellTermTerm <$> input "B"
  248     ts <- cellTermTerm <$> input "S"
  249 
  250     width <- connWidth "A"
  251     widthBv <- liftIO . SC.scBitvector sc $ connWidthNat "A"
  252     swidth <- connWidth "S"
  253     bool <- liftIO $ SC.scBoolType sc
  254     nat <- liftIO $ SC.scNatType sc
  255     splitb <- liftIO $ SC.scSplit sc swidth width bool tb
  256     zero <- liftIO $ SC.scNat sc 0
  257     accTy <- liftIO $ SC.scPairType sc nat widthBv
  258     defaultAcc <- liftIO $ SC.scPairValue sc zero ta
  259 
  260     bitEC <- liftIO $ SC.scFreshEC sc "bit" bool
  261     accEC <- liftIO $ SC.scFreshEC sc "acc" accTy
  262     fun <- liftIO . SC.scAbstractExts sc [bitEC, accEC] =<< do
  263       bit <- liftIO $ SC.scExtCns sc bitEC
  264       acc <- liftIO $ SC.scExtCns sc accEC
  265       idx <- liftIO $ SC.scPairLeft sc acc
  266       aval <- liftIO $ SC.scPairRight sc acc
  267       bval <- liftIO $ SC.scAtWithDefault sc swidth widthBv aval splitb idx
  268       newidx <- liftIO $ SC.scAddNat sc idx width
  269       newval <- liftIO $ SC.scIte sc widthBv bit bval aval
  270       liftIO $ SC.scPairValue sc newidx newval
  271 
  272     scFoldr <- liftIO . SC.scLookupDef sc $ SC.mkIdent SC.preludeName "foldr"
  273     resPair <- liftIO $ SC.scApplyAll sc scFoldr [bool, accTy, swidth, fun, defaultAcc, ts]
  274     res <- liftIO $ SC.scPairRight sc resPair
  275     output $ CellTerm res (connWidthNat "A") (connSigned "Y")
  276   CellTypeBmux -> do
  277     ia <- input "A"
  278     is <- input "S"
  279     let swidth = cellTermWidth is
  280     let ywidth = connWidthNat "Y"
  281     -- Split input A into chunks
  282     chunks <- liftIO (SC.scNat sc (2 ^ swidth))
  283     ywidth' <- liftIO (SC.scNat sc ywidth)
  284     bool <- liftIO (SC.scBoolType sc)
  285     splitA <- liftIO (SC.scSplit sc chunks ywidth' bool (cellTermTerm ia))
  286     -- reverse to put index 0 on the left
  287     outputType <- liftIO (SC.scBitvector sc ywidth)
  288     revA <- liftIO (SC.scGlobalApply sc "Prelude.reverse" [chunks, outputType, splitA])
  289     -- Select chunk from output
  290     ixWidth <- liftIO (SC.scNat sc swidth)
  291     elt <- liftIO (SC.scBvAt sc chunks outputType ixWidth revA (cellTermTerm is))
  292     output (CellTerm elt ywidth (connSigned "Y"))
  293   -- "$demux" -> _
  294   -- "$lut" -> _
  295   -- "$slice" -> _
  296   -- "$concat" -> _
  297   CellTypeDff -> pure Nothing
  298   CellTypeFf -> pure Nothing
  299   CellTypeUnsupportedPrimitive _ -> pure Nothing
  300   CellTypeUserType _ -> pure Nothing
  301   where
  302     nm :: Text
  303     nm = Text.pack $ show $ c ^. cellType
  304 
  305     textBinNat :: Text -> Natural
  306     textBinNat = fromIntegral . Text.foldl' (\a x -> digitToInt x + a * 2) 0
  307     connSigned :: Text -> Bool
  308     connSigned onm =
  309       case Map.lookup (onm <> "_SIGNED") $ c ^. cellParameters of
  310         Just t -> textBinNat t > 0
  311         Nothing -> False
  312     connWidthNat :: Text -> Natural
  313     connWidthNat onm =
  314       case Map.lookup onm $ c ^. cellConnections of
  315         Nothing -> panic "cellToTerm" ["Missing expected output name for " <> nm <> " cell"]
  316         Just bits -> fromIntegral $ length bits
  317     connWidth :: Text -> m SC.Term
  318     connWidth onm = liftIO . SC.scNat sc $ connWidthNat onm
  319     outputWidth = connWidth "Y"
  320 
  321     input :: Text -> m CellTerm
  322     input inpNm = case Map.lookup inpNm args of
  323       Nothing -> panic "cellToTerm" [nm <> " missing input " <> inpNm]
  324       Just a -> pure $ CellTerm a (connWidthNat inpNm) (connSigned inpNm)
  325 
  326     output :: CellTerm -> m (Maybe (Map Text SC.Term))
  327     output (CellTerm ct cw _) = do
  328       let res = CellTerm ct cw (connSigned "Y")
  329       CellTerm t _ _ <- extTrunc sc (connWidthNat "Y") res
  330       pure . Just $ Map.fromList
  331         [ ("Y", t)
  332         ]
  333 
  334     outputBit :: SC.Term -> m (Maybe (Map Text SC.Term))
  335     outputBit res = do
  336       bool <- liftIO $ SC.scBoolType sc
  337       vres <- liftIO $ SC.scSingle sc bool res
  338       pure . Just $ Map.fromList
  339         [ ("Y", vres)
  340         ]
  341 
  342     -- convert input to big endian
  343     bvUnaryOp :: (CellTerm -> m CellTerm) -> m (Maybe (Map Text SC.Term))
  344     bvUnaryOp f = do
  345       t <- input "A"
  346       res <- f t
  347       output res
  348     -- extend inputs to output width
  349     bvBinaryOp :: (CellTerm -> CellTerm -> m CellTerm) -> m (Maybe (Map Text SC.Term))
  350     bvBinaryOp f = do
  351       let w = connWidthNat "Y"
  352       ta <- extTrunc sc w =<< input "A"
  353       tb <- extTrunc sc w =<< input "B"
  354       res <- f ta tb
  355       output res
  356     -- extend inputs to max input width, output is a single bit
  357     bvBinaryCmp :: (CellTerm -> CellTerm -> m SC.Term) -> m (Maybe (Map Text SC.Term))
  358     bvBinaryCmp f = do
  359       ta <- input "A"
  360       tb <- input "B"
  361       res <- uncurry f =<< extMax sc ta tb
  362       outputBit res
  363     bvReduce :: Bool -> SC.Term -> m (Maybe (Map Text SC.Term))
  364     bvReduce boolIdentity boolFun = do
  365       CellTerm t _ _ <- input "A"
  366       w <- connWidth "A"
  367       boolTy <- liftIO $ SC.scBoolType sc
  368       identity <- liftIO $ SC.scBool sc boolIdentity
  369       scFoldr <- liftIO . SC.scLookupDef sc $ SC.mkIdent SC.preludeName "foldr"
  370       bit <- liftIO $ SC.scApplyAll sc scFoldr [boolTy, boolTy, w, boolFun, identity, t]
  371       outputBit bit
