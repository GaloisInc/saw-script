   48 -- Values and Thunks

   50 {- | SAWCore Values. Values are parameterized by the /name/ of an
   51 instantiation.
   52 The concrete parameters to use are computed from the name using
   53 a collection of type families (e.g., 'EvalM', 'VBool', etc.). -}
   54 data Value l
   55   = VFun !(Thunk l -> MValue l)
   56   | VUnit
   57   | VPair (Thunk l) (Thunk l) -- TODO: should second component be strict?
   58   | VCtorApp !Name !(TValue l) ![Thunk l] ![Thunk l]
   59   | VCtorMux ![Thunk l] !(IntMap (VBool l, Name, TValue l, [Thunk l]))
   60     -- ^ A mux tree of possible constructor values of a data type.
   61     -- The list of data type parameters is kept outside the mux.
   62     -- The 'IntMap' keys are 'VarIndex'es of each constructor name.
   63     -- The 'VBool' predicates must be mutually-exclusive and one
   64     -- must always be true.
   65   | VVector !(Vector (Thunk l))
   66   | VBool (VBool l)
   67   | VWord (VWord l)
   68   | VBVToNat !Int (Value l) -- TODO: don't use @Int@ for this, use @Natural@
   69   | VIntToNat (Value l)
   70   | VNat !Natural
   71   | VInt (VInt l)
   72   | VIntMod !Natural (VInt l)
   73   | VArray (VArray l)
   74   | VString !Text
   75   | VRecordValue ![(FieldName, Thunk l)]
   76   | VExtra (Extra l)
   77   | TValue (TValue l)
   78 
   79 data VRecursor l
   80   = VRecursor
   81      !Name -- data type name
   82      ![Value l]  -- data type parameters
   83      !Int        -- number of index parameters
   84      ![Name]     -- constructor order
   85      !(Value l)  -- motive function
   86 
   87 -- | The subset of values that represent types.
   88 data TValue l
   89   = VVecType !Natural !(TValue l)
   90   | VBoolType
   91   | VIntType
   92   | VIntModType !Natural
   93   | VArrayType !(TValue l) !(TValue l)
   94   | VPiType !(TValue l) !(PiBody l)
   95   | VStringType
   96   | VUnitType
   97   | VPairType !(TValue l) !(TValue l)
   98   | VDataType !Name !(TValue l) ![Value l] ![Value l]
   99   | VRecordType ![(FieldName, TValue l)]
  100   | VSort !Sort
  101   | VTyTerm !Sort !Term
  102 
  103 data PiBody l
  104   = VDependentPi !(Thunk l -> EvalM l (TValue l))
  105   | VNondependentPi !(TValue l)
  106 
  107 type Thunk l = Lazy (EvalM l) (Value l)
  108 
  109 -- | Evaluation monad for value instantiation 'l'
  110 type family EvalM l :: Type -> Type
  111 -- | Booleans for value instantiation 'l'
  112 type family VBool l :: Type
  113 -- | Words for value instantiation 'l'
  114 type family VWord l :: Type
  115 -- | Integers for value instantiation 'l'
  116 type family VInt  l :: Type
  117 -- | SMT arrays for value instantiation 'l'
  118 type family VArray l :: Type
  119 -- | Additional constructors for instantiation 'l'
  120 type family Extra l :: Type
  121 
  122 -- | Short-hand for a monadic value.
  123 type MValue l     = EvalM l (Value l)
  124 
  125 -- | Short-hand for a monadic boolean.
  126 type MBool  l     = EvalM l (VBool l)
  127 
  128 -- | Short-hand for a monadic word.
  129 type MWord l      = EvalM l (VWord l)
  130 
  131 -- | Short-hand for a monadic integer.
  132 type MInt l       = EvalM l (VInt  l)
  133 
  134 -- | Short-hand for a monadic array.
  135 type MArray l     = EvalM l (VArray l)
  136 
  137 -- | Short hand to specify that the evaluation monad is a monad (very common)
  138 type VMonad l     = Monad (EvalM l)
  139 
  140 -- | Short hand to specify that the evaluation monad is a lazy monad.
  141 type VMonadLazy l = MonadLazy (EvalM l)

  146 -- | Language instantiations with a specific monad.
  147 data WithM (m :: Type -> Type) l
  148 type instance EvalM (WithM m l) = m
  149 type instance VBool (WithM m l) = VBool l
  150 type instance VWord (WithM m l) = VWord l
  151 type instance VInt  (WithM m l) = VInt l
  152 type instance VArray (WithM m l) = VArray l
  153 type instance Extra (WithM m l) = Extra l

  157 instance Show (Extra l) => Show (Value l) where
  158   showsPrec p v =
  159     case v of
  160       VFun {}        -> showString "<<fun>>"
  161       VUnit          -> showString "()"
  162       VPair{}        -> showString "<<tuple>>"
  163       VCtorApp c _ty _ps _xv -> shows (toAbsoluteName (nameInfo c))
  164       VCtorMux {}    -> showString "<<constructor>>"
  165       VVector xv     -> showList (toList xv)
  166       VBool _        -> showString "<<boolean>>"
  167       VWord _        -> showString "<<bitvector>>"
  168       VBVToNat n x   -> showString "bvToNat " . shows n . showString " " . showParen True (shows x)
  169       VIntToNat x    -> showString "intToNat " . showParen True (shows x)
  170       VNat n         -> shows n
  171       VInt _         -> showString "<<integer>>"
  172       VIntMod n _    -> showString ("<<Z " ++ show n ++ ">>")
  173       VArray{}       -> showString "<<array>>"
  174       VString s      -> shows s
  175       VRecordValue [] -> showString "{}"
  176       VRecordValue ((fld,_):_) ->
  177         showString "{" . showString (Text.unpack fld) . showString " = _, ...}"
  178       VExtra x       -> showsPrec p x
  179       TValue x       -> showsPrec p x
  180     where
  181       toList = map (const Nil) . V.toList
  182 
  183 instance Show (Extra l) => Show (TValue l) where
  184   showsPrec p v =
  185     case v of
  186       VBoolType      -> showString "Bool"
  187       VStringType    -> showString "String"
  188       VIntType       -> showString "Integer"
  189       VIntModType n  -> showParen True (showString "IntMod " . shows n)
  190       VArrayType{}   -> showString "Array"
  191       VPiType t _    -> showParen True
  192                         (shows t . showString " -> ...")
  193       VUnitType      -> showString "#()"
  194       VPairType x y  -> showParen True (shows x . showString " * " . shows y)
  195       VDataType s _ ps vs
  196         | null (ps++vs) -> shows s
  197         | otherwise  -> shows s . showList (ps++vs)
  198       VRecordType [] -> showString "{}"
  199       VRecordType ((fld,_):_) ->
  200         showString "{" . showString (Text.unpack fld) . showString " :: _, ...}"
  201       VVecType n a   -> showString "Vec " . shows n
  202                         . showString " " . showParen True (showsPrec p a)
  203       VSort s        -> shows s
  204 
  205       VTyTerm _ tm   -> showString "TyTerm (" . (\x -> showTerm tm ++ x) . showString ")"
  206

  209 instance Show Nil where
  210   show Nil = "_"

  213 -- Basic operations on values
  214 
  215 vTuple :: VMonad l => [Thunk l] -> Value l
  216 vTuple [] = VUnit
  217 vTuple [_] = error "vTuple: unsupported 1-tuple"
  218 vTuple [x, y] = VPair x y
  219 vTuple (x : xs) = VPair x (ready (vTuple xs))
  220 
  221 vTupleType :: VMonad l => [TValue l] -> TValue l
  222 vTupleType [] = VUnitType
  223 vTupleType [t] = t
  224 vTupleType (t : ts) = VPairType t (vTupleType ts)
  225 
  226 valPairLeft :: (HasCallStack, VMonad l, Show (Extra l)) => Value l -> MValue l
  227 valPairLeft (VPair t1 _) = force t1
  228 valPairLeft v = panic "valPairLeft" ["Not a pair value: " <> Text.pack (show v)]
  229 
  230 valPairRight :: (HasCallStack, VMonad l, Show (Extra l)) => Value l -> MValue l
  231 valPairRight (VPair _ t2) = force t2
  232 valPairRight v = panic "valPairRight" ["Not a pair value: " <> Text.pack (show v)]
  233 
  234 vRecord :: Map FieldName (Thunk l) -> Value l
  235 vRecord m = VRecordValue (Map.assocs m)
  236 
  237 valRecordProj :: (HasCallStack, VMonad l, Show (Extra l)) => Value l -> FieldName -> MValue l
  238 valRecordProj (VRecordValue fld_map) fld
  239   | Just t <- lookup fld fld_map = force t
  240 valRecordProj v@(VRecordValue _) fld =
  241   panic "valRecordProj" [
  242       "Record field " <> Text.pack (show fld) <> " not found in value: " <>
  243           Text.pack (show v)
  244   ]
  245 valRecordProj v _ =
  246   panic "valRecordProj" ["Not a record value: " <> Text.pack (show v)]
  247 
  248 apply :: (HasCallStack, VMonad l, Show (Extra l)) => Value l -> Thunk l -> MValue l
  249 apply (VFun f) x = f x
  250 apply (TValue (VPiType _ body)) x = TValue <$> applyPiBody body x
  251 
  252 apply v _x = panic "apply" ["Not a function value: " <> Text.pack (show v)]
  253 
  254 applyAll :: (VMonad l, Show (Extra l)) => Value l -> [Thunk l] -> MValue l
  255 applyAll = foldM apply
  256 
  257 {-# INLINE applyPiBody #-}
  258 applyPiBody :: VMonad l => PiBody l -> Thunk l -> EvalM l (TValue l)
  259 applyPiBody (VDependentPi f) x    = f x
  260 applyPiBody (VNondependentPi t) _ = pure t

  262 asFiniteTypeValue :: Value l -> Maybe FiniteType
  263 asFiniteTypeValue v =
  264   case v of
  265     TValue tv -> asFiniteTypeTValue tv
  266     _ -> Nothing
  267 
  268 asFiniteTypeTValue :: TValue l -> Maybe FiniteType
  269 asFiniteTypeTValue v =
  270   case v of
  271     VBoolType -> return FTBit
  272     VVecType n v1 -> do
  273       t1 <- asFiniteTypeTValue v1
  274       return (FTVec n t1)
  275     VUnitType -> return (FTTuple [])
  276     VPairType v1 v2 -> do
  277       t1 <- asFiniteTypeTValue v1
  278       t2 <- asFiniteTypeTValue v2
  279       case t2 of
  280         FTTuple ts -> return (FTTuple (t1 : ts))
  281         _ -> return (FTTuple [t1, t2])
  282     VRecordType elem_tps ->
  283       FTRec <$> Map.fromList <$>
  284       mapM (\(fld,tp) -> (fld,) <$> asFiniteTypeTValue tp) elem_tps
  285     _ -> Nothing
  286 
  287 asFirstOrderTypeValue :: Value l -> Maybe FirstOrderType
  288 asFirstOrderTypeValue v =
  289   case v of
  290     TValue tv -> asFirstOrderTypeTValue tv
  291     _ -> Nothing
  292 
  293 asFirstOrderTypeTValue :: TValue l -> Maybe FirstOrderType
  294 asFirstOrderTypeTValue v =
  295   case v of
  296     VBoolType     -> return FOTBit
  297     VVecType n v1 -> FOTVec n <$> asFirstOrderTypeTValue v1
  298     VIntType      -> return FOTInt
  299     VIntModType m -> return (FOTIntMod m)
  300     VArrayType a b ->
  301       FOTArray <$> asFirstOrderTypeTValue a <*> asFirstOrderTypeTValue b
  302     VUnitType -> return (FOTTuple [])
  303     VPairType v1 v2 -> do
  304       t1 <- asFirstOrderTypeTValue v1
  305       t2 <- asFirstOrderTypeTValue v2
  306       case t2 of
  307         FOTTuple ts -> return (FOTTuple (t1 : ts))
  308         _ -> return (FOTTuple [t1, t2])
  309     VRecordType elem_tps ->
  310       FOTRec . Map.fromList <$>
  311         mapM (traverse asFirstOrderTypeTValue) elem_tps
  312 
  313     VStringType   -> Nothing
  314     VPiType{}   -> Nothing
  315     VDataType{} -> Nothing
  316     VSort{}     -> Nothing
  317     VTyTerm{}   -> Nothing
  318 
  319 -- | A (partial) injective mapping from type values to strings. These
  320 -- are intended to be useful as suffixes for names of type instances
  321 -- of uninterpreted constants.
  322 suffixTValue :: TValue sym -> Maybe String
  323 suffixTValue tv =
  324   case tv of
  325     VVecType n a ->
  326       do a' <- suffixTValue a
  327          Just ("_Vec_" ++ show n ++ a')
  328     VBoolType -> Just "_Bool"
  329     VIntType -> Just "_Int"
  330     VIntModType n -> Just ("_IntMod_" ++ show n)
  331     VArrayType a b ->
  332       do a' <- suffixTValue a
  333          b' <- suffixTValue b
  334          Just ("_Array" ++ a' ++ b')
  335     VPiType _ _ -> Nothing
  336     VUnitType -> Just "_Unit"
  337     VPairType a b ->
  338       do a' <- suffixTValue a
  339          b' <- suffixTValue b
  340          Just ("_Pair" ++ a' ++ b')
  341 
  342     VStringType -> Nothing
  343     VDataType {} -> Nothing
  344     VRecordType {} -> Nothing
  345     VSort {} -> Nothing
  346     VTyTerm{} -> Nothing
