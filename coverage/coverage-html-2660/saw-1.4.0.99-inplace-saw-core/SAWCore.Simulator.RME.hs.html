   56 -- | Evaluator for shared terms.
   57 evalSharedTerm :: ModuleMap -> Map Ident RPrim -> Term -> RValue
   58 evalSharedTerm m addlPrims t =
   59   runIdentity $ do
   60     cfg <- Sim.evalGlobal m (Map.union constMap addlPrims)
   61            extcns (\_ _ -> Nothing) neutral primHandler
   62            (Prims.lazyMuxValue prims)
   63     Sim.evalSharedTerm cfg t
   64   where
   65     extcns ec = return $ Prim.userError $ "Unimplemented: external constant " ++ show (ecName ec)
   66     neutral _env nt = return $ Prim.userError $ "Could not evaluate neutral term\n:" ++ show nt
   67     primHandler nm _ msg env _tv =
   68       return $ Prim.userError $ unlines
   69         [ "Could not evaluate primitive " ++ Text.unpack (toAbsoluteName (nameInfo nm))
   70         , "On argument " ++ show (length env)
   71         , Text.unpack msg
   72         ]

   98 toBool :: RValue -> RME

  102 vWord :: Vector RME -> RValue

  106 toWord (VWord x) = x

  117 wordFun :: (Vector RME -> RPrim) -> RPrim
  118 wordFun f = Prims.strictFun (\x -> f (toWord x))

  120 genShift :: (a -> b -> b -> b) -> (b -> Integer -> b) -> b -> Vector a -> b
  121 genShift cond f x0 v = go x0 (V.toList v)
  122   where
  123     go x [] = x
  124     go x (y : ys) = go (cond y (f x (2 ^ length ys)) x) ys

  126 -- | op : (w : Nat) -> Vec w Bool -> Nat -> Vec w Bool;
  127 bvShiftOp :: (Vector RME -> Integer -> Vector RME) -> RPrim
  128 bvShiftOp op =
  129   Prims.constFun $
  130   wordFun $ \x ->
  131   Prims.strictFun $ \y ->
  132   Prims.PrimValue $
  133     case y of
  134       VNat n   -> vWord (op x (toInteger n))
  135       VBVToNat _sz v -> vWord (genShift muxRMEV op x (toWord v))
  136       VIntToNat _i   -> error "RME.shiftOp: intToNat TODO"
  137       _        -> error $ unwords ["SAWCore.Simulator.RME.shiftOp", show y]

  145 pure2 f x y = pure (f x y)

  148 pure3 f x y z = pure (f x y z)

  278 -- primitive intToBv : (n : Nat) -> Integer -> Vec n Bool;
  279 intToBvOp :: RPrim
  280 intToBvOp =
  281   Prims.natFun $ \n ->
  282   Prims.intFun $ \x -> Prims.PrimValue $
  283     VWord (V.reverse (V.generate (fromIntegral n) (RME.constant . testBit x)))

  285 muxRMEV :: RME -> Vector RME -> Vector RME -> Vector RME
  286 muxRMEV b = V.zipWith (RME.mux b)
  287 
  288 muxInt :: RME -> Integer -> Integer -> Integer
  289 muxInt b x y =
  290   case RME.isBool b of
  291     Just c -> if c then x else y
  292     Nothing -> if x == y then x else error $ "muxRValue: VInt " ++ show (x, y)
  293 
  294 muxExtra :: TValue ReedMuller -> RME -> RExtra -> RExtra -> RExtra
  295 muxExtra (VDataType (nameInfo -> ModuleIdentifier "Prelude.Stream") _ [TValue tp] []) b (AStream xs) (AStream ys) =
  296   AStream (muxRValue tp b <$> xs <*> ys)
  297 muxExtra tp _ _ _ = panic "muxExtra" ["Type mismatch: " <> Text.pack (show tp)]
  298 
  299 muxRValue :: TValue ReedMuller -> RME -> RValue -> RValue -> RValue
  300 muxRValue tp b x y = runIdentity $ Prims.muxValue prims tp b x y

  302 -- | Signed shift right simply copies the high order bit
  303 --   into the shifted places.  We special case the zero
  304 --   length vector to avoid a possible out-of-bounds error.
  305 vSignedShiftR :: V.Vector a -> Integer -> V.Vector a
  306 vSignedShiftR xs i
  307   | V.length xs > 0 = Prims.vShiftR x xs i
  308   | otherwise       = xs
  309  where x = xs V.! 0

  347 -- MkStream :: (a :: sort 0) -> (Nat -> a) -> Stream a;
  348 mkStreamOp :: RPrim
  349 mkStreamOp =
  350   Prims.constFun $
  351   Prims.strictFun $ \f ->
  352     Prims.PrimValue (vStream (fmap (\n -> runIdentity (apply f (ready (VNat n)))) IntTrie.identity))
  353 
  354 -- streamGet :: (a :: sort 0) -> Stream a -> Nat -> a;
  355 streamGetOp :: RPrim
  356 streamGetOp =
  357   Prims.tvalFun   $ \tp ->
  358   Prims.strictFun $ \xs ->
  359   Prims.strictFun $ \ix -> Prims.Prim $ case ix of
  360     VNat n -> pure $ IntTrie.apply (toStream xs) (toInteger n)
  361     VIntToNat _i -> error "RME.streamGetOp : symbolic integer TODO"
  362     VBVToNat _sz bv ->
  363       do let trie = toStream xs
  364              loop k [] = IntTrie.apply trie k
  365              loop k (b:bs)
  366                | Just True <- RME.isBool b
  367                = loop k1 bs
  368                | Just False <- RME.isBool b
  369                = loop k0 bs
  370                | otherwise
  371                = muxRValue tp b (loop k1 bs) (loop k0 bs)
  372               where
  373                k0 = k `shiftL` 1
  374                k1 = k0 + 1
  375          pure $ loop (0::Integer) (V.toList (toWord bv))
  376 
  377     _ -> panic "streamGetOp" ["Expected Nat value; found " <> Text.pack (show ix)]

  383 newVars :: FiniteType -> State Int RValue
  384 newVars FTBit = do
  385   i <- get
  386   put (i + 1)
  387   return (vBool (RME.lit i))
  388 newVars (FTVec n t) = VVector <$> V.replicateM (fromIntegral n) (newVars' t)
  389 newVars (FTTuple ts) = vTuple <$> traverse newVars' ts
  390 newVars (FTRec tm) = vRecord <$> traverse newVars' tm
  391 
  392 newVars' :: FiniteType -> State Int RThunk
  393 newVars' shape = ready <$> newVars shape

  396 -- Bit-blasting primitives.
  397 
  398 bitBlastBasic :: ModuleMap
  399               -> Map Ident RPrim
  400               -> Map VarIndex RValue
  401               -> Term
  402               -> RValue
  403 bitBlastBasic m addlPrims ecMap t = runIdentity $ do
  404   let neutral _env nt = return $ Prim.userError $ "Could not evaluate neutral term\n:" ++ show nt
  405   let primHandler nm _ msg env _tv =
  406          return $ Prim.userError $ unlines
  407            [ "Could not evaluate primitive " ++ Text.unpack (toAbsoluteName (nameInfo nm))
  408            , "On argument " ++ show (length env)
  409            , Text.unpack msg
  410            ]
  411 
  412   cfg <- Sim.evalGlobal m (Map.union constMap addlPrims)
  413          (\ec -> case Map.lookup (ecVarIndex ec) ecMap of
  414                    Just v -> pure v
  415                    Nothing -> error ("RME: unknown ExtCns: " ++ show (ecName ec)))
  416          (\_ _ -> Nothing)
  417          neutral
  418          primHandler
  419          (Prims.lazyMuxValue prims)
  420   Sim.evalSharedTerm cfg t

  423 processVar ::
  424   (ExtCns Term, FirstOrderType) ->
  425   IO (ExtCns Term, FiniteType)
  426 processVar (ec, fot) =
  427   case toFiniteType fot of
  428     Nothing -> fail ("RME solver does not support variables of type " ++ show fot)
  429     Just ft -> pure (ec, ft)

  431 withBitBlastedSATQuery ::
  432   SharedContext ->
  433   Map Ident RPrim ->
  434   SATQuery ->
  435   (RME -> [(ExtCns Term, FiniteType)] -> IO a) ->
  436   IO a
  437 withBitBlastedSATQuery sc addlPrims satq cont =
  438   do unless (Set.null (satUninterp satq)) $ fail
  439         "RME prover does not support uninterpreted symbols"
  440      t <- satQueryAsTerm sc satq
  441      varShapes <- mapM processVar (Map.toList (satVariables satq))
  442      modmap <- scGetModuleMap sc
  443      let vars = evalState (traverse (traverse newVars) varShapes) 0
  444      let varMap = Map.fromList [ (ecVarIndex ec, v) | (ec,v) <- vars ]
  445      let bval = bitBlastBasic modmap addlPrims varMap t
  446      case bval of
  447        VBool anf -> cont anf varShapes
  448        _ -> panic "withBitBlastedSATQuery" ["Non-boolean result type " <> Text.pack (show bval)]
