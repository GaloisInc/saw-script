   88 -- Module Names ----------------------------------------------------------------
   89 
   90 newtype ModuleName = ModuleName Text
   91   deriving (Eq, Ord, Generic, TH.Lift)
   92 
   93 instance Hashable ModuleName -- automatically derived
   94 
   95 instance Show ModuleName where
   96   show (ModuleName s) = Text.unpack s
   97 
   98 
   99 moduleNameText :: ModuleName -> Text
  100 moduleNameText (ModuleName x) = x
  101 
  102 moduleNamePieces :: ModuleName -> [Text]
  103 moduleNamePieces (ModuleName x) = Text.splitOn "." x
  104 
  105 -- | Create a module name given a list of strings with the top-most
  106 -- module name given first.
  107 mkModuleName :: [Text] -> ModuleName
  108 mkModuleName [] = panic "mkModuleName" ["Empty module name"]
  109 mkModuleName nms = ModuleName $ Text.intercalate "." (reverse nms)
  110 
  111 preludeName :: ModuleName
  112 preludeName = mkModuleName ["Prelude"]

  115 -- Identifiers -----------------------------------------------------------------
  116 
  117 data Ident =
  118   Ident
  119   { identModule :: ModuleName
  120   , identBaseName :: Text
  121   }
  122   deriving (Eq, Ord, Generic)
  123 
  124 instance Hashable Ident -- automatically derived
  125 
  126 instance Show Ident where
  127   show (Ident m s) = shows m ('.' : Text.unpack s)
  128 
  129 identText :: Ident -> Text
  130 identText i = moduleNameText (identModule i) <> Text.pack "." <> identBaseName i
  131 
  132 identPieces :: Ident -> NonEmpty Text
  133 identPieces i =
  134   case moduleNamePieces (identModule i) of
  135     [] -> identBaseName i :| []
  136     (x:xs) -> x :| (xs ++ [identBaseName i])
  137 
  138 identName :: Ident -> String
  139 identName = Text.unpack . identBaseName
  140 
  141 instance Read Ident where
  142   readsPrec _ str =
  143     let (str1, str2) = break (not . isIdChar) str in
  144     [(parseIdent str1, str2)]
  145 
  146 mkIdent :: ModuleName -> Text -> Ident
  147 mkIdent m s = Ident m s
  148 
  149 -- | Make a \"coq-safe\" identifier from a string that might contain
  150 -- non-identifier characters, where we use the SAW core notion of identifier
  151 -- characters as letters, digits, underscore and primes. Any disallowed
  152 -- character is mapped to the string @__xNN@, where @NN@ is the hexadecimal code
  153 -- for that character. Additionally, a SAW core identifier is not allowed to
  154 -- start with a prime, so a leading underscore is added in such a case.
  155 mkSafeIdent :: ModuleName -> String -> Ident
  156 mkSafeIdent _ [] = fromString "_"
  157 mkSafeIdent mnm nm =
  158   let is_safe_char c = isAlphaNum c || c == '_' || c == '\'' in
  159   mkIdent mnm $ Text.pack $
  160   (if nm!!0 == '\'' then ('_' :) else id) $
  161   concatMap
  162   (\c -> if is_safe_char c then [c] else
  163            "__x" ++ showHex (ord c) "")
  164   nm
  165 
  166 -- | Parse a fully qualified identifier.
  167 parseIdent :: String -> Ident
  168 parseIdent s0 =
  169     case reverse (breakEach s0) of
  170       (_:[]) -> panic "parseIdent" ["Empty module name"]
  171       (nm:rMod) -> mkIdent (mkModuleName (reverse rMod)) nm
  172       _ -> panic "parseIdent" ["Bad identifier " <> Text.pack s0]
  173   where breakEach s =
  174           case break (=='.') s of
  175             (h, []) -> [Text.pack h]
  176             (h, _ : r) -> Text.pack h : breakEach r
  177 
  178 instance IsString Ident where
  179   fromString = parseIdent
  180 
  181 isIdent :: String -> Bool
  182 isIdent (c:l) = isAlpha c && all isIdChar l
  183 isIdent [] = False
  184 
  185 -- | Returns true if character can appear in identifier.
  186 isIdChar :: Char -> Bool
  187 isIdChar c = isAlphaNum c || (c == '_') || (c == '\'') || (c == '.')

  191 -- NameInfo
  192 
  193 
  194 -- | Descriptions of the origins of names that may be in scope
  195 data NameInfo
  196   = -- | This name arises from an exported declaration from a module
  197     ModuleIdentifier Ident
  198 
  199   | -- | This name was imported from some other programming language/scope
  200     ImportedName
  201       URI      -- ^ An absolutely-qualified name, which is required to be unique
  202       [Text]   -- ^ A collection of aliases for this name.  Shorter or "less-qualified"
  203                --   aliases should be nearer the front of the list
  204 
  205  deriving (Eq,Ord,Show)
  206 
  207 nameURI :: NameInfo -> URI
  208 nameURI =
  209   \case
  210     ModuleIdentifier i -> moduleIdentToURI i
  211     ImportedName uri _ -> uri
  212 
  213 nameAliases :: NameInfo -> [Text]
  214 nameAliases =
  215   \case
  216     ModuleIdentifier i -> [identBaseName i, identText i]
  217     ImportedName _ aliases -> aliases
  218 
  219 toShortName :: NameInfo -> Text
  220 toShortName (ModuleIdentifier i) = identBaseName i
  221 toShortName (ImportedName uri []) = render uri
  222 toShortName (ImportedName _ (x:_)) = x
  223 
  224 toAbsoluteName :: NameInfo -> Text
  225 toAbsoluteName (ModuleIdentifier i) = identText i
  226 toAbsoluteName (ImportedName uri _) = render uri
  227 
  228 moduleIdentToURI :: Ident -> URI
  229 moduleIdentToURI ident = fromMaybe (panic "moduleIdentToURI" ["Failed to construct ident URI: " <> identText ident]) $
  230   do sch  <- mkScheme "sawcore"
  231      path <- mapM mkPathPiece (identPieces ident)
  232      pure URI
  233        { uriScheme = Just sch
  234        , uriAuthority = Left True -- absolute path
  235        , uriPath   = Just (False, path)
  236        , uriQuery  = []
  237        , uriFragment = Nothing
  238        }

  241 -- External Constants ----------------------------------------------------------
  242 
  243 type VarIndex = Int
  244 
  245 -- | An external constant with a name.
  246 -- Names are not necessarily unique, but the var index should be.
  247 data ExtCns e =
  248   EC
  249   { ecVarIndex :: !VarIndex
  250   , ecName :: !NameInfo
  251   , ecType :: !e
  252   }
  253   deriving (Show, Functor, Foldable, Traversable)
  254 
  255 instance Eq (ExtCns e) where
  256   x == y = ecVarIndex x == ecVarIndex y
  257 
  258 instance Ord (ExtCns e) where
  259   compare x y = compare (ecVarIndex x) (ecVarIndex y)
  260 
  261 instance Hashable (ExtCns e) where
  262   hashWithSalt x ec = hashWithSalt x (ecVarIndex ec)
  263

  265 -- Primitive Names ------------------------------------------------------------
  266 
  267 -- | Names of SAWCore primitives, data types and data type constructors.
  268 data PrimName e =
  269   PrimName
  270   { primVarIndex :: !VarIndex
  271   , primName     :: !Ident
  272   , primType     :: e
  273   }
  274   deriving (Show, Functor, Foldable, Traversable)
  275 
  276 instance Eq (PrimName e) where
  277   x == y = primName x == primName y
  278 
  279 instance Ord (PrimName e) where
  280   compare x y = compare (primName x) (primName y)
  281 
  282 instance Hashable (PrimName e) where
  283   hashWithSalt x pn = hashWithSalt x (primName pn)
  284 
  285 primNameToExtCns :: PrimName e -> ExtCns e
  286 primNameToExtCns (PrimName varIdx nm tp) = EC varIdx (ModuleIdentifier nm) tp
  287 
  288 scFreshNameURI :: Text -> VarIndex -> URI
  289 scFreshNameURI nm i = fromMaybe (panic "scFreshNameURI" ["Failed to construct name URI: <> " <> nm <> "  " <> Text.pack (show i)]) $
  290   do sch <- mkScheme "fresh"
  291      nm' <- mkPathPiece (if Text.null nm then "_" else nm)
  292      i'  <- mkFragment (Text.pack (show i))
  293      pure URI
  294        { uriScheme = Just sch
  295        , uriAuthority = Left False -- relative path
  296        , uriPath   = Just (False, (nm' :| []))
  297        , uriQuery  = []
  298        , uriFragment = Just i'
  299        }

  302 -- Display Name Environments --------------------------------------------------------
  303 
  304 -- | A 'DisplayNameEnv' encodes the mappings between display names (type
  305 -- 'Text') and internal names (type 'VarIndex'). Multiple display
  306 -- names may be associated with each internal name; these are stored
  307 -- in priority order, with preferred display names first.
  308 data DisplayNameEnv =
  309   DisplayNameEnv
  310   { displayNames :: !(IntMap [Text]) -- Keyed by VarIndex; preferred names come first.
  311   , displayIndexes :: !(Map Text IntSet)
  312   }
  313 -- Invariants: The 'displayNames' and 'displayIndexes' maps should be
  314 -- inverses of each other. That is, 'displayNames' maps @i@ to a list
  315 -- containing @x@ if and only if 'displayIndexes' maps @x@ to a set
  316 -- containing @i@.
  317 
  318 emptyDisplayNameEnv :: DisplayNameEnv
  319 emptyDisplayNameEnv = DisplayNameEnv mempty mempty
  320 
  321 -- | Extend a 'DisplayNameEnv' by providing new 'Text' display names
  322 -- for a given 'VarIndex'. Display names should be provided in
  323 -- priority order, with preferred names first. If the 'VarIndex'
  324 -- already exists in the map, then add the new display names with
  325 -- higher priority while keeping the old ones.
  326 extendDisplayNameEnv :: VarIndex -> [Text] -> DisplayNameEnv -> DisplayNameEnv
  327 extendDisplayNameEnv i aliases env =
  328   DisplayNameEnv
  329   { displayNames = IntMap.insertWith List.union i aliases (displayNames env)
  330   , displayIndexes = foldr insertAlias (displayIndexes env) aliases
  331   }
  332   where
  333     insertAlias :: Text -> Map Text IntSet -> Map Text IntSet
  334     insertAlias x m = Map.insertWith IntSet.union x (IntSet.singleton i) m
  335 
  336 -- | Filter display names in a 'DisplayNameEnv' that satisfy a predicate.
  337 filterDisplayNameEnv :: (Text -> Bool) -> DisplayNameEnv -> DisplayNameEnv
  338 filterDisplayNameEnv p env =
  339   DisplayNameEnv
  340   { displayNames = IntMap.filter (not . null) $ fmap (filter p) $ displayNames env
  341   , displayIndexes = Map.filterWithKey (\k _ -> p k) $ displayIndexes env
  342   }
  343 
  344 -- | Merge two 'DisplayNameEnv's, giving higher priority to display
  345 -- names from the first argument.
  346 mergeDisplayNameEnv :: DisplayNameEnv -> DisplayNameEnv -> DisplayNameEnv
  347 mergeDisplayNameEnv env1 env2 =
  348   DisplayNameEnv
  349   { displayNames = IntMap.unionWith List.union (displayNames env1) (displayNames env2)
  350   , displayIndexes = Map.unionWith IntSet.union (displayIndexes env1) (displayIndexes env2)
  351   }
  352 
  353 -- | Look up a 'Text' display name in an environment, returning the
  354 -- list of indexes it could resolve to.
  355 resolveDisplayName :: DisplayNameEnv -> Text -> [VarIndex]
  356 resolveDisplayName env x =
  357   case Map.lookup x (displayIndexes env) of
  358     Nothing -> []
  359     Just s -> IntSet.elems s
  360 
  361 -- | Return the first display name for the given 'VarIndex' that is
  362 -- unambiguous in the naming environment. If there is no unambiguous
  363 -- alias, then return 'Nothing'.
  364 bestDisplayName :: DisplayNameEnv -> VarIndex -> Maybe Text
  365 bestDisplayName env i =
  366   do aliases <- IntMap.lookup i (displayNames env)
  367      go aliases
  368   where
  369     go [] = Nothing
  370     go (x : xs) =
  371       case Map.lookup x (displayIndexes env) of
  372         Nothing -> go xs -- should never happen
  373         Just vs
  374           | IntSet.size vs == 1 -> Just x
  375           | otherwise -> go xs

  377 -- Naming Environments ---------------------------------------------------------
  378 
  379 data SAWNamingEnv = SAWNamingEnv
  380   { resolvedNames :: !(Map VarIndex NameInfo)
  381   , absoluteNames :: !(Map URI VarIndex)
  382   , aliasNames    :: !(Map Text (Set VarIndex))
  383   }
  384 -- Invariants: The 'resolvedNames' and 'absoluteNames' maps should be
  385 -- inverses of each other. That is, 'resolvedNames' maps @i@ to @n@ if
  386 -- and only if 'absoluteNames' maps @nameURI n@ to @i@. Also, every
  387 -- 'VarIndex' appearing in 'aliasNames' must be present as a key in
  388 -- 'resolvedNames'.
  389 
  390 emptySAWNamingEnv :: SAWNamingEnv
  391 emptySAWNamingEnv = SAWNamingEnv mempty mempty mempty
  392 
  393 -- | Add a new name entry in a 'SAWNamingEnv'. Returns 'Left' if
  394 -- there is already an entry under the same URI.
  395 registerName :: VarIndex -> NameInfo -> SAWNamingEnv -> Either URI SAWNamingEnv
  396 registerName i nmi env =
  397   case Map.lookup uri (absoluteNames env) of
  398     Just _ -> Left uri
  399     Nothing ->
  400       Right $
  401       SAWNamingEnv
  402       { resolvedNames = Map.insert i nmi (resolvedNames env)
  403       , absoluteNames = Map.insert uri i (absoluteNames env)
  404       , aliasNames    = foldr insertAlias (aliasNames env) aliases
  405       }
  406   where
  407     uri = nameURI nmi
  408     aliases = render uri : nameAliases nmi
  409 
  410     insertAlias :: Text -> Map Text (Set VarIndex) -> Map Text (Set VarIndex)
  411     insertAlias x m = Map.insertWith Set.union x (Set.singleton i) m
  412 
  413 resolveURI :: SAWNamingEnv -> URI -> Maybe VarIndex
  414 resolveURI env uri = Map.lookup uri (absoluteNames env)
  415 
  416 resolveName :: SAWNamingEnv -> Text -> [(VarIndex, NameInfo)]
  417 resolveName env nm =
  418   case Map.lookup nm (aliasNames env) of
  419     Nothing -> []
  420     Just vs -> [ (v, findName v (resolvedNames env)) | v <- Set.toList vs ]
  421   where
  422     findName v m =
  423       case Map.lookup v m of
  424         Just nmi -> nmi
  425         Nothing ->
  426             panic "resolveName" [
  427                 "Unbound VarIndex when resolving name: " <> nm,
  428                 "Index: " <> Text.pack (show v)
  429             ]
  430 
  431 -- | Return the first alias (according to 'nameAliases') that is
  432 -- unambiguous in the naming environment. If there is no unambiguous
  433 -- alias, then return the URI.
  434 bestAlias :: SAWNamingEnv -> NameInfo -> Either URI Text
  435 bestAlias env nmi = go (nameAliases nmi)
  436   where
  437     go [] = Left (nameURI nmi)
  438     go (x : xs) =
  439       case Map.lookup x (aliasNames env) of
  440         Nothing -> go xs
  441         Just vs
  442           | Set.size vs == 1 -> Right x
  443           | otherwise -> go xs
  444 
  445 -- | Convert from a 'SAWNamingEnv' to a 'DisplayNameEnv'.
  446 toDisplayNameEnv :: SAWNamingEnv -> DisplayNameEnv
  447 toDisplayNameEnv env =
  448   DisplayNameEnv
  449   { displayNames = IntMap.fromList [ (vi, nameAliases nmi) | (vi, nmi) <- Map.assocs (resolvedNames env) ]
  450   , displayIndexes = fmap (IntSet.fromList . Set.toList) (aliasNames env)
  451   }
