   56 module_skeleton ::
   57   Some LLVMModule ->
   58   TopLevel ModuleSkeleton
   59 module_skeleton m = do
   60   let ast = viewSome modAST m
   61   liftIO $ moduleSkeleton ast
   62 
   63 function_skeleton ::
   64   ModuleSkeleton ->
   65   String ->
   66   TopLevel FunctionSkeleton
   67 function_skeleton mskel nm =
   68   case mskel ^. modSkelFunctions . at (Text.pack nm) of
   69     Just fskel -> pure fskel
   70     Nothing -> throwTopLevel $ mconcat ["No skeleton exists for function \"", nm, "\""]
   71 
   72 -- | Manually add a new guess for the size of the argument at the given index
   73 skeleton_resize_arg_index ::
   74   FunctionSkeleton ->
   75   Int ->
   76   Int ->
   77   Bool ->
   78   TopLevel FunctionSkeleton
   79 skeleton_resize_arg_index skel idx sz initialized =
   80   pure (skel & funSkelArgs . ix idx . argSkelType . typeSkelSizeGuesses %~ (guess:))
   81   where
   82     guess :: SizeGuess
   83     guess = SizeGuess
   84       { _sizeGuessElems = sz
   85       , _sizeGuessInitialized = initialized
   86       , _sizeGuessSource = "user guess"
   87       }
   88 
   89 skelArgIndex ::
   90   FunctionSkeleton ->
   91   String ->
   92   Maybe Int
   93 skelArgIndex skel nm = flip findIndex (skel ^. funSkelArgs) $ \arg ->
   94   arg ^. argSkelName == Just (Text.pack nm)
   95 
   96 -- | Manually add a new guess for the size of the argument with the given name
   97 skeleton_resize_arg ::
   98   FunctionSkeleton ->
   99   String ->
  100   Int ->
  101   Bool ->
  102   TopLevel FunctionSkeleton
  103 skeleton_resize_arg skel nm sz initialized
  104   | Just idx <- skelArgIndex skel nm
  105   = skeleton_resize_arg_index skel idx sz initialized
  106   | otherwise = throwTopLevel $ mconcat
  107     [ "No argument named \""
  108     , nm
  109     , "\" (enabling debug symbols when compiling might help)"
  110     ]

  112 llvmTypeSize :: C.LLVM.TypeContext -> LLVM.Type -> Int
  113 llvmTypeSize tc t =
  114   let ?lc = tc in
  115     case C.LLVM.liftMemType t of
  116       Left _ -> 1
  117       Right m -> fromIntegral $ C.LLVM.memTypeSize (C.LLVM.llvmDataLayout ?lc) m

  119 skeleton_guess_arg_sizes ::
  120   ModuleSkeleton ->
  121   Some LLVMModule ->
  122   [(String, [FunctionProfile])] ->
  123   TopLevel ModuleSkeleton
  124 skeleton_guess_arg_sizes mskel (Some m) profiles = do
  125   let (_, tc) = C.LLVM.typeContextFromModule $ modAST m
  126   fskels <- forM (mskel ^. modSkelFunctions) $ \skel -> do
  127     case lookup (Text.unpack $ skel ^. funSkelName) profiles of
  128       Just (prof:_) -> let
  129         updateArg (a, p)
  130           | a ^. argSkelType . typeSkelIsPointer
  131           , Just s <- p ^. argProfileSize
  132           = a & argSkelType . typeSkelSizeGuesses
  133             %~ (SizeGuess
  134                  (quot s $ llvmTypeSize tc $ a ^. argSkelType . typeSkelLLVMType)
  135                  (p ^. argProfileInitialized)
  136                  "checking sizes in the simulator":)
  137           | otherwise = a
  138         uargs args = updateArg <$> zip args (prof ^. funProfileArgs)
  139         in pure (skel & funSkelArgs %~ uargs)
  140       _ -> pure skel
  141   pure $ mskel & modSkelFunctions .~ fskels

  146 skeleton_globals_pre ::
  147   ModuleSkeleton ->
  148   LLVMCrucibleSetupM ()
  149 skeleton_globals_pre mskel =
  150   forM_ (mskel ^. modSkelGlobals) $ \gskel ->
  151     when (gskel ^. globSkelMutable) $ do
  152       let gname = Text.unpack $ gskel ^. globSkelName
  153       llvm_alloc_global gname
  154       when (gskel ^. globSkelInitialized)
  155         . llvm_points_to True (anySetupGlobal gname)
  156         $ anySetupGlobalInitializer gname
  157 
  158 skeleton_globals_post ::
  159   ModuleSkeleton ->
  160   LLVMCrucibleSetupM ()
  161 skeleton_globals_post mskel =
  162   forM_ (mskel ^. modSkelGlobals) $ \gskel -> do
  163     when (gskel ^. globSkelMutable && gskel ^. globSkelInitialized) $ do
  164       let gname = Text.unpack $ gskel ^. globSkelName
  165       llvm_points_to True (anySetupGlobal gname)
  166         $ anySetupGlobalInitializer gname

  168 buildArg ::
  169   ArgSkeleton ->
  170   Int ->
  171   LLVMCrucibleSetupM (Maybe TypedTerm, Maybe (AllLLVM SetupValue), Maybe Text)
  172 buildArg arg idx
  173   | arg ^. argSkelType . typeSkelIsPointer
  174   = let
  175       pt = arg ^. argSkelType . typeSkelLLVMType
  176       (t, initialized) = case arg ^. argSkelType . typeSkelSizeGuesses of
  177         (s:_)
  178           | s ^. sizeGuessElems > 1
  179             -> (LLVM.Array (fromIntegral $ s ^. sizeGuessElems) pt, s ^. sizeGuessInitialized)
  180           | otherwise -> (pt, s ^. sizeGuessInitialized)
  181         _ -> (pt, False)
  182     in do
  183       ptr <- llvm_alloc t
  184       mval <- if initialized
  185         then do
  186         val <- llvm_fresh_var ident t
  187         llvm_points_to True ptr (anySetupTerm val)
  188         pure $ Just val
  189         else pure Nothing
  190       pure (mval, Just ptr, arg ^. argSkelName)
  191   | otherwise
  192   = do
  193       val <- llvm_fresh_var ident
  194         $ arg ^. argSkelType . typeSkelLLVMType
  195       pure (Just val, Nothing, arg ^. argSkelName)
  196   where
  197     ident = maybe ("arg" <> Text.pack (show idx)) id $ arg ^. argSkelName
  198 
  199 skeleton_prestate ::
  200   FunctionSkeleton ->
  201   LLVMCrucibleSetupM SkeletonState
  202 skeleton_prestate skel = do
  203   _skelArgs <- mapM (uncurry buildArg) $ zip (skel ^. funSkelArgs) [1,2..]
  204   pure $ SkeletonState{..}
  205 
  206 skeleton_exec ::
  207   SkeletonState ->
  208   LLVMCrucibleSetupM ()
  209 skeleton_exec prestate = do
  210   args <- forM (prestate ^. skelArgs) $ \(mval, mptr, _) ->
  211     case (mval, mptr) of
  212       (_, Just ptr) -> pure ptr
  213       (Just val, Nothing) -> pure $ anySetupTerm val
  214       (Nothing, Nothing) -> throwLLVMFun "skeleton_exec" "Invalid pointer-pointee combination on skeleton argument"
  215   llvm_execute_func args
  216 
  217 rebuildArg ::
  218   (ArgSkeleton, (Maybe TypedTerm, Maybe (AllLLVM SetupValue), Maybe Text))  ->
  219   Int ->
  220   LLVMCrucibleSetupM (Maybe TypedTerm, Maybe (AllLLVM SetupValue), Maybe Text)
  221 rebuildArg (arg, prearg) idx
  222   | arg ^. argSkelType . typeSkelIsPointer
  223   , (_, Just ptr, nm) <- prearg
  224   = let
  225       pt = arg ^. argSkelType . typeSkelLLVMType
  226       t = case arg ^. argSkelType . typeSkelSizeGuesses of
  227         (s:_)
  228           | s ^. sizeGuessElems > 1
  229             -> LLVM.Array (fromIntegral $ s ^. sizeGuessElems) pt
  230           | otherwise -> pt
  231         _ -> pt
  232       ident = maybe ("arg" <> Text.pack (show idx)) id nm
  233     in do
  234       val' <- llvm_fresh_var ident t
  235       llvm_points_to True ptr $ anySetupTerm val'
  236       pure (Just val', Just ptr, nm)
  237   | otherwise = pure prearg
  238 
  239 skeleton_poststate ::
  240   FunctionSkeleton ->
  241   SkeletonState ->
  242   LLVMCrucibleSetupM SkeletonState
  243 skeleton_poststate skel prestate = do
  244   _skelArgs <- zipWithM rebuildArg
  245     (zip (skel ^. funSkelArgs) (prestate ^. skelArgs))
  246     [1,2..]
  247   case skel ^. funSkelRet . typeSkelLLVMType of
  248     LLVM.PrimType LLVM.Void -> pure ()
  249     t -> do
  250       ret <- llvm_fresh_var ("return value of " <> (skel ^. funSkelName)) t
  251       llvm_return $ anySetupTerm ret
  252   pure $ SkeletonState{..}

  254 skeleton_arg_index ::
  255   SkeletonState ->
  256   Int ->
  257   LLVMCrucibleSetupM TypedTerm
  258 skeleton_arg_index state idx
  259   | idx < length (state ^. skelArgs)
  260   , (Just t, _, _) <- (state ^. skelArgs) !! idx
  261   = pure t
  262   | otherwise = throwLLVMFun "skeleton_arg_index" $ mconcat
  263     [ "No initialized argument at index "
  264     , show idx
  265     ]
  266 
  267 stateArgIndex ::
  268   SkeletonState ->
  269   String ->
  270   Maybe Int
  271 stateArgIndex state nm = flip findIndex (state ^. skelArgs) $ \(_, _, mnm) ->
  272   mnm == Just (Text.pack nm)
  273 
  274 skeleton_arg ::
  275   SkeletonState ->
  276   String ->
  277   LLVMCrucibleSetupM TypedTerm
  278 skeleton_arg state nm
  279   | Just idx <- stateArgIndex state nm
  280   = skeleton_arg_index state idx
  281   | otherwise = throwLLVMFun "skeleton_arg" $ mconcat
  282     [ "No initialized argument named \""
  283     , nm
  284     , "\" (enabling debug symbols when compiling might help)"
  285     ]

  287 skeleton_arg_index_pointer ::
  288   SkeletonState ->
  289   Int ->
  290   LLVMCrucibleSetupM (AllLLVM SetupValue)
  291 skeleton_arg_index_pointer state idx
  292   | idx < length (state ^. skelArgs)
  293   , (_, mp, _) <- (state ^. skelArgs) !! idx
  294   = case mp of
  295       Just p -> pure p
  296       Nothing -> throwLLVMFun "skeleton_arg_index_pointer" $ mconcat
  297         [ "Argument at index "
  298         , show idx
  299         , " is not a pointer or array"
  300         ]
  301   | otherwise = throwLLVMFun "skeleton_arg_index_pointer" $ mconcat
  302     [ "No argument at index "
  303     , show idx
  304     ]
  305 
  306 skeleton_arg_pointer ::
  307   SkeletonState ->
  308   String ->
  309   LLVMCrucibleSetupM (AllLLVM SetupValue)
  310 skeleton_arg_pointer state nm
  311   | Just idx <- stateArgIndex state nm
  312   = skeleton_arg_index_pointer state idx
  313   | otherwise = do
  314       throwLLVMFun "skeleton_arg_pointer" $ mconcat
  315         [ "No argument named \""
  316         , nm
  317         , "\" (enabling debug symbols when compiling might help)"
