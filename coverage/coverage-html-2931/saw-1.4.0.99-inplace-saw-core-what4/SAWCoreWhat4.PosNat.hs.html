   38 -- We include the KnownNat constraint here so that we can avoid using
   39 -- withKnownNat outside this module.  The downside is that we have two
   40 -- different runtime representations of these positive nats --- the
   41 -- one stored in the KnownNat, and the one stored in the NatRepr.
   42 
   43 -- | Runtime representation of a non-zero natural number
   44 data PosNat (n :: Nat) where
   45   PosNat :: (1 <= n, KnownNat n) => NatRepr n -> PosNat n
   46 
   47 -- | Check whether an integer is a positive nat
   48 somePosNat :: Integral a => a -> Maybe (Some PosNat)
   49 somePosNat n
   50   | Just (Some nr) <- someNat n,
   51     Just LeqProof  <- testLeq (knownNat @1) nr
   52   = withKnownNat nr $ Just (Some (PosNat nr))
   53   | otherwise
   54   = Nothing
   55

   58 -- positivity can be removed. Annoyingly, we cannot do 'testLeq'
   59 -- without already knowing that w >= 0)

   62 -- | Add two numbers together and return a proof that their
   63 -- result is positive.
   64 addPosNat :: forall w1 w2.
   65   (1 <= w1, 1 <= w2, KnownNat w1, KnownNat w2) => PosNat (w1 + w2)
   66 addPosNat =
   67   let w1  = knownNat @w1
   68       w2  = knownNat @w2
   69       sm  = addNat w1 w2 in
   70   case leqAddPos w1 w2 of
   71     LeqProof -> withKnownNat sm $ PosNat sm
   72
