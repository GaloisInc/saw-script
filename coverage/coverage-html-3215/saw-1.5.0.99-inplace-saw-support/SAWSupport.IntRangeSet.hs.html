    1 {- |
    2 Module      : SAWSupport.IntRangeSet
    3 Description : A type representing sets of ranges of integers
    4 License     : BSD3
    5 Maintainer  : saw@galois.com
    6 Stability   : provisional
    7 
    8 An 'IntRangeSet' represents a set of 'Int's, much like an @IntSet@,
    9 but it also supports insertion and deletion of whole contiguous ranges
   10 of 'Int's at once.
   11 
   12 Ranges are merged so that every set of 'Int's has a canonical
   13 representation as an 'IntRangeSet'.
   14 
   15 The complexity of operations scales not with the number of 'Int's in a
   16 set, but rather the number of separate contiguous ranges of 'Int's.
   17 -}
   18 
   19 module SAWSupport.IntRangeSet (
   20     IntRangeSet,
   21     empty,
   22     singleton,
   23     insert,
   24     delete,
   25     member,
   26     union,
   27     difference,
   28     fromList,
   29     toList
   30   ) where
   31 
   32 import Data.IntMap.Strict (IntMap)
   33 import qualified Data.IntMap.Strict as IntMap
   34 
   35 -- | An 'IntRangeSet' represents a set of disjoint contiguous
   36 -- intervals of type 'Int'.
   37 -- Each interval @(i, j)@ with @i <= j@ represents the set of values
   38 -- @[i..j]@.
   39 newtype IntRangeSet = IntRangeSet (IntMap Int)
   40   deriving Eq
   41 
   42 instance Show IntRangeSet where
   43   show (IntRangeSet m) = show m
   44 
   45 -- | The empty set.
   46 empty :: IntRangeSet
   47 empty = IntRangeSet IntMap.empty
   48 
   49 -- | A set containing a single interval @[i..j]@.
   50 singleton :: (Int, Int) -> IntRangeSet
   51 singleton (i, j)
   52   | i <= j = IntRangeSet (IntMap.singleton i j)
   53   | otherwise = empty
   54 
   55 -- | Add an interval @[i..j]@ to the set.
   56 insert :: (Int, Int) -> IntRangeSet -> IntRangeSet
   57 insert (i, j) (IntRangeSet m)
   58   | i > j = IntRangeSet m
   59   | otherwise =
   60     let -- m1 contains all intervals strictly below i
   61         -- also possibly one interval starting at i'
   62         m1 = IntMap.takeWhileAntitone (< i) m
   63         -- m2 contains all intervals strictly above j
   64         m2 = IntMap.dropWhileAntitone (<= j+1) m
   65         i' = case IntMap.lookupLE i m of
   66                Nothing -> i
   67                Just (u, v) -> if i <= v+1 then u else i
   68         j' = case IntMap.lookupLE (j+1) m of
   69                Nothing -> j
   70                Just (_, v) -> max v j
   71     in IntRangeSet (IntMap.insert i' j' (IntMap.union m1 m2))
   72     -- Any interval [u..v] with v+1 < i stays unchanged.
   73     -- Any interval [u..v] with u <= i <= v+1 is overwritten.
   74     -- Any interval [u..v] with i < u <= j is removed.
   75     -- Any interval [u..v] with j < u stays unchanged.
   76     -- If i falls inside (or is adjacent to the upper bound of) an
   77     -- interval [u..v], set i' = u; else i' = i.
   78     -- If j falls inside (or is adjacent to the lower bound of) an
   79     -- interval [u..v], let j' = v; else j' = j.
   80     -- Always insert a new interval [i'..j'].
   81 
   82 -- | Remove an interval @[i..j]@ from the set.
   83 delete :: (Int, Int) -> IntRangeSet -> IntRangeSet
   84 delete (i, j) (IntRangeSet m)
   85   | i > j = IntRangeSet m
   86   | otherwise =
   87     let m1 = IntMap.takeWhileAntitone (< i) m
   88         m2 = IntMap.dropWhileAntitone (<= j) m
   89         m1' = case IntMap.lookupLT i m1 of
   90                 Nothing -> m1
   91                 Just (u, v) -> if i <= v then IntMap.insert u (i-1) m1 else m1
   92         m2' = case IntMap.lookupLE j m of
   93                 Nothing -> m2
   94                 Just (_, v) -> if j < v then IntMap.insert (j+1) v m2 else m2
   95     in IntRangeSet (IntMap.union m1' m2')
   96     -- Any interval [u..v] with v < i stays unchanged.
   97     -- Any interval [u..v] with u < i <= v is overwritten.
   98     -- Any interval [u..v] with i <= u <= j is removed.
   99     -- Any interval [u..v] with j < u stays unchanged.
  100     -- If i falls inside an interval [u..v], replace it with interval [u..i-1].
  101     -- If j falls inside an interval [u..v], insert a new interval [j+1..v].
  102 
  103 -- | Is a specific value in the set?
  104 member :: Int -> IntRangeSet -> Bool
  105 member x (IntRangeSet m) =
  106   case IntMap.lookupLE x m of
  107     Nothing -> False
  108     Just (_, j) -> x <= j
  109 
  110 -- | The union of two sets.
  111 union :: IntRangeSet -> IntRangeSet -> IntRangeSet
  112 union s1 s2 = foldr insert s1 (toList s2)
  113 
  114 -- | The difference of two sets: Remove all members of the second set
  115 -- from the first set.
  116 difference :: IntRangeSet -> IntRangeSet -> IntRangeSet
  117 difference s1 s2 = foldr delete s1 (toList s2)
  118 
  119 -- | Create a set from a list of intervals.
  120 fromList :: [(Int, Int)] -> IntRangeSet
  121 fromList = foldr insert empty
  122 
  123 -- | Convert the set to a list of disjoint intervals in ascending
  124 -- order.
  125 toList :: IntRangeSet -> [(Int, Int)]
  126 toList (IntRangeSet m) = IntMap.toList m
