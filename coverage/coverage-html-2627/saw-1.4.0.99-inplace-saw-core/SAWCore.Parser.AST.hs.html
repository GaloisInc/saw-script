   62 data UTerm
   63   = Name (PosPair Text)
   64   | Sort Pos Sort SortFlags
   65   | App UTerm UTerm
   66   | Lambda Pos UTermCtx UTerm
   67   | Pi Pos UTermCtx UTerm
   68   | Recursor (PosPair Text)
   69   | UnitValue Pos
   70   | UnitType Pos
   71     -- | New-style records
   72   | RecordValue Pos [(PosPair FieldName, UTerm)]
   73   | RecordType Pos [(PosPair FieldName, UTerm)]
   74   | RecordProj UTerm FieldName
   75     -- | Simple pairs
   76   | PairValue Pos UTerm UTerm
   77   | PairType Pos UTerm UTerm
   78   | PairLeft UTerm
   79   | PairRight UTerm
   80     -- | Identifies a type constraint on the term, i.e., a type ascription
   81   | TypeConstraint UTerm Pos UTerm
   82   | NatLit Pos Natural
   83   | StringLit Pos Text
   84     -- | Vector literal.
   85   | VecLit Pos [UTerm]
   86     -- | Bitvector literal.
   87   | BVLit Pos [Bool]
   88   | BadTerm Pos
   89  deriving (Show, TH.Lift)
   90 
   91 -- | A pattern used for matching a variable.
   92 data UTermVar
   93   = TermVar (PosPair LocalName)
   94   | UnusedVar Pos
   95   deriving (Eq, Ord, Show, TH.Lift)
   96 
   97 -- | Return the 'String' name associated with a 'UTermVar'
   98 termVarString :: UTermVar -> String
   99 termVarString (TermVar (PosPair _ str)) = Text.unpack str
  100 termVarString (UnusedVar _) = "_"
  101 
  102 -- | Return the 'LocalName' associated with a 'UTermVar'
  103 termVarLocalName :: UTermVar -> LocalName
  104 termVarLocalName (TermVar (PosPair _ str)) = str
  105 termVarLocalName (UnusedVar _) = "_"
  106 
  107 -- | A context of 0 or more variable bindings, with types
  108 type UTermCtx = [(UTermVar, UTerm)]
  109 
  110 instance Positioned UTerm where
  111   pos t =
  112     case t of
  113       Name i               -> pos i
  114       Sort p _ _           -> p
  115       Lambda p _ _         -> p
  116       App x _              -> pos x
  117       Pi p _ _             -> p
  118       Recursor i           -> pos i
  119       UnitValue p          -> p
  120       UnitType p           -> p
  121       RecordValue p _      -> p
  122       RecordType p _       -> p
  123       RecordProj x _       -> pos x
  124       PairValue p _ _      -> p
  125       PairType p _ _       -> p
  126       PairLeft x           -> pos x
  127       PairRight x          -> pos x
  128       TypeConstraint _ p _ -> p
  129       NatLit p _           -> p
  130       StringLit p _        -> p
  131       VecLit p _           -> p
  132       BVLit p _            -> p
  133       BadTerm p            -> p
  134 
  135 instance Positioned UTermVar where
  136   pos (TermVar i) = pos i
  137   pos (UnusedVar p) = p
  138 
  139 badTerm :: Pos -> UTerm
  140 badTerm = BadTerm

  142 -- | A constructor declaration of the form @c (x1 :: tp1) .. (xn :: tpn) :: tp@
  143 data CtorDecl = Ctor (PosPair Text) UTermCtx UTerm
  144   deriving (Show, TH.Lift)
  145 
  146 -- | A top-level declaration in a saw-core file
  147 data Decl
  148    = TypeDecl DefQualifier (PosPair Text) UTerm
  149      -- ^ A declaration of something having a type, where the declaration
  150      -- qualifier states what flavor of thing it is
  151    | DataDecl (PosPair Text) UTermCtx UTerm [CtorDecl]
  152      -- ^ A declaration of an inductive data types, with a name, a parameter
  153      -- context, a return type, and a list of constructor declarations
  154    | TermDef (PosPair Text) [UTermVar] UTerm
  155      -- ^ A declaration of a term having a definition, with variables
  156    | TypedDef (PosPair Text) [(UTermVar, UTerm)] UTerm UTerm
  157      -- ^ A definition of something with a specific type, with parameters
  158    | InjectCodeDecl Text Text
  159      -- ^ Some raw text to inject into a translation
  160   deriving (Show, TH.Lift)

  162 -- | A set of constraints on what 'String' names to import from a module
  163 data ImportConstraint
  164   = SpecificImports [String]
  165     -- ^ Only import the given names
  166   | HidingImports [String]
  167     -- ^ Import all but the given names
  168  deriving (Eq, Ord, Show, TH.Lift)
  169 
  170 -- | An import declaration
  171 data Import = Import { importModName :: PosPair ModuleName
  172                        -- ^ The name of the module to import
  173                      , importConstraints :: Maybe ImportConstraint
  174                        -- ^ The constraints on what to import
  175                      }
  176             deriving (Show, TH.Lift)
  177 
  178 -- | Test whether a 'String' name satisfies the constraints of an 'Import'
  179 nameSatsConstraint :: Maybe ImportConstraint -> String -> Bool
  180 nameSatsConstraint Nothing _ = True
  181 nameSatsConstraint (Just (SpecificImports ns)) n = elem n ns
  182 nameSatsConstraint (Just (HidingImports ns)) n = notElem n ns
  183

  185 -- | A module declaration gives:
  186 -- * A name for the module;
  187 -- * A list of imports; AND
  188 -- * A list of top-level declarations
  189 data Module = Module (PosPair ModuleName) [Import] [Decl]
  190   deriving (Show, TH.Lift)
  191 
  192 moduleName :: Module -> ModuleName
  193 moduleName (Module (PosPair _ mnm) _ _) = mnm
  194 
  195 -- | Get a list of all names (i.e., definitions, axioms, or primitives) declared
  196 -- in a module, along with their types and qualifiers
  197 moduleTypedDecls :: Module -> [(Text, UTerm)]
  198 moduleTypedDecls (Module _ _ decls) = concatMap helper decls where
  199   helper :: Decl -> [(Text, UTerm)]
  200   helper (TypeDecl _ (PosPair _ nm) tm) = [(nm,tm)]
  201   helper _ = []
  202 
  203 -- | Get a list of all datatypes declared in a module
  204 moduleDataDecls :: Module -> [(Text, UTermCtx, UTerm, [CtorDecl])]
  205 moduleDataDecls (Module _ _ decls) = concatMap helper decls where
  206   helper :: Decl -> [(Text, UTermCtx, UTerm, [CtorDecl])]
  207   helper (DataDecl (PosPair _ nm) params tp ctors) = [(nm, params, tp, ctors)]
  208   helper _ = []
  209 
  210 moduleTypedDataDecls :: Module -> [(Text, UTerm)]
  211 moduleTypedDataDecls =
  212   map (\(nm,p_ctx,tp,_) ->
  213         (nm, Pi (pos tp) p_ctx tp)) . moduleDataDecls
  214 
  215 -- | Get a list of all constructors declared in a module, along with the context
  216 -- of parameters for each one
  217 moduleCtorDecls :: Module -> [(UTermCtx, CtorDecl)]
  218 moduleCtorDecls =
  219   concatMap (\(_,p_ctx,_,ctors) -> map (p_ctx,) ctors) . moduleDataDecls
  220 
  221 -- | Get a list of the names and types of all the constructors in a module
  222 moduleTypedCtorDecls :: Module -> [(Text, UTerm)]
  223 moduleTypedCtorDecls =
  224   concatMap (\(_,p_ctx,_,ctors) ->
  225               map (\(Ctor (PosPair _ nm) ctx tp) ->
  226                     (nm, Pi (pos tp) (p_ctx ++ ctx) tp)) ctors)
  227   . moduleDataDecls
  228 
  229 asPiList :: UTerm -> (UTermCtx, UTerm)
  230 asPiList (Pi _ ctx1 body1) =
  231   let (ctx2,body2) = asPiList body1 in
  232   (ctx1 ++ ctx2, body2)
  233 asPiList t = ([], t)
  234 
  235 asApp :: UTerm -> (UTerm, [UTerm])
  236 asApp = go []
  237   where go l (App t u)   = go (u:l) t
  238         go l t = (t,l)
  239 
  240 -- | Build a tuple value @(x1, .., xn)@.
  241 mkTupleValue :: Pos -> [UTerm] -> UTerm
  242 mkTupleValue p [] = UnitValue p
  243 mkTupleValue _ [x] = x
  244 mkTupleValue p (x:xs) = PairValue (pos x) x (mkTupleValue p xs)
  245 
  246 -- | Build a tuple type @#(x1, .., xn)@.
  247 mkTupleType :: Pos -> [UTerm] -> UTerm
  248 mkTupleType p [] = UnitType p
  249 mkTupleType _ [x] = x
  250 mkTupleType p (x:xs) = PairType (pos x) x (mkTupleType p xs)
  251 
  252 -- | Build a projection @t.i@ of a tuple. NOTE: This function does not
  253 -- work to access the last component in a tuple, since it always
  254 -- generates a @PairLeft@.
  255 mkTupleSelector :: UTerm -> Natural -> UTerm
  256 mkTupleSelector t i
  257   | i == 1    = PairLeft t
  258   | i > 1     = mkTupleSelector (PairRight t) (i - 1)
  259   | otherwise = error "mkTupleSelector: non-positive index"
