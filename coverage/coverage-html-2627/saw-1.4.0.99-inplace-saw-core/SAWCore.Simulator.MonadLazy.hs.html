    9 module SAWCore.Simulator.MonadLazy where
   10 
   11 import Control.Monad (liftM)
   12 import Control.Monad.Identity (Identity)
   13 import Control.Monad.IO.Class
   14 import Data.IORef
   15 
   16 newtype Lazy m a = Lazy (m a)
   17 
   18 class Monad m => MonadLazy m where
   19   delay :: m a -> m (Lazy m a)
   20 
   21 force :: Lazy m a -> m a
   22 force (Lazy m) = m
   23 
   24 ready :: Monad m => a -> Lazy m a
   25 ready x = Lazy (return x)
   26 
   27 instance MonadLazy Identity where
   28   delay m = return (Lazy m)
   29 
   30 instance MonadLazy IO where
   31   delay = delayIO
   32 
   33 delayIO :: MonadIO m => m a -> m (Lazy m a)
   34 delayIO m = liftM pull (liftIO (newIORef Nothing))
   35   where
   36     pull ref = Lazy $ do
   37       r <- liftIO (readIORef ref)
   38       case r of
   39         Nothing -> do
   40           x <- m
   41           liftIO (writeIORef ref (Just x))
   42           return x
   43         Just x -> return x
