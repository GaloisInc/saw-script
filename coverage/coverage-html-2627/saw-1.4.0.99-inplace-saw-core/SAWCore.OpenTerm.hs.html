   10 {- |
   11 Module      : SAWCore.OpenTerm
   12 Copyright   : Galois, Inc. 2018
   13 License     : BSD3
   14 Stability   : experimental
   15 Portability : non-portable (language extensions)
   16 
   17 This module defines an interface to building SAW core terms in an incrementally
   18 type-checked way, meaning that type-checking is performed as the terms are
   19 built. The interface provides a convenient DSL for building terms in a pure way,
   20 where sub-terms can be freely composed and combined into super-terms without
   21 monadic sequencing or 'IO' computations; the 'IO' computation is only run at the
   22 top level when all the term-building is complete. Users of this interface can
   23 also build binding constructs like lambda- and pi-abstractions without worrying
   24 about deBruijn indices, lifting, and free variables. Instead, a key feature of
   25 this interface is that it uses higher-order abstract syntax for lambda- and
   26 pi-abstractions, meaning that the bodies of these term constructs are specified
   27 as Haskell functions that take in terms for the bound variables. The library
   28 takes care of all the deBruijn indices under the hood.
   29 
   30 To use the 'OpenTerm' API, the caller builds up 'OpenTerm's using a variety of
   31 combinators that mirror the SAW core 'Term' structure. As some useful examples
   32 of 'OpenTerm' operations, 'applyOpenTerm' applies one 'OpenTerm' to another,
   33 'globalOpenTerm' builds an 'OpenTerm' for a global identifier, and
   34 'lambdaOpenTerm' builds a lambda-abstraction. For instance, the SAW core term
   35 
   36 > \ (f : Bool -> Bool) (x : Bool) -> f x
   37 
   38 can be built with the 'OpenTerm' expression
   39 
   40 > let bool = globalOpenTerm "Prelude.Bool" in
   41 > lambdaOpenTerm "f" (arrowOpenTerm bool bool) $ \f ->
   42 > lambdaOpenTerm "x" (globalOpenTerm "Prelude.Bool") $ \x ->
   43 > applyOpenTerm f x
   44 
   45 Existing SAW core 'Term's can be used in 'OpenTerm' by applying 'closedOpenTerm'
   46 if the 'Term' is closed (meaning it has no free variables) or 'openOpenTerm' if
   47 it does, where the latter requires the context of free variables to be
   48 specified. At the top level, 'completeOpenTerm' then "completes" an 'OpenTerm'
   49 by running its underlying 'IO' computation to build and type-check the resulting
   50 SAW core 'Term'.
   51 -}

  116 -- | An open term is represented as a type-checking computation that computes a
  117 -- SAW core term and its type
  118 newtype OpenTerm = OpenTerm { unOpenTerm :: TCM SCTypedTerm }
  119 
  120 -- | \"Complete\" an 'OpenTerm' to a closed term or 'fail' on type-checking
  121 -- error
  122 completeOpenTerm :: SharedContext -> OpenTerm -> IO Term
  123 completeOpenTerm sc (OpenTerm termM) =
  124   either (fail . show) return =<<
  125   runTCM (typedVal <$> termM) sc []
  126 
  127 -- | \"Complete\" an 'OpenTerm' to a closed term and 'betaNormalize' the result
  128 completeNormOpenTerm :: SharedContext -> OpenTerm -> IO Term
  129 completeNormOpenTerm sc m =
  130   completeOpenTerm sc m >>= sawLetMinimize sc >>= betaNormalize sc
  131 
  132 -- | \"Complete\" an 'OpenTerm' to a closed term for its type
  133 completeOpenTermType :: SharedContext -> OpenTerm -> IO Term
  134 completeOpenTermType sc (OpenTerm termM) =
  135   either (fail . show) return =<<
  136   runTCM (typedType <$> termM) sc []

  138 -- | Embed a closed 'Term' into an 'OpenTerm'
  139 closedOpenTerm :: Term -> OpenTerm
  140 closedOpenTerm t = OpenTerm $ typeInferComplete t
  141 
  142 -- | Embed a 'Term' in the given typing context into an 'OpenTerm'
  143 openOpenTerm :: [(LocalName, Term)] -> Term -> OpenTerm
  144 openOpenTerm ctx t =
  145   -- Extend the local type-checking context, wherever this OpenTerm gets used,
  146   -- by appending ctx to the end, so that variables 0..length ctx-1 all get
  147   -- type-checked with ctx. If these are really the only free variables, then it
  148   -- won't matter what the rest of the ambient context is.
  149   --
  150   -- FIXME: we should check that the free variables of t are all < length ctx
  151   OpenTerm $ withCtx ctx $ typeInferComplete t
  152 
  153 -- | Build an 'OpenTerm' that 'fail's in the underlying monad when completed
  154 failOpenTerm :: String -> OpenTerm
  155 failOpenTerm str = OpenTerm $ fail str
  156 
  157 -- | Bind the result of a type-checking computation in building an 'OpenTerm'.
  158 -- NOTE: this operation should be considered \"unsafe\" because it can create
  159 -- malformed 'OpenTerm's if the result of the 'TCM' computation is used as part
  160 -- of the resulting 'OpenTerm'. For instance, @a@ should not be 'OpenTerm'.
  161 bindTCMOpenTerm :: TCM a -> (a -> OpenTerm) -> OpenTerm
  162 bindTCMOpenTerm m f = OpenTerm (m >>= unOpenTerm . f)
  163 
  164 -- | Bind the result of pretty-printing an 'OpenTerm' while building another
  165 bindPPOpenTerm :: OpenTerm -> (String -> OpenTerm) -> OpenTerm
  166 bindPPOpenTerm (OpenTerm m) f =
  167   OpenTerm $
  168   do ctx <- askCtx
  169      t <- typedVal <$> m
  170      -- XXX: this could use scPrettyTermInCtx (which builds in the call to
  171      -- PPS.render) except that it's slightly different under the covers
  172      -- (in its use of the "global" flag, and it isn't entirely clear what
  173      -- that actually does)
  174      unOpenTerm $ f $ PPS.render PPS.defaultOpts $
  175        ppTermInCtx PPS.defaultOpts (map fst ctx) t

  177 -- | Return type type of an 'OpenTerm' as an 'OpenTerm

  179 openTermType (OpenTerm m) =

  249 -- | Build an 'OpenTerm' for a pair

  258 pairLeftOpenTerm :: OpenTerm -> OpenTerm

  265 -- | Build a right-nested tuple as an 'OpenTerm'
  266 tupleOpenTerm :: [OpenTerm] -> OpenTerm
  267 tupleOpenTerm = foldr pairOpenTerm unitOpenTerm
  268 
  269 -- | Build a right-nested tuple type as an 'OpenTerm'
  270 tupleTypeOpenTerm :: [OpenTerm] -> OpenTerm
  271 tupleTypeOpenTerm = foldr pairTypeOpenTerm unitTypeOpenTerm

  273 -- | Project the @n@th element of a right-nested tuple type
  274 projTupleOpenTerm :: Integer -> OpenTerm -> OpenTerm
  275 projTupleOpenTerm 0 t = pairLeftOpenTerm t
  276 projTupleOpenTerm i t = projTupleOpenTerm (i-1) (pairRightOpenTerm t)
  277 
  278 -- | Build a right-nested tuple as an 'OpenTerm' but without adding a final unit
  279 -- as the right-most element
  280 tupleOpenTerm' :: [OpenTerm] -> OpenTerm
  281 tupleOpenTerm' [] = unitOpenTerm
  282 tupleOpenTerm' ts = foldr1 pairOpenTerm ts
  283 
  284 -- | Build a right-nested tuple type as an 'OpenTerm' but without adding a final
  285 -- unit type as the right-most element
  286 tupleTypeOpenTerm' :: [OpenTerm] -> OpenTerm
  287 tupleTypeOpenTerm' [] = unitTypeOpenTerm
  288 tupleTypeOpenTerm' ts = foldr1 pairTypeOpenTerm ts
  289 
  290 -- | Project the @i@th element from a term of a right-nested tuple term that
  291 -- does not have a final unit type as the right-most type. The first argument is
  292 -- the number of types used to make the tuple type and the second is the index.
  293 projTupleOpenTerm' :: Natural -> Natural -> OpenTerm -> OpenTerm
  294 projTupleOpenTerm' 0 _ _ =
  295   panic "projTupleOpenTerm'" ["Projection of empty tuple!"]
  296 projTupleOpenTerm' 1 0 tup = tup
  297 projTupleOpenTerm' _ 0 tup = pairLeftOpenTerm tup
  298 projTupleOpenTerm' len i tup =
  299   projTupleOpenTerm' (len-1) (i-1) $ pairRightOpenTerm tup
  300 
  301 -- | Build a record value as an 'OpenTerm'
  302 recordOpenTerm :: [(FieldName, OpenTerm)] -> OpenTerm
  303 recordOpenTerm flds_ts =
  304   OpenTerm $ do let (flds,ots) = unzip flds_ts
  305                 ts <- mapM unOpenTerm ots
  306                 typeInferComplete $ RecordValue $ zip flds ts

  308 -- | Build a record type as an 'OpenTerm'
  309 recordTypeOpenTerm :: [(FieldName, OpenTerm)] -> OpenTerm
  310 recordTypeOpenTerm flds_ts =
  311   OpenTerm $ do let (flds,ots) = unzip flds_ts
  312                 ts <- mapM unOpenTerm ots
  313                 typeInferComplete $ RecordType $ zip flds ts

  315 -- | Project a field from a record
  316 projRecordOpenTerm :: OpenTerm -> FieldName -> OpenTerm
  317 projRecordOpenTerm (OpenTerm m) f =
  318   OpenTerm $ do t <- m

  321 -- | Build an 'OpenTerm' for a constructor applied to its arguments
  322 ctorOpenTerm :: Ident -> [OpenTerm] -> OpenTerm
  323 ctorOpenTerm c all_args = applyOpenTermMulti ctor_open_term all_args
  324   where
  325     ctor_open_term =
  326       OpenTerm $
  327       do maybe_ctor <- liftTCM scFindCtor c
  328          ctor <- case maybe_ctor of
  329                    Just ctor -> pure ctor
  330                    Nothing -> throwTCError $ NoSuchCtor (ModuleIdentifier c)
  331          typeInferComplete (Constant (ctorName ctor) :: TermF Term)
  332 
  333 -- | Build an 'OpenTerm' for a datatype applied to its arguments
  334 dataTypeOpenTerm :: Ident -> [OpenTerm] -> OpenTerm
  335 dataTypeOpenTerm d all_args = applyOpenTermMulti dt_open_term all_args
  336   where
  337     dt_open_term =
  338       OpenTerm $
  339       do maybe_dt <- liftTCM scFindDataType d
  340          dt <- case maybe_dt of
  341                  Just dt -> pure dt
  342                  Nothing -> throwTCError $ NoSuchDataType (ModuleIdentifier d)
  343          typeInferComplete (Constant (dtName dt) :: TermF Term)
  344 
  345 -- | Build an 'OpenTerm' for a global name with a definition
  346 globalOpenTerm :: Ident -> OpenTerm
  347 globalOpenTerm ident =
  348   OpenTerm (liftTCM scGlobalTypedTerm ident)

  350 -- | Build an 'OpenTerm' for an 'Ident', which can either refer to a definition,

  352 identOpenTerm :: Ident -> OpenTerm

  356 -- | Build an 'OpenTerm' for a named variable.
  357 variableOpenTerm :: ExtCns Term -> OpenTerm
  358 variableOpenTerm ec = OpenTerm (liftTCM scVariable ec >>= typeInferComplete)

  360 -- | Apply an 'OpenTerm' to another
  361 applyOpenTerm :: OpenTerm -> OpenTerm -> OpenTerm
  362 applyOpenTerm (OpenTerm f) (OpenTerm arg) =
  363   OpenTerm ((App <$> f <*> arg) >>= typeInferComplete)
  364 
  365 -- | Apply an 'OpenTerm' to 0 or more arguments
  366 applyOpenTermMulti :: OpenTerm -> [OpenTerm] -> OpenTerm
  367 applyOpenTermMulti = foldl applyOpenTerm
  368 
  369 -- | Apply a named global to 0 or more arguments
  370 applyGlobalOpenTerm :: Ident -> [OpenTerm] -> OpenTerm
  371 applyGlobalOpenTerm ident = applyOpenTermMulti (globalOpenTerm ident)

  373 -- | Compute the output type of applying a function of a given type to an
  374 -- argument. That is, given @tp@ and @arg@, compute the type of applying any @f@
  375 -- of type @tp@ to @arg@.
  376 applyPiOpenTerm :: OpenTerm -> OpenTerm -> OpenTerm
  377 applyPiOpenTerm (OpenTerm m_f) (OpenTerm m_arg) =
  378   OpenTerm $
  379   do f <- m_f
  380      arg <- m_arg
  381      ret <- applyPiTyped (NotFuncTypeInApp f arg) (typedVal f) arg
  382      typeInferComplete ret
  383 
  384 -- | Get the argument type of a function type, 'fail'ing if the input term is
  385 -- not a function type
  386 piArgOpenTerm :: OpenTerm -> OpenTerm
  387 piArgOpenTerm (OpenTerm m) =
  388   OpenTerm $ m >>= \case
  389   (unwrapTermF . typedVal -> Pi _ tp _) -> typeInferComplete tp
  390   t ->
  391     do ctx <- askCtx
  392        fail ("piArgOpenTerm: not a pi type: " ++
  393              scPrettyTermInCtx PPS.defaultOpts (map fst ctx) (typedVal t))
  394 
  395 -- | Build an 'OpenTerm' for the top variable in the current context, by
  396 -- building the 'TCM' computation which checks how much longer the context has
  397 -- gotten since the variable was created and uses this to compute its deBruijn
  398 -- index
  399 openTermTopVar :: TCM OpenTerm
  400 openTermTopVar =
  401   do outer_ctx <- askCtx
  402      return $ OpenTerm $ do
  403        inner_ctx <- askCtx
  404        typeInferComplete (LocalVar (length inner_ctx
  405                                     - length outer_ctx) :: TermF Term)
  406 
  407 -- | Build an open term inside a binder of a variable with the given name and
  408 -- type, where the binder is represented as a Haskell function on 'OpenTerm's
  409 bindOpenTerm :: LocalName -> SCTypedTerm -> (OpenTerm -> OpenTerm) ->
  410                 TCM SCTypedTerm
  411 bindOpenTerm x tp body_f =
  412   do tp_whnf <- typeCheckWHNF $ typedVal tp
  413      withVar x tp_whnf (openTermTopVar >>= (unOpenTerm . body_f))
  414 
  415 -- | Build a lambda abstraction as an 'OpenTerm'
  416 lambdaOpenTerm :: LocalName -> OpenTerm -> (OpenTerm -> OpenTerm) -> OpenTerm
  417 lambdaOpenTerm x (OpenTerm tpM) body_f = OpenTerm $
  418   do tp <- tpM
  419      body <- bindOpenTerm x tp body_f
  420      typeInferComplete $ Lambda x tp body
  421 
  422 -- | Build a nested sequence of lambda abstractions as an 'OpenTerm'
  423 lambdaOpenTermMulti :: [(LocalName, OpenTerm)] -> ([OpenTerm] -> OpenTerm) ->
  424                        OpenTerm
  425 lambdaOpenTermMulti xs_tps body_f =
  426   foldr (\(x,tp) rest_f xs ->
  427           lambdaOpenTerm x tp (rest_f . (:xs))) (body_f . reverse) xs_tps []
  428 
  429 -- | Build a Pi abstraction as an 'OpenTerm'
  430 piOpenTerm :: LocalName -> OpenTerm -> (OpenTerm -> OpenTerm) -> OpenTerm
  431 piOpenTerm x (OpenTerm tpM) body_f = OpenTerm $
  432   do tp <- tpM
  433      body <- bindOpenTerm x tp body_f
  434      typeInferComplete $ Pi x tp body
  435 
  436 -- | Build a non-dependent function type.
  437 arrowOpenTerm :: LocalName -> OpenTerm -> OpenTerm -> OpenTerm
  438 arrowOpenTerm x tp body = piOpenTerm x tp (const body)
  439 
  440 -- | Build a nested sequence of Pi abstractions as an 'OpenTerm'
  441 piOpenTermMulti :: [(LocalName, OpenTerm)] -> ([OpenTerm] -> OpenTerm) ->
  442                    OpenTerm
  443 piOpenTermMulti xs_tps body_f =
  444   foldr (\(x,tp) rest_f xs ->
  445           piOpenTerm x tp (rest_f . (:xs))) (body_f . reverse) xs_tps []

  447 -- | Build a let expression as an 'OpenTerm'. This is equivalent to
  448 -- > 'applyOpenTerm' ('lambdaOpenTerm' x tp body) rhs
  449 letOpenTerm :: LocalName -> OpenTerm -> OpenTerm -> (OpenTerm -> OpenTerm) ->
  450                OpenTerm
  451 letOpenTerm x tp rhs body_f = applyOpenTerm (lambdaOpenTerm x tp body_f) rhs
  452 
  453 -- | Build a let expression as an 'OpenTerm'. This is equivalent to
  454 -- > 'applyOpenTerm' ('lambdaOpenTerm' x tp body) rhs
  455 sawLetOpenTerm :: LocalName -> OpenTerm -> OpenTerm -> OpenTerm ->
  456                   (OpenTerm -> OpenTerm) -> OpenTerm
  457 sawLetOpenTerm x tp tp_ret rhs body_f =
  458   applyOpenTermMulti (globalOpenTerm "Prelude.sawLet")
  459   [tp, tp_ret, rhs, lambdaOpenTerm x tp body_f]

  461 -- | Build a bitvector type with the given length
  462 bitvectorTypeOpenTerm :: OpenTerm -> OpenTerm
  463 bitvectorTypeOpenTerm w =
  464   applyGlobalOpenTerm "Prelude.Vec" [w, globalOpenTerm "Prelude.Bool"]

  466 -- | Build a SAW core term for a list with the given element type
  467 listOpenTerm :: OpenTerm -> [OpenTerm] -> OpenTerm
  468 listOpenTerm tp elems =
  469   foldr (\x l -> ctorOpenTerm "Prelude.Cons" [tp, x, l])

  472 -- | Build the type @Either a b@ from types @a@ and @b@

  477 -- Monadic operations for building terms including 'IO' actions
  478 
  479 -- | The monad for building 'OpenTerm's if you want to add in 'IO' actions. This
  480 -- is just the type-checking monad, but we give it a new name to keep this
  481 -- module self-contained.
  482 newtype OpenTermM a = OpenTermM { unOpenTermM :: TCM a }
  483                     deriving (Functor, Applicative, Monad)
  484 
  485 instance MonadIO OpenTermM where
  486   liftIO = OpenTermM . liftIO
  487 
  488 -- | \"Run\" an 'OpenTermM' computation to produce an 'OpenTerm'
  489 runOpenTermM :: OpenTermM OpenTerm -> OpenTerm
  490 runOpenTermM (OpenTermM m) =
  491   OpenTerm $ join $ fmap unOpenTerm m
  492 
  493 -- | \"Complete\" an 'OpenTerm' build in 'OpenTermM' to a closed term, or 'fail'
  494 -- on a type-checking error
  495 completeOpenTermM :: SharedContext -> OpenTermM OpenTerm -> IO Term
  496 completeOpenTermM sc m = completeOpenTerm sc (runOpenTermM m)
  497 
  498 -- | \"De-duplicate\" an open term, so that duplicating the returned 'OpenTerm'
  499 -- does not lead to duplicated WHNF work
  500 dedupOpenTermM :: OpenTerm -> OpenTermM OpenTerm
  501 dedupOpenTermM (OpenTerm trmM) =
  502   OpenTermM $ do trm <- trmM
  503                  return $ OpenTerm $ return trm
  504 
  505 -- | Build an open term inside a binder of a variable with the given name and
  506 -- type, where the binder is represented as a monadic Haskell function on
  507 -- 'OpenTerm's that also returns an auxiliary value. Returns the normalized type
  508 -- and the body, along with the auxiliary result returned by the body-generating
  509 -- function.
  510 bindOpenTermAuxM ::
  511   LocalName -> OpenTerm ->
  512   (OpenTerm -> OpenTermM (OpenTerm, a)) ->
  513   OpenTermM (SCTypedTerm, SCTypedTerm, a)
  514 bindOpenTermAuxM x (OpenTerm tpM) body_f =
  515   OpenTermM $
  516   do tp <- tpM
  517      tp_whnf <- liftTCM scTypedTermWHNF tp
  518      (OpenTerm bodyM, a) <-
  519        withVar x (typedVal tp_whnf) (openTermTopVar >>= (unOpenTermM . body_f))
  520      body <- bodyM
  521      return (tp_whnf, body, a)
  522 
  523 -- | Build a lambda abstraction in the 'OpenTermM' monad
  524 lambdaOpenTermM ::
  525   LocalName -> OpenTerm -> (OpenTerm -> OpenTermM OpenTerm) ->
  526   OpenTermM OpenTerm
  527 lambdaOpenTermM x tp body_f =
  528   fst <$> lambdaOpenTermAuxM x tp (body_f >=> (\t -> return (t, ())))
  529 
  530 -- | Build a pi abstraction in the 'OpenTermM' monad
  531 piOpenTermM ::
  532   LocalName -> OpenTerm -> (OpenTerm -> OpenTermM OpenTerm) ->
  533   OpenTermM OpenTerm
  534 piOpenTermM x tp body_f =
  535   fst <$> piOpenTermAuxM x tp (body_f >=> (\t -> return (t, ())))
  536 
  537 -- | Build a lambda abstraction with an auxiliary return value in the
  538 -- 'OpenTermM' monad
  539 lambdaOpenTermAuxM ::
  540   LocalName -> OpenTerm ->
  541   (OpenTerm -> OpenTermM (OpenTerm, a)) ->
  542   OpenTermM (OpenTerm, a)
  543 lambdaOpenTermAuxM x tp body_f =
  544   do (tp', body, a) <- bindOpenTermAuxM x tp body_f
  545      return (OpenTerm (typeInferComplete $ Lambda x tp' body), a)

  548 -- monad
  549 piOpenTermAuxM ::
  550   LocalName -> OpenTerm -> (OpenTerm -> OpenTermM (OpenTerm, a)) ->
  551   OpenTermM (OpenTerm, a)
  552 piOpenTermAuxM x tp body_f =
  553   do (tp', body, a) <- bindOpenTermAuxM x tp body_f
  554      return (OpenTerm (typeInferComplete $ Pi x tp' body), a)

  560 class OpenTermLike t where
  561   -- | Convert an 'OpenTerm' to a @t@
  562   openTermLike :: OpenTerm -> t
  563   -- | Get the type of a @t@
  564   typeOfTermLike :: t -> t
  565   -- | Build a @t@ from a 'FlatTermF'
  566   flatTermLike :: FlatTermF t -> t
  567   -- | Apply a @t@ to another @t@
  568   applyTermLike :: t -> t -> t
  569   -- | Build a lambda abstraction as a @t@
  570   lambdaTermLike :: LocalName -> t -> (t -> t) -> t
  571   -- | Build a pi abstraction as a @t@
  572   piTermLike :: LocalName -> t -> (t -> t) -> t
  573   -- | Build a @t@ for a constructor applied to its arguments
  574   ctorTermLike :: Ident -> [t] -> t
  575   -- | Build a @t@ for a datatype applied to its arguments
  576   dataTypeTermLike :: Ident -> [t] -> t

  578 instance OpenTermLike OpenTerm where
  579   openTermLike = id
  580   typeOfTermLike = openTermType
  581   flatTermLike = flatOpenTerm
  582   applyTermLike = applyOpenTerm
  583   lambdaTermLike = lambdaOpenTerm
  584   piTermLike = piOpenTerm
  585   ctorTermLike = ctorOpenTerm
  586   dataTypeTermLike = dataTypeOpenTerm

  588 -- Lift an OpenTermLike instance from t to functions from some type a to t,
  589 -- where the OpenTermLike methods pass the same input a argument to all subterms
  590 instance OpenTermLike t => OpenTermLike (a -> t) where
  591   openTermLike t = const $ openTermLike t
  592   typeOfTermLike t = \x -> typeOfTermLike (t x)
  593   flatTermLike ftf = \x -> flatTermLike (fmap ($ x) ftf)
  594   applyTermLike f arg = \x -> applyTermLike (f x) (arg x)
  595   lambdaTermLike nm tp bodyF =
  596     \x -> lambdaTermLike nm (tp x) (\y -> bodyF (const y) x)
  597   piTermLike nm tp bodyF =
  598     \x -> piTermLike nm (tp x) (\y -> bodyF (const y) x)
  599   ctorTermLike c args = \x -> ctorTermLike c (map ($ x) args)
  600   dataTypeTermLike d args = \x -> dataTypeTermLike d (map ($ x) args)
  601 
  602 -- This is the same as the function instance above
  603 instance OpenTermLike t => OpenTermLike (Reader r t) where
  604   openTermLike t = reader $ openTermLike t
  605   typeOfTermLike t = reader $ typeOfTermLike $ runReader t
  606   flatTermLike ftf = reader $ flatTermLike $ fmap runReader ftf
  607   applyTermLike f arg = reader $ applyTermLike (runReader f) (runReader arg)
  608   lambdaTermLike x tp body =
  609     reader $ lambdaTermLike x (runReader tp) (runReader . body . reader)
  610   piTermLike x tp body =
  611     reader $ piTermLike x (runReader tp) (runReader . body . reader)
  612   ctorTermLike c args = reader $ ctorTermLike c $ map runReader args
  613   dataTypeTermLike d args = reader $ dataTypeTermLike d $ map runReader args

  739 -- | A map from each deBruijn index to a count of its occurrences in a term
  740 newtype VarOccs = VarOccs [Integer]
  741 
  742 -- | Make a 'VarOccs' with a single occurrence of a deBruijn index
  743 varOccs1 :: DeBruijnIndex -> VarOccs
  744 varOccs1 i = VarOccs (take i (repeat 0) ++ [1])
  745 
  746 -- | Move a 'VarOccs' out of a binder by returning the number of occurrences of
  747 -- deBruijn index 0 along with the result of subtracting 1 from all other indices
  748 unconsVarOccs :: VarOccs -> (Integer, VarOccs)
  749 unconsVarOccs (VarOccs []) = (0, VarOccs [])
  750 unconsVarOccs (VarOccs (cnt:occs)) = (cnt, VarOccs occs)
  751 
  752 -- | Multiply every index in a 'VarOccs' by a constant
  753 multVarOccs :: Integer -> VarOccs -> VarOccs
  754 multVarOccs i (VarOccs occs) = VarOccs $ map (* i) occs
  755 
  756 -- | The infinite list of zeroes
  757 zeroes :: [Integer]
  758 zeroes = 0:zeroes
  759 
  760 instance Semigroup VarOccs where
  761   (VarOccs occs1) <> (VarOccs occs2)
  762     | length occs1 < length occs2
  763     = VarOccs (zipWith (+) (occs1 ++ zeroes) occs2)
  764   (VarOccs occs1) <> (VarOccs occs2)
  765     = VarOccs (zipWith (+) occs1 (occs2 ++ zeroes))
  766 
  767 instance Monoid VarOccs where
  768   mempty = VarOccs []
  769 
  770 -- | 'listen' to the output of a writer computation and return that output but
  771 -- drop it from the writer output of the computation
  772 listenDrop :: MonadWriter w m => m a -> m (a, w)
  773 listenDrop m = pass (listen m >>= \aw -> return (aw, const mempty))
  774 
  775 -- | The monad for sawLet minimization
  776 type SLMinM = StateT (IntMap (Term, VarOccs)) (WriterT VarOccs IO)
  777 
  778 -- | Find every subterm of the form @sawLet a b rhs (\ x -> body)@ and, whenever
  779 -- @x@ occurs at most once in @body@, unfold the @sawLet@ by substituting @rhs@
  780 -- into @body@
  781 sawLetMinimize :: SharedContext -> Term -> IO Term
  782 sawLetMinimize sc t_top =
  783   fst <$> runWriterT (evalStateT (slMinTerm t_top) IntMap.empty) where
  784   slMinTerm :: Term -> SLMinM Term
  785   slMinTerm (Unshared tf) = slMinTermF tf
  786   slMinTerm t@(STApp { stAppIndex = i }) =
  787     do memo_table <- get
  788        case IntMap.lookup i memo_table of
  789          Just (t', occs) ->
  790            -- NOTE: the fact that we explicitly tell occs here means that we are
  791            -- going to double-count variable occurrences for multiple
  792            -- occurrences of the same subterm. That is, a variable occurence
  793            -- counts for each copy of a shared subterm.
  794            tell occs >> return t'
  795          Nothing ->
  796            do (t', occs) <- listen $ slMinTermF (unwrapTermF t)
  797               modify $ IntMap.insert i (t', occs)
  798               return t'
  799 
  800   slMinTermF :: TermF Term -> SLMinM Term
  801   slMinTermF tf@(App (asApplyAll ->
  802                       (isGlobalDef "Prelude.sawLet" -> Just _, [_a, _b, rhs]))
  803                  (asLambda -> Just (_, _, body))) =
  804     do (body', (unconsVarOccs ->
  805                 (x_cnt, body_occs))) <- listenDrop $ slMinTerm body
  806        if x_cnt > 1 then slMinTermF' tf else
  807          do (rhs', rhs_occs) <- listenDrop $ slMinTerm rhs
  808             tell (multVarOccs x_cnt rhs_occs <> body_occs)
  809             liftIO $ instantiateVar sc 0 rhs' body'
  810   slMinTermF tf = slMinTermF' tf
  811 
  812   slMinTermF' :: TermF Term -> SLMinM Term
  813   slMinTermF' (FTermF ftf) = slMinFTermF ftf
  814   slMinTermF' (App f arg) =
  815     do f' <- slMinTerm f
  816        arg' <- slMinTerm arg
  817        liftIO $ scTermF sc (App f' arg')
  818   slMinTermF' (Lambda x tp body) =
  819     do tp' <- slMinTerm tp
  820        (body', body_occs) <- listenDrop $ slMinTerm body
  821        tell $ snd $ unconsVarOccs body_occs
  822        liftIO $ scTermF sc (Lambda x tp' body')
  823   slMinTermF' (Pi x tp body) =
  824     do tp' <- slMinTerm tp
  825        (body', body_occs) <- listenDrop $ slMinTerm body
  826        tell $ snd $ unconsVarOccs body_occs
  827        liftIO $ scTermF sc (Pi x tp' body')
  828   slMinTermF' tf@(LocalVar i) =
  829     tell (varOccs1 i) >> liftIO (scTermF sc tf)
  830   slMinTermF' tf@(Constant _) = liftIO (scTermF sc tf)
  831   slMinTermF' tf@(Variable _) = liftIO (scTermF sc tf)
  832 
  833   slMinFTermF :: FlatTermF Term -> SLMinM Term
  834   slMinFTermF ftf = traverse slMinTerm ftf >>= liftIO . scFlatTermF sc
