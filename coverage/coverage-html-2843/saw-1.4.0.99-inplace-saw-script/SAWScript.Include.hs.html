   19 -- Type shorthand for the file-reader function we want.
   20 --
   21 -- (The file-reader function is passed down to avoid a dependency
   22 -- cycle.)
   23 --
   24 -- The `Bool` argument is @once@ meaning `True` for @include_once@
   25 -- and `False` for ordinary repeating @include@.
   26 --
   27 type Reader = FilePath -> Bool -> IO (Either [Text] [Stmt])
   28 
   29 -- Type shorthand for the type of process*.
   30 type Processor a = Reader -> a -> IO (Either [Text] a)
   31 
   32 -- | Context for the pass: the reader for include files
   33 --   plus the error messages accumulated so far.
   34 --
   35 --   The error list is a list of lists of Text, the idea being that we
   36 --   cons each batch of errors we get back from the reader on the
   37 --   front of the outer list, and read it off at the end with @concat
   38 --   $ reverse errs@.
   39 --
   40 --   Note that we do not _generate_ errors, only pass on errors
   41 --   returned by the reader. For the moment, that does not include
   42 --   warnings, so there's no need to handle warnings in here either.
   43 --
   44 data Ctx = Ctx {
   45     ctxReader :: Reader,
   46     ctxErrs :: IORef [[Text]]
   47 }
   48 
   49 ctxAddErrs :: Ctx -> [Text] -> IO ()
   50 ctxAddErrs ctx errs = do
   51     let errsRef = ctxErrs ctx
   52     errlists <- IORef.readIORef errsRef
   53     IORef.writeIORef errsRef (errs : errlists)
   54

   57 -- Main recursive pass
   58 
   59 -- | Resolve includes, for expressions.
   60 incs'expr :: Ctx -> Expr -> IO Expr
   61 incs'expr ctx e0 =
   62   case e0 of
   63     Bool{} -> pure e0
   64     String{} -> pure e0
   65     Int{} -> pure e0
   66     Code{} -> pure e0
   67     CType{} -> pure e0
   68     Array pos es -> do
   69         es' <- mapM (incs'expr ctx) es
   70         pure $ Array pos es'
   71     Block pos (stmts, lastexpr) -> do
   72         stmts' <- incs'stmts ctx stmts
   73         lastexpr' <- incs'expr ctx lastexpr
   74         pure $ Block pos (stmts', lastexpr')
   75     Tuple pos es -> do
   76         es' <- mapM (incs'expr ctx) es
   77         pure $ Tuple pos es'
   78     Record pos members -> do
   79         members' <- traverse (incs'expr ctx) members
   80         pure $ Record pos members'
   81     Index pos e1 e2 -> do
   82         e1' <- incs'expr ctx e1
   83         e2' <- incs'expr ctx e2
   84         pure $ Index pos e1' e2'
   85     Lookup pos e1 field -> do
   86         e1' <- incs'expr ctx e1
   87         pure $ Lookup pos e1' field
   88     TLookup pos e1 ix -> do
   89         e1' <- incs'expr ctx e1
   90         pure $ TLookup pos e1' ix
   91     Var{} -> pure e0
   92     Lambda pos mname pat e1 -> do
   93         e1' <- incs'expr ctx e1
   94         pure $ Lambda pos mname pat e1'
   95     Application pos e1 e2 -> do
   96         e1' <- incs'expr ctx e1
   97         e2' <- incs'expr ctx e2
   98         pure $ Application pos e1' e2'
   99     Let pos ds e1 -> do
  100         ds' <- incs'declgroup ctx ds
  101         e1' <- incs'expr ctx e1
  102         pure $ Let pos ds' e1'
  103     TSig pos e1 ty -> do
  104         e1' <- incs'expr ctx e1
  105         pure $ TSig pos e1' ty
  106     IfThenElse pos e1 e2 e3 -> do
  107         e1' <- incs'expr ctx e1
  108         e2' <- incs'expr ctx e2
  109         e3' <- incs'expr ctx e3
  110         pure $ IfThenElse pos e1' e2' e3'
  111 
  112 -- | Resolve includes, for statements.
  113 --
  114 --   Returns a list of statements so included files can be inserted.
  115 incs'stmt :: Ctx -> Stmt -> IO [Stmt]
  116 incs'stmt ctx s0 = case s0 of
  117     StmtBind pos pat e1 -> do
  118         e1' <- incs'expr ctx e1
  119         let s0' = StmtBind pos pat e1'
  120         pure [s0']
  121     StmtLet pos rebindable ds -> do
  122         ds' <- incs'declgroup ctx ds
  123         let s0' = StmtLet pos rebindable ds'
  124         pure [s0']
  125     StmtCode{} -> pure [s0]
  126     StmtImport{} -> pure [s0]
  127     StmtInclude _pos name once -> do
  128         result <- (ctxReader ctx) (Text.unpack name) once
  129         case result of
  130             Left errs -> do
  131                 ctxAddErrs ctx errs
  132                 pure []
  133             Right stmts' ->
  134                 pure stmts'
  135     StmtTypedef{} -> pure [s0]
  136     StmtPushdir{} -> pure [s0]
  137     StmtPopdir{} -> pure [s0]
  138 
  139 -- | Resolve includes, for a list of statements.
  140 --
  141 --   Includes the implicit call to `concat`.
  142 incs'stmts :: Ctx -> [Stmt] -> IO [Stmt]
  143 incs'stmts ctx stmts = do
  144     stmtses <- mapM (incs'stmt ctx) stmts
  145     pure $ concat stmtses
  146 
  147 -- | Resolve includes, for a single declaration.
  148 incs'decl :: Ctx -> Decl -> IO Decl
  149 incs'decl ctx d = do
  150     def' <- incs'expr ctx (dDef d)
  151     pure $ d { dDef = def' }
  152 
  153 -- | Resolve includes, for a declaration group.
  154 incs'declgroup :: Ctx -> DeclGroup -> IO DeclGroup
  155 incs'declgroup ctx dg = case dg of
  156     Recursive ds -> do
  157         ds' <- mapM (incs'decl ctx) ds
  158         pure $ Recursive ds'
  159     NonRecursive d -> do
  160         d' <- incs'decl ctx d
  161         pure $ NonRecursive d'
  162

  165 -- External interface
  166 
  167 process :: Reader -> (Ctx -> a -> IO a) -> a -> IO (Either [Text] a)
  168 process reader entrypoint tree = do
  169     errsRef <- IORef.newIORef []
  170     let ctx = Ctx { ctxReader = reader, ctxErrs = errsRef }
  171     tree' <- entrypoint ctx tree
  172     errlists <- IORef.readIORef errsRef
  173     case errlists of
  174         [] -> return $ Right tree'
  175         _ -> return $ Left (concat $ reverse errlists)
  176 
  177 processExpr :: Reader -> Expr -> IO (Either [Text] Expr)
  178 processExpr reader e = process reader incs'expr e
  179 
  180 processStmts :: Reader -> [Stmt] -> IO (Either [Text] [Stmt])
  181 processStmts reader stmts = process reader incs'stmts stmts
