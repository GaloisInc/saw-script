   55 -- | Infer the type of an untyped term and complete it to a 'Term'.
   56 inferCompleteTerm ::
   57   SharedContext -> Maybe ModuleName -> Un.UTerm -> IO (Either PPS.Doc Term)
   58 inferCompleteTerm sc mnm t =
   59   do res <- runTCM (typeInferCompleteUTerm t) sc mnm
   60      case res of
   61        Right t' -> pure (Right t')
   62        Left err ->
   63          do ne <- scGetNamingEnv sc
   64             let opts = PPS.defaultOpts
   65             pure (Left (prettyTCError opts ne err))
   66 
   67 -- | Pretty-print a type-checking error
   68 ppTCError :: TCError -> String
   69 ppTCError e =
   70   PPS.render PPS.defaultOpts $
   71   prettyTCError PPS.defaultOpts emptyDisplayNameEnv e
   72 
   73 -- | Pretty-print a type-checking error
   74 prettyTCError :: PPS.Opts -> DisplayNameEnv -> TCError -> PPS.Doc
   75 prettyTCError opts ne e = helper Nothing e where
   76 
   77   ppWithPos :: Maybe Pos -> [PPS.Doc] -> PPS.Doc
   78   ppWithPos maybe_p strs =
   79     case maybe_p of
   80       Just p -> vcat (pretty (ppPos p) : strs)
   81       Nothing -> vcat strs
   82 
   83   helper :: Maybe Pos -> TCError -> PPS.Doc
   84   helper mp err =
   85     case err of
   86       UnboundName str ->
   87         ppWithPos mp [ "Unbound name:" <+> pretty str ]
   88       EmptyVectorLit ->
   89         ppWithPos mp [ "Empty vector literal"]
   90       NoSuchDataType d ->
   91         ppWithPos mp [ "No such data type:" <+> pretty (show d) ]
   92       DeclError nm reason ->
   93         ppWithPos mp [ "Malformed declaration for" <+> pretty (show nm), pretty reason ]
   94       ErrorPos p err' ->
   95         helper (Just p) err'
   96       ErrorUTerm t err' ->
   97         ppWithPos mp [ "While typechecking term:"
   98                      , indent 2 $ Un.prettyUTerm t
   99                      , helper mp err'
  100                      ]
  101       TermError err' ->
  102         ppWithPos mp [ prettyTermError opts ne err' ]

  107 -- | The 'ReaderT' environment for a computation to typecheck a
  108 -- SAWCore parser AST.
  109 data TCEnv =
  110   TCEnv
  111   { tcSharedContext :: SharedContext -- ^ the SAW context
  112   , tcModName :: Maybe ModuleName -- ^ the current module name
  113   , tcLocals :: Map LocalName (VarName, Term) -- ^ the mapping of display names to variables
  114   }
  115 
  116 -- | Errors that can occur during type-checking
  117 data TCError
  118   = UnboundName Text
  119   | EmptyVectorLit
  120   | NoSuchDataType NameInfo
  121   | DeclError Text String
  122   | ErrorPos Pos TCError
  123   | ErrorUTerm Un.UTerm TCError
  124   | TermError SC.TermError
  125 
  126 -- | The monad for computations to typecheck a SAWCore parser AST.
  127 newtype TCM a = TCM (ReaderT TCEnv (ExceptT TCError IO) a)
  128   deriving (Functor, Applicative, Monad, MonadIO)
  129 
  130 runTCM ::
  131   TCM a -> SharedContext -> Maybe ModuleName -> IO (Either TCError a)
  132 runTCM (TCM m) sc mnm =
  133   runExceptT (runReaderT m (TCEnv sc mnm Map.empty))
  134 
  135 -- | Read the current module name
  136 askSharedContext :: TCM SharedContext
  137 askSharedContext = TCM (asks tcSharedContext)
  138 
  139 -- | Read the current module name
  140 askModName :: TCM (Maybe ModuleName)
  141 askModName = TCM (asks tcModName)
  142 
  143 -- | Read the current context of named variables
  144 askLocals :: TCM (Map LocalName (VarName, Term))
  145 askLocals = TCM (asks tcLocals)
  146 
  147 -- | Look up the current module name, raising an error if it is not set
  148 getModuleName :: TCM ModuleName
  149 getModuleName =
  150   do maybe_modname <- askModName
  151      case maybe_modname of
  152        Just mnm -> return mnm
  153        Nothing ->
  154          panic "getModuleName" ["Current module name not set during typechecking"]
  155 
  156 -- | Throw a type-checking error.
  157 throwTCError :: TCError -> TCM a
  158 throwTCError e = TCM (throwError e)
  159 
  160 -- | Augment and rethrow any 'TCError' thrown by the given computation.
  161 rethrowTCError :: (TCError -> TCError) -> TCM a -> TCM a
  162 rethrowTCError f (TCM m) = TCM (catchError m (throwError . f))
  163 
  164 -- | Run a type-checking computation @m@ and tag any error it throws with the
  165 -- 'ErrorUTerm' constructor.
  166 withErrorUTerm :: Un.UTerm -> TCM a -> TCM a
  167 withErrorUTerm t = rethrowTCError (ErrorUTerm t)
  168 
  169 -- | Run a type-checking computation @m@ and tag any error it throws with the
  170 -- given position, using the 'ErrorPos' constructor, unless that error is
  171 -- already tagged with a position.
  172 atPos :: Pos -> TCM a -> TCM a
  173 atPos p = rethrowTCError (ErrorPos p)
  174 
  175 -- | Lift a computation from the SAWCore term-building monad 'SC.SCM'
  176 -- into 'TCM'.
  177 liftSCM :: SC.SCM a -> TCM a
  178 liftSCM m =
  179   do sc <- askSharedContext
  180      result <- liftIO $ SC.runSCM sc m
  181      case result of
  182        Left e -> throwTCError (TermError e)
  183        Right x -> pure x

  187 -- | Resolve a name.
  188 inferResolveName :: Text -> TCM Term
  189 inferResolveName n =
  190   do nctx <- askLocals
  191      mnm <- getModuleName
  192      sc <- askSharedContext
  193      mm <- liftIO $ scGetModuleMap sc
  194      let ident = mkIdent mnm n
  195      case (Map.lookup n nctx, resolveNameInMap mm ident) of
  196        (Just (vn, tp), _) ->
  197          liftSCM $ SC.scmVariable vn tp
  198        (_, Just rn) ->
  199          do let c = resolvedNameName rn
  200             liftSCM $ SC.scmConst c
  201        (Nothing, Nothing) ->
  202          throwTCError $ UnboundName n
  203 
  204 -- | The debugging level
  205 debugLevel :: Int
  206 debugLevel = 0

  208 -- | Print debugging output if 'debugLevel' is greater than 0
  209 typeInferDebug :: String -> TCM ()
  210 typeInferDebug str | debugLevel > 0 = liftIO $ traceIO str
  211 typeInferDebug _ = return ()

  213 -- | Completely typecheck an untyped SAWCore AST.
  214 typeInferCompleteUTerm :: Un.UTerm -> TCM Term
  215 typeInferCompleteUTerm t =
  216   do typeInferDebug ("typechecking term: " ++ show t)
  217      res <- atPos (pos t) $ withErrorUTerm t $ typeInferCompleteTerm t
  218      ty <- liftSCM $ SC.scmTypeOf res
  219      typeInferDebug ("completed typechecking term: " ++ show t ++ "\n"
  220                      ++ "type = " ++ show ty)
  221      return res
  222 
  223 -- | Main workhorse function for type inference on untyped terms
  224 typeInferCompleteTerm :: Un.UTerm -> TCM Term
  225 typeInferCompleteTerm uterm =
  226   case uterm of
  227     Un.Name (PosPair _ n) ->
  228       inferResolveName n
  229 
  230     Un.Sort _ srt h ->
  231       liftSCM $ SC.scmSortWithFlags srt h
  232 
  233     Un.Recursor (PosPair _ str) s ->
  234       do mnm <- getModuleName
  235          sc <- askSharedContext
  236          mm <- liftIO $ scGetModuleMap sc
  237          let dt_ident = mkIdent mnm str
  238          dt <- case findDataTypeInMap dt_ident mm of
  239            Just d -> return d
  240            Nothing -> throwTCError $ NoSuchDataType (ModuleIdentifier dt_ident)
  241          liftSCM $ SC.scmRecursor (dtName dt) s
  242 
  243     Un.App f arg ->
  244       -- Only call typeInferCompleteUTerm on the function arguments, to
  245       -- avoid adding too many 'atPos' and 'withErrorUTerm' annotations.
  246       do f' <- typeInferCompleteTerm f
  247          arg' <- typeInferCompleteUTerm arg
  248          liftSCM $ SC.scmApply f' arg'
  249 
  250     Un.Lambda _ [] t ->
  251       typeInferCompleteUTerm t
  252     Un.Lambda p ((Un.termVarLocalName -> x, tp) : ctx) t ->
  253       do tp_trm <- typeInferCompleteUTerm tp
  254          _ <- liftSCM $ SC.scmEnsureSortType tp_trm
  255          vn <- liftSCM $ SC.scmFreshVarName x
  256          body <- withVar x vn tp_trm $
  257            typeInferCompleteUTerm $ Un.Lambda p ctx t
  258          liftSCM $ SC.scmLambda vn tp_trm body
  259 
  260     Un.Pi _ [] t ->
  261       typeInferCompleteUTerm t
  262     Un.Pi p ((Un.termVarLocalName -> x, tp) : ctx) t ->
  263       do tp' <- typeInferCompleteUTerm tp
  264          vn <- liftSCM $ SC.scmFreshVarName x
  265          body <- withVar x vn tp' $
  266            typeInferCompleteUTerm $ Un.Pi p ctx t
  267          liftSCM $ SC.scmPi vn tp' body
  268 
  269     Un.RecordValue _ elems ->
  270       do typed_elems <-
  271            mapM (\(PosPair _ fld, t) -> (fld,) <$> typeInferCompleteUTerm t) elems
  272          liftSCM $ SC.scmRecordValue typed_elems
  273     Un.RecordType _ elems ->
  274       do typed_elems <-
  275            mapM (\(PosPair _ fld, t) -> (fld,) <$> typeInferCompleteUTerm t) elems
  276          liftSCM $ SC.scmRecordType typed_elems
  277     Un.RecordProj t prj ->
  278       do t' <- typeInferCompleteUTerm t
  279          liftSCM $ SC.scmRecordSelect t' prj
  280 
  281     Un.UnitValue _ ->
  282       liftSCM $ SC.scmUnitValue
  283     Un.UnitType _ ->
  284       liftSCM $ SC.scmUnitType
  285 
  286     Un.PairValue _ t1 t2 ->
  287       do t1' <- typeInferCompleteUTerm t1
  288          t2' <- typeInferCompleteUTerm t2
  289          liftSCM $ SC.scmPairValue t1' t2'
  290     Un.PairType _ t1 t2 ->
  291       do t1' <- typeInferCompleteUTerm t1
  292          t2' <- typeInferCompleteUTerm t2
  293          liftSCM $ SC.scmPairType t1' t2'
  294     Un.PairLeft t ->
  295       do t' <- typeInferCompleteUTerm t
  296          liftSCM $ SC.scmPairLeft t'
  297     Un.PairRight t ->
  298       do t' <- typeInferCompleteUTerm t
  299          liftSCM $ SC.scmPairRight t'
  300 
  301     Un.TypeConstraint t _ tp ->
  302       do t' <- typeInferCompleteUTerm t
  303          tp' <- typeInferCompleteUTerm tp
  304          liftSCM $ SC.scmAscribe t' tp'
  305 
  306     Un.NatLit _ i ->
  307       liftSCM $ SC.scmNat i
  308     Un.StringLit _ str ->
  309       liftSCM $ SC.scmString str
  310     Un.VecLit _ ts ->
  311       do typed_ts <- mapM typeInferCompleteUTerm ts
  312          typed_tp <-
  313            case typed_ts of
  314              (t1:_) -> liftSCM $ SC.scmTypeOf t1
  315              [] -> throwTCError $ EmptyVectorLit
  316          liftSCM $ SC.scmVector typed_tp typed_ts
  317     Un.BVLit _ bits ->
  318       do tp <- liftSCM $ SC.scmGlobalDef "Prelude.Bool"
  319          let bit b = SC.scmGlobalDef (if b then "Prelude.True" else "Prelude.False")
  320          bit_tms <- liftSCM $ traverse bit bits
  321          liftSCM $ SC.scmVector tp bit_tms
  322 
  323     Un.BadTerm _ ->
  324       -- Should be unreachable, since BadTerms represent parse errors, that should
  325       -- already have been signaled before type inference
  326       panic "typeInferCompleteTerm" ["Type inference encountered a BadTerm"]

  333 -- | Type-check a list of declarations and insert them into the current module
  334 processDecls :: [Un.Decl] -> TCM ()
  335 processDecls [] = return ()
  336 
  337 processDecls (Un.InjectCodeDecl ns txt : rest) =
  338   do mnm <- getModuleName
  339      sc <- askSharedContext
  340      liftIO $ scInjectCode sc mnm ns txt
  341      processDecls rest
  342 
  343 processDecls (Un.TypedDef nm params rty body : rest) =
  344   processDecls (Un.TypeDecl NoQualifier nm (Un.Pi (pos nm) params rty) :
  345                 Un.TermDef nm (map fst params) body : rest)
  346 
  347 processDecls (Un.TypeDecl NoQualifier (PosPair p nm) tp :
  348               Un.TermDef (PosPair _ ((== nm) -> True)) vars body : rest) =
  349   -- Type-checking for definitions
  350   (atPos p $
  351    do
  352      sc <- askSharedContext
  353      -- Step 1: type-check the type annotation, and make sure it is a type
  354      typed_tp <- typeInferCompleteUTerm tp
  355      void $ liftSCM $ SC.scmEnsureSortType typed_tp
  356      let def_tp = typed_tp
  357      def_tp_whnf <- liftIO $ scWhnf sc def_tp
  358 
  359      -- Step 2: assign types to the bound variables of the definition, by
  360      -- peeling off the pi-abstraction variables in the type annotation. Any
  361      -- remaining body of the pi-type is the required type for the def body.
  362      matchResult <-
  363        runMaybeT $ matchPiWithNames (map Un.termVarLocalName vars) def_tp_whnf
  364      (ctx, req_body_tp) <-
  365        case matchResult of
  366          Just x -> return x
  367          Nothing ->
  368              throwTCError $
  369              DeclError nm ("More variables " ++ show (map Un.termVarLocalName vars) ++
  370                            " than length of function type:\n" ++
  371                            ppTermPureDefaults typed_tp)
  372 
  373      -- Step 3: type-check the body of the definition in the context of its
  374      -- variables, and build a function that takes in those variables
  375      def_tm <-
  376        withCtx ctx $
  377        do body' <- typeInferCompleteUTerm body
  378           body_ty <- liftSCM $ SC.scmTypeOf body'
  379           ok <- liftIO $ scSubtype sc body_ty req_body_tp
  380           unless ok $ throwTCError (TermError (SC.ApplyNotSubtype req_body_tp body')) -- FIXME: Wrong error
  381           let vts = map (\(_, v, t) -> (v, t)) ctx
  382           result <- liftIO $ scLambdaList sc vts body'
  383           liftSCM $ SC.scmAscribe result def_tp
  384 
  385      -- Step 4: add the definition to the current module
  386      mnm <- getModuleName
  387      let nmi = ModuleIdentifier (mkIdent mnm nm)
  388      void $ liftIO $ scDefineConstant sc nmi def_tm) >>
  389   processDecls rest
  390 
  391 processDecls (Un.TypeDecl NoQualifier (PosPair p nm) _ : _) =
  392   atPos p $ throwTCError $ DeclError nm "Definition without defining equation"
  393 
  394 processDecls (Un.TypeDecl _ (PosPair p nm) _ :
  395               Un.TermDef (PosPair _ ((== nm) -> True)) _ _ : _) =
  396   atPos p $ throwTCError $ DeclError nm "Primitive or axiom with definition"
  397 
  398 processDecls (Un.TypeDecl q (PosPair p nm) tp : rest) =
  399   (atPos p $
  400    do typed_tp <- typeInferCompleteUTerm tp
  401       void $ liftSCM $ SC.scmEnsureSortType typed_tp
  402       mnm <- getModuleName
  403       let ident = mkIdent mnm nm
  404       let def_tp = typed_tp
  405       sc <- askSharedContext
  406       liftIO $ scDeclarePrim sc ident q def_tp) >>
  407   processDecls rest
  408 
  409 processDecls (Un.TermDef (PosPair p nm) _ _ : _) =
  410   atPos p $ throwTCError $ DeclError nm "Dangling definition without a type"
  411 
  412 processDecls (Un.DataDecl nm param_ctx dt_tp c_decls : rest) =
  413   do processDataDecl nm param_ctx dt_tp c_decls
  414      processDecls rest
  415 
  416 -- | Type-check a data type declaration and insert it into the current module.
  417 processDataDecl :: PosPair Text -> Un.UTermCtx -> Un.UTerm -> [Un.CtorDecl] -> TCM ()
  418 processDataDecl (PosPair p nm) param_ctx dt_tp c_decls =
  419   let param_ctx' = map (\(x, t) -> (Un.termVarLocalName x, t)) param_ctx in
  420   atPos p $
  421   -- Step 1: type-check the parameters
  422   typeInferCompleteInCtx param_ctx' $ \params -> do
  423   let dtParams = map (\(_, v, t, _) -> (v, t)) params
  424   let param_sort = maxSort (map (\(_, _, _, s) -> s) params)
  425   let err :: String -> TCM a
  426       err msg = throwTCError $ DeclError nm msg
  427 
  428   -- Step 2: type-check the type given for d, and make sure it is of the form
  429   -- (i1:ix1) -> ... -> (in:ixn) -> Type s for some sort s. Then form the full
  430   -- type of d as (p1:param1) -> ... -> (i1:ix1) -> ... -> Type s
  431   (dt_ixs, dtSort) <-
  432     case Un.asPiList dt_tp of
  433       (ixs, Un.Sort _ s h) | h == noFlags ->
  434         return (ixs, s) -- NB, don't allow `isort`, etc.
  435       _ -> err "Wrong form for type of datatype"
  436   let dt_ixs' = map (\(x, t) -> (Un.termVarLocalName x, t)) dt_ixs
  437   dt_ixs_typed <- typeInferCompleteCtx dt_ixs'
  438   let dtIndices = map (\(_, v, t, _) -> (v, t)) dt_ixs_typed
  439       ixs_max_sort = maxSort (map (\(_, _, _, s) -> s) dt_ixs_typed)
  440 
  441   -- Step 3: do the necessary universe inclusion checking for any predicative
  442   -- (non-Prop) inductive type, which includes:
  443   --
  444   -- 1. All ix types must be of sort dtSort; AND
  445   -- 2. All param types must be of sort dtSort+1
  446   if dtSort /= propSort && param_sort > sortOf dtSort then
  447     err ("Universe level of parameters should be no greater" ++
  448          " than that of the datatype")
  449     else return ()
  450   if dtSort /= propSort && ixs_max_sort > dtSort then
  451     err ("Universe level of indices should be strictly contained" ++
  452          " in that of the datatype")
  453     else return ()
  454 
  455   -- Step 4: Add d as an empty datatype, so we can typecheck the constructors
  456   mnm <- getModuleName
  457   let dtIdent = mkIdent mnm nm
  458   pn <- liftSCM $ SC.scmBeginDataType dtIdent dtParams dtIndices dtSort
  459 
  460   -- Step 5: typecheck the constructors, and build Ctors for them
  461   typed_ctors <-
  462     mapM (\(Un.Ctor (PosPair p' c) ctx body) ->
  463            (c,) <$> typeInferCompleteUTerm (Un.Pi p' ctx body)) c_decls
  464   sc <- askSharedContext
  465   ctors <-
  466     forM typed_ctors $ \(c, typed_tp) ->
  467     -- Check that the universe level of the type of each constructor
  468     do case termSortOrType typed_tp of
  469            Left ctor_sort
  470              | dtSort /= propSort && ctor_sort > dtSort ->
  471                err ("Universe level of constructors should be strictly" ++
  472                     " contained in that of the datatype")
  473            Left _ ->
  474                return ()
  475            Right ty ->
  476                panic "processDecls" [
  477                    "Type of the type of constructor is not a sort!",
  478                    "Constructor type: " <> Text.pack (ppTermPureDefaults typed_tp),
  479                    "Type of that type: " <> Text.pack (ppTermPureDefaults ty)
  480                ]
  481        let tp = typed_tp
  482        let result = mkCtorArgStruct pn dtParams dtIndices tp
  483        case result of
  484          Just arg_struct ->
  485            liftIO $ scBuildCtor sc pn (mkIdent mnm c) arg_struct
  486          Nothing -> err ("Malformed type form constructor: " ++ show c)
  487 
  488   -- Step 6: complete the datatype with the given ctors
  489   liftSCM $ SC.scmCompleteDataType dtIdent ctors
  490

  492 -- | Typecheck a module and, on success, insert it into the current context
  493 tcInsertModule :: SharedContext -> Un.Module -> IO ()
  494 tcInsertModule sc (Un.Module (PosPair _ mnm) imports decls) = do
  495   -- First, insert an empty module for mnm
  496   scLoadModule sc $ emptyModule mnm
  497   -- Next, process all the imports
  498   forM_ imports $ \imp ->
  499     do let imn = val $ Un.importModName imp
  500        i_exists <- scModuleIsLoaded sc imn
  501        unless i_exists $ fail $ "Imported module not found: " ++ show imn
  502        scImportModule sc (Un.nameSatsConstraint (Un.importConstraints imp) . Text.unpack) imn mnm
  503   -- Finally, process all the decls
  504   decls_res <- runTCM (processDecls decls) sc (Just mnm)
  505   case decls_res of
  506     Left err -> fail $ ppTCError err
  507     Right _ -> return ()

  514 -- | Pattern match a nested pi-abstraction, like 'asPiList', but only match as
  515 -- far as the supplied list of variables, and use them as the new names
  516 matchPiWithNames ::
  517   [LocalName] -> Term -> MaybeT TCM ([(LocalName, VarName, Term)], Term)
  518 matchPiWithNames [] tp = pure ([], tp)
  519 matchPiWithNames (var : vars) tp =
  520   case asPi tp of
  521     Nothing -> mzero
  522     Just (x, arg_tp, body_tp) ->
  523       do vn <-
  524            if IntSet.member (vnIndex x) (freeVars body_tp)
  525            then -- dependent function type: use name from Pi type
  526              pure x
  527            else -- non-dependent function: use 'var' as base name
  528              lift $ liftSCM $ SC.scmFreshVarName var
  529          (ctx, body) <- matchPiWithNames vars body_tp
  530          pure ((var, vn, arg_tp) : ctx, body)

  532 -- | Run a type-checking computation in a typing context extended with a new
  533 -- variable with the given name and type.
  534 withVar :: LocalName -> VarName -> Term -> TCM a -> TCM a
  535 withVar x vn tp (TCM m) =
  536   TCM $
  537   local (\env -> env { tcLocals = Map.insert x (vn, tp) (tcLocals env) }) m
  538 
  539 -- | Run a type-checking computation in a typing context extended by a list of
  540 -- variables and their types. See 'withVar'.
  541 withCtx :: [(LocalName, VarName, Term)] -> TCM a -> TCM a
  542 withCtx = flip (foldr (\(x, vn, tp) -> withVar x vn tp))
  543 
  544 -- | Perform type inference on a context, i.e., a list of variable names and
  545 -- their associated types. This will give us 'Term's for each type, as
  546 -- well as their 'Sort's, since the type of any type is a 'Sort'.
  547 typeInferCompleteCtx ::
  548   [(LocalName, Un.UTerm)] -> TCM [(LocalName, VarName, Term, Sort)]
  549 typeInferCompleteCtx [] = pure []
  550 typeInferCompleteCtx ((x, tp) : ctx) =
  551   do tp' <- typeInferCompleteUTerm tp
  552      s <- liftSCM $ SC.scmEnsureSortType tp'
  553      vn <- liftSCM $ SC.scmFreshVarName x
  554      ctx' <- withVar x vn tp' (typeInferCompleteCtx ctx)
  555      pure ((x, vn, tp', s) : ctx')
  556 
  557 -- | Perform type inference on a context via 'typeInferCompleteCtx', and then
  558 -- run a computation in that context via 'withCtx', also passing in that context
  559 -- to the computation
  560 typeInferCompleteInCtx ::
  561   [(LocalName, Un.UTerm)] ->
  562   ([(LocalName, VarName, Term, Sort)] -> TCM a) -> TCM a
  563 typeInferCompleteInCtx ctx f =
  564   do typed_ctx <- typeInferCompleteCtx ctx
  565      withCtx (map (\(x, v, t, _) -> (x, v, t)) typed_ctx) (f typed_ctx)
