   28 -- | A @MapCache m k a@ is a mutable reference to a 'Map' with keys of
   29 -- type @k@ and values of type @a@, for use in monad @m@.
   30 newtype MapCache m k a = MapCache (T m (Map k a))
   31 
   32 -- | Create a new empty 'MapCache'.
   33 newMapCache :: (C m, Ord k) => m (MapCache m k a)
   34 newMapCache = MapCache <$> new Map.empty
   35 
   36 -- | Memoize a computation of type @k -> m a@ using a 'MapCache'.
   37 -- If the cache already contains an entry for the given key, return
   38 -- the value immediately without running the given monadic action.
   39 -- Otherwise run the monadic action, store the result in the cache,
   40 -- and also return the newly-computed value.
   41 useMapCache :: (C m, Ord k) => MapCache m k a -> k -> m a -> m a
   42 useMapCache (MapCache ref) k action =
   43   do m <- Data.Ref.read ref
   44      case Map.lookup k m of
   45        Just x -> pure x
   46        Nothing ->
   47          do x <- action
   48             modify ref (Map.insert k x)
   49             pure x
   50 
   51 -- | An @IntCache m k a@ is a mutable reference to an 'IntMap' with
   52 -- keys of type 'Int' and values of type @a@, for use in monad @m@.
   53 -- @IntCache m a@ is a more efficient alternative to @MapCache m Int
   54 -- a@.
   55 newtype IntCache m a = IntCache (T m (IntMap a))
   56 
   57 -- | Create a new empty 'IntCache'.
   58 newIntCache :: (C m) => m (IntCache m a)
   59 newIntCache = IntCache <$> new IntMap.empty
   60 
   61 -- | Memoize a computation of type @Int -> m a@ using an 'IntCache'.
   62 -- If the cache already contains an entry for the given key, return
   63 -- the value immediately without running the given monadic action.
   64 -- Otherwise run the monadic action, store the result in the cache,
   65 -- and also return the newly-computed value.
   66 useIntCache :: C m => IntCache m a -> Int -> m a -> m a
   67 useIntCache (IntCache ref) k action =
   68   do m <- Data.Ref.read ref
   69      case IntMap.lookup k m of
   70        Just x -> pure x
   71        Nothing ->
   72          do x <- action
   73             modify ref (IntMap.insert k x)
   74             pure x
