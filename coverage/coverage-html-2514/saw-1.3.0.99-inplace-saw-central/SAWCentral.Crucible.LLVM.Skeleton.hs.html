  101 parseType :: LLVM.Type -> IO TypeSkeleton
  102 parseType (LLVM.PtrTo t) = pure $ TypeSkeleton t True [SizeGuess 1 True "default guess of size 1"]
  103 -- It is unclear how to combine opaque pointers with type skeletons due to the
  104 -- lack of a pointee type. For now, we simply fail if we encounter one
  105 -- (see #1877).
  106 parseType LLVM.PtrOpaque =
  107   panic "SAWCentral.Crucible.LLVM.Skeleton.parseType"
  108         [ "Skeleton generation does not support opaque pointers"
  109         , "Please report this at: https://github.com/GaloisInc/saw-script/issues/1877"
  110         ]
  111 parseType (LLVM.Array i t) = pure $ TypeSkeleton t True
  112   [ SizeGuess (fromIntegral i) True $ "default guess of size " <> Text.pack (show i)
  113   ]
  114 parseType t = pure $ TypeSkeleton t False []
  115 
  116 parseGlobal :: Map String Int -> LLVM.Global -> IO GlobalSkeleton
  117 parseGlobal ls LLVM.Global
  118   { LLVM.globalSym = LLVM.Symbol s
  119   , LLVM.globalType = t
  120   , LLVM.globalValue = v
  121   , LLVM.globalAttrs = LLVM.GlobalAttrs { LLVM.gaConstant = c }
  122   } = do
  123   ty <- parseType t
  124   pure GlobalSkeleton
  125     { _globSkelName = Text.pack s
  126     , _globSkelLoc = flip Location Nothing <$> Map.lookup s ls
  127     , _globSkelType = ty
  128     , _globSkelMutable = not c
  129     , _globSkelInitialized = isJust v
  130     }
  131 
  132 parseArg :: LLVM.Typed LLVM.Ident -> (Maybe String, Maybe Location) -> IO ArgSkeleton
  133 parseArg LLVM.Typed { LLVM.typedType = t } (nm, loc) = do
  134   ty <- parseType t
  135   pure ArgSkeleton
  136     { _argSkelName = Text.pack <$> nm
  137     , _argSkelLoc = loc
  138     , _argSkelType = ty
  139     }

  141 stmtCalls :: [LLVM.Stmt] -> Set Text
  142 stmtCalls [] = Set.empty
  143 stmtCalls (LLVM.Result _ (LLVM.Call _ _ (LLVM.ValSymbol (LLVM.Symbol s)) _) _:stmts) =
  144   Set.insert (Text.pack s) $ stmtCalls stmts
  145 stmtCalls (LLVM.Effect (LLVM.Call _ _ (LLVM.ValSymbol (LLVM.Symbol s)) _) _:stmts) =
  146   Set.insert (Text.pack s) $ stmtCalls stmts
  147 stmtCalls (_:stmts) = stmtCalls stmts
  148 
  149 stmtDebugDeclares :: [LLVM.Stmt] -> Map Int Location
  150 stmtDebugDeclares [] = Map.empty
  151 stmtDebugDeclares
  152   (LLVM.Result _
  153     (LLVM.Call _ _
  154       (LLVM.ValSymbol (LLVM.Symbol s))
  155       [ _
  156       , LLVM.Typed
  157         { LLVM.typedValue =
  158           LLVM.ValMd (LLVM.ValMdDebugInfo (LLVM.DebugInfoLocalVariable LLVM.DILocalVariable { LLVM.dilvArg = a }))
  159         }
  160       , _
  161       ]) md:stmts)
  162   | s == "llvm.dbg.declare" || s == "llvm.dbg.value"
  163   , Just (LLVM.ValMdLoc LLVM.DebugLoc { LLVM.dlLine = line, LLVM.dlCol = col }) <- lookup "dbg" md
  164   = Map.insert (fromIntegral a) (Location (fromIntegral line) . Just $ fromIntegral col) $ stmtDebugDeclares stmts
  165 stmtDebugDeclares
  166   (LLVM.Effect
  167     (LLVM.Call _ _
  168       (LLVM.ValSymbol (LLVM.Symbol s))
  169       [ _
  170       , LLVM.Typed
  171         { LLVM.typedValue =
  172           LLVM.ValMd (LLVM.ValMdDebugInfo (LLVM.DebugInfoLocalVariable LLVM.DILocalVariable { LLVM.dilvArg = a }))
  173         }
  174       , _
  175       ]) md:stmts)
  176   | s == "llvm.dbg.declare" || s == "llvm.dbg.value"
  177   , Just (LLVM.ValMdLoc LLVM.DebugLoc { LLVM.dlLine = line, LLVM.dlCol = col }) <- lookup "dbg" md
  178   = Map.insert (fromIntegral a) (Location (fromIntegral line) . Just $ fromIntegral col) $ stmtDebugDeclares stmts
  179 stmtDebugDeclares (_:stmts) = stmtDebugDeclares stmts
  180 
  181 defineName :: LLVM.Define -> Text
  182 defineName LLVM.Define { LLVM.defName = LLVM.Symbol s } = Text.pack s
  183 
  184 parseDefine :: Map String Int -> LLVM.Module -> LLVM.Define -> IO FunctionSkeleton
  185 parseDefine _ _ LLVM.Define { LLVM.defVarArgs = True } =
  186   fail "Skeleton generation does not support varargs"
  187 parseDefine ls m d@LLVM.Define
  188   { LLVM.defName = LLVM.Symbol s
  189   , LLVM.defArgs = args
  190   , LLVM.defBody = body
  191   , LLVM.defRetType = ret
  192   } = do
  193   let stmts = mconcat $ LLVM.bbStmts <$> body
  194   let argNames = LLVM.debugInfoArgNames m d
  195   let debugDeclares = stmtDebugDeclares stmts
  196   argSkels <- zipWithM parseArg args $ (flip IntMap.lookup argNames &&& flip Map.lookup debugDeclares) <$> [1, 2..]
  197   retTy <- parseType ret
  198   pure FunctionSkeleton
  199     { _funSkelName = Text.pack s
  200     , _funSkelLoc = flip Location Nothing <$> Map.lookup s ls
  201     , _funSkelArgs = argSkels
  202     , _funSkelRet = retTy
  203     , _funSkelCalls = Set.intersection
  204       (Set.fromList $ defineName <$> LLVM.modDefines m)
  205       (stmtCalls stmts)
  206     }
  207 
  208 moduleSkeleton :: LLVM.Module -> IO ModuleSkeleton
  209 moduleSkeleton ast = do
  210   globs <- mapM (parseGlobal $ LLVM.debugInfoGlobalLines ast) $ LLVM.modGlobals ast
  211   funs <- mapM (parseDefine (LLVM.debugInfoDefineLines ast) ast) $ LLVM.modDefines ast
  212   pure $ ModuleSkeleton
  213     { _modSkelGlobals = Map.fromList $ (\g -> (g ^. globSkelName, g)) <$> globs
  214     , _modSkelFunctions = Map.fromList $ (\f -> (f ^. funSkelName, f)) <$> funs
  215     }
