   45 data YosysTheorem = YosysTheorem
   46   { _theoremURI :: URI.URI -- ^ URI identifying overridden module
   47   , _theoremInputCryptolType :: C.Type -- ^ cryptol type of r
   48   , _theoremOutputCryptolType :: C.Type -- ^ cryptol type of (module r)
   49   , _theoremInputType :: SC.Term -- ^ type of r
   50   , _theoremOutputType :: SC.Term -- ^ type of (module r)
   51   , _theoremModule :: SC.Term -- ^ {{ \r -> module r }}
   52   , _theoremPrecond :: Maybe SC.Term -- ^ {{ \r -> precond r }}
   53   , _theoremBody :: SC.Term -- ^ {{ \r -> other r }}
   54   } 
   55 makeLenses ''YosysTheorem

   57 -- | Construct a SAWCore proposition for the given theorem.
   58 -- In pseudo-Cryptol, this looks like {{ \r -> precond r ==> (module r == body r) }}
   59 theoremProp ::
   60   (MonadIO m, MonadThrow m) =>
   61   SC.SharedContext ->
   62   YosysTheorem ->
   63   m SC.TypedTerm
   64 theoremProp sc thm = do
   65   ec <- liftIO $ SC.scFreshEC sc "r" $ thm ^. theoremInputType
   66   r <- liftIO $ SC.scExtCns sc ec
   67   modr <- liftIO $ SC.scApply sc (thm ^. theoremModule) r
   68   bodyr <- liftIO $ SC.scApply sc (thm ^. theoremBody) r
   69   equality <- liftIO $ eqBvRecords sc (thm ^. theoremOutputCryptolType) modr bodyr
   70   res <- case thm ^. theoremPrecond of
   71     Nothing -> pure equality
   72     Just pc -> do
   73       pcr <- liftIO $ SC.scApply sc pc r
   74       liftIO $ SC.scImplies sc pcr equality
   75   func <- liftIO $ SC.scAbstractExts sc [ec] res
   76   let cty = C.tFun (thm ^. theoremInputCryptolType) C.tBit
   77   SC.TypedTerm (SC.TypedTermSchema $ C.tMono cty)
   78     <$> validateTerm sc
   79     ("constructing a proposition while verifying " <> URI.render (thm ^. theoremURI))
   80     func

   82 -- | Construct a SAWCore proposition for the given theorem.
   83 -- In pseudo-Cryptol, this looks like {{ \r -> if precond r then body r else module r }}
   84 theoremReplacement ::
   85   (MonadIO m, MonadThrow m) =>
   86   SC.SharedContext ->
   87   YosysTheorem ->
   88   m SC.Term
   89 theoremReplacement sc thm = do
   90   ec <- liftIO $ SC.scFreshEC sc "r" $ thm ^. theoremInputType
   91   r <- liftIO $ SC.scExtCns sc ec
   92   body <- case thm ^. theoremPrecond of
   93     Nothing -> liftIO $ SC.scApply sc (thm ^. theoremBody) r
   94     Just pc -> do
   95       precond <- liftIO $ SC.scApply sc pc r
   96       thenCase <- liftIO $ SC.scApply sc (thm ^. theoremBody) r
   97       elseCase <- liftIO $ SC.scApply sc (thm ^. theoremModule) r
   98       liftIO $ SC.scIte sc (thm ^. theoremOutputType) precond thenCase elseCase
   99   ft <- liftIO $ SC.scAbstractExts sc [ec] body
  100   validateTerm sc
  101     ("constructing an override replacement for " <> URI.render (thm ^. theoremURI))
  102     ft

  104 -- | Given a SAWCore term corresponding to an HDL module, a specification, and a precondition:
  105 -- Construct a theorem summarizing the relationship between the module and the specification.
  106 buildTheorem ::
  107   (MonadIO m, MonadThrow m) =>
  108   SC.SharedContext ->
  109   SC.TypedTerm ->
  110   SC.Term ->
  111   Maybe SC.TypedTerm ->
  112   SC.TypedTerm ->
  113   m YosysTheorem
  114 buildTheorem sc ymod newmod precond body = do
  115   cty <- case SC.ttType ymod of
  116     SC.TypedTermSchema (C.Forall [] [] cty) -> pure cty
  117     _ -> throw YosysErrorInvalidOverrideTarget
  118   (cinpTy, coutTy) <- case cty of
  119     C.TCon (C.TC C.TCFun) [ci, co] -> pure (ci, co)
  120     _ -> throw YosysErrorInvalidOverrideTarget
  121   inpTy <- liftIO $ CSC.importType sc CSC.emptyEnv cinpTy
  122   outTy <- liftIO $ CSC.importType sc CSC.emptyEnv coutTy
  123   nmi <- case SC.ttTerm ymod of
  124     (R.asConstant -> Just (SC.Name _ nmi)) -> pure nmi
  125     _ -> throw YosysErrorInvalidOverrideTarget
  126   uri <- case nmi of
  127     SC.ImportedName uri _ -> pure uri
  128     _ -> throw YosysErrorInvalidOverrideTarget
  129   pure YosysTheorem
  130     { _theoremURI = uri
  131     , _theoremInputCryptolType = cinpTy
  132     , _theoremOutputCryptolType = coutTy
  133     , _theoremInputType = inpTy
  134     , _theoremOutputType = outTy
  135     , _theoremModule = newmod
  136     , _theoremPrecond = SC.ttTerm <$> precond
  137     , _theoremBody = SC.ttTerm body
  138     }

  140 -- | Applying a theorem thm as an "override" in a Yosys-derived term t proceeds as follows:
  141 --  1) unfold all names except thm.theoremURI in t
  142 --  2) traverse t, looking for constants named thm.theoremURI
  143 --  4) replace the constant term with either thm.theoremBody, or
  144 --     {{ \r -> if thm.theoremPrecond r then thm.theoremBody r else thm.theoremURI r }}
  145 --     in the presence of a precondition
  146 applyOverride ::
  147   forall m.
  148   (MonadIO m, MonadThrow m) =>
  149   SC.SharedContext ->
  150   YosysTheorem ->
  151   SC.Term ->
  152   m SC.Term
  153 applyOverride sc thm t = do
  154   tidx <- liftIO (SC.scResolveNameByURI sc $ thm ^. theoremURI) >>= \case
  155     Nothing -> throw . YosysErrorOverrideNameNotFound . URI.render $ thm ^. theoremURI
  156     Just i -> pure i
  157   unfolded <- liftIO $ SC.scUnfoldConstantSet sc False (Set.singleton tidx) t
  158   cache <- liftIO SC.newCache
  159   let
  160     go :: SC.Term -> IO SC.Term
  161     go s@(SC.Unshared tf) = case tf of
  162       SC.Constant (SC.Name idx _)
  163         | idx == tidx -> theoremReplacement sc thm
  164         | otherwise -> pure s
  165       _ -> SC.Unshared <$> traverse go tf
  166     go s@SC.STApp { SC.stAppIndex = aidx, SC.stAppTermF = tf } = SC.useCache cache aidx $ case tf of
  167       SC.Constant (SC.Name idx _)
  168         | idx == tidx -> theoremReplacement sc thm
  169         | otherwise -> pure s
  170       _ -> SC.scTermF sc =<< traverse go tf
  171   ft <- liftIO $ go unfolded
  172   validateTerm sc
  173     ("applying an override for " <> URI.render (thm ^. theoremURI))
  174     ft
