   32 -- | Parse a Cryptol module into a list of declarations
   33 --   Yields an Interaction so that we can talk to the user about what went wrong
   34 getDeclsCryptol :: FilePath -> IO (Interaction (Maybe [Decl]))
   35 getDeclsCryptol path = do
   36    let evalOpts = EvalOpts quietLogger defaultPPOpts
   37    modEnv <- M.initialModuleEnv
   38    let minp = M.ModuleInput True (pure evalOpts) BS.readFile modEnv
   39    (result, warnings) <-
   40      SMT.withSolver (return ()) (meSolverConfig modEnv) $ \s ->
   41      M.loadModuleByPath path (minp s)
   42    return $ do
   43       forM_ warnings $ liftF . flip Warning () . pretty
   44       case result of
   45          Left err -> liftF $ Failure True (pretty err) Nothing
   46          Right ((_, top), _) ->
   47             case top of
   48               AST.TCTopSignature {} ->
   49                  liftF $ Failure True "Expected a module but found an intreface" Nothing
   50               AST.TCTopModule AST.Module { mDecls } ->
   51                 let stdDecls = catMaybes . for (concatMap flattenDeclGroup mDecls) $
   52                       \(AST.Decl{dName, dSignature, dDefinition}) -> do
   53                          funcName <- sourceName dName
   54                          argNames <- mapM sourceName =<< tupleLambdaBindings =<< declDefExpr dDefinition
   55                          (argTypes, retType) <- tupleFunctionType (AST.sType dSignature)
   56                          return $ Decl funcName retType (zipWith Arg argNames argTypes)
   57                 in return $ Just (stdDecls :: [Decl])

   59 -- All this is just sifting through the Cryptol typechecker's AST to get the information we want:
   60 
   61 -- Things will break if Cryptol's internals shift the way they desugar tuple bindings in declarations:
   62 -- Currently, a declaration like @foo (a,b,c) = ...@ turns into something
   63 -- like @foo = \x -> ... where a = x.0.; b = x.1; c = x.2@
   64 -- We rely on this behavior to parse out the information we need.
   65 
   66 -- | We don't care about recursive bindings, so we flatten them out
   67 flattenDeclGroup :: AST.DeclGroup -> [AST.Decl]
   68 flattenDeclGroup (AST.Recursive decls)   = decls
   69 flattenDeclGroup (AST.NonRecursive decl) = [decl]
   70 
   71 -- | If the expression is a tuple projection, get the name of the tuple and the index projected
   72 tupleSelInfo :: AST.Expr -> Maybe (AST.Name, Int)
   73 tupleSelInfo (AST.ESel (AST.EVar name) (AST.TupleSel i _)) = return (name, i)
   74 tupleSelInfo _                                             = Nothing
   75 
   76 -- | If an expression is a where binding, get all its local declarations
   77 whereBindings :: AST.Expr -> Maybe [AST.Decl]
   78 whereBindings (AST.EWhere _ bindings) = return $ concatMap flattenDeclGroup bindings
   79 whereBindings _                       = Nothing
   80 
   81 -- | Find the expression inside a definition
   82 --   We can't handle primitives currently
   83 declDefExpr :: AST.DeclDef -> Maybe AST.Expr
   84 declDefExpr = \case
   85    AST.DPrim            -> Nothing
   86    AST.DForeign _ mexpr -> mexpr
   87    AST.DExpr expr       -> Just expr
   88 
   89 -- | If a lambda is of the form @\(a,b,...,z) -> ...)@ then give the list of names bound in the tuple
   90 tupleLambdaBindings :: AST.Expr -> Maybe [AST.Name]
   91 tupleLambdaBindings (AST.EAbs tupleName _ whereClause) = do
   92    bindings <- whereBindings whereClause
   93    return . map snd . sortBy (comparing fst) . catMaybes . for bindings $
   94       \AST.Decl{dDefinition, dName} -> do
   95          (from, index) <- tupleSelInfo =<< declDefExpr dDefinition
   96          guard (from == tupleName)
   97          return (index, dName)
   98 tupleLambdaBindings (AST.ETAbs _ expr)     = tupleLambdaBindings expr   --   \
   99 tupleLambdaBindings (AST.EProofAbs _ expr) = tupleLambdaBindings expr   --    >- drill down past casts & type abstractions
  100 tupleLambdaBindings _ = Nothing
  101 
  102 -- | If the type given is of the form @(a,b,...,y) -> z@ then give the pair of arguments and result
  103 tupleFunctionType :: AST.Type -> Maybe ([AST.Type], AST.Type)
  104 tupleFunctionType (AST.TCon (AST.TC AST.TCFun) [AST.TCon (AST.TC (AST.TCTuple _)) inputs, output]) = return (inputs, output)
  105 tupleFunctionType _                                                                                = Nothing
  106 
  107 -- | Find the name from the source if one exists
  108 sourceName :: Name -> Maybe Text
  109 sourceName nm = Just (identText (nameIdent nm))
