   45 -- | Find all of the flip-flop cells in a network graph.
   46 findDffs ::
   47   Netgraph ->
   48   Map Text (Cell [Bitrep])
   49 findDffs ng =
   50   Map.fromList
   51   . filter (\(_, c) -> c ^. cellType `elem` [CellTypeDff, CellTypeFf])
   52   . fmap (\v -> let (n, nm, _) = ng ^. netgraphNodeFromVertex $ v in (cellIdentifier nm, n))
   53   . Graph.vertices
   54   $ ng ^. netgraphGraph

   56 -- ^ A SAWCore translation of an HDL module alongside some type information that is useful to keep around.
   57 data YosysSequential = YosysSequential
   58   { _yosysSequentialTerm :: SC.TypedTerm -- ^ The "raw" SAWCore term derived from the module, which includes a __state__ field in the input and output.
   59   , _yosysSequentialStateFields :: Map Text (SC.Term, C.Type) -- ^ A mapping from each state field name to a SAWCore and Cryptol type.
   60   , _yosysSequentialInputFields :: Map Text (SC.Term, C.Type) -- ^ A mapping from each input to a SAWCore and Cryptol type.
   61   , _yosysSequentialOutputFields :: Map Text (SC.Term, C.Type) -- ^ A mapping from each output to a SAWCore and Cryptol type.
   62   , _yosysSequentialInputWidths :: Map Text Natural -- ^ A mapping from each input to a width.
   63   , _yosysSequentialOutputWidths :: Map Text Natural -- ^ A mapping from each output to a width.
   64   , _yosysSequentialStateWidths :: Map Text Natural -- ^ A mapping from each state field to a width.
   65   }
   66 makeLenses ''YosysSequential

   68 -- | Add a record-typed field named __states__ to the given mapping of field names to types.
   69 insertStateField ::
   70   MonadIO m =>
   71   SC.SharedContext ->
   72   Map Text (SC.Term, C.Type) {- ^ The field types of "__states__" -} ->
   73   Map Text (SC.Term, C.Type) {- ^ The mapping to update -} ->
   74   m (Map Text (SC.Term, C.Type))
   75 insertStateField sc stateFields fields = do
   76   stateRecordType <- fieldsToType sc stateFields
   77   stateRecordCryptolType <- fieldsToCryptolType stateFields
   78   pure $ Map.insert "__state__" (stateRecordType, stateRecordCryptolType) fields

   80 -- | Translate a stateful HDL module into SAWCore
   81 convertModuleInline ::
   82   MonadIO m =>
   83   SC.SharedContext ->
   84   Module ->
   85   m YosysSequential
   86 convertModuleInline sc m = do
   87   let ng = moduleNetgraph m
   88 
   89   -- construct SAWCore and Cryptol types
   90   let dffs = findDffs ng
   91 
   92   stateWidths <- forM dffs $ \c ->
   93     case Map.lookup "Q" $ c ^. cellConnections of
   94       Nothing -> panic "convertModuleInline" ["Missing expected output name for $dff cell"]
   95       Just b -> pure . fromIntegral $ length b
   96 
   97   stateFields <- forM stateWidths $ \w -> do
   98     t <- liftIO $ SC.scBitvector sc w
   99     let cty = C.tWord $ C.tNum w
  100     pure (t, cty)
  101 
  102   let inputPorts = moduleInputPorts m
  103   let outputPorts = moduleOutputPorts m
  104   inputFields <- forM inputPorts $ \inp -> do
  105     ty <- liftIO . SC.scBitvector sc . fromIntegral $ length inp
  106     let cty = C.tWord . C.tNum $ length inp
  107     pure (ty, cty)
  108   outputFields <- forM outputPorts $ \out -> do
  109     ty <- liftIO . SC.scBitvector sc . fromIntegral $ length out
  110     let cty = C.tWord . C.tNum $ length out
  111     pure (ty, cty)
  112 
  113   domainFields <- insertStateField sc stateFields inputFields
  114   codomainFields <- insertStateField sc stateFields outputFields
  115 
  116   domainRecordType <- fieldsToType sc domainFields
  117   domainCryptolRecordType <- fieldsToCryptolType domainFields
  118   -- codomainRecordType <- fieldsToType sc codomainFields
  119   codomainCryptolRecordType <- fieldsToCryptolType codomainFields
  120 
  121   -- convert module into term
  122   domainRecordEC <- liftIO $ SC.scFreshEC sc "input" domainRecordType
  123   domainRecord <- liftIO $ SC.scExtCns sc domainRecordEC
  124 
  125   derivedInputs <- forM (Map.assocs inputPorts) $ \(nm, inp) -> do
  126     t <- liftIO $ cryptolRecordSelect sc domainFields domainRecord nm
  127     deriveTermsByIndices sc inp t
  128 
  129   preStateRecord <- liftIO $ cryptolRecordSelect sc domainFields domainRecord "__state__"
  130   derivedPreState <- forM (Map.assocs dffs) $ \(cnm, c) ->
  131     case Map.lookup "Q" $ c ^. cellConnections of
  132       Nothing -> panic "convertModuleInline" ["Missing expected output name for $dff cell"]
  133       Just b -> do
  134         t <- liftIO $ cryptolRecordSelect sc stateFields preStateRecord cnm
  135         deriveTermsByIndices sc b t
  136 
  137   zeroTerm <- liftIO $ SC.scBvConst sc 1 0
  138   oneTerm <- liftIO $ SC.scBvConst sc 1 1
  139   oneBitType <- liftIO $ SC.scBitvector sc 1
  140   xMsg <- liftIO $ SC.scString sc "Attempted to read X bit"
  141   xTerm <- liftIO $ SC.scGlobalApply sc (SC.mkIdent SC.preludeName "error") [oneBitType, xMsg]
  142   zMsg <- liftIO $ SC.scString sc "Attempted to read Z bit"
  143   zTerm <- liftIO $ SC.scGlobalApply sc (SC.mkIdent SC.preludeName "error") [oneBitType, zMsg]
  144   let inputs = Map.unions $ mconcat
  145         [ [ Map.fromList
  146             [ ( [BitrepZero], zeroTerm)
  147             , ( [BitrepOne], oneTerm )
  148             , ( [BitrepX], xTerm )
  149             , ( [BitrepZ], zTerm )
  150             ]
  151           ]
  152         , derivedInputs
  153         , derivedPreState
  154         ]
  155 
  156   terms <- netgraphToTerms sc Map.empty ng inputs
  157 
  158   postStateFields <- mapForWithKeyM dffs $ \cnm c ->
  159     case Map.lookup "D" $ c ^. cellConnections of
  160       Nothing -> panic "convertModuleInline" ["Missing expected input name for $dff cell"]
  161       Just b -> do
  162         t <- lookupPatternTerm sc (YosysBitvecConsumerCell cnm "D") b terms
  163         pure t
  164   postStateRecord <- cryptolRecord sc postStateFields
  165 
  166   outputRecord <- cryptolRecord sc . Map.insert "__state__" postStateRecord =<< mapForWithKeyM outputPorts
  167     (\onm out -> lookupPatternTerm sc (YosysBitvecConsumerOutputPort onm) out terms)
  168 
  169   -- construct result
  170   t <- liftIO $ SC.scAbstractExts sc [domainRecordEC] outputRecord
  171   -- ty <- liftIO $ SC.scFun sc domainRecordType codomainRecordType
  172   _ <- validateTerm sc "translating a sequential circuit" t
  173   let cty = C.tFun domainCryptolRecordType codomainCryptolRecordType
  174   pure YosysSequential
  175     { _yosysSequentialTerm = SC.TypedTerm (SC.TypedTermSchema $ C.tMono cty) t
  176     , _yosysSequentialStateFields = stateFields
  177     , _yosysSequentialInputFields = inputFields
  178     , _yosysSequentialOutputFields = outputFields
  179     , _yosysSequentialInputWidths = fromIntegral . length <$> inputPorts
  180     , _yosysSequentialOutputWidths = fromIntegral . length <$> outputPorts
  181     , _yosysSequentialStateWidths = stateWidths
  182     }

  184 -- | Given a SAWCore term with an explicit state, iterate the term the given number of times.
  185 -- | The resulting term has a parameter for the initial state, the resulting Cryptol types does not.
  186 composeYosysSequentialHelper ::
  187   forall m.
  188   MonadIO m =>
  189   SC.SharedContext ->
  190   YosysSequential ->
  191   Integer ->
  192   m (SC.Term, C.Type)
  193 composeYosysSequentialHelper sc s n = do
  194   let t = SC.ttTerm $ s ^. yosysSequentialTerm
  195 
  196   width <- liftIO . SC.scNat sc $ fromIntegral n
  197   extendedInputFields <- forM (s ^. yosysSequentialInputFields) $ \(ty, cty) -> do
  198     exty <- liftIO $ SC.scVecType sc width ty
  199     let excty = C.tSeq (C.tNum n) cty
  200     pure (exty, excty)
  201   extendedOutputFields <- forM (s ^. yosysSequentialOutputFields) $ \(ty, cty) -> do
  202     exty <- liftIO $ SC.scVecType sc width ty
  203     let excty = C.tSeq (C.tNum n) cty
  204     pure (exty, excty)
  205   extendedInputType <- fieldsToType sc extendedInputFields
  206   extendedInputCryptolType <- fieldsToCryptolType extendedInputFields
  207   extendedInputRecordEC <- liftIO $ SC.scFreshEC sc "input" extendedInputType
  208   extendedInputRecord <- liftIO $ SC.scExtCns sc extendedInputRecordEC
  209   extendedOutputCryptolType <- fieldsToCryptolType extendedOutputFields
  210 
  211   allInputs <- fmap Map.fromList . forM (Map.keys extendedInputFields) $ \nm -> do
  212     inp <- liftIO $ cryptolRecordSelect sc extendedInputFields extendedInputRecord nm
  213     pure (nm, inp)
  214 
  215   codomainFields <- insertStateField sc (s ^. yosysSequentialStateFields) $ s ^. yosysSequentialOutputFields
  216 
  217   let
  218     buildIntermediateInput :: Integer -> SC.Term -> m SC.Term
  219     buildIntermediateInput i st = do
  220       inps <- fmap Map.fromList . forM (Map.assocs allInputs) $ \(nm, inp) -> do
  221         case Map.lookup nm $ s ^. yosysSequentialInputFields of
  222           Nothing -> throw . YosysError $ "Invalid input: " <> nm
  223           Just (elemty, _) -> do
  224             idx <- liftIO . SC.scNat sc $ fromIntegral i
  225             idxed <- liftIO $ SC.scAt sc width elemty inp idx
  226             pure (nm, idxed)
  227       let inpsWithSt = Map.insert "__state__" st inps
  228       cryptolRecord sc inpsWithSt
  229 
  230     summarizeOutput :: SC.Term -> m (SC.Term, Map Text SC.Term)
  231     summarizeOutput outrec = do
  232       outstate <- liftIO $ cryptolRecordSelect sc codomainFields outrec "__state__"
  233       outputs <- fmap Map.fromList . forM (Map.assocs $ s ^. yosysSequentialOutputFields) $ \(nm, (ty, _)) -> do
  234         out <- liftIO $ cryptolRecordSelect sc codomainFields outrec nm
  235         wrapped <- liftIO $ SC.scSingle sc ty out
  236         pure (nm, wrapped)
  237       pure (outstate, outputs)
  238 
  239     compose1 :: Integer -> (SC.Term, Map Text SC.Term) -> m (SC.Term, Map Text SC.Term)
  240     compose1 i (st, outs) = do
  241       inprec <- buildIntermediateInput i st
  242       outrec <- liftIO $ SC.scApply sc t inprec
  243       (st', outs') <- summarizeOutput outrec
  244       mergedOuts <- fmap Map.fromList . forM (Map.assocs outs') $ \(nm, arr) -> do
  245         case (Map.lookup nm $ s ^. yosysSequentialOutputFields, Map.lookup nm outs) of
  246           (Just (ty, _), Just rest) -> do
  247             restlen <- liftIO . SC.scNat sc $ fromIntegral i
  248             arrlen <- liftIO $ SC.scNat sc 1
  249             appended <- liftIO $ SC.scAppend sc restlen arrlen ty rest arr
  250             pure (nm, appended)
  251           _ -> pure (nm, arr)
  252       pure (st', mergedOuts)
  253 
  254   stateType <- fieldsToType sc $ s ^. yosysSequentialStateFields
  255   initialStateEC <- liftIO $ SC.scFreshEC sc "initial_state" stateType
  256   initialState <- liftIO $ SC.scExtCns sc initialStateEC
  257   (_, outputs) <- foldM (\acc i -> compose1 i acc) (initialState, Map.empty) [0..n]
  258 
  259   outputRecord <- cryptolRecord sc outputs
  260   res <- liftIO $ SC.scAbstractExts sc [initialStateEC, extendedInputRecordEC] outputRecord
  261   let cty = C.tFun extendedInputCryptolType extendedOutputCryptolType
  262 
  263   pure (res, cty)

  265 -- | Given a SAWCore term with an explicit state, iterate the term the given number of times.
  266 -- | Accessing the initial state produces an error.
  267 composeYosysSequential ::
  268   forall m.
  269   MonadIO m =>
  270   SC.SharedContext ->
  271   YosysSequential ->
  272   Integer ->
  273   m SC.TypedTerm
  274 composeYosysSequential sc s n = do
  275   (t, cty) <- composeYosysSequentialHelper sc s n
  276   stateType <- fieldsToType sc $ s ^. yosysSequentialStateFields
  277   initialStateMsg <- liftIO $ SC.scString sc "Attempted to read initial state of sequential circuit"
  278   initialState <- liftIO $ SC.scGlobalApply sc (SC.mkIdent SC.preludeName "error") [stateType, initialStateMsg]
  279   res <- liftIO $ SC.scApply sc t initialState
  280   pure $ SC.TypedTerm (SC.TypedTermSchema $ C.tMono cty) res

  282 -- | Given a SAWCore term with an explicit state, iterate the term the given number of times.
  283 -- | The resulting term has a parameter for the initial state.
  284 composeYosysSequentialWithState ::
  285   forall m.
  286   MonadIO m =>
  287   SC.SharedContext ->
  288   YosysSequential ->
  289   Integer ->
  290   m SC.TypedTerm
  291 composeYosysSequentialWithState sc s n = do
  292   (t, cty) <- composeYosysSequentialHelper sc s n
  293   scty <- fieldsToCryptolType $ s ^. yosysSequentialStateFields
  294   pure $ SC.TypedTerm (SC.TypedTermSchema . C.tMono $ C.tFun scty cty) t
