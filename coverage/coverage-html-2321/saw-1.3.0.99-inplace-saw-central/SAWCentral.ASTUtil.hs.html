   13 -- NamedTyVars {{{
   14 
   15 --
   16 -- namedTyVars is a type-class-polymorphic function for extracting named
   17 -- type variables from a type or type schema. It returns a set of Name
   18 -- (Name is just Text) manifested as a map from those Names to their
   19 -- positions/provenance.
   20 --
   21 
   22 class NamedTyVars t where
   23   namedTyVars :: t -> Map Name Pos
   24 
   25 instance (Ord k, NamedTyVars a) => NamedTyVars (Map k a) where
   26   namedTyVars = namedTyVars . M.elems
   27 
   28 instance (NamedTyVars a) => NamedTyVars [a] where
   29   namedTyVars = M.unionsWith choosePos . map namedTyVars
   30 
   31 instance NamedTyVars Type where
   32   namedTyVars t = case t of
   33     TyCon _ _ ts      -> namedTyVars ts
   34     TyRecord _ tm     -> namedTyVars tm
   35     TyVar pos n       -> M.singleton n pos
   36     TyUnifyVar _ _    -> M.empty
   37 
   38 instance NamedTyVars Schema where
   39   namedTyVars (Forall ns t) = namedTyVars t M.\\ M.fromList ns'
   40     where ns' = map (\(pos, n) -> (n, pos)) ns
   41

   44 -- SubstituteTyVars {{{
   45 
   46 --
   47 -- substituteTyVars is a typeclass-polymorphic function for
   48 -- substituting named type variables (such as those declared with
   49 -- typedef) in a Type.
   50 --
   51 -- Note: substituteTyVars is exposed from this module and reused by
   52 -- the interpreter as part of its handling of typedefs during
   53 -- execution.
   54 --
   55 
   56 class SubstituteTyVars t where
   57   -- | @substituteTyVars m x@ applies the map @m@ to type variables in @x@.
   58   substituteTyVars :: Map Name NamedType -> t -> t
   59 
   60 instance (SubstituteTyVars a) => SubstituteTyVars (Maybe a) where
   61   substituteTyVars tyenv = fmap (substituteTyVars tyenv)
   62 
   63 instance (SubstituteTyVars a) => SubstituteTyVars [a] where
   64   substituteTyVars tyenv = map (substituteTyVars tyenv)
   65 
   66 instance SubstituteTyVars Type where
   67   substituteTyVars tyenv ty = case ty of
   68     TyCon pos tc ts     -> TyCon pos tc (substituteTyVars tyenv ts)
   69     TyRecord pos fs     -> TyRecord pos (fmap (substituteTyVars tyenv) fs)
   70     TyUnifyVar _ _      -> ty
   71     TyVar _ n           ->
   72         case M.lookup n tyenv of
   73             Nothing -> ty
   74             Just AbstractType -> ty
   75             Just (ConcreteType ty') -> ty'
   76 
   77 -- }}}
