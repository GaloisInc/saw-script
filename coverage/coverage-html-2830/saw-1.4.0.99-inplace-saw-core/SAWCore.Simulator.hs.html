   89 data SimulatorConfig l =
   90   SimulatorConfig
   91   { simPrimitive :: Name -> MValue l
   92   -- ^ Interpretation of 'Primitive' terms.
   93   , simVariable :: Term -> VarName -> TValue l -> MValue l
   94   -- ^ Interpretation of free 'Variable' terms.
   95   , simConstant :: Name -> TValue l -> Maybe (MValue l)
   96   -- ^ Interpretation of 'Constant' terms. 'Nothing' indicates that
   97   -- the body of the constant should be evaluated. 'Just' indicates
   98   -- that the constant's definition should be overridden.
   99   , simRecursor :: Name -> Sort -> Maybe (MValue l)
  100   -- ^ Interpretation of 'Recursor' terms. 'Nothing' indicates that
  101   -- the generic recursor implementation should be used, while 'Just'
  102   -- indicates that the recursor should be overridden.
  103   , simModMap :: ModuleMap
  104   , simLazyMux :: VBool l -> MValue l -> MValue l -> MValue l
  105   }

  124 {-# SPECIALIZE
  125   evalTermF :: Show (Extra l) =>
  126     SimulatorConfigIn IO l ->

  132 -- | Generic evaluator for TermFs.
  133 evalTermF :: forall l. (VMonadLazy l, Show (Extra l)) =>
  134              SimulatorConfig l          -- ^ Evaluator for global constants
  135           -> (Term -> OpenValue l)      -- ^ Evaluator for subterms under binders
  136           -> (Term -> MValue l)         -- ^ Evaluator for subterms in the same bound variable context
  137           -> TermF Term -> OpenValue l
  138 evalTermF cfg lam recEval tf env =
  139   case tf of
  140     App t1 t2               -> recEval t1 >>= \case
  141                                  VFun f ->
  142                                    do x <- recEvalDelay t2
  143                                       f x
  144                                  _ -> panic "evalTermF" ["Expected VFun"]
  145     Lambda nm _tp t         -> pure $ VFun (\x -> lam t (IntMap.insert (vnIndex nm) x env))
  146     Pi nm t1 t2             -> do v <- evalType t1
  147                                   body <-
  148                                     if IntSet.member (vnIndex nm) (freeVars t2) then
  149                                       pure (VDependentPi (\x -> toTValue <$> lam t2 (IntMap.insert (vnIndex nm) x env)))
  150                                     else
  151                                       VNondependentPi . toTValue <$> lam t2 env
  152                                   return $ TValue $ VPiType v body
  153 
  154     Constant nm             -> do let r = requireNameInMap nm (simModMap cfg)
  155                                   ty' <- evalType (resolvedNameType r)
  156                                   case simConstant cfg nm ty' of
  157                                     Just override -> override
  158                                     Nothing ->
  159                                       case r of
  160                                         ResolvedCtor ctor ->
  161                                           ctorValue nm ty' (ctorNumParams ctor) (ctorNumArgs ctor)
  162                                         ResolvedDataType dt ->
  163                                           dtValue nm (dtNumParams dt) (dtNumIndices dt)
  164                                         ResolvedDef d ->
  165                                           case defBody d of
  166                                             Just t -> recEval t
  167                                             Nothing -> simPrimitive cfg nm
  168 
  169     Variable nm tp          -> case IntMap.lookup (vnIndex nm) env of
  170                                  Just x -> force x
  171                                  Nothing ->
  172                                    do tp' <- evalType tp
  173                                       simVariable cfg tp nm tp'
  174     FTermF ftf              ->
  175       case ftf of
  176         UnitValue           -> return VUnit
  177 
  178         UnitType            -> return $ TValue VUnitType
  179 
  180         PairValue x y       -> do tx <- recEvalDelay x
  181                                   ty <- recEvalDelay y
  182                                   return $ VPair tx ty
  183 
  184         PairType x y        -> do vx <- evalType x
  185                                   vy <- evalType y
  186                                   return $ TValue $ VPairType vx vy
  187 
  188         PairLeft x          -> recEval x >>= \case
  189                                  VPair l _r -> force l
  190                                  _ -> panic "evalTermF" ["Expected VPair"]
  191 
  192         PairRight x         -> recEval x >>= \case
  193                                  VPair _l r -> force r
  194                                  _ -> panic "evalTermF" ["Expected VPair"]
  195 
  196         Recursor r ->
  197           case simRecursor cfg (recursorDataType r) (recursorSort r) of
  198             Just v -> v
  199             Nothing ->
  200               do let dname = recursorDataType r
  201                  let nparams = recursorNumParams r
  202                  let nixs = recursorNumIxs r
  203                  let cnames = recursorCtorOrder r
  204                  vFunList nparams $ \_ps_thunks ->
  205                    pure $ VFun $ \_motive ->
  206                    vFunList (length cnames) $ \elim_thunks ->
  207                    do let es = Map.fromList (zip (map nameIndex cnames) elim_thunks)
  208                       let vrec = VRecursor dname nixs es
  209                       vFunList nixs (\_ixs -> pure (evalRecursor vrec))
  210 
  211         RecordType elem_tps ->
  212           TValue . VRecordType <$> traverse (traverse evalType) elem_tps
  213 
  214         RecordValue elems   ->
  215           VRecordValue <$> mapM (\(fld,t) -> (fld,) <$> recEvalDelay t) elems
  216 
  217         RecordProj t fld    -> recEval t >>= \case
  218                                  v@VRecordValue{} -> valRecordProj v fld
  219                                  _ -> panic "evalTermF" ["Expected VRecordValue"]
  220 
  221         Sort s _h           -> return $ TValue (VSort s)
  222 
  223         ArrayValue _ tv     -> liftM VVector $ mapM recEvalDelay tv
  224 
  225         StringLit s         -> return $ VString s
  226   where
  227     evalType :: Term -> EvalM l (TValue l)
  228     evalType t = toTValue <$> recEval t
  229 
  230     toTValue :: HasCallStack => Value l -> TValue l
  231     toTValue (TValue x) = x
  232     toTValue t = panic "evalTermF / toTValue" ["Not a type value: " <> Text.pack (show t)]
  233 
  234     evalRecursor :: VRecursor l -> Value l
  235     evalRecursor vrec@(VRecursor d _nixs ps_fs) =
  236       vStrictFun $ \argv ->
  237       case evalConstructor argv of
  238         Just (ctor, args)
  239           | Just elim <- Map.lookup (nameIndex (ctorName ctor)) ps_fs ->
  240               do elimv <- force elim
  241                  reduceRecursor (evalRecursor vrec) elimv args (ctorArgStruct ctor)
  242 
  243           | otherwise ->
  244               panic "evalTermF / evalRecursor"
  245               ["Could not find info for constructor: " <> toAbsoluteName (nameInfo (ctorName ctor))]
  246         Nothing ->
  247           case argv of
  248             VCtorMux _ps branches ->
  249               do alts <- traverse (evalCtorMuxBranch vrec) (IntMap.elems branches)
  250                  combineAlts alts
  251             VBVToNat{} ->
  252               panic "evalTerF / evalRecursor"
  253               ["Unsupported symbolic recursor argument of type Nat"]
  254             _ ->
  255               panic "evalTermF / evalRecursor"
  256               ["Expected constructor for datatype: " <> toAbsoluteName (nameInfo d)]
  257 
  258     evalCtorMuxBranch ::
  259       VRecursor l ->
  260       (VBool l, Name, TValue l, [Thunk l]) ->
  261       EvalM l (VBool l, EvalM l (Value l))
  262     evalCtorMuxBranch r (p, c, _ct, args) =
  263       case r of
  264         VRecursor _d _nixs ps_fs ->
  265           do let i = nameIndex c
  266              case (lookupVarIndexInMap i (simModMap cfg), Map.lookup i ps_fs) of
  267                (Just (ResolvedCtor ctor), Just elim) ->
  268                  do elimv <- force elim
  269                     pure (p, reduceRecursor (evalRecursor r) elimv args (ctorArgStruct ctor))
  270                _ -> panic "evalTermF / evalCtorMuxBranch"
  271                     ["could not find info for constructor: " <> toAbsoluteName (nameInfo c)]
  272 
  273     combineAlts :: [(VBool l, EvalM l (Value l))] -> EvalM l (Value l)
  274     combineAlts [] = panic "evalTermF / combineAlts" ["no alternatives"]
  275     combineAlts [(_, x)] = x
  276     combineAlts ((p, x) : alts) = simLazyMux cfg p x (combineAlts alts)
  277 
  278     evalConstructor :: Value l -> Maybe (Ctor, [Thunk l])
  279     evalConstructor (VCtorApp c _tv _ps args) =
  280       case lookupVarIndexInMap (nameIndex c) (simModMap cfg) of
  281         Just (ResolvedCtor ctor) -> Just (ctor, args)
  282         _ -> Nothing
  283     evalConstructor _ =
  284        Nothing
  285 
  286     recEvalDelay :: Term -> EvalM l (Thunk l)
  287     recEvalDelay = delay . recEval
  288 
  289     ctorValue :: Name -> TValue l -> Int -> Int -> MValue l
  290     ctorValue nm tv i j =
  291       vFunList i $ \params ->
  292       vFunList j $ \args ->
  293       pure $ VCtorApp nm tv params args
  294 
  295     dtValue :: Name -> Int -> Int -> MValue l
  296     dtValue nm i j =
  297       vStrictFunList i $ \params ->
  298       vStrictFunList j $ \idxs ->
  299       pure $ TValue $ VDataType nm params idxs

  301 -- | Evaluate a recursor applied to a specific data constructor.
  302 reduceRecursor ::
  303   forall l. (VMonadLazy l, Show (Extra l)) =>
  304   Value l {- ^ recursor function expecting datatype argument -} ->
  305   Value l {- ^ constructor eliminator function -} ->
  306   [Thunk l] {- ^ constructor arguments -} ->
  307   CtorArgStruct {- ^ constructor formal argument descriptor -} ->
  308   MValue l
  309 reduceRecursor r elim c_args argstruct = go elim c_args (map snd (ctorArgs argstruct))
  310   where
  311     go :: Value l -> [Thunk l] -> [CtorArg] -> MValue l
  312     go e [] [] = pure e
  313     go e (x : xs) (arg : args) =
  314       case arg of
  315         ConstArg _ ->
  316           do e' <- apply e x
  317              go e' xs args
  318         RecursiveArg zs _ixs ->
  319           do e1 <- apply e x
  320              recx <- delay (mk_rec_arg (length zs) x)
  321              e2 <- apply e1 recx
  322              go e2 xs args
  323     go _ _ _ = panic "reduceRecursor" ["Wrong number of constructor arguments"]
  324 
  325     -- For a recursive argument, we need a value of the form
  326     -- > \z1 .. zk -> r (x z1 .. zk)
  327     mk_rec_arg :: Int -> Thunk l -> MValue l
  328     mk_rec_arg k x =
  329       vFunList k $ \zs ->
  330       do x_zs <- delay (force x >>= \v -> applyAll v zs)
  331          apply r x_zs

  393 -- | A variant of 'evalGlobal' that lets the uninterpreted function
  394 -- symbol and external-constant callbacks have access to the 'TermF'.
  395 evalGlobal' ::
  396   forall l. (VMonadLazy l, Show (Extra l)) =>
  397   ModuleMap ->
  398   -- | Implementations of 'Primitive' terms, plus overrides for 'Constant' and 'CtorApp' terms
  399   Map Ident (Prims.Prim l) ->
  400   -- | Implementations of free 'Variable' terms
  401   (Term -> VarName -> TValue l -> MValue l) ->
  402   -- | Overrides for Constant terms (e.g. uninterpreted functions)
  403   (Name -> TValue l -> Maybe (MValue l)) ->
  404   -- | Overrides for Recursor terms
  405   (Name -> Sort -> Maybe (MValue l)) ->
  406   -- | Handler for stuck primitives
  407   (Name -> Text -> [Thunk l] -> MValue l) ->
  408   -- | Lazy mux operation
  409   (VBool l -> MValue l -> MValue l -> MValue l) ->
  410   EvalM l (SimulatorConfig l)
  411 evalGlobal' modmap prims variable constant recursor primHandler lazymux =
  412   do checkPrimitives modmap prims
  413      return (SimulatorConfig primitive variable constant' recursor modmap lazymux)
  414   where
  415     constant' :: Name -> TValue l -> Maybe (MValue l)
  416     constant' nm tv =
  417       case constant nm tv of
  418         Just v -> Just v
  419         Nothing ->
  420           case nameInfo nm of
  421             ModuleIdentifier ident ->
  422               evalPrim (primHandler nm) <$> Map.lookup ident prims
  423             ImportedName{} -> Nothing
  424 
  425     primitive :: Name -> MValue l
  426     primitive nm =
  427       case nameInfo nm of
  428         ImportedName {} ->
  429           panic "evalGlobal'" ["Unimplemented global: " <> toAbsoluteName (nameInfo nm)]
  430         ModuleIdentifier ident ->
  431           case Map.lookup ident prims of
  432             Just v  -> evalPrim (primHandler nm) v
  433             Nothing -> panic "evalGlobal'" ["Unimplemented global: " <> identText ident]

  435 -- | Check that all the primitives declared in the given module
  436 --   are implemented, and that terms with implementations are not
  437 --   overridden.
  438 checkPrimitives :: forall l. (VMonadLazy l, Show (Extra l))
  439                 => ModuleMap
  440                 -> Map Ident (Prims.Prim l)
  441                 -> EvalM l ()
  442 checkPrimitives modmap prims = do
  443    -- FIXME this is downgraded to a warning temporarily while we work out a
  444    -- solution to issue GaloisInc/saw-script#48
  445    --   when (not $ null unimplementedPrims) (panic $ unimplementedMsg)
  446    -- (if null unimplementedPrims then id else Debug.trace (unimplementedMsg++"\n")) $
  447 --   (if null overridePrims then id else Debug.trace (overrideMsg++"\n")) $
  448      return ()
  449 
  450   where _unimplementedMsg = unwords $
  451             ("WARNING unimplemented primitives:" : (map show unimplementedPrims))
  452         _overrideMsg = unwords $
  453             ("WARNING overridden definitions:" : (map show overridePrims))
  454 
  455         primSet = Set.fromList $ mapMaybe defIdent $ allModulePrimitives modmap
  456         defSet  = Set.fromList $ mapMaybe defIdent $ allModuleActualDefs modmap
  457         implementedPrims = Map.keysSet prims
  458 
  459         unimplementedPrims = Set.toList $ Set.difference primSet implementedPrims
  460         overridePrims = Set.toList $ Set.intersection defSet implementedPrims

  469 -- The evaluation strategy for SharedTerms involves two memo tables:
  470 -- The first, @memoClosed@, is precomputed and contains the result of
  471 -- evaluating all _closed_ subterms. The same @memoClosed@ table is
  472 -- used for evaluation under lambdas, since the meaning of a closed
  473 -- term does not depend on the local variable context. The second memo
  474 -- table is @memoLocal@, which additionally includes the result of
  475 -- evaluating _open_ terms in the current variable context. It is
  476 -- reinitialized to @memoClosed@ whenever we descend under a lambda
  477 -- binder.
  478 
  479 {-# SPECIALIZE evalSharedTerm ::
  480   Show (Extra l) => SimulatorConfigIn Id l -> Term -> MValueIn Id l #-}
  481 {-# SPECIALIZE evalSharedTerm ::
  482   Show (Extra l) => SimulatorConfigIn IO l -> Term -> MValueIn IO l #-}
  483 
  484 -- | Evaluator for shared terms.
  485 evalSharedTerm :: (VMonadLazy l, MonadFix (EvalM l), Show (Extra l)) =>
  486                   SimulatorConfig l -> Term -> MValue l
  487 evalSharedTerm cfg t = do
  488   memoClosed <- mkMemoClosed cfg t
  489   evalOpen cfg memoClosed t IntMap.empty
  490 
  491 {-# SPECIALIZE mkMemoClosed ::
  492   Show (Extra l) =>
  493   SimulatorConfigIn Id l -> Term -> Id (IntMap (ThunkIn Id l)) #-}
  494 {-# SPECIALIZE mkMemoClosed ::
  495   Show (Extra l) =>
  496   SimulatorConfigIn IO l -> Term -> IO (IntMap (ThunkIn IO l)) #-}
  497 
  498 -- | Precomputing the memo table for closed subterms.
  499 mkMemoClosed :: forall l. (VMonadLazy l, MonadFix (EvalM l), Show (Extra l)) =>
  500                 SimulatorConfig l -> Term -> EvalM l (IntMap (Thunk l))
  501 mkMemoClosed cfg t =
  502   mfix $ \memoClosed -> mapM (delay . evalClosedTermF cfg memoClosed) subterms
  503   where
  504     -- | Map of all closed subterms of t.
  505     subterms :: IntMap (TermF Term)
  506     subterms = fmap fst $ IntMap.filter (IntSet.null . snd) $ State.execState (go t) IntMap.empty
  507 
  508     go :: Term -> State.State (IntMap (TermF Term, IntSet)) IntSet
  509     go (STApp{ stAppIndex = i, stAppTermF = tf }) =
  510       do memo <- State.get
  511          case IntMap.lookup i memo of
  512            Just (_, b) -> pure b
  513            Nothing ->
  514              do b <- termf tf
  515                 State.modify (IntMap.insert i (tf, b))
  516                 pure b
  517 
  518     termf :: TermF Term -> State.State (IntMap (TermF Term, IntSet)) IntSet
  519     termf tf =
  520       case tf of
  521         Constant nm ->
  522           -- if tf is a defined constant, traverse the definition body and type
  523           do let r = requireNameInMap nm (simModMap cfg)
  524              void $ go (resolvedNameType r)
  525              case r of
  526                ResolvedDef (defBody -> Just body) -> go body
  527                _ -> pure IntSet.empty
  528         Lambda x _ty body ->
  529           -- skip type, which is not used for simulation
  530           IntSet.delete (vnIndex x) <$> go body
  531         _ ->
  532           freesTermF <$> traverse go tf

  534 {-# SPECIALIZE evalClosedTermF ::
  535   Show (Extra l) =>

  537   IntMap (ThunkIn Id l) ->
  538   TermF Term ->
  539   MValueIn Id l #-}
  540 {-# SPECIALIZE evalClosedTermF ::

  547 -- | Evaluator for closed terms, used to populate @memoClosed@.
  548 evalClosedTermF :: (VMonadLazy l, Show (Extra l)) =>
  549                    SimulatorConfig l
  550                 -> IntMap (Thunk l)
  551                 -> TermF Term -> MValue l
  552 evalClosedTermF cfg memoClosed tf = evalTermF cfg lam recEval tf IntMap.empty
  553   where
  554     lam = evalOpen cfg memoClosed
  555     recEval (STApp{ stAppIndex = i }) =
  556       case IntMap.lookup i memoClosed of
  557         Just x -> force x
  558         Nothing -> panic "evalClosedTermF" ["internal error"]

  560 {-# SPECIALIZE mkMemoLocal ::
  561   Show (Extra l) =>
  562   SimulatorConfigIn Id l ->
  563   IntMap (ThunkIn Id l) ->
  564   Term ->
  565   EnvIn Id l ->
  566   Id (IntMap (ThunkIn Id l)) #-}
  567 {-# SPECIALIZE mkMemoLocal ::
  568   Show (Extra l) =>
  569   SimulatorConfigIn IO l ->
  570   IntMap (ThunkIn IO l) ->
  571   Term ->
  572   EnvIn IO l ->
  573   IO (IntMap (ThunkIn IO l)) #-}
  574 
  575 -- | Precomputing the memo table for open subterms in the current context.
  576 mkMemoLocal :: forall l. (VMonadLazy l, Show (Extra l)) =>
  577                SimulatorConfig l -> IntMap (Thunk l) ->
  578                Term -> Env l -> EvalM l (IntMap (Thunk l))
  579 mkMemoLocal cfg memoClosed t env = go mempty t
  580   where
  581     go :: IntMap (Thunk l) -> Term -> EvalM l (IntMap (Thunk l))
  582     go memo (t'@STApp{ stAppIndex = i, stAppTermF = tf })
  583       | closedTerm t' = pure memo
  584       | otherwise =
  585         case IntMap.lookup i memo of
  586           Just _ -> pure memo
  587           Nothing ->
  588             do memo' <- goTermF memo tf
  589                thunk <- delay (evalLocalTermF cfg memoClosed memo' tf env)
  590                pure (IntMap.insert i thunk memo')
  591     goTermF :: IntMap (Thunk l) -> TermF Term -> EvalM l (IntMap (Thunk l))
  592     goTermF memo tf =
  593       case tf of
  594         FTermF ftf      -> foldlM go memo ftf
  595         App t1 t2       -> do memo' <- goTermF memo (unwrapTermF t1)
  596                               go memo' t2
  597         Lambda{}        -> pure memo
  598         Pi _ t1 _       -> go memo t1
  599         Constant{}      -> pure memo
  600         Variable _nm tp -> go memo tp
  601 
  602 {-# SPECIALIZE evalLocalTermF ::
  603   Show (Extra l) =>
  604   SimulatorConfigIn Id l ->
  605   IntMap (ThunkIn Id l) ->
  606   IntMap (ThunkIn Id l) ->
  607   TermF Term ->
  608   OpenValueIn Id l #-}
  609 {-# SPECIALIZE evalLocalTermF ::
  610   Show (Extra l) =>
  611   SimulatorConfigIn IO l ->
  612   IntMap (ThunkIn IO l) ->
  613   IntMap (ThunkIn IO l) ->
  614   TermF Term ->
  615   OpenValueIn IO l #-}
  616 -- | Evaluator for open terms, used to populate @memoLocal@.
  617 evalLocalTermF :: (VMonadLazy l, Show (Extra l)) =>
  618                    SimulatorConfig l
  619                 -> IntMap (Thunk l) -> IntMap (Thunk l)
  620                 -> TermF Term -> OpenValue l
  621 evalLocalTermF cfg memoClosed memoLocal tf0 env = evalTermF cfg lam recEval tf0 env
  622   where
  623     lam = evalOpen cfg memoClosed
  624     recEval (t@STApp{ stAppIndex = i, stAppTermF = tf }) =
  625       case IntMap.lookup i memo of
  626         Just x -> force x
  627         Nothing -> evalTermF cfg lam recEval tf env
  628       where memo = if closedTerm t then memoClosed else memoLocal

  630 {-# SPECIALIZE evalOpen ::
  631   Show (Extra l) =>
  632   SimulatorConfigIn Id l ->
  633   IntMap (ThunkIn Id l) ->
  634   Term ->
  635   OpenValueIn Id l #-}
  636 
  637 {-# SPECIALIZE evalOpen ::
  638   Show (Extra l) =>
  639   SimulatorConfigIn IO l ->
  640   IntMap (ThunkIn IO l) ->
  641   Term ->
  642   OpenValueIn IO l #-}
  643 
  644 -- | Evaluator for open terms; parameterized by a precomputed table @memoClosed@.
  645 evalOpen :: forall l. (VMonadLazy l, Show (Extra l)) =>
  646             SimulatorConfig l
  647          -> IntMap (Thunk l)
  648          -> Term -> OpenValue l
  649 evalOpen cfg memoClosed t env = do
  650   memoLocal <- mkMemoLocal cfg memoClosed t env
  651   let eval :: Term -> MValue l
  652       eval (t'@STApp{ stAppIndex = i, stAppTermF = tf }) =
  653         case IntMap.lookup i memo of
  654           Just x -> force x
  655           Nothing -> evalF tf
  656         where memo = if closedTerm t' then memoClosed else memoLocal
  657       evalF :: TermF Term -> MValue l
  658       evalF tf = evalTermF cfg (evalOpen cfg memoClosed) eval tf env
  659   eval t

  662 {-# SPECIALIZE evalPrim ::
  663   Show (Extra l) =>
  664   (Text -> [ThunkIn Id l] -> MValueIn Id l) ->
  665   PrimIn Id l ->
  666   MValueIn Id l
  667  #-}
  668 {-# SPECIALIZE evalPrim ::
  669   Show (Extra l) =>
  670   (Text -> [ThunkIn IO l] -> MValueIn IO l) ->
  671   PrimIn IO l ->
  672   MValueIn IO l
  673  #-}
  674 evalPrim :: forall l. (VMonadLazy l, Show (Extra l)) =>
  675   (Text -> [Thunk l] -> MValue l) ->
  676   Prims.Prim l ->
  677   MValue l
  678 evalPrim fallback = loop []
  679   where
  680     loop :: [Thunk l] -> Prims.Prim l -> MValue l
  681     loop env (Prims.PrimFun f) =
  682       pure $ VFun $ \x ->
  683         loop (x : env) (f x)
  684 
  685     loop env (Prims.PrimStrict f) =
  686       pure $ vStrictFun $ \x ->
  687         loop (ready x : env) (f x)
  688 
  689     loop env (Prims.PrimFilterFun msg r f) =
  690       pure $ vStrictFun $ \x ->
  691         runMaybeT (r x) >>= \case
  692           Just v -> loop (ready x : env) (f v)
  693           _ -> fallback msg (ready x : env)
  694 
  695     loop env (Prims.PrimExcept m) =
  696       runExceptT m >>= \case
  697         Right v  -> pure v
  698         Left msg -> fallback msg env
  699 
  700     loop _env (Prims.Prim m) = m
  701     loop _env (Prims.PrimValue v) = pure v
  702 
  703 -- | A basic handler for stuck primitives.
  704 defaultPrimHandler ::
  705   (VMonadLazy l, MonadFail (EvalM l)) =>
  706   Name -> Text -> [Thunk l] -> MValue l
  707 defaultPrimHandler nm msg env =
  708   fail $ unlines
  709   [ "Could not evaluate primitive " ++ Text.unpack (toAbsoluteName (nameInfo nm))
  710   , "On argument " ++ show (length env)
  711   , Text.unpack msg
