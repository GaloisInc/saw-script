   30 -- | Given an 'SBV.Solver' from @SBV@, attempt to query the solver for its
   31 -- version and return the result as a string.
   32 -- Adapted from @what4/test/ProbeSolvers.hs@
   33 getSolverVersion :: SBV.Solver -> IO (Maybe String)
   34 getSolverVersion s =
   35   let s' = SBV.solver $ SBV.defaultSolverConfig s
   36       nope what = panic "getSolverVersion" [
   37              "Version request for unsupported solver " <> what
   38            ]
   39 
   40       (args, pref) = case SBV.name s' of
   41         -- n.b. abc will return a non-zero exit code if asked for command usage.
   42         SBV.ABC       -> (["s", "-q", "version;quit"], "UC Berkeley, ABC ")
   43         SBV.Boolector -> (["--version"]              , "")
   44         SBV.Bitwuzla  -> (["--version"]              , "")
   45         SBV.CVC4      -> (["--version"]              , "This is CVC4 version ")
   46         SBV.CVC5      -> (["--version"]              , "This is cvc5 version ")
   47         SBV.DReal     -> (["--version"]              , "dReal v")
   48         SBV.MathSAT   -> (["-version"]               , "MathSAT5 version ")
   49         SBV.OpenSMT   -> nope "OpenSMT"
   50         SBV.Yices     -> (["--version"]              , "Yices ")
   51         SBV.Z3        -> (["--version"]              , "Z3 version ")
   52   in
   53   try (readProcessWithExitCode (SBV.executable s') args "") >>= \case
   54     Right (ExitSuccess,o,_) | (l:_) <- lines o ->
   55       return $ Just $ dropPrefix pref l
   56     Right _                   -> return Nothing
   57     Left (_ :: SomeException) -> return Nothing
   58   where dropPrefix (x:xs) (y:ys) | x == y = dropPrefix xs ys
   59         dropPrefix _ ys = ys

   61 -- | Make a characteristic string from a git hash (which might
   62 --   actually be a tag with a version, and might not) and a list of
   63 --   Cabal packages and versions.
   64 makeSubmoduleVersion :: [(Text, Text)] -> Maybe Text -> Text
   65 makeSubmoduleVersion vers mhash =
   66     let vers' = Text.intercalate ";" (map (\(n, v) -> n <> " " <> v) vers) in
   67     case (mhash, foundGit) of
   68         -- git wasn't found
   69         (Nothing, False) -> vers' <> ": <VCS-less build>"
   70         -- release or snapshot version, cabal versions should be sufficient
   71         (Nothing, True) -> vers'
   72         -- got a git hash, include it
   73         (Just hash, _) -> vers' <> ": " <> hash

   75 -- | Get the 'SolverBackendVersion' of a 'SolverBackend'
   76 getSolverBackendVersion :: SolverBackend -> IO (Maybe String)
   77 getSolverBackendVersion backend = case backend of
   78   What4     -> pure $ Just $ Text.unpack $ makeSubmoduleVersion what4Versions what4Hash
   79   SBV       -> pure $ Just SBV.version
   80   AIG       -> pure $ Just $ Text.unpack $ makeSubmoduleVersion aigVersions aigHash
   81   RME       -> pure $ Just $ Text.unpack $ makeSubmoduleVersion rmeVersions rmeHash
   82   -- We use individual cases for the remaining constructors to ensure that if
   83   -- a new backend is added, a warning is generated for this pattern match
   84   ABC       -> getSolverVersion SBV.ABC
   85   Bitwuzla  -> getSolverVersion SBV.Bitwuzla
   86   Boolector -> getSolverVersion SBV.Boolector
   87   CVC4      -> getSolverVersion SBV.CVC4
   88   CVC5      -> getSolverVersion SBV.CVC5
   89   DReal     -> getSolverVersion SBV.DReal
   90   MathSAT   -> getSolverVersion SBV.MathSAT
   91   OpenSMT   -> getSolverVersion SBV.OpenSMT
   92   Yices     -> getSolverVersion SBV.Yices
   93   Z3        -> getSolverVersion SBV.Z3
   94 
   95 -- | Get the set of 'SolverBackendVersions' of a list of 'SolverBackend's
   96 getSolverBackendVersions :: [SolverBackend] -> IO SolverBackendVersions
   97 getSolverBackendVersions backends = Map.fromList <$>
   98   mapM (\backend -> (backend,) <$> getSolverBackendVersion backend) backends
