   75 data TypeError = TypeError Pos String
   76   deriving Show
   77 
   78 mkNuMatching [t| TypeError |]
   79 
   80 instance Closable TypeError where
   81   toClosed = unsafeClose
   82 
   83 instance Liftable TypeError where
   84   mbLift = unClosed . mbLift . fmap toClosed

   86 ----------------------------------------------------------------------
   87 -- * Type-checking type
   88 ----------------------------------------------------------------------
   89 
   90 -- | Type-checking computations carrying a 'TcEnv' and which
   91 -- can fail. Access the environment with 'tcLocal' and 'tcAsk'
   92 -- and fail with 'tcError'.
   93 newtype Tc a = Tc { runTc :: TcEnv -> Either TypeError a }
   94 
   95 -- | Run a type-checking computation given an initial permission
   96 -- environment.
   97 startTc ::
   98   Tc a                  {- ^ typechecking action        -} ->
   99   PermEnv               {- ^ permission environment     -} ->
  100   Either TypeError a
  101 startTc tc env = runTc tc (TcEnv [] env)
  102 
  103 -- | 'fmap' derived from 'Monad'
  104 instance Functor Tc where
  105   fmap = liftM
  106 
  107 -- | ('<*>') derived from 'Monad'
  108 instance Applicative Tc where
  109   pure x = Tc \_ -> Right x
  110   (<*>) = ap
  111 
  112 instance Monad Tc where
  113   Tc f >>= g = Tc \env ->
  114     do x <- f env
  115        runTc (g x) env
  116 
  117 instance MonadBind Tc where
  118   mbM m = Tc \env ->
  119     case mbMatch $ fmap (`runTc` env) m of
  120       [nuMP| Left e  |] -> Left (mbLift e)
  121       [nuMP| Right x |] -> Right x
  122 
  123 -- | Run type-checking computation with local changes to the
  124 -- type-checking environment.
  125 tcLocal ::
  126   (TcEnv -> TcEnv)      {- ^ environment update         -} ->
  127   Tc a -> Tc a
  128 tcLocal f (Tc k) = Tc (k . f)
  129 
  130 -- | Get current type-checking environment
  131 tcAsk :: Tc TcEnv
  132 tcAsk = Tc Right
  133 
  134 -- | Abort checking with an error message
  135 tcError ::
  136   Pos                   {- ^ error location             -} ->
  137   String                {- ^ error message              -} ->
  138   Tc a
  139 tcError p err = Tc (\_ -> Left (TypeError p err))

  142 -- * Casting
  143 ----------------------------------------------------------------------

  145 -- | Cast a typed value to the requested type or
  146 -- raise an error in 'Tc'
  147 tcCastTyped ::
  148   Pos                   {- ^ position of expression     -} ->
  149   TypeRepr a            {- ^ target type                -} ->
  150   Typed f b             {- ^ expression                 -} ->
  151   Tc (f a)              {- ^ casted expression          -}
  152 tcCastTyped p tp (Typed tp' f) =
  153   case testEquality tp tp' of
  154     Just Refl -> pure f
  155     Nothing   -> tcError p ("Expected type " ++ show tp ++ ", got type " ++ show tp')

  161 -- | Run a parsing computation in a context extended with an expression variable
  162 withExprVar ::
  163   String                {- ^ identifier                 -} ->
  164   TypeRepr tp           {- ^ type of identifer          -} ->
  165   ExprVar tp            {- ^ implementation             -} ->
  166   Tc a -> Tc a
  167 withExprVar str tp x = tcLocal \env ->
  168   env { tcEnvExprVars = (str, Some (Typed tp x)) : tcEnvExprVars env }
  169 
  170 -- | Run a parsing computation in a context extended with 0 or more expression
  171 -- variables
  172 withExprVars ::
  173   RAssign (Constant String) ctx ->
  174   CruCtx ctx ->
  175   RAssign Name ctx ->
  176   Tc a ->
  177   Tc a
  178 withExprVars MNil                CruCtxNil           MNil       m = m
  179 withExprVars (xs :>: Constant x) (CruCtxCons ctx tp) (ns :>: n) m = withExprVars xs ctx ns (withExprVar x tp n m)

  185 -- | Check an 'AstType' as a 'TypeRepr'
  186 tcType :: AstType -> Tc (Some TypeRepr)
  187 tcType t = mapSome unKnownReprObj <$> tcTypeKnown t
  188 
  189 -- | Check an 'AstType' and build a @'KnownRepr' 'TypeRepr'@ instance for it
  190 tcTypeKnown :: AstType -> Tc (Some (KnownReprObj TypeRepr))
  191 tcTypeKnown t =
  192   case t of
  193     TyUnit          {} -> pure (Some (mkKnownReprObj UnitRepr))
  194     TyBool          {} -> pure (Some (mkKnownReprObj BoolRepr))
  195     TyNat           {} -> pure (Some (mkKnownReprObj NatRepr))
  196     TyLifetime      {} -> pure (Some (mkKnownReprObj LifetimeRepr))
  197     TyRwModality    {} -> pure (Some (mkKnownReprObj RWModalityRepr))
  198     TyPermList      {} -> pure (Some (mkKnownReprObj PermListRepr))
  199 
  200     TyBV p n ->
  201       withPositive p "Zero bitvector width not allowed" n \w ->
  202         pure (Some (mkKnownReprObj (BVRepr w)))
  203     TyLlvmPtr p n ->
  204       withPositive p "Zero LLVM Ptr width not allowed" n \w ->
  205         pure (Some (mkKnownReprObj (LLVMPointerRepr w)))
  206     TyLlvmFrame p n ->
  207       withPositive p "Zero LLVM Frame width not allowed" n \w ->
  208         pure (Some (mkKnownReprObj (LLVMFrameRepr w)))
  209     TyLlvmShape p n ->
  210       withPositive p "Zero LLVM Shape width not allowed" n \w ->
  211         pure (Some (mkKnownReprObj (LLVMShapeRepr w)))
  212     TyLlvmBlock p n ->
  213       withPositive p "Zero LLVM Block width not allowed" n \w ->
  214         pure (Some (mkKnownReprObj (LLVMBlockRepr w)))
  215 
  216     TyStruct _ fs ->
  217       do fs1 <- traverse tcTypeKnown fs
  218          let fs2 = foldl structAdd (Some (mkKnownReprObj Ctx.empty)) fs1
  219          case fs2 of
  220            Some xs@KnownReprObj -> pure (Some (mkKnownReprObj (StructRepr (unKnownReprObj xs))))
  221 
  222     TyPerm _ x ->
  223       do Some tp@KnownReprObj <- tcTypeKnown x
  224          pure (Some (mkKnownReprObj (ValuePermRepr (unKnownReprObj tp))))
  225 
  226 -- | Helper function for building struct type lists
  227 structAdd ::
  228   Some (KnownReprObj (Ctx.Assignment TypeRepr)) ->
  229   Some (KnownReprObj TypeRepr) ->
  230   Some (KnownReprObj (Ctx.Assignment TypeRepr))
  231 structAdd (Some acc@KnownReprObj) (Some x@KnownReprObj) =
  232   Some (mkKnownReprObj (Ctx.extend (unKnownReprObj acc) (unKnownReprObj x)))

  238 -- | Parse an identifier as an expression variable of a specific type
  239 tcVar ::
  240   TypeRepr a            {- ^ expected type              -} ->
  241   Pos                   {- ^ identifier position        -} ->
  242   String                {- ^ identifier                 -} ->
  243   Tc (ExprVar a)
  244 tcVar ty p name =
  245   do Some tn <- tcTypedName p name
  246      tcCastTyped p ty tn
  247 
  248 -- | Check a valid identifier string as an expression variable
  249 tcTypedName ::
  250   Pos                   {- ^ identifier position        -} ->
  251   String                {- ^ identifier                 -} ->
  252   Tc TypedName
  253 tcTypedName p name =
  254   do env <- tcAsk
  255      case lookup name (tcEnvExprVars env) of
  256        Nothing -> tcError p ("Unknown variable:" ++ name)
  257        Just stn -> pure stn
  258 
  259 -- | Check an 'AstExpr' as a 'PermExpr' with a known type.
  260 tcKExpr :: KnownRepr TypeRepr a => AstExpr -> Tc (PermExpr a)
  261 tcKExpr = tcExpr knownRepr
  262 
  263 -- | Check an 'AstExpr' as a 'PermExpr' with a given type.
  264 -- This is a top-level entry-point to the checker that will
  265 -- resolve variables.
  266 tcExpr :: TypeRepr a -> AstExpr -> Tc (PermExpr a)
  267 tcExpr ty (ExVar p name Nothing Nothing) = PExpr_Var <$> tcVar ty p name
  268 
  269 tcExpr tp@(LLVMShapeRepr w) (ExVar p name (Just args) Nothing) =
  270   do env <- tcAsk
  271      case lookupNamedShape (tcEnvPermEnv env) name of
  272        Just (SomeNamedShape nmsh)
  273          | Just Refl <- testEquality w (natRepr nmsh) ->
  274            do sub <- tcExprs p (namedShapeArgs nmsh) args
  275               pure (PExpr_NamedShape Nothing Nothing nmsh sub)
  276        Just (SomeNamedShape nmsh) ->
  277          tcError p $ renderDoc $ sep
  278          [ pretty "Named shape" <+> pretty name <+>
  279            pretty "is of incorrect type"
  280          , pretty "Expected:" <+> permPretty emptyPPInfo tp
  281          , pretty "Found:" <+>
  282            permPretty emptyPPInfo (LLVMShapeRepr (natRepr nmsh)) ]
  283        Nothing -> tcError p ("Unknown shape name: " ++ name)
  284 
  285 tcExpr tp@(ValuePermRepr sub_tp) (ExVar p name (Just args) Nothing) =
  286   do env <- tcAsk
  287      case lookupNamedPermName (tcEnvPermEnv env) name of
  288        Just (SomeNamedPermName npn)
  289          | Just Refl <- testEquality (namedPermNameType npn) sub_tp ->
  290            do arg_exprs <- tcExprs p (namedPermNameArgs npn) args
  291               pure (PExpr_ValPerm $ ValPerm_Named npn arg_exprs NoPermOffset)
  292        Just (SomeNamedPermName npn) ->
  293          tcError p $ renderDoc $ sep
  294          [ pretty "Named permission" <+> pretty (namedPermNameName npn) <+>
  295            pretty "is of incorrect type"
  296          , pretty "Expected:" <+> permPretty emptyPPInfo tp
  297          , pretty "Found:" <+> permPretty emptyPPInfo (namedPermNameType npn) ]
  298        Nothing -> tcError p ("Unknown shape name: " ++ name)
  299 
  300 tcExpr _ (ExVar p _ Just{} _) = tcError p "Unexpected variable instantiation"
  301 tcExpr _ (ExVar p _ _ Just{}) = tcError p "Unexpected variable offset"
  302 
  303 tcExpr UnitRepr         e = tcUnit e
  304 tcExpr NatRepr          e = tcNat e
  305 tcExpr (BVRepr w)       e = withKnownNat w (normalizeBVExpr <$> tcBV e)
  306 tcExpr (StructRepr fs)  e = tcStruct fs e
  307 tcExpr LifetimeRepr     e = tcLifetimeLit e
  308 tcExpr (LLVMPointerRepr w) e = withKnownNat w (tcLLVMPointer w e)
  309 tcExpr FunctionHandleRepr{} e = tcError (pos e) "Expected functionhandle" -- no literals
  310 tcExpr PermListRepr     e = tcError (pos e) "Expected permlist" -- no literals
  311 tcExpr RWModalityRepr   e = tcRWModality e
  312 tcExpr (ValuePermRepr t) e = permToExpr <$> tcValPerm t e
  313 tcExpr (LLVMShapeRepr w) e = withKnownNat w (tcLLVMShape e)
  314 
  315 tcExpr (IntrinsicRepr s _) e = tcError (pos e) ("Expected intrinsic type: " ++ show s)
  316

  338 -- | Check for a unit literal
  339 tcUnit :: AstExpr -> Tc (PermExpr UnitType)
  340 tcUnit ExUnit{} = pure PExpr_Unit
  341 tcUnit e        = tcError (pos e) "Expected unit"
  342 
  343 -- | Check for a nat literal
  344 tcNat :: AstExpr -> Tc (PermExpr NatType)
  345 tcNat (ExNat _ i) = pure (PExpr_Nat i)
  346 tcNat e           = tcError (pos e) "Expected integer"
  347 
  348 -- | Check for a bitvector expression
  349 tcBV :: (KnownNat w, 1 <= w) => AstExpr -> Tc (PermExpr (BVType w))
  350 tcBV (ExAdd _ x y) = bvAdd <$> tcBV x <*> tcBV y
  351 tcBV (ExNeg _ x)   = bvNegate <$> tcBV x
  352 tcBV e             = tcBVFactor e
  353 
  354 -- | Check for a bitvector factor. This is limited to
  355 -- variables, constants, and multiplication by a constant.
  356 tcBVFactor :: (KnownNat w, 1 <= w) => AstExpr -> Tc (PermExpr (BVType w))
  357 -- Constants
  358 tcBVFactor (ExNat _ i) = pure (bvInt (fromIntegral i))
  359 -- Multiplication by a constant
  360 tcBVFactor (ExMul _ c (ExVar p name Nothing Nothing)) =
  361   do c' <- asBVConst c
  362      Some tn <- tcTypedName p name
  363      bvMultBV c' . PExpr_Var <$> tcCastTyped p knownRepr tn
  364 tcBVFactor (ExMul _ (ExVar p name Nothing Nothing) c) =
  365   do c' <- asBVConst c
  366      Some tn <- tcTypedName p name
  367      bvMultBV c' . PExpr_Var <$> tcCastTyped p knownRepr tn
  368 -- Variables
  369 tcBVFactor (ExVar p name Nothing Nothing) =
  370   do Some tn <- tcTypedName p name
  371      PExpr_Var <$> tcCastTyped p knownRepr tn
  372 tcBVFactor e = tcError (pos e) "Expected BV factor"
  373 
  374 -- | Check for an integer literal, which can be negative. If one is found,
  375 -- convert it to a 'BV'. Fail otherwise.
  376 asBVConst :: (KnownNat w, 1 <= w) => AstExpr -> Tc (BV w)
  377 asBVConst (ExNat _ i) =
  378   pure $ BV.mkBV knownNat $ toInteger i
  379 asBVConst (ExNeg _ (ExNat _ i)) =
  380   pure $ BV.negate knownNat $ BV.mkBV knownNat $ toInteger i
  381 asBVConst e =
  382   tcError (pos e) "Expected integer or negated integer"
  383 
  384 -- | Check for a struct literal
  385 tcStruct :: CtxRepr fs -> AstExpr -> Tc (PermExpr (StructType fs))
  386 tcStruct ts (ExStruct p es) = PExpr_Struct <$> tcExprs p (mkCruCtx ts) es
  387 tcStruct _ e = tcError (pos e) "Expected struct"

  389 -- | Check a list of expressions. In case of arity issues
  390 -- an arity error is reported at the given position.
  391 tcExprs ::
  392   Pos                   {- ^ position for arity error   -} ->
  393   CruCtx fs             {- ^ expected types             -} ->
  394   [AstExpr]             {- ^ expressions                -} ->
  395   Tc (PermExprs fs)
  396 tcExprs p tys es = tcExprs' p tys (reverse es)
  397 
  398 -- | Helper for 'tcExprs'
  399 tcExprs' :: Pos -> CruCtx fs -> [AstExpr] -> Tc (PermExprs fs)
  400 tcExprs' _ CruCtxNil [] = pure PExprs_Nil
  401 tcExprs' p (CruCtxCons xs x) (y:ys) =
  402   do zs <- tcExprs' p xs ys
  403      z  <- tcExpr x y
  404      pure (zs :>: z)
  405 tcExprs' p _ _ = tcError p "Bad arity"
  406 
  407 -- | Parse a sequence of permissions of some given types
  408 tcValuePerms :: Pos -> RAssign TypeRepr tys -> [AstExpr] -> Tc (RAssign ValuePerm tys)
  409 tcValuePerms p tys es = tcValuePerms' p tys (reverse es)
  410 
  411 -- | Helper for 'tcValuePerms'
  412 tcValuePerms' :: Pos -> RAssign TypeRepr tps -> [AstExpr] -> Tc (RAssign ValuePerm tps)
  413 tcValuePerms' _ MNil [] = pure MNil
  414 tcValuePerms' p (xs :>: x) (y:ys) =
  415   do zs <- tcValuePerms' p xs ys
  416      z  <- tcValPerm x y
  417      pure (zs :>: z)
  418 tcValuePerms' p _ _ = tcError p "Bad arity"

  436 -- | Check an LLVM shape expression
  437 tcLLVMShape :: (KnownNat w, 1 <= w) => AstExpr -> Tc (PermExpr (LLVMShapeType w))
  438 tcLLVMShape (ExOrSh _ x y) = PExpr_OrShape <$> tcKExpr x <*> tcKExpr y
  439 tcLLVMShape (ExExSh _ var vartype sh) =
  440   do Some ktp'@KnownReprObj <- tcTypeKnown vartype
  441      fmap PExpr_ExShape $ mbM $ nu \z ->
  442        withExprVar var (unKnownReprObj ktp') z (tcKExpr sh)
  443 tcLLVMShape (ExSeqSh _ x y) = PExpr_SeqShape <$> tcKExpr x <*> tcKExpr y
  444 tcLLVMShape ExEmptySh{} = pure PExpr_EmptyShape
  445 tcLLVMShape (ExEqSh _ len v) = PExpr_EqShape <$> tcKExpr len <*> tcKExpr v
  446 tcLLVMShape (ExPtrSh _ maybe_l maybe_rw sh) =
  447   PExpr_PtrShape
  448   <$> traverse tcKExpr maybe_l
  449   <*> traverse tcKExpr maybe_rw
  450   <*> tcKExpr sh
  451 tcLLVMShape (ExFieldSh _ w fld) = PExpr_FieldShape <$> tcLLVMFieldShape_ w fld
  452 tcLLVMShape (ExArraySh _ len stride sh) =
  453   PExpr_ArrayShape
  454   <$> tcKExpr len
  455   <*> (Bytes . fromIntegral <$> tcNatural stride)
  456   <*> tcKExpr sh
  457 tcLLVMShape (ExTupleSh _ sh) = PExpr_TupShape <$> tcKExpr sh
  458 tcLLVMShape (ExFalseSh _) = pure PExpr_FalseShape
  459 tcLLVMShape e = tcError (pos e) "Expected shape"
  460 
  461 -- | Field and array helper for 'tcLLVMShape'
  462 tcLLVMFieldShape_ ::
  463   forall w. (KnownNat w, 1 <= w) => Maybe AstExpr -> AstExpr -> Tc (LLVMFieldShape w)
  464 tcLLVMFieldShape_ Nothing e = tcLLVMFieldShape (knownNat :: NatRepr w) e
  465 tcLLVMFieldShape_ (Just w) e =
  466   do Some (Pair nr LeqProof) <- tcPositive w
  467      withKnownNat nr (tcLLVMFieldShape nr e)
  468 
  469 -- | Check a single field or array element shape
  470 tcLLVMFieldShape ::
  471   forall (w :: Nat) (v :: Nat).
  472   (KnownNat w, 1 <= w) =>
  473   NatRepr w -> AstExpr -> Tc (LLVMFieldShape v)
  474 tcLLVMFieldShape nr e = LLVMFieldShape <$> tcValPerm (LLVMPointerRepr nr) e
  475 
  476 -- | Check a LLVM pointer expression
  477 tcLLVMPointer :: (KnownNat w, 1 <= w) => NatRepr w -> AstExpr -> Tc (PermExpr (LLVMPointerType w))
  478 tcLLVMPointer _ (ExLlvmWord _ e) = PExpr_LLVMWord <$> tcKExpr e
  479 tcLLVMPointer w (ExAdd _ (ExVar p name Nothing Nothing) off) = PExpr_LLVMOffset <$> tcVar (LLVMPointerRepr w) p name <*> tcKExpr off
  480 tcLLVMPointer _ e = tcError (pos e) "Expected llvmpointer"

  482 -- | Check a value permission of a known type in a given context
  483 tcValPermInCtx :: ParsedCtx ctx -> TypeRepr a -> AstExpr -> Tc (Mb ctx (ValuePerm a))
  484 tcValPermInCtx ctx tp = inParsedCtxM ctx . const . tcValPerm tp
  485 
  486 -- | Parse a value permission of a known type
  487 tcValPerm :: TypeRepr a -> AstExpr -> Tc (ValuePerm a)
  488 tcValPerm _  ExTrue{} = pure ValPerm_True
  489 tcValPerm _  ExFalse{} = pure ValPerm_False
  490 tcValPerm ty (ExOr _ x y) = ValPerm_Or <$> tcValPerm ty x <*> tcValPerm ty y
  491 tcValPerm ty (ExEq _ e) = ValPerm_Eq <$> tcExpr ty e
  492 tcValPerm ty (ExExists _ var vartype e) =
  493   do Some ktp'@KnownReprObj <- tcTypeKnown vartype
  494      fmap ValPerm_Exists $ mbM $ nu \z ->
  495        withExprVar var (unKnownReprObj ktp') z (tcValPerm ty e)
  496 tcValPerm ty (ExVar p n (Just argEs) maybe_off) =
  497   do env <- tcEnvPermEnv <$> tcAsk
  498      case lookupNamedPermName env n of
  499        Just (SomeNamedPermName rpn)
  500          | Just Refl <- testEquality (namedPermNameType rpn) ty ->
  501            do args <- tcExprs p (namedPermNameArgs rpn) argEs
  502               off <- tcPermOffset ty p maybe_off
  503               pure (ValPerm_Named rpn args off)
  504        Just (SomeNamedPermName rpn) ->
  505          tcError p $ renderDoc $ sep
  506          [ pretty "Named permission" <+> pretty n <+>
  507            pretty "is of incorrect type"
  508          , pretty "Expected:" <+> permPretty emptyPPInfo ty
  509          , pretty "Found:" <+>
  510            permPretty emptyPPInfo (namedPermNameType rpn) ]
  511        Nothing ->
  512          tcError p ("Unknown named permission '" ++ n ++ "'")
  513 tcValPerm ty (ExVar p n Nothing off) =
  514   ValPerm_Var <$> tcVar (ValuePermRepr ty) p n <*> tcPermOffset ty p off
  515 tcValPerm ty e = ValPerm_Conj <$> tcAtomicPerms ty e
  516 
  517 -- | Parse a @*@-separated list of atomic permissions
  518 tcAtomicPerms :: TypeRepr a -> AstExpr -> Tc [AtomicPerm a]
  519 tcAtomicPerms ty (ExMul _ x y) = (++) <$> tcAtomicPerms ty x <*> tcAtomicPerms ty y
  520 tcAtomicPerms ty e = pure <$> tcAtomicPerm ty e
  521 
  522 -- | Parse an atomic permission of a specific type
  523 tcAtomicPerm :: TypeRepr a -> AstExpr -> Tc (AtomicPerm a)
  524 tcAtomicPerm ty (ExVar p n (Just argEs) maybe_off) =
  525   do env <- tcEnvPermEnv <$> tcAsk
  526      case lookupNamedPermName env n of
  527        Just (SomeNamedPermName npn)
  528          | Just Refl <- testEquality (namedPermNameType npn) ty
  529          , TrueRepr <- nameIsConjRepr npn ->
  530              do args <- tcExprs p (namedPermNameArgs npn) argEs
  531                 off <- tcPermOffset ty p maybe_off
  532                 return (Perm_NamedConj npn args off)
  533        Just (SomeNamedPermName npn)
  534          | Just Refl <- testEquality (namedPermNameType npn) ty ->
  535          tcError p ("Non-conjoinable permission name '" ++ n
  536                   ++ "' used in conjunctive context")
  537        Just (SomeNamedPermName _) ->
  538          tcError p ("Permission name '" ++ n ++ "' has incorrect type")
  539        Nothing ->
  540          tcError p ("Unknown permission name '" ++ n ++ "'")
  541 tcAtomicPerm (LLVMPointerRepr w) e = withKnownNat w (tcPointerAtomic e)
  542 tcAtomicPerm (LLVMFrameRepr w) e = withKnownNat w (tcFrameAtomic e)
  543 tcAtomicPerm (LLVMBlockRepr w) e = withKnownNat w (tcBlockAtomic e)
  544 tcAtomicPerm (StructRepr tys) e = tcStructAtomic tys e
  545 tcAtomicPerm LifetimeRepr e = tcLifetimeAtomic e
  546 tcAtomicPerm _ (ExAny _) = return Perm_Any
  547 tcAtomicPerm _ e = tcError (pos e) "Expected perm"
  548 
  549 -- | Build a field permission using an 'LLVMFieldShape'
  550 fieldPermFromShape :: (KnownNat w, 1 <= w) => PermExpr RWModalityType ->
  551                       PermExpr LifetimeType -> PermExpr (BVType w) ->
  552                       LLVMFieldShape w -> AtomicPerm (LLVMPointerType w)
  553 fieldPermFromShape rw l off (LLVMFieldShape p) =
  554   Perm_LLVMField $ LLVMFieldPerm rw l off p

  556 -- | Check an LLVM pointer atomic permission expression
  557 tcPointerAtomic :: (KnownNat w, 1 <= w) => AstExpr -> Tc (AtomicPerm (LLVMPointerType w))
  558 tcPointerAtomic (ExPtr _ l rw off sz c) =
  559   fieldPermFromShape
  560   <$> tcKExpr rw
  561   <*> tcOptLifetime l
  562   <*> tcKExpr off
  563   <*> tcLLVMFieldShape_ sz c
  564 tcPointerAtomic (ExArray _ l rw off len stride sh) =
  565   Perm_LLVMArray <$> tcArrayAtomic l rw off len stride sh
  566 tcPointerAtomic (ExMemblock _ l rw off len sh) = Perm_LLVMBlock <$> tcMemblock l rw off len sh
  567 tcPointerAtomic (ExFree      _ x  ) = Perm_LLVMFree <$> tcKExpr x
  568 tcPointerAtomic (ExLlvmFunPtr _ n w f) = tcFunPtrAtomic n w f
  569 tcPointerAtomic (ExEqual     _ x y) = Perm_BVProp <$> (BVProp_Eq   <$> tcKExpr x <*> tcKExpr y)
  570 tcPointerAtomic (ExNotEqual  _ x y) = Perm_BVProp <$> (BVProp_Neq  <$> tcKExpr x <*> tcKExpr y)
  571 tcPointerAtomic (ExLessThan  _ x y) = Perm_BVProp <$> (BVProp_ULt  <$> tcKExpr x <*> tcKExpr y)
  572 tcPointerAtomic (ExLessEqual _ x y) = Perm_BVProp <$> (BVProp_ULeq <$> tcKExpr x <*> tcKExpr y)
  573 tcPointerAtomic e = tcError (pos e) "Expected pointer perm"
  574 
  575 -- | Check a function pointer permission literal
  576 tcFunPtrAtomic ::
  577   (KnownNat w, 1 <= w) =>
  578   AstExpr -> AstExpr -> AstFunPerm -> Tc (AtomicPerm (LLVMPointerType w))
  579 tcFunPtrAtomic x y fun =
  580   do Some args_no            <- mkNatRepr <$> tcNatural x
  581      Some (Pair w' LeqProof) <- tcPositive y
  582      Some args               <- pure (cruCtxReplicate args_no (LLVMPointerRepr w'))
  583      SomeFunPerm fun_perm    <- tcFunPerm args (LLVMPointerRepr w') fun
  584      pure (mkPermLLVMFunPtr knownNat fun_perm)

  586 -- | Check a memblock permission literal
  587 tcMemblock ::
  588   (KnownNat w, 1 <= w) =>
  589   Maybe AstExpr ->
  590   AstExpr -> AstExpr -> AstExpr -> AstExpr -> Tc (LLVMBlockPerm w)
  591 tcMemblock l rw off len sh =
  592   do llvmBlockLifetime <- tcOptLifetime l
  593      llvmBlockRW       <- tcKExpr rw
  594      llvmBlockOffset   <- tcKExpr off
  595      llvmBlockLen      <- tcKExpr len
  596      llvmBlockShape    <- tcKExpr sh
  597      pure LLVMBlockPerm{..}

  599 -- | Check an atomic array permission literal
  600 tcArrayAtomic ::
  601   (KnownNat w, 1 <= w) => Maybe AstExpr -> AstExpr -> AstExpr -> AstExpr ->
  602   AstExpr -> AstExpr -> Tc (LLVMArrayPerm w)
  603 tcArrayAtomic l rw off len stride sh =
  604   LLVMArrayPerm
  605   <$> tcKExpr rw
  606   <*> tcOptLifetime l
  607   <*> tcKExpr off
  608   <*> tcKExpr len
  609   <*> (Bytes . fromIntegral <$> tcNatural stride)
  610   <*> tcKExpr sh
  611   <*> pure []

  629 -- | Check a lifetime permission literal
  630 tcLifetimeAtomic :: AstExpr -> Tc (AtomicPerm LifetimeType)
  631 tcLifetimeAtomic (ExLOwned _ ls ps_in ps_out) =
  632   do Some ps_in' <- tcDistPerms ps_in
  633      Some ps_out' <- tcDistPerms ps_out
  634      ls' <- mapM tcKExpr ls
  635      let eps_in = distPermsToExprPerms $ unTypeDistPerms ps_in'
  636      let eps_out = distPermsToExprPerms $ unTypeDistPerms ps_out'
  637      pure (Perm_LOwned ls' (typedDistPermsCtx ps_in')
  638            (typedDistPermsCtx ps_out') eps_in eps_out)
  639 tcLifetimeAtomic (ExLCurrent _ l) = Perm_LCurrent <$> tcOptLifetime l
  640 tcLifetimeAtomic (ExLFinished _) = return Perm_LFinished
  641 tcLifetimeAtomic e = tcError (pos e) "Expected lifetime perm"

  643 -- | Check a sequence @x1:p1, ..., xn:pn@ of variables and permissions
  644 tcDistPerms :: [(Located String,AstExpr)] -> Tc (Some TypedDistPerms)
  645 tcDistPerms [] = pure (Some MNil)
  646 tcDistPerms ((Located p n,e):xs) =
  647   do Some (Typed tp x) <- tcTypedName p n
  648      perm <- tcValPerm tp e
  649      Some ps <- tcDistPerms xs
  650      pure (Some (ps :>: Typed tp (VarAndPerm x perm)))

  653 tcPermOffset :: TypeRepr a -> Pos -> Maybe AstExpr -> Tc (PermOffset a)
  654 tcPermOffset _ _ Nothing = pure NoPermOffset
  655 tcPermOffset (LLVMPointerRepr w) _ (Just i) = withKnownNat w (LLVMPermOffset <$> tcKExpr i)

  658 -- | Check for a number literal
  659 tcNatural :: AstExpr -> Tc Natural
  660 tcNatural (ExNat _ i) = pure i
  661 tcNatural e = tcError (pos e) "Expected integer literal"

  663 -- | Ensure a natural nubmer is positive
  664 withPositive ::
  665   Pos                   {- ^ location of literal        -} ->
  666   String                {- ^ error message              -} ->
  667   Natural               {- ^ number                     -} ->
  668   (forall w. (1 <= w, KnownNat w) => NatRepr w -> Tc a)
  669                         {- ^ continuation               -} ->
  670   Tc a
  671 withPositive p err n k =
  672   case someNatGeq1 n of
  673     Nothing -> tcError p err
  674     Just (Some (Pair w LeqProof)) -> withKnownNat w (k w)
  675 
  676 -- | Check for a positive number literal
  677 tcPositive :: AstExpr -> Tc (Some (Product NatRepr (LeqProof 1)))
  678 tcPositive e =
  679   do i <- tcNatural e
  680      withPositive (pos e) "positive required" i \w -> pure (Some (Pair w LeqProof))

  682 -- | Check a typing context @x1:tp1, x2:tp2, ...@
  683 tcCtx :: [(Located String, AstType)] -> Tc (Some ParsedCtx)
  684 tcCtx []         = pure (Some emptyParsedCtx)
  685 tcCtx ((n,t):xs) = preconsSomeParsedCtx (locThing n) <$> tcType t <*> tcCtx xs
  686 
  687 -- | Check a sequence @x1:p1, x2:p2, ...@ of variables and their permissions,
  688 -- where each variable occurs at most once. The input list says which variables
  689 -- can occur and which have already been seen. Return a sequence of the
  690 -- permissions in the same order as the input list of variables.
  691 tcSortedValuePerms ::
  692   VarPermSpecs ctx -> [(Located String, AstExpr)] -> Tc (ValuePerms ctx)
  693 tcSortedValuePerms var_specs [] = pure (varSpecsToPerms var_specs)
  694 tcSortedValuePerms var_specs ((Located p var, x):xs) =
  695   do Some (Typed tp n) <- tcTypedName p var
  696      perm              <- tcValPerm tp x
  697      var_specs'        <- tcSetVarSpecs p var n perm var_specs
  698      tcSortedValuePerms var_specs' xs
  699 
  700 -- | Check a sequence @x1:p1, x2:p2, ...@ of variables and their permissions,
  701 -- and sort the result into a 'ValuePerms' in a multi-binding that is in the
  702 -- same order as the 'ParsedCtx' supplied on input
  703 tcSortedMbValuePerms ::
  704   ParsedCtx ctx -> [(Located String, AstExpr)] -> Tc (MbValuePerms ctx)
  705 tcSortedMbValuePerms ctx perms =
  706   inParsedCtxM ctx \ns ->
  707   tcSortedValuePerms (mkVarPermSpecs ns) perms
  708 
  709 -- | Check a function permission of the form
  710 --
  711 -- > (x1:tp1, ...). arg1:p1, ... -o
  712 -- >   (y1:tp1', ..., ym:tpm'). arg1:p1', ..., argn:pn', ret:p_ret
  713 --
  714 -- for some arbitrary context @x1:tp1, ...@ of ghost variables
  715 tcFunPerm :: CruCtx args -> TypeRepr ret -> AstFunPerm -> Tc (SomeFunPerm args ret)
  716 tcFunPerm args ret (AstFunPerm _ untyCtx ins untyCtxOut outs) =
  717   do Some ghosts_ctx@(ParsedCtx _ ghosts) <- tcCtx untyCtx
  718      Some gouts_ctx@(ParsedCtx _ gouts) <- tcCtx untyCtxOut
  719      let args_ctx = mkArgsParsedCtx args
  720          perms_in_ctx = appendParsedCtx ghosts_ctx args_ctx
  721          perms_out_ctx =
  722            appendParsedCtx (appendParsedCtx ghosts_ctx args_ctx)
  723            (consParsedCtx "ret" ret gouts_ctx)
  724      perms_in  <- tcSortedMbValuePerms perms_in_ctx ins
  725      perms_out <- tcSortedMbValuePerms perms_out_ctx outs
  726      pure (SomeFunPerm (FunPerm ghosts args gouts ret perms_in perms_out))

  732 -- | Helper type for 'parseValuePerms' that represents whether a pair @x:p@ has
  733 -- been parsed yet for a specific variable @x@ and, if so, contains that @p@
  734 data VarPermSpec a = VarPermSpec (Name a) (Maybe (ValuePerm a))
  735 
  736 -- | A sequence of variables @x@ and what pairs @x:p@ have been parsed so far
  737 type VarPermSpecs = RAssign VarPermSpec
  738 
  739 -- | Build a 'VarPermSpecs' from a list of names
  740 mkVarPermSpecs :: RAssign Name ctx -> VarPermSpecs ctx
  741 mkVarPermSpecs = RL.map (\n -> VarPermSpec n Nothing)
  742 
  743 -- | Find a 'VarPermSpec' for a particular variable
  744 findVarPermSpec :: Name (a :: CrucibleType) ->
  745                    VarPermSpecs ctx -> Maybe (Member ctx a)
  746 findVarPermSpec _ MNil = Nothing
  747 findVarPermSpec n (_ :>: VarPermSpec n' _)
  748   | Just Refl <- testEquality n n'
  749   = Just Member_Base
  750 findVarPermSpec n (specs :>: _) = Member_Step <$> findVarPermSpec n specs
  751 
  752 -- | Try to set the permission for a variable in a 'VarPermSpecs' list, raising
  753 -- a parse error if the variable already has a permission or is one of the
  754 -- expected variables
  755 tcSetVarSpecs ::
  756   Pos -> String -> Name tp -> ValuePerm tp -> VarPermSpecs ctx ->
  757   Tc (VarPermSpecs ctx)
  758 tcSetVarSpecs p var n perm var_specs =
  759   case findVarPermSpec n var_specs of
  760     Nothing -> tcError p ("Unknown variable: " ++ var)
  761     Just memb ->
  762       case RL.get memb var_specs of
  763         VarPermSpec _ Nothing ->
  764           pure (RL.modify memb (const (VarPermSpec n (Just perm))) var_specs)
  765         _ -> tcError p ("Variable " ++ var ++ " occurs more than once!")
  766 
  767 -- | Convert a 'VarPermSpecs' sequence to a sequence of permissions, using the
  768 -- @true@ permission for any variables without permissions
  769 varSpecsToPerms :: VarPermSpecs ctx -> ValuePerms ctx
  770 varSpecsToPerms MNil = ValPerms_Nil
  771 varSpecsToPerms (var_specs :>: VarPermSpec _ (Just p)) =
  772   ValPerms_Cons (varSpecsToPerms var_specs) p
  773 varSpecsToPerms (var_specs :>: VarPermSpec _ Nothing) =
  774   ValPerms_Cons (varSpecsToPerms var_specs) ValPerm_True

  776 -- | Run a parsing computation inside a name-binding for expressions variables
  777 -- given by a 'ParsedCtx'. Returning the results inside a name-binding.
  778 inParsedCtxM ::
  779   NuMatching a =>
  780   ParsedCtx ctx -> (RAssign Name ctx -> Tc a) -> Tc (Mb ctx a)
  781 inParsedCtxM (ParsedCtx ids tps) f =
  782   mbM (nuMulti (cruCtxProxies tps) \ns -> withExprVars ids tps ns (f ns))
