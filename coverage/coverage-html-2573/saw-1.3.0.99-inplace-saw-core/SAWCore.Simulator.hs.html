   93 -- Simulator configuration

   95 data SimulatorConfig l =
   96   SimulatorConfig
   97   { simPrimitive :: ExtCns (TValue l) -> MValue l
   98   -- ^ Interpretation of 'Primitive' terms.
   99   , simExtCns :: TermF Term -> ExtCns (TValue l) -> MValue l
  100   -- ^ Interpretation of 'ExtCns' terms.
  101   , simConstant :: TermF Term -> ExtCns (TValue l) -> Maybe (MValue l)
  102   -- ^ Interpretation of 'Constant' terms. 'Nothing' indicates that
  103   -- the body of the constant should be evaluated. 'Just' indicates
  104   -- that the constant's definition should be overridden.
  105   , simCtorApp :: ExtCns (TValue l) -> Maybe (MValue l)
  106   -- ^ Interpretation of constructor terms. 'Nothing' indicates that
  107   -- the constructor is treated as normal. 'Just' replaces the
  108   -- constructor with a custom implementation.
  109   , simNeutral :: Env l -> NeutralTerm -> MValue l
  110   -- ^ Handler that fires if the simulator encounters a term that it
  111   --   cannot otherwise evaluate because it is blocked. For some simulators,
  112   --   this is just an error condition; for others, sensible action can
  113   --   be taken.
  114   , simModMap :: ModuleMap
  115   , simLazyMux :: TValue l -> VBool l -> MValue l -> MValue l -> MValue l
  116   }

  136   evalTermF :: Show (Extra l) =>

  143 -- | Generic evaluator for TermFs.
  144 evalTermF :: forall l. (VMonadLazy l, Show (Extra l)) =>
  145              SimulatorConfig l          -- ^ Evaluator for global constants
  146           -> (Term -> OpenValue l)      -- ^ Evaluator for subterms under binders
  147           -> (Term -> MValue l)         -- ^ Evaluator for subterms in the same bound variable context
  148           -> TermF Term -> OpenValue l
  149 evalTermF cfg lam recEval tf env =
  150   case tf of
  151     App t1 t2               -> recEval t1 >>= \case
  152                                  VFun _ f ->
  153                                    do x <- recEvalDelay t2
  154                                       f x
  155                                  _ -> simNeutral cfg env (NeutralApp (NeutralBox t1) t2)
  156     Lambda nm tp t          -> do v <- evalType tp
  157                                   return $ VFun nm (\x -> lam t ((x,v) : env))
  158     Pi nm t1 t2             -> do v <- evalType t1
  159                                   body <-
  160                                     if inBitSet 0 (looseVars t2) then
  161                                       pure (VDependentPi (\x -> toTValue <$> lam t2 ((x,v) : env)))
  162                                     else
  163                                       do -- put dummy values in the environment; the term should never reference them
  164                                          let val = ready (VTuple mempty)
  165                                          let tp  = VTupleType mempty
  166                                          VNondependentPi . toTValue <$> lam t2 ((val,tp):env)
  167                                   return $ TValue $ VPiType nm v body
  168 
  169     LocalVar i              -> force (fst (env !! i))
  170 
  171     Constant nm             -> do let r = requireNameInMap nm (simModMap cfg)
  172                                   ty' <- evalType (resolvedNameType r)
  173                                   let ec' = EC nm ty'
  174                                   case simConstant cfg tf ec' of
  175                                     Just override -> override
  176                                     Nothing ->
  177                                       case r of
  178                                         ResolvedCtor ctor ->
  179                                           ctorValue ec' (ctorNumParams ctor) (ctorNumArgs ctor)
  180                                         ResolvedDataType dt ->
  181                                           dtValue ec' (dtNumParams dt) (dtNumIndices dt)
  182                                         ResolvedDef d ->
  183                                           case defBody d of
  184                                             Just t -> recEval t
  185                                             Nothing ->
  186                                               case defQualifier d of
  187                                                 NoQualifier -> simNeutral cfg env (NeutralConstant nm)
  188                                                 PrimQualifier -> simPrimitive cfg ec'
  189                                                 AxiomQualifier -> simPrimitive cfg ec'
  190 
  191     Variable ec             -> do ec' <- traverse evalType ec
  192                                   simExtCns cfg tf ec'
  193     FTermF ftf              ->
  194       case ftf of
  195         TupleValue xs       -> VTuple <$> traverse recEvalDelay xs
  196 
  197         TupleSelector x i   -> recEval x >>= \case
  198                                  VTuple ys -> force (ys V.! i)
  199                                  _ -> simNeutral cfg env (NeutralTupleProj (NeutralBox x) i)
  200 
  201         RecursorType d ps m mtp ->
  202           do dty <- evalType (resolvedNameType (requireNameInMap d (simModMap cfg)))
  203              TValue <$> (VRecursorType <$>
  204                pure (EC d dty) <*>
  205                mapM recEval ps <*>
  206                recEval m <*>
  207                (evalType mtp))
  208 
  209         Recursor r ->
  210           do let f (e,ety) = do v  <- recEvalDelay e
  211                                 ty <- evalType ety
  212                                 pure (v,ty)
  213              let dname = recursorDataType r
  214              dty <- evalType (resolvedNameType (requireNameInMap dname (simModMap cfg)))
  215              let d = EC dname dty
  216              ps  <- traverse recEval (recursorParams r)
  217              m   <- recEval (recursorMotive r)
  218              mty <- evalType (recursorMotiveTy r)
  219              es  <- traverse f (recursorElims r)
  220              pure (VRecursor d ps m mty es)
  221 
  222         RecursorApp rectm ixs arg ->
  223           do r <- recEval rectm
  224              case r of
  225                VRecursor d ps motive motiveTy ps_fs ->
  226                  do argv <- recEval arg
  227                     case evalConstructor argv of
  228                       Just (ctor, args)
  229                         | Just (elim,elimTy) <- Map.lookup (nameIndex (ctorName ctor)) ps_fs
  230                         -> do let rTy = VRecursorType d ps motive motiveTy
  231                               ctorTy <- toTValue <$> lam (ctorType ctor) []
  232                               allArgs <- processRecArgs ps args ctorTy [(elim,elimTy),(ready r,rTy)]
  233                               lam (ctorIotaTemplate ctor) allArgs
  234 
  235                         | otherwise -> panic "evalTermF / RecursorApp" ["could not find info for constructor: " <> Text.pack (show ctor)]
  236                       Nothing ->
  237                         case argv of
  238                           VCtorMux _ps branches ->
  239                             do alts <- traverse (evalCtorMuxBranch r) (IntMap.elems branches)
  240                                -- compute return type of recursor application
  241                                ixvs <- traverse recEval ixs
  242                                retTy <- toTValue <$> applyAll motive (map ready (ixvs ++ [argv]))
  243                                combineAlts retTy alts
  244                           _ -> simNeutral cfg env (NeutralRecursorArg rectm ixs (NeutralBox arg))
  245                _ -> simNeutral cfg env (NeutralRecursor (NeutralBox rectm) ixs arg)
  246 
  247         RecordType elem_tps ->
  248           TValue . VRecordType <$> traverse (traverse evalType) elem_tps
  249 
  250         RecordValue elems   ->
  251           VRecordValue <$> mapM (\(fld,t) -> (fld,) <$> recEvalDelay t) elems
  252 
  253         RecordProj t fld    -> recEval t >>= \case
  254                                  v@VRecordValue{} -> valRecordProj v fld
  255                                  _ -> simNeutral cfg env (NeutralRecordProj (NeutralBox t) fld)
  256 
  257         Sort s _h           -> return $ TValue (VSort s)
  258 
  259         NatLit n            -> return $ VNat n
  260 
  261         ArrayValue _ tv     -> liftM VVector $ mapM recEvalDelay tv
  262 
  263         StringLit s         -> return $ VString s
  264   where
  265     evalType :: Term -> EvalM l (TValue l)
  266     evalType t = toTValue <$> recEval t
  267 
  268     toTValue :: HasCallStack => Value l -> TValue l
  269     toTValue (TValue x) = x
  270     toTValue t = panic "evalTermF / toTValue" ["Not a type value: " <> Text.pack (show t)]
  271 
  272     evalCtorMuxBranch ::
  273       Value l ->
  274       (VBool l, ExtCns (TValue l), [Thunk l]) ->
  275       EvalM l (VBool l, EvalM l (Value l))
  276     evalCtorMuxBranch r (p, c, args) =
  277       case r of
  278         VRecursor d ps motive motiveTy ps_fs ->
  279           do let i = ecVarIndex c
  280              let rTy = VRecursorType d ps motive motiveTy
  281              case (lookupVarIndexInMap i (simModMap cfg), Map.lookup i ps_fs) of
  282                (Just (ResolvedCtor ctor), Just (elim, elimTy)) ->
  283                  do allArgs <- processRecArgs ps args (ecType c) [(elim, elimTy), (ready r, rTy)]
  284                     pure (p, lam (ctorIotaTemplate ctor) allArgs)
  285                _ -> panic "evalTermF / evalCtorMuxBranch"
  286                     ["could not find info for constructor: " <> toAbsoluteName (ecNameInfo c)]
  287         _ -> panic "evalTermF / evalCtorMuxBranch" ["expected VRecursor"]
  288 
  289     combineAlts :: TValue l -> [(VBool l, EvalM l (Value l))] -> EvalM l (Value l)
  290     combineAlts _ [] = panic "evalTermF / combineAlts" ["no alternatives"]
  291     combineAlts _ [(_, x)] = x
  292     combineAlts tp ((p, x) : alts) = simLazyMux cfg tp p x (combineAlts tp alts)
  293 
  294     evalConstructor :: Value l -> Maybe (Ctor, [Thunk l])
  295     evalConstructor (VCtorApp c _ps args) =
  296       case lookupVarIndexInMap (ecVarIndex c) (simModMap cfg) of
  297         Just (ResolvedCtor ctor) -> Just (ctor, args)
  298         _ -> Nothing
  299     evalConstructor (VNat 0) =
  300        do ctor <- findCtorInMap preludeZeroIdent (simModMap cfg)
  301           Just (ctor, [])
  302     evalConstructor (VNat n) =
  303        do ctor <- findCtorInMap preludeSuccIdent (simModMap cfg)
  304           Just (ctor, [ ready (VNat (pred n)) ])
  305     evalConstructor _ =
  306        Nothing
  307 
  308     recEvalDelay :: Term -> EvalM l (Thunk l)
  309     recEvalDelay = delay . recEval
  310 
  311     ctorValue :: ExtCns (TValue l) -> Int -> Int -> MValue l
  312     ctorValue ec i j =
  313       vFunList (replicate i "_") $ \params ->
  314       vFunList (replicate j "_") $ \args ->
  315       pure $ VCtorApp ec params args
  316 
  317     dtValue :: ExtCns (TValue l) -> Int -> Int -> MValue l
  318     dtValue ec i j =
  319       vStrictFunList (replicate i "_") $ \params ->
  320       vStrictFunList (replicate j "_") $ \idxs ->
  321       pure $ TValue $ VDataType ec params idxs

  323 -- | Create a 'Value' for a lazy multi-argument function.
  324 vFunList :: forall l. VMonad l => [LocalName] -> ([Thunk l] -> MValue l) -> MValue l
  325 vFunList names k = go [] names
  326   where
  327     go :: [Thunk l] -> [LocalName] -> MValue l
  328     go args [] = k (reverse args)
  329     go args (n : ns) = pure $ VFun n (\x -> go (x : args) ns)
  330 
  331 -- | Create a 'Value' for a strict multi-argument function.
  332 vStrictFunList :: forall l. VMonad l => [LocalName] -> ([Value l] -> MValue l) -> MValue l
  333 vStrictFunList names k = go [] names
  334   where
  335     go :: [Value l] -> [LocalName] -> MValue l
  336     go args [] = k (reverse args)
  337     go args (n : ns) = pure $ VFun n (\x -> force x >>= \v -> go (v : args) ns)

  339 processRecArgs ::
  340   (VMonadLazy l, Show (Extra l)) =>
  341   [Value l] ->
  342   [Thunk l] ->
  343   TValue l ->
  344   Env l ->
  345   EvalM l (Env l)
  346 processRecArgs (p:ps) args (VPiType _ _ body) env =
  347   do tp' <- applyPiBody body (ready p)
  348      processRecArgs ps args tp' env
  349 processRecArgs [] (x:xs) (VPiType _ tp body) env =
  350   do tp' <- applyPiBody body x
  351      processRecArgs [] xs tp' ((x,tp):env)
  352 processRecArgs [] [] _ env = pure env
  353 processRecArgs _ _ ty _ =
  354   panic "processRegArgs" [
  355      "Expected Pi type!",
  356      "Found: " <> Text.pack (show ty)

  412 -- | A variant of 'evalGlobal' that lets the uninterpreted function
  413 -- symbol and external-constant callbacks have access to the 'TermF'.
  414 evalGlobal' ::
  415   forall l. (VMonadLazy l, Show (Extra l)) =>
  416   ModuleMap ->
  417   -- | Implementations of 'Primitive' terms, plus overrides for 'Constant' and 'CtorApp' terms
  418   Map Ident (Prims.Prim l) ->
  419   -- | Implementations of ExtCns terms
  420   (TermF Term -> ExtCns (TValue l) -> MValue l) ->
  421   -- | Overrides for Constant terms (e.g. uninterpreted functions)
  422   (TermF Term -> ExtCns (TValue l) -> Maybe (MValue l)) ->
  423   -- | Handler for neutral terms
  424   (Env l -> NeutralTerm -> MValue l) ->
  425   -- | Handler for stuck primitives
  426   (ExtCns (TValue l) -> Text -> Env l -> TValue l -> MValue l) ->
  427   -- | Lazy mux operation
  428   (TValue l -> VBool l -> MValue l -> MValue l -> MValue l) ->
  429   EvalM l (SimulatorConfig l)
  430 evalGlobal' modmap prims extcns constant neutral primHandler lazymux =
  431   do checkPrimitives modmap prims
  432      return (SimulatorConfig primitive extcns constant' ctors neutral modmap lazymux)
  433   where
  434     constant' :: TermF Term -> ExtCns (TValue l) -> Maybe (MValue l)
  435     constant' tf ec =
  436       case constant tf ec of
  437         Just v -> Just v
  438         Nothing ->
  439           case ecNameInfo ec of
  440             ModuleIdentifier ident ->
  441               evalPrim (primHandler ec) ec <$> Map.lookup ident prims
  442             ImportedName{} -> Nothing
  443 
  444     ctors :: ExtCns (TValue l) -> Maybe (MValue l)
  445     ctors ec =
  446       case ecNameInfo ec of
  447         ModuleIdentifier ident ->
  448           evalPrim (primHandler ec) ec <$> Map.lookup ident prims
  449         ImportedName{} -> Nothing
  450 
  451     primitive :: ExtCns (TValue l) -> MValue l
  452     primitive ec =
  453       case ecNameInfo ec of
  454         ImportedName {} ->
  455           panic "evalGlobal'" ["Unimplemented global: " <> toAbsoluteName (ecNameInfo ec)]
  456         ModuleIdentifier ident ->
  457           case Map.lookup ident prims of
  458             Just v  -> evalPrim (primHandler ec) ec v
  459             Nothing -> panic "evalGlobal'" ["Unimplemented global: " <> identText ident]

  461 -- | Check that all the primitives declared in the given module
  462 --   are implemented, and that terms with implementations are not
  463 --   overridden.
  464 checkPrimitives :: forall l. (VMonadLazy l, Show (Extra l))
  465                 => ModuleMap
  466                 -> Map Ident (Prims.Prim l)
  467                 -> EvalM l ()
  468 checkPrimitives modmap prims = do
  469    -- FIXME this is downgraded to a warning temporarily while we work out a
  470    -- solution to issue GaloisInc/saw-script#48
  471    --   when (not $ null unimplementedPrims) (panic $ unimplementedMsg)
  472    -- (if null unimplementedPrims then id else Debug.trace (unimplementedMsg++"\n")) $
  473 --   (if null overridePrims then id else Debug.trace (overrideMsg++"\n")) $
  474      return ()
  475 
  476   where _unimplementedMsg = unwords $
  477             ("WARNING unimplemented primitives:" : (map show unimplementedPrims))
  478         _overrideMsg = unwords $
  479             ("WARNING overridden definitions:" : (map show overridePrims))
  480 
  481         primSet = Set.fromList $ mapMaybe defIdent $ allModulePrimitives modmap
  482         defSet  = Set.fromList $ mapMaybe defIdent $ allModuleActualDefs modmap
  483         implementedPrims = Map.keysSet prims
  484 
  485         unimplementedPrims = Set.toList $ Set.difference primSet implementedPrims
  486         overridePrims = Set.toList $ Set.intersection defSet implementedPrims

  495 -- The evaluation strategy for SharedTerms involves two memo tables:
  496 -- The first, @memoClosed@, is precomputed and contains the result of
  497 -- evaluating all _closed_ subterms. The same @memoClosed@ table is
  498 -- used for evaluation under lambdas, since the meaning of a closed
  499 -- term does not depend on the local variable context. The second memo
  500 -- table is @memoLocal@, which additionally includes the result of
  501 -- evaluating _open_ terms in the current variable context. It is
  502 -- reinitialized to @memoClosed@ whenever we descend under a lambda
  503 -- binder.
  504 
  505 {-# SPECIALIZE evalSharedTerm ::
  506   Show (Extra l) => SimulatorConfigIn Id l -> Term -> MValueIn Id l #-}
  507 {-# SPECIALIZE evalSharedTerm ::
  508   Show (Extra l) => SimulatorConfigIn IO l -> Term -> MValueIn IO l #-}
  509 
  510 -- | Evaluator for shared terms.
  511 evalSharedTerm :: (VMonadLazy l, MonadFix (EvalM l), Show (Extra l)) =>
  512                   SimulatorConfig l -> Term -> MValue l
  513 evalSharedTerm cfg t = do
  514   memoClosed <- mkMemoClosed cfg t
  515   evalOpen cfg memoClosed t []
  516 
  517 {-# SPECIALIZE mkMemoClosed ::
  518   Show (Extra l) =>
  519   SimulatorConfigIn Id l -> Term -> Id (IntMap (ThunkIn Id l)) #-}
  520 {-# SPECIALIZE mkMemoClosed ::
  521   Show (Extra l) =>
  522   SimulatorConfigIn IO l -> Term -> IO (IntMap (ThunkIn IO l)) #-}
  523 
  524 -- | Precomputing the memo table for closed subterms.
  525 mkMemoClosed :: forall l. (VMonadLazy l, MonadFix (EvalM l), Show (Extra l)) =>
  526                 SimulatorConfig l -> Term -> EvalM l (IntMap (Thunk l))
  527 mkMemoClosed cfg t =
  528   mfix $ \memoClosed -> mapM (delay . evalClosedTermF cfg memoClosed) subterms
  529   where
  530     -- | Map of all closed subterms of t.
  531     subterms :: IntMap (TermF Term)
  532     subterms = fmap fst $ IMap.filter ((== emptyBitSet) . snd) $ State.execState (go t) IMap.empty
  533 
  534     go :: Term -> State.State (IntMap (TermF Term, BitSet)) BitSet
  535     go (Unshared tf) = termf tf
  536     go (STApp{ stAppIndex = i, stAppTermF = tf }) =
  537       do memo <- State.get
  538          case IMap.lookup i memo of
  539            Just (_, b) -> pure b
  540            Nothing ->
  541              do b <- termf tf
  542                 State.modify (IMap.insert i (tf, b))
  543                 pure b
  544 
  545     termf :: TermF Term -> State.State (IntMap (TermF Term, BitSet)) BitSet
  546     termf tf =
  547       do -- if tf is a defined constant, traverse the definition body and type
  548          case tf of
  549            Constant nm ->
  550              do let r = requireNameInMap nm (simModMap cfg)
  551                 void $ go (resolvedNameType r)
  552                 case r of
  553                   ResolvedDef (defBody -> Just body) -> void $ go body
  554                   _ -> pure ()
  555            _ -> pure ()
  556          looseTermF <$> traverse go tf
  557 
  558 {-# SPECIALIZE evalClosedTermF ::
  559   Show (Extra l) =>
  560   SimulatorConfigIn Id l ->
  561   IntMap (ThunkIn Id l) ->
  562   TermF Term ->
  563   MValueIn Id l #-}
  564 {-# SPECIALIZE evalClosedTermF ::
  565   Show (Extra l) =>
  566   SimulatorConfigIn IO l ->
  567   IntMap (ThunkIn IO l) ->
  568   TermF Term ->
  569   MValueIn IO l #-}
  570 
  571 -- | Evaluator for closed terms, used to populate @memoClosed@.
  572 evalClosedTermF :: (VMonadLazy l, Show (Extra l)) =>
  573                    SimulatorConfig l
  574                 -> IntMap (Thunk l)
  575                 -> TermF Term -> MValue l
  576 evalClosedTermF cfg memoClosed tf = evalTermF cfg lam recEval tf []
  577   where
  578     lam = evalOpen cfg memoClosed
  579     recEval (Unshared tf') = evalTermF cfg lam recEval tf' []
  580     recEval (STApp{ stAppIndex = i }) =
  581       case IMap.lookup i memoClosed of
  582         Just x -> force x
  583         Nothing -> panic "evalClosedTermF" ["internal error"]
  584 
  585 {-# SPECIALIZE mkMemoLocal ::
  586   Show (Extra l) =>
  587   SimulatorConfigIn Id l ->
  588   IntMap (ThunkIn Id l) ->
  589   Term ->
  590   EnvIn Id l ->
  591   Id (IntMap (ThunkIn Id l)) #-}
  592 {-# SPECIALIZE mkMemoLocal ::
  593   Show (Extra l) =>
  594   SimulatorConfigIn IO l ->
  595   IntMap (ThunkIn IO l) ->
  596   Term ->
  597   EnvIn IO l ->
  598   IO (IntMap (ThunkIn IO l)) #-}
  599 
  600 -- | Precomputing the memo table for open subterms in the current context.
  601 mkMemoLocal :: forall l. (VMonadLazy l, Show (Extra l)) =>
  602                SimulatorConfig l -> IntMap (Thunk l) ->
  603                Term -> Env l -> EvalM l (IntMap (Thunk l))
  604 mkMemoLocal cfg memoClosed t env = go mempty t
  605   where
  606     go :: IntMap (Thunk l) -> Term -> EvalM l (IntMap (Thunk l))
  607     go memo (Unshared tf) = goTermF memo tf
  608     go memo (t'@STApp{ stAppIndex = i, stAppTermF = tf })
  609       | termIsClosed t' = pure memo
  610       | otherwise =
  611         case IMap.lookup i memo of
  612           Just _ -> pure memo
  613           Nothing ->
  614             do memo' <- goTermF memo tf
  615                thunk <- delay (evalLocalTermF cfg memoClosed memo' tf env)
  616                pure (IMap.insert i thunk memo')
  617     goTermF :: IntMap (Thunk l) -> TermF Term -> EvalM l (IntMap (Thunk l))
  618     goTermF memo tf =
  619       case tf of
  620         FTermF ftf      -> foldlM go memo ftf
  621         App t1 t2       -> do memo' <- goTermF memo (unwrapTermF t1)
  622                               go memo' t2
  623         Lambda _ t1 _   -> go memo t1
  624         Pi _ t1 _       -> go memo t1
  625         LocalVar _      -> pure memo
  626         Constant{}      -> pure memo
  627         Variable ec     -> go memo (ecType ec)
  628 
  629 {-# SPECIALIZE evalLocalTermF ::
  630   Show (Extra l) =>
  631   SimulatorConfigIn Id l ->
  632   IntMap (ThunkIn Id l) ->
  633   IntMap (ThunkIn Id l) ->
  634   TermF Term ->
  635   OpenValueIn Id l #-}
  636 {-# SPECIALIZE evalLocalTermF ::
  637   Show (Extra l) =>
  638   SimulatorConfigIn IO l ->
  639   IntMap (ThunkIn IO l) ->
  640   IntMap (ThunkIn IO l) ->
  641   TermF Term ->
  642   OpenValueIn IO l #-}
  643 -- | Evaluator for open terms, used to populate @memoLocal@.
  644 evalLocalTermF :: (VMonadLazy l, Show (Extra l)) =>
  645                    SimulatorConfig l
  646                 -> IntMap (Thunk l) -> IntMap (Thunk l)
  647                 -> TermF Term -> OpenValue l
  648 evalLocalTermF cfg memoClosed memoLocal tf0 env = evalTermF cfg lam recEval tf0 env
  649   where
  650     lam = evalOpen cfg memoClosed
  651     recEval (Unshared tf) = evalTermF cfg lam recEval tf env
  652     recEval (t@STApp{ stAppIndex = i, stAppTermF = tf }) =
  653       case IMap.lookup i memo of
  654         Just x -> force x
  655         Nothing -> evalTermF cfg lam recEval tf env
  656       where memo = if termIsClosed t then memoClosed else memoLocal
  657 
  658 {-# SPECIALIZE evalOpen ::
  659   Show (Extra l) =>
  660   SimulatorConfigIn Id l ->
  661   IntMap (ThunkIn Id l) ->
  662   Term ->
  663   OpenValueIn Id l #-}
  664 
  665 {-# SPECIALIZE evalOpen ::
  666   Show (Extra l) =>
  667   SimulatorConfigIn IO l ->
  668   IntMap (ThunkIn IO l) ->
  669   Term ->
  670   OpenValueIn IO l #-}
  671 
  672 -- | Evaluator for open terms; parameterized by a precomputed table @memoClosed@.
  673 evalOpen :: forall l. (VMonadLazy l, Show (Extra l)) =>
  674             SimulatorConfig l
  675          -> IntMap (Thunk l)
  676          -> Term -> OpenValue l
  677 evalOpen cfg memoClosed t env = do
  678   memoLocal <- mkMemoLocal cfg memoClosed t env
  679   let eval :: Term -> MValue l
  680       eval (Unshared tf) = evalF tf
  681       eval (t'@STApp{ stAppIndex = i, stAppTermF = tf }) =
  682         case IMap.lookup i memo of
  683           Just x -> force x
  684           Nothing -> evalF tf
  685         where memo = if termIsClosed t' then memoClosed else memoLocal
  686       evalF :: TermF Term -> MValue l
  687       evalF tf = evalTermF cfg (evalOpen cfg memoClosed) eval tf env
  688   eval t

  691 {-# SPECIALIZE evalPrim ::
  692   Show (Extra l) =>
  693   (Text -> EnvIn Id l -> TValue (WithM Id l) -> MValueIn Id l) ->
  694   ExtCns (TValue (WithM Id l)) ->
  695   PrimIn Id l ->
  696   MValueIn Id l
  697  #-}
  698 {-# SPECIALIZE evalPrim ::
  699   Show (Extra l) =>
  700   (Text -> EnvIn IO l -> TValue (WithM IO l) -> MValueIn IO l) ->
  701   ExtCns (TValue (WithM IO l)) ->
  702   PrimIn IO l ->
  703   MValueIn IO l
  704  #-}
  705 evalPrim :: forall l. (VMonadLazy l, Show (Extra l)) =>
  706   (Text -> Env l -> TValue l -> MValue l) ->
  707   ExtCns (TValue l) ->
  708   Prims.Prim l ->
  709   MValue l
  710 evalPrim fallback ec = loop [] (ecType ec)
  711   where
  712     loop :: Env l -> TValue l -> Prims.Prim l -> MValue l
  713     loop env (VPiType nm t body) (Prims.PrimFun f) =
  714       pure $ VFun nm $ \x ->
  715         do tp' <- applyPiBody body x
  716            loop ((x,t):env) tp' (f x)
  717 
  718     loop env (VPiType nm t body) (Prims.PrimStrict f) =
  719       pure $ VFun nm $ \x ->
  720         do tp' <- applyPiBody body x
  721            x'  <- force x
  722            loop ((ready x',t):env) tp' (f x')
  723 
  724     loop env (VPiType nm t body) (Prims.PrimFilterFun msg r f) =
  725       pure $ VFun nm $ \x ->
  726         do tp' <- applyPiBody body x
  727            x'  <- force x
  728            runMaybeT (r x') >>= \case
  729              Just v -> loop ((ready x',t):env) tp' (f v)
  730              _ -> fallback msg ((ready x',t):env) tp'
  731 
  732     loop env tp (Prims.PrimExcept m) =
  733       runExceptT m >>= \case
  734         Right v  -> pure v
  735         Left msg -> fallback msg env tp
  736 
  737     loop _env _tp (Prims.Prim m) = m
  738     loop _env _tp (Prims.PrimValue v) = pure v
  739 
  740     loop _env _tp _p =
  741       panic "evalPrim" [
  742           "Type mismatch in primitive: " <> toAbsoluteName (ecNameInfo ec)]
  743 
  744 -- | A basic handler for stuck primitives.
  745 defaultPrimHandler ::
  746   (VMonadLazy l, MonadFail (EvalM l)) =>
  747   ExtCns (TValue l) -> Text -> Env l -> TValue l -> MValue l
  748 defaultPrimHandler ec msg env _tv =
  749   fail $ unlines
  750   [ "Could not evaluate primitive " ++ Text.unpack (toAbsoluteName (ecNameInfo ec))
  751   , "On argument " ++ show (length env)
  752   , Text.unpack msg
  753   ]
