   41 parseSBV :: SharedContext -> NodeCache -> SBV.SBV -> IO (Natural, Term)
   42 parseSBV sc _ (SBV.SBV size (Left num)) =
   43     do t <- scBvConst sc (fromInteger size) num
   44        return (fromInteger size, t)
   45 parseSBV _ nodes (SBV.SBV size (Right nodeid)) =
   46     case Map.lookup nodeid nodes of
   47       Just t -> return (fromIntegral size, t)
   48       Nothing -> fail "parseSBV"

   52 parseSBVExpr :: Options -> SharedContext -> UnintMap -> NodeCache ->
   53                 Natural -> SBV.SBVExpr -> IO Term
   54 parseSBVExpr _opts sc _unint nodes _size (SBV.SBVAtom sbv) =
   55     liftM snd $ parseSBV sc nodes sbv
   56 parseSBVExpr opts sc unint nodes size (SBV.SBVApp operator sbvs) =
   57     case operator of
   58       SBV.BVAdd -> binop scBvAdd sbvs
   59       SBV.BVSub -> binop scBvSub sbvs
   60       SBV.BVMul -> binop scBvMul sbvs
   61       SBV.BVDiv _loc -> binop (error "bvDiv") sbvs
   62       SBV.BVMod _loc -> binop (error "bvMod") sbvs
   63       SBV.BVPow -> binop (error "bvPow") sbvs
   64       SBV.BVIte ->
   65           case sbvs of
   66             [sbv1, sbv2, sbv3] ->
   67                 do (_size1, arg1) <- parseSBV sc nodes sbv1
   68                    (_size2, arg2) <- parseSBV sc nodes sbv2
   69                    (_size3, arg3) <- parseSBV sc nodes sbv3
   70                    -- assert size1 == 1 && size2 == size && size3 == size
   71                    s <- scBitvector sc size
   72                    cond <- scBv1ToBool sc arg1
   73                    scIte sc s cond arg2 arg3
   74             _ -> fail "parseSBVExpr: wrong number of arguments for if-then-else"
   75       SBV.BVShl -> shiftop scBvShiftL sbvs
   76       SBV.BVShr -> shiftop scBvShiftR sbvs
   77       SBV.BVRol -> shiftop scBvRotateL sbvs
   78       SBV.BVRor -> shiftop scBvRotateR sbvs
   79       SBV.BVExt hi lo | lo >= 0 && hi >= lo ->
   80           case sbvs of
   81             [sbv1] ->
   82                 do (size1, arg1) <- parseSBV sc nodes sbv1
   83                    unless (size == fromInteger (hi + 1 - lo)) (fail $ "parseSBVExpr BVExt: size mismatch " ++ show (size, hi, lo))
   84                    b <- scBoolType sc
   85                    s1 <- scNat sc (size1 - 1 - fromInteger hi)
   86                    s2 <- scNat sc size
   87                    s3 <- scNat sc (fromInteger lo)
   88                    -- SBV indexes bits starting with 0 = lsb.
   89                    scSlice sc b s1 s2 s3 arg1
   90             _ -> fail "parseSBVExpr: wrong number of arguments for extract"
   91       SBV.BVExt{} -> fail "parseSBVExpr: BVExt bad arguments"
   92       SBV.BVAnd -> binop scBvAnd sbvs
   93       SBV.BVOr  -> binop scBvOr  sbvs
   94       SBV.BVXor -> binop scBvXor sbvs
   95       SBV.BVNot ->
   96           case sbvs of
   97             [sbv1] ->
   98                 do (size1, arg1) <- parseSBV sc nodes sbv1
   99                    s1 <- scNat sc size1
  100                    unless (size == size1) (fail $ "parseSBVExpr BVNot: size mismatch " ++ show (size, size1))
  101                    scBvNot sc s1 arg1
  102             _ -> fail "parseSBVExpr: wrong number of arguments for Not"
  103       SBV.BVEq  -> binrel scBvEq  sbvs
  104       SBV.BVGeq -> binrel scBvUGe sbvs
  105       SBV.BVLeq -> binrel scBvULe sbvs
  106       SBV.BVGt  -> binrel scBvUGt sbvs
  107       SBV.BVLt  -> binrel scBvULt sbvs
  108       SBV.BVApp ->
  109           case sbvs of
  110             [sbv1, sbv2] ->
  111                 do (size1, arg1) <- parseSBV sc nodes sbv1
  112                    (size2, arg2) <- parseSBV sc nodes sbv2
  113                    s1 <- scNat sc size1
  114                    s2 <- scNat sc size2
  115                    unless (size == size1 + size2) (fail $ "parseSBVExpr BVApp: size mismatch " ++ show (size, size1, size2))
  116                    b <- scBoolType sc
  117                    -- SBV append takes the most-significant argument
  118                    -- first, as SAWCore does.
  119                    scAppend sc s1 s2 b arg1 arg2
  120             _ -> fail "parseSBVExpr: wrong number of arguments for append"
  121       SBV.BVLkUp indexSize resultSize ->
  122           do (size1 : inSizes, arg1 : args) <- liftM unzip $ mapM (parseSBV sc nodes) sbvs
  123              unless (size1 == fromInteger indexSize && all (== (fromInteger resultSize)) inSizes)
  124                         (fail $ "parseSBVExpr BVLkUp: size mismatch")
  125              e <- scBitvector sc (fromInteger resultSize)
  126              scMultiMux sc (fromInteger indexSize) e arg1 args
  127       SBV.BVUnint _loc _codegen (name, irtyp) ->
  128           do let typ = parseIRType irtyp
  129              t <- case unint name typ of
  130                Just t -> return t
  131                Nothing ->
  132                    do printOutLn opts Warn ("WARNING: unknown uninterpreted function " ++ show (name, typ, size))
  133                       printOutLn opts Info ("Using Prelude." ++ name)
  134                       scGlobalDef sc (mkIdent preludeName (Text.pack name))
  135              args <- mapM (parseSBV sc nodes) sbvs
  136              let inSizes = map fst args
  137              (inTyp, outTyp) <-
  138                case typ of
  139                  TFun inTyp outTyp -> pure (inTyp, outTyp)
  140                  _ -> fail "parseSBVExpr: expected function type"
  141              unless (sum (typSizes inTyp) == sum (map fromIntegral inSizes)) $ do
  142                printOutLn opts Error ("ERROR parseSBVPgm: input size mismatch in " ++ name)
  143                -- these used to print without newlines but that seems totally wrong
  144                printOutLn opts Error (show inTyp)
  145                printOutLn opts Error (show inSizes)
  146              argument <- combineOutputs sc inTyp args
  147              result <- scApply sc t argument
  148              results <- splitInputs sc outTyp result
  149              let outSizes = typSizes outTyp
  150              -- Append bitvector components of result value in lsb order
  151              scAppendAll sc (reverse (zip results outSizes))
  152     where
  153       -- | scMkOp : (n : Nat) -> Vec n Bool -> Vec n Bool -> Vec n Bool;
  154       binop scMkOp [sbv1, sbv2] =
  155           do (size1, arg1) <- parseSBV sc nodes sbv1
  156              (size2, arg2) <- parseSBV sc nodes sbv2
  157              unless (size1 == size && size2 == size) (fail $ "parseSBVExpr binop: size mismatch " ++ show (size, size1, size2))
  158              s <- scNat sc size
  159              scMkOp sc s arg1 arg2
  160       binop _ _ = fail "parseSBVExpr: wrong number of arguments for binop"
  161       -- | scMkRel : (n : Nat) -> Vec n Bool -> Vec n Bool -> Bool;
  162       binrel scMkRel [sbv1, sbv2] =
  163           do (size1, arg1) <- parseSBV sc nodes sbv1
  164              (size2, arg2) <- parseSBV sc nodes sbv2
  165              unless (size == 1 && size1 == size2) (fail $ "parseSBVExpr binrel: size mismatch " ++ show (size, size1, size2))
  166              s <- scNat sc size1
  167              t <- scMkRel sc s arg1 arg2
  168              scBoolToBv1 sc t
  169       binrel _ _ = fail "parseSBVExpr: wrong number of arguments for binrel"
  170       -- | scMkOp : (n : Nat) -> Vec n Bool -> Nat -> Vec n Bool;
  171       shiftop scMkOp [sbv1, sbv2] =
  172           do (size1, arg1) <- parseSBV sc nodes sbv1
  173              (size2, arg2) <- parseSBV sc nodes sbv2
  174              unless (size1 == size) (fail "parseSBVExpr shiftop: size mismatch")
  175              s1 <- scNat sc size1
  176              s2 <- scNat sc size2
  177              c <- scBool sc False
  178              boolTy <- scBoolType sc
  179              scMkOp s1 boolTy s2 c arg1 arg2
  180       shiftop _ _ = fail "parseSBVExpr: wrong number of arguments for binop"
  181       scBvShiftL n ty w c v amt = scGlobalApply sc "Prelude.bvShiftL" [n, ty, w, c, v, amt]
  182       scBvShiftR n ty w c v amt = scGlobalApply sc "Prelude.bvShiftR" [n, ty, w, c, v, amt]
  183       scBvRotateL n ty w _ v amt = scGlobalApply sc "Prelude.bvRotateL" [n, ty, w, v, amt]
  184       scBvRotateR n ty w _ v amt = scGlobalApply sc "Prelude.bvRotateR" [n, ty, w, v, amt]

  188 data SBVAssign = SBVAssign SBV.Size SBV.NodeId SBV.SBVExpr
  189   deriving Show
  190 data SBVInput = SBVInput SBV.Size SBV.NodeId
  191   deriving Show
  192 type SBVOutput = SBV.SBV
  193 
  194 partitionSBVCommands :: [SBV.SBVCommand] -> ([SBVAssign], [SBVInput], [SBVOutput])
  195 partitionSBVCommands = foldr select ([], [], [])
  196     where
  197       select cmd (assigns, inputs, outputs) =
  198           case cmd of
  199             SBV.Decl _ (SBV.SBV _ (Left _)) _ ->
  200                 error "invalid SBV command: ground value on lhs"
  201             SBV.Decl _ (SBV.SBV size (Right nodeid)) Nothing ->
  202                 (assigns, SBVInput size nodeid : inputs, outputs)
  203             SBV.Decl _ (SBV.SBV size (Right nodeid)) (Just expr) ->
  204                 (SBVAssign size nodeid expr : assigns, inputs, outputs)
  205             SBV.Output sbv ->
  206                 (assigns, inputs, sbv : outputs)
  207 
  208 -- TODO: Should I use a state monad transformer?
  209 parseSBVAssign :: Options -> SharedContext -> UnintMap -> NodeCache -> SBVAssign -> IO NodeCache
  210 parseSBVAssign opts sc unint nodes (SBVAssign size nodeid expr) =
  211     do term <- parseSBVExpr opts sc unint nodes (fromInteger size) expr
  212        return (Map.insert nodeid term nodes)

  216 data Typ
  217   = TBool
  218   | TFun Typ Typ
  219   | TVec SBV.Size Typ
  220   | TTuple [Typ]
  221   | TRecord [(FieldName, Typ)]
  222 
  223 instance Show Typ where
  224   show TBool = "."
  225   show (TFun t1 t2) = "(" ++ show t1 ++ " -> " ++ show t2 ++ ")"
  226   show (TVec size t) = "[" ++ show size ++ "]" ++ show t
  227   show (TTuple ts) = "(" ++ intercalate "," (map show ts) ++ ")"
  228   show (TRecord fields) = "{" ++ intercalate "," (map showField fields) ++ "}"
  229     where showField (s, t) = Text.unpack s ++ ":" ++ show t
  230 
  231 parseIRType :: SBV.IRType -> Typ
  232 parseIRType (SBV.TApp "." []) = TBool
  233 parseIRType (SBV.TApp "->" [a, b]) = TFun (parseIRType a) (parseIRType b)
  234 parseIRType (SBV.TApp ":" [SBV.TInt n, a]) = TVec n (parseIRType a)
  235 parseIRType (SBV.TApp c ts)
  236   | c == "(" ++ replicate (length ts - 1) ',' ++ ")" = TTuple (map parseIRType ts)
  237 parseIRType (SBV.TRecord fields) =
  238     TRecord [ (Text.pack name, parseIRType t) | (name, SBV.Scheme [] [] [] t) <- fields ]
  239 parseIRType t = error ("parseIRType: " ++ show t)
  240 
  241 typSizes :: Typ -> [SBV.Size]
  242 typSizes TBool = [1]
  243 typSizes (TTuple ts) = concatMap typSizes ts
  244 typSizes (TVec n TBool) = [n]
  245 typSizes (TVec n t) = concat (replicate (fromIntegral n) (typSizes t))
  246 typSizes (TRecord fields) = concatMap (typSizes . snd) fields
  247 typSizes (TFun _ _) = error "typSizes: not a first-order type"
  248 
  249 scTyp :: SharedContext -> Typ -> IO Term
  250 scTyp sc TBool = scBoolType sc
  251 scTyp sc (TFun a b) =
  252     do s <- scTyp sc a
  253        t <- scTyp sc b
  254        scFun sc s t
  255 scTyp sc (TVec n TBool) =
  256     do scBitvector sc (fromInteger n)
  257 scTyp sc (TVec n t) =
  258     do ty <- scTyp sc t
  259        ntm <- scNat sc (fromInteger n)
  260        scVecType sc ntm ty
  261 scTyp sc (TTuple as) =
  262     do ts <- mapM (scTyp sc) as
  263        scTupleType sc ts
  264 scTyp sc (TRecord fields) =
  265     do tm <- mapM (\(f,t) -> (f,) <$> scTyp sc t) fields
  266        scRecordType sc tm

  268 -- | projects all the components out of the input term
  269 -- TODO: rename to splitInput?
  270 splitInputs :: SharedContext -> Typ -> Term -> IO [Term]
  271 splitInputs _sc TBool x = return [x]
  272 splitInputs sc (TTuple ts) x =
  273     do xs <- mapM (\i -> scTupleSelector sc x i) [0 .. length ts - 1]
  274        yss <- sequence (zipWith (splitInputs sc) ts xs)
  275        return (concat yss)
  276 splitInputs _ (TVec _ TBool) x = return [x]
  277 splitInputs sc (TVec n t) x =
  278     do nt <- scNat sc (fromIntegral n)
  279        idxs <- mapM (scNat sc . fromIntegral) [0 .. (n - 1)]
  280        ty <- scTyp sc t
  281        xs <- mapM (scAt sc nt ty x) idxs
  282        yss <- mapM (splitInputs sc t) xs
  283        return (concat yss)
  284 splitInputs _ (TFun _ _) _ = error "splitInputs TFun: not a first-order type"
  285 splitInputs sc (TRecord fields) x =
  286     do let (names, ts) = unzip fields
  287        xs <- mapM (scRecordSelect sc x) names
  288        yss <- sequence (zipWith (splitInputs sc) ts xs)
  289        return (concat yss)

  293 -- | Combines outputs into a data structure according to Typ
  294 combineOutputs :: SharedContext -> Typ -> [(Natural, Term)] -> IO Term
  295 combineOutputs sc ty xs0 =
  296     do (z, ys) <- runStateT (go ty) xs0
  297        unless (null ys) (fail $ "combineOutputs: too many outputs: " ++
  298                                 show (length ys) ++ " remaining")
  299        return z
  300     where
  301       pop :: StateT [(Natural, Term)] IO (Natural, Term)
  302       pop = do xs <- get
  303                case xs of
  304                  [] -> fail "combineOutputs: too few outputs"
  305                  y : ys -> put ys >> return y
  306       go :: Typ -> StateT [(Natural, Term)] IO Term
  307       go TBool =
  308           do (_, x) <- pop
  309              lift (scBv1ToBool sc x)
  310       go (TTuple ts) =
  311           do xs <- mapM go ts
  312              lift (scTuple sc xs)
  313       -- | SBV files may encode values of type '[n]Bool' in one of two
  314       -- ways: as a single n-bit word, or as a list of n 1-bit words.
  315       go (TVec n TBool) =
  316           do (n', x) <- pop
  317              case () of
  318                () | n' == fromIntegral n -> return x
  319                   | n' == 1 ->
  320                     do (sizes, xs) <- fmap unzip $ replicateM (fromIntegral n - 1) pop
  321                        unless (all (== 1) sizes) $
  322                          fail $ "combineOutputs: can't read SBV bitvector: " ++
  323                                 show sizes ++ " doesn't equal " ++ show n
  324                        -- Append 1-bit words, lsb first (TODO: is this right?)
  325                        lift $ scAppendAll sc $ reverse [ (t, 1) | t <- x : xs ]
  326                   | otherwise -> fail $ "combineOutputs: can't read SBV bitvector from " ++
  327                                         show n' ++ " arguments"
  328       go (TVec n t) =
  329           do xs <- replicateM (fromIntegral n) (go t)
  330              ety <- lift (scTyp sc t)
  331              lift (scVector sc ety xs)
  332       go (TRecord fields) =
  333           do let (names, ts) = unzip fields
  334              xs <- mapM go ts
  335              lift (scRecord sc (Map.fromList (zip names xs)))
  336       go (TFun _ _) =
  337           fail "combineOutputs: not a first-order type"

  341 parseSBVPgm :: Options -> SharedContext -> UnintMap -> SBV.SBVPgm -> IO Term
  342 parseSBVPgm opts sc unint (SBV.SBVPgm (_version, irtype, revcmds, _vcs, _warnings, _uninterps)) =
  343     do (inTyp, outTyp) <-
  344          case parseIRType irtype of
  345            TFun inTyp outTyp -> pure (inTyp, outTyp)
  346            _ -> fail "parseSBVPgm: expected function type"
  347        let cmds = reverse revcmds
  348        let (assigns, inputs, outputs) = partitionSBVCommands cmds
  349        let inSizes = [ size | SBVInput size _ <- inputs ]
  350        let inNodes = [ node | SBVInput _ node <- inputs ]
  351        unless (typSizes inTyp == inSizes) (fail "parseSBVPgm: input size mismatch")
  352        inputType <- scTyp sc inTyp
  353        inputVar <- scFreshGlobal sc "x" inputType
  354        inputTerms <- splitInputs sc inTyp inputVar
  355        let nodes0 = Map.fromList (zip inNodes inputTerms)
  356        nodes <- foldM (parseSBVAssign opts sc unint) nodes0 assigns
  357        outputTerms <- mapM (parseSBV sc nodes) outputs
  358        outputTerm <- combineOutputs sc outTyp outputTerms
  359        scAbstractTerms sc [inputVar] outputTerm

  364 -- | bv1ToBool : Vec 1 Bool -> Bool
  365 -- bv1ToBool x = bvAt 1 Bool 1 x (bv 1 0)
  366 scBv1ToBool :: SharedContext -> Term -> IO Term
  367 scBv1ToBool sc x =
  368     do n0 <- scNat sc 0
  369        n1 <- scNat sc 1
  370        b <- scBoolType sc
  371        bv <- scBvNat sc n1 n0
  372        scBvAt sc n1 b n1 x bv
  373 
  374 -- | boolToBv1 :: Bool -> Vec 1 Bool
  375 scBoolToBv1 :: SharedContext -> Term -> IO Term
  376 scBoolToBv1 sc x =
  377     do b <- scBoolType sc
  378        scSingle sc b x
  379 
  380 -- see if it's the same error
  381 scMultiMux :: SharedContext -> Natural -> Term
  382            -> Term -> [Term] -> IO Term
  383 scMultiMux sc iSize e i args = do
  384     vec <- scVector sc e args
  385     w <- scNat sc iSize
  386     m <- scNat sc (fromIntegral (length args))
  387     scBvAt sc m e w vec i
  388 
  389 scAppendAll :: SharedContext -> [(Term, Integer)] -> IO Term
  390 scAppendAll _ [] = error "scAppendAll: unimplemented"
  391 scAppendAll _ [(x, _)] = return x
  392 scAppendAll sc ((x, size1) : xs) =
  393     do let size2 = sum (map snd xs)
  394        b <- scBoolType sc
  395        s1 <- scNat sc (fromInteger size1)
  396        s2 <- scNat sc (fromInteger size2)
  397        y <- scAppendAll sc xs
  398        scAppend sc s1 s2 b x y
