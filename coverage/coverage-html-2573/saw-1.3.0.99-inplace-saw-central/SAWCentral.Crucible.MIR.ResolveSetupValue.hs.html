  125 -- | Pretty-print a 'MIRVal'.
  126 ppMIRVal ::
  127   forall ann.
  128   Sym ->
  129   MIRVal ->
  130   PP.Doc ann
  131 ppMIRVal sym (MIRVal shp val) =
  132   case shp of
  133     UnitShape _ ->
  134       PP.pretty val
  135     PrimShape _ _ ->
  136       W4.printSymExpr val
  137     TupleShape _ elems -> prettyAggregate elems val
  138     ArrayShape _ _ shp' ->
  139       case val of
  140         Mir.MirVector_Vector vec ->
  141           PP.brackets $ commaList $ V.toList $
  142           fmap (\v -> ppMIRVal sym (MIRVal shp' v)) vec
  143         Mir.MirVector_PartialVector vec ->
  144           PP.braces $ commaList $ V.toList $
  145           fmap (\v -> let v' = readMaybeType sym "vector element" (shapeType shp') v in
  146                       ppMIRVal sym (MIRVal shp' v')) vec
  147         Mir.MirVector_Array arr ->
  148           W4.printSymExpr arr
  149     StructShape _ _ fldShp ->
  150       PP.braces $ prettyAdtOrTuple fldShp val
  151     EnumShape _ _ variantShps _ _
  152       |  Ctx.Empty Ctx.:> RV _ Ctx.:> RV variants <- val
  153       -> case firstConcreteVariant variantShps variants of
  154            Just (Some (Functor.Pair fldShps fldVals)) ->
  155              PP.braces $ prettyAdtOrTuple fldShps fldVals
  156            Nothing ->
  157              "<symbolic enum>"
  158     TransparentShape _ shp' ->
  159       ppMIRVal sym $ MIRVal shp' val
  160     RefShape _ _ _ _  ->
  161       "<reference>"
  162     SliceShape _ _ _ _ ->
  163       "<slice>"
  164     FnPtrShape _ _ _ ->
  165       PP.viaShow val
  166   where
  167     commaList :: [PP.Doc ann] -> PP.Doc ann
  168     commaList []     = PP.emptyDoc
  169     commaList (x:xs) = x PP.<> PP.hcat (map (\y -> PP.comma PP.<+> y) xs)
  170 
  171     -- Return Just the first VariantBranch whose predicate is concretely equal
  172     -- to True. If none of the VariantBranches satisfy this property, then
  173     -- return Nothing.
  174     firstConcreteVariant ::
  175       Ctx.Assignment VariantShape ctx ->
  176       Ctx.Assignment (VariantBranch Sym) ctx ->
  177       Maybe (Some (Functor.Product
  178         (Ctx.Assignment FieldShape)
  179         (Ctx.Assignment (RegValue' Sym))))
  180     firstConcreteVariant variantShapes variantBranches =
  181       List.firstJust
  182         (\(Some (Functor.Pair (VariantShape fldShps) (VB branch))) ->
  183           case branch of
  184             W4.PE fldPred fldVals
  185               |  W4.asConstantPred fldPred == Just True
  186               -> Just $ Some $ Functor.Pair fldShps fldVals
  187               |  otherwise
  188               -> Nothing
  189             W4.Unassigned ->
  190               Nothing) $
  191       FC.toListFC Some $
  192       Ctx.zipWith Functor.Pair variantShapes variantBranches
  193 
  194     prettyAdtOrTuple ::
  195       forall ctx.
  196       Ctx.Assignment FieldShape ctx ->
  197       Ctx.Assignment (RegValue' Sym) ctx ->
  198       PP.Doc ann
  199     prettyAdtOrTuple fldShp fldVals =
  200       commaList $
  201       map (\(Some (Functor.Pair shp' (RV v))) -> prettyField shp' v) $
  202       FC.toListFC Some $
  203       Ctx.zipWith Functor.Pair fldShp fldVals
  204 
  205     prettyField ::
  206       forall tp.
  207       FieldShape tp ->
  208       RegValue Sym tp ->
  209       PP.Doc ann
  210     prettyField fldShp val' =
  211       case fldShp of
  212         OptField shp' ->
  213           ppMIRVal sym $ MIRVal shp' $
  214           readMaybeType sym "field" (shapeType shp') val'
  215         ReqField shp' ->
  216           ppMIRVal sym $ MIRVal shp' val'

  218     prettyAggregate ::
  219       [AgElemShape] ->
  220       Mir.MirAggregate Sym ->
  221       PP.Doc ann
  222     prettyAggregate elems (Mir.MirAggregate _sz m) =
  223       PP.braces $ commaList (map (prettyAgElem m) elems)
  224 
  225     prettyAgElem ::
  226       IntMap (Mir.MirAggregateEntry Sym) ->
  227       AgElemShape ->
  228       PP.Doc ann
  229     prettyAgElem m (AgElemShape off _sz shp') =
  230       let valDoc = case IntMap.lookup (fromIntegral off) m of
  231             Just (Mir.MirAggregateEntry _sz tpr rv)
  232               | Just Refl <- W4.testEquality tpr (shapeType shp') ->
  233                   ppMIRVal sym $ MIRVal shp' $
  234                   readMaybeType sym "elem" tpr rv
  235               | otherwise -> "<type mismatch>"
  236             Nothing -> "<unset>"
  237       in
  238       PP.viaShow off PP.<+> "->" PP.<+> valDoc

  241 -- | Wrapper around `buildMirAggregate` for the case where the additional
  242 -- values are `MIRVal`s.  This automatically checks that the `MIRVal`'s
  243 -- `TypeShape` matches the shape of the `Mir.MirAggregateEntry` and just passes
  244 -- the `MIRVal`'s inner `RegValue` to the callback.
  245 buildMirAggregateWithVal ::
  246   (Monad m, MonadFail m) =>
  247   Sym ->
  248   [AgElemShape] ->
  249   [MIRVal] ->
  250   (forall tp. Word -> Word -> TypeShape tp -> RegValue Sym tp -> m (RegValue Sym tp)) ->
  251   m (Mir.MirAggregate Sym)
  252 buildMirAggregateWithVal sym elems vals f =
  253   buildMirAggregate sym elems vals $
  254     \off sz shp (MIRVal shp' rv) ->
  255       case W4.testEquality shp shp' of
  256         Just Refl -> f off sz shp rv
  257         Nothing -> fail $ "buildMirAggregateWithVal: type mismatch at offset "
  258           ++ show off ++ ": expected " ++ show shp ++ ", but the MIRVal contained " ++ show shp'

  263 data MIRTypeOfError
  264   = MIRPolymorphicType Cryptol.Schema
  265   | MIRNonRepresentableType Cryptol.Type ToMIRTypeErr
  266   | MIRInvalidTypedTerm TypedTermType
  267   | MIRInvalidIdentifier String
  268   | MIRStaticNotFound Mir.DefId
  269   | MIRSliceNonReference Mir.Ty
  270   | MIRSliceNonArrayReference Mir.Ty
  271   | MIRSliceWrongTy Mir.Ty
  272   | MIRStrSliceNonU8Array Mir.Ty
  273   | MIRMuxNonBoolCondition Mir.Ty
  274   | MIRMuxDifferentBranchTypes Mir.Ty Mir.Ty
  275   | MIRCastNonRawPtr Mir.Ty
  276   | MIRIndexOutOfBounds
  277       Mir.Ty -- ^ sequence type
  278       Int    -- ^ sequence length
  279       Int    -- ^ attempted index
  280   | MIRIndexWrongTy MirIndexingMode Mir.Ty

  369 typeOfSetupValue ::
  370   forall m.
  371   X.MonadThrow m =>
  372   MIRCrucibleContext ->
  373   Map AllocIndex (Some MirAllocSpec) ->
  374   Map AllocIndex Text ->
  375   SetupValue ->
  376   m Mir.Ty
  377 typeOfSetupValue mcc env nameEnv val =
  378   case val of
  379     MS.SetupVar i ->
  380       case Map.lookup i env of
  381         Nothing ->
  382            panic "MIRSetup (in typeOfSetupValue)" [
  383                "Unresolved prestate variable: " <> Text.pack (show i)
  384            ]
  385         Just (Some alloc) ->
  386           pure $ ptrKindToTy (alloc^.maPtrKind) (alloc^.maMirType) (alloc^.maMutbl)
  387     MS.SetupTerm tt ->
  388       typeOfTypedTerm tt
  389     MS.SetupArray elemTy vs ->
  390       pure $ Mir.TyArray elemTy (length vs)
  391     MS.SetupStruct adt _ ->
  392       pure $ mirAdtToTy adt
  393     MS.SetupEnum enum_ ->
  394       case enum_ of
  395         MirSetupEnumVariant adt _ _ _ ->
  396           pure $ mirAdtToTy adt
  397         MirSetupEnumSymbolic adt _ _ ->
  398           pure $ mirAdtToTy adt
  399     MS.SetupTuple () vals -> do
  400       tys <- traverse (typeOfSetupValue mcc env nameEnv) vals
  401       pure $ Mir.TyTuple tys
  402     MS.SetupGlobal () name ->
  403       staticTyRef <$> findStatic cs name
  404     MS.SetupGlobalInitializer () name -> do
  405       static <- findStatic cs name
  406       pure $ static ^. Mir.sTy
  407     MS.SetupSlice slice ->
  408       case slice of
  409         MirSetupSliceRaw{} ->
  410           panic "MIRSetup (in typeOfSetupValue)" [
  411               "MirSetupSliceRaw not yet implemented"
  412           ]
  413         MirSetupSlice sliceInfo arrRef ->
  414           typeOfSliceFromArrayRef sliceInfo arrRef
  415         MirSetupSliceRange sliceInfo arrRef _ _ ->
  416           typeOfSliceFromArrayRef sliceInfo arrRef
  417     MS.SetupMux () c t f -> do
  418       cTy <- typeOfTypedTerm c
  419       unless (cTy == Mir.TyBool) $
  420         X.throwM $ MIRMuxNonBoolCondition cTy
  421       tTy <- typeOfSetupValue mcc env nameEnv t
  422       fTy <- typeOfSetupValue mcc env nameEnv f
  423       unless (checkCompatibleTys tTy fTy) $
  424         X.throwM $ MIRMuxDifferentBranchTypes tTy fTy
  425       pure tTy
  426     MS.SetupCast newPointeeTy oldPtr -> do
  427       -- Make sure the cast is valid
  428       oldPtrTy <- typeOfSetupValue mcc env nameEnv oldPtr
  429       case oldPtrTy of
  430         Mir.TyRawPtr _ mutbl ->
  431           pure $ Mir.TyRawPtr newPointeeTy mutbl
  432         _ -> X.throwM $ MIRCastNonRawPtr oldPtrTy
  433 
  434     MS.SetupElem ixMode xs i -> do
  435       xsTy <- typeOfSetupValue mcc env nameEnv xs
  436       let boundsCheck len res
  437             | i >= 0 && i < len = pure res
  438             | otherwise = X.throwM $ MIRIndexOutOfBounds xsTy len i
  439           throwWrongTy = X.throwM $ MIRIndexWrongTy ixMode xsTy
  440       case ixMode of
  441         MirIndexIntoVal ->
  442           case xsTy of
  443             Mir.TyArray elemTy len -> boundsCheck len elemTy
  444             _ -> throwWrongTy
  445         MirIndexIntoRef ->
  446           case tyToShape col xsTy of
  447             Some (RefShape ptrTy (Mir.TyArray elemTy len) mutbl _) ->
  448               boundsCheck len $ ptrKindToTy (tyToPtrKind ptrTy) elemTy mutbl
  449             _ -> throwWrongTy
  450         MirIndexOffsetRef ->
  451           panic "MIRSetup (in typeOfSetupValue)"
  452             ["MirIndexOffsetRef not yet implemented"]
  453     MS.SetupField _ _ _ ->
  454       panic "MIRSetup (in typeOfSetupValue)" ["fields not yet implemented"]
  455 
  456     MS.SetupNull empty                -> absurd empty
  457     MS.SetupUnion empty _ _           -> absurd empty
  458   where
  459     cs = mcc ^. mccRustModule . Mir.rmCS
  460     col = cs ^. Mir.collection
  461 
  462     typeOfSliceFromArrayRef :: MirSliceInfo -> SetupValue -> m Mir.Ty
  463     typeOfSliceFromArrayRef sliceInfo arrRef = do
  464       arrRefTy <- typeOfSetupValue mcc env nameEnv arrRef
  465       case arrRefTy of
  466         Mir.TyRef arrTy mut -> do
  467           (sliceTy, _elemTy, _len) <- arrayToSliceTys sliceInfo mut arrTy
  468           pure $ Mir.TyRef sliceTy mut

  472     typeOfTypedTerm :: TypedTerm -> m Mir.Ty
  473     typeOfTypedTerm tt =
  474       case ttType tt of

  491 -- | Translate a SetupValue into a Crucible MIR value, resolving
  492 -- references
  493 resolveSetupVal ::
  494   MIRCrucibleContext ->
  495   Map AllocIndex (Some (MirPointer Sym)) ->
  496   Map AllocIndex (Some MirAllocSpec) ->
  497   Map AllocIndex Text ->
  498   SetupValue ->
  499   IO MIRVal
  500 resolveSetupVal mcc env tyenv nameEnv val =
  501   mccWithBackend mcc $ \bak ->
  502   let sym = backendGetSym bak in
  503   case val of
  504     MS.SetupVar i -> do
  505       Some ptr <- pure $ lookupAllocIndex env i
  506       let pointeeType = ptr ^. mpMirType
  507       pure $ MIRVal (RefShape
  508                         (ptrKindToTy (ptr ^. mpKind) pointeeType (ptr ^. mpMutbl))
  509                         pointeeType
  510                         (ptr ^. mpMutbl)
  511                         (ptr ^. mpType))
  512                     (ptr ^. mpRef)
  513     MS.SetupTerm tm -> resolveTypedTerm mcc tm
  514     MS.SetupNull empty                -> absurd empty
  515     MS.SetupStruct adt flds ->
  516       case adt of
  517         _ | adt ^. Mir.adtReprTransparent,
  518             Just i <- Mir.findReprTransparentField col adt,
  519             Just fld <- flds ^? ix i ->
  520           resolveTransparentSetupVal adt fld
  521         Mir.Adt nm Mir.Struct variants _ _ _ _ -> do
  522           -- First, retrieve the struct variant.
  523           variant <-
  524             case variants of
  525               [variant] ->
  526                 pure variant
  527               _ ->
  528                 panic "resolveSetupVal" [
  529                     "Encountered struct Adt with " <>
  530                         Text.pack (show (length variants)) <>
  531                         " variants: ",
  532                     "   " <> Text.pack (show nm)
  533                 ]
  534 
  535           -- Next, resolve the field values and check that they have the
  536           -- expected types.
  537           flds' <- traverse (resolveSetupVal mcc env tyenv nameEnv) flds
  538           let expectedFlds = variant ^. Mir.vfields
  539           let actualFldTys = map (\(MIRVal shp _) -> shapeMirTy shp) flds'
  540           checkFields nm "Struct" "struct fields" expectedFlds actualFldTys
  541 
  542           -- Finally, construct a MIRVal of the appropriate shape.
  543           Some (Functor.Pair fldShpAssn valAssn) <-
  544             pure $ variantFieldsToAssns sym flds'
  545           let structShp = StructShape (mirAdtToTy adt) actualFldTys fldShpAssn
  546           pure $ MIRVal structShp valAssn
  547         Mir.Adt nm (Mir.Enum _) _ _ _ _ _ ->
  548           panic "resolveSetupVal" [
  549               "Expected struct type, received enum: " <> Text.pack (show nm)
  550           ]
  551         Mir.Adt nm Mir.Union _ _ _ _ _ ->
  552           panic "resolveSetupVal" [
  553               "Expected struct type, received union: " <> Text.pack (show nm)
  554           ]
  555     MS.SetupEnum enum_ ->
  556       case enum_ of
  557         MirSetupEnumVariant adt variant variantIdxInt flds ->
  558           case adt of
  559             _ | adt ^. Mir.adtReprTransparent,
  560                 Just i <- Mir.findReprTransparentField col adt,
  561                 Just fld <- flds ^? ix i -> do
  562               resolveTransparentSetupVal adt fld
  563             Mir.Adt nm (Mir.Enum discrTp) variants _ _ _ _ -> do
  564               -- Resolve the field values and check that they have the expected
  565               -- types.
  566               flds' <- traverse (resolveSetupVal mcc env tyenv nameEnv) flds
  567               let expectedFlds = variant ^. Mir.vfields
  568               let actualFldTys = map (\(MIRVal shp _) -> shapeMirTy shp) flds'
  569               checkFields
  570                 nm
  571                 "Enum"
  572                 ("fields in enum variant " ++ show (variant ^. Mir.vname))
  573                 expectedFlds
  574                 actualFldTys
  575 
  576               -- Ensure that the discriminant has an integral type and build
  577               -- a symbolic bitvector from it.
  578               Some discrTpr <- case Mir.tyToRepr col discrTp of
  579                 Left err -> panic "resolveSetupVal" ["Unsupported type", Text.pack err]
  580                 Right x -> return x
  581               let discrShp = tyToShapeEq col discrTp discrTpr
  582               IsBVShape _ discrW <- pure $ testDiscriminantIsBV discrShp
  583               let discr = getEnumVariantDiscr variant
  584               discrVal <- W4.bvLit sym discrW $ BV.mkBV discrW discr
  585 
  586               -- Construct an EnumShape and RustEnumRepr. This requires
  587               -- processing /all/ variants, not just the particular variant that
  588               -- we are building.
  589               SomeEnumShape expectedVariantShps enumShp <- pure $

 1012       fail "resolveSAWTerm: unimplemented type Float e p (FIXME)"

 1014       fail "resolveSAWTerm: unimplemented type Array a b (FIXME)"

 1016       fail "resolveSAWTerm: unimplemented type Rational (FIXME)"

 1018       case sz of
 1019         8   -> bv_term Mir.B8   $ W4.knownNat @8

 1027                 => Mir.BaseSize -> W4.NatRepr w -> IO MIRVal

 1417 -- | Check if two 'Mir.Ty's are compatible in SAW. This is a slightly coarser
 1418 -- notion of equality to reflect the fact that MIR's type system is richer than
 1419 -- Cryptol's type system, and some types which would be distinct in MIR are in
 1420 -- fact equal when converted to the equivalent Cryptol types. In particular:
 1421 --
 1422 -- 1. A @u<N>@ type is always compatible with an @i<N>@ type. For instance, @u8@
 1423 --    is compatible with @i8@, and @u16@ is compatible with @i16@. Note that the
 1424 --    bit sizes of both types must be the same. For instance, @u8@ is /not/
 1425 --    compatible with @i16@.
 1426 --
 1427 -- 2. The @usize@/@isize@ types are always compatible with @u<N>@/@i<N>@, where
 1428 --    @N@ is the number of bits corresponding to the 'Mir.SizeBits' type in
 1429 --    "Mir.Intrinsics". (This is a bit unsavory, as the actual size of
 1430 --    @usize@/@isize@ is platform-dependent, but this is the current approach.)
 1431 --
 1432 -- 3. Compatibility applies recursively. For instance, @[ty_1; N]@ is compatible
 1433 --    with @[ty_2; N]@ iff @ty_1@ and @ty_2@ are compatibile. Similarly, a tuple
 1434 --    typle @(ty_1_a, ..., ty_n_a)@ is compatible with @(ty_1_b, ..., ty_n_b)@
 1435 --    iff @ty_1_a@ is compatible with @ty_1_b@, ..., and @ty_n_a@ is compatible
 1436 --    with @ty_n_b@.
 1437 --
 1438 -- See also @checkRegisterCompatibility@ in "SAWCentral.Crucible.LLVM.Builtins"
 1439 -- and @registerCompatible@ in "SAWCentral.Crucible.JVM.Builtins", which fill a
 1440 -- similar niche in the LLVM and JVM backends, respectively.
 1441 checkCompatibleTys :: Mir.Ty -> Mir.Ty -> Bool
 1442 checkCompatibleTys ty1 ty2 = tyView ty1 == tyView ty2

 1713 -- | Given a definition name @defName@, attempt to look up its corresponding
 1714 -- 'Mir.DefId'. If successful, return it with 'Right'. Currently, the following
 1715 -- types of definition names are permitted:
 1716 --
 1717 -- * @<crate_name>/<disambiguator>::<def_name>: A fully disambiguated name.
 1718 --
 1719 -- * @<crate_name>::<def_name>: A name without a disambiguator. In this
 1720 --   case, SAW will attempt to look up a disambiguator from the
 1721 --   'Mir.crateHashesMap'. If none can be found, or if there are multiple
 1722 --   disambiguators for the given @<crate_name>@, then this will return an error
 1723 --   message with 'Left'.
 1724 --
 1725 -- This also consults the 'Mir.langItems' so that if a user looks up the
 1726 -- original 'DefId' for a lang item (e.g., @core::option::Option@), then this
 1727 -- function will return the @$lang@-based 'DefId' instead (e.g.,
 1728 -- @$lang::Option@), as the latter 'DefId' is what will be used throughout the
 1729 -- rest of the MIR code.
 1730 findDefIdEither :: Mir.CollectionState -> Text -> Either String Mir.DefId
 1731 findDefIdEither cs defName = do
 1732     (crate, path) <-
 1733       case edid of
 1734         crate:path -> pure (crate, path)
 1735         [] -> Left $ unlines
 1736                 [ "The definition `" ++ defNameStr ++ "` lacks a crate."
 1737                 , "Consider providing one, e.g., `<crate_name>::" ++ defNameStr ++ "`."
 1738                 ]
 1739     let crateStr = Text.unpack crate
 1740     origDefId <-
 1741       case Text.splitOn "/" crate of
 1742         [crateNoDisambig, disambig] ->
 1743           Right $ Mir.textId $ Text.intercalate "::"
 1744                 $ (crateNoDisambig <> "/" <> disambig) : path
 1745         [_] ->
 1746           case Map.lookup crate crateDisambigs of
 1747               Just allDisambigs@(disambig :| otherDisambigs)
 1748                 |  F.null otherDisambigs
 1749                 -> Right $ Mir.textId $ Text.intercalate "::"
 1750                          $ (crate <> "/" <> disambig) : path
 1751                 |  otherwise
 1752                 -> Left $ unlines $
 1753                      [ "ambiguous crate " ++ crateStr
 1754                      , "crate disambiguators:"
 1755                      ] ++ F.toList (Text.unpack <$> allDisambigs)
 1756               Nothing -> Left $ "unknown crate " ++ crateStr
 1757         _ -> Left $ "Malformed crate name: " ++ show crateStr
 1758     Right $ Map.findWithDefault origDefId origDefId langItemDefIds
 1759   where
 1760     crateDisambigs = cs ^. Mir.crateHashesMap
 1761     langItemDefIds = cs ^. Mir.collection . Mir.langItems
 1762 
 1763     defNameStr = Text.unpack defName
 1764     edid = Text.splitOn "::" defName

 1880 -- | Check if there is a 'SetupCast' somewhere in a 'SetupValue'.
 1881 containsCast :: SetupValue -> Bool
 1882 containsCast (MS.SetupVar _) = False
 1883 containsCast (MS.SetupTerm _) = False
 1884 containsCast (MS.SetupNull empty) = absurd empty
 1885 containsCast (MS.SetupStruct _ vs) = any containsCast vs
 1886 containsCast (MS.SetupArray _ vs) = any containsCast vs
 1887 containsCast (MS.SetupElem _ v _) = containsCast v
 1888 containsCast (MS.SetupField () v _) = containsCast v
 1889 containsCast (MS.SetupCast _ _) = True
 1890 containsCast (MS.SetupUnion empty _ _) = absurd empty
 1891 containsCast (MS.SetupTuple () vs) = any containsCast vs
 1892 containsCast (MS.SetupSlice slice) =
 1893   case slice of
 1894     MirSetupSliceRaw ptr _ -> containsCast ptr
 1895     MirSetupSlice _ ref -> containsCast ref
 1896     MirSetupSliceRange _ ref _ _ -> containsCast ref
 1897 containsCast (MS.SetupEnum enum_) =
 1898   case enum_ of
 1899     MirSetupEnumVariant _ _ _ vs -> any containsCast vs
 1900     MirSetupEnumSymbolic _ disc vss ->
 1901       containsCast disc || any (any containsCast) vss
 1902 containsCast (MS.SetupGlobal () _) = False
 1903 containsCast (MS.SetupGlobalInitializer () _) = False
 1904 containsCast (MS.SetupMux () _ vt vf) = containsCast vt || containsCast vf
