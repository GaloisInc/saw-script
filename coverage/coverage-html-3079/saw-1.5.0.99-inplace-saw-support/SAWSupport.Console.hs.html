    2 {- |
    3 Module      : SAWSupport.Console
    4 Description : Console messages and error printing infrastructure
    5 License     : BSD3
    6 Maintainer  : saw@galois.com
    7 Stability   : provisional
    8 
    9 This module contains the infrastructure for printing messages to the
   10 console from inside SAW.
   11 
   12 Panic messages should use `panic`. Each component of SAW has its own
   13 `Panic` module that defines the appropriate `panic` function for code
   14 in that component.
   15 
   16 All other messages, apart from directly user-facing UI logic, should
   17 be printed through this interface. If it proves inadequate, it should
   18 be strengthened. (Just printing to standard output or standard error
   19 will not do the right thing with the remote API. That is ok for
   20 temporary debug prints, but not for committed code.)
   21 
   22 Note that the SAWScript interpreter is not user-facing UI code; only
   23 the REPL code qualifies.
   24 
   25 However, the terminal-based interactive logic in the AutoMatch code
   26 will be treated as user-facing UI code; for now at least it will
   27 continue to do its own thing and not go through this interface.
   28 
   29 Note that the `SAWConsole` instance for `IO` is meant as temporary
   30 scaffolding and expected to go away in the long run. `IO` by itself
   31 doesn't provide adequate infrastructure to collect messages, only to
   32 print them, and the latter isn't good enough for the remote API.
   33 
   34 This module is intended to be imported unqualified.
   35 -}

   54 -- SAWConsole class
   55 
   56 -- | Monad class for SAW message printing.
   57 --
   58 --   There are three basic kinds of diagnostic messages:
   59 --      - @err@ prints an error;
   60 --      - @warn@ prints a warning;
   61 --      - @note@ prints a notice.
   62 --
   63 --   These are then further qualified as follows:
   64 --      - @N@ prints without any position;
   65 --      - @P@ prints with an explicit position;
   66 --      - @X@ prints with the current execution position.
   67 --      - The primed versions generate a prettyprinter doc rather than `Text`.
   68 --
   69 --   The @D@ variants of the error functions defer exit: they record
   70 --   that a fatal error occurred but continue execution. This is
   71 --   desirable behavior in a number of contexts (e.g. typechecking).
   72 --
   73 --   Use `checkFail` at suitable points to bail out if an error has
   74 --   previously been recorded.
   75 --
   76 --   Use `unimplemented` (rather than either panic or @err@) to report
   77 --   unimplemented features / missing code paths. We haven't been able
   78 --   to decide if unimplemented features should be panics or errors,
   79 --   and I think the reason is that they have properties of both:
   80 --      - they are a problem in SAW, not something the user did, which
   81 --        is like a panic;
   82 --      - it typically does not make sense to continue executing with
   83 --        a dummy value, which is like a panic;
   84 --      - they do not represent a corrupted internal state and there's
   85 --        no need to shut down, which is like an error.
   86 --
   87 --   The arguments of `unimplemented` are like `panic`. The first is
   88 --   the "location", generally the function in which something's
   89 --   missing, and the second is the message. The message is restricted
   90 --   to a single line -- that should be all that is needed to name
   91 --   what's missing, and there should be no need to print anything
   92 --   else. The offending location in the SAW source is printed via
   93 --   HasCallStack.
   94 --
   95 --   The @progress@ functions are for progress reporting. The idea is
   96 --   that the actual printing will be throttled by the infrastructure,
   97 --   so nothing happens until something's already taken a while. This
   98 --   will not work as intended in the `IO` instance, so widespread
   99 --   deployment needs to wait until we can get rid of it.
  100 --
  101 --   FUTURE: there is also eventually intended to be infrastructure in
  102 --   here for printing execution traces, and for dumping internal
  103 --   representations at key points.
  104 --
  105 --   Note that while we could provide workable default implementations
  106 --   for most of these functions, this would require some additional
  107 --   functions to use as building blocks, and we don't really want to
  108 --   expose those functions.
  109 --
  110 class (Monad m) => SAWConsole m where
  111     errN :: Text -> m a
  112     errP :: IsPosition pos => pos -> Text -> m a
  113     errX :: Text -> m a
  114 
  115     errDN :: Text -> m ()
  116     errDP :: IsPosition pos => pos -> Text -> m ()
  117     errDX :: Text -> m ()
  118 
  119     warnN :: Text -> m ()
  120     warnP :: IsPosition pos => pos -> Text -> m ()
  121     warnX :: Text -> m ()
  122 
  123     noteN :: Text -> m ()
  124     noteP :: IsPosition pos => pos -> Text -> m ()
  125     noteX :: Text -> m ()
  126 
  127     errN' :: PPS.Doc -> m a
  128     errP' :: IsPosition pos => pos -> PPS.Doc -> m a
  129     errX' :: PPS.Doc -> m a
  130 
  131     errDN' :: PPS.Doc -> m ()
  132     errDP' :: IsPosition pos => pos -> PPS.Doc -> m ()
  133     errDX' :: PPS.Doc -> m ()
  134 
  135     warnN' :: PPS.Doc -> m ()
  136     warnP' :: IsPosition pos => pos -> PPS.Doc -> m ()
  137     warnX' :: PPS.Doc -> m ()
  138 
  139     noteN' :: PPS.Doc -> m ()
  140     noteP' :: IsPosition pos => pos -> PPS.Doc -> m ()
  141     noteX' :: PPS.Doc -> m ()
  142 
  143     checkFail :: m ()
  144 
  145     unimplemented :: HasCallStack => Text -> Text -> m a
  146 
  147     progressStart :: Text -> m ()
  148     progressStep :: Text -> m ()
  149     progressAmount :: Int -> Maybe Int -> m ()
  150     progressInterim :: Text -> m ()
  151     progressInterim' :: PPS.Doc -> m ()
  152     progressEnd :: m ()

  156 -- IO instanace
  157 
  158 -- | Internal function for the `IO` instance.
  159 say :: Text -> IO ()
  160 say msg = TextIO.putStrLn msg
  161 
  162 -- | Dummy position used by the `IO` instance.
  163 newtype DummyPos = DummyPos ()
  164 instance IsPosition DummyPos where
  165     ppPosition (DummyPos ()) = "Unknown position"
  166 dummyPos :: DummyPos
  167 dummyPos = DummyPos ()
  168 
  169 -- | `SAWConsole` instance for `IO`. This does not (and cannot) handle
  170 --   deferred exit correctly, or the progress interface either, and is
  171 --   expected to be removed in the long run.
  172 --
  173 --   Also note that there are, properly, prettyprinter options that
  174 --   should get passed to `render`, which `IO` doesn't and can't
  175 --   track. And it doesn't know what the current execution position
  176 --   means.
  177 instance SAWConsole IO where
  178     errN msg = Supp.errN say msg >> throwIO (Fatal False)
  179     errP pos msg = Supp.errP say pos msg >> throwIO (Fatal False)
  180     -- XXX: for now throw a Fatal with True and, provided we're
  181     -- actually inside TopLevel somewhere, the exception glop in
  182     -- Value.hs will see it and print a stack trace. (If we aren't
  183     -- actually inside TopLevel, it won't and there'll be no stack
  184     -- trace.) This is the best we can do if all we have is `IO`. All
  185     -- of this should be removed: we want all error calls to happen in
  186     -- a monad that has access to state, and then the stack trace will
  187     -- be available to print before throwing as intended.
  188     --
  189     errX msg = Supp.errN say msg >> throwIO (Fatal True)
  190 
  191     -- XXX: These can't be handled correctly in IO; they need state.
  192     errDN msg = Supp.errN say msg >> throwIO (Fatal False)
  193     errDP pos msg = Supp.errP say pos msg >> throwIO (Fatal False)
  194     -- XXX: same as above
  195     errDX msg = Supp.errN say msg >> throwIO (Fatal True)
  196 
  197     warnN msg = Supp.warnN say msg
  198     warnP pos msg = Supp.warnP say pos msg
  199     warnX msg = Supp.warnP say dummyPos msg
  200 
  201     noteN msg = Supp.noteN say msg
  202     noteP pos msg = Supp.noteP say pos msg
  203     noteX msg = Supp.noteP say dummyPos msg
  204 
  205     errN' msg = Supp.errN' say PPS.defaultOpts msg >> throwIO (Fatal False)
  206     errP' pos msg = Supp.errP' say PPS.defaultOpts pos msg >> throwIO (Fatal False)
  207     -- XXX: same as above
  208     errX' msg = Supp.errN' say PPS.defaultOpts msg >> throwIO (Fatal True)
  209 
  210     -- XXX: These can't be handled correctly in IO; they need state.
  211     errDN' msg = Supp.errN' say PPS.defaultOpts msg >> throwIO (Fatal False)
  212     errDP' pos msg = Supp.errP' say PPS.defaultOpts pos msg >> throwIO (Fatal False)
  213     -- XXX: same as above
  214     errDX' msg = Supp.errN' say PPS.defaultOpts msg >> throwIO (Fatal True)
  215 
  216     warnN' msg = Supp.warnN' say PPS.defaultOpts msg
  217     warnP' pos msg = Supp.warnP' say PPS.defaultOpts pos msg
  218     warnX' msg = Supp.warnP' say PPS.defaultOpts dummyPos msg
  219 
  220     noteN' msg = Supp.noteN' say PPS.defaultOpts msg
  221     noteP' pos msg = Supp.noteP' say PPS.defaultOpts pos msg
  222     noteX' msg = Supp.noteP' say PPS.defaultOpts dummyPos msg
  223 
  224     checkFail = pure ()
  225 
  226     unimplemented loc msg = do
  227         let msglines = Supp.unimplementedMessage loc msg
  228         mapM_ say msglines
  229         throwIO (Fatal False)
  230 
  231     -- XXX: this implementation is a very weak placeholder
  232     progressStart phase = say phase
  233     progressStep phase = say phase
  234     progressAmount n mm = do
  235         let n' = Text.pack $ show n
  236             m' = case mm of
  237                 Nothing -> "unknown"
  238                 Just m -> Text.pack $ show m
  239         TextIO.putStr $ "(" <> n' <> " of " <> m' <> ")\r"
  240     progressInterim msg = do
  241         TextIO.putStr "\n"
  242         say msg
  243     progressInterim' msg = do
  244         TextIO.putStr "\n"
  245         say $ PPS.renderText PPS.defaultOpts msg
  246     progressEnd =
  247         TextIO.putStr "\n"
