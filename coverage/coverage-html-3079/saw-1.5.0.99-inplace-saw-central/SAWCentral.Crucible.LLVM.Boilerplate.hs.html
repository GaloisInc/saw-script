   26 sortByDeps :: [FunctionSkeleton] -> [FunctionSkeleton]
   27 sortByDeps skels = reverse $ (\(f, _, _) -> f) . fromVertex <$> Graph.topSort g
   28   where
   29     (g, fromVertex, _) = Graph.graphFromEdges $ adjacency <$> skels
   30     adjacency :: FunctionSkeleton -> (FunctionSkeleton, Text, [Text])
   31     adjacency s = (s, s ^. funSkelName, Set.toList $ s ^. funSkelCalls)

   33 preludeBoilerplate :: Bool -> ModuleSkeleton -> Text
   34 preludeBoilerplate True _mskel = "enable_experimental;\nMODULE_SKEL <- module_skeleton MODULE;\n\n"
   35 preludeBoilerplate False _mskel = "\n";
   36 
   37 globalsPreBoilerplate :: Bool -> ModuleSkeleton -> Text
   38 globalsPreBoilerplate True _mskel = "skeleton_globals_pre MODULE_SKEL;\n"
   39 globalsPreBoilerplate False _mskel = "\n"
   40 
   41 globalsPostBoilerplate :: Bool -> ModuleSkeleton -> Text
   42 globalsPostBoilerplate True _mskel = "skeleton_globals_post MODULE_SKEL;\n"
   43 globalsPostBoilerplate False _mskel = "\n"

   45 llvmTypeBoilerplate :: LLVM.Type -> Text
   46 llvmTypeBoilerplate (LLVM.PrimType (LLVM.Integer n)) = mconcat
   47   [ "(llvm_int ", Text.pack $ show n, ")"
   48   ]
   49 llvmTypeBoilerplate (LLVM.PrimType (LLVM.FloatType LLVM.Float)) = "llvm_float"
   50 llvmTypeBoilerplate (LLVM.PrimType (LLVM.FloatType LLVM.Double)) = "llvm_double"
   51 llvmTypeBoilerplate (LLVM.Alias (LLVM.Ident n)) = mconcat
   52   [ "(llvm_type \"%", Text.pack n, ")"
   53   ]
   54 llvmTypeBoilerplate _ = "undefined"
   55 
   56 typeBoilerplate :: TypeSkeleton -> Text
   57 typeBoilerplate t
   58   | t ^. typeSkelIsPointer
   59   , (s:_) <- t ^. typeSkelSizeGuesses
   60   = mconcat
   61     [ "(llvm_array "
   62     , Text.pack . show $ s ^. sizeGuessElems
   63     , " "
   64     , llvmTypeBoilerplate $ t ^. typeSkelLLVMType
   65     , ")"
   66     ]
   67   | otherwise = llvmTypeBoilerplate $ t ^. typeSkelLLVMType
   68 
   69 bodyBoilerplate :: Bool -> FunctionSkeleton -> Text -> Text
   70 bodyBoilerplate True _fskel skelName = mconcat
   71   [ "  prestate <- skeleton_prestate ", skelName, ";\n"
   72   , "  skeleton_exec prestate;\n"
   73   , "  poststate <- skeleton_poststate ", skelName,  " prestate;\n"
   74   ]
   75 bodyBoilerplate False fskel _ = mconcat
   76   [ mconcat $ uncurry argPrecondition <$> numberedArgs
   77   , "  llvm_execute_func ["
   78   , Text.intercalate ", " $ uncurry argTerm <$> numberedArgs
   79   , "];\n\n"
   80   , if | LLVM.PrimType LLVM.Void <- fskel ^. funSkelRet . typeSkelLLVMType -> ""
   81        | fskel ^. funSkelRet . typeSkelIsPointer -> "  llvm_return undefined;\n\n"
   82        | otherwise -> mconcat
   83          [ "  __return <- llvm_fresh_var \"(return ", fskel ^. funSkelName
   84          , ")\" ", typeBoilerplate $ fskel ^. funSkelRet, ";\n"
   85          , "  llvm_return (llvm_term __return);\n\n"
   86          ]
   87   , mconcat $ uncurry argPostcondition <$> numberedArgs
   88   ]
   89   where
   90     numberedArgs = zip [0..] $ fskel ^. funSkelArgs
   91     argName :: Int -> ArgSkeleton -> Text
   92     argName i a
   93       | Just n <- a ^. argSkelName = n
   94       | otherwise = "arg" <> Text.pack (show i)
   95     argPrecondition :: Int -> ArgSkeleton -> Text
   96     argPrecondition i a
   97       | a ^. argSkelType . typeSkelIsPointer
   98       , (s:_) <- a ^. argSkelType . typeSkelSizeGuesses
   99       , s ^. sizeGuessInitialized
  100       = mconcat
  101         [ "  ", argName i a, " <- llvm_fresh_var \"", argName i a, "\" ", typeBoilerplate (a ^. argSkelType), ";\n"
  102         , "  ", argName i a, "_ptr <- llvm_alloc ", typeBoilerplate (a ^. argSkelType), ";\n"
  103         , "  llvm_points_to ", argName i a, "_ptr (llvm_term ", argName i a, ");\n\n"
  104         ]
  105       | a ^. argSkelType . typeSkelIsPointer
  106       = mconcat
  107         [ "  ", argName i a, "_ptr <- llvm_alloc ", typeBoilerplate (a ^. argSkelType), ";\n\n"
  108         ]
  109       | otherwise
  110       = mconcat
  111         [ "  ", argName i a, " <- llvm_fresh_var \"", argName i a, "\" ", typeBoilerplate (a ^. argSkelType), ";\n\n"
  112         ]
  113     argPostcondition :: Int -> ArgSkeleton -> Text
  114     argPostcondition i a
  115       | a ^. argSkelType . typeSkelIsPointer
  116       = mconcat
  117         [ "  ", argName i a, "_post <- llvm_fresh_var \"", argName i a, "_post\" ", typeBoilerplate (a ^. argSkelType), ";\n"
  118         , "  llvm_points_to ", argName i a, "_ptr (llvm_term ", argName i a, "_post);\n\n"
  119         ]
  120       | otherwise = ""
  121     argTerm :: Int -> ArgSkeleton -> Text
  122     argTerm i a
  123       | a ^. argSkelType . typeSkelIsPointer
  124       = argName i a <> "_ptr"
  125       | otherwise
  126       = "(llvm_term " <> argName i a <> ")"

  128 functionBoilerplate :: Bool -> ModuleSkeleton -> FunctionSkeleton -> Text
  129 functionBoilerplate skelBuiltins mskel fskel = mconcat
  130   [ if skelBuiltins
  131     then mconcat [skelName, " <- function_skeleton MODULE_SKEL \"", name, "\";\n"]
  132     else ""
  133   , "let ", specName, " = do {\n"
  134   , "  ", globalsPreBoilerplate skelBuiltins mskel
  135   , bodyBoilerplate skelBuiltins fskel skelName
  136   , "  ", globalsPostBoilerplate skelBuiltins mskel
  137   , "};\n"
  138   , overrideName, " <- llvm_verify MODULE \"", name, "\" ["
  139   , Text.intercalate ", " $ (<>"_override") <$> (Set.toList $ fskel ^. funSkelCalls)
  140   , "] false ", specName, " z3;\n"
  141   ]
  142   where
  143     name = fskel ^. funSkelName
  144     specName = name <> "_spec"
  145     skelName = name <> "_skel"
  146     overrideName = name <> "_override"
  147 
  148 llvm_boilerplate :: FilePath -> ModuleSkeleton -> Bool -> TopLevel ()
  149 llvm_boilerplate path mskel skelBuiltins = do
  150   let fskels = sortByDeps . Map.elems $ mskel ^. modSkelFunctions
  151   liftIO . withFile path WriteMode $ \h -> Text.IO.hPutStrLn h $ mconcat
  152     [ preludeBoilerplate skelBuiltins mskel
  153     , Text.unlines $ functionBoilerplate skelBuiltins mskel <$> fskels
  154     ]
