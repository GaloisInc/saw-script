   42 -- | Optional rewrites to do when resolving a term
   43 data ResolveRewrite = ResolveRewrite {
   44   rrBasicSS :: Maybe (Simpset TheoremNonce),
   45   -- ^ Rewrite terms using these rewrites
   46 
   47   rrWhat4Eval :: Bool
   48   -- ^ Also simplify terms using What4 evaluation
   49 }
   50 
   51 -- | Don't do any rewriting
   52 noResolveRewrite :: ResolveRewrite
   53 noResolveRewrite = ResolveRewrite { rrBasicSS = Nothing, rrWhat4Eval = False }
   54 
   55 -- Convert a term to a What4 expression, trying to simplify it in the process.
   56 -- Currently this expects the terms to be either boolean or bit-vectors, as
   57 -- that's all we use it for.   It should be fairly straightforward to generalize
   58 -- if need be.
   59 resolveTerm ::
   60   Sym {- ^ Backend state -} ->
   61   Set VarIndex {- ^ Keep these opaque -} ->
   62   W4.BaseTypeRepr t  {- ^ Type of term -} ->
   63   ResolveRewrite {- ^ Optional rewrites to do on the term -} ->
   64   Term {- ^ Term to process -} ->
   65   IO (Crucible.RegValue Sym (Crucible.BaseToType t))
   66 resolveTerm sym unint bt rr tm =
   67   do
   68     st       <- sawCoreState sym
   69     let sc = saw_sc st
   70     checkType sc
   71     tm'      <- basicRewrite sc tm
   72     canFold <- isConstFoldTerm sc unint tm'
   73     case () of
   74       _ | canFold ->
   75           do -- Evaluate as constant
   76             modmap <- scGetModuleMap sc
   77             let v = Concrete.evalSharedTerm modmap mempty mempty tm'
   78             case bt of
   79               W4.BaseBoolRepr -> pure (W4.backendPred sym (Concrete.toBool v))
   80               W4.BaseBVRepr w -> W4.bvLit sym w (BV.mkBV w (Prim.unsigned (Concrete.toWord v)))
   81               _ -> fail "resolveTerm: expected `Bool` or bit-vector"
   82 
   83         | rrWhat4Eval rr ->
   84           do (_, _, _, p) <- w4EvalAny sym st sc mempty unint tm'
   85              case valueToSymExpr p of
   86                Just (Some y)
   87                  | Just Refl <- testEquality bt ty -> pure y
   88                  | otherwise -> typeError (show ty)
   89                  where ty = W4.exprType y
   90                _ -> fail ("resolveTerm: unexpected w4Eval result " ++ show p)
   91 
   92           -- Just bind the term
   93         | otherwise -> bindSAWTerm sym st bt tm'
   94 
   95   where
   96   basicRewrite sc =
   97     case rrBasicSS rr of
   98       Nothing -> pure
   99       Just ss -> \t -> snd <$> rewriteSharedTerm sc ss t
  100 
  101   checkType sc =
  102     do
  103       schema <- ttType <$> mkTypedTerm sc tm
  104       case ttIsMono schema of
  105           Just ty
  106             | tIsBit ty, W4.BaseBoolRepr <- bt -> pure ()
  107             | Just (n,el) <- (tIsSeq ty)
  108             , tIsBit el, Just i <- tIsNum n, W4.BaseBVRepr w <- bt
  109             , intValue w == i -> pure ()
  110             | otherwise -> typeError (Text.unpack (CryPP.pp ty)) :: IO ()
  111           Nothing -> do
  112             schema' <- prettyTypedTermType sc PPS.defaultOpts schema
  113             typeError (PPS.render PPS.defaultOpts schema')
  114 
  115   typeError :: String -> IO a
  116   typeError t = fail $ unlines [
  117     "Expected type: " ++ (
  118       case bt of
  119         W4.BaseBoolRepr  -> "Bit"
  120         W4.BaseBVRepr w  -> "[" ++ show w ++ "]"
  121         _             -> show bt),
  122     "Actual type:   " ++ t
  123     ]
  124 
  125 -- 'resolveTerm' specialized to booleans.
  126 resolveBoolTerm' :: Sym -> Set VarIndex -> ResolveRewrite -> Term -> IO (W4.Pred Sym)
  127 resolveBoolTerm' sym unint = resolveTerm sym unint W4.BaseBoolRepr
  128 
  129 -- 'resolveTerm' specialized to booleans, without rewriting.
  130 resolveBoolTerm :: Sym -> Set VarIndex -> Term -> IO (W4.Pred Sym)
  131 resolveBoolTerm sym unint = resolveBoolTerm' sym unint noResolveRewrite
  132 
  133 -- 'resolveTerm' specialized to bit-vectors.
  134 resolveBitvectorTerm' ::
  135   (1 W4.<= w) => Sym -> Set VarIndex -> W4.NatRepr w -> ResolveRewrite -> Term -> IO (W4.SymBV Sym w)
  136 resolveBitvectorTerm' sym unint w = resolveTerm sym unint (W4.BaseBVRepr w)
  137 
  138 -- 'resolveTerm' specialized to bit-vectors, without rewriting.
  139 resolveBitvectorTerm ::
  140   (1 W4.<= w) => Sym -> Set VarIndex -> W4.NatRepr w -> Term -> IO (W4.SymBV Sym w)
  141 resolveBitvectorTerm sym unint w = resolveBitvectorTerm' sym unint w noResolveRewrite
