   95 -- *** PointsTo

   97 overlapPointsTo :: JVMPointsTo -> JVMPointsTo -> Bool
   98 overlapPointsTo =
   99   \case
  100     JVMPointsToField _ p1 f1 _ ->
  101       \case
  102         JVMPointsToField _ p2 f2 _ -> p1 == p2 && f1 == f2
  103         _                          -> False
  104     JVMPointsToStatic _ f1 _ ->
  105       \case
  106         JVMPointsToStatic _ f2 _   -> f1 == f2
  107         _                          -> False
  108     JVMPointsToElem _ p1 i1 _ ->
  109       \case
  110         JVMPointsToElem _ p2 i2 _  -> p1 == p2 && i1 == i2
  111         JVMPointsToArray _ p2 _    -> p1 == p2
  112         _                          -> False
  113     JVMPointsToArray _ p1 _ ->
  114       \case
  115         JVMPointsToElem _ p2 _ _   -> p1 == p2
  116         JVMPointsToArray _ p2 _    -> p1 == p2
  117         _                          -> False
  118 
  119 prettyJVMPointsTo :: SharedContext -> PPS.Opts -> JVMPointsTo -> IO PPS.Doc
  120 prettyJVMPointsTo sc opts =
  121   \case
  122     JVMPointsToField _loc ptr fid val -> do
  123         let ptr' = MS.prettyAllocIndex ptr
  124             fid' = PP.pretty $ J.fieldIdName fid
  125         val' <- opt MS.prettySetupValue val
  126         pure $ ptr' <> "." <> fid' <+> "points to" <+> val'
  127     JVMPointsToStatic _loc fid val -> do
  128         let fidclass' = PP.pretty $ J.unClassName $ J.fieldIdClass fid
  129             fidname' = PP.pretty $ J.fieldIdName fid
  130         val' <- opt MS.prettySetupValue val
  131         pure $ fidclass' <> "." <> fidname' <+> "points to" <+> val'
  132     JVMPointsToElem _loc ptr idx val -> do
  133         let ptr' = MS.prettyAllocIndex ptr
  134             idx' = PP.pretty idx
  135         val' <- opt MS.prettySetupValue val
  136         pure $ ptr' <>  "[" <> idx' <> "]" <+> "points to" <+> val'
  137     JVMPointsToArray _loc ptr val -> do
  138         let ptr' = MS.prettyAllocIndex ptr
  139         val' <- opt prettyTypedTerm val
  140         pure $ ptr' <+> "points to" <+> val'
  141   where
  142     opt prt x = case x of
  143       Nothing -> pure "<unspecified>"
  144       Just x' -> prt sc opts x'

  162 initialDefCrucibleMethodSpecIR ::
  163   CB.Codebase ->
  164   J.ClassName ->
  165   J.Method ->
  166   ProgramLoc ->
  167   MS.CrucibleMethodSpecIR CJ.JVM
  168 initialDefCrucibleMethodSpecIR cb cname method loc =
  169   let methId = JVMMethodId (J.methodKey method) cname
  170       retTy = J.methodReturnType method
  171       argTys = thisType ++ J.methodParameterTypes method
  172   in MS.makeCrucibleMethodSpecIR methId argTys retTy loc cb
  173   where thisType = if J.methodIsStatic method then [] else [J.ClassType cname]
  174 
  175 initialCrucibleSetupState ::
  176   JVMCrucibleContext ->
  177   (J.Class, J.Method) ->
  178   ProgramLoc ->
  179   Setup.CrucibleSetupState CJ.JVM
  180 initialCrucibleSetupState cc (cls, method) loc =
  181   Setup.makeCrucibleSetupState () cc $
  182     initialDefCrucibleMethodSpecIR
  183       (cc ^. jccCodebase)
  184       (J.className cls)
  185       method
  186       loc

  200 {-
  201 -- | Represent `CrucibleMethodSpecIR` as a function term in SAW-Core.
  202 methodSpecToTerm :: SharedContext -> CrucibleMethodSpecIR -> MaybeT IO Term
  203 methodSpecToTerm sc spec =
  204       -- 1. fill in the post-state user variable holes with final
  205       -- symbolic state
  206   let _ppts = _csPointsTos $ _csPostState $ instantiateUserVars spec
  207       -- 2. get the free variables in post points to's (note: these
  208       -- should be contained in variables bound by pre-points-tos)
  209 
  210       -- 3. abstract the free variables in each post-points-to
  211 
  212       -- 4. put every abstracted post-points-to in a tuple
  213 
  214       -- 5. Create struct type with fields being names of free variables
  215 
  216       -- 6. Create a lambda term bound to a struct-typed variable that returns the tuple
  217   in lift $ scLambda sc undefined undefined undefined
  218 
  219 -- | Rewrite the `csPostPointsTos` to substitute the elements of the
  220 -- final symbolic state for the fresh variables created by the user in
  221 -- the post-state.
  222 instantiateUserVars :: CrucibleMethodSpecIR -> CrucibleMethodSpecIR
  223 instantiateUserVars _spec = undefined
