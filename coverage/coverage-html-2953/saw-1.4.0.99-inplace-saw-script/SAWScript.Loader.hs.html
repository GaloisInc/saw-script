   54 -- | Type shorthand for an operation that can return warnings and/or
   55 --   errors. On error it returns Left and a list of messages, at least
   56 --   one of which is an error and therefore fatal. On success, it
   57 --   returns a (possibly empty) list of messages, which are all
   58 --   warnings and not fatal, and a result of type a.
   59 --
   60 --   The position is a Maybe because errors from the parser come back
   61 --   with a position already stuffed into the message text, and we would
   62 --   like to not print another vacuous one along with it.
   63 --
   64 --   The text is a Doc because we've improved the parser so it sometimes
   65 --   prints tabular messages.
   66 --
   67 type WithMsgs a = Either [(Pos, PPS.Doc)] ([(Pos, PPS.Doc)], a)

   69 -- | Type shorthand for an include path.
   70 --
   71 --   The LHS is the directory the current file was included from
   72 --   (starts as ".") and the RHS is the user's search path.

   75 -- | Type shorthand for the set of includes already seen.
   76 --
   77 --   We pass it around as an IORef because it needs to accumulate
   78 --   as we go.
   79 type SeenSet = IORef (Set FilePath)

   81 -- | Construct an empty SeenSet.
   82 emptySeenSet :: IO SeenSet
   83 emptySeenSet = IORef.newIORef Set.empty

   85 -- | Check for membership in a SeenSet.
   86 seenSetMember :: FilePath -> SeenSet -> IO Bool
   87 seenSetMember x ref = do
   88     set <- IORef.readIORef ref
   89     pure $ Set.member x set
   90 
   91 -- | Insert into a SeenSet.
   92 seenSetInsert :: FilePath -> SeenSet -> IO ()
   93 seenSetInsert x ref = do
   94     set <- IORef.readIORef ref
   95     IORef.writeIORef ref (Set.insert x set)
   96     pure ()

   98 -- | Wrap statements in a pushd/popd pair.
   99 --
  100 --   See the notes in the interpreter for why we do this.
  101 --
  102 --   FUTURE: if we keep this thing, it would be nice to attach the
  103 --   source position of the include to the push and pop statements.
  104 --
  105 wrapDir :: FilePath -> IO [Stmt] -> IO [Stmt]
  106 wrapDir dir m = do
  107     stmts <- m
  108     let pos = PosInternal "pushd/popd derived from include"
  109     pure $ [StmtPushdir pos dir] ++ stmts ++ [StmtPopdir pos]

  111 -- | Read some SAWScript text, using the selected parser entry point.
  112 --
  113 --   Returns only in Either; caller is responsible for handling the
  114 --   resulting warnings and errors. We need it to be this way because
  115 --   processing the builtin table at startup has to be pure.
  116 --
  117 --   The FilePath argument is the name of the file the text comes
  118 --   from. If the text didn't come from a file, it should have a
  119 --   suitable placeholder string; that's what ends up in the source
  120 --   positions in the results.
  121 --
  122 --   The second text argument (which goes with the parser function) is
  123 --   the EOF token name passed to `prettyParseError`, which should
  124 --   generally be either "end of line" or "end of file".
  125 --
  126 readAny :: FilePath -> Text -> Text -> ([Token Pos] -> Either ParseError a) -> WithMsgs a
  127 readAny fileName str eofName parser =
  128     case lexSAW fileName eofName str of
  129         Left (_verbosity, pos, msg) ->
  130             Left [(pos, PP.pretty msg)]
  131         Right (tokens, optmsg) ->
  132             let msgs = case optmsg of
  133                   Nothing -> []
  134                   Just (Options.Error, _, _) ->
  135                       panic "readAny" ["Lexer returned an error in the warning slot"]
  136                   Just (_verbosity, pos, msg) ->
  137                       [(pos, PP.pretty msg)]
  138             in
  139             case parser tokens of
  140                 Left err ->
  141                     let (optpos, err') = prettyParseError eofName err
  142                         pos = case optpos of
  143                             Nothing ->
  144                                 -- Happy is unable to provide the
  145                                 -- position of an error at EOF. Here
  146                                 -- we have the input token list, so we
  147                                 -- can pick up after it. Use the
  148                                 -- trailing edge of the last token.
  149                                 case uncons (reverse tokens) of
  150                                     Nothing ->
  151                                         Range fileName 1 1 1 1
  152                                     Just (t, _) ->
  153                                         Pos.trailingPos $ tokPos t
  154                             Just p ->
  155                                 p
  156                     in
  157                     Left (msgs ++ [(pos, err')])
  158                 Right tree ->
  159                     -- Note: there's no such things as warnings in a
  160                     -- Happy parser, so there are no more messages to
  161                     -- add in this branch.
  162                     Right (msgs, tree)

  164 -- | Use the readAny result to panic if any messages were generated.
  165 panicOnMsgs :: Text -> WithMsgs a -> a
  166 panicOnMsgs whoAmI result =
  167   -- Properly, printing the positions with the errors should be done
  168   -- by the error infrastructure. However, the error infrastructure
  169   -- necessarily needs to run in `IO`, and we need to _not_ run in
  170   -- `IO` here because we're part of initializing the builtins table.
  171   -- Do it by hand instead. I don't think it's worth adding extra
  172   -- stuff to SAWConsole just to avoid needing this code.
  173   let pp (pos, msg) =
  174         let msg' = PP.pretty (PosSupport.ppPosition pos) <> ":" PP.<+> msg in
  175         PPS.renderText PPS.defaultOpts msg'
  176   in
  177   case result of
  178    Left errs ->
  179        panic whoAmI ("Unexpected errors:" : map pp errs)
  180    Right ([], tree) ->
  181        tree
  182    Right (warns, _) ->
  183        panic whoAmI ("Unexpected warnings:" : map pp warns)
  184 
  185 -- | Like `panicOnMsgs` but for typechecker results. XXX: the
  186 --   typechecker should issue its own messages; if not, it at least
  187 --   shouldn't be arbitrarily different.
  188 panicOnMsgs' :: Text -> (Either [(Pos, String)] a, [(Pos, String)]) -> a
  189 panicOnMsgs' whoAmI (errs_or_results, warns) =
  190     let pp (pos, msg) =
  191           PosSupport.ppPosition pos <> ": " <> Text.pack msg
  192     in
  193     case warns of
  194         [] -> case errs_or_results of
  195             Left errs -> panic whoAmI ("Unexpected errors: " : map pp errs)
  196             Right result -> result
  197         _ -> panic whoAmI ("Unexpected warnings: " : map pp warns)

  199 -- | Handle the readAny result in IO.
  200 --
  201 --   Add HasCallStack because if the panic happens we'll want to know
  202 --   where we came from. XXX: figure out how to get rid of the panic
  203 --   and remove HasCallStack again.
  204 dispatchMsgs :: HasCallStack => WithMsgs a -> IO a
  205 dispatchMsgs result =
  206     case result of
  207         Left errs -> do
  208             let pp (pos, msg) = Cons.errDP' pos msg
  209             mapM_ pp errs
  210             Cons.checkFail
  211             panic "dispatchMsgs" ["checkFail didn't fail"]
  212         Right (msgs, tree) -> do
  213             let pp (pos, msg) = Cons.warnP' pos msg
  214             mapM_ pp msgs
  215             pure tree
  216 
  217 -- | Handle a typechecker result in IO.
  218 --
  219 --   Add HasCallStack because if the panic happens we'll want to know
  220 --   where we came from. XXX: figure out how to get rid of the panic
  221 --   and remove HasCallStack again.
  222 dispatchMsgs' :: HasCallStack => (Either [(Pos, String)] a, [(Pos, String)]) -> IO a
  223 dispatchMsgs' (errs_or_result, warns) = do
  224     mapM_ (\(pos, msg) -> Cons.warnP pos $ Text.pack msg) warns
  225     case errs_or_result of
  226         Left errs -> do
  227             mapM_ (\(pos, msg) -> Cons.errDP pos $ Text.pack msg) errs
  228             Cons.checkFail
  229             panic "dispatchMsgs'" ["checkFail didn't fail"]
  230         Right tree ->
  231             pure tree

  234 readAnyPure :: FilePath -> Text -> Text -> ([Token Pos] -> Either ParseError a) -> Text -> a
  235 readAnyPure fileName str eofName parser whoAmI =
  236     panicOnMsgs whoAmI $ readAny fileName str eofName parser

  243 -- | Run the readAny result through the `Include` module to resolve

  246 --   If we fail after collecting warnings, print the warnings before

  254 -- | Read a type schema from a string. This is used to digest the type
  255 --   signatures for builtins, and the expansions for builtin typedefs.
  256 --
  257 -- Pure function that panics on any error or warning. This is
  258 -- appropriate since we use it only to handle the builtins; any
  259 -- glitches there are properly panics.
  260 --
  261 -- The first argument (fileName) is a string to pass as the
  262 -- filename for the lexer, which (complete with line and column
  263 -- numbering of dubious value) will go into the positions of the
  264 -- elements of the resulting type.
  265 --
  266 -- The fake file name names the builtin object involved, so we'll also
  267 -- use it as the panic location string if we need to panic.
  268 --
  269 -- FUTURE: we should figure out how to generate more meaningful
  270 -- positions (like "third argument of concat") but this at least
  271 -- allows telling the user which builtin the type came from.
  272 --
  273 -- Note: there is no way to reach @include@ statements from type
  274 -- schemas, so no need to process includes in what we read.
  275 --
  276 readSchemaPure ::
  277     FilePath ->
  278     PrimitiveLifecycle ->
  279     ScopedMap Name (PrimitiveLifecycle, NamedType) ->
  280     Text ->
  281     Schema
  282 readSchemaPure fakeFileName lc tyenv str =
  283     let schema = readAnyPure fakeFileName str "end-of-input" parseSchema (Text.pack fakeFileName) in
  284     panicOnMsgs' (Text.pack fakeFileName) $ checkSchema lc tyenv schema

  286 -- | Read a schema pattern from a string. This is used by the
  287 --   :search REPL command.
  288 --
  289 --   Also runs the typechecker to check the pattern.
  290 --
  291 -- Note: there is no way to reach @include@ statements from schema
  292 -- patterns, so no need to process includes in what we read.
  293 --
  294 readSchemaPattern ::
  295     Options ->
  296     FilePath -> Environ -> RebindableEnv -> Set PrimitiveLifecycle -> Text ->
  297     IO SchemaPattern
  298 readSchemaPattern _opts fileName environ rbenv avail str = do
  299   pat <- readAnyIO fileName str "end-of-line" parseSchemaPattern
  300   let Environ varenv tyenv _cryenv = environ
  301 
  302   -- XXX it should not be necessary to do this munging
  303   --
  304   -- Note that we need to flatten the scopes we have here
  305   -- (outside the typechecker) in order to have the union with
  306   -- the rebindable env to work right.
  307   let squash (pos, lc, ty, _val, _doc) = (pos, lc, ReadOnlyVar, ty)
  308       varenv' = Map.map squash $ ScopedMap.flatten varenv
  309       rbsquash (pos, ty, _val) = (pos, Current, RebindableVar, ty)
  310       rbenv' = Map.map rbsquash rbenv
  311       varenv'' = Map.union varenv' rbenv'
  312       varenv''' = ScopedMap.seed varenv''
  313 
  314   dispatchMsgs' $ checkSchemaPattern avail varenv''' tyenv pat

  316 -- | Read an expression from a string. This is used by the
  317 --   :type REPL command.
  318 --
  319 -- It is possible for someone to provide an expression that includes
  320 -- an @include@ in a do-block, so just in case call into `Include`
  321 -- to resolve any that appear.
  322 --
  323 readExpression ::
  324     Options ->
  325     FilePath -> Environ -> RebindableEnv -> Set PrimitiveLifecycle -> Text ->
  326     IO (Schema, Expr)
  327 readExpression opts fileName environ rbenv avail str = do
  328   seen <- emptySeenSet
  329   let incpath = (".", Options.importPath opts)
  330 
  331   expr0 <- readAnyIO fileName str "end-of-line" parseExpression
  332   expr <- resolveIncludes 0{-depth-} seen incpath opts Inc.processExpr expr0
  333   let Environ varenv tyenv _cryenvs = environ
  334 
  335   -- XXX it should not be necessary to do this munging
  336   let squash (defpos, lc, ty, _val, _doc) = (defpos, lc, ReadOnlyVar, ty)
  337       varenv' = Map.map squash $ ScopedMap.flatten varenv
  338       rbsquash (defpos, ty, _val) = (defpos, Current, RebindableVar, ty)
  339       rbenv' = Map.map rbsquash rbenv
  340       varenv'' = Map.union varenv' rbenv'
  341       varenv''' = ScopedMap.seed varenv''
  342 
  343   -- XXX: also it shouldn't be necessary to do this wrappery
  344   let pos = Pos.getPos expr
  345       decl = Decl pos (PWild pos Nothing) Nothing expr
  346 
  347   decl' <- dispatchMsgs' $ checkDecl avail varenv''' tyenv decl
  348 
  349   let expr' = dDef decl'
  350       schema = case dType decl' of
  351         Just sch -> sch
  352         Nothing ->
  353             -- If the typechecker didn't insert a type, it's bust,
  354             -- so panic. Not much point in printing the expression
  355             -- or position in panic, since it's what the user just
  356             -- typed.
  357             panic "readExpressionChecked" [
  358                 "Typechecker failed to produce a type"
  359             ]
  360 
  361   pure (schema, expr')

  376 -- | Find a file, potentially looking in a list of multiple search paths (as
  377 -- specified via the @SAW_IMPORT_PATH@ environment variable or
  378 -- @-i@/@--import-path@ command-line options). If the file was successfully
  379 -- found, return the full path. If not, fail by returning `Left`.
  380 --
  381 locateFile :: IncludePath -> FilePath -> IO FilePath
  382 locateFile (current, rawdirs) file = do
  383   -- Replace "." in the search path with the directory the current
  384   -- file came from, which might or might not be ".".
  385   let dirs = map adjust rawdirs
  386       adjust "." = current
  387       adjust dir = dir
  388 
  389   mfname <- findFile dirs file
  390   case mfname of
  391     Nothing -> do
  392         -- XXX make this a pp-doc instead
  393         let msg = Text.unlines ([
  394                       "Couldn't find file: " <> Text.pack file,
  395                       "  Searched in directories:"
  396                   ] ++ map (\p -> "    " <> Text.pack p) dirs)
  397         -- XXX pass in the include position
  398         Cons.errN msg
  399     Just fname ->
  400       -- NB: Normalise the path name. The default SAW_IMPORT_PATH contains ".",
  401       -- and the behavior of filepath's 'normalise' function is to prepend a
  402       -- search path to the front of the file path that is found, which can
  403       -- cause paths like "./foo.saw" to be returned. This looks ugly in error
  404       -- messages, where we would rather display "foo.saw" instead.
  405       return $ normalise fname

  407 -- | Load the 'Stmt's in a @.saw@ file.
  408 --   Doesn't run the typechecker (yet).
  409 includeFile ::
  410     Int -> SeenSet -> IncludePath -> Options -> FilePath -> Bool -> IO [Stmt]
  411 includeFile depth seen incpath opts fname once = do
  412   fname' <- locateFile incpath fname
  413   alreadySeen <- seenSetMember fname' seen 
  414   if depth > 128 then
  415       -- XXX pass in the include position as well
  416       Cons.errX $ "Maximum include depth exceeded"
  417   else if once && alreadySeen then do
  418       -- include_once and we've already seen this file
  419       Cons.noteN $ "Skipping already-included file \"" <>
  420                    Text.pack fname' <> "\""
  421       pure []
  422   else do
  423       seenSetInsert fname' seen
  424       let (_current, dirs) = incpath
  425           current' = takeDirectory fname'
  426           incpath' = (current', dirs)
  427 
  428       Cons.noteN $ "Loading file \"" <> Text.pack fname' <> "\""
  429       ftext <- TextIO.readFile fname'
  430 
  431       stmts <- wrapDir current' $ readAnyIO fname ftext "end-of-file" parseModule
  432       resolveIncludes (depth + 1) seen incpath' opts Inc.processStmts stmts

  434 -- | Find a file, potentially looking in a list of multiple search paths (as
  435 -- specified via the @SAW_IMPORT_PATH@ environment variable or
  436 -- @-i@/@--import-path@ command-line options). If the file was successfully
  437 -- found, load it. If not, fail by returning `Left`.
  438 --
  439 -- Doesn't run the typechecker (yet).
  440 findAndLoadFileUnchecked :: Options -> FilePath -> IO [Stmt]
  441 findAndLoadFileUnchecked opts fp = do
  442   let depth = 0
  443   seen <- emptySeenSet
  444   let incpath = (".", Options.importPath opts)
  445       once = False
  446   includeFile depth seen incpath opts fp once
