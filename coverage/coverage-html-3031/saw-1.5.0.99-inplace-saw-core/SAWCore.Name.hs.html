   79 -- Module Names ----------------------------------------------------------------
   80 
   81 newtype ModuleName = ModuleName Text
   82   deriving (Eq, Ord, Generic, TH.Lift)
   83 
   84 instance Hashable ModuleName -- automatically derived
   85 
   86 instance Show ModuleName where
   87   show (ModuleName s) = Text.unpack s
   88 
   89 
   90 moduleNameText :: ModuleName -> Text
   91 moduleNameText (ModuleName x) = x
   92 
   93 moduleNamePieces :: ModuleName -> [Text]
   94 moduleNamePieces (ModuleName x) = Text.splitOn "." x
   95 
   96 -- | Create a module name given a list of strings with the top-most
   97 -- module name given first.
   98 mkModuleName :: [Text] -> ModuleName
   99 mkModuleName nms = ModuleName $ Text.intercalate "." nms
  100 
  101 preludeName :: ModuleName
  102 preludeName = mkModuleName ["Prelude"]

  105 -- Identifiers -----------------------------------------------------------------
  106 
  107 data Ident =
  108   Ident
  109   { identModule :: ModuleName
  110   , identBaseName :: Text
  111   }
  112   deriving (Eq, Ord, Generic)
  113 
  114 instance Hashable Ident -- automatically derived
  115 
  116 instance Show Ident where
  117   show (Ident m s) = shows m ('.' : Text.unpack s)
  118 
  119 identText :: Ident -> Text
  120 identText i = moduleNameText (identModule i) <> Text.pack "." <> identBaseName i
  121 
  122 identPieces :: Ident -> NonEmpty Text
  123 identPieces i =
  124   case moduleNamePieces (identModule i) of
  125     [] -> identBaseName i :| []
  126     (x:xs) -> x :| (xs ++ [identBaseName i])
  127 
  128 identName :: Ident -> String
  129 identName = Text.unpack . identBaseName
  130 
  131 instance Read Ident where
  132   readsPrec _ str =
  133     let (str1, str2) = break (not . isIdChar) str in
  134     [(parseIdent str1, str2)]
  135 
  136 mkIdent :: ModuleName -> Text -> Ident
  137 mkIdent m s = Ident m s
  138 
  139 -- | Make a \"rocq-safe\" identifier from a string that might contain
  140 -- non-identifier characters, where we use the SAW core notion of identifier
  141 -- characters as letters, digits, underscore and primes. Any disallowed
  142 -- character is mapped to the string @__xNN@, where @NN@ is the hexadecimal code
  143 -- for that character. Additionally, a SAW core identifier is not allowed to
  144 -- start with a prime, so a leading underscore is added in such a case.
  145 mkSafeIdent :: ModuleName -> String -> Ident
  146 mkSafeIdent _ [] = fromString "_"
  147 mkSafeIdent mnm nm =
  148   let is_safe_char c = isAlphaNum c || c == '_' || c == '\'' in
  149   mkIdent mnm $ Text.pack $
  150   (if nm!!0 == '\'' then ('_' :) else id) $
  151   concatMap
  152   (\c -> if is_safe_char c then [c] else
  153            "__x" ++ showHex (ord c) "")
  154   nm
  155 
  156 -- | Parse a fully qualified identifier.
  157 parseIdent :: String -> Ident
  158 parseIdent s0 =
  159     case reverse (breakEach s0) of
  160       (_:[]) -> panic "parseIdent" ["Empty module name"]
  161       (nm:rMod) -> mkIdent (mkModuleName (reverse rMod)) nm
  162       _ -> panic "parseIdent" ["Bad identifier " <> Text.pack s0]
  163   where breakEach s =
  164           case break (=='.') s of
  165             (h, []) -> [Text.pack h]
  166             (h, _ : r) -> Text.pack h : breakEach r
  167 
  168 instance IsString Ident where
  169   fromString = parseIdent
  170 
  171 isIdent :: String -> Bool
  172 isIdent (c:l) = isAlpha c && all isIdChar l
  173 isIdent [] = False
  174 
  175 -- | Returns true if character can appear in identifier.
  176 isIdChar :: Char -> Bool
  177 isIdChar c = isAlphaNum c || (c == '_') || (c == '\'') || (c == '.')

  181 -- NameInfo
  182 
  183 
  184 -- | Descriptions of the origins of names that may be in scope
  185 data NameInfo
  186   = -- | This name arises from an exported declaration from a module
  187     ModuleIdentifier Ident
  188 
  189   | -- | This name was imported from some other programming language/scope
  190     ImportedName
  191       URI      -- ^ An absolutely-qualified name, which is required to be unique
  192       [Text]   -- ^ A collection of aliases for this name.  Shorter or "less-qualified"
  193                --   aliases should be nearer the front of the list
  194 
  195  deriving (Eq,Ord,Show)
  196 
  197 instance Hashable NameInfo where
  198   hashWithSalt x nmi =
  199     case nmi of
  200       ModuleIdentifier ident -> hashWithSalt x ident
  201       ImportedName uri _ -> hashWithSalt x uri
  202 
  203 nameURI :: NameInfo -> URI
  204 nameURI =
  205   \case
  206     ModuleIdentifier i -> moduleIdentToURI i
  207     ImportedName uri _ -> uri
  208 
  209 nameAliases :: NameInfo -> [Text]
  210 nameAliases =
  211   \case
  212     ModuleIdentifier i -> [identBaseName i, identText i]
  213     ImportedName _ aliases -> aliases
  214 
  215 toShortName :: NameInfo -> Text
  216 toShortName (ModuleIdentifier i) = identBaseName i
  217 toShortName (ImportedName uri []) = renderURI uri
  218 toShortName (ImportedName _ (x:_)) = x
  219 
  220 toAbsoluteName :: NameInfo -> Text
  221 toAbsoluteName (ModuleIdentifier i) = identText i
  222 toAbsoluteName (ImportedName uri _) = renderURI uri
  223 
  224 moduleIdentToURI :: Ident -> URI
  225 moduleIdentToURI ident = fromMaybe (panic "moduleIdentToURI" ["Failed to construct ident URI: " <> identText ident]) $
  226   mkURI NameSpaceCore (identPieces ident) (-1)
  227 
  228 scFreshNameURI :: Text -> VarIndex -> URI
  229 scFreshNameURI nm i = fromMaybe (panic "scFreshNameURI" ["Failed to construct name URI: <> " <> nm <> "  " <> Text.pack (show i)]) $
  230   mkURI NameSpaceFresh [(if Text.null nm then "_" else nm)] i
  231 
  232 -- Global Names ----------------------------------------------------------------
  233 
  234 type VarIndex = Int
  235 
  236 -- | A global name with a unique ID. We maintain a global invariant
  237 -- that the 'VarIndex' and the 'NameInfo' must be in a strict
  238 -- one-to-one correspondence: Each 'VarIndex' is paired with a unique
  239 -- 'NameInfo', and each 'NameInfo' is paired with a unique 'VarIndex'.
  240 data Name =
  241   Name
  242   { nameIndex :: !VarIndex
  243   , nameInfo :: !NameInfo
  244   }
  245   deriving (Show)
  246 
  247 -- | Because of the global uniqueness invariant, comparing the
  248 -- 'VarIndex' is sufficient to ensure equality of names.
  249 instance Eq Name where
  250   x == y = nameIndex x == nameIndex y
  251 
  252 instance Ord Name where
  253   compare x y = compare (nameIndex x) (nameIndex y)
  254 
  255 -- | For hashing, we consider only the 'NameInfo' and not the
  256 -- 'VarIndex'; this gives a stable hash value for a particular name,
  257 -- even if the unique IDs are assigned differently from run to run.
  258 instance Hashable Name where
  259   hashWithSalt x nm = hashWithSalt x (nameInfo nm)
  260

  262 -- Variable Names --------------------------------------------------------------
  263 
  264 -- | A variable name with a unique ID.
  265 -- We assume a global uniqueness invariant: The 'VarIndex' determines the name.
  266 data VarName =
  267   VarName
  268   { vnIndex :: !VarIndex
  269   , vnName :: !Text
  270   }
  271   deriving (Show)
  272 
  273 -- | Because of the global uniqueness invariant, comparing the
  274 -- 'VarIndex' is sufficient to ensure equality of variable names.
  275 instance Eq VarName where
  276   x == y = vnIndex x == vnIndex y
  277 
  278 instance Ord VarName where
  279   compare x y = compare (vnIndex x) (vnIndex y)
  280 
  281 -- | For hashing, we consider only the base name and not the
  282 -- 'VarIndex'; this gives a stable hash value for a particular name,
  283 -- even if the unique IDs are assigned differently from run to run.
  284 instance Hashable VarName where
  285   hashWithSalt x vn = hashWithSalt x (vnName vn)
  286 
  287 -- | A wildcard variable name consisting of an underscore "_".
  288 wildcardVarName :: VarName
  289 wildcardVarName = VarName 0 "_"
  290 
  291 -- | A data type representing a context of bound 'VarName's.
  292 -- It maps each 'VarName' to a de Bruijn index.
  293 data VarCtx =
  294   VarCtx
  295   !Int -- ^ Context size
  296   !(IntMap Int) -- ^ Mapping from VarIndex to (size - de Bruijn index)
  297 
  298 -- | The empty 'VarName' context.
  299 emptyVarCtx :: VarCtx
  300 emptyVarCtx = VarCtx 0 IntMap.empty
  301 
  302 -- | Extend a 'VarCtx' with a new bound 'VarName'.
  303 -- The new name has de Bruijn index 0; the index of each previous name
  304 -- is incremented by one.
  305 consVarCtx :: VarName -> VarCtx -> VarCtx
  306 consVarCtx x (VarCtx size m) =
  307   let size' = size + 1
  308   in VarCtx size' (IntMap.insert (vnIndex x) size' m)
  309 
  310 -- | Look up the de Bruijn index of the given 'VarName' in a 'VarCtx'.
  311 -- Return 'Nothing' if the name is not present in the context.
  312 lookupVarCtx :: VarName -> VarCtx -> Maybe Int
  313 lookupVarCtx x (VarCtx size m) =
  314   case IntMap.lookup (vnIndex x) m of
  315     Just i -> Just (size - i)
  316     Nothing -> Nothing

  319 -- Display Name Environments --------------------------------------------------------
  320 
  321 -- | A 'DisplayNameEnv' encodes the mappings between display names (type
  322 -- 'Text') and internal names (type 'VarIndex'). Multiple display
  323 -- names may be associated with each internal name; these are stored
  324 -- in priority order, with preferred display names first.
  325 data DisplayNameEnv =
  326   DisplayNameEnv
  327   { displayNames :: !(IntMap [Text]) -- Keyed by VarIndex; preferred names come first.
  328   , displayIndexes :: !(Map Text IntSet)
  329   }
  330 -- Invariants: The 'displayNames' and 'displayIndexes' maps should be
  331 -- inverses of each other. That is, 'displayNames' maps @i@ to a list
  332 -- containing @x@ if and only if 'displayIndexes' maps @x@ to a set
  333 -- containing @i@.
  334 
  335 emptyDisplayNameEnv :: DisplayNameEnv
  336 emptyDisplayNameEnv = DisplayNameEnv mempty mempty
  337 
  338 -- | Extend a 'DisplayNameEnv' by providing new 'Text' display names
  339 -- for a given 'VarIndex'. Display names should be provided in
  340 -- priority order, with preferred names first. If the 'VarIndex'
  341 -- already exists in the map, then add the new display names with
  342 -- higher priority while keeping the old ones.
  343 extendDisplayNameEnv :: VarIndex -> [Text] -> DisplayNameEnv -> DisplayNameEnv
  344 extendDisplayNameEnv i aliases env =
  345   DisplayNameEnv
  346   { displayNames = IntMap.insertWith List.union i aliases (displayNames env)
  347   , displayIndexes = foldr insertAlias (displayIndexes env) aliases
  348   }
  349   where
  350     insertAlias :: Text -> Map Text IntSet -> Map Text IntSet
  351     insertAlias x m = Map.insertWith IntSet.union x (IntSet.singleton i) m
  352 
  353 -- | Filter display names in a 'DisplayNameEnv' that satisfy a predicate.
  354 filterDisplayNameEnv :: (Text -> Bool) -> DisplayNameEnv -> DisplayNameEnv
  355 filterDisplayNameEnv p env =
  356   DisplayNameEnv
  357   { displayNames = IntMap.filter (not . null) $ fmap (filter p) $ displayNames env
  358   , displayIndexes = Map.filterWithKey (\k _ -> p k) $ displayIndexes env
  359   }
  360 
  361 -- | Merge two 'DisplayNameEnv's, giving higher priority to display
  362 -- names from the first argument.
  363 mergeDisplayNameEnv :: DisplayNameEnv -> DisplayNameEnv -> DisplayNameEnv
  364 mergeDisplayNameEnv env1 env2 =
  365   DisplayNameEnv
  366   { displayNames = IntMap.unionWith List.union (displayNames env1) (displayNames env2)
  367   , displayIndexes = Map.unionWith IntSet.union (displayIndexes env1) (displayIndexes env2)
  368   }
  369 
  370 -- | Look up a 'Text' display name in an environment, returning the
  371 -- list of indexes it could resolve to.
  372 resolveDisplayName :: DisplayNameEnv -> Text -> [VarIndex]
  373 resolveDisplayName env x =
  374   case Map.lookup x (displayIndexes env) of
  375     Nothing -> []
  376     Just s -> IntSet.elems s
  377 
  378 -- | Return the first display name for the given 'VarIndex' that is
  379 -- unambiguous in the naming environment. If there is no unambiguous
  380 -- alias, then return 'Nothing'.
  381 bestDisplayName :: DisplayNameEnv -> VarIndex -> Maybe Text
  382 bestDisplayName env i =
  383   do aliases <- IntMap.lookup i (displayNames env)
  384      go aliases
  385   where
  386     go [] = Nothing
  387     go (x : xs) =
  388       case Map.lookup x (displayIndexes env) of
  389         Nothing -> panic "bestDisplayName" ["Invariant violated: Missing key " <> x]
  390         Just vs
  391           | IntSet.size vs == 1 -> Just x
  392           | otherwise -> go xs
