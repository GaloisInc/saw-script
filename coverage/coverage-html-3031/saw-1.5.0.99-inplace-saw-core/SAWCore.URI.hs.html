   30 data URI = URI 
   31   { uriScheme :: Maybe Text 
   32   , uriAuthority :: Either Bool ()
   33   , uriPath :: Maybe (Bool, NonEmpty Text)
   34   , uriQuery :: [()]
   35   , uriFragment :: Maybe Text
   36   }
   37  deriving (Eq,Ord,Show)
   38 
   39 instance Hashable URI where
   40   hashWithSalt s (URI a b c d e) = s
   41     `hashWithSalt` a 
   42     `hashWithSalt` b
   43     `hashWithSalt` c
   44     `hashWithSalt` d
   45     `hashWithSalt` e

   47 mkScheme :: Text -> Maybe Text
   48 mkScheme nm = Just nm
   49 
   50 mkPathPiece :: Text -> Maybe Text
   51 mkPathPiece nm = Just nm
   52 
   53 mkFragment :: Text -> Maybe Text
   54 mkFragment nm = Just nm

   57 nonEmpty :: MonadFail m => Text -> m ()
   58 nonEmpty txt = case Text.null txt of
   59   True -> fail "nonEmpty"
   60   False -> return ()
   61 
   62 splitM :: MonadFail m => (Char -> Bool) -> Text -> m (NonEmpty Text)
   63 splitM f txt = do
   64   nonEmpty txt
   65   (p:ps) <- return $ Text.split f txt
   66   return $ p :| ps
   67 
   68 breakM :: MonadFail m => (Char -> Bool) -> Text -> m (Text,Text)
   69 breakM f txt = nonEmpty txt >> (return $ Text.break f txt)
   70 
   71 sepM :: MonadFail m => Char -> Text -> m (Maybe (Text, Text))
   72 sepM c txt = do
   73   (ns,rest) <- breakM (==c) txt
   74   nonEmpty ns
   75   case Text.uncons rest of
   76     Nothing -> return $ Nothing
   77     Just (_,rest') -> return $ Just (ns,rest')
   78 
   79 mkURI :: Text -> Maybe URI
   80 mkURI txt = do
   81   (mns, txt') <- sepM ':' txt >>= \case
   82     Nothing -> return (Nothing, txt)
   83     Just (ns,txt') -> return (Just ns, txt')
   84   (mf, txt'') <- sepM '#' txt' >>= \case
   85     Nothing -> return (Nothing, txt')
   86     Just (txt'', f) -> return (Just f, txt'')
   87   ps <- splitM (=='/') txt''
   88   ps' <- case ps of
   89     (p :| (p':ps')) | Text.null p -> return $ p' :| ps'
   90     _ -> return ps
   91   return $ URI mns (Left True) (Just (False, ps')) [] mf
   92 
   93 render :: URI -> Text
   94 render uri = schemePrefix uri <> pathBody uri <> fragmentSuffix uri
   95 
   96 schemePrefix :: URI -> Text
   97 schemePrefix uri = case uriScheme uri of
   98   Just s -> s <> ":"
   99   Nothing -> Text.empty
  100 
  101 fragmentSuffix :: URI -> Text
  102 fragmentSuffix uri = case uriFragment uri of
  103   Just f -> "#" <> f
  104   Nothing -> Text.empty
  105 
  106 pathBody :: URI -> Text
  107 pathBody uri = case uriPath uri of
  108   Just (_,p :| ps) -> case ps of
  109     [] -> p
  110     _ -> "/" <> Text.intercalate "/" (p:ps)
  111   Nothing -> Text.empty
