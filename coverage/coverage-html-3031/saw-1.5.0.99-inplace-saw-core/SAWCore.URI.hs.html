   15 module SAWCore.URI 
   16   ( NameSpace(..)
   17   , URI
   18   , mkPathURI
   19   , mkIdxURI
   20   , parseURI
   21   , renderURI
   22   ) where
   23 
   24 import qualified Data.Foldable as Foldable
   25 import           Data.Hashable
   26 import qualified Data.List as List
   27 import           Data.Map (Map)
   28 import qualified Data.Map as Map
   29 
   30 import           Data.Text (Text)
   31 import qualified Data.Text as Text
   32 import qualified Data.Text.Read as Text

   34 data NameSpace =
   35   NameSpaceCore | NameSpaceCryptol | NameSpaceFresh | NameSpaceYoSys | NameSpaceLLVM
   36   deriving (Eq, Ord, Enum, Bounded)
   37 
   38 instance Hashable NameSpace where
   39   hashWithSalt s ns = hashWithSalt s (fromEnum ns)
   40 
   41 renderNameSpace :: NameSpace -> Text
   42 renderNameSpace = \case
   43   NameSpaceCore -> "core"
   44   NameSpaceCryptol -> "cryptol"
   45   NameSpaceFresh -> "fresh"
   46   NameSpaceYoSys -> "yosys"
   47   NameSpaceLLVM -> "llvm"
   48 
   49 instance Show NameSpace where
   50   show ns = Text.unpack $ renderNameSpace ns
   51 
   52 nameSpaceMap :: Map Text NameSpace
   53 nameSpaceMap = Map.fromList $ map (\ns -> (renderNameSpace ns, ns)) [minBound..maxBound]
   54 
   55 readNameSpace :: MonadFail m => Text -> m NameSpace
   56 readNameSpace txt = case Map.lookup txt nameSpaceMap of
   57   Just ns -> pure ns
   58   Nothing -> fail $ "readNameSpace: namespace not found: " ++ Text.unpack txt
   59 
   60 parsePath :: MonadFail m => Text -> m ([Text], Text)
   61 parsePath txt
   62   | not (Text.null txt)
   63   , ps <- map percentDecode $ Text.splitOn "/" txt
   64   = case ps of
   65       -- allow for a leading '/'
   66       (p:ps') | Text.null p -> mkPath ps'
   67       _ -> mkPath ps
   68 parsePath txt =
   69   fail $ "parsePath: invalid path: " ++ (Text.unpack txt)
   70 
   71 data URI = URI 
   72   { uriNameSpace :: NameSpace
   73   , uriBaseName :: Text
   74   , uriPathIdx :: Either [Text] Int
   75   }
   76   deriving (Eq, Ord)
   77 
   78 uriQualifier :: URI -> [Text]
   79 uriQualifier uri = case uriPathIdx uri of
   80   Left ps -> ps
   81   Right _ -> []
   82 
   83 instance Hashable URI where
   84   hashWithSalt s (URI a b c) = s
   85     `hashWithSalt` a 
   86     `hashWithSalt` b
   87     `hashWithSalt` c
   88 
   89 percentEncode :: Text -> Text
   90 percentEncode txt = Text.replace "/" "%2F" $ Text.replace "%" "%25" txt
   91 
   92 percentDecode :: Text -> Text
   93 percentDecode txt = Text.replace "%25" "%" $ Text.replace "%2F" "/" txt
   94 
   95 mkPath :: (Foldable t, MonadFail m) => t Text -> m ([Text], Text)
   96 mkPath ps = case (List.any Text.null ps',ps') of
   97   (False, _:_) -> return (List.init ps',List.last ps')
   98   _ -> fail $ "mkPath: invalid path: " ++ show ps'
   99   where
  100     ps' = Foldable.toList ps
  101 
  102 mkPathURI :: (Foldable t, MonadFail m) => NameSpace -> t Text -> m URI
  103 mkPathURI ns ps = do
  104   (ps', nm) <- mkPath ps
  105   return $ URI ns nm (Left ps')
  106 
  107 mkIdxURI :: (MonadFail m) => NameSpace -> Text -> Int -> m URI
  108 mkIdxURI ns nm i = do
  109   ([],nm') <- mkPath [nm]
  110   return $ URI ns nm' (Right i)
  111 
  112 nonEmpty :: MonadFail m => String -> Text -> m ()
  113 nonEmpty msg txt = case Text.null txt of
  114   True -> fail msg
  115   False -> return ()
  116 
  117 splitM :: MonadFail m => Char -> Text -> m (Text,Text)
  118 splitM c txt = do
  119   nonEmpty err txt
  120   let (lhs, rhs) = Text.break (==c) txt
  121   nonEmpty err lhs
  122   case Text.uncons rhs of
  123     Nothing -> return (lhs, Text.empty)
  124     Just(_,rhs') -> return (lhs, rhs')
  125   where
  126     err = "splitM: failed to split on char '" ++ [c] ++ "' in '"
  127       ++ Text.unpack txt
  128 
  129 parseURI :: MonadFail m => Text -> m URI
  130 parseURI txt0 = do
  131   (ns, txt1) <- splitM ':' txt0
  132   ns' <- readNameSpace ns
  133   (si_rev, txt2_rev) <- splitM '#' (Text.reverse txt1)
  134   let si = Text.reverse si_rev
  135   (mi, txt2) <- case Text.null txt2_rev of
  136     False
  137       | Right (i,s) <- Text.decimal si
  138       , Text.null s
  139       -> pure $ (Just i, Text.reverse txt2_rev)
  140     _ -> pure (Nothing, txt1)
  141   case mi of
  142     Just i -> do
  143       ([],nm) <- parsePath txt2
  144       return $ URI ns' nm (Right i)
  145     Nothing -> do
  146       (ps,nm) <- parsePath txt2
  147       return $ URI ns' nm (Left ps)
  148 
  149 renderURI :: URI -> Text
  150 renderURI uri =
  151      renderNameSpace (uriNameSpace uri)
  152   <> ":" <> pathBody <> suffix
  153   where
  154     suffix :: Text
  155     suffix = case uriPathIdx uri of
  156       Left _ -> Text.empty
  157       Right i -> "#" <> Text.pack (show i)
  158 
  159     pathBody :: Text
  160     pathBody = case uriQualifier uri of
  161       [] -> percentEncode $ uriBaseName uri
  162       ps -> "/" <> Text.intercalate "/" (map percentEncode (ps ++ [uriBaseName uri]))
  163 
  164 instance Show URI where
  165   show uri = Text.unpack (renderURI uri)
