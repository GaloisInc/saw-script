   86 {- | SAWCore Values. Values are parameterized by the /name/ of an
   87 instantiation.
   88 The concrete parameters to use are computed from the name using
   89 a collection of type families (e.g., 'EvalM', 'VBool', etc.). -}
   90 data Value l
   91   = VFun !(Thunk l -> MValue l)
   92   | VUnit
   93   | VPair (Thunk l) (Thunk l) -- TODO: should second component be strict?
   94   | VCtorApp !Name !(TValue l) ![Thunk l] ![Thunk l]
   95   | VCtorMux ![Thunk l] !(IntMap (VBool l, Name, TValue l, [Thunk l]))
   96     -- ^ A mux tree of possible constructor values of a data type.
   97     -- The list of data type parameters is kept outside the mux.
   98     -- The 'IntMap' keys are 'VarIndex'es of each constructor name.
   99     -- The 'VBool' predicates must be mutually-exclusive and one
  100     -- must always be true.
  101   | VVector !(Vector (Thunk l))
  102   | VBool (VBool l)
  103   | VWord (VWord l)
  104   | VBVToNat !Int (Value l) -- TODO: don't use @Int@ for this, use @Natural@
  105   | VIntToNat (Value l)
  106   | VNat !Natural
  107   | VInt (VInt l)
  108   | VIntMod !Natural (VInt l)
  109   | VArray (VArray l)
  110   | VString !Text
  111   | VEmptyRecord
  112   | VRecordValue !FieldName (Thunk l) !(Value l) -- strict in spine of record
  113   | VExtra (Extra l)
  114   | TValue (TValue l)
  115 
  116 data VRecursor l
  117   = VRecursor
  118      !Name -- data type name
  119      !Int        -- number of index parameters
  120      !(Map VarIndex (Thunk l)) -- constructor eliminators
  121 
  122 -- | The subset of values that represent types.
  123 data TValue l
  124   = VVecType !Natural !(TValue l)
  125   | VBoolType
  126   | VIntType
  127   | VIntModType !Natural
  128   | VArrayType !(TValue l) !(TValue l)
  129   | VPiType !(TValue l) !(PiBody l)
  130   | VStringType
  131   | VUnitType
  132   | VPairType !(TValue l) !(TValue l)
  133   | VDataType !Name ![Value l] ![Value l] -- ^ name, parameters, indices
  134   | VEmptyRecordType
  135   | VRecordType !FieldName !(TValue l) !(TValue l)
  136   | VSort !Sort
  137   | VTyTerm !Sort !Term
  138 
  139 data PiBody l
  140   = VDependentPi !(Thunk l -> EvalM l (TValue l))
  141   | VNondependentPi !(TValue l)
  142 
  143 type Thunk l = Lazy (EvalM l) (Value l)
  144 
  145 -- | Evaluation monad for value instantiation 'l'
  146 type family EvalM l :: Type -> Type
  147 -- | Booleans for value instantiation 'l'
  148 type family VBool l :: Type
  149 -- | Words for value instantiation 'l'
  150 type family VWord l :: Type
  151 -- | Integers for value instantiation 'l'
  152 type family VInt  l :: Type
  153 -- | SMT arrays for value instantiation 'l'
  154 type family VArray l :: Type
  155 -- | Additional constructors for instantiation 'l'
  156 type family Extra l :: Type
  157 
  158 -- | Short-hand for a monadic value.
  159 type MValue l     = EvalM l (Value l)
  160 
  161 -- | Short-hand for a monadic boolean.
  162 type MBool  l     = EvalM l (VBool l)
  163 
  164 -- | Short-hand for a monadic word.
  165 type MWord l      = EvalM l (VWord l)
  166 
  167 -- | Short-hand for a monadic integer.
  168 type MInt l       = EvalM l (VInt  l)
  169 
  170 -- | Short-hand for a monadic array.
  171 type MArray l     = EvalM l (VArray l)
  172 
  173 -- | Short hand to specify that the evaluation monad is a monad (very common)
  174 type VMonad l     = Monad (EvalM l)
  175 
  176 -- | Short hand to specify that the evaluation monad is a lazy monad.
  177 type VMonadLazy l = MonadLazy (EvalM l)

  182 -- | Language instantiations with a specific monad.
  183 data WithM (m :: Type -> Type) l
  184 type instance EvalM (WithM m l) = m
  185 type instance VBool (WithM m l) = VBool l
  186 type instance VWord (WithM m l) = VWord l
  187 type instance VInt  (WithM m l) = VInt l
  188 type instance VArray (WithM m l) = VArray l
  189 type instance Extra (WithM m l) = Extra l

  193 instance Show (Extra l) => Show (Value l) where
  194   showsPrec p v =
  195     case v of
  196       VFun {}        -> showString "<<fun>>"
  197       VUnit          -> showString "()"
  198       VPair{}        -> showString "<<tuple>>"
  199       VCtorApp c _ty _ps _xv -> shows (toAbsoluteName (nameInfo c))
  200       VCtorMux {}    -> showString "<<constructor>>"
  201       VVector xv     -> showList (toList xv)
  202       VBool _        -> showString "<<boolean>>"
  203       VWord _        -> showString "<<bitvector>>"
  204       VBVToNat n x   -> showString "bvToNat " . shows n . showString " " . showParen True (shows x)
  205       VIntToNat x    -> showString "intToNat " . showParen True (shows x)
  206       VNat n         -> shows n
  207       VInt _         -> showString "<<integer>>"
  208       VIntMod n _    -> showString ("<<Z " ++ show n ++ ">>")
  209       VArray{}       -> showString "<<array>>"
  210       VString s      -> shows s
  211       VEmptyRecord   -> showString "{}"
  212       VRecordValue fld _ _ ->
  213         showString "{" . showString (Text.unpack fld) . showString " = _, ...}"
  214       VExtra x       -> showsPrec p x
  215       TValue x       -> showsPrec p x
  216     where
  217       toList = map (const Nil) . V.toList
  218 
  219 instance Show (Extra l) => Show (TValue l) where
  220   showsPrec p v =
  221     case v of
  222       VBoolType      -> showString "Bool"
  223       VStringType    -> showString "String"
  224       VIntType       -> showString "Integer"
  225       VIntModType n  -> showParen True (showString "IntMod " . shows n)
  226       VArrayType{}   -> showString "Array"
  227       VPiType t _    -> showParen True
  228                         (shows t . showString " -> ...")
  229       VUnitType      -> showString "#()"
  230       VPairType x y  -> showParen True (shows x . showString " * " . shows y)
  231       VDataType s ps vs
  232         | null (ps++vs) -> shows s
  233         | otherwise  -> shows s . showList (ps++vs)
  234       VEmptyRecordType -> showString "{}"
  235       VRecordType fld _ _ ->
  236         showString "{" . showString (Text.unpack fld) . showString " :: _, ...}"
  237       VVecType n a   -> showString "Vec " . shows n
  238                         . showString " " . showParen True (showsPrec p a)
  239       VSort s        -> shows s
  240 
  241       VTyTerm _ tm   -> showString "TyTerm (" . (\x -> ppTermPureDefaults tm ++ x) . showString ")"
  242

  245 instance Show Nil where
  246   show Nil = "_"

  251 -- | Create a 'Value' for a strict function.
  252 vStrictFun :: VMonad l => (Value l -> MValue l) -> Value l
  253 vStrictFun k = VFun $ \x -> force x >>= k
  254 
  255 -- | Create a 'Value' for a lazy multi-argument function.
  256 vFunList :: forall l. VMonad l => Int -> ([Thunk l] -> MValue l) -> MValue l
  257 vFunList n0 k = go n0 []
  258   where
  259     go :: Int -> [Thunk l] -> MValue l
  260     go 0 args = k (reverse args)
  261     go n args = pure $ VFun (\x -> go (n - 1) (x : args))
  262 
  263 -- | Create a 'Value' for a strict multi-argument function.
  264 vStrictFunList :: forall l. VMonad l => Int -> ([Value l] -> MValue l) -> MValue l
  265 vStrictFunList n0 k = go n0 []
  266   where
  267     go :: Int -> [Value l] -> MValue l
  268     go 0 args = k (reverse args)
  269     go n args = pure $ vStrictFun $ \v -> go (n - 1) (v : args)
  270 
  271 vTuple :: VMonad l => [Thunk l] -> Value l
  272 vTuple [] = VUnit
  273 vTuple [_] = error "vTuple: unsupported 1-tuple"
  274 vTuple [x, y] = VPair x y
  275 vTuple (x : xs) = VPair x (ready (vTuple xs))
  276 
  277 vTupleType :: VMonad l => [TValue l] -> TValue l
  278 vTupleType [] = VUnitType
  279 vTupleType [t] = t
  280 vTupleType (t : ts) = VPairType t (vTupleType ts)
  281 
  282 valPairLeft :: (HasCallStack, VMonad l, Show (Extra l)) => Value l -> MValue l
  283 valPairLeft (VPair t1 _) = force t1
  284 valPairLeft v = panic "valPairLeft" ["Not a pair value: " <> Text.pack (show v)]
  285 
  286 valPairRight :: (HasCallStack, VMonad l, Show (Extra l)) => Value l -> MValue l
  287 valPairRight (VPair _ t2) = force t2
  288 valPairRight v = panic "valPairRight" ["Not a pair value: " <> Text.pack (show v)]
  289 
  290 vRecord :: Map FieldName (Thunk l) -> Value l
  291 vRecord m = foldr (\(f, t) -> VRecordValue f t) VEmptyRecord (Map.assocs m)
  292 
  293 valRecordProj :: (HasCallStack, VMonad l, Show (Extra l)) => Value l -> FieldName -> MValue l
  294 valRecordProj v fld = go v
  295   where
  296     go (VRecordValue fld1 t1 v1)
  297       | fld == fld1 = force t1
  298       | otherwise = go v1
  299     go VEmptyRecord =
  300       panic "valRecordProj"
  301       [ "Record field " <> Text.pack (show fld) <> " not found in value: " <>
  302         Text.pack (show v)
  303       ]
  304     go _ =
  305       panic "valRecordProj" ["Not a record value: " <> Text.pack (show v)]
  306 
  307 apply :: (HasCallStack, VMonad l, Show (Extra l)) => Value l -> Thunk l -> MValue l
  308 apply (VFun f) x = f x
  309 apply (TValue (VPiType _ body)) x = TValue <$> applyPiBody body x
  310 
  311 apply v _x = panic "apply" ["Not a function value: " <> Text.pack (show v)]
  312 
  313 applyAll :: (VMonad l, Show (Extra l)) => Value l -> [Thunk l] -> MValue l
  314 applyAll = foldM apply
  315 
  316 {-# INLINE applyPiBody #-}
  317 applyPiBody :: VMonad l => PiBody l -> Thunk l -> EvalM l (TValue l)
  318 applyPiBody (VDependentPi f) x    = f x
  319 applyPiBody (VNondependentPi t) _ = pure t

  321 -- | Return the 'FiniteType' corresponding to the given 'Value', if
  322 -- one exists.
  323 -- If @asFiniteTypeValue v = Just t@, then the term returned by
  324 -- 'SAWCore.FiniteValue.scFiniteType' applied to @t@ should evaluate
  325 -- to @v@.
  326 asFiniteTypeValue :: Value l -> Maybe FiniteType
  327 asFiniteTypeValue v =
  328   case v of
  329     TValue tv -> asFiniteTypeTValue tv
  330     _ -> Nothing
  331 
  332 -- | Return the 'FiniteType' corresponding to the given 'TValue', if
  333 -- one exists.
  334 -- If @asFiniteTypeValue tv = Just t@, then the term returned by
  335 -- 'SAWCore.FiniteValue.scFiniteType' applied to @t@ should evaluate
  336 -- to @TValue tv@.
  337 asFiniteTypeTValue :: TValue l -> Maybe FiniteType
  338 asFiniteTypeTValue v =
  339   case v of
  340     VBoolType -> return FTBit
  341     VVecType n v1 -> do
  342       t1 <- asFiniteTypeTValue v1
  343       return (FTVec n t1)
  344     VUnitType -> return (FTTuple [])
  345     VPairType v1 v2 -> do
  346       t1 <- asFiniteTypeTValue v1
  347       t2 <- asFiniteTypeTValue v2
  348       case t2 of
  349         FTTuple ts -> return (FTTuple (t1 : ts))
  350         _ -> return (FTTuple [t1, t2])
  351     VEmptyRecordType -> Just (FTRec Map.empty)
  352     VRecordType fname v1 v2 ->
  353       do t1 <- asFiniteTypeTValue v1
  354          t2 <- asFiniteTypeTValue v2
  355          -- scFiniteType only produces nested record types with field
  356          -- names in strictly increasing order.
  357          -- In the case of duplicate or non-sorted field names, this
  358          -- TValue does not correspond to any canonical record type,
  359          -- so we return Nothing.
  360          case t2 of
  361            FTRec tm | lessThanKeys fname tm -> Just (FTRec (Map.insert fname t1 tm))
  362            _ -> Nothing
  363     VStringType   -> Nothing
  364     VPiType{}     -> Nothing
  365     VDataType{}   -> Nothing
  366     VSort{}       -> Nothing
  367     VTyTerm{}     -> Nothing
  368     VIntType      -> Nothing
  369     VIntModType{} -> Nothing
  370     VArrayType{}  -> Nothing
  371   where
  372 
  373 asFirstOrderTypeValue :: Value l -> Maybe FirstOrderType
  374 asFirstOrderTypeValue v =
  375   case v of
  376     TValue tv -> asFirstOrderTypeTValue tv
  377     _ -> Nothing
  378 
  379 asFirstOrderTypeTValue :: TValue l -> Maybe FirstOrderType
  380 asFirstOrderTypeTValue v =
  381   case v of
  382     VBoolType     -> return FOTBit
  383     VVecType n v1 -> FOTVec n <$> asFirstOrderTypeTValue v1
  384     VIntType      -> return FOTInt
  385     VIntModType m -> return (FOTIntMod m)
  386     VArrayType a b ->
  387       FOTArray <$> asFirstOrderTypeTValue a <*> asFirstOrderTypeTValue b
  388     VUnitType -> return (FOTTuple [])
  389     VPairType v1 v2 -> do
  390       t1 <- asFirstOrderTypeTValue v1
  391       t2 <- asFirstOrderTypeTValue v2
  392       case t2 of
  393         FOTTuple ts -> return (FOTTuple (t1 : ts))
  394         _ -> return (FOTTuple [t1, t2])
  395     VEmptyRecordType -> Just (FOTRec Map.empty)
  396     VRecordType fname v1 v2 ->
  397       do t1 <- asFirstOrderTypeTValue v1
  398          t2 <- asFirstOrderTypeTValue v2
  399          -- scFirstOrderType only produces nested record types with
  400          -- field names in strictly increasing order.
  401          -- In the case of duplicate or non-sorted field names, this
  402          -- TValue does not correspond to any canonical record type,
  403          -- so we return Nothing.
  404          case t2 of
  405            FOTRec tm | lessThanKeys fname tm -> Just (FOTRec (Map.insert fname t1 tm))
  406            _ -> Nothing
  407 
  408     VStringType -> Nothing
  409     VPiType{}   -> Nothing
  410     VDataType{} -> Nothing
  411     VSort{}     -> Nothing
  412     VTyTerm{}   -> Nothing
  413 
  414 -- | Is the given key less than all keys in the given map?
  415 lessThanKeys :: (Ord k) => k -> Map k a -> Bool
  416 lessThanKeys k m =
  417   case Map.minViewWithKey m of
  418     Just ((k', _), _) -> k < k'
  419     Nothing -> True

  421 -- | A (partial) injective mapping from type values to strings. These
  422 -- are intended to be useful as suffixes for names of type instances
  423 -- of uninterpreted constants.
  424 suffixTValue :: TValue sym -> Maybe String
  425 suffixTValue tv =
  426   case tv of
  427     VVecType n a ->
  428       do a' <- suffixTValue a
  429          Just ("_Vec_" ++ show n ++ a')
  430     VBoolType -> Just "_Bool"
  431     VIntType -> Just "_Int"
  432     VIntModType n -> Just ("_IntMod_" ++ show n)
  433     VArrayType a b ->
  434       do a' <- suffixTValue a
  435          b' <- suffixTValue b
  436          Just ("_Array" ++ a' ++ b')
  437     VPiType _ _ -> Nothing
  438     VUnitType -> Just "_Unit"
  439     VPairType a b ->
  440       do a' <- suffixTValue a
  441          b' <- suffixTValue b
  442          Just ("_Pair" ++ a' ++ b')
  443 
  444     VStringType -> Nothing
  445     VDataType {} -> Nothing
  446     VEmptyRecordType -> Nothing
  447     VRecordType {} -> Nothing
  448     VSort {} -> Nothing
  449     VTyTerm{} -> Nothing
