   50 -- | Class of types that can be uniformly exported as JSON
   51 -- using the Heapster pretty-printing information for names
   52 class JsonExport a where
   53     jsonExport :: (?ppi::PPInfo) => a -> Value
   54     default jsonExport :: ToJSON a => (?ppi::PPInfo) => a -> Value
   55     jsonExport = toJSON
   56 
   57 
   58 -- This code generates generic JSON generation instances for
   59 -- algebraic data types.
   60 --
   61 -- - All instances will generate an object.
   62 -- - The object will have a @tag@ field containing the name
   63 --   of the constructor used.
   64 -- - Record constructors will add each record field to the
   65 --   object using the field name
   66 -- - Normal constructors with fields will have a field called
   67 --   @contents@. If this constructor has more than one parameter
   68 --   the @contents@ field will have a list. Otherwise it will
   69 --   have a single element.
   70 let fields :: String -> TH.ConstructorVariant -> [TH.ExpQ] -> TH.ExpQ
   71 
   72     -- Record constructor, use record field names as JSON field names
   73     fields tag (TH.RecordConstructor fieldNames) xs =
   74         TH.listE
   75           $ [| ("tag", tag) |]
   76           : [ [| (n, $x) |] | n <- TH.nameBase <$> fieldNames | x <- xs]
   77 
   78     -- No fields, so just report the constructor tag
   79     fields tag _ []  = [| [("tag", tag)] |]
   80 
   81     -- One field, just report that field as @contents@
   82     fields tag _ [x] = [| [("tag", tag), ("contents", $x)] |]
   83 
   84     -- Multiple fields, report them as a list as @contents@
   85     fields tag _ xs  = [| [("tag", tag), ("contents", Array $(TH.listE xs))] |]
   86 
   87     clauses :: TH.DatatypeInfo -> [TH.ClauseQ]
   88     clauses info =
   89         [do fieldVars <- for [0..length (TH.constructorFields con)-1] $ \i ->
   90                             TH.newName ("x" ++ show i)
   91             TH.clause
   92               [TH.conP (TH.constructorName con) (TH.varP <$> fieldVars)]
   93               (TH.normalB [|
   94                 object
   95                     $(fields
   96                         (TH.nameBase (TH.constructorName con))
   97                         (TH.constructorVariant con)
   98                         [ [| jsonExport $(TH.varE v) |] | v <- fieldVars ]) |])
   99               []
  100         | con <- TH.datatypeCons info ]
  101 
  102     generateJsonExport :: TH.Name -> TH.DecQ
  103     generateJsonExport n =
  104       do info <- TH.reifyDatatype n
  105          let t = foldl TH.appT (TH.conT n)
  106                $ zipWith (\c _ -> TH.varT (TH.mkName [c])) ['a'..]
  107                $ TH.datatypeInstTypes info
  108          TH.instanceD (TH.cxt []) [t|JsonExport $t|]
  109            [TH.funD 'jsonExport (clauses info)]
  110 
  111     typesNeeded :: [TH.Name]
  112     typesNeeded =
  113         [''AtomicPerm, ''BaseTypeRepr, ''BoolRepr, ''BVFactor, ''BVProp,
  114         ''BVRange, ''CruCtx, ''FloatInfoRepr, ''FloatPrecisionRepr,
  115         ''FnHandle, ''FunPerm, ''LLVMArrayBorrow,
  116         ''LLVMArrayIndex, ''LLVMArrayPerm, ''LLVMBlockPerm, ''LLVMFieldPerm,
  117         ''LLVMFieldShape, ''NamedPermName, ''NamedShape,
  118         ''NamedShapeBody, ''NameReachConstr, ''NameSortRepr, ''NatRepr,
  119         ''PermExpr, ''PermOffset, ''StringInfoRepr, ''SymbolRepr, ''TypeRepr,
  120         ''ValuePerm, ''RWModality, ''PermImpl1, ''Member, ''SimplImpl,
  121         ''VarAndPerm, ''LocalPermImpl, ''LifetimeFunctor, ''NamedPerm,
  122         ''RecPerm, ''OpaquePerm, ''DefinedPerm, ''ReachMethods, ''MbPermImpls,
  123         ''ExprAndPerm, ''OrListDisj, ''EndianForm
  124         ]
  125 
  126  in traverse generateJsonExport typesNeeded

  138 instance JsonExport b => JsonExport (Mb (a :: RList CrucibleType) b) where
  139     jsonExport mb = mbLift $ flip nuMultiWithElim1 mb $ \names body ->
  140         let ?ppi = ppInfoAddExprNames "x" names ?ppi in
  141         object [
  142             ("args", jsonExport names),
  143             ("body", jsonExport body)
  144         ]

  179 -- Custom instance avoids the polymorphic field on the Done case
  180 instance JsonExport (PermImpl f ps) where
  181     jsonExport (PermImpl_Done _eq) =
  182         object [("tag", "PermImpl_Done")]
  183     jsonExport (PermImpl_Step rs mb) =
  184         object
  185             [("tag", "PermImpl_Step"),
  186              ("contents", Array
  187                 [jsonExport rs,
  188                  jsonExport mb])]
