    1 {-# Language ScopedTypeVariables #-}
    2 {-# Language GADTs #-}
    3 {-# Language TypeOperators #-}
    4 module Heapster.ParsedCtx where
    5 
    6 import Data.Functor.Constant
    7 
    8 import Data.Binding.Hobbits
    9 
   10 import qualified Data.Type.RList as RL
   11 
   12 import Data.Parameterized.Some (Some(Some))
   13 
   14 import Lang.Crucible.Types
   15 
   16 import Heapster.CruUtil
   17 
   18 -- | A sequence of variable names and their types
   19 data ParsedCtx ctx = ParsedCtx {
   20   parsedCtxNames :: RAssign (Constant String) ctx,
   21   parsedCtxCtx :: CruCtx ctx
   22   }
   23 
   24 -- | Remove the last variable in a 'ParsedCtx'
   25 parsedCtxUncons :: ParsedCtx (ctx :> tp) -> ParsedCtx ctx
   26 parsedCtxUncons (ParsedCtx (xs :>: _) (CruCtxCons ctx _)) = ParsedCtx xs ctx
   27 
   28 -- | Add a variable name and type to a 'ParsedCtx'
   29 consParsedCtx :: String -> TypeRepr tp -> ParsedCtx ctx ->
   30                  ParsedCtx (ctx :> tp)
   31 consParsedCtx x tp (ParsedCtx xs ctx) =
   32   ParsedCtx (xs :>: Constant x) (CruCtxCons ctx tp)
   33 
   34 -- | An empty 'ParsedCtx'
   35 emptyParsedCtx :: ParsedCtx RNil
   36 emptyParsedCtx = ParsedCtx MNil CruCtxNil
   37 
   38 -- | A 'ParsedCtx' with a single element
   39 singletonParsedCtx :: String -> TypeRepr tp -> ParsedCtx (RNil :> tp)
   40 singletonParsedCtx x tp =
   41   ParsedCtx (MNil :>: Constant x) (CruCtxCons CruCtxNil tp)
   42 
   43 -- | Append two 'ParsedCtx's
   44 appendParsedCtx :: ParsedCtx ctx1 -> ParsedCtx ctx2 ->
   45                    ParsedCtx (ctx1 :++: ctx2)
   46 appendParsedCtx (ParsedCtx ns1 ctx1) (ParsedCtx ns2 ctx2) =
   47   ParsedCtx (RL.append ns1 ns2) (appendCruCtx ctx1 ctx2)
   48 
   49 -- | Add a variable name and type to the beginning of an unknown 'ParsedCtx'
   50 preconsSomeParsedCtx :: String -> Some TypeRepr -> Some ParsedCtx ->
   51                         Some ParsedCtx
   52 preconsSomeParsedCtx x (Some (tp :: TypeRepr tp)) (Some (ParsedCtx ns tps)) =
   53   Some $ ParsedCtx
   54   (RL.append (MNil :>: (Constant x :: Constant String tp)) ns)
   55   (appendCruCtx (singletonCruCtx tp) tps)
   56 
   57 -- | Make a 'ParsedCtx' where the string names are @"arg0,arg1,..."@
   58 mkArgsParsedCtx :: CruCtx ctx -> ParsedCtx ctx
   59 mkArgsParsedCtx = mkPrefixParsedCtx "arg"
   60 
   61 mkPrefixParsedCtx :: String -> CruCtx ctx -> ParsedCtx ctx
   62 mkPrefixParsedCtx prefix ctx = ParsedCtx (mkPrefixParsedCtx' prefix ctx) ctx
   63 
   64 mkPrefixParsedCtx' :: String -> CruCtx ctx -> RAssign (Constant String) ctx
   65 mkPrefixParsedCtx' _ CruCtxNil = MNil
   66 mkPrefixParsedCtx' prefix (CruCtxCons ctx _) =
   67   mkPrefixParsedCtx' prefix ctx :>: Constant (prefix ++ show (cruCtxLen ctx))
   68 
   69 -- | Change the type of the last element of a 'ParsedCtx'
   70 parsedCtxSetLastType :: TypeRepr tp -> ParsedCtx (ctx :> tp') ->
   71                         ParsedCtx (ctx :> tp)
   72 parsedCtxSetLastType tp (ParsedCtx (xs :>: Constant str) (CruCtxCons ctx _)) =
   73   (ParsedCtx (xs :>: Constant str) (CruCtxCons ctx tp))
   74 
   75 -- | Extract out the last element of a 'ParsedCtx' as a singleton 'ParsedCtx'
   76 parsedCtxLast :: ParsedCtx (ctx :> tp) -> ParsedCtx (RNil :> tp)
   77 parsedCtxLast (ParsedCtx (_ :>: Constant str) (CruCtxCons _ tp)) =
   78   ParsedCtx (MNil :>: Constant str) (CruCtxCons CruCtxNil tp)
