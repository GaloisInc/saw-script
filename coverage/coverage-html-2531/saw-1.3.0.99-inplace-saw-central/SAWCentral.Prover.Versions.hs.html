    9 data Version
   10   = Version [Int]
   11   deriving (Eq, Ord, Show)
   12 
   13 parseVersion :: String -> Version
   14 parseVersion = Version . fixup . go [[]]
   15   where
   16     go (n : ns) (c : cs) | '0' <= c && c <= '9' = go ((c : n) : ns) cs
   17     go (n : ns) ('.' : cs) = go ([] : n : ns) cs
   18     go ns _ = ns
   19     fixup = reverse . catMaybes . map (readMaybe . reverse)
   20 
   21 prettyVersion :: Version -> String
   22 prettyVersion (Version xs) = intercalate "." (map show xs)
   23 
   24 parseYicesVersion :: String -> Version
   25 parseYicesVersion = go . map words . lines
   26   where
   27     go (("Yices" : version : _) : _) = parseVersion version
   28     go _ = Version []
   29 
   30 parseZ3Version :: String -> Version
   31 parseZ3Version = go . map words . lines
   32   where
   33     go (("Z3" : "version" : version : _) : _) = parseVersion version
   34     go _ = Version []
   35 
   36 getYicesVersion :: IO (Maybe Version)
   37 getYicesVersion = do
   38    mpath <- findExecutable "yices"
   39    case mpath of
   40      Just yices -> (Just . parseYicesVersion) <$> readProcess yices ["--version"] ""
   41      Nothing -> return Nothing
   42 
   43 getZ3Version :: IO (Maybe Version)
   44 getZ3Version = do
   45    mpath <- findExecutable "z3"
   46    case mpath of
   47      Just z3 -> (Just . parseZ3Version) <$> readProcess z3 ["--version"] ""
   48      Nothing -> return Nothing
   49 
   50 checkYicesVersion :: IO ()
   51 checkYicesVersion = do
   52   mv <- getYicesVersion
   53   case mv of
   54     Just v | v < Version [2,6,1] ->
   55       fail $ unlines
   56         [ "Yices version " ++ prettyVersion v ++ " is not supported."
   57         , "Version 2.6.1 or greater is required for path satisfiability checking."
   58         ]
   59     Nothing ->
   60       fail $ unlines
   61         [ "Yices executable 'yices' not found."
   62         , "Version 2.6.1 or greater is required for path satisfiability checking."
   63         ]
   64     _ -> return ()
