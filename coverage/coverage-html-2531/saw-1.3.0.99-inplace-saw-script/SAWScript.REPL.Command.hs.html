   81 -- | Commands.
   82 data Command
   83   = Command (REPL ())         -- ^ Successfully parsed command
   84   | Ambiguous String [String] -- ^ Ambiguous command, list of conflicting
   85                               --   commands
   86   | Unknown String            -- ^ The unknown command
   87 
   88 -- | Command builder.
   89 data CommandDescr = CommandDescr
   90   { cName :: String
   91   , cAliases :: [String]
   92   , cBody :: CommandBody
   93   , cHelp :: String
   94   }
   95 
   96 instance Show CommandDescr where
   97   show = cName
   98 
   99 instance Eq CommandDescr where
  100   (==) = (==) `on` cName
  101 
  102 instance Ord CommandDescr where
  103   compare = compare `on` cName
  104 
  105 data CommandBody
  106   = ExprArg     (String   -> REPL ())
  107   | FilenameArg (FilePath -> REPL ())
  108   | ShellArg    (String   -> REPL ())
  109   | NoArg       (REPL ())
  110

  112 -- | Convert the command list to a Trie, expanding aliases.
  113 makeCommands :: [CommandDescr] -> CommandMap
  114 makeCommands list  = foldl insert emptyTrie (concatMap expandAliases list)
  115   where
  116   insert m (name, d) = insertTrie name d m
  117   expandAliases :: CommandDescr -> [(String, CommandDescr)]
  118   expandAliases d = (cName d, d) : zip (cAliases d) (repeat d)

  156 genHelp :: [CommandDescr] -> [String]
  157 genHelp cs = map cmdHelp cs
  158   where
  159   cmdHelp cmd = concat [ "  ", cName cmd, pad (cName cmd), cHelp cmd ]
  160   padding     = 2 + maximum (map (length . cName) cs)
  161   pad n       = replicate (max 0 (padding - length n)) ' '

  166 -- | Run a command.
  167 runCommand :: Command -> REPL ()
  168 runCommand c = case c of
  169 
  170   Command cmd -> exceptionProtect cmd
  171 
  172   Unknown cmd -> io (putStrLn ("Unknown command: " ++ cmd))
  173 
  174   Ambiguous cmd cmds -> io $ do
  175     putStrLn (cmd ++ " is ambiguous, it could mean one of:")
  176     putStrLn ("\t" ++ intercalate ", " cmds)

  179 lexSAW :: String -> Text.Text -> REPL [Token Pos]
  180 lexSAW fileName str = do
  181   -- XXX wrap printing of positions in the message-printing infrastructure
  182   case SAWScript.Lexer.lexSAW fileName str of
  183     Left (_, pos, msg) ->
  184          fail $ show pos ++ ": " ++ Text.unpack msg
  185     Right (tokens', Nothing) -> pure tokens'
  186     Right (_, Just (SAWCentral.Options.Error, pos, msg)) ->
  187          fail $ show pos ++ ": " ++ Text.unpack msg
  188     Right (tokens', Just _) -> pure tokens'
  189 
  190 typeOfCmd :: String -> REPL ()
  191 typeOfCmd str
  192   | null str = do io $ putStrLn "[error] :type requires an argument"
  193   | otherwise =
  194   do tokens <- lexSAW replFileName (Text.pack str)
  195      expr <- case SAWScript.Parser.parseExpression tokens of
  196        Left err -> fail (show err)
  197        Right expr -> return expr
  198      let pos = getPos expr
  199          decl = SS.Decl pos (SS.PWild pos Nothing) Nothing expr
  200      rw <- getValueEnvironment
  201      decl' <- do
  202        let primsAvail = rwPrimsAvail rw
  203            valueInfo = rwValueInfo rw
  204            valueInfo' = Map.map (\(lc, ty, _val) -> (lc, ty)) valueInfo
  205            typeInfo = rwTypeInfo rw
  206        let (errs_or_results, warns) = checkDecl primsAvail valueInfo' typeInfo decl
  207        let issueWarning (msgpos, msg) =
  208              -- XXX the print functions should be what knows how to show positions...
  209              putStrLn (show msgpos ++ ": Warning: " ++ msg)
  210        io $ mapM_ issueWarning warns
  211        either failTypecheck return errs_or_results
  212      let ~(SS.Decl _pos _ (Just schema) _expr') = decl'
  213      io $ putStrLn $ PPS.pShow schema
  214 
  215 searchCmd :: String -> REPL ()
  216 searchCmd str
  217   | null str = do io $ putStrLn $ "[error] :search requires at least one argument"
  218   | otherwise =
  219   do tokens <- lexSAW replFileName (Text.pack str)
  220      pat <- case SAWScript.Parser.parseSchemaPattern tokens of
  221        Left err -> fail (show err)
  222        Right pat -> return pat
  223      rw <- getValueEnvironment
  224 
  225      -- Always search the entire environment and recognize all type
  226      -- names in the user's pattern, regardless of whether
  227      -- enable_experimental or enable_deprecated is in effect. It is
  228      -- definitely confusing to search for a hidden type and have
  229      -- search treat that as a free type variable and match all kinds
  230      -- of things you didn't intend. It is also better to retrieve
  231      -- invisible/deprecated items and report their existence than to
  232      -- hide them from the search.
  233 
  234      -- FUTURE: it would be nice to be able to use the words
  235      -- "experimental" and "deprecated" in the search term to match
  236      -- against the lifecycle, to allow doing stuff like searching
  237      -- for deprecated functions that take Terms.
  238 
  239      let primsAvail = rwPrimsAvail rw
  240          valueInfo = rwValueInfo rw
  241          valueInfo' = Map.map (\(lc, ty, _val) -> (lc, ty)) valueInfo
  242          typeInfo = rwTypeInfo rw
  243          (errs_or_results, warns) = checkSchemaPattern everythingAvailable valueInfo' typeInfo pat
  244      let issueWarning (msgpos, msg) =
  245            -- XXX the print functions should be what knows how to show positions...
  246            putStrLn (show msgpos ++ ": Warning: " ++ msg)
  247      io $ mapM_ issueWarning warns
  248      pat' <- either failTypecheck return $ errs_or_results
  249      let search = compileSearchPattern typeInfo pat'
  250          allMatches = Map.filter (\(_lc, ty) -> matchSearchPattern search ty) valueInfo'
  251 
  252          -- Divide the results into visible, experimental-not-visible,
  253          -- and deprecated-not-visible.
  254          inspect ::
  255              SS.LName ->
  256              (PrimitiveLifecycle, SS.Schema) ->
  257              (Map.Map SS.LName (PrimitiveLifecycle, SS.Schema),
  258               Map.Map SS.LName (PrimitiveLifecycle, SS.Schema),
  259               Map.Map SS.LName (PrimitiveLifecycle, SS.Schema)) ->
  260              (Map.Map SS.LName (PrimitiveLifecycle, SS.Schema),
  261               Map.Map SS.LName (PrimitiveLifecycle, SS.Schema),
  262               Map.Map SS.LName (PrimitiveLifecycle, SS.Schema))
  263          inspect name (lc, ty) (vis, ex, dep) =
  264              if Set.member lc primsAvail then
  265                  (Map.insert name (lc, ty) vis, ex, dep)
  266              else case lc of
  267                Current -> oops
  268                WarnDeprecated -> oops
  269                HideDeprecated ->
  270                  (vis, ex, Map.insert name (lc, ty) dep)
  271                Experimental ->
  272                  (vis, Map.insert name (lc, ty) ex, dep)
  273              where
  274                oops =
  275                  panic "searchCmd" [
  276                      "Found non-visible object " <> Text.pack (show name) <>
  277                      " with unexpected lifecycle " <> Text.pack (show lc)
  278                  ]
  279          (visMatches, expMatches, depMatches) =
  280              Map.foldrWithKey inspect (Map.empty, Map.empty, Map.empty) allMatches
  281 
  282          printMatch (lname, (lc, ty)) = do
  283            let name = Text.unpack $ SS.getVal lname
  284                ty' = PPS.pShow ty
  285                lc' = case lc of
  286                    Current -> ""
  287                    WarnDeprecated -> "  (DEPRECATED AND WILL WARN)"
  288                    HideDeprecated -> "  (DEPRECATED AND UNAVAILABLE BY DEFAULT)"
  289                    Experimental -> "  (EXPERIMENTAL)"
  290            putStrLn (name ++ " : " ++ ty' ++ lc')
  291          printMatches matches =
  292            io $ mapM_ printMatch (Map.assocs matches)
  293 
  294          moreMatches matches =
  295              let n = Map.size matches in
  296              if n == 1 then "1 more match"
  297              else show n ++ " more matches"
  298          alsoExperimental =
  299              if not (Map.null expMatches) then
  300                  io $ putStrLn $ moreMatches expMatches ++ " tagged " ++
  301                                  "experimental; use enable_experimental to " ++
  302                                  "see them"
  303              else
  304                  pure ()
  305          alsoDeprecated =
  306              if not (Map.null depMatches) then
  307                  io $ putStrLn $ moreMatches depMatches ++ " tagged " ++
  308                                  "deprecated; use enable_deprecated to " ++
  309                                  "see them"
  310              else
  311                  pure ()
  312 
  313      if not (Map.null visMatches) then do
  314          printMatches visMatches
  315          alsoExperimental
  316          alsoDeprecated
  317      else do
  318          io $ putStrLn "No matches."
  319          if not (Map.null expMatches) then do
  320              io $ putStrLn $ "The following experimental matches require " ++
  321                              "enable_experimental:"
  322              printMatches expMatches
  323              alsoDeprecated
  324          else if not (Map.null depMatches) then do
  325              io $ putStrLn $ "The following deprecated matches require " ++
  326                              "enable_deprecated:"
  327              printMatches depMatches
  328          else
  329              pure ()

  337 envCmd = do
  338   rw <- getValueEnvironment
  339   let showLName = Text.unpack . SS.getVal

  342       valueInfo' = Map.filter (\(lc, _ty, _v) -> Set.member lc avail) valueInfo

  346 tenvCmd = do

  349       typeInfo = rwTypeInfo rw
  350       typeInfo' = Map.filter (\(lc, _ty) -> Set.member lc avail) typeInfo
  351   io $ sequence_ [ TextIO.putStrLn (a <> " : " <> Text.pack (PPS.pShow ty)) | (a, (_lc, ty)) <- Map.assocs typeInfo' ]

  368 cdCmd :: FilePath -> REPL ()
  369 cdCmd f | null f = io $ putStrLn $ "[error] :cd requires a path argument"
  370         | otherwise = do
  371   exists <- io $ doesDirectoryExist f
  372   if exists
  373     then io $ setCurrentDirectory f
  374     else raise $ DirectoryNotFound f

  379 -- SAWScript commands ----------------------------------------------------------
  380 
  381 {- Evaluation is fairly straightforward; however, there are a few important
  382 caveats:
  383 
  384   1. 'return' is type-polymorphic.  This means that 'return "foo"' will produce
  385      a type-polymorphic function 'm -> m String', for any monad 'm'.  It also
  386      means that if you type 'return "foo"' into a naively-written interpreter,
  387      you won't see 'foo'!  The solution is to force each statement that comes
  388      into the REPL to have type 'TopLevel t' ('TopLevel' is the SAWScript
  389      version of 'IO').  This gets done as soon as the statement is parsed.
  390 
  391   2. Handling binding variables to values is somewhat tricky.  When you're
  392      allowed to bind variables in the REPL, you're basically working in the
  393      context of a partially evaluated module: all the results of all the old
  394      computations are in scope, and you're allowed to add new computations that
  395      depend on them.  It's insufficient to merely hang onto the AST for the
  396      old computations, as that could cause them to be evaluated multiple times;
  397      it could also cause their type to be inferred differently several times,
  398      which is bad.  Instead, we hang onto the inferred types of previous
  399      computations and use them to seed the name resolver and the typechecker;
  400      we also hang onto the results and use them to seed the interpreter. -}
  401 sawScriptCmd :: String -> REPL ()
  402 sawScriptCmd str = do
  403   tokens <- lexSAW replFileName (Text.pack str)
  404   case SAWScript.Parser.parseStmtSemi tokens of
  405     Left err -> io $ print err
  406     Right stmt ->
  407       do mr <- getProofStateRef
  408          case mr of
  409            Nothing -> void $ liftTopLevel (interpretTopStmt True stmt)
  410            Just r  -> void $ liftProofScript (interpretTopStmt True stmt) r
  411 
  412 replFileName :: String
  413 replFileName = "<stdin>"

  435 -- | Strip leading space.
  436 sanitize :: String -> String
  437 sanitize  = dropWhile isSpace
  438 
  439 -- | Strip trailing space.
  440 sanitizeEnd :: String -> String
  441 sanitizeEnd = reverse . sanitize . reverse
  442 
  443 -- | Split at the first word boundary.
  444 splitCommand :: String -> Maybe (String,String)
  445 splitCommand txt =
  446   case sanitize txt of
  447     ':' : more
  448       | (as,bs) <- span (\x -> isPunctuation x || isSymbol x) more
  449       , not (null as) -> Just (':' : as, sanitize bs)
  450 
  451       | (as,bs) <- break isSpace more
  452       , not (null as) -> Just (':' : as, sanitize bs)
  453 
  454       | otherwise -> Nothing
  455 
  456     expr -> guard (not (null expr)) >> return (expr,[])
  457 
  458 -- | Look up a string in a command list. If given a string that's both
  459 -- itself a command and a prefix of something else, choose that
  460 -- command; otherwise such commands are inaccessible. Also, deduplicate
  461 -- the list of results to avoid silliness with command aliases.
  462 findSomeCommand :: String -> CommandMap -> [CommandDescr]
  463 findSomeCommand str commandTable = nub $ lookupTrieWithExact str commandTable
  464 
  465 -- | Look up a string in the command list.
  466 findCommand :: String -> [CommandDescr]
  467 findCommand str = findSomeCommand str commands
  468 
  469 -- | Look up a string in the notebook-safe command list.
  470 findNbCommand :: String -> [CommandDescr]
  471 findNbCommand str = findSomeCommand str nbCommands

  473 -- | Parse a line as a command.
  474 parseCommand :: (String -> [CommandDescr]) -> String -> Maybe Command
  475 parseCommand findCmd line = do
  476   (cmd,args) <- splitCommand line
  477   let args' = sanitizeEnd args
  478   case findCmd cmd of
  479     -- nothing matched; if it doesn't begin with a colon, eval it
  480     [] -> case cmd of
  481       []      -> Nothing
  482       ':' : _ -> Just (Unknown cmd)
  483       _       -> Just (Command (sawScriptCmd line))
  484 
  485     -- matched exactly one command; run it
  486     [c] -> case cBody c of
  487       ExprArg     body -> Just (Command (body args'))
  488       FilenameArg body -> Just (Command (body =<< expandHome args'))
  489       ShellArg    body -> Just (Command (body args'))
  490       NoArg       body -> Just (Command  body)
  491 
  492     -- matched several things; complain
  493     cs -> Just (Ambiguous cmd (map cName cs))
  494 
  495   where
  496   expandHome path =
  497     case path of
  498       '~' : c : more | isPathSeparator c -> do dir <- io getHomeDirectory
  499                                                return (dir </> more)
  500       _ -> return path
