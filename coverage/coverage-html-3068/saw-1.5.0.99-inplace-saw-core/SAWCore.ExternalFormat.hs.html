   31 -- External text format
   32 
   33 type WriteM = State.State (Map TermIndex Int, Map VarIndex (Either Text NameInfo), [String], Int)
   34 
   35 renderNames :: Map VarIndex (Either Text NameInfo) -> String
   36 renderNames nms = show
   37   [ (idx, f nmi)
   38   | (idx,nmi) <- Map.toList nms
   39   ]
   40  where
   41    f (Left s) = Left s
   42    f (Right (ModuleIdentifier i))  = Right (Left (show i))
   43    f (Right (ImportedName uri as)) = Right (Right (renderURI uri, as))
   44 
   45 readNames :: String -> Maybe (Map VarIndex (Either Text NameInfo))
   46 readNames xs = Map.fromList <$> (mapM readName =<< readMaybe xs)
   47  where
   48    readName :: (VarIndex, Either Text (Either Text (Text,[Text]))) -> Maybe (VarIndex, Either Text NameInfo)
   49    readName (idx, Left x) = pure (idx, Left x)
   50    readName (idx, Right (Left i)) = pure (idx, Right (ModuleIdentifier (parseIdent (Text.unpack i))))
   51    readName (idx, Right (Right (uri,as))) =
   52        do uri' <- parseURI uri
   53           pure (idx, Right (ImportedName uri' as))
   54 
   55 -- | Render to external text format
   56 scWriteExternal :: Term -> String
   57 scWriteExternal t0 =
   58     let (x, (_, nms, lns, _)) = State.runState (go t0) (Map.empty, Map.empty, [], 1)
   59     in unlines $
   60         [ unwords ["SAWCoreTerm", show x]
   61         , renderNames nms
   62         ] ++ reverse lns
   63   where
   64     output :: String -> WriteM ()
   65     output l =
   66        do (m, nms, lns, x) <- State.get
   67           State.put (m, nms, l:lns, x)
   68     memoize :: TermIndex -> WriteM Int
   69     memoize i =
   70        do (m, nms, lns, x) <- State.get
   71           State.put (Map.insert i x m, nms, lns, x+1)
   72           return x
   73     stashName :: Name -> WriteM ()
   74     stashName ec =
   75        do (m, nms, lns, x) <- State.get
   76           State.put (m, Map.insert (nameIndex ec) (Right (nameInfo ec)) nms, lns, x)
   77     stashVarName :: VarName -> WriteM ()
   78     stashVarName vn =
   79        do (m, nms, lns, x) <- State.get
   80           State.put (m, Map.insert (vnIndex vn) (Left (vnName vn)) nms, lns, x)
   81 
   82     go :: Term -> WriteM Int
   83     go t =
   84       do let i = termIndex t
   85          (memo, _, _, _) <- State.get
   86          case Map.lookup i memo of
   87            Just x -> pure x
   88            Nothing -> do
   89              tf' <- traverse go (unwrapTermF t)
   90              body <- writeTermF tf'
   91              x <- memoize i
   92              output (unwords [show x, body])
   93              pure x
   94 
   95     writeTermF :: TermF Int -> WriteM String
   96     writeTermF tf =
   97       case tf of
   98         App e1 e2      -> pure $ unwords ["App", show e1, show e2]
   99         Lambda s t e   ->
  100           do stashVarName s
  101              pure $ unwords ["Lam", show (vnIndex s), show t, show e]
  102         Pi s t e       ->
  103           do stashVarName s
  104              pure $ unwords ["Pi", show (vnIndex s), show t, show e]
  105         Constant nm    ->
  106             do stashName nm
  107                pure $ unwords ["Constant", show (nameIndex nm)]
  108         Variable nm tp ->
  109            do stashVarName nm
  110               pure $ unwords ["Variable", show (vnIndex nm), show tp]
  111         FTermF ftf     ->
  112           case ftf of
  113             PairValue x y       -> pure $ unwords ["Pair", show x, show y]
  114             PairType x y        -> pure $ unwords ["PairT", show x, show y]
  115             PairLeft e          -> pure $ unwords ["ProjL", show e]
  116             PairRight e         -> pure $ unwords ["ProjR", show e]
  117 
  118             Recursor (CompiledRecursor d s _ _ _) ->
  119               do stashName d
  120                  let show_s = if s == propSort then "Prop" else drop 5 (show s)
  121                  pure $ unwords ["Recursor", show (nameIndex d), show_s]
  122 
  123             Sort s h
  124               | s == propSort -> pure $ unwords ("Prop" : map show (sortFlagsToList h))
  125               | otherwise     -> pure $ unwords ("Sort" : drop 5 (show s) : map show (sortFlagsToList h))
  126                                                         -- /\ Ugly hack to drop "sort "
  127             ArrayValue e v      -> pure $ unwords ("Array" : show e :
  128                                             map show (V.toList v))
  129             StringLit s         -> pure $ unwords ["String", show s]
  130

  132 -- | During parsing, we maintain various maps used for renumbering.
  133 -- We do not reuse any such numbers that appear in the external file,
  134 -- but generate fresh ones that are valid in the current
  135 -- 'SharedContext'.
  136 data ReadState =
  137   ReadState
  138   { rsTerms :: Map Int Term
  139     -- ^ Map 'Int' term identifiers from external core file to SAWCore terms
  140   , rsNames :: Map VarIndex (Either Text NameInfo)
  141     -- ^ Map 'VarIndex'es from external core file to global names
  142   , rsVars :: Map VarIndex VarIndex
  143     -- ^ Map 'VarIndex'es from external core file to variables
  144   }
  145 
  146 type ReadM = State.StateT ReadState IO
  147 
  148 scReadExternal :: SharedContext -> String -> IO Term
  149 scReadExternal sc input =
  150   case lines input of
  151     ( (words -> ["SAWCoreTerm", final]) : nmlist : rows ) ->
  152       case readNames nmlist of
  153         Nothing -> fail "scReadExternal: failed to parse name table"
  154         Just nms ->
  155           State.evalStateT (mapM_ (go . words) rows >> readIdx final) (ReadState Map.empty nms Map.empty)
  156 
  157     _ -> fail "scReadExternal: failed to parse input file"
  158   where
  159     go :: [String] -> ReadM ()
  160     go (tok : tokens) =
  161       do i <- readM tok
  162          t <- parse tokens
  163          State.modify $ \s -> s { rsTerms = Map.insert i t (rsTerms s) }
  164     go [] = pure () -- empty lines are ignored
  165 
  166     readM :: forall a. Read a => String -> ReadM a
  167     readM tok =
  168       case readMaybe tok of
  169         Nothing -> fail $ "scReadExternal: parse error: " ++ show tok
  170         Just x -> pure x
  171 
  172     getTerm :: Int -> ReadM Term
  173     getTerm i =
  174       do ts <- State.gets rsTerms
  175          case Map.lookup i ts of
  176            Nothing -> fail $ "scReadExternal: invalid term index: " ++ show i
  177            Just t -> pure t
  178 
  179     readIdx :: String -> ReadM Term
  180     readIdx tok = getTerm =<< readM tok
  181 
  182     readName' :: VarIndex -> ReadM Name
  183     readName' vi =
  184       do nms <- State.gets rsNames
  185          vs <- State.gets rsVars
  186          nmi <- case Map.lookup vi nms of
  187                   Just (Right nmi) -> pure nmi
  188                   _ -> lift $ fail $ "scReadExternal: Name missing name info: " ++ show vi
  189          case nmi of
  190            ModuleIdentifier ident ->
  191              lift (scResolveNameByURI sc (moduleIdentToURI ident)) >>= \case
  192                Just vi' -> pure (Name vi' nmi)
  193                Nothing  -> lift $ fail $ "scReadExternal: missing module identifier: " ++ show ident
  194            ImportedName uri _aliases ->
  195              lift (scResolveNameByURI sc uri) >>= \case
  196                Just vi' -> pure (Name vi' nmi)
  197                Nothing -> case Map.lookup vi vs of
  198                  Just vi' -> pure $ Name vi' nmi
  199                  Nothing ->
  200                    do nm <- lift $ scRegisterName sc nmi
  201                       State.modify $ \s -> s { rsVars = Map.insert vi (nameIndex nm) (rsVars s) }
  202                       pure nm
  203 
  204     readName :: String -> ReadM Name
  205     readName i =
  206       do vi <- readM i
  207          readName' vi
  208 
  209     readVarName' :: VarIndex -> ReadM VarName
  210     readVarName' vi =
  211       do nms <- State.gets rsNames
  212          vs <- State.gets rsVars
  213          case Map.lookup vi nms of
  214            Just (Left x) ->
  215              case Map.lookup vi vs of
  216                Just vi' -> pure (VarName vi' x)
  217                Nothing ->
  218                  do vn <- lift $ scFreshVarName sc x
  219                     State.modify $ \s -> s { rsVars = Map.insert vi (vnIndex vn) (rsVars s) }
  220                     pure vn
  221            _ -> lift $ fail $ "scReadExternal: VarName missing name: " ++ show vi
  222 
  223     readVarName :: String -> ReadM VarName
  224     readVarName i =
  225       do vi <- readM i
  226          readVarName' vi
  227 
  228     parse :: [String] -> ReadM Term
  229     parse tokens =
  230       case tokens of
  231         ["App", e1, e2]     -> do t1 <- readIdx e1
  232                                   t2 <- readIdx e2
  233                                   lift $ scApply sc t1 t2
  234         ["Lam", s, e1, e2]  -> do x <- readVarName s
  235                                   t1 <- readIdx e1
  236                                   t2 <- readIdx e2
  237                                   lift $ scLambda sc x t1 t2
  238         ["Pi", s, e1, e2]   -> do x <- readVarName s
  239                                   t1 <- readIdx e1
  240                                   t2 <- readIdx e2
  241                                   lift $ scPi sc x t1 t2
  242         ["Constant", i]     -> do nm <- readName i
  243                                   lift $ scConst sc nm
  244         ["Unit"]            -> lift $ scUnitValue sc
  245         ["UnitT"]           -> lift $ scUnitType sc
  246         ["Pair", e1, e2]    -> do t1 <- readIdx e1
  247                                   t2 <- readIdx e2
  248                                   lift $ scPairValue sc t1 t2
  249         ["PairT", e1, e2]   -> do t1 <- readIdx e1
  250                                   t2 <- readIdx e2
  251                                   lift $ scPairType sc t1 t2
  252         ["ProjL", e1]       -> do t1 <- readIdx e1
  253                                   lift $ scPairLeft sc t1
  254         ["ProjR", e1]       -> do t1 <- readIdx e1
  255                                   lift $ scPairRight sc t1
  256 
  257         ["Recursor", i, s]  ->
  258           do nm <- readName i
  259              s' <- if s == "Prop" then pure propSort else mkSort <$> readM s
  260              lift $ scRecursor sc nm s'
  261 
  262         ("Prop" : h)        -> do flags <- sortFlagsFromList <$> mapM readM h
  263                                   lift $ scSortWithFlags sc propSort flags
  264         ("Sort" : s : h)    -> do s' <- mkSort <$> readM s
  265                                   flags <- sortFlagsFromList <$> mapM readM h
  266                                   lift $ scSortWithFlags sc s' flags
  267         ("Array" : e : es)  -> do t <- readIdx e
  268                                   ts <- traverse readIdx es
  269                                   lift $ scVector sc t ts
  270         ("String" : ts)     -> do str <- readM (unwords ts)
  271                                   lift $ scString sc str
  272         ["Variable", i, t]  -> do vn <- readVarName i
  273                                   tp <- readIdx t
  274                                   lift $ scVariable sc vn tp
  275         _ -> fail $ "Parse error: " ++ unwords tokens
