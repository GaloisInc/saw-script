   22 -- | Replace all occurrences of the double quote character @"@ with the string
   23 -- @""@, i.e., two copies of it, as this is how Coq escapes double quote
   24 -- characters.
   25 escapeStringLit :: String -> String
   26 escapeStringLit = concatMap (\c -> if c == '"' then "\"\"" else [c])

   39 -- | Like hsep, but behaves usefully for lists that might be empty:
   40 --   returns the empty doc like hsep for empty lists, but for
   41 --   non-empty lists prepends horizontal space. The result can then
   42 --   be inserted like this: x <> result <+> y, and will produce
   43 --   either "x y" or "x result y" but not "x  y", which is what
   44 --   happens if you do x <+> hsep ... <+> y.
   45 --
   46 --   This baloney arises because d1 <+> emptyDoc <+> d2 prints two
   47 --   spaces instead of one, which I'd describe as a bug. If you
   48 --   _really_ want to accumulate multiple spaces by concatenating
   49 --   empty docs, which seems against the concept of prettyprinting
   50 --   layout anyway, there can be another way to do that, like the
   51 --   latex "phantom" object.
   52 hsep' :: [Doc ann] -> Doc ann
   53 hsep' docs = case docs of
   54     [] -> emptyDoc
   55     _ : _ -> emptyDoc <+> hsep docs
   56 
   57 -- This version glues the separator to the end of each element.
   58 tightSepList :: Doc ann -> [Doc ann] -> Doc ann
   59 tightSepList _ [] = mempty
   60 tightSepList _ [d] = d
   61 tightSepList s (d:l) = d <> s <+> tightSepList s l
   62 
   63 -- This version issues space before the separator.
   64 looseSepList :: Doc ann -> [Doc ann] -> Doc ann
   65 looseSepList _ [] = mempty
   66 looseSepList _ [d] = d
   67 looseSepList s (d:l) = d <+> s <+> looseSepList s l
   68 
   69 commaSepList, starSepList, semiSepList :: [Doc ann] -> Doc ann
   70 commaSepList = tightSepList comma
   71 starSepList = looseSepList "*"
   72 semiSepList = tightSepList semi
   73 
   74 period :: Doc ann
   75 period = "."

   77 prettyIdent :: Ident -> Doc ann
   78 prettyIdent (Ident s) = pretty s
   79 
   80 prettyNameType :: Ident -> Type -> Doc ann
   81 prettyNameType x ty = prettyIdent x <+> colon <+> prettyTerm PrecNone ty
   82 
   83 prettyBinder :: Binder -> Doc ann
   84 prettyBinder b = case b of
   85     Binder Explicit x Nothing   -> prettyIdent x
   86     Binder Explicit x (Just ty) -> parens $ prettyNameType x ty
   87     Binder Implicit x Nothing    -> braces $ prettyIdent x
   88     Binder Implicit x (Just ty)  -> braces $ prettyNameType x ty
   89 
   90 prettyPiBinder :: PiBinder -> Doc ann
   91 prettyPiBinder b = case b of
   92     PiBinder Explicit Nothing ty ->
   93         prettyTerm PrecApp ty <+> "->"
   94     PiBinder Explicit (Just x) ty ->
   95         "forall" <+> parens (prettyNameType x ty) <> comma
   96     PiBinder Implicit Nothing ty ->
   97         braces (prettyTerm PrecApp ty) <+> "->"
   98     PiBinder Implicit (Just x) ty ->
   99         "forall" <+> braces (prettyNameType x ty) <> comma
  100 
  101 prettyBinders :: [Binder] -> Doc ann
  102 prettyBinders bs = hsep $ map prettyBinder bs
  103 
  104 prettyMaybeTy :: Maybe Type -> Doc ann
  105 prettyMaybeTy Nothing = mempty
  106 prettyMaybeTy (Just ty) = colon <+> prettyTerm PrecNone ty
  107 
  108 prettySort :: Sort -> Doc ann
  109 prettySort s = case s of
  110     Prop -> "Prop"
  111     Set -> "Set"
  112     Type -> "Type"
  113 
  114 prettyPiBinders :: [PiBinder] -> Doc ann
  115 prettyPiBinders bs = hsep $ map prettyPiBinder bs

  117 data Prec
  118   = PrecNone
  119   | PrecLambda
  120   | PrecApp
  121   | PrecAtom
  122   deriving (Eq, Ord)
  123 
  124 parensIf :: Bool -> Doc ann -> Doc ann
  125 parensIf p d = if p then parens d else d
  126 
  127 prettyTerm :: Prec -> Term -> Doc ann
  128 prettyTerm p e =
  129   case e of
  130     Lambda bs t ->
  131       let bs' = prettyBinders bs
  132           t' = prettyTerm PrecLambda t
  133       in
  134       parensIf (p > PrecLambda) $ "fun" <+> bs' <+> "=>" <+> t'
  135     Fix ident binders returnType body ->
  136       let ident' = prettyIdent ident
  137           binders' = prettyBinders binders
  138           returnType' = prettyTerm PrecNone returnType
  139           body' = prettyTerm PrecLambda body
  140       in
  141       parensIf (p > PrecLambda) $
  142           "fix" <+> ident' <+> binders' <+> ":" <+> returnType' <+> ":=" <+> body'
  143     Pi bs t ->
  144       let bs' = prettyPiBinders bs
  145           t' = prettyTerm PrecLambda t
  146       in
  147       parensIf (p > PrecLambda) $ bs' <+> t'
  148     Let x bs mty t body ->
  149       let x' = prettyIdent x
  150           bs' = prettyBinders bs
  151           mty' = prettyMaybeTy mty
  152           t' = prettyTerm PrecNone t
  153           body' = prettyTerm PrecLambda body
  154       in
  155       parensIf (p > PrecLambda) $ fillSep [
  156           "let" <+> x' <+> bs' <+> mty' <+> ":=" <+> t' <+> "in",
  157           body'
  158       ]
  159     If c t f ->
  160       let c' = prettyTerm PrecNone c
  161           t' = prettyTerm PrecNone t
  162           f' = prettyTerm PrecLambda f
  163       in
  164       parensIf (p > PrecLambda) $
  165           "if" <+> c' <+> "then" <+> t' <+> "else" <+> f'
  166     App f [] ->
  167       prettyTerm p f
  168     App f args ->
  169       let f' = prettyTerm PrecApp f
  170           args' = map (prettyTerm PrecAtom) args
  171       in
  172       parensIf (p > PrecApp) $ hsep (f' : args')
  173     Sort s ->
  174       prettySort s
  175     Var x ->
  176       prettyIdent x
  177     ExplVar x ->
  178       let x' = prettyIdent x in
  179       parensIf (p > PrecApp) $ "@" <> x'
  180     Ascription tm tp ->
  181       let tm' = prettyTerm PrecApp tm
  182           tp' = prettyTerm PrecApp tp
  183       in
  184       parensIf (p > PrecLambda) $ tm' <+> ":" <+> tp'
  185     NatLit i ->
  186       if i > 1000 then
  187         -- Explicitly convert from Z if an integer is too big
  188         parensIf (p > PrecLambda) ("Z.to_nat" <+> integer i <> "%Z")
  189       else
  190         integer i
  191     ZLit i ->
  192       -- we use hex unless our integer is a positive or negative digit
  193       -- XXX: this cannot possibly work as intended for negative values
  194       if abs i > 9 then
  195           let ui = toInteger (fromInteger i :: Word64)
  196               ui' = showHex ui []
  197           in
  198           text ("0x" ++ ui' ++ "%Z")
  199       else if i < 0 then
  200           text ("(" ++ show i ++ ")%Z")
  201       else
  202           text (show i ++ "%Z")
  203     List ts ->
  204       let ts' = map (prettyTerm PrecNone) ts in
  205       brackets $ semiSepList ts'
  206     StringLit s ->
  207       dquotes (string $ escapeStringLit s)
  208     Scope term scope ->
  209       let term' = prettyTerm PrecAtom term
  210           scope' = text scope
  211       in
  212       term' <> "%" <> scope'
  213     Ltac s ->
  214       "ltac:" <> parens (string s)
  215 
  216 prettyBasicDecl :: Doc ann -> Ident -> Type -> Doc ann
  217 prettyBasicDecl what nm ty =
  218   let nm' = prettyIdent nm
  219       ty' = prettyTerm PrecNone ty
  220   in
  221   nest 2 (what <+> nm' <+> ":" <+> ty' <+> period) <> hardline
  222 
  223 prettyDecl :: Decl -> Doc ann
  224 prettyDecl decl = case decl of
  225   Axiom nm ty -> prettyBasicDecl "Axiom" nm ty
  226   Parameter nm ty -> prettyBasicDecl "Parameter" nm ty
  227   Variable nm ty -> prettyBasicDecl "Variable" nm ty
  228   Comment s ->
  229     "(*" <+> text s <+> "*)" <> hardline
  230   Definition nm bs mty body ->
  231     let nm' = prettyIdent nm
  232         bs' = hsep' $ map prettyBinder bs
  233         mty' = prettyMaybeTy mty
  234         body' = prettyTerm PrecNone body
  235     in
  236     nest 2 (
  237       vsep [
  238           "Definition" <+> nm' <> bs' <+> mty' <+> ":=",
  239           body' <> period
  240       ]
  241     ) <> hardline
  242   InductiveDecl ind ->
  243     prettyInductive ind
  244   Section nm ds ->
  245     let nm' = prettyIdent nm
  246         ds' = map (indent 2 . prettyDecl) ds
  247         header = "Section" <+> nm' <+> period
  248         footer = "End" <+> nm' <+> period
  249     in
  250     -- XXX vsep issues soft newlines and there should be a hard newline
  251     -- after the head and after the foot. (Note that every other Decl
  252     -- always ends in a hard newline, so ds' is ok.)
  253     -- (XXX: Does `vsep` on top of `hardline` generate two lines?)
  254     vsep $ [header] ++ ds' ++ [footer]
  255   Snippet s ->
  256     text s
  257 
  258 prettyConstructor :: Constructor -> Doc ann
  259 prettyConstructor (Constructor {..}) =
  260   let name' = prettyIdent constructorName
  261       ty' = prettyTerm PrecNone constructorType
  262   in
  263   nest 2 $ "|" <+> name' <+> ":" <+> ty'
  264 
  265 prettyInductive :: Inductive -> Doc ann
  266 prettyInductive (Inductive {..}) =
  267   let name' = prettyIdent inductiveName
  268       params' = hsep' $ map prettyBinder inductiveParameters
  269       indices' = hsep' $ map prettyPiBinder inductiveIndices
  270       sort' = prettySort inductiveSort
  271       ctors' = map prettyConstructor inductiveConstructors
  272       header = "Inductive" <+> name' <> params' <+> ":" <> indices' <+> sort' <+> ":="
  273   in
  274   vsep ([nest 2 header] ++ ctors' ++ [period]) <> hardline
