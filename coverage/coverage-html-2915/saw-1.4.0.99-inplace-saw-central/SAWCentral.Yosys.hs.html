   68 -- ** Building the module graph from Yosys IR

   70 data Modgraph = Modgraph
   71   { _modgraphGraph :: Graph.Graph
   72   , _modgraphNodeFromVertex :: Graph.Vertex -> (Module, Text, [Text])
   73   -- , _modgraphVertexFromKey :: Text -> Maybe Graph.Vertex
   74   }
   75 makeLenses ''Modgraph
   76 
   77 -- | Given a Yosys IR, construct a graph of intermodule dependencies.
   78 yosysIRModgraph :: YosysIR -> Modgraph
   79 yosysIRModgraph ir =
   80   let
   81     moduleToNode :: (Text, Module) -> (Module, Text, [Text])
   82     moduleToNode (nm, m) = (m, nm, deps)
   83       where
   84         deps = Text.pack . show . view cellType <$> Map.elems (m ^. moduleCells)
   85     nodes = moduleToNode <$> Map.assocs (ir ^. yosysModules)
   86     (_modgraphGraph, _modgraphNodeFromVertex, _modgraphVertexFromKey)
   87       = Graph.graphFromEdges nodes
   88   in Modgraph{..}

   90 -- | Given a Yosys IR, construct a map from module names to SAWCore terms alongside SAWCore and Cryptol types
   91 convertYosysIR ::
   92   SC.SharedContext ->
   93   YosysIR ->
   94   IO (Map Text ConvertedModule)
   95 convertYosysIR sc ir =
   96   do let mg = yosysIRModgraph ir
   97      let sorted = reverseTopSort $ mg ^. modgraphGraph
   98      foldM
   99        (\env v ->
  100           do let (m, nm, _) = mg ^. modgraphNodeFromVertex $ v
  101              cm <- convertModule sc env m
  102              n <- Nonce.freshNonce Nonce.globalNonceGenerator
  103              let frag = Text.pack . show $ Nonce.indexValue n
  104              let uri = URI.URI
  105                    { URI.uriScheme = URI.mkScheme "yosys"
  106                    , URI.uriAuthority = Left True
  107                    , URI.uriPath = (False,) <$> mapM URI.mkPathPiece (nm NE.:| [])
  108                    , URI.uriQuery = []
  109                    , URI.uriFragment = URI.mkFragment frag
  110                    }
  111              let ni = SC.ImportedName uri [nm]
  112              body <- SC.scAscribe sc (cm ^. convertedModuleTerm) (cm ^. convertedModuleType)
  113              tc <- SC.scDefineConstant sc ni body
  114              let cm' = cm { _convertedModuleTerm = tc }
  115              pure $ Map.insert nm cm' env
  116        )
  117        Map.empty
  118        sorted

  120 -- | Given a Yosys IR, construct a map from module names to TypedTerms
  121 yosysIRToTypedTerms ::
  122   SC.SharedContext ->
  123   YosysIR ->
  124   IO (Map Text SC.TypedTerm)
  125 yosysIRToTypedTerms sc ir =
  126   do env <- convertYosysIR sc ir
  127      pure $ flip fmap env $ \cm ->
  128        SC.TypedTerm
  129        (SC.TypedTermSchema $ C.tMono $ cm ^. convertedModuleCryptolType)
  130        $ cm ^. convertedModuleTerm
  131 
  132 -- | Given a Yosys IR, construct a SAWCore record containing terms for each module
  133 yosysIRToRecordTerm ::
  134   SC.SharedContext ->
  135   YosysIR ->
  136   IO SC.TypedTerm
  137 yosysIRToRecordTerm sc ir =
  138   do env <- convertYosysIR sc ir
  139      record <- cryptolRecord sc $ view convertedModuleTerm <$> env
  140      let cty = C.tRec . C.recordFromFields $ (\(nm, cm) -> (C.mkIdent nm, cm ^. convertedModuleCryptolType)) <$> Map.assocs env
  141      let tt = SC.TypedTerm (SC.TypedTermSchema $ C.tMono cty) record
  142      pure tt

  144 -- | Given a Yosys IR, construct a value representing a specific module with all submodules inlined
  145 yosysIRToSequential ::
  146   SC.SharedContext ->
  147   YosysIR ->
  148   Text ->
  149   IO YosysSequential
  150 yosysIRToSequential sc ir nm =
  151   case Map.lookup nm $ ir ^. yosysModules of
  152     Nothing -> throw . YosysError $ mconcat
  153       [ "Could not find module: "
  154       , nm
  155       ]
  156     Just m -> convertModuleInline sc m

  163 -- | Produces a Term given the path to a JSON file produced by the
  164 -- Yosys write_json command. The resulting term is a Cryptol record,
  165 -- where each field corresponds to one HDL module exported by Yosys.
  166 -- Each HDL module is in turn represented by a function from a record
  167 -- of input port values to a record of output port values.
  168 yosys_import :: FilePath -> TopLevel SC.TypedTerm
  169 yosys_import path =
  170   do sc <- getSharedContext
  171      ir <- liftIO $ loadYosysIR path
  172      liftIO $ yosysIRToRecordTerm sc ir

  174 -- | Proves equality between a combinational HDL module and a
  175 -- specification. Note that terms derived from HDL modules are first
  176 -- class, and are not restricted to yosys_verify: they may also be
  177 -- used with SAW's typical Term infrastructure like sat, prove_print,
  178 -- term rewriting, etc. yosys_verify simply provides a convenient and
  179 -- familiar interface, similar to llvm_verify or jvm_verify.
  180 yosys_verify ::
  181   SC.TypedTerm {- ^ Term corresponding to the HDL module -} ->
  182   [SC.TypedTerm] {- ^ Preconditions for the equality -} ->
  183   SC.TypedTerm {- ^ Specification term of the same type as the HDL module -} ->
  184   [YosysTheorem] {- ^ Overrides to apply -} ->
  185   ProofScript () ->
  186   TopLevel YosysTheorem
  187 yosys_verify ymod preconds other specs tactic =
  188   do sc <- getSharedContext
  189      newmod <- liftIO $ foldM (flip $ applyOverride sc)
  190        (SC.ttTerm ymod)
  191        specs
  192      mpc <-
  193        case preconds of
  194          [] -> pure Nothing
  195          (pc:pcs) ->
  196            do t <- foldM (\a b -> liftIO $ SC.scAnd sc a b) (SC.ttTerm pc) (SC.ttTerm <$> pcs)
  197               pure . Just $ SC.TypedTerm (SC.ttType pc) t
  198      thm <- liftIO $ buildTheorem sc ymod newmod mpc other
  199      prop <- liftIO $ theoremProp sc thm
  200      _ <- Builtins.provePrintPrim tactic prop
  201      pure thm

  203 -- | Import a single sequential HDL module. N.B. SAW expects the
  204 -- sequential module to exist entirely within a single Yosys module.
  205 yosys_import_sequential ::
  206   Text {- ^ Name of the HDL module -} ->
  207   FilePath {- ^ Path to the Yosys JSON file -} ->
  208   TopLevel YosysSequential
  209 yosys_import_sequential nm path =
  210   do sc <- getSharedContext
  211      ir <- liftIO $ loadYosysIR path
  212      liftIO $ yosysIRToSequential sc ir nm

  214 -- | Extracts a term from the given sequential module with the state
  215 -- eliminated by iterating the term over the given concrete number of
  216 -- cycles. The resulting term has no state field in the inputs or
  217 -- outputs. Each input and output field is replaced with an array of
  218 -- that field's type (array length being the number of cycles
  219 -- specified).
  220 yosys_extract_sequential ::
  221   YosysSequential ->
  222   Integer {- ^ Number of cycles to iterate term -} ->
  223   TopLevel SC.TypedTerm
  224 yosys_extract_sequential s n =
  225   do sc <- getSharedContext
  226      liftIO $ composeYosysSequential sc s n
  227 
  228 -- | Like `yosys_extract_sequential`, but the resulting term has an
  229 -- additional parameter to specify the initial state.
  230 yosys_extract_sequential_with_state ::
  231   YosysSequential ->
  232   Integer {- ^ Number of cycles to iterate term -} ->
  233   TopLevel SC.TypedTerm
  234 yosys_extract_sequential_with_state s n =
  235   do sc <- getSharedContext
  236      liftIO $ composeYosysSequentialWithState sc s n

  238 -- | Extracts a term from the given sequential module. This term has
  239 -- explicit fields for the state of the circuit in the input and
  240 -- output record types.
  241 yosys_extract_sequential_raw :: YosysSequential -> TopLevel SC.TypedTerm
  242 yosys_extract_sequential_raw s = pure $ s ^. yosysSequentialTerm

  244 -- | Export a query over the given sequential module to an input file for the Sally model checker.
  245 yosys_verify_sequential_sally ::
  246   YosysSequential ->
  247   FilePath {- ^ Path to write the resulting Sally input -} ->
  248   SC.TypedTerm {- ^ A boolean function of three parameters: an 8-bit cycle counter, a record of "fixed" inputs, and a record of circuit outputs -} ->
  249   [Text] {- ^ Names of circuit inputs that are fixed -} ->
  250   TopLevel ()
  251 yosys_verify_sequential_sally s path q fixed =
  252   do sc <- getSharedContext
  253      sym <- liftIO $ Common.newSAWCoreExprBuilder sc False
  254      liftIO $ queryModelChecker sym sc s path q $ Set.fromList fixed
