   24 -- NamedTyVars {{{
   25 
   26 --
   27 -- namedTyVars is a type-class-polymorphic function for extracting named
   28 -- type variables from a type or type schema. It returns a set of Name
   29 -- (Name is just Text) manifested as a map from those Names to their
   30 -- positions/provenance.
   31 --
   32 
   33 class NamedTyVars t where
   34   namedTyVars :: t -> Map Name Pos
   35 
   36 instance (Ord k, NamedTyVars a) => NamedTyVars (Map k a) where
   37   namedTyVars = namedTyVars . Map.elems
   38 
   39 instance (NamedTyVars a) => NamedTyVars [a] where
   40   namedTyVars = Map.unionsWith choosePos . map namedTyVars
   41 
   42 instance NamedTyVars Type where
   43   namedTyVars t = case t of
   44     TyCon _ _ ts      -> namedTyVars ts
   45     TyRecord _ tm     -> namedTyVars tm
   46     TyVar pos n       -> Map.singleton n pos
   47     TyUnifyVar _ _    -> Map.empty
   48 
   49 instance NamedTyVars Schema where
   50   namedTyVars (Forall ns t) = namedTyVars t Map.\\ Map.fromList ns'
   51     where ns' = map (\(pos, n) -> (n, pos)) ns

   55 -- SubstituteTyVars {{{
   56 
   57 --
   58 -- substituteTyVars is a typeclass-polymorphic function for
   59 -- substituting named type variables (such as those declared with
   60 -- typedef) in a Type.
   61 --
   62 -- Panics if we try to substitute a definition that isn't visible.
   63 --
   64 -- Note: substituteTyVars is reused by the interpreter as part of its
   65 -- handling of typedefs during execution as well as by the
   66 -- typechecker.
   67 --
   68 -- XXX: it's not clear that the instances for Maybe and List warrant
   69 -- setting up the typeclass.
   70 --
   71 
   72 class SubstituteTyVars t where
   73   -- | @substituteTyVars m x@ applies the (scoped) map @m@ to type variables in @x@.
   74   substituteTyVars ::
   75       Set PrimitiveLifecycle ->
   76       ScopedMap Name (PrimitiveLifecycle, NamedType) ->
   77       t -> t
   78 
   79 instance (SubstituteTyVars a) => SubstituteTyVars (Maybe a) where
   80   substituteTyVars avail tyenv = fmap (substituteTyVars avail tyenv)
   81 
   82 instance (SubstituteTyVars a) => SubstituteTyVars [a] where
   83   substituteTyVars avail tyenv = map (substituteTyVars avail tyenv)
   84 
   85 instance SubstituteTyVars Type where
   86   substituteTyVars avail tyenv ty = case ty of
   87     TyCon pos tc ts     -> TyCon pos tc (substituteTyVars avail tyenv ts)
   88     TyRecord pos fs     -> TyRecord pos (fmap (substituteTyVars avail tyenv) fs)
   89     TyUnifyVar _ _      -> ty
   90     TyVar _ n           ->
   91         case ScopedMap.lookup n tyenv of
   92             Nothing -> ty
   93             Just (lc, expansion) ->
   94                 if not (Set.member lc avail) then
   95                     panic "substituteTyVars" [
   96                         "Found reference to non-visible typedef: " <> n,
   97                         "Lifecycle setting: " <> Text.pack (show lc)
   98                     ]
   99                 else case expansion of
  100                     AbstractType _kind  -> ty
  101                     ConcreteType ty' -> ty'
  102 
  103 --
  104 -- The prime version uses an ordinary map.
  105 --
  106 -- This is used by the typechecker for the time being until the
  107 -- typechecker gets taught to use ScopedMap.
  108 --
  109 
  110 class SubstituteTyVars' t where
  111   -- | @substituteTyVars' m x@ applies the (ordinary) map @m@ to type variables in @x@.
  112   substituteTyVars' ::
  113       Set PrimitiveLifecycle ->
  114       Map Name (PrimitiveLifecycle, NamedType) ->
  115       t -> t
  116 
  117 instance (SubstituteTyVars' a) => SubstituteTyVars' (Maybe a) where
  118   substituteTyVars' avail tyenv = fmap (substituteTyVars' avail tyenv)
  119 
  120 instance (SubstituteTyVars' a) => SubstituteTyVars' [a] where
  121   substituteTyVars' avail tyenv = map (substituteTyVars' avail tyenv)
  122 
  123 instance SubstituteTyVars' Type where
  124   substituteTyVars' avail tyenv ty = case ty of
  125     TyCon pos tc ts     -> TyCon pos tc (substituteTyVars' avail tyenv ts)
  126     TyRecord pos fs     -> TyRecord pos (fmap (substituteTyVars' avail tyenv) fs)
  127     TyUnifyVar _ _      -> ty
  128     TyVar _ n           ->
  129         case Map.lookup n tyenv of
  130             Nothing -> ty
  131             Just (lc, expansion) ->
  132                 if not (Set.member lc avail) then
  133                     panic "substituteTyVars'" [
  134                         "Found reference to non-visible typedef: " <> n,
  135                         "Lifecycle setting: " <> Text.pack (show lc)
  136                     ]
  137                 else case expansion of
  138                     AbstractType _kind -> ty
  139                     ConcreteType ty' -> ty'
  140

  145 -- Deprecation {{{
  146 
  147 isDeprecated :: PrimitiveLifecycle -> Bool
  148 isDeprecated lc = case lc of
  149     Current -> False
  150     WarnDeprecated -> True
  151     HideDeprecated -> True
  152     Experimental -> False
