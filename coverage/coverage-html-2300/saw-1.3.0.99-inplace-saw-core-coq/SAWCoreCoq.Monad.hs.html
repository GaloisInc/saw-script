   34 data TranslationError a
   35   = NotSupported a
   36   | NotExpr a
   37   | NotType a
   38   | LocalVarOutOfBounds a
   39   | BadTerm a
   40   | CannotCreateDefaultValue a
   41 
   42 instance {-# OVERLAPPING #-} Show (TranslationError Term) where
   43   show = showError showTerm
   44 
   45 instance {-# OVERLAPPABLE #-} Show a => Show (TranslationError a) where
   46   show = showError show
   47 
   48 showError :: (a -> String) -> TranslationError a -> String
   49 showError printer err = case err of
   50   NotSupported a -> "Not supported: " ++ printer a
   51   NotExpr a      -> "Expecting an expression term: " ++ printer a
   52   NotType a      -> "Expecting a type term: " ++ printer a
   53   LocalVarOutOfBounds a -> "Local variable reference is out of bounds: " ++ printer a
   54   BadTerm a -> "Malformed term: " ++ printer a
   55   CannotCreateDefaultValue a -> "Unable to generate a default value of the given type: " ++ printer a
   56 
   57 data TranslationConfiguration = TranslationConfiguration
   58   { constantRenaming :: [(String, String)]
   59   -- ^ A map from 'ImportedName's of constants to names that should be used to
   60   -- realize them in Coq; this is generally used to map cryptol operators like
   61   -- @||@ or @&&@ to nicer names in Coq, but works on any imported names
   62   , constantSkips :: [String]
   63   -- ^ A list of 'ImportedName's to skip, i.e., not translate when encountered
   64   , monadicTranslation :: Bool
   65   -- ^ Whether to wrap everything in a free monad construction.
   66   -- - Advantage: fixpoints can be readily represented
   67   -- - Disadvantage: pure computations look more gnarly
   68   , postPreamble :: String
   69   -- ^ Text to be concatenated at the end of the Coq preamble, before the code
   70   -- generated by the translation.  Usually consists of extra file imports,
   71   -- module imports, scopes openings.
   72   , vectorModule       :: String
   73   -- ^ all vector operations will be prepended with this module name, i.e.
   74   -- "<VectorModule>.append", etc.  So that it can be retargeted easily.
   75   -- Current provided options are:
   76   -- - SAWCoreVectorsAsCoqLists
   77   -- - SAWCoreVectorsAsCoqVectors
   78   -- Currently considering adding:
   79   -- - SAWCoreVectorsAsSSReflectSeqs
   80   }
   81 
   82 -- | The functional dependency of 'MonadReader' makes it not compositional, so
   83 -- we have to jam together different structures that want to be in the 'Reader'
   84 -- into a single datatype.  This type allows adding extra configuration on top
   85 -- of the translation configuration.
   86 data WithTranslationConfiguration r = WithTranslationConfiguration
   87   { translationConfiguration :: TranslationConfiguration
   88   , otherConfiguration :: r
   89   }
   90 
   91 -- | Some computations will rely solely on access to the configuration, so we
   92 -- provide it separately.
   93 type TranslationConfigurationMonad r m =
   94   ( MonadReader (WithTranslationConfiguration r) m
   95   )
   96 
   97 type TranslationMonad r s m =
   98   ( Except.MonadError (TranslationError Term)  m
   99   , TranslationConfigurationMonad r m
  100   , MonadState s m
  101   )
  102 
  103 runTranslationMonad ::
  104   TranslationConfiguration ->
  105   r ->
  106   s ->
  107   (forall m. TranslationMonad r s m => m a) ->
  108   Either (TranslationError Term) (a, s)
  109 runTranslationMonad configuration r s m =
  110   runStateT (runReaderT m (WithTranslationConfiguration configuration r)) s
