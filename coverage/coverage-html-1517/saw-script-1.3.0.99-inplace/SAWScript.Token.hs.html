    8 {-# LANGUAGE DeriveFunctor #-}
    9 module SAWScript.Token where
   10 
   11 import Data.Text (Text)
   12 
   13 import SAWScript.Position (Positioned(..))
   14 
   15 -- | Lexer tokens for saw-script.
   16 --
   17 -- All tokens have tokStr carrying the matched text (even TEOF where
   18 -- it doesn't entirely make sense) so that tokStr can be applied to
   19 -- any variant of the type.
   20 --
   21 -- The comment and EOL tokens are internal to the lexer and not seen
   22 -- downstream.
   23 --
   24 -- The tokens are:
   25 --   TVar       variable/identifier
   26 --   TQVar      qualified variable/identifier
   27 --   TLit       string constant
   28 --   TCode      Cryptol code block
   29 --   TCType     Cryptol type bllock
   30 --   TUnknown   Anything else that doesn't fit
   31 --   TPunct     Punctuation tokens used in the grammar
   32 --   TReserved  All reserved words
   33 --   TOp        Punctuation tokens apparently not used in the grammar
   34 --   TNum       Number
   35 --   TCommentS  Start of a block comment
   36 --   TCommentE  End of a block comment
   37 --   TCommentL  Start of a line comment
   38 --   TEOL       End of line
   39 --   TEOF       End of file/input
   40 --
   41 -- FUTURE: many of these could stand to be renamed
   42 --
   43 data Token p = TVar      { tokPos :: p, tokStr :: Text                               }
   44              | TQVar     { tokPos :: p, tokStr :: Text, tokVars :: ([Text], Text)    }
   45              | TLit      { tokPos :: p, tokStr :: Text                               }
   46              | TCode     { tokPos :: p, tokStr :: Text                               }
   47              | TCType    { tokPos :: p, tokStr :: Text                               }
   48              | TUnknown  { tokPos :: p, tokStr :: Text                               }
   49              | TPunct    { tokPos :: p, tokStr :: Text                               }
   50              | TReserved { tokPos :: p, tokStr :: Text                               }
   51              | TOp       { tokPos :: p, tokStr :: Text                               }
   52              | TNum      { tokPos :: p, tokStr :: Text, tokNum :: Integer            }
   53              | TEOF      { tokPos :: p, tokStr :: Text                               }
   54              deriving (Show, Functor)
   55 
   56 instance Positioned p => Positioned (Token p) where
   57   getPos = getPos . tokPos
