   24 -- | Bit-blast a proposition and check its validity using SBV.
   25 -- Constants with names in @unints@ are kept as uninterpreted
   26 -- functions.
   27 proveUnintSBV ::
   28   SBV.SMTConfig {- ^ SBV configuration -} ->
   29   Maybe Integer {- ^ Timeout in milliseconds -} ->
   30   SATQuery      {- ^ A query to be proved -} ->
   31   TopLevel (Maybe CEX, String)
   32     -- ^ (example/counter-example, solver statistics)
   33 proveUnintSBV conf timeout satq =
   34   do sc <- getSharedContext
   35      io $ proveUnintSBVIO sc conf timeout satq
   36 
   37 proveUnintSBVIO ::
   38   SharedContext ->
   39   SBV.SMTConfig {- ^ SBV configuration -} ->
   40   Maybe Integer {- ^ Timeout in milliseconds -} ->
   41   SATQuery      {- ^ A query to be proved -} ->
   42   IO (Maybe CEX, String)
   43     -- ^ (example/counter-example, solver statistics)
   44 proveUnintSBVIO sc conf timeout satq =
   45   do p <- findExecutable . SBV.executable $ SBV.solver conf
   46      unless (isJust p) . fail $ mconcat
   47        [ "Unable to locate the executable \""
   48        , SBV.executable $ SBV.solver conf
   49        , "\" needed to run the solver "
   50        , show . SBV.name $ SBV.solver conf
   51        ]
   52 
   53      (labels, argNames, lit) <- SBVSim.sbvSATQuery sc mempty satq
   54 
   55      let script = maybe (return ()) SBV.setTimeOut timeout >> lit
   56 
   57      SBV.SatResult r <- SBV.satWith conf script
   58 
   59      let solver_name = "SBV->" ++ show (SBV.name (SBV.solver conf))
   60 
   61      case r of
   62        SBV.Unsatisfiable {} -> return (Nothing, solver_name)
   63 
   64        SBV.Satisfiable {} ->
   65          do let dict = SBV.getModelDictionary r
   66                 r'   = SBVSim.getLabels labels dict argNames
   67             return (r', solver_name)
   68 
   69        SBV.DeltaSat{} -> fail "Prover returned an unexpected DeltaSat result"
   70 
   71        SBV.SatExtField {} -> fail "Prover returned model in extension field"
   72 
   73        SBV.Unknown {} -> fail "Prover returned Unknown"
   74 
   75        SBV.ProofError _ ls _ -> fail . unlines $ "Prover returned error: " : ls
