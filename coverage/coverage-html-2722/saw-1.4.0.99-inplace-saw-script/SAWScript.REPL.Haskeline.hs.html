   35 -- | Haskeline-specific repl implementation.
   36 --
   37 -- XXX this needs to handle Ctrl-C, which at the moment will just cause
   38 -- haskeline to exit.  See the function 'withInterrupt' for more info on how to
   39 -- handle this.
   40 repl :: Maybe FilePath -> Options -> REPL () -> IO ()
   41 repl mbBatch opts begin =
   42   runREPL (isJust mbBatch) opts (replBody mbBatch begin)
   43 
   44 
   45 replBody :: Maybe FilePath -> REPL () -> REPL ()
   46 replBody mbBatch begin =
   47   do settings <- MTL.liftIO (setHistoryFile replSettings)
   48      enableSubshell (runInputTBehavior style settings body)
   49   where
   50   body = withInterrupt $ do
   51     MTL.lift begin
   52     loop
   53 
   54   style = case mbBatch of
   55     Nothing   -> defaultBehavior
   56     Just path -> useFile path
   57 
   58   enableSubshell m =
   59     REPL $ \refs ->
   60       do let ro' = (rTopLevelRO refs){ roSubshell = subshell (runInputT replSettings (withInterrupt loop)) }
   61          unREPL m refs{ rTopLevelRO = ro' }
   62 
   63   loop = do
   64     prompt <- MTL.lift getPrompt
   65     mb     <- handleInterrupt (return (Just "")) (getInputLines prompt [])
   66     case mb of
   67       Just line
   68         | Just cmd <- parseCommand findCommand line -> do
   69           continue <- MTL.lift $ do
   70             handleInterrupt handleCtrlC (runCommand cmd)
   71             shouldContinue
   72           when continue loop
   73 
   74         | otherwise -> loop
   75 
   76       Nothing -> return ()
   77 
   78   getInputLines prompt ls =
   79     do mb <- fmap (filter (/= '\r')) <$> getInputLine prompt
   80        let newPrompt = map (\_ -> ' ') prompt
   81        case mb of
   82           Nothing -> return Nothing
   83           Just l | not (null l) && last l == '\\' ->
   84                                       getInputLines newPrompt (init l : ls)
   85                  | otherwise -> return $ Just $ unlines $ reverse $ l : ls

   89 -- | Try to set the history file.
   90 setHistoryFile :: Settings REPL -> IO (Settings REPL)
   91 setHistoryFile ss =
   92   do dir <- getAppUserDataDirectory "saw"
   93      createDirectoryIfMissing True dir
   94      return ss { historyFile = Just (dir </> "history") }
   95    `X.catch` \(X.SomeException {}) -> return ss
   96 
   97 -- | Haskeline settings for the REPL.
   98 replSettings :: Settings REPL
   99 replSettings  = Settings
  100   { complete       = replComp
  101   , historyFile    = Nothing
  102   , autoAddHistory = True
  103   }

  125 instance E.MonadThrow REPL where
  126   throwM e = REPL $ \_ -> X.throwIO e
  127 
  128 instance E.MonadCatch REPL where
  129   catch repl_op handler = REPL $ \ioref ->
  130     E.catch (unREPL repl_op ioref) (\e -> unREPL (handler e) ioref)
  131 
  132 instance E.MonadMask REPL where
  133   mask repl_op = REPL $ \ioref ->
  134     E.mask (\runIO -> unREPL (repl_op (\f -> REPL (\ioref' -> runIO (unREPL f ioref')))) ioref)
  135   uninterruptibleMask repl_op = REPL $ \ioref ->
  136     E.uninterruptibleMask (\runIO -> unREPL (repl_op (\f -> REPL (\ioref' -> runIO (unREPL f ioref')))) ioref)
  137   generalBracket acquire release repl_op = REPL $ \ioref ->
  138     E.generalBracket
  139     (unREPL acquire ioref)
  140     (\rsrc exitCase -> unREPL (release rsrc exitCase) ioref)
  141     (\rsrc -> unREPL (repl_op rsrc) ioref)
  142 #endif

  147 -- | Top-level completion for the REPL.
  148 replComp :: CompletionFunc REPL
  149 replComp cursor@(l,r)
  150   | ":" `isPrefixOf` l'
  151   , Just (cmd,rest) <- splitCommand l' = case findCommand cmd of
  152 
  153       [c] | null rest && not (any isSpace l') -> do
  154             return (l, [cmdComp cmd c])
  155           | otherwise -> do
  156             (rest',cs) <- cmdArgument (cBody c) (reverse (sanitize rest),r)
  157             return (unwords [rest', reverse cmd],cs)
  158 
  159       cmds ->
  160         return (l, [ cmdComp l' c | c <- cmds ])
  161 
  162   | l' == ":" =
  163       return (l, [ cmdComp l' c | c <- findCommand "" ])
  164   | otherwise = completeSAWScriptValue cursor
  165   where
  166   l' = sanitize (reverse l)
  167 
  168 -- | Generate a completion from a REPL command definition.
  169 cmdComp :: String -> CommandDescr -> Completion
  170 cmdComp prefix c = Completion
  171   { replacement = drop (length prefix) (cName c)
  172   , display     = cName c
  173   , isFinished  = True
  174   }
  175 
  176 -- | Dispatch to a completion function based on the kind of completion the
  177 -- command is expecting.
  178 cmdArgument :: CommandBody -> CompletionFunc REPL
  179 cmdArgument ct cursor@(l,_) = case ct of
  180   ExprArg _     -> completeSAWScriptValue cursor
  181   TypeArg _     -> completeSAWScriptType cursor
  182   FilenameArg _ -> completeFilename cursor
  183   ShellArg _    -> completeFilename cursor
  184   NoArg       _ -> return (l,[])
  185 
  186 data LexerMode = ModeNormal | ModeCryptol | ModeCryType | ModeQuote
  187 
  188 lexerMode :: String -> LexerMode
  189 lexerMode = normal
  190   where
  191     normal [] = ModeNormal
  192     normal ('{' : '{' : s) = cryptol s
  193     normal ('{' : '|' : s) = crytype s
  194     normal ('\"' : s) = quote s
  195     normal (_ : s) = normal s
  196 
  197     cryptol [] = ModeCryptol
  198     cryptol ('}' : '}' : s) = normal s
  199     cryptol (_ : s) = cryptol s
  200 
  201     crytype [] = ModeCryType
  202     crytype ('|' : '}' : s) = normal s
  203     crytype (_ : s) = crytype s
  204 
  205     quote [] = ModeQuote
  206     quote ('\"' : s) = normal s
  207     quote (_ : s) = quote s
  208 
  209 isIdentChar :: Char -> Bool
  210 isIdentChar c = isAlphaNum c || c `elem` "_\'"
  211 
  212 -- | Complete a name from the SAWScript value environment.
  213 completeSAWScriptValue :: CompletionFunc REPL
  214 completeSAWScriptValue cursor@(l, _) = do
  215   ns1 <- getSAWScriptValueNames
  216   ns2 <- getCryptolExprNames
  217   ns3 <- getCryptolTypeNames
  218   let n = reverse (takeWhile isIdentChar l)
  219       nameComps prefix ns = map (nameComp prefix) (filter (prefix `isPrefixOf`) ns)
  220   case lexerMode (reverse l) of
  221     ModeNormal  -> return (l, nameComps n ns1)
  222     ModeCryptol -> return (l, nameComps n ns2)
  223     ModeCryType -> return (l, nameComps n ns3)
  224     ModeQuote   -> completeFilename cursor
  225 
  226 -- | Complete a name from the SAWScript type environment.
  227 completeSAWScriptType :: CompletionFunc REPL
  228 completeSAWScriptType (l, _) = do
  229   ns <- getSAWScriptTypeNames
  230   let prefix = reverse (takeWhile isIdentChar l)
  231       nameComps = map (nameComp prefix) (filter (prefix `isPrefixOf`) ns)
  232   return (l, nameComps)
  233 
  234 -- | Generate a completion from a prefix and a name.
  235 nameComp :: String -> String -> Completion
  236 nameComp prefix c = Completion
  237   { replacement = drop (length prefix) c
  238   , display     = c
  239   , isFinished  = True
  240   }
