  103 -- | Make sure the class is in the database and allocate handles for its
  104 -- methods and static fields
  105 --
  106 loadJavaClass :: Text -> TopLevel J.Class
  107 loadJavaClass str = do
  108   -- XXX fix prepareClassTopLevel to return c so we don't have to look
  109   -- it up twice
  110   cb <- getJavaCodebase
  111   c <- io $ findClass cb (Text.unpack str)
  112   prepareClassTopLevel str
  113   return c
  114 
  115 -----------------------------------------------------------------------
  116 -- | Allocate the method handles/global static variables for the given
  117 -- class and add them to the current translation context
  118 prepareClassTopLevel :: Text -> TopLevel ()
  119 prepareClassTopLevel str = do
  120 
  121    cb <- getJavaCodebase
  122 
  123    -- get class from codebase
  124    c <- io $ findClass cb (Text.unpack str)
  125 
  126    -- get current ctx
  127    ctx0 <- getJVMTrans
  128 
  129    -- make sure that we haven't already processed this class
  130    unless (Map.member (J.className c) (CJ.classTable ctx0)) $ do
  131 
  132      -- add handles/global variables for this class
  133      halloc <- getHandleAlloc
  134      ctx <- io $ execStateT (CJ.extendJVMContext halloc c) ctx0
  135 
  136      -- update ctx
  137      putJVMTrans ctx
  138

  143 -- | Extract a JVM method to saw-core
  144 --
  145 jvm_extract :: J.Class -> Text -> TopLevel TypedTerm
  146 jvm_extract c mname = do
  147   sc <- getSharedContext
  148   cb <- getJavaCodebase
  149   opts <- getOptions
  150   pathSatSolver <- gets rwPathSatSolver
  151   let verbosity = simVerbose opts
  152   let gen       = Nonce.globalNonceGenerator
  153 
  154 
  155   traceM $ "extracting " ++ Text.unpack mname
  156   (mcls, meth) <- io $ CJ.findMethod cb (Text.unpack mname) c
  157   when (not (J.methodIsStatic meth)) $ do
  158        fail $ unlines [ "Crucible can only extract static methods" ]
  159 
  160   let className = J.className c
  161 
  162   -- allocate all of the handles/static vars that are directly referenced by
  163   -- this class
  164   let refs = CJ.initClasses ++ Set.toList (CJ.classRefs c)
  165   mapM_ (prepareClassTopLevel . Text.pack . J.unClassName) refs
  166 
  167   halloc <- getHandleAlloc
  168   ctx <- getJVMTrans
  169 
  170   io $ do -- only the IO monad, nothing else
  171           sym <- newSAWCoreExprBuilder sc False
  172           SomeOnlineBackend bak <- newSAWCoreBackend pathSatSolver sym
  173           st  <- sawCoreState sym
  174           CJ.setSimulatorVerbosity verbosity sym
  175 
  176           (CJ.JVMHandleInfo _m2 h) <- CJ.findMethodHandle ctx mcls meth
  177 
  178           (ecs, args) <- setupArgs sc sym h
  179 
  180           res <- CJ.runMethodHandle bak SAWCruciblePersonality halloc
  181                      ctx verbosity className h args
  182 
  183           case res of
  184             Crucible.FinishedResult _ pr -> do
  185               gp <- getGlobalPair opts pr
  186               let regval = gp^.Crucible.gpValue
  187               let regty = Crucible.regType regval
  188               let failure :: forall a. IO a
  189                   failure = fail $ unwords ["Unexpected return type:", show regty]
  190               t <- Crucible.asSymExpr regval (toSC sym st) failure
  191               cty <-
  192                 case Crucible.asBaseType regty of
  193                   Crucible.NotBaseType -> failure
  194                   Crucible.AsBaseType bt ->
  195                     case baseCryptolType bt of
  196                       Nothing -> failure
  197                       Just cty -> return cty
  198               let tt = TypedTerm (TypedTermSchema (Cryptol.tMono cty)) t
  199               abstractTypedVars sc (toList ecs) tt
  200             Crucible.AbortedResult _ _ar -> do
  201               fail $ unlines [ "Symbolic execution failed." ]
  202             Crucible.TimeoutResult _cxt -> do
  203               fail $ unlines [ "Symbolic execution timed out." ]

  205 -- | Create a fresh argument variable of the appropriate type, suitable for use
  206 -- in an extracted function derived from @jvm_extract@.
  207 setupArg ::
  208   SharedContext ->
  209   Sym ->
  210   -- | The function arguments extracted so far.
  211   IORef (Seq TypedVariable) ->
  212   Crucible.TypeRepr tp ->
  213   IO (Crucible.RegEntry Sym tp)
  214 setupArg sc sym ecRef tp = do
  215   (cty, scTp) <- typeReprToSAWTypes sym sc tp
  216   ecs <- readIORef ecRef
  217   vn <- scFreshVarName sc ("arg_" <> Text.pack (show (length ecs)))
  218   writeIORef ecRef (ecs Seq.|> TypedVariable cty vn scTp)
  219   t <- scVariable sc vn scTp
  220   Crucible.RegEntry tp <$> termToRegValue sym tp t
  221 
  222 -- | Create fresh argument variables of the appropriate types, suitable for use
  223 -- in an extracted function derived from @jvm_extract@.
  224 setupArgs ::
  225   SharedContext ->
  226   Sym ->
  227   Crucible.FnHandle init ret ->
  228   IO (Seq TypedVariable, Crucible.RegMap Sym init)
  229 setupArgs sc sym fn =
  230   do ecRef  <- newIORef Seq.empty
  231      regmap <- Crucible.RegMap <$> FC.traverseFC (setupArg sc sym ecRef) (Crucible.handleArgTypes fn)
  232      ecs    <- readIORef ecRef
  233      return (ecs, regmap)
