   41 randomFirstOrderValue :: (Applicative m, Functor m, MonadRandom m) =>
   42   FirstOrderType -> Compose Maybe m FirstOrderValue
   43 randomFirstOrderValue FOTBit =
   44   Compose (Just (FOVBit <$> getRandom))
   45 randomFirstOrderValue FOTInt =
   46   Compose (Just (FOVInt <$> randomInt))
   47 randomFirstOrderValue (FOTIntMod m) =
   48   Compose (Just (FOVIntMod m <$> getRandomR (0, toInteger m - 1)))
   49 randomFirstOrderValue (FOTVec n FOTBit) =
   50   Compose (Just (FOVWord n <$> getRandomR (0, 2^n - 1)))
   51 randomFirstOrderValue (FOTVec n t) =
   52   FOVVec t <$> replicateM (fromIntegral n) (randomFirstOrderValue t)
   53 randomFirstOrderValue (FOTTuple ts) =
   54   FOVTuple <$> traverse randomFirstOrderValue ts
   55 randomFirstOrderValue (FOTRec fs) =
   56   FOVRec <$> traverse randomFirstOrderValue fs
   57 randomFirstOrderValue (FOTArray _ _) = Compose Nothing

   66 execTest ::
   67   (F.MonadFail m, MonadRandom m, MonadIO m) =>
   68   SharedContext ->
   69   ModuleMap ->
   70   Map.Map VarName (m FirstOrderValue) ->
   71   Term ->
   72   m (Maybe [(VarName, FirstOrderValue)])
   73 execTest sc mmap vars tm =
   74   do testVec <- sequence vars
   75      tm' <- liftIO $
   76              do argMap0 <- traverse (scFirstOrderValue sc) testVec
   77                 let argMap = IntMap.fromList [ (vnIndex x, v) | (x, v) <- Map.toList argMap0 ]
   78                 scInstantiateExt sc argMap tm
   79      case evalSharedTerm mmap Map.empty Map.empty tm' of
   80        -- satisfaible, return counterexample
   81        VBool True  -> return (Just (Map.toList testVec))
   82        -- not satisfied by this test vector
   83        VBool False -> return Nothing
   84        _ -> fail "execTest: expected boolean result from random testing"
   85 
   86 prepareSATQuery ::
   87   (MonadRandom m, F.MonadFail m, MonadIO m) =>
   88   SharedContext ->
   89   SATQuery ->
   90   IO (m (Maybe [(VarName, FirstOrderValue)]))
   91 prepareSATQuery sc satq
   92   | Set.null (satUninterp satq) =
   93        do varmap <- traverse prepareVar (satVariables satq)
   94           t <- satQueryAsTerm sc satq
   95           mmap <- scGetModuleMap sc
   96           return (execTest sc mmap varmap t)
   97   | otherwise = fail "Random testing cannot handle uninterpreted functions"
   98 
   99  where
  100    prepareVar fot =
  101      case randomFirstOrderValue fot of
  102        Compose (Just v) -> pure v
  103        _ -> fail ("Cannot randomly test argument of type: " ++ show fot)

  105 runManyTests ::
  106   RandT TFGen IO (Maybe [(VarName, FirstOrderValue)]) ->
  107   Integer ->
  108   IO (Maybe [(VarName, FirstOrderValue)])
  109 runManyTests m numtests = evalRandT (loop numtests) =<< newTFGen
  110   where
  111     loop n
  112       | n > 0 =
  113           m >>= \case
  114             Nothing  -> loop (n-1)
  115             Just cex -> return (Just cex)
  116 
  117       | otherwise = return Nothing
