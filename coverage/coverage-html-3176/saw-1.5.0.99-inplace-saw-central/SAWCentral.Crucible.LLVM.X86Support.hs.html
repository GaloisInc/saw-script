   87 data Unit = Bytes | Words | DWords | QWords | V128s | V256s deriving Show
   88 
   89 
   90 (*.) :: Integer -> Unit -> Bytes
   91 n *. u = toBytes (fromInteger n * bs)
   92   where bs = unitByteSize u natValue :: Natural
   93 
   94 unitByteSize :: Unit -> (forall w. (1 <= w) => NatRepr w -> a) -> a
   95 unitByteSize u k =
   96   case u of
   97     Bytes  -> k (knownNat @1)
   98     Words  -> k (knownNat @2)
   99     DWords -> k (knownNat @4)
  100     QWords -> k (knownNat @8)
  101     V128s  -> k (knownNat @16)
  102     V256s  -> k (knownNat @32)

  107 freshRegister :: Sym -> Ctx.Index ctx tp -> TypeRepr tp -> IO (RegValue' Sym tp)
  108 freshRegister sym idx repr = RV <$> freshVal sym repr True ("reg" ++ show idx)
  109 
  110 freshVal ::
  111   Sym -> TypeRepr t -> Bool {- ptrOK ?-}-> String -> IO (RegValue Sym t)
  112 freshVal sym t ptrOk nm =
  113   case t of
  114     BoolRepr -> do
  115       sn <- symName nm
  116       freshConstant sym sn BaseBoolRepr
  117     LLVMPointerRepr w
  118       | ptrOk, Just Refl <- testEquality w (knownNat @64) -> do
  119           sn_base <- symName (nm ++ "_base")
  120           sn_off <- symName (nm ++ "_off")
  121           base <- freshNat sym sn_base
  122           off <- freshConstant sym sn_off (BaseBVRepr w)
  123           return (LLVMPointer base off)
  124       | otherwise -> do
  125           sn <- symName nm
  126           base <- natLit sym 0
  127           off <- freshConstant sym sn (BaseBVRepr w)
  128           return (LLVMPointer base off)
  129     it -> fail ("[freshVal] Unexpected type repr: " ++ show it)
  130 
  131   where
  132   symName s =
  133     case userSymbol ("macaw_" ++ s) of
  134       Left err -> error ("Invalid symbol name " ++ show s ++ ": " ++ show err)
  135       Right a -> return a

  140 -- | Implements a layer to map 'LLVMPtr's to their underlying allocations, as
  141 -- tracked by the 'RegionIndex' map
  142 --
  143 -- NOTE: If the initial obvious mapping (where the concrete nat is in the map)
  144 -- fails, there are two possibilities:
  145 --
  146 -- 1. The region ID is concrete but not in the map.  We should just pass it
  147 --    through without modifying it, since it is a valid LLVM pointer
  148 -- 2. The region ID is symbolic.  In this case, we need to generate a mux that
  149 --    dispatches to the entries in the map when they match, or otherwise passes
  150 --    the pointer through untouched.
  151 --
  152 -- If the region ID is concretely zero, it should be the case that the
  153 -- 'RegionIndex' map would translate it into a real 'LLVMPtr' since the only map
  154 -- entry (established in 'setupGlobals') is for 0.
  155 mkGlobalMap ::
  156   (?memOpts::Crucible.MemOptions, Crucible.HasLLVMAnn Sym) =>
  157   Map RegionIndex (LLVMPtr Sym 64) ->
  158   GlobalMap Sym Crucible.Mem 64
  159 mkGlobalMap rmap = GlobalMap $ \bak mem region off ->
  160   let
  161     sym = backendGetSym bak
  162 
  163     mapConcreteRegion = maybe mzero id (addOffset <$> thisRegion)
  164     thisRegion = join (findRegion <$> asNat region)
  165     findRegion r = Map.lookup (fromIntegral r) rmap
  166     addOffset p = doPtrAddOffset bak mem p off
  167       where ?ptrWidth = knownNat
  168     passThroughConcreteRegion =
  169       case asNat region of
  170         Nothing -> mzero
  171         Just _ -> return (LLVMPointer region off)
  172     -- If the symbolic nat is (symbolically) equal to any of the entries in the
  173     -- rmap, we need to do the translation; otherwise, we pass it through
  174     mapSymbolicRegion = foldlM muxSymbolicRegion (LLVMPointer region off) (Map.toList rmap)
  175     muxSymbolicRegion others (regionNum, basePtr) = do
  176       thisRegionNat <- natLit sym (fromIntegral regionNum)
  177       isEqRegion <- natEq sym thisRegionNat region
  178       adjustedPtr <- addOffset basePtr
  179       muxLLVMPtr sym isEqRegion adjustedPtr others
  180 
  181   in mapConcreteRegion <|> passThroughConcreteRegion <|> mapSymbolicRegion

  187 -- | These are the parts of the ELF file that we care about.
  188 data RelevantElf = RelevantElf
  189   { memory    :: Memory 64
  190   , funSymMap :: AddrSymMap 64
  191   , symMap    :: AddrSymMap 64
  192   }
  193 
  194 -- | Parse an elf file.
  195 getElf :: FilePath -> IO (Elf.ElfHeaderInfo 64)
  196 getElf path =
  197   do bs <- BS.readFile path
  198      case Elf.decodeElfHeaderInfo bs of
  199        Left (off, msg) ->
  200            malformed path $ "Invalid ELF header at offset " <> Text.pack (show off) <>
  201                             ": " <> Text.pack msg
  202        Right (Elf.SomeElf hdr) ->
  203            let elfmachine = Elf.headerMachine (Elf.header hdr)
  204                elfclass = Elf.headerClass (Elf.header hdr)
  205            in case (elfmachine, elfclass) of
  206                (Elf.EM_X86_64, Elf.ELFCLASS64) ->
  207                    pure hdr
  208                (Elf.EM_X86_64, _) ->
  209                    -- Note that 32-bit x86 is a different machine; if
  210                    -- we do see a 32-bit x86_64 bin though it might be
  211                    -- one of the several 32-on-64 ABIs (akin to mips
  212                    -- N32) that haven't caught on, so call it
  213                    -- unsupported rather than malformed.
  214                    unsupported path $ "Unexpected ELF class " ++ show elfclass
  215                (_, _) ->
  216                    unsupported path $ "Unexpected ELF machine " ++ show elfmachine
  217

  219 -- | Extract a Macaw "memory" from an ELF file and resolve symbols.
  220 getRelevant :: FilePath -> Elf.ElfHeaderInfo 64 -> IO RelevantElf
  221 getRelevant path elf =
  222   case (memoryForElf opts elf, memoryForElfAllSymbols opts elf) of
  223     (Left err, _) -> malformed path $ Text.pack err
  224     (_, Left err) -> malformed path $ Text.pack err
  225     (Right (mem, faddrs, _warnings, _errs), Right (_, addrs, _, _)) ->
  226       do let toEntry msym = (memSymbolStart msym, memSymbolName msym)
  227          return RelevantElf { memory = mem
  228                             , funSymMap = Map.fromList (map toEntry faddrs)
  229                             , symMap = Map.fromList (map toEntry addrs)
  230                             }
  231 
  232   where
  233   -- XXX: What options do we want?
  234   opts = LoadOptions { loadOffset = Just 0
  235                      }

  240 -- | Find the address(es) of a symbol by name.
  241 findSymbols :: AddrSymMap 64 -> ByteString -> [ MemSegmentOff 64 ]
  242 findSymbols addrs nm = Map.findWithDefault [] nm invertedMap
  243   where
  244   invertedMap = Map.fromListWith (++) [ (y,[x]) | (x,y) <- Map.toList addrs ]
  245 
  246 loadGlobal ::
  247   RelevantElf ->
  248   (ByteString, Integer, Unit) ->
  249   IO [(String, Integer, Unit, [Integer])]
  250 loadGlobal elf (nm,n,u) =
  251   case findSymbols (symMap elf) nm of
  252     [] -> do print $ symMap elf
  253              err "Global not found"
  254     _  -> mapM loadLoc (findSymbols (symMap elf) nm)
  255   where
  256   mem   = memory elf
  257   sname = BSC.unpack nm
  258 
  259   readOne a = case u of
  260                 Bytes  -> check (readWord8    mem a)
  261                 Words  -> check (readWord16le mem a)
  262                 DWords -> check (readWord32le mem a)
  263                 QWords -> check (readWord64le mem a)
  264                 _      -> err ("unsupported global size: " ++ show u)
  265 
  266   nextAddr = incAddr (bytesToInteger (1 *. u))
  267 
  268   addrsFor o = take (fromIntegral n) (iterate nextAddr o)
  269 
  270   check :: (Show b, Integral a) => Either b a -> IO Integer
  271   check res = case res of
  272                 Left e  -> err (show e)
  273                 Right a -> return (fromIntegral a)
  274 
  275 
  276   loadLoc off = do let start = segoffAddr off
  277                        a  = memWordToUnsigned (addrOffset start)
  278                    is <- mapM readOne (addrsFor start)
  279                    return (sname, a, u, is)
  280 
  281   err :: [Char] -> IO a
  282   err xs = fail $ unlines
  283                     [ "Failed to load global."
  284                     , "*** Global: " ++ show nm
  285                     , "*** Error: " ++ xs
  286                     ]

  288 -- | The position associated with a specific location.
  289 posFn :: MemSegmentOff 64 -> Position
  290 posFn = OtherPos . Text.pack . show

  294 -- Calling Convention
  295 -- see: http://refspecs.linuxfoundation.org/elf/x86_64-abi-0.99.pdf
  296 -- Need to preserve: %rbp, %rbx, %r12--%r15
  297 -- Preserve control bits in MXCSR
  298 -- Preserve x87 control word.
  299 -- On entry:
  300 --   CPU is in x87 mode
  301 --   DF in $rFLAGS is clear one entry and return.
  302 -- "Red zone" 128 bytes past the end of the stack %rsp.
  303 --    * not modified by interrupts

  309 -- | Exception for hitting an unsupported object or feature. The arguments
  310 --   are the filename we were looking at, and a message.
  311 data X86Unsupported = X86Unsupported FilePath String deriving Show
  312 
  313 -- | Exception for miscellaneous errors during verification. The arguments
  314 --   are the filename we were looking at, also optionally a function/symbol
  315 --   name, and a message.
  316 data X86Error       = X86Error FilePath (Maybe Text) Text deriving Show
  317 
  318 instance Exception X86Unsupported
  319 instance Exception X86Error
  320 
  321 unsupported :: FilePath -> String -> IO a
  322 unsupported path x = throwIO (X86Unsupported path x)
  323 
  324 malformed :: FilePath -> Text -> IO a
  325 malformed path x = throwIO (X86Error path Nothing x)
