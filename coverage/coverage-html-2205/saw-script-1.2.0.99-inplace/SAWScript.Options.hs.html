   49 -- | Verbosity is currently a linear setting (vs a mask or tree).  Any given
   50 -- level includes the outputs of all lower levels.
   51 data Verbosity
   52   = Silent
   53   | OnlyCounterExamples
   54   | Error
   55   | Warn
   56   | Info
   57   | Debug
   58   | ExtraDebug
   59     deriving (Show,Eq,Ord)

   87 printOutWith :: Verbosity -> Verbosity -> String -> IO ()
   88 printOutWith setting level msg
   89     | setting >= level = do
   90       t <- formatTime defaultTimeLocale "%T.%3q" <$> getCurrentTime
   91       putStr $ "[" ++ t ++ "] " ++ msg
   92     | otherwise        = return ()
   93 
   94 printOutLn :: Options -> Verbosity -> String -> IO ()
   95 printOutLn o v s = printOutFn o v (s ++ "\n")

  188 -- Try to read verbosity as either a string or number and default to 'Debug'.
  189 readVerbosity :: String -> Verbosity
  190 readVerbosity s | Just (n::Integer) <- readMaybe s =
  191      case n of
  192          0 -> Silent
  193          1 -> OnlyCounterExamples
  194          2 -> Error
  195          3 -> Warn
  196          4 -> Info
  197          _ -> Debug
  198 readVerbosity s =
  199     case map toLower s of
  200         "silent"              -> Silent
  201         "counterexamples"     -> OnlyCounterExamples
  202         "onlycounterexamples" -> OnlyCounterExamples
  203         "error"               -> Error
  204         "warn"                -> Warn
  205         "warning"             -> Warn
  206         "info"                -> Info
  207         "debug"               -> Debug
  208         _                     -> Debug

  210 -- | Perform some additional post-processing on an 'Options' value based on
  211 -- whether certain environment variables are defined.
  212 processEnv :: Options -> IO Options
  213 processEnv opts = do
  214   curEnv <- getEnvironment
  215   opts' <- addJavaBinDirInducedOpts opts
  216   return $ foldr addSawOpt opts' curEnv
  217   where
  218     -- If a Java executable's path is specified (either by way of
  219     -- --java-bin-dirs or PATH, see the Haddocks for findJavaIn), then use that
  220     -- to detect the path to Java's standard rt.jar file and add it to the
  221     -- jarList on Java 8 or earlier. (Later versions of Java do not use
  222     -- rt.jar—see Note [Loading classes from JIMAGE files] in
  223     -- Lang.JVM.Codebase in crucible-jvm.)
  224     -- If Java's path is not specified, return the Options unchanged.
  225     addJavaBinDirInducedOpts :: Options -> IO Options
  226     addJavaBinDirInducedOpts os@Options{javaBinDirs} = do
  227       mbJavaPath <- findJavaIn javaBinDirs
  228       case mbJavaPath of
  229         Nothing       -> pure os
  230         Just javaPath -> do
  231           javaMajorVersion <- findJavaMajorVersion javaPath
  232           if javaMajorVersion >= 9
  233              then pure os
  234              else addRTJar javaPath os
  235 
  236     -- rt.jar lives in a standard location relative to @java.home@. At least,
  237     -- this is the case on every operating system I've tested.
  238     addRTJar :: FilePath -> Options -> IO Options
  239     addRTJar javaPath os = do
  240       mbJavaHome <- findJavaProperty javaPath "java.home"
  241       case mbJavaHome of
  242         Nothing -> fail $ "Could not find where rt.jar lives for " ++ javaPath
  243         Just javaHome ->
  244           let rtJarPath = javaHome </> "lib" </> "rt.jar" in
  245           pure $ os{ jarList = rtJarPath : jarList os }
  246 
  247     addSawOpt :: (String, String) -> Options -> Options
  248     addSawOpt ("SAW_IMPORT_PATH", p) os =
  249       os { importPath = importPath os ++ splitSearchPath p }
  250     addSawOpt ("SAW_JDK_JAR", p) os = os { jarList = p : jarList os }
  251     addSawOpt ("CLASSPATH", p) os = os { jarList = jarList os ++ jars
  252                                        , classPath = classPath os ++ dirs
  253                                        }
  254                                     where
  255                                       es = splitSearchPath p
  256                                       (jars, dirs) = partition (".jar" `isExtensionOf`) es
  257     addSawOpt _ os = os
