   83 -- | ImportVisibility is an enumeration that indicates how we handle
   84 -- the visibility of the symbols in an imported module.
   85 --
   86 -- `OnlyPublic` makes only the public/exported symbols of a module
   87 -- visible from SAW. `PublicAndPrivate` instead makes all symbols
   88 -- visible, as if one is working inside it. The latter is often useful
   89 -- (or necessary) for verification and code generation.
   90 --
   91 -- `PublicAndPrivate` is akin to setting the module focus in the
   92 -- Cryptol REPL; however, it uses a simpler internal mechanism and is
   93 -- only settable at import time.
   94 --
   95 -- (See 'CryptolEnv.importCryptolModule'.)
   96 --
   97 -- NOTE: this notion of public vs. private symbols is specific to
   98 -- SAWScript and distinct from Cryptol's notion of private
   99 -- definitions.
  100 --
  101 -- FUTURE: this should probably be replaced with a way to manipulate
  102 -- the module focus like the Cryptol REPL uses. What you really want
  103 -- is not to expose module innards that weren't meant to be exposed,
  104 -- but to go inside to prove things in the module's internal context.
  105 --
  106 data ImportVisibility
  107   = OnlyPublic       -- ^ behaves like a normal Cryptol @import@
  108   | PublicAndPrivate -- ^ allows viewing of both @private@ sections
  109                      --   and (arbitrarily nested) submodules.
  110   deriving (Eq, Show)

  146 -- | Restore a `GlobalCryptolEnv` from a checkpoint. The first argument
  147 --   @chkEnv@ is the `GlobalCryptolEnv` saved by / copied into the
  148 --   checkpoint; the second argument @newEnv@ is the current one
  149 --   we wish to overwrite by rolling back to the checkpoint.
  150 --   The 'ME.meNameSeeds' and 'ME.meSupply' from the
  151 --   module environment are not rolled back, to avoid re-using old
  152 --   names.
  153 --   NOTE: This effectively
  154 --   invalidates any translated 'Term's or Cryptol expressions created
  155 --   after the checkpoint. Attempting to use them in the restored
  156 --   environment will have unpredictable results, and likely will
  157 --   result in a panic. Similarly, 'CryptolEnv's captured after the
  158 --   checkpoint are no longer safe to use in the resulting environment.
  159 
  160 --   We also ought to invalidate terms constructed since the checkpoint
  161 --   was taken, like SAWCore does. See #2859.
  162 
  163 --   We could, for example, have 'CryptolScope' track which
  164 --   identifiers it references, and check that they are in a valid
  165 --   range with respect to the corresponding global environment
  166 --   before combining them into a 'CryptolEnv'.
  167   restoreMetadata chk now = return $
  168     let newMEnv = geModuleEnv now
  169         chkMEnv = geModuleEnv chk
  170     in chk { geModuleEnv = chkMEnv 
  171                { ME.meNameSeeds = ME.meNameSeeds newMEnv
  172                , ME.meSupply = ME.meSupply newMEnv 
  173                }
  174            }

  176 -- | A scope frame that captures which Cryptol names are accessible.
  177 --  `fNamingEnv` is the local naming environment, which can be extended
  178 --  ad-hoc with additional declarations. `fImports` is a list of all
  179 --  the modules that have been imported, and a visibility setting for
  180 --  each. This does not include, for example, builtin modules that
  181 --  exist but that have not been imported.
  182 
  183 data CryptolFrame =
  184   CryptolFrame { fNamingEnv :: MR.NamingEnv
  185                , fImports :: [(ImportVisibility, C.Import)] 
  186                }

  188 initFrame :: CryptolFrame
  189 initFrame = CryptolFrame mempty mempty

  191 -- | A nonempty list of 'CryptolFrame's, where the first element is the
  192 --   "bottom" frame that takes highest precedence when looking up
  193 --   names. Each individual frame only contains values declared at
  194 --   exactly that level. The full scope is computed by collecting
  195 --   everything in this stack, via 'eExtraNaming' and 'eImports'.
  196 newtype CryptolEnv = CryptolEnv (NonEmpty CryptolFrame)
  197 
  198 initEnv :: CryptolEnv
  199 initEnv = CryptolEnv (initFrame :| [])
  200 
  201 isToplevel :: CryptolEnv -> Bool
  202 isToplevel (CryptolEnv (_ :| frames)) = null frames

  204 -- | Test if the scopes have the same number of frames pushed.
  205 sameHeight :: CryptolEnv -> CryptolEnv -> Bool
  206 sameHeight (CryptolEnv scope1) (CryptolEnv scope2) = 
  207   NE.length scope1 == NE.length scope2
  208 
  209 mapCurFrame :: 
  210   (CryptolFrame -> CryptolFrame) -> 
  211   CryptolEnv -> 
  212   CryptolEnv
  213 mapCurFrame f (CryptolEnv (frame :| frames)) =
  214   CryptolEnv (f frame :| frames)
  215 
  216 -- | Push a fresh frame onto the stack.
  217 pushScope :: CryptolEnv -> CryptolEnv
  218 pushScope (CryptolEnv frames) = CryptolEnv (initFrame <| frames)
  219 
  220 -- | Pop the current frame from the stack, discarding its
  221 --   contents. Panics if this is the only frame.
  222 popScope :: CryptolEnv -> CryptolEnv
  223 popScope (CryptolEnv frames) = case snd (NE.uncons frames) of
  224   Nothing -> panic "popScope" [ "Popping topmost scope"]
  225   Just frames' -> CryptolEnv frames'
  226 
  227 -- | Map the naming environment of the frame currently in scope.
  228 mapNaming :: 
  229   (MR.NamingEnv -> MR.NamingEnv) -> 
  230   CryptolEnv -> 
  231   CryptolEnv
  232 mapNaming f = mapCurFrame $ 
  233       \fr -> fr {fNamingEnv = f (fNamingEnv fr) }
  234 
  235 -- | Map the module imports of the frame currently in scope.
  236 mapImports :: 
  237   ([(ImportVisibility, C.Import)] -> [(ImportVisibility, C.Import)] ) -> 
  238   CryptolEnv -> 
  239   CryptolEnv
  240 mapImports f = mapCurFrame $ 
  241       \fr -> fr {fImports = f (fImports fr) }

  243 -- | Run the inner action bracketed new frame pushed/popped on
  244 --   the 'CryptolScope' stack.
  245 --   Fails if the inner action changes the scope height
  246 --   (i.e. it does not properly bracket its pushes and pops).
  247 withFreshScope :: 
  248   MonadFail m =>
  249   CryptolEnv -> 
  250   (CryptolEnv -> 
  251   m (a, CryptolEnv)) -> 
  252   m (a, CryptolEnv)
  253 withFreshScope env0 f = do
  254   let env1 = pushScope env0
  255   (a, env2) <- f env1
  256   unless (sameHeight env1 env2) $
  257     panic "withFreshScope" ["mismatched push/pops"]
  258   let env3 = popScope env2
  259   return (a, env3)

  261 -- | Access the 'C.Supply' in the global 'ME.ModuleEnv' for generating
  262 --   fresh names. More efficient than 'runModuleM', as it avoids any other
  263 --   bookkeeping.
  264 useModEnvSupply :: SharedContext -> (C.Supply -> (a, C.Supply)) -> IO a
  265 useModEnvSupply sc f = do
  266   modEnv <- eModuleEnv sc
  267   let (a, supply) = f $ ME.meSupply modEnv
  268   mapModEnv sc (\modEnv_ -> modEnv_ { ME.meSupply = supply } )
  269   return a

  283 -- The "getters" below were historically fields in 'CryptolEnv', which
  284 -- are now defined functions that access either the 'GlobalCryptolEnv'
  285 -- or the 'CryptolScope' as appropriate. In contrast, updates were
  286 -- historically made by directly accessing 'CryptolEnv' fields, and are
  287 -- now restricted to only adding new entries to the maps in
  288 -- 'GlobalCryptolEnv' (e.g. 'addExtraVars'). This enforces the policy
  289 -- that the global environment is intended to be a complete record of
  290 -- all work done during translation/import, regardless of scope. Since
  291 -- the maps in the global environment are all keyed on globally-unique
  292 -- values, these operations are expected to only add entries and never
  293 -- overwrite existing ones.
  294 --
  295 -- NOTE: We could enforce a write-once policy here, but it's
  296 -- not immediately obvious if we need to support key clashes
  297 -- for equal entries (not possible for
  298 -- some types like 'C.Expr'), or if it is even useful to do so.

  368 -- | Run an 'MM.ModuleM' action using the module environment from
  369 -- 'eModuleEnv'. If the action is successful, updates the module
  370 -- environment in the 'SharedContext' and returns the result of the
  371 -- action. Otherwise, the environment is left unchanged and a
  372 -- 'MM.ModuleError' is returned.
  373 runModuleM ::
  374  SharedContext -> MM.ModuleM a -> IO (Either MM.ModuleError a, [MM.ModuleWarning])
  375 runModuleM sc m = do
  376   fileReader <- scFileReader sc
  377   env <- eModuleEnv sc
  378   let minp solver = MM.ModuleInput {
  379           MM.minpCallStacks = True,
  380           MM.minpSaveRenamed = False,
  381           MM.minpEvalOpts = pure defaultEvalOpts,
  382           MM.minpByteReader = fileReader,
  383           MM.minpModuleEnv = env,
  384           MM.minpTCSolver = solver
  385       }
  386   (res, ws) <- SMT.withSolver (return ()) (meSolverConfig env) $ \solver ->
  387     MM.runModuleM (minp solver) m
  388   case res of
  389     Right (a, env') -> do
  390       mapGlobal sc $ \genv ->
  391         refreshCryptolEnv $ genv { geModuleEnv = env' }
  392       return (Right a, ws)
  393     Left err -> return (Left err, ws)

  518 -- | The "extra" naming environment that captures Cryptol names
  519 --   which don't correspond to any imported module. Generally these
  520 --   result from names created in SAW which have been reflected into
  521 --   the Cryptol environment.
  522 --   This is scoped content, where the accessible names are expected

  526 -- FUTURE: Cryptol has its own functionality for additional bindings
  527 -- (it uses it for things created from the Cryptol REPL) and we ought

  533 eExtraNaming (CryptolEnv (frame :| frames)) = 
  534   foldr (\fr ne -> ne `MR.shadowing` (fNamingEnv fr)) (fNamingEnv frame) frames

  536 -- | The list of Cryptol modules which have been brought into the
  537 --   current scope. This essentially acts as a filter on the full set
  538 --   of loaded modules in the global module environment ('eModuleEnv'),
  539 --   where the contents of the selected modules are brought into scope
  540 --   according to the associated 'ImportVisibility'. The modules here
  541 --   should only correspond to modules that are present in the module
  542 --   environment *and* have been translated into SAWCore.
  543 eImports :: CryptolEnv -> [(ImportVisibility, C.Import)]

  545   concat $ map fImports $ NE.toList frames

  548 -- | Refresh 'geAllVars' after updating the module environment.
  549 --   Previously (before 'GlobalCryptolEnv'), this would overwrite the
  550 --   'eAllVars' field. Now this will add new vars (i.e. from
  551 --   newly-added modules) but keep existing ones, as the
  552 --   'GlobalCryptolEnv' should never drop entries. If the module
  553 --   environment has been managed properly, we expect any clashing keys
  554 --   to have equal elements, but this is not checked/enforced. Not
  555 --   exported.
  556 refreshCryptolEnv :: GlobalCryptolEnv -> GlobalCryptolEnv
  557 refreshCryptolEnv genv =
  558      let modEnv = geModuleEnv genv
  559          ifaceDecls = getAllIfaceDecls modEnv
  560          newtypeCons = Map.fromList
  561                          [ con
  562                          | nt <- Map.elems (MI.ifNominalTypes ifaceDecls)
  563                          , con <- C.nominalTypeConTypes nt
  564                          ]
  565          vars = Map.map MI.ifDeclSig $ MI.ifDecls ifaceDecls
  566          -- note that we don't need to re-add geExtraVars, because
  567          -- it is already included in the existing geAllVars
  568          allvars = newtypeCons `Map.union` vars
  569          allvars' = Map.union allvars (geAllVars genv)
  570      in genv { geAllVars = allvars' }
