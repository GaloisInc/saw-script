    4 module CryptolSAWCore.FileReader 
    5   ( withFileReader
    6   , scFileReader 
    7   ) where
    8 
    9 import Control.Monad.IO.Class
   10 import Data.ByteString (ByteString)
   11 import qualified Data.ByteString as BS
   12 import Data.Typeable
   13 
   14 import SAWCore.SharedTerm (SharedContext, IsMetadata(..), scGetData, scWithData)
   15 
   16 newtype FileReader = FileReader (FilePath -> IO ByteString)
   17   deriving Typeable
   18 
   19 instance IsMetadata FileReader where
   20   initMetadata = return $ FileReader BS.readFile
   21 
   22 -- Override the result of 'scFileReader' with the given function for
   23 -- the duration of an action.
   24 withFileReader :: (MonadIO m) => SharedContext -> (FilePath -> IO ByteString) -> m a -> m a
   25 withFileReader sc fileReader = scWithData sc (\_ -> FileReader fileReader)
   26 
   27 -- | Get the file reader from the 'SharedContext'. Returns
   28 -- 'Data.ByteString.readFile' unless overridden by 'withFileReader'.
   29 scFileReader :: (MonadIO m) => SharedContext -> m (FilePath -> IO ByteString)
   30 scFileReader sc = do
   31   FileReader fileReader <- liftIO $ scGetData sc
   32   return fileReader
