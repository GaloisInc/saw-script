   12 module SAWCore.Cache
   13   ( Cache
   14   , newCache
   15   , newCacheMap
   16   , newCacheMap'
   17   , newCacheIntMap
   18   , newCacheIntMap'
   19   , useCache
   20   )
   21 where
   22 
   23 import           Control.Monad (liftM)
   24 import qualified Data.IntMap as IntMap
   25 import qualified Data.Map as Map
   26 import           Data.Ref
   27 import           Prelude hiding (lookup)
   28 
   29 data Cache m k a = forall t. Cache (T m t) (k -> t -> Maybe a) (k -> a -> t -> t)
   30 
   31 useCache :: C m => Cache m k a -> k -> m a -> m a
   32 useCache (Cache ref lookup update) k action = do
   33   result <- liftM (lookup k) (Data.Ref.read ref)
   34   case result of
   35     Just x -> return x
   36     Nothing -> do
   37       x <- action
   38       modify ref (update k x)
   39       return x
   40 
   41 newCache :: (C m, Ord k) => m (Cache m k a)
   42 newCache = newCacheMap
   43 
   44 newCacheMap :: (C m, Ord k) => m (Cache m k a)
   45 newCacheMap = newCacheMap' Map.empty
   46 
   47 newCacheMap' :: (C m, Ord k) => Map.Map k a -> m (Cache m k a)
   48 newCacheMap' initialMap = do
   49   ref <- new initialMap
   50   return (Cache ref Map.lookup Map.insert)
   51 
   52 newCacheIntMap :: (C m) => m (Cache m Int a)
   53 newCacheIntMap = newCacheIntMap' IntMap.empty
   54 
   55 newCacheIntMap' :: (C m) => IntMap.IntMap a -> m (Cache m Int a)
   56 newCacheIntMap' initialMap = do
   57   ref <- new initialMap
   58   return (Cache ref IntMap.lookup IntMap.insert)
