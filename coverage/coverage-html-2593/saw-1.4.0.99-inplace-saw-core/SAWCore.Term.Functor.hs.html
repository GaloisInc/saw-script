   60 -- Sorts -----------------------------------------------------------------------
   61 
   62 -- | The sorts, also known as universes, which can either be a predicative
   63 -- universe with level i or the impredicative universe Prop.
   64 data Sort
   65   = TypeSort Natural
   66   | PropSort
   67   deriving (Eq, Generic, TH.Lift)
   68 
   69 -- Prop is the lowest sort
   70 instance Ord Sort where
   71   PropSort <= _ = True
   72   (TypeSort _) <= PropSort = False
   73   (TypeSort i) <= (TypeSort j) = i <= j
   74 
   75 instance Hashable Sort -- automatically derived
   76 
   77 instance Show Sort where
   78   showsPrec p (TypeSort i) = showParen (p >= 10) (showString "sort " . shows i)
   79   showsPrec _ PropSort = showString "Prop"
   80 
   81 -- | Create sort @Type i@ for the given natural number @i@.
   82 mkSort :: Natural -> Sort
   83 mkSort i = TypeSort i
   84 
   85 -- | Wrapper around 'PropSort', for export
   86 propSort :: Sort
   87 propSort = PropSort
   88 
   89 -- | Returns sort of the given sort.
   90 sortOf :: Sort -> Sort
   91 sortOf (TypeSort i) = TypeSort (i + 1)
   92 sortOf PropSort = TypeSort 0
   93 
   94 -- | Get the maximum sort in a list, returning Prop for the empty list
   95 maxSort :: [Sort] -> Sort
   96 maxSort [] = propSort
   97 maxSort ss = maximum ss
   98 
   99 -- | This type represents a set of advisory flags for 'Sort's that are mostly
  100 -- ignored, but are used in the Coq export process to indicate where various
  101 -- typeclass instances are necessary in function definitions. In the concrete
  102 -- syntax "isort", "qsort", etc. is used to indicate cases where these flags
  103 -- are set. Note in particular that these flags do not affect typechecking,
  104 -- so missing or overeager "isort"/"qsort" annotations will only be detected
  105 -- via the Coq export.
  106 --
  107 -- * If 'flagInhabited' is 'True', an implicit @Inhabited@ typeclass argument
  108 --   will be added during Coq translation. In the concrete syntax, an "i" is
  109 --   prepended to the sort (e.g. "isort").
  110 -- * If 'flagQuantType' is 'True', an implicit @QuantType@ typeclass argument
  111 --   will be added during Coq translation. In the concrete syntax, an "q" is
  112 --   prepended to the sort (e.g. "qsort", "qisort").
  113 data SortFlags = SortFlags { flagInhabited :: Bool
  114                            , flagQuantType :: Bool }
  115     deriving (Eq, Ord, Generic, TH.Lift)
  116 
  117 instance Hashable SortFlags -- automatically derived
  118 
  119 instance Show SortFlags where
  120   showsPrec _ (SortFlags i q) = showString $
  121     concatMap (\(b,s) -> if b then s else "")
  122               [(q,"q"), (i,"i")]
  123 
  124 -- | The 'SortFlags' object with no flags set
  125 noFlags :: SortFlags
  126 noFlags = SortFlags False False
  127 
  128 -- | Apply a binary operation to corresponding flags of two 'SortFlags'
  129 sortFlagsLift2 :: (Bool -> Bool -> Bool) -> SortFlags -> SortFlags -> SortFlags
  130 sortFlagsLift2 f (SortFlags i1 q1) (SortFlags i2 q2) = SortFlags (f i1 i2) (f q1 q2)
  131 
  132 -- | Convert a 'SortFlags' to a list of 'Bool's, indicating which flags are set
  133 sortFlagsToList :: SortFlags -> [Bool]
  134 sortFlagsToList (SortFlags i q) = [i, q]
  135 
  136 -- | Build a 'SortFlags' from a list of 'Bool's indicating which flags are set
  137 sortFlagsFromList :: [Bool] -> SortFlags
  138 sortFlagsFromList bs = SortFlags (isSet 0) (isSet 1)
  139   where isSet i = i < length bs && bs !! i

  142 -- Flat Terms ------------------------------------------------------------------
  143 
  144 -- | The "flat terms", which are the built-in atomic constructs of SAW core.
  145 --
  146 -- NB: If you add constructors to FlatTermF, make sure you update
  147 --     zipWithFlatTermF!
  148 data FlatTermF e
  149     -- Tuples are represented as nested pairs, grouped to the right,
  150     -- terminated with unit at the end.
  151   = UnitValue
  152   | UnitType
  153   | PairValue e e
  154   | PairType e e
  155   | PairLeft e
  156   | PairRight e
  157 
  158     -- | A recursor, which is specified by a 'CompiledRecursor'
  159     -- comprising the datatype name, elimination sort, and other data
  160     -- about the recursor.
  161   | Recursor CompiledRecursor
  162 
  163     -- | Non-dependent record types, i.e., N-ary tuple types with named
  164     -- fields. These are considered equal up to reordering of fields. Actual
  165     -- tuple types are represented with field names @"1"@, @"2"@, etc.
  166   | RecordType ![(FieldName, e)]
  167     -- | Non-dependent records, i.e., N-ary tuples with named fields. These are
  168     -- considered equal up to reordering of fields. Actual tuples are
  169     -- represented with field names @"1"@, @"2"@, etc.
  170   | RecordValue ![(FieldName, e)]
  171     -- | Non-dependent record projection
  172   | RecordProj e !FieldName
  173 
  174     -- | Sorts, aka universes, are the types of types; i.e., an object is a
  175     -- "type" iff it has type @Sort s@ for some s. See 'SortFlags' for an
  176     -- explanation of the extra argument.
  177   | Sort !Sort !SortFlags
  178 
  179     -- Primitive builtin values
  180     -- | Natural number with given value.
  181   | NatLit !Natural
  182     -- | Array value includes type of elements followed by elements.
  183   | ArrayValue e (Vector e)
  184     -- | String literal
  185   | StringLit !Text
  186   deriving (Eq, Ord, Show, Functor, Foldable, Traversable, Generic)
  187 
  188 instance Hashable e => Hashable (FlatTermF e) -- automatically derived
  189 
  190 -- | A 'CompiledRecursor' comprises the datatype name and elimination
  191 -- sort of a recursor, along with some other data derived from details
  192 -- of the datatype definition.
  193 data CompiledRecursor =
  194   CompiledRecursor
  195   { recursorDataType  :: Name
  196   , recursorSort      :: Sort
  197   , recursorNumParams :: Int
  198   , recursorNumIxs    :: Int
  199   , recursorCtorOrder :: [Name]
  200   }
  201  deriving (Eq, Ord, Show, Generic)
  202 
  203 instance Hashable CompiledRecursor -- automatically derived
  204 
  205 -- | Test if the association list used in a 'RecordType' or 'RecordValue' uses
  206 -- precisely the given field names and no more. If so, return the values
  207 -- associated with those field names, in the order given in the input, and
  208 -- otherwise return 'Nothing'
  209 alistAllFields :: Eq k => [k] -> [(k, a)] -> Maybe [a]
  210 alistAllFields [] [] = Just []
  211 alistAllFields (fld:flds) alist
  212   | Just val <- lookup fld alist =
  213     (val :) <$> alistAllFields flds (deleteField fld alist)
  214   where
  215     deleteField _ [] = error "deleteField"
  216     deleteField f ((f',_):rest) | f == f' = rest
  217     deleteField f (x:rest) = x : deleteField f rest
  218 alistAllFields _ _ = Nothing
  219 
  220 zipName :: Name -> Name -> Maybe Name
  221 zipName x y
  222   | x == y = Just x
  223   | otherwise = Nothing
  224 
  225 zipRec :: CompiledRecursor -> CompiledRecursor -> Maybe CompiledRecursor
  226 zipRec (CompiledRecursor d1 s1 ps1 n1 ord1) (CompiledRecursor d2 s2 _ n2 ord2)
  227   | n1 == n2 && s1 == s2
  228   = do d <- zipName d1 d2
  229        ord <- sequence (zipWith zipName ord1 ord2)
  230        pure $ CompiledRecursor
  231               d
  232               s1
  233               ps1
  234               n1
  235               ord
  236   | otherwise = Nothing
  237 
  238 -- | Zip a binary function @f@ over a pair of 'FlatTermF's by applying @f@
  239 -- pointwise to immediate subterms, if the two 'FlatTermF's are the same
  240 -- constructor; otherwise, return 'Nothing' if they use different constructors
  241 zipWithFlatTermF :: (x -> y -> z) -> FlatTermF x -> FlatTermF y ->
  242                     Maybe (FlatTermF z)
  243 zipWithFlatTermF f = go
  244   where
  245     go UnitValue UnitValue = Just UnitValue
  246     go UnitType UnitType = Just UnitType
  247     go (PairValue x1 x2) (PairValue y1 y2) = Just (PairValue (f x1 y1) (f x2 y2))
  248     go (PairType x1 x2) (PairType y1 y2) = Just (PairType (f x1 y1) (f x2 y2))
  249     go (PairLeft x) (PairLeft y) = Just (PairLeft (f x y))
  250     go (PairRight x) (PairRight y) = Just (PairLeft (f x y))
  251 
  252     go (Recursor rec1) (Recursor rec2) =
  253       Recursor <$> zipRec rec1 rec2
  254 
  255     go (RecordType elems1) (RecordType elems2)
  256       | Just vals2 <- alistAllFields (map fst elems1) elems2 =
  257         Just $ RecordType $ zipWith (\(fld,x) y -> (fld, f x y)) elems1 vals2
  258     go (RecordValue elems1) (RecordValue elems2)
  259       | Just vals2 <- alistAllFields (map fst elems1) elems2 =
  260         Just $ RecordValue $ zipWith (\(fld,x) y -> (fld, f x y)) elems1 vals2
  261     go (RecordProj e1 fld1) (RecordProj e2 fld2)
  262       | fld1 == fld2 = Just $ RecordProj (f e1 e2) fld1
  263 
  264     go (Sort sx hx) (Sort sy hy) | sx == sy = Just (Sort sx (sortFlagsLift2 (&&) hx hy))
  265          -- /\ NB, it's not entirely clear how the flags should be propagated
  266     go (NatLit i) (NatLit j) | i == j = Just (NatLit i)
  267     go (StringLit s) (StringLit t) | s == t = Just (StringLit s)
  268     go (ArrayValue tx vx) (ArrayValue ty vy)
  269       | V.length vx == V.length vy
  270       = Just $ ArrayValue (f tx ty) (V.zipWith f vx vy)
  271 
  272     go UnitValue      _ = Nothing
  273     go UnitType       _ = Nothing
  274     go PairValue{}    _ = Nothing
  275     go PairType{}     _ = Nothing
  276     go PairLeft{}     _ = Nothing
  277     go PairRight{}    _ = Nothing
  278     go Recursor{}     _ = Nothing
  279     go RecordType{}   _ = Nothing
  280     go RecordValue{}  _ = Nothing
  281     go RecordProj{}   _ = Nothing
  282     go Sort{}         _ = Nothing
  283     go NatLit{}       _ = Nothing
  284     go ArrayValue{}   _ = Nothing
  285     go StringLit{}    _ = Nothing

  288 -- Term Functor ----------------------------------------------------------------
  289 
  290 -- | A \"knot-tying\" structure for representing terms and term-like things.
  291 -- Often, this appears in context as the type \"'TermF' 'Term'\", in which case
  292 -- it represents a full 'Term' AST. The \"F\" stands for 'Functor', or
  293 -- occasionally for \"Former\".
  294 data TermF e
  295     = FTermF !(FlatTermF e)
  296       -- ^ The atomic, or builtin, term constructs
  297     | App !e !e
  298       -- ^ Applications of functions
  299     | Lambda !VarName !e !e
  300       -- ^ Function abstractions
  301     | Pi !VarName !e !e
  302       -- ^ The type of a (possibly) dependent function
  303     | Constant !Name
  304       -- ^ A global constant identified by its name.
  305     | Variable !VarName !e
  306       -- ^ A named variable with a type.
  307   deriving (Eq, Ord, Show, Functor, Foldable, Traversable, Generic)
  308 
  309 instance Hashable e => Hashable (TermF e) -- automatically derived

  312 -- Free Named Variables --------------------------------------------------------
  313 
  314 -- | Compute an 'IntSet' containing the 'VarIndex' of the free
  315 -- variables of a term, given the free variables for its immediate
  316 -- subterms.
  317 freesTermF :: TermF IntSet -> IntSet
  318 freesTermF tf =
  319   case tf of
  320     FTermF ftf -> Foldable.foldl' IntSet.union IntSet.empty ftf
  321     App l r -> IntSet.union l r
  322     Lambda nm tp rhs -> IntSet.union tp (IntSet.delete (vnIndex nm) rhs)
  323     Pi nm lhs rhs -> IntSet.union lhs (IntSet.delete (vnIndex nm) rhs)
  324     Constant {} -> IntSet.empty
  325     Variable nm tp -> IntSet.insert (vnIndex nm) tp
