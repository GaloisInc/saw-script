   62 --------------------------------------------------------------------------------
   63 -- * Certified typed terms
   64 
   65 -- | An abstract datatype pairing a well-formed 'Raw.Term' with its type.
   66 -- A 'Term' represents a typing judgment of the form @e : t@, where
   67 -- @e@ is the raw term and @t@ is its type.
   68 data Term =
   69   Term
   70   Raw.Term -- ^ value
   71   Raw.Term -- ^ type
   72 
   73 -- | The raw term of a 'Term'.
   74 rawTerm :: Term -> Raw.Term
   75 rawTerm (Term trm _) = trm
   76 
   77 -- | The type of a 'Term' as a raw term.
   78 rawType :: Term -> Raw.Term
   79 rawType (Term _ typ) = typ
   80 
   81 --------------------------------------------------------------------------------
   82 
   83 -- | Reduce the given 'Raw.Term' to WHNF, using all reductions allowed by
   84 -- the SAWCore type system.
   85 scTypeCheckWHNF :: SharedContext -> Raw.Term -> IO Raw.Term
   86 scTypeCheckWHNF sc t =
   87   do (_, t') <- rewriteSharedTermTypeSafe sc (addConvs natConversions emptySimpset :: Simpset ()) t
   88      Raw.scWhnf sc t'
   89 
   90 -- | Check if two terms are "convertible for type-checking", meaning that they
   91 -- are convertible up to 'natConversions'.
   92 scTypeConvertible :: SharedContext -> Raw.Term -> Raw.Term -> IO Bool
   93 scTypeConvertible sc t1 t2 = Raw.scConvertibleEval sc scTypeCheckWHNF True t1 t2
   94 
   95 -- | Check whether one type is a subtype of another: Either they are
   96 -- convertible, or they are both Pi types with convertible argument
   97 -- types and result sorts @s1@ and @s2@ with @s1 <= s2@.
   98 scSubtype :: SharedContext -> Raw.Term -> Raw.Term -> IO Bool
   99 scSubtype sc t1 t2
  100   | alphaEquiv t1 t2 = pure True
  101   | otherwise =
  102     do t1' <- Raw.scWhnf sc t1
  103        t2' <- Raw.scWhnf sc t2
  104        case (t1', t2') of
  105          (asSort -> Just s1, asSort -> Just s2) ->
  106            pure (s1 <= s2)
  107          (unwrapTermF -> Pi x1 a1 b1, unwrapTermF -> Pi x2 a2 b2)
  108            | x1 == x2 ->
  109              (&&) <$> scTypeConvertible sc a1 a2 <*> scSubtype sc b1 b2
  110            | otherwise ->
  111              do conv1 <- scTypeConvertible sc a1 a2
  112                 var1 <- Raw.scVariable sc x1 a1
  113                 b2' <- Raw.scInstantiate sc (IntMap.singleton (vnIndex x2) var1) b2
  114                 conv2 <- scSubtype sc b1 b2'
  115                 pure (conv1 && conv2)
  116          _ ->
  117            scTypeConvertible sc t1' t2'

  122 -- | Compute the type of a 'Term'.
  123 scTypeOf :: SharedContext -> Term -> IO Term
  124 scTypeOf sc (Term _ tp) =
  125   do tp_tp <- Raw.scTypeOf sc tp
  126      pure (Term tp tp_tp)
  127 
  128 -- | Reduce a 'Cterm' to WHNF (see also 'scTypeCheckWHNF').
  129 scWHNF :: SharedContext -> Term -> IO Term
  130 scWHNF sc (Term tm tp) =
  131   do tm' <- scTypeCheckWHNF sc tm
  132      pure (Term tm' tp)

  134 scGlobal :: SharedContext -> Ident -> IO Term
  135 scGlobal sc ident =
  136   do tm <- Raw.scGlobalDef sc ident
  137      tp <- Raw.scTypeOfIdent sc ident
  138      pure (Term tm tp)

  141 -- * Building certified terms
  142 
  143 -- possible errors: not a pi type, bad argument type, context mismatch
  144 scApply :: SharedContext -> Term -> Term -> IO Term
  145 scApply sc f arg =
  146   do tm <- Raw.scApply sc (rawTerm f) (rawTerm arg)
  147      (vnIndex -> i, t1, t2) <- ensurePi sc (rawType f)
  148      ok <- scSubtype sc (rawType arg) t1
  149      unless ok $ fail $ unlines $
  150        ["Not a subtype", "expected: " ++ showTerm t1, "got: " ++ showTerm (rawType arg)]
  151      tp <- Raw.scInstantiate sc (IntMap.singleton i (rawTerm arg)) t2
  152      pure (Term tm tp)
  153 
  154 -- possible errors: not a type, context mismatch, variable free in context
  155 scLambda :: SharedContext -> VarName -> Term -> Term -> IO Term
  156 scLambda sc x t body =
  157   do _s <- ensureSort sc (rawType t)
  158      tm <- Raw.scLambda sc x (rawTerm t) (rawTerm body)
  159      tp <- Raw.scPi sc x (rawTerm t) (rawType body)
  160      pure (Term tm tp)
  161 
  162 -- possible errors: not a variable, context mismatch, variable free in context
  163 scAbstract :: SharedContext -> Term -> Term -> IO Term
  164 scAbstract sc var body =
  165   case asVariable (rawTerm var) of
  166     Nothing -> fail "scAbstract: Not a variable"
  167     Just (x, _) ->
  168       do tm <- Raw.scLambda sc x (rawType var) (rawTerm body)
  169          tp <- Raw.scPi sc x (rawType var) (rawType body)
  170          pure (Term tm tp)
  171 
  172 -- possible errors: not a type, context mismatch, variable free in context
  173 scPi :: SharedContext -> VarName -> Term -> Term -> IO Term
  174 scPi sc x t body =
  175   do tm <- Raw.scPi sc x (rawTerm t) (rawTerm body)
  176      s1 <- ensureSort sc (rawType t)
  177      s2 <- ensureSort sc (rawType body)
  178      tp <- Raw.scSort sc (piSort s1 s2)
  179      pure (Term tm tp)
  180 
  181 scGeneralize :: SharedContext -> Term -> Term -> IO Term
  182 scGeneralize sc var body =
  183   case asVariable (rawTerm var) of
  184     Nothing -> fail "scGeneralize: Not a variable"
  185     Just (x, _) ->
  186       do tp <- scTypeOf sc var
  187          scPi sc x tp body
  188 
  189 -- possible errors: not a type, context mismatch
  190 scFun :: SharedContext -> Term -> Term -> IO Term
  191 scFun sc a b =
  192   do tm <- Raw.scFun sc (rawTerm a) (rawTerm b)
  193      sa <- ensureSort sc (rawType a)
  194      sb <- ensureSort sc (rawType b)
  195      tp <- Raw.scSort sc (piSort sa sb)
  196      pure (Term tm tp)

  198 -- possible errors: constant not defined
  199 scConstant :: SharedContext -> Name -> IO Term
  200 scConstant sc nm =
  201   do tm <- Raw.scConst sc nm
  202      tp <- Raw.scTypeOfName sc nm

  205 -- possible errors: not a type
  206 scVariable :: SharedContext -> VarName -> Term -> IO Term
  207 scVariable sc vn t =
  208   do _s <- ensureSort sc (rawType t)
  209      let tp = rawTerm t
  210      tm <- Raw.scVariable sc vn tp
  211      pure (Term tm tp)

  227 -- possible errors: none (could redesign to require types in sort 0)
  228 scPairValue :: SharedContext -> Term -> Term -> IO Term
  229 scPairValue sc x y =
  230   do tm <- Raw.scPairValue sc (rawTerm x) (rawTerm y)
  231      tp <- Raw.scPairType sc (rawType x) (rawType y)
  232      pure (Term tm tp)

  234 -- possible errors: not a type
  235 scPairType :: SharedContext -> Term -> Term -> IO Term
  236 scPairType sc x y =
  237   do tm <- Raw.scPairType sc (rawTerm x) (rawTerm y)
  238      sx <- ensureSort sc (rawType x)
  239      sy <- ensureSort sc (rawType y)
  240      tp <- Raw.scSort sc (max sx sy)
  241      pure (Term tm tp)
  242 
  243 -- possible errors: not a pair
  244 scPairLeft :: SharedContext -> Term -> IO Term
  245 scPairLeft sc x =
  246   do tm <- Raw.scPairLeft sc (rawTerm x)
  247      tp <- fst <$> ensurePairType sc (rawType x)
  248      pure (Term tm tp)
  249 
  250 -- possible errors: not a pair
  251 scPairRight :: SharedContext -> Term -> IO Term
  252 scPairRight sc x =
  253   do tm <- Raw.scPairRight sc (rawTerm x)
  254      tp <- snd <$> ensurePairType sc (rawType x)
  255      pure (Term tm tp)

  257 -- possible errors: not a datatype, bad elimination sort
  258 scRecursor :: SharedContext -> Name -> Sort -> IO Term
  259 scRecursor sc nm s =
  260   do mm <- Raw.scGetModuleMap sc
  261      case lookupVarIndexInMap (nameIndex nm) mm of
  262        Just (ResolvedDataType dt) ->
  263          do unless (Raw.allowedElimSort dt s) $ fail "Disallowed propositional elimination"
  264             let d = dtName dt
  265             let nparams = length (dtParams dt)
  266             let nixs = length (dtIndices dt)
  267             let ctorOrder = map ctorName (dtCtors dt)
  268             let crec = CompiledRecursor d s nparams nixs ctorOrder
  269             tm <- Raw.scFlatTermF sc (Recursor crec)
  270             tp <- Raw.scRecursorType sc dt s
  271             pure (Term tm tp)
  272        _ ->
  273          fail "datatype not found"
  274 
  275 -- possible errors: field not a type, context mismatch
  276 scRecordType :: SharedContext -> [(FieldName, Term)] -> IO Term
  277 scRecordType sc fields =
  278   do tm <- Raw.scRecordType sc (map (fmap rawTerm) fields)
  279      sorts <- traverse (ensureSort sc . rawType . snd) fields
  280      tp <- Raw.scSort sc (foldl max (mkSort 0) sorts)
  281      pure (Term tm tp)
  282 
  283 -- possible errors: duplicate field name
  284 scRecordValue :: SharedContext -> [(FieldName, Term)] -> IO Term
  285 scRecordValue sc fields =
  286   do tm <- Raw.scFlatTermF sc $ RecordValue (map (fmap rawTerm) fields)
  287      tp <- Raw.scRecordType sc (map (fmap rawType) fields)
  288      pure (Term tm tp)
  289 
  290 -- possible errors: not a record type, field name not found
  291 scRecordProj :: SharedContext -> Term -> FieldName -> IO Term
  292 scRecordProj sc t fname =
  293   do tm <- Raw.scRecordSelect sc (rawTerm t) fname
  294      tps <- ensureRecordType sc (rawType t)
  295      case Map.lookup fname tps of
  296        Nothing -> fail "scRecordProj: field name not found"
  297        Just tp -> pure (Term tm tp)

  299 -- no possible errors
  300 scSort :: SharedContext -> Sort -> IO Term
  301 scSort sc s = scSortWithFlags sc s noFlags
  302 
  303 -- | A variant of 'scSort' that also takes a 'SortFlags' argument.
  304 -- No possible errors.
  305 scSortWithFlags :: SharedContext -> Sort -> SortFlags -> IO Term
  306 scSortWithFlags sc s flags =
  307   do tm <- Raw.scFlatTermF sc (Sort s flags)
  308      tp <- Raw.scSort sc (sortOf s)
  309      pure (Term tm tp)

  318 -- possible errors: context mismatch, element type not a type, element wrong type

  321   do -- TODO: check that all xs have type e

  337 ensureRecognizer :: String -> SharedContext -> (Raw.Term -> Maybe a) -> Raw.Term -> IO a
  338 ensureRecognizer s sc f trm =
  339   case f trm of
  340     Just a -> pure a
  341     Nothing ->
  342       do trm' <- scTypeCheckWHNF sc trm
  343          case f trm' of
  344            Just a -> pure a
  345            Nothing ->
  346              fail $ "ensureRecognizer: Expected " ++ s ++ ", found: " ++ showTerm trm'
  347 
  348 ensureSort :: SharedContext -> Raw.Term -> IO Sort
  349 ensureSort sc tp = ensureRecognizer "Sort" sc asSort tp
  350 
  351 ensurePi :: SharedContext -> Raw.Term -> IO (VarName, Raw.Term, Raw.Term)
  352 ensurePi sc tp = ensureRecognizer "Pi" sc asPi tp
  353 
  354 ensurePairType :: SharedContext -> Raw.Term -> IO (Raw.Term, Raw.Term)
  355 ensurePairType sc tp = ensureRecognizer "PairType" sc asPairType tp
  356 
  357 ensureRecordType :: SharedContext -> Raw.Term -> IO (Map FieldName Raw.Term)
  358 ensureRecordType sc tp = ensureRecognizer "RecordType" sc asRecordType tp

  360 piSort :: Sort -> Sort -> Sort
  361 piSort s1 s2 = if s2 == propSort then propSort else max s1 s2
