   44 -- | Convert a type expression from SAW-core to What4
   45 fotToBaseType :: FirstOrderType -> Maybe (Some BaseTypeRepr)
   46 fotToBaseType FOTBit
   47   = Just (Some BaseBoolRepr)
   48 fotToBaseType FOTInt
   49   = Just (Some BaseIntegerRepr)
   50 fotToBaseType (FOTIntMod _n)
   51   = Just (Some BaseIntegerRepr)
   52 fotToBaseType (FOTVec nat FOTBit)
   53   | Just (Some (PosNat nr)) <- somePosNat nat
   54   = Just (Some (BaseBVRepr nr))
   55   | otherwise = Nothing
   56 
   57 fotToBaseType (FOTVec nat fot)
   58   | Just (Some assn) <- listToAssn (replicate (fromIntegral nat) fot)
   59   = Just (Some (BaseStructRepr assn))
   60 fotToBaseType (FOTArray fot1 fot2)
   61   | Just (Some ty1) <- fotToBaseType fot1
   62   , Just (Some ty2) <- fotToBaseType fot2
   63   = Just (Some (BaseArrayRepr (Empty :> ty1) ty2))
   64 fotToBaseType (FOTTuple fots)
   65   | Just (Some assn) <- listToAssn fots
   66   = Just (Some (BaseStructRepr assn))
   67 
   68 -- TODO: convert to What4 records ?!
   69 fotToBaseType _ = Nothing
   70 
   71 listToAssn :: [FirstOrderType] -> Maybe (Some (Assignment BaseTypeRepr))
   72 listToAssn [] = Just (Some empty)
   73 listToAssn (fot:rest) =
   74   case (fotToBaseType fot, listToAssn rest) of
   75     (Just (Some bt), Just (Some assn)) -> Just (Some (extend assn bt))
   76     _ -> Nothing

   79 -- | Convert a type expression from What4 to SAW-core
   80 
   81 typeReprToFOT :: BaseTypeRepr ty -> Either String FirstOrderType
   82 typeReprToFOT BaseBoolRepr            = pure FOTBit
   83 typeReprToFOT BaseIntegerRepr         = pure FOTInt
   84 typeReprToFOT (BaseBVRepr w)          = pure $ FOTVec (natValue w) FOTBit
   85 typeReprToFOT BaseRealRepr            = Left "No FO Real"
   86 typeReprToFOT BaseComplexRepr         = Left "No FO Complex"
   87 typeReprToFOT (BaseStringRepr _)      = Left "No FO String"
   88 typeReprToFOT (BaseArrayRepr (Empty :> ty) b)
   89   | Right fot1 <- typeReprToFOT ty
   90   , Right fot2 <- typeReprToFOT b
   91   = pure $ FOTArray fot1 fot2
   92 typeReprToFOT ty@(BaseArrayRepr _ctx _b) = Left $ "Unsupported FO Array: " ++ show ty
   93 typeReprToFOT (BaseFloatRepr _)       = Left "No FO Floating point"
   94 typeReprToFOT (BaseStructRepr ctx)    = FOTTuple <$> assnToList ctx
   95 
   96 assnToList :: Assignment BaseTypeRepr ctx -> Either String [FirstOrderType]
   97 assnToList = foldrFC g (Right []) where
   98   g :: BaseTypeRepr x -> Either String [FirstOrderType] -> Either String [FirstOrderType]
   99   g ty l = (:) <$> typeReprToFOT ty <*> l

  103 -- | Convert between What4 GroundValues and saw-core FirstOrderValues
  104 
  105 groundToFOV :: BaseTypeRepr ty -> GroundValue ty -> Either String FirstOrderValue
  106 groundToFOV BaseBoolRepr    b         = pure $ FOVBit b
  107 groundToFOV BaseIntegerRepr i         = pure $ FOVInt i
  108 groundToFOV (BaseBVRepr w) bv         = pure $ FOVWord (natValue w) (BV.asUnsigned bv)
  109 groundToFOV BaseRealRepr    _         = Left "Real is not FOV"
  110 groundToFOV BaseComplexRepr         _ = Left "Complex is not FOV"
  111 groundToFOV (BaseStringRepr _)      _ = Left "String is not FOV"
  112 groundToFOV (BaseFloatRepr _)       _ = Left "Floating point is not FOV"
  113 groundToFOV (BaseArrayRepr (Empty :> ty_idx) ty_val) (ArrayMapping _) = do
  114     -- ArrayMapping is an array represented as a function call we can
  115     -- use to extract values. We can't do anything useful with this
  116     -- because we don't have any clue what key values to look at and
  117     -- no good way to figure that out either. So return a placeholder
  118     -- value in FirstOrderValue.
  119     --
  120     -- (The frustrating part is that in many cases the _user_ will
  121     -- know what key values to look at, but has no way to tell us.)
  122     --
  123     -- In principle we could change groundToFOV to return any of an
  124     -- error, a FirstOrderValue, or a separate FunctionArrayValue
  125     -- type, and use that channel to return the actual function
  126     -- handle. (Including the function handle in a FirstOrderValue
  127     -- value is problematic as things stand.) However, there's nothing
  128     -- to be gained by doing this without a way to figure out what
  129     -- key values to look at. And realistically, if we ever _do_
  130     -- come up with a way to figure out what key values to look at,
  131     -- it should be implemented in What4 so What4 never returns
  132     -- ArrayMapping.
  133     --
  134     -- (See Note [FOVArray] in SAWCore.FiniteValue
  135     -- where FirstOrderValue is defined.)
  136     ty_idx' <- typeReprToFOT ty_idx
  137     ty_val' <- typeReprToFOT ty_val
  138     pure $ FOVOpaqueArray ty_idx' ty_val'
  139 groundToFOV (BaseArrayRepr (Empty :> ty_idx) ty_val) (ArrayConcrete dfl values) = do
  140     ty_idx' <- typeReprToFOT ty_idx
  141     dfl' <- groundToFOV ty_val dfl
  142     let convert idx values' v = do
  143           let idx' = indexToFOV idx
  144           v' <- groundToFOV ty_val v
  145           return $ Map.insert idx' v' values'
  146     values' <- ifoldlM convert Map.empty values
  147     pure $ FOVArray ty_idx' dfl' values'
  148 groundToFOV (BaseArrayRepr _ _) _ =
  149     Left "Unsupported FOV array (unexpected key type)"
  150 groundToFOV (BaseStructRepr ctx) tup  = FOVTuple <$> tupleToList ctx tup
  151 
  152 indexToFOV :: Assignment IndexLit (EmptyCtx ::> ty) -> FirstOrderValue
  153 indexToFOV (Empty :> IntIndexLit k) =
  154     FOVInt k
  155 indexToFOV (Empty :> BVIndexLit w k) =
  156     FOVWord (natValue w) (BV.asUnsigned k)

  159 tupleToList :: Assignment BaseTypeRepr ctx ->
  160              Assignment GroundValueWrapper ctx -> Either String [FirstOrderValue]
  161 tupleToList (viewAssign -> AssignEmpty) (viewAssign -> AssignEmpty) = Right []
  162 tupleToList (viewAssign -> AssignExtend rs r) (viewAssign -> AssignExtend gvs gv) =
  163   (:) <$> groundToFOV r (unGVW gv) <*> tupleToList rs gvs
  164 #if !MIN_VERSION_GLASGOW_HASKELL(8,10,0,0)
  165 tupleToList _ _ = error "GADTs should rule this out."
  166 #endif
