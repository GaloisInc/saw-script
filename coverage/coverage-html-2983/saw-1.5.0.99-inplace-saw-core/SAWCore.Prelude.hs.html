   34 -- | Given two terms, compute a term representing a decidable
   35 --   equality test between them.  The terms are assumed to
   36 --   be of the same type, which must be a first-order type.
   37 --   The returned term will be of type @Bool@.
   38 scEq :: SharedContext -> Term -> Term -> IO Term
   39 scEq sc x y = do
   40   xty <- scTypeOf sc x
   41   mmap <- scGetModuleMap sc
   42   case asFirstOrderTypeValue (evalSharedTerm mmap mempty mempty xty) of
   43     Just fot -> scDecEq sc fot (Just (x,y))
   44     Nothing  -> do
   45         xty' <- ppTerm sc PPS.defaultOpts xty
   46         fail $ "scEq: expected first order type, but got: " ++ xty'
   47 
   48 -- | Given a first-order type, return the decidable equality
   49 --   operation on that type.  If arguments are provided, they
   50 --   will be applied, returning a value of type @Bool@.  If no
   51 --   arguments are provided a function of type @tp -> tp -> Bool@
   52 --   will be returned.
   53 scDecEq ::
   54   SharedContext ->
   55   FirstOrderType    {- ^ Type of elements to test for equality -} ->
   56   Maybe (Term,Term) {- ^ optional arguments to apply -} ->
   57   IO Term
   58 scDecEq sc fot args = case fot of
   59   FOTBit ->
   60     do fn <- scGlobalDef sc "Prelude.boolEq"
   61        case args of
   62          Nothing    -> return fn
   63          Just (x,y) -> scApplyAll sc fn [x,y]
   64 
   65   FOTInt ->
   66     do fn <- scGlobalDef sc "Prelude.intEq"
   67        case args of
   68          Nothing    -> return fn
   69          Just (x,y) -> scApplyAll sc fn [x,y]
   70 
   71   FOTIntMod m ->
   72     do fn <- scGlobalDef sc "Prelude.intModEq"
   73        m' <- scNat sc m
   74        case args of
   75          Nothing    -> scApply sc fn m'
   76          Just (x,y) -> scApplyAll sc fn [m',x,y]
   77 
   78   FOTVec w FOTBit ->
   79     do fn <- scGlobalDef sc "Prelude.bvEq"
   80        w' <- scNat sc w
   81        case args of
   82          Nothing    -> scApply sc fn w'
   83          Just (x,y) -> scApplyAll sc fn [w',x,y]
   84 
   85   FOTVec w t ->
   86     do fn <- scGlobalDef sc "Prelude.vecEq"
   87        w' <- scNat sc w
   88        t' <- scFirstOrderType sc t
   89        subFn <- scDecEq sc t Nothing
   90        case args of
   91          Nothing    -> scApplyAll sc fn [w',t',subFn]
   92          Just (x,y) -> scApplyAll sc fn [w',t',subFn,x,y]
   93 
   94   FOTArray a b ->
   95     do a' <- scFirstOrderType sc a
   96        b' <- scFirstOrderType sc b
   97        fn <- scGlobalDef sc "Prelude.arrayEq"
   98        case args of
   99          Nothing    -> scApplyAll sc fn [a',b']
  100          Just (x,y) -> scApplyAll sc fn [a',b',x,y]
  101 
  102   FOTTuple []  ->
  103     case args of
  104       Nothing -> scGlobalDef sc "Prelude.unitEq"
  105       Just _  -> scBool sc True
  106 
  107   FOTTuple [t] -> scDecEq sc t args
  108 
  109   FOTTuple (t:ts) ->
  110     do fnLeft  <- scDecEq sc t Nothing
  111        fnRight <- scDecEq sc (FOTTuple ts) Nothing
  112        fn      <- scGlobalDef sc "Prelude.pairEq"
  113        t'      <- scFirstOrderType sc t
  114        ts'     <- scFirstOrderType sc (FOTTuple ts)
  115        case args of
  116          Nothing    -> scApplyAll sc fn [t',ts',fnLeft,fnRight]
  117          Just (x,y) -> scApplyAll sc fn [t',ts',fnLeft,fnRight,x,y]
  118 
  119   FOTRec fs ->
  120     case args of
  121       Just (x,y) ->
  122            mkRecordEqBody (Map.toList fs) x y
  123 
  124       Nothing ->
  125         do tp <- scFirstOrderType sc fot
  126            x <- scFreshVariable sc "x" tp
  127            y <- scFreshVariable sc "y" tp
  128            body <- mkRecordEqBody (Map.toList fs) x y
  129            scAbstractTerms sc [x, y] body
  130 
  131  where
  132   mkRecordEqBody [] _x _y = scBool sc True
  133   mkRecordEqBody [(f,tp)] x y =
  134      do xf <- scRecordSelect sc x f
  135         yf <- scRecordSelect sc y f
  136         scDecEq sc tp (Just (xf,yf))
  137   mkRecordEqBody ((f,tp):fs) x y =
  138      do xf   <- scRecordSelect sc x f
  139         yf   <- scRecordSelect sc y f
  140         fp  <- scDecEq sc tp (Just (xf,yf))
  141         fsp <- mkRecordEqBody fs x y
  142         scAnd sc fp fsp
