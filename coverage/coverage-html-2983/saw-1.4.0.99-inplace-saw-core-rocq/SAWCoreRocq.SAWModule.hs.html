   43   M.TranslationConfiguration ->

   51 dropPi :: Rocq.Term -> Rocq.Term
   52 dropPi (Rocq.Pi (_ : t) r) = Rocq.Pi t r
   53 dropPi (Rocq.Pi _       r) = dropPi r
   54 dropPi t                   = t

   56 translateCtor ::
   57   ModuleTranslationMonad m =>
   58   [Rocq.Binder] -> -- list of parameters to drop from `ctorType`
   59   Ctor -> m Rocq.Constructor
   60 translateCtor inductiveParameters (Ctor {..}) = do
   61   maybe_constructorName <-
   62     case nameInfo ctorName of
   63       ModuleIdentifier ident -> liftTermTranslationMonad $ TermTranslation.translateIdentToIdent ident
   64       ImportedName{} -> pure Nothing
   65   let constructorName = case maybe_constructorName of
   66         -- Drop qualifiers from constructor name
   67         Just (Rocq.Ident n) -> Rocq.Ident (reverse (takeWhile (/= '.') (reverse n)))
   68         Nothing -> error "translateCtor: unexpected translation for constructor"
   69   constructorType <-
   70     -- Unfortunately, `ctorType` qualifies the inductive type's name in the
   71     -- return type.
   72     -- dropModuleNameWithinCtor <$>
   73     -- Unfortunately, `ctorType` comes with the inductive parameters universally
   74     -- quantified.
   75     (\ t -> iterate dropPi t !! length inductiveParameters) <$>
   76     (liftTermTranslationMonad (TermTranslation.translateTerm ctorType))
   77   return $ Rocq.Constructor
   78     { constructorName
   79     , constructorType
   80     }

   82 translateDataType :: ModuleTranslationMonad m => DataType -> m Rocq.Decl
   83 -- translateDataType (DataType {..})
   84 --   | trace ("translateDataType: " ++ show dtName) False = undefined
   85 translateDataType (DataType {..}) =
   86   case nameInfo dtName of
   87     ModuleIdentifier dtIdent ->
   88       atDefSite <$> findSpecialTreatment dtIdent >>= \case
   89       DefPreserve            -> translateNamed $ Rocq.Ident (identName dtIdent)
   90       DefRename   targetName -> translateNamed $ targetName
   91       DefReplace  str        -> return $ Rocq.Snippet str
   92       DefSkip                -> return $ skipped dtIdent
   93     ImportedName{} ->
   94       translateNamed $ Rocq.Ident (Text.unpack (toShortName (nameInfo dtName)))
   95   where
   96     translateNamed :: ModuleTranslationMonad m => Rocq.Ident -> m Rocq.Decl
   97     translateNamed name = do
   98       let inductiveName = name
   99       (inductiveParameters, inductiveIndices) <-
  100         liftTermTranslationMonad $
  101         TermTranslation.translateParams dtParams $ \ps ->
  102         TermTranslation.translateParams dtIndices $ \ixs ->
  103         -- Translating the indices of a data type should never yield Inhabited
  104         -- constraints, so the result of calling `translateParams dtIndices` above should
  105         -- only return explicit, not implicit, Binders.  Moreover, `translateParams`
  106         -- always returns Binders where the second field is `Just t`, where `t` is the type.
  107         let errorBecause msg = error $ "translateDataType.translateNamed: " ++ msg in
  108         let bs = map (\case Rocq.Binder Rocq.Explicit s (Just t) ->
  109                               Rocq.PiBinder Rocq.Explicit (Just s) t
  110                             Rocq.Binder Rocq.Explicit _ Nothing ->
  111                               errorBecause "encountered a Binder without a Type"
  112                             Rocq.Binder Rocq.Implicit _ _ ->
  113                               errorBecause "encountered an implicit binder")
  114                      ixs in
  115         return (ps, bs)
  116       let inductiveSort = TermTranslation.translateSort dtSort
  117       inductiveConstructors <- mapM (translateCtor inductiveParameters) dtCtors
  118       return $ Rocq.InductiveDecl $ Rocq.Inductive
  119         { inductiveName
  120         , inductiveParameters
  121         , inductiveIndices
  122         , inductiveSort
  123         , inductiveConstructors
  124         }

  143 translateDef :: ModuleTranslationMonad m => Def -> m Rocq.Decl
  144 translateDef (Def {..}) = {- trace ("translateDef " ++ show defIdent) $ -} do
  145   specialTreatment <- findSpecialTreatment' (nameInfo defName)
  146   translateAccordingly (atDefSite specialTreatment)
  147 
  148   where
  149 
  150     translateAccordingly :: ModuleTranslationMonad m => DefSiteTreatment -> m Rocq.Decl
  151     translateAccordingly  DefPreserve           = translateNamed $ Rocq.Ident (Text.unpack (toShortName (nameInfo defName)))
  152     translateAccordingly (DefRename targetName) = translateNamed $ targetName
  153     translateAccordingly (DefReplace  str)      = return $ Rocq.Snippet str
  154     translateAccordingly  DefSkip               = return $ skipped' (nameInfo defName)
  155 
  156     translateNamed :: ModuleTranslationMonad m => Rocq.Ident -> m Rocq.Decl
  157     translateNamed name = liftTermTranslationMonad (go defQualifier defBody)
  158 
  159       where
  160 
  161         go :: TermTranslation.TermTranslationMonad m => DefQualifier -> Maybe Term -> m Rocq.Decl
  162         go NoQualifier    Nothing     = error "Terms should have a body (unless axiom/primitive)"
  163         go NoQualifier    (Just body) = Rocq.Definition
  164                                         <$> pure name
  165                                         <*> pure []
  166                                         <*> (Just <$> TermTranslation.translateTerm defType)
  167                                         <*> TermTranslation.translateTerm body
  168         go AxiomQualifier _           = Rocq.Axiom
  169                                         <$> pure name
  170                                         <*> TermTranslation.translateTerm defType
  171         go PrimQualifier  _           = Rocq.Axiom
  172                                         <$> pure name
  173                                         <*> TermTranslation.translateTerm defType
  174 
  175 liftTermTranslationMonad ::
  176   (forall n. TermTranslation.TermTranslationMonad   n => n a) ->
  177   (forall m. ModuleTranslationMonad m => m a)
  178 liftTermTranslationMonad n = do
  179   configuration <- asks translationConfiguration
  180   (modname, mm) <- asks otherConfiguration
  181   let r = TermTranslation.runTermTranslationMonad configuration modname mm [] [] n
  182   case r of
  183     Left  e      -> Except.throwError e
  184     Right (a, _) -> return a

  186 translateDecl ::
  187   M.TranslationConfiguration ->
  188   Maybe ModuleName ->
  189   ModuleMap ->
  190   ModuleDecl ->
  191   Doc ann
  192 translateDecl configuration modname mm decl =
  193   case decl of
  194     TypeDecl td -> do
  195       case runModuleTranslationMonad configuration modname mm (translateDataType td) of
  196         Left e           -> error $ show e
  197         Right (tdecl, _) -> Rocq.prettyDecl tdecl
  198     DefDecl dd -> do
  199       case runModuleTranslationMonad configuration modname mm (translateDef dd) of
  200         Left e           -> error $ show e
  201         Right (tdecl, _) -> Rocq.prettyDecl tdecl
  202     InjectCodeDecl ns txt
  203       | ns == "Rocq" -> pretty txt
  204       | otherwise    -> mempty
