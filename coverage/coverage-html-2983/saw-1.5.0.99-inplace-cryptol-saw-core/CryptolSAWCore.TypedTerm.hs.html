   57 -- Typed terms -----------------------------------------------------------------
   58 
   59 -- | Within SAWScript, we represent an object language term as a
   60 -- SAWCore shared term paired with a Cryptol type schema. The Cryptol
   61 -- type is used for type inference/checking of inline Cryptol
   62 -- expressions.
   63 
   64 data TypedTerm =
   65   TypedTerm
   66   { ttType :: TypedTermType
   67   , ttTerm :: Term
   68   }
   69   deriving Show
   70 
   71 
   72 -- | The different notion of Cryptol types that
   73 --   a SAWCore term might have.
   74 data TypedTermType
   75   = TypedTermSchema C.Schema
   76   | TypedTermKind   C.Kind
   77   | TypedTermOther  Term
   78  deriving Show
   79 
   80 
   81 -- It is not ideal to have two copies of these but it's not that
   82 -- helpful to try to share. In the long run the pure ones should
   83 -- probably go away as they produce worse output.
   84 
   85 prettyTypedTerm :: SharedContext -> PPS.Opts -> TypedTerm -> IO (PP.Doc ann)
   86 prettyTypedTerm sc opts (TypedTerm tp tm) = do
   87   tm' <- prettyTerm sc opts tm
   88   tp' <- prettyTypedTermType sc opts tp
   89   pure $ PP.unAnnotate tm' PP.<+> ":" PP.<+> tp'
   90 
   91 prettyTypedTermType :: SharedContext -> PPS.Opts -> TypedTermType -> IO (PP.Doc ann)
   92 prettyTypedTermType _sc _opts (TypedTermSchema sch) =
   93   pure $ PP.viaShow (C.ppPrec 0 sch)
   94 prettyTypedTermType _sc _opts (TypedTermKind k) =
   95   pure $ PP.viaShow (C.ppPrec 0 k)
   96 prettyTypedTermType sc opts (TypedTermOther tp) = do
   97   tp' <- prettyTerm sc opts tp
   98   pure $ PP.unAnnotate tp'
   99 
  100 prettyTypedTermPure :: TypedTerm -> PP.Doc ann
  101 prettyTypedTermPure (TypedTerm tp tm) =
  102   PP.unAnnotate (prettyTermPure PPS.defaultOpts tm)
  103   PP.<+> ":" PP.<+>
  104   prettyTypedTermTypePure tp
  105 
  106 prettyTypedTermTypePure :: TypedTermType -> PP.Doc ann
  107 prettyTypedTermTypePure (TypedTermSchema sch) =
  108   PP.viaShow (C.ppPrec 0 sch)
  109 prettyTypedTermTypePure (TypedTermKind k) =
  110   PP.viaShow (C.ppPrec 0 k)
  111 prettyTypedTermTypePure (TypedTermOther tp) =
  112   PP.unAnnotate (prettyTermPure PPS.defaultOpts tp)

  114 prettyTypedVariable :: TypedVariable -> PP.Doc ann
  115 prettyTypedVariable (TypedVariable ctp vn _tp) =
  116   PP.unAnnotate (PP.pretty (vnName vn))
  117   PP.<+> ":" PP.<+>
  118   PP.viaShow (C.ppPrec 0 ctp)

  121 -- | Convert the 'ttType' field of a 'TypedTerm' to a SAW core term
  122 ttTypeAsTerm :: SharedContext -> Env -> TypedTerm -> IO Term
  123 ttTypeAsTerm sc env (TypedTerm (TypedTermSchema schema) _) =
  124   importSchema sc env schema
  125 ttTypeAsTerm sc _ (TypedTerm (TypedTermKind k) _) = importKind sc k
  126 ttTypeAsTerm _ _ (TypedTerm (TypedTermOther tp) _) = return tp
  127 
  128 ttTermLens :: Functor f => (Term -> f Term) -> TypedTerm -> f TypedTerm
  129 ttTermLens f tt = tt `seq` fmap (\x -> tt{ttTerm = x}) (f (ttTerm tt))
  130 
  131 ttIsMono :: TypedTermType -> Maybe C.Type
  132 ttIsMono ttp =
  133   case ttp of
  134     TypedTermSchema sch -> C.isMono sch
  135     _ -> Nothing
  136 
  137 mkTypedTerm :: SharedContext -> Term -> IO TypedTerm
  138 mkTypedTerm sc trm = do
  139   ty <- scTypeOf sc trm
  140   ct <- scCryptolType sc ty
  141   let ttt = case ct of
  142         Nothing        -> TypedTermOther ty
  143         Just (Left k)  -> TypedTermKind k
  144         Just (Right t) -> TypedTermSchema (C.tMono t)
  145   return (TypedTerm ttt trm)
  146 
  147 -- | Apply a 'TypedTerm' to a list of arguments. This operation fails
  148 -- if the first 'TypedTerm' does not have a function type of
  149 -- sufficient arity, or if the types of the arguments do not match
  150 -- the type of the function.
  151 applyTypedTerms :: SharedContext -> TypedTerm -> [TypedTerm] -> IO TypedTerm
  152 applyTypedTerms sc (TypedTerm _ fn) args =
  153   do trm <- foldM (scApply sc) fn (map ttTerm args)
  154      ty <- scTypeOf sc trm
  155      -- NB, scCryptolType can behave in strange ways due to the non-injectivity
  156      -- of the mapping from Cryptol to SAWCore types. Perhaps we would be better
  157      -- to combine the incoming type information instead of applying and then
  158      -- reconstructing here.
  159      ct <- scCryptolType sc ty
  160      let ttt = case ct of
  161            Nothing        -> TypedTermOther ty
  162            Just (Left k)  -> TypedTermKind k
  163            Just (Right t) -> TypedTermSchema (C.tMono t)
  164      return (TypedTerm ttt trm)

  166 -- First order types and values ------------------------------------------------
  167 
  168 cryptolTypeOfFirstOrderType :: FirstOrderType -> C.Type
  169 cryptolTypeOfFirstOrderType fot =
  170   case fot of
  171     FOTBit -> C.tBit
  172     FOTInt -> C.tInteger
  173     FOTIntMod n -> C.tIntMod (C.tNum n)
  174     FOTVec n t -> C.tSeq (C.tNum n) (cryptolTypeOfFirstOrderType t)
  175     -- NB, special case, don't produce 1-tuples
  176     FOTTuple [x] -> cryptolTypeOfFirstOrderType x
  177     FOTTuple ts -> C.tTuple (map cryptolTypeOfFirstOrderType ts)
  178     FOTArray a b ->
  179       C.tArray
  180       (cryptolTypeOfFirstOrderType a)
  181       (cryptolTypeOfFirstOrderType b)
  182     FOTRec m ->
  183       C.tRec $
  184       C.recordFromFields $
  185       [ (C.mkIdent l, cryptolTypeOfFirstOrderType t)
  186       | (l, t) <- Map.assocs m ]
  187 
  188 typedTermOfFirstOrderValue :: SharedContext -> FirstOrderValue -> IO TypedTerm
  189 typedTermOfFirstOrderValue sc fov =
  190   do let fot = firstOrderTypeOf fov
  191      let cty = cryptolTypeOfFirstOrderType fot
  192      t <- scFirstOrderValue sc fov
  193      pure $ TypedTerm (TypedTermSchema (C.tMono cty)) t

  195 -- Typed variables -------------------------------------------------------------

  197 data TypedVariable =
  198   TypedVariable
  199   { tvCType :: C.Type
  200   , tvName :: VarName
  201   , tvType :: Term
  202   }
  203   deriving Show
  204 
  205 -- | Recognize 'TypedTerm's that are variables.
  206 asTypedVariable :: TypedTerm -> Maybe TypedVariable
  207 asTypedVariable (TypedTerm tp t) =
  208   do cty <- ttIsMono tp
  209      (x, ty) <- asVariable t
  210      pure $ TypedVariable cty x ty
  211 
  212 -- | Make a 'TypedTerm' from a 'TypedVariable'.
  213 typedTermOfVariable :: SharedContext -> TypedVariable -> IO TypedTerm
  214 typedTermOfVariable sc (TypedVariable cty vn ty) =
  215   TypedTerm (TypedTermSchema (C.tMono cty)) <$> scVariable sc vn ty
  216 
  217 abstractTypedVars :: SharedContext -> [TypedVariable] -> TypedTerm -> IO TypedTerm
  218 abstractTypedVars sc tvars (TypedTerm (TypedTermSchema (C.Forall params props ty)) trm) =
  219   do let tys = map tvCType tvars
  220      let vars = map (\(TypedVariable _ vn tp) -> (vn, tp)) tvars
  221      let ty' = foldr C.tFun ty tys
  222      trm' <- scLambdaList sc vars trm
  223      pure $ TypedTerm (TypedTermSchema (C.Forall params props ty')) trm'
  224 abstractTypedVars sc tvars (TypedTerm (TypedTermKind k) trm) =
  225   do let vars = map (\(TypedVariable _ vn tp) -> (vn, tp)) tvars
  226      trm' <- scLambdaList sc vars trm
  227      pure $ TypedTerm (TypedTermKind k) trm'
  228 abstractTypedVars sc tvars (TypedTerm (TypedTermOther _tp) trm) =
  229   do let vars = map (\(TypedVariable _ vn tp) -> (vn, tp)) tvars
  230      trm' <- scLambdaList sc vars trm
  231      tp'  <- scTypeOf sc trm'
  232      pure $ TypedTerm (TypedTermOther tp') trm'

  234 -- Typed modules ---------------------------------------------------------------
  235 
  236 -- | In SAWScript, we can refer to a Cryptol module as a first class
  237 -- value. These are represented simply as maps from names to typed
  238 -- terms.
  239 
  240 data CryptolModule =
  241   CryptolModule
  242     (Map C.Name C.TySyn)    -- type synonyms
  243     (Map C.Name TypedTerm)  -- symbols (mapping to SawCore things).
  244 
  245 -- Note: Cryptol's prettyprinter isn't directly compatible with SAW's.
  246 --
  247 -- Also note that its naming conventions are opposite ours; it uses
  248 -- "pp" to make Docs and "pretty" to make Strings.
  249 --
  250 prettyCryptolModule :: CryptolModule -> PP.Doc ann
  251 prettyCryptolModule (CryptolModule sm tm) =
  252   let cpp item = PP.viaShow $ C.pp item
  253       prettyTSyn (C.TySyn name params _props rhs _doc) =
  254         let name' = cpp (nameIdent name)
  255             params' = map cpp params
  256             rhs' = cpp rhs
  257         in
  258         PP.indent 4 $ PP.hsep (name' : params') PP.<+> "=" PP.<+> rhs'
  259 
  260       prettyBinding (name, TypedTerm (TypedTermSchema schema) _) =
  261         [PP.indent 4 $ cpp (nameIdent name) PP.<+> ":" PP.<+> cpp schema]
  262       prettyBinding _ =
  263         []
  264       synonyms =
  265         if Map.null sm then []
  266         else
  267             "Type Synonyms" : "=============" : map prettyTSyn (Map.elems sm) ++ [""]
  268       symbols =
  269         "Symbols" : "=======" : concatMap prettyBinding (Map.assocs tm)
  270   in
  271   PP.vsep $ synonyms ++ symbols
