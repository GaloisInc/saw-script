   69 data JVMVal
   70   = RVal (Crucible.RegValue Sym CJ.JVMRefType)
   71   | IVal (Crucible.RegValue Sym CJ.JVMIntType)
   72   | LVal (Crucible.RegValue Sym CJ.JVMLongType)

   74 instance Show JVMVal where
   75   show (RVal _) = "RVal"
   76   show (IVal _) = "IVal"
   77   show (LVal _) = "LVal"

   81 data JVMTypeOfError
   82   = JVMPolymorphicType Cryptol.Schema
   83   | JVMNonRepresentableType Cryptol.Type
   84   | JVMInvalidTypedTerm TypedTermType
   85 
   86 instance Show JVMTypeOfError where
   87   show (JVMPolymorphicType s) =
   88     unlines
   89     [ "Expected monomorphic term"
   90     , "instead got:"
   91     , show (Cryptol.pp s)
   92     ]
   93   show (JVMNonRepresentableType ty) =
   94     unlines
   95     [ "Type not representable in JVM:"
   96     , show (Cryptol.pp ty)
   97     ]
   98   show (JVMInvalidTypedTerm tp) =
   99     unlines
  100     [ "Expected typed term with Cryptol representable type, but got"
  101     , show (prettyTypedTermTypePure tp)
  102     ]
  103 
  104 instance X.Exception JVMTypeOfError

  106 typeOfSetupValue ::
  107   X.MonadThrow m =>
  108   JVMCrucibleContext ->
  109   Map AllocIndex (MS.ConditionMetadata, Allocation) ->
  110   Map AllocIndex JIdent ->
  111   SetupValue ->
  112   m J.Type
  113 typeOfSetupValue _cc env _nameEnv val =
  114   case val of
  115     MS.SetupVar i ->
  116       case Map.lookup i env of
  117         Nothing ->
  118             panic "JVMSetup (in typeOfSetupValue)" [
  119                 "Unresolved prestate variable: " <> Text.pack (show i)
  120             ]
  121         Just (_, alloc) -> return (allocationType alloc)
  122     MS.SetupTerm tt ->
  123       case ttType tt of
  124         TypedTermSchema (Cryptol.Forall [] [] ty) ->
  125           case toJVMType (Cryptol.evalValType mempty ty) of
  126             Nothing -> X.throwM (JVMNonRepresentableType ty)
  127             Just jty -> return jty
  128         TypedTermSchema s -> X.throwM (JVMPolymorphicType s)
  129         tp -> X.throwM (JVMInvalidTypedTerm tp)
  130 
  131     MS.SetupNull () ->
  132       -- We arbitrarily set the type of NULL to java.lang.Object,
  133       -- because a) it is memory-compatible with any type that NULL
  134       -- can be used at, and b) it prevents us from doing any
  135       -- type-safe field accesses.
  136       return (J.ClassType (J.mkClassName "java/lang/Object"))
  137     MS.SetupGlobal empty _            -> absurd empty
  138     MS.SetupStruct empty _            -> absurd empty
  139     MS.SetupEnum empty                -> absurd empty
  140     MS.SetupTuple empty _             -> absurd empty
  141     MS.SetupSlice empty               -> absurd empty
  142     MS.SetupArray empty _             -> absurd empty
  143     MS.SetupElem empty _ _            -> absurd empty
  144     MS.SetupField empty _ _           -> absurd empty
  145     MS.SetupCast empty _              -> absurd empty
  146     MS.SetupUnion empty _ _           -> absurd empty
  147     MS.SetupGlobalInitializer empty _ -> absurd empty
  148     MS.SetupMux empty _ _ _           -> absurd empty
  149 
  150 lookupAllocIndex :: Map AllocIndex a -> AllocIndex -> a
  151 lookupAllocIndex env i =
  152   case Map.lookup i env of
  153     Just x -> x
  154     Nothing ->
  155       panic "JVMSetup (in lookupAllocIndex" [
  156           "Unresolved prestate variable: " <> Text.pack (show i)
  157       ]

  159 -- | Translate a SetupValue into a Crucible JVM value, resolving
  160 -- references
  161 resolveSetupVal ::
  162   JVMCrucibleContext ->
  163   Map AllocIndex JVMRefVal ->
  164   Map AllocIndex (MS.ConditionMetadata, Allocation) ->
  165   Map AllocIndex JIdent ->
  166   SetupValue ->
  167   IO JVMVal
  168 resolveSetupVal cc env _tyenv _nameEnv val =
  169   case val of
  170     MS.SetupVar i ->
  171       pure (RVal (lookupAllocIndex env i))
  172     MS.SetupTerm tm -> resolveTypedTerm cc tm
  173     MS.SetupNull () ->
  174       return (RVal (W4.maybePartExpr sym Nothing))
  175     MS.SetupGlobal empty _            -> absurd empty
  176     MS.SetupStruct empty _            -> absurd empty
  177     MS.SetupEnum empty                -> absurd empty
  178     MS.SetupTuple empty _             -> absurd empty
  179     MS.SetupSlice empty               -> absurd empty
  180     MS.SetupArray empty _             -> absurd empty
  181     MS.SetupElem empty _ _            -> absurd empty
  182     MS.SetupField empty _ _           -> absurd empty
  183     MS.SetupCast empty _              -> absurd empty
  184     MS.SetupUnion empty _ _           -> absurd empty
  185     MS.SetupGlobalInitializer empty _ -> absurd empty
  186     MS.SetupMux empty _ _ _           -> absurd empty
  187   where
  188     sym = cc^.jccSym
  189 
  190 resolveTypedTerm ::
  191   JVMCrucibleContext ->
  192   TypedTerm       ->
  193   IO JVMVal
  194 resolveTypedTerm cc tm =
  195   case ttType tm of
  196     TypedTermSchema (Cryptol.Forall [] [] ty) ->
  197       resolveSAWTerm cc (Cryptol.evalValType mempty ty) (ttTerm tm)
  198     tp -> fail $ unlines
  199             [ "resolveSetupVal: expected monomorphic term"
  200             , "but got a term of type"
  201             , show (prettyTypedTermTypePure tp)
  202             ]
  203 
  204 resolveSAWPred ::
  205   JVMCrucibleContext ->
  206   Term ->
  207   IO (W4.Pred Sym)
  208 resolveSAWPred cc = resolveBoolTerm (cc ^. jccSym) (cc ^. jccUninterp)
  209 
  210 resolveSAWTerm ::
  211   JVMCrucibleContext ->
  212   Cryptol.TValue ->
  213   Term ->
  214   IO JVMVal
  215 resolveSAWTerm cc tp tm =
  216   case tp of
  217     Cryptol.TVBit ->
  218       do b <- resolveBoolTerm sym (cc ^. jccUninterp) tm
  219          x0 <- W4.bvLit sym W4.knownNat (BV.zero W4.knownNat)
  220          x1 <- W4.bvLit sym W4.knownNat (BV.one  W4.knownNat)
  221          x <- W4.bvIte sym b x1 x0
  222          return (IVal x)
  223     Cryptol.TVInteger ->
  224       fail "resolveSAWTerm: unimplemented type Integer (FIXME)"
  225     Cryptol.TVIntMod _ ->
  226       fail "resolveSAWTerm: unimplemented type Z n (FIXME)"
  227     Cryptol.TVFloat{} ->
  228       fail "resolveSAWTerm: unimplemented type Float e p (FIXME)"
  229     Cryptol.TVArray{} ->
  230       fail "resolveSAWTerm: unimplemented type Array a b (FIXME)"
  231     Cryptol.TVRational ->
  232       fail "resolveSAWTerm: unimplemented type Rational (FIXME)"
  233     Cryptol.TVSeq sz Cryptol.TVBit ->
  234       let unint = cc ^. jccUninterp in
  235       case sz of
  236         8  -> do x <- resolveBitvectorTerm sym unint (W4.knownNat @8) tm
  237                  IVal <$> W4.bvSext sym W4.knownNat x
  238         16 -> do x <- resolveBitvectorTerm sym unint (W4.knownNat @16) tm
  239                  IVal <$> W4.bvSext sym W4.knownNat x
  240         32 -> IVal <$> resolveBitvectorTerm sym unint W4.knownNat tm
  241         64 -> LVal <$> resolveBitvectorTerm sym unint W4.knownNat tm
  242         _  -> fail ("Invalid bitvector width: " ++ show sz)
  243     Cryptol.TVSeq _sz _tp' ->
  244       fail "resolveSAWTerm: unimplemented sequence type"
  245     Cryptol.TVStream _tp' ->
  246       fail "resolveSAWTerm: unsupported infinite stream type"
  247     Cryptol.TVTuple _tps ->
  248       fail "resolveSAWTerm: unsupported tuple type"
  249     Cryptol.TVRec _flds ->
  250       fail "resolveSAWTerm: unsupported record type"
  251     Cryptol.TVFun _ _ ->
  252       fail "resolveSAWTerm: unsupported function type"
  253     Cryptol.TVNominal {} ->
  254       fail "resolveSAWTerm: unsupported nominal type"
  255   where
  256     sym = cc^.jccSym
  257 
  258 toJVMType :: Cryptol.TValue -> Maybe J.Type
  259 toJVMType tp =
  260   case tp of
  261     Cryptol.TVBit -> Just J.BooleanType
  262     Cryptol.TVInteger -> Nothing
  263     Cryptol.TVIntMod _ -> Nothing
  264     Cryptol.TVFloat{} -> Nothing
  265     Cryptol.TVArray{} -> Nothing
  266     Cryptol.TVRational -> Nothing
  267     Cryptol.TVSeq n Cryptol.TVBit ->
  268       case n of
  269         8  -> Just J.ByteType
  270         16 -> Just J.ShortType
  271         32 -> Just J.IntType
  272         64 -> Just J.LongType
  273         _  -> Nothing
  274     Cryptol.TVSeq _n t ->
  275       do t' <- toJVMType t
  276          Just (J.ArrayType t')
  277     Cryptol.TVStream _tp' -> Nothing
  278     Cryptol.TVTuple _tps -> Nothing
  279     Cryptol.TVRec _flds -> Nothing
  280     Cryptol.TVFun _ _ -> Nothing
  281     Cryptol.TVNominal {} -> Nothing

  283 equalValsPred ::
  284   JVMCrucibleContext ->
  285   JVMVal ->
  286   JVMVal ->
  287   IO (W4.Pred Sym)
  288 equalValsPred cc v1 v2 = go (v1, v2)
  289   where
  290   go :: (JVMVal, JVMVal) -> IO (W4.Pred Sym)
  291   go (RVal r1, RVal r2) = CJ.refIsEqual sym r1 r2
  292   go (IVal i1, IVal i2) = W4.bvEq sym i1 i2
  293   go (LVal l1, LVal l2) = W4.bvEq sym l1 l2
  294   go _ = return (W4.falsePred sym)
  295 
  296   sym = cc^.jccSym
