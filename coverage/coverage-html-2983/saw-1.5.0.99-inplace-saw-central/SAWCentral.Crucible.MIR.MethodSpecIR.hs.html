  125 mutIso :: Iso' M.Mutability Bool
  126 mutIso =
  127   iso
  128     (\case
  129       M.Mut -> True
  130       M.Immut -> False)
  131     (\case
  132       True -> M.Mut
  133       False -> M.Immut)

  138 -- | Converts a 'MirPointerKind' to a 'M.Ty' constructor.
  139 ptrKindToTy :: MirPointerKind -> M.Ty -> M.Mutability -> M.Ty
  140 ptrKindToTy MirPointerRef = M.TyRef
  141 ptrKindToTy MirPointerRaw = M.TyRawPtr
  142 
  143 -- | Gets the 'MirPointerKind' for a reference or pointer type. Panics if the
  144 -- given 'M.Ty' is not a reference or pointer.
  145 tyToPtrKind :: M.Ty -> MirPointerKind
  146 tyToPtrKind (M.TyRef _ _) = MirPointerRef
  147 tyToPtrKind (M.TyRawPtr _ _) = MirPointerRaw
  148 tyToPtrKind ty = panic "tyToPtrKind"
  149   ["Type is not a reference or pointer: " <> Text.pack (show ty)]

  153 initialDefCrucibleMethodSpecIR ::
  154   CollectionState ->
  155   M.Fn ->
  156   ProgramLoc ->
  157   MS.CrucibleMethodSpecIR MIR
  158 initialDefCrucibleMethodSpecIR rm fn loc =
  159   let fname = fn ^. M.fname
  160       fsig = fn ^. M.fsig
  161       argTys = fsig ^. M.fsarg_tys
  162       retTy = case fsig ^. M.fsreturn_ty of
  163                 M.TyTuple [] -> Nothing
  164                 ty -> Just ty in
  165   MS.makeCrucibleMethodSpecIR fname argTys retTy loc rm
  166 
  167 initialCrucibleSetupState ::
  168   MIRCrucibleContext ->
  169   M.Fn ->
  170   ProgramLoc ->
  171   Setup.CrucibleSetupState MIR
  172 initialCrucibleSetupState cc fn loc =
  173   Setup.makeCrucibleSetupState () cc $
  174     initialDefCrucibleMethodSpecIR
  175       (cc ^. mccRustModule ^. rmCS)
  176       fn
  177       loc
