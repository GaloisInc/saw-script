   46 -- | Generate a preamble for a Coq file, containing a list of Coq imports. This
   47 -- includes standard imports, one of which is the @VectorNotations@ module to
   48 -- support the vector literals used to translate SAW core array values, along
   49 -- with any user-supplied imports in the 'postPreamble' field of the
   50 -- supplied 'TranslationConfiguration'.
   51 preamble :: TranslationConfiguration -> Doc ann
   52 preamble (TranslationConfiguration { vectorModule, postPreamble }) = text [i|
   53 (** Mandatory imports from saw-core-coq *)
   54 From Coq Require Import Lists.List.
   55 From Coq Require Import String.
   56 From Coq Require Import Vectors.Vector.
   57 From CryptolToCoq Require Import SAWCoreScaffolding.
   58 From CryptolToCoq Require Import #{vectorModule}.
   59 Import VectorNotations.
   60 
   61 (** Post-preamble section specified by you *)
   62 #{postPreamble}
   63 
   64 (** Code generated by saw-core-coq *)
   65 |]

   67 translateTermAsDeclImports ::
   68   TranslationConfiguration -> ModuleMap -> Coq.Ident -> Term -> Term ->
   69   Either (TranslationError Term) (Doc ann)
   70 translateTermAsDeclImports configuration mm name t tp = do
   71   doc <-
   72     TermTranslation.translateDefDoc
   73       configuration
   74       Nothing
   75       mm
   76       [] name t tp
   77   return $ vcat [preamble configuration, hardline <> doc]

   79 -- | Translate a SAW core module to a Coq module
   80 translateSAWModule :: TranslationConfiguration -> ModuleMap -> Module -> Doc ann
   81 translateSAWModule configuration mm m =
   82   let name = show $ translateModuleName (moduleName m)
   83   in
   84   vcat $ []
   85   ++ [ text $ "Module " ++ name ++ "."
   86      , ""
   87      ]
   88   ++ [ SAWModuleTranslation.translateDecl configuration (Just $ moduleName m) mm decl
   89      | decl <- moduleDecls m ]
   90   ++ [ text $ "End " ++ name ++ "."
   91      , ""
   92      ]
   93 
   94 -- | Translate a Cryptol module to a Coq module
   95 translateCryptolModule ::
   96   SharedContext -> Env ->
   97   Coq.Ident {- ^ Section name -} ->
   98   TranslationConfiguration ->
   99   -- | List of already translated global declarations
  100   [Coq.Ident] ->
  101   CryptolModule ->
  102   IO (Either (TranslationError Term) (Doc ann))
  103 translateCryptolModule sc env nm configuration globalDecls m = do
  104   translated <- CMT.translateCryptolModule sc env configuration globalDecls m
  105   return $ Coq.ppDecl . Coq.Section (escapeIdent nm) <$> translated

  107 -- | Extract out the 'String' name of a declaration in a SAW core module
  108 moduleDeclName :: ModuleDecl -> Maybe String
  109 moduleDeclName (TypeDecl (DataType { dtNameInfo })) = Just (Text.unpack (toShortName dtNameInfo))
  110 moduleDeclName (DefDecl  (Def      { defNameInfo })) = Just (Text.unpack (toShortName defNameInfo))
  111 moduleDeclName InjectCodeDecl{} = Nothing
