  144 -- | Look up a name in a map containing Cryptol names.
  145 --
  146 -- The string corresponds to the Cryptol name we are looking for.
  147 --
  148 -- If it is unqualifed, then we return any entry associated with the given
  149 -- name.  If the string is qualified (i.e., has @::@), then we only consider
  150 -- entries from the module named by the qualification.
  151 --
  152 -- The result is either the corresponding value, or a list of the candidate
  153 -- fully-qualified names.
  154 --
  155 lookupIn :: Text -> Map T.Name b -> Either [T.Name] b
  156 lookupIn nm mp =
  157   case [ x | x <- Map.toList mp, matches (fst x) ] of
  158     [ (_,v) ] -> Right v
  159     opts      -> Left (map fst opts)
  160   where
  161   matches = nameMatcher nm
  162 
  163 
  164 -- | Parse a string into a function that will match Cryptol names.
  165 --
  166 -- If the string is unqualified (i.e., no `::`), then we match all
  167 -- names with the given identifier.  Otherwise, we only match the
  168 -- ones in the module specified by the qualification.
  169 --
  170 -- The first argument is the name we're looking for; the second
  171 -- argument (or the first argument of the returned function, in the
  172 -- intended usage) is a candidate name to match.
  173 nameMatcher :: Text -> T.Name -> Bool
  174 nameMatcher nm0 =
  175     case C.modNameChunksText (textToModName nm0) of
  176       []  -> const False
  177       [x] -> (x ==) . C.identText . MN.nameIdent
  178       cs  -> \n ->
  179                 case MN.nameInfo n of
  180                   MN.LocalName {} -> False
  181                   MN.GlobalName _ og ->
  182                     let (top,ns) = C.modPathSplit (C.ogModule og)
  183                     in last cs == identText (C.ogName og) &&
  184                        init cs == C.modNameChunksText top ++ map identText ns

  189 -- | initCryptolEnv - Create initial CryptolEnv. This involves loading
  190 --   the built-in modules (preludeName, arrayName,
  191 --   preludeReferenceName) and translating them into SAWCore, and
  192 --   putting them into scope.
  193 --
  194 --   NOTE: submodules in these built-in modules are supported in this code.
  195 --
  196 initCryptolEnv ::
  197   (?fileReader :: FilePath -> IO ByteString) =>
  198   SharedContext -> IO CryptolEnv
  199 initCryptolEnv sc = do
  200   modEnv0 <- M.initialModuleEnv
  201 
  202   -- Set the Cryptol include path (TODO: we may want to do this differently)
  203   (binDir, _) <- splitExecutablePath
  204   let instDir = normalise . joinPath . init . splitPath $ binDir
  205   mCryptolPath <- lookupEnv "CRYPTOLPATH"
  206   let cryptolPaths =
  207         case mCryptolPath of
  208           Nothing -> []
  209           Just path ->
  210 #if defined(mingw32_HOST_OS) || defined(__MINGW32__)
  211             -- Windows paths search from end to beginning
  212             reverse (splitSearchPath path)
  213 #else
  214             splitSearchPath path
  215 #endif
  216   let modEnv1 = modEnv0 { ME.meSearchPath = cryptolPaths ++
  217                            (instDir </> "lib") : ME.meSearchPath modEnv0 }
  218 
  219   -- Load Cryptol prelude and magic Array module
  220   (_, modEnv2) <-
  221     liftModuleM modEnv1 $
  222       do _ <- MB.loadModuleFrom False (MM.FromModule preludeName)
  223          _ <- MB.loadModuleFrom False (MM.FromModule arrayName)
  224          return ()
  225 
  226   -- Load Cryptol reference implementation
  227   ((_,refTop), modEnv3) <-
  228     liftModuleM modEnv2 $
  229       MB.loadModuleFrom False (MM.FromModule preludeReferenceName)
  230   let refMod = T.tcTopEntityToModule refTop
  231 
  232   -- Set up reference implementation redirections
  233   let refDecls = T.mDecls refMod
  234   let nms = Set.toList (MI.ifsPublic (TIface.genIfaceNames refMod))
  235 
  236   let refPrims = Map.fromList
  237                   [ (prelPrim (identText (MN.nameIdent nm)), T.EWhere (T.EVar nm) refDecls)
  238                   | nm <- nms ]
  239 
  240   -- The module names in P.Import are now Located, so give them an empty position.
  241   let preludeName'          = locatedUnknown preludeName
  242       preludeReferenceName' = locatedUnknown preludeReferenceName
  243       arrayName'            = locatedUnknown arrayName
  244 
  245   let env0 = CryptolEnv
  246         { eImports    =
  247             [ mkImport OnlyPublic preludeName'          Nothing Nothing
  248             , mkImport OnlyPublic preludeReferenceName' (Just preludeReferenceName) Nothing
  249             , mkImport OnlyPublic arrayName'            Nothing Nothing
  250             ]
  251         , eModuleEnv  = modEnv3
  252         , eExtraNaming = mempty
  253         , eExtraVars  = Map.empty
  254         , eExtraTySyns = Map.empty
  255         , eAllVars    = Map.empty
  256         , eTyVars     = Map.empty
  257         , eTyProps    = Map.empty
  258         , eAllTerms   = Map.empty
  259         , eRefPrims   = refPrims
  260         , ePrims      = Map.empty
  261         , ePrimTypes  = Map.empty
  262         , eFFITypes   = Map.empty
  263         }
  264 
  265   -- Generate SAWCore translations for all values in scope
  266   env1 <- genTermEnv sc modEnv3 env0
  267 
  268   -- Clear `eRefPrims`. This preserves the behavior from before
  269   -- `CryptolEnv` and the old additional `Env` type were merged. It
  270   -- isn't clear if this is correct or not, but I don't want the code
  271   -- cleanup to change the behavior.
  272   return env1 {
  273       eRefPrims = Map.empty
  274   }

  277 -- | Translate all declarations in all loaded modules to SAWCore terms.
  278 --   NOTE: used only for initialization code.
  279 --
  280 genTermEnv :: SharedContext -> ME.ModuleEnv -> C.CryptolEnv -> IO C.CryptolEnv
  281 genTermEnv sc modEnv env0 = do
  282   let declGroups = concatMap T.mDecls
  283                  $ filter (not . T.isParametrizedModule)
  284                  $ ME.loadedModules modEnv
  285       nominals   = loadedNonParamNominalTypes modEnv
  286   -- These update eAllTerms and eAllVars and leave the rest alone
  287   env1 <- C.genCodeForNominalTypes sc nominals env0
  288   env2 <- C.importTopLevelDeclGroups sc C.defaultPrimitiveOptions env1 declGroups
  289   return env2

  328 -- | Get the full 'MR.NamingEnv' based on all the imports (from
  329 --   `eImports`), plus all the local "extra" decls too. Note that the
  330 --   imports are combined with `mconcat`, which uses the `Semigroup`
  331 --   instance for `MR.NamingEnv` to ambiguate any names that appear
  332 --   more than once, but the "extra" decls are (specifically) bolted
  333 --   on with `MR.shadowing` so they hide any previous occurrences.
  334 --
  335 --   Note that while `eImports` is (mostly) maintained with more
  336 --   recent imports at the front of the list, this should be
  337 --   irrelevant to name resolution.
  338 --
  339 getNamingEnv :: CryptolEnv -> MR.NamingEnv
  340 getNamingEnv env =
  341   eExtraNaming env
  342   `MR.shadowing`
  343   (mconcat $ map (getNamingEnvForImport (eModuleEnv env))
  344                  (eImports env)
  345   )
  346 
  347 -- | Get the `MR.NamingEnv` for one `T.Import`.
  348 getNamingEnvForImport :: ME.ModuleEnv
  349                       -> (ImportVisibility, T.Import)
  350                       -> MR.NamingEnv
  351 getNamingEnvForImport modEnv (vis, imprt) =
  352     MN.interpImportEnv'
  353       MN.nameToPNameWithQualifiers (T.iAs imprt) (T.iSpec imprt)
  354          -- adjusting for qualified imports
  355   $ MN.namingEnvNames
  356   $ computeNamingEnv lm vis
  357 
  358   where
  359   modName :: C.ModName
  360   modName = P.thing $ T.iModule imprt
  361 
  362   lm = case ME.lookupModule modName modEnv of
  363          Just lm' -> lm'
  364          Nothing  -> panic "getNamingEnvForImport"
  365                        ["cannot lookupModule: " <> CryPP.pp modName]
  366 
  367 -- | Compute a `MR.NamingEnv` for a module based on the
  368 --   `ImportVisibility`.
  369 computeNamingEnv :: ME.LoadedModule -> ImportVisibility -> MR.NamingEnv
  370 computeNamingEnv lm vis =
  371   case vis of
  372     PublicAndPrivate -> envPublicAndPrivate  -- all names defined, pub & pri
  373     OnlyPublic       -> envPublic            -- i.e., what's exported.
  374 
  375   where
  376   -- NamingEnvs: --
  377 
  378     -- | envTopLevels
  379     --    - Does not include privates in submodules (which makes for
  380     --      much of the complications of this function).
  381     --    - Includes everything in scope at the toplevel of 'lm' module
  382 
  383     envTopLevels :: MR.NamingEnv
  384     envTopLevels = ME.lmNamingEnv lm
  385 
  386     -- | envPublicAndPrivate - awkward as envTopLevels excludes privates
  387     envPublicAndPrivate :: MR.NamingEnv
  388     envPublicAndPrivate =
  389        -- nab all the names defined in module (from toplevel scope):
  390        MN.filterUNames (`Set.member` nmsDefined) envTopLevels
  391        <>
  392        -- we must create a new NamingEnv (since the privates are not
  393        -- in `envTopLevels`):
  394        MN.namingEnvFromNames' MN.nameToPNameWithQualifiers nmsPrivate
  395 
  396     envPublic :: MR.NamingEnv
  397     envPublic = MN.filterUNames
  398                   (`Set.member` nmsPublic)
  399                   envTopLevels
  400     
  401   -- Name Sets: --
  402 
  403     -- | names in scope at Top level of module
  404     nmsTopLevels :: Set MN.Name
  405     nmsTopLevels = MN.namingEnvNames envTopLevels
  406 
  407     -- | names defined in module and in submodules
  408     --   - this includes `PublicAndPrivate` names!
  409     --   - includes submodule names, type synonyms, and nominal types
  410     nmsDefined :: Set MN.Name
  411     nmsDefined =
  412         -- definitions from all submodules:
  413         ( Set.unions
  414         $ map (MI.ifsDefines . T.smIface)
  415         $ Map.elems
  416         $ T.mSubmodules
  417         $ ME.lmModule lm
  418         )
  419       `Set.union`
  420         -- definitions at the top module:
  421         (MI.ifsDefines $ MI.ifNames $ ME.lmInterface lm)
  422 
  423 
  424     nmsPublic :: Set MN.Name
  425     nmsPublic = MI.ifsPublic $ MI.ifNames $ ME.lmInterface lm
  426 
  427     nmsPrivate :: Set MN.Name
  428     nmsPrivate = nmsDefined Set.\\ nmsTopLevels

  460 -- | Restore a `CryptolEnv` from a checkpoint. The first argument
  461 --   @chkEnv@ is the `CryptolEnv` saved by / copied into the
  462 --   checkpoint; the second argument @newEnv@ is the current one
  463 --   we wish to overwrite by rolling back to the checkpoint.
  464 --
  465 --   We need to keep the newer name supply so as to not reuse names
  466 --   already issued, in case some of those are still floating around
  467 --   after the restore. (They should not... but bugs happen.)
  468 --
  469 --   We also ought to invalidate terms constructed since the checkpoint
  470 --   was taken, like SAWCore does. See #2859.
  471 --
  472 restoreCryptolEnv :: CryptolEnv -> CryptolEnv -> CryptolEnv
  473 restoreCryptolEnv chkEnv newEnv =
  474     let newMEnv = eModuleEnv newEnv
  475         chkMEnv = eModuleEnv chkEnv
  476         menv' = chkMEnv { ME.meNameSeeds = ME.meNameSeeds newMEnv }
  477     in
  478     chkEnv { eModuleEnv = menv' }

  604 -- as a `CryptolModule` structure.

  606 -- This is used in two places: in `loadCryptolModule`, which is a
  607 -- legacy code path that would ideally be removed, and in
  608 -- `loadExtCryptolModule` as part of generating the print output.
  609 -- Ideally all of this could be consolidated.
  610 --
  611 mkCryptolModule :: T.Module -> CryptolEnv -> CryptolModule
  612 mkCryptolModule m env =
  613   let
  614       ifaceDecls = C.getAllIfaceDecls (eModuleEnv env)
  615       types = Map.map MI.ifDeclSig (MI.ifDecls ifaceDecls)
  616       -- we're keeping only the exports of `m`:
  617       vNameSet = MEx.exported C.NSValue (T.mExports m)
  618       tNameSet = MEx.exported C.NSType  (T.mExports m)
  619   in
  620       CryptolModule
  621         -- create Map of type synonyms:
  622         (Map.filterWithKey
  623            (\k _ -> Set.member k tNameSet)
  624            (T.mTySyns m)
  625         )
  626 
  627         -- create Map of the `TypedTerm` s:
  628         ( Map.filterWithKey (\k _ -> Set.member k vNameSet)
  629         $ Map.intersectionWith
  630              (\t x -> TypedTerm (TypedTermSchema t) x)
  631              types
  632              (eAllTerms env)
  633         )

  635 -- | bindExtCryptolModule - add extra bindings to the Cryptol
  636 --     environment {{-}}; this happens when an `ExtCryptolModule` is
  637 --     bound in the SAWScript code.  (This may be referred to as a
  638 --     "magic bind").
  639 --
  640 --     The code that does this in the SAWScript interpreter is in
  641 --     @extendEnv@, which currently lives in Value.hs.
  642 --
  643 --   NOTE RE CALLS TO THIS: Three SAWScript variants get us here:
  644 --
  645 --      > D <- cryptol_load "PATH"
  646 --
  647 --    which results in a call to `bindLoadedModule`.
  648 --    And each of these
  649 --
  650 --      > x <- return (cryptol_prims ())
  651 --      > let x = cryptol_prims ()
  652 --
  653 --    will result in calling `bindCryptolModule`.

  655 -- NOTE:
  656 --  - The `ExtCryptolModule` datatype and these functions are a bit
  657 --    ad hoc.
  658 --  - Ideally thse would go away with further improvements to the
  659 --    external interface and corresponding implementation changes.
  660 --    - See #2569.
  661 --    - Re `bindCryptolModule` below
  662 --      - It is more general than what is needed.
  663 --      - It is somewhat duplicating functionality that we already have with
  664 --        `importCryptolModule`, this could go away in the future.
  665 --
  666 --  - See also the discusion of the SAWScript-level @cryptol_load@ in
  667 --    @CHANGES.md@.

  672   case ecm of
  673     ECM_CryptolModule cm   -> bindCryptolModule (modName, cm)
  674     ECM_LoadedModule  nm _ -> bindLoadedModule  (modName, nm)

  676 -- | bindLoadedModule - when we have a @cryptol_load@ created object,
  677 -- add the module into the import list.
  678 bindLoadedModule ::
  679   (P.ModName, P.Located C.ModName) -> CryptolEnv -> CryptolEnv
  680 bindLoadedModule (asName, origName) env =
  681   env {eImports = mkImport PublicAndPrivate origName (Just asName) Nothing

  685 -- | Undo `bindLoadedModule`. Not a general removal function. Not
  686 --  exported, and used exactly once below where we want to add a
  687 --  module to the import list temporarily.

  695 -- | bindCryptolModule - when we have the @cryptol_prims ()@ created
  696 --   object, add the `CryptolModule` to the relevant maps in the
  697 --   `CryptolEnv` See `bindExtCryptolModule` above.
  698 --
  699 --   Note that this stuffs the module contents in as "extra"
  700 --   declarations, which can be subtly different from importing it,
  701 --   e.g. in the presence of overlapping names. This is only used for
  702 --   @cryptol_prims@, and when we eventually manage to figure out how
  703 --   to handle that stuff better / more like a real module (#2645), it
  704 --   can and should be removed.
  705 --
  706 bindCryptolModule :: (P.ModName, CryptolModule) -> CryptolEnv -> CryptolEnv
  707 bindCryptolModule (modName, CryptolModule sm tm) env =
  708   env { eExtraNaming = flip (foldr addName) (Map.keys tm') $
  709                       flip (foldr addTSyn) (Map.keys sm) $
  710                       eExtraNaming env
  711       , eExtraTySyns = Map.union sm (eExtraTySyns env)
  712       , eExtraVars  = Map.union (fmap fst tm') (eExtraVars env)
  713       , eAllTerms   = Map.union (fmap snd tm') (eAllTerms env)
  714       }
  715   where
  716     -- | `tm'` is the typed terms from `tm` that have Cryptol schemas
  717     tm' = Map.mapMaybe f tm
  718           where
  719           f (TypedTerm (TypedTermSchema s) x) = Just (s,x)
  720           f _                                 = Nothing
  721 
  722     addName name =
  723       MN.shadowing
  724        (MN.singletonNS C.NSValue (P.mkQual modName (MN.nameIdent name)) name)
  725 
  726     addTSyn name =
  727       MN.shadowing
  728         (MN.singletonNS C.NSType (P.mkQual modName (MN.nameIdent name)) name)

  784 -- | Load a Cryptol module and translate its contents to SAWCore.
  785 --
  786 -- There are three paths here:
  787 --    - `importCryptolModule`, which is the back end for SAWScript @import@
  788 --    - `loadExtCryptolModule`, which is the back end for SAWScript @cryptol_load@
  789 --    - `loadCryptolModule`, which is used for Rocq export and from crux-mir-comp
  790 --
  791 -- These can probably be unified.
  792 --
  793 loadAndTranslateModule ::
  794   (?fileReader :: FilePath -> IO ByteString) =>
  795   SharedContext             {- ^ Shared context for creating terms -} ->
  796   CryptolEnv                {- ^ Extend this environment -} ->
  797   Either FilePath P.ModName {- ^ Where to find the module -} ->
  798   IO (T.Module, CryptolEnv)
  799 loadAndTranslateModule sc env0 src =
  800   do let modEnv = eModuleEnv env0
  801      (mtop, modEnv') <- liftModuleM modEnv $
  802        case src of
  803          Left path -> MB.loadModuleByPath True path
  804          Right mn  -> snd <$> MB.loadModuleFrom True (MM.FromModule mn)
  805      m <- case mtop of
  806             T.TCTopModule mod'  -> pure mod'
  807             T.TCTopSignature {} ->
  808               fail $
  809                 "Expected a module, but "
  810                 ++ (case src of
  811                       Left  path -> show path
  812                       Right mn   -> Text.unpack (CryPP.pp mn)
  813                    )
  814                 ++ " is an interface."
  815 
  816      checkNotParameterized m
  817      let env1 = env0 { eModuleEnv = modEnv' }
  818 
  819      -- Regenerate SharedTerm environment:
  820      let oldModNames   = map ME.lmName
  821                        $ ME.lmLoadedModules
  822                        $ ME.meLoadedModules modEnv
  823          isNew m'      = T.mName m' `notElem` oldModNames
  824          newModules    = filter isNew
  825                        $ map ME.lmModule
  826                        $ ME.lmLoadedModules
  827                        $ ME.meLoadedModules modEnv'
  828          newDeclGroups = concatMap T.mDecls newModules
  829          newNominal    = Map.difference (loadedNonParamNominalTypes modEnv')
  830                                         (loadedNonParamNominalTypes modEnv)
  831 
  832      env2 <- C.refreshCryptolEnv env1
  833 
  834      -- These update eAllTerms and eAllVars and leave the rest alone
  835      env3 <- C.genCodeForNominalTypes sc newNominal env2
  836      env4 <- C.importTopLevelDeclGroups
  837                         sc C.defaultPrimitiveOptions env3 newDeclGroups
  838 
  839      ffiTypes' <- updateFFITypes sc m (eAllTerms env4) (eFFITypes env4)
  840      let env5 = env4 { eFFITypes  = ffiTypes' }
  841 
  842      return (m, env5)
  843 
  844 -- | Reject unapplied functors.
  845 checkNotParameterized :: T.Module -> IO ()
  846 checkNotParameterized m =
  847   when (T.isParametrizedModule m) $
  848     fail $ unlines [ "Cannot load parameterized modules directly."
  849                    , "Either use a ` import, or make a module instantiation."
  850                    ]

  852 -- | Helper for updating `eFFITypes` in `CryptolEnv`.
  853 updateFFITypes ::
  854   SharedContext ->
  855   T.Module ->
  856   Map MN.Name Term ->
  857   Map NameInfo T.FFI ->
  858   IO (Map NameInfo T.FFI)
  859 updateFFITypes sc m allTerms' eFFITypes' = do
  860   let getNameInfo (nm, ty) = do
  861         let tm = case Map.lookup nm allTerms' of
  862               Just tm' -> tm'
  863               Nothing ->
  864                   panic "updateFFITypes" [
  865                       "Cannot find foreign function in term environment: " <>
  866                           CryPP.pp nm
  867                   ]
  868         info <- case asConstant tm of
  869               Just n ->
  870                 pure $ nameInfo n
  871               Nothing -> do
  872                 tm' <- ppTerm sc tm
  873                 panic "updateFFITypes" [
  874                     "SAWCore term of Cryptol name is not a Constant",
  875                     "Name: " <> CryPP.pp nm,
  876                     "Term: " <> Text.pack tm'
  877                  ]
  878         pure $ (info, ty)
  879   decls <- mapM getNameInfo (T.findForeignDecls m)
  880   pure $ foldr (\(info, ty) decl -> Map.insert info ty decl) eFFITypes' decls

  996 -- | Like `bindExtraVar` but temporary within a passed-in operation.
  997 --
  998 --   That is, it adds a new variable as an "extra" declaration while
  999 --   running the passed in @op@ on the `CryptolEnv`, then drops it
 1000 --   again, preserving unrelated changes to the `CryptolEnv`.
 1001 --
 1002 --   XXX: This will come unstuck if the wrapped operation touches
 1003 --   XXX: `eExtraNaming`, `eExtraVars`, or `eAllTerms`. We need a
 1004 --   XXX: better way to do this; however, there's no way to undo
 1005 --   XXX: `MR.shadowing` so there aren't many choices.
 1006 --
 1007 --   The right way to do this is probably to extend `CryptolEnv` to
 1008 --   support scopes, and then to have the caller create a temporary
 1009 --   scope and use `bindExtraVar`. (Scope support should happen
 1010 --   anyway. Right now the SAWScript interpreter creates stacks of
 1011 --   environments to handle scopes, which was an ad hoc solution to an
 1012 --   immediate need, isn't the right way, and creates its own
 1013 --   problems.)
 1014 --
 1015 withExtraVar ::
 1016     (Ident, TypedTerm) ->
 1017     CryptolEnv ->
 1018     (CryptolEnv -> IO (a, CryptolEnv)) ->
 1019     IO (a, CryptolEnv)
 1020 withExtraVar (ident, TypedTerm (TypedTermSchema schema) trm) env_0 op = do
 1021   -- Note: bindIdent only updates the name supply, arguably it is misnamed
 1022   let (name, env_1) = bindIdent ident env_0
 1023 
 1024   -- Extract the original state
 1025   let naming_1 = eExtraNaming env_1
 1026       extravars_1 = eExtraVars env_1
 1027       allterms_1 = eAllTerms env_1
 1028 
 1029   -- Generate an updated state and a working environment
 1030   let pname = P.mkUnqual ident
 1031       naming_2 = MR.shadowing (MN.singletonNS C.NSValue pname name) naming_1
 1032       extravars_2 = Map.insert name schema extravars_1
 1033       allterms_2 = Map.insert name trm allterms_1
 1034   let env_2 = env_1 {
 1035         eExtraNaming = naming_2,
 1036         eExtraVars = extravars_2,
 1037         eAllTerms = allterms_2
 1038       }
 1039 
 1040   -- Call the op
 1041   (ret, env_3) <- op env_2
 1042 
 1043   -- Restore the original state
 1044   let env_4 = env_3 {
 1045         eExtraNaming = naming_1,
 1046         eExtraVars = extravars_1,
 1047         eAllTerms = allterms_1
 1048       }
 1049 
 1050   -- done
 1051   pure (ret, env_4)
 1052

 1054 -- something, so it'd be a mistake if they passed in a binding that
 1055 -- can't be made visible.
 1056 withExtraVar _ env_0 op =
 1057   op env_0

 1376 -- | Process an `M.ModuleRes` (result of a `MM.ModuleM` computation)
 1377 --   Print errors and warnings.
 1378 --
 1379 --   Suppress warnings about defaulting types. 
 1380 --
 1381 moduleCmdResult :: M.ModuleRes a -> IO (a, ME.ModuleEnv)
 1382 moduleCmdResult (res, ws) = do
 1383   mapM_ (warnN' . CryPP.pretty) (concatMap suppressDefaulting ws)
 1384   case res of
 1385     Right (a, me) -> return (a, me)
 1386     Left err      -> errX' $ "Cryptol:" <+> CryPP.pretty err
 1387   where
 1388     -- If all warnings are about type defaults, pretend there are no warnings at
 1389     -- all to avoid displaying an empty warning container.
 1390     suppressDefaulting :: MM.ModuleWarning -> [MM.ModuleWarning]
 1391     suppressDefaulting w =
 1392       case w of
 1393         MM.RenamerWarnings xs -> [MM.RenamerWarnings xs]
 1394         MM.TypeCheckWarnings nm xs ->
 1395           case filter (notDefaulting . snd) xs of
 1396             [] -> []
 1397             xs' -> [MM.TypeCheckWarnings nm xs']
 1398 
 1399     notDefaulting :: TE.Warning -> Bool
 1400     notDefaulting (TE.DefaultingTo {}) = False
 1401     notDefaulting _ = True
