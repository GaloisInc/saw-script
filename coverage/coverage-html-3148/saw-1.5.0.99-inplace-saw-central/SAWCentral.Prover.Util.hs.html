   18 -- | Is this a bool, or something that returns bool.
   19 checkBooleanType :: C.Type -> IO ()
   20 checkBooleanType ty
   21   | C.tIsBit ty                 = return ()
   22   | Just (_,ty') <- C.tIsFun ty = checkBooleanType ty'
   23   | otherwise = fail ("Invalid non-boolean type: " ++ Text.unpack (CryPP.pp ty))
   24 
   25 -- | Make sure that this schema is monomorphic, and either boolean,
   26 -- or something that returns a boolean.
   27 checkBooleanSchema :: SharedContext -> TypedTermType -> IO ()
   28 checkBooleanSchema _ (TypedTermSchema (C.Forall [] [] t)) = checkBooleanType t
   29 checkBooleanSchema _ (TypedTermSchema s) = fail $ "Invalid polymorphic type: " ++ Text.unpack (CryPP.pp s)
   30 checkBooleanSchema sc tp = do
   31   tp' <- ppTypedTermType sc tp
   32   fail $ "Expected boolean type, but got " ++ (Text.unpack tp')

   34 bindAllExts :: SharedContext -> Term -> IO Term
   35 bindAllExts sc body = scLambdaList sc (getAllVars body) body

   37 -- | Lift a counterexample containing big-endian words
   38 --
   39 --   XXX: this should be explicitly tagged BE
   40 liftCexBB :: [FiniteType] -> [Bool] -> Either String [FiniteValue]
   41 liftCexBB tys bs =
   42   case readFiniteValues tys bs of
   43     Nothing -> Left "Failed to lift counterexample"
   44     Just fvs -> Right fvs
   45 
   46 -- | Lift a counterexample containing little-endian words
   47 liftLECexBB :: [FiniteType] -> [Bool] -> Either String [FiniteValue]
   48 liftLECexBB tys bs =
   49   case readFiniteValuesLE tys bs of
   50     Nothing -> Left "Failed to lift counterexample"
   51     Just fvs -> Right fvs
