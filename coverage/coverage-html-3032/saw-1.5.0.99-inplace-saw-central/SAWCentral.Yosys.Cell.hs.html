   34 -- | A SAWCore bitvector term along with its width and whether it should be interpreted as signed.
   35 data CellTerm = CellTerm
   36   { cellTermTerm :: SC.Term
   37   , cellTermWidth :: Natural
   38   , cellTermSigned :: Bool
   39   }
   40 
   41 cellTermNat :: SC.SharedContext -> CellTerm -> IO SC.Term
   42 cellTermNat sc (CellTerm { cellTermTerm = t, cellTermWidth = w }) = SC.scBvToNat sc w t
   43 
   44 -- | Apply the appropriate (possibly signed) extension or truncation to make the given bitvector
   45 -- match the given width.
   46 extTrunc :: SC.SharedContext -> Natural -> CellTerm -> IO CellTerm
   47 extTrunc sc width (CellTerm { cellTermTerm = t, cellTermWidth = bvw, cellTermSigned = signed}) =
   48   do wterm <- SC.scNat sc width
   49      bvwterm <- SC.scNat sc bvw
   50      res <- if
   51        | bvw > width ->
   52            do diffterm <- SC.scNat sc $ bvw - width
   53               SC.scBvTrunc sc diffterm wterm t
   54        | width > bvw && bvw == 0 ->
   55            do zero <- SC.scNat sc 0
   56               SC.scBvNat sc wterm zero
   57        | width > bvw && signed ->
   58            do bvwpredterm <- SC.scNat sc $ bvw - 1
   59               diffterm <- SC.scNat sc $ width - bvw
   60               SC.scBvSExt sc diffterm bvwpredterm t
   61        | width > bvw && not signed ->
   62            do diffterm <- SC.scNat sc $ width - bvw
   63               SC.scBvUExt sc diffterm bvwterm t
   64        | otherwise -> pure t
   65      pure $ CellTerm res width signed
   66 
   67 -- | Given two bitvectors, extend the narrower bitvector to match the wider.
   68 extMax :: SC.SharedContext -> CellTerm -> CellTerm -> IO (CellTerm, CellTerm)
   69 extMax sc c1 c2 =
   70   do let
   71        w1 = cellTermWidth c1
   72        w2 = cellTermWidth c2
   73        w = max w1 w2
   74      res1 <- extTrunc sc w c1
   75      res2 <- extTrunc sc w c2
   76      pure (res1, res2)
   77 
   78 liftUnary ::
   79   SC.SharedContext ->
   80   (SC.Term -> SC.Term -> IO SC.Term) -> -- (w : Nat) -> [w] -> [w]
   81   CellTerm -> IO CellTerm
   82 liftUnary sc f c@(CellTerm { cellTermTerm = t }) =
   83   do wt <- SC.scNat sc $ cellTermWidth c
   84      res <- f wt t
   85      pure $ c { cellTermTerm = res }
   86 
   87 liftBinary ::
   88   SC.SharedContext ->
   89   (SC.Term -> SC.Term -> SC.Term -> IO SC.Term) -> -- (w : Nat) -> [w] -> [w] -> [w]
   90   CellTerm -> CellTerm -> IO CellTerm
   91 liftBinary sc f c1@(CellTerm { cellTermTerm = t1 }) (CellTerm { cellTermTerm = t2 }) =
   92   do wt <- SC.scNat sc $ cellTermWidth c1
   93      res <- f wt t1 t2
   94      pure $ c1 { cellTermTerm = res }
   95 
   96 liftBinaryCmp ::
   97   SC.SharedContext ->
   98   (SC.Term -> SC.Term -> SC.Term -> IO SC.Term) -> -- (w : Nat) -> [w] -> [w] -> Bool
   99   CellTerm -> CellTerm -> IO SC.Term
  100 liftBinaryCmp sc f c1@(CellTerm { cellTermTerm = t1 }) (CellTerm { cellTermTerm = t2 }) =
  101   do wt <- SC.scNat sc $ cellTermWidth c1
  102      f wt t1 t2
  103 
  104 -- | Perform a two's complement negation of a bit vector.
  105 bvneg :: SC.SharedContext -> CellTerm -> IO CellTerm
  106 bvneg sc a =
  107   do w <- SC.scNat sc (cellTermWidth a)
  108      res <- SC.scBvNeg sc w (cellTermTerm a)
  109      pure $ a {cellTermTerm = res}
  110 
  111 -- | Given an input and a shift amount, perform a logical left shift.
  112 shl :: SC.SharedContext -> CellTerm -> CellTerm -> IO CellTerm
  113 shl sc a b =
  114   do w <- SC.scNat sc (cellTermWidth a)
  115      nb <- cellTermNat sc b
  116      res <- SC.scBvShl sc w (cellTermTerm a) nb
  117      pure $ a {cellTermTerm = res}
  118 
  119 -- | Given an input and a shift amount, perform a logical right shift.
  120 shr :: SC.SharedContext -> CellTerm -> CellTerm -> IO CellTerm
  121 shr sc a b =
  122   do w <- SC.scNat sc (cellTermWidth a)
  123      nb <- cellTermNat sc b
  124      res <- SC.scBvShr sc w (cellTermTerm a) nb
  125      pure $ a {cellTermTerm = res}
  126 
  127 -- | Given an input and a shift amount, perform an signed right shift
  128 -- if the input is signed, and a logical right shift if not.
  129 sshr :: SC.SharedContext -> CellTerm -> CellTerm -> IO CellTerm
  130 sshr sc a b
  131   | cellTermSigned a =
  132     do wt1 <- SC.scNat sc (cellTermWidth a - 1)
  133        nb <- cellTermNat sc b
  134        res <- SC.scBvSShr sc wt1 (cellTermTerm a) nb
  135        pure $ a {cellTermTerm = res}
  136   | otherwise =
  137     shr sc a b
  138 
  139 -- | Given an input and a shift amount, perform a logical left shift
  140 -- if the shift amount is negative, and a logical right shift if the
  141 -- shift amount is positive or unsigned.
  142 shift :: SC.SharedContext -> CellTerm -> CellTerm -> IO CellTerm
  143 shift sc a b
  144   | cellTermSigned b =
  145     -- If the shift amount is signed, then shift right by b if b >= 0
  146     -- and left by -b otherwise.
  147     do let wa = cellTermWidth a
  148        let ta = cellTermTerm a
  149        let wb = cellTermWidth b
  150        let tb = cellTermTerm b
  151        wt <- SC.scNat sc wa
  152        wbt <- SC.scNat sc wb
  153        zero <- SC.scBvConst sc wb 0
  154        tbn <- SC.scBvToNat sc wb tb
  155        tbneg <- SC.scBvNeg sc wbt tb
  156        tbnegn <- SC.scBvToNat sc wb tbneg
  157        cond <- SC.scBvSGe sc wbt tb zero
  158        tcase <- SC.scBvShr sc wt ta tbn
  159        ecase <- SC.scBvShl sc wt ta tbnegn
  160        ty <- SC.scBitvector sc wa
  161        res <- SC.scIte sc ty cond tcase ecase
  162        pure $ a {cellTermTerm = res}
  163   | otherwise =
  164     -- If the shift amount is unsigned, then unconditionally shift right.
  165     shr sc a b

  167 -- | Given a primitive Yosys cell and a map of terms for its
  168 -- arguments, construct a record term representing the output. If the
  169 -- provided cell is not a primitive, return Nothing.
  170 primCellToTerm ::
  171   forall b.
  172   SC.SharedContext ->
  173   Cell [b] {- ^ Cell type -} ->
  174   Map Text SC.Term {- ^ Mapping of input names to input terms -} ->
  175   IO (Maybe SC.Term)
  176 primCellToTerm sc c args =
  177   do mm <- primCellToMap sc c args
  178      traverse (cryptolRecord sc) mm
  179 
  180 primCellToMap ::
  181   forall b.
  182   SC.SharedContext ->
  183   Cell [b] {- ^ Cell type -} ->
  184   Map Text SC.Term {- ^ Mapping of input names to input terms -} ->
  185   IO (Maybe (Map Text SC.Term))
  186 primCellToMap sc c args =
  187   case c ^. cellType of
  188     CellTypeNot -> bvUnaryOp . liftUnary sc $ SC.scBvNot sc
  189     CellTypePos ->
  190       do res <- input "A"
  191          output res
  192     CellTypeNeg ->
  193       -- If output size is larger, then we must extend before negating
  194       -- to compute the high bits correctly.
  195       -- If output size is smaller, truncating first is safe.
  196       do a <- extTrunc sc (connWidthNat "Y") =<< input "A"
  197          output =<< bvneg sc a
  198     CellTypeAnd -> bvBinaryOp . liftBinary sc $ SC.scBvAnd sc
  199     CellTypeOr -> bvBinaryOp . liftBinary sc $ SC.scBvOr sc
  200     CellTypeXor -> bvBinaryOp . liftBinary sc $ SC.scBvXor sc
  201     CellTypeXnor ->
  202       bvBinaryOp $ liftBinary sc $ \w x y ->
  203       do r <- SC.scBvXor sc w x y
  204          SC.scBvNot sc w r
  205     CellTypeReduceAnd ->
  206       do r <- SC.scGlobalDef sc $ SC.mkIdent SC.preludeName "and"
  207          bvReduce True r
  208     CellTypeReduceOr ->
  209       do r <- SC.scGlobalDef sc $ SC.mkIdent SC.preludeName "or"
  210          bvReduce False r
  211     CellTypeReduceXor ->
  212       do r <- SC.scGlobalDef sc $ SC.mkIdent SC.preludeName "xor"
  213          bvReduce False r
  214     CellTypeReduceXnor ->
  215       do boolTy <- SC.scBoolType sc
  216          x <- SC.scFreshVariable sc "x" boolTy
  217          y <- SC.scFreshVariable sc "y" boolTy
  218          r <- SC.scXor sc x y
  219          res <- SC.scNot sc r
  220          t <- SC.scAbstractTerms sc [x, y] res
  221          bvReduce True t
  222     CellTypeReduceBool ->
  223       do r <- SC.scGlobalDef sc $ SC.mkIdent SC.preludeName "or"
  224          bvReduce False r
  225     CellTypeShl ->
  226       -- If output size is larger, then we must extend before shifting
  227       -- to avoid losing high bits.
  228       -- If output size is smaller, truncating first is safe.
  229       do a <- extTrunc sc (connWidthNat "Y") =<< input "A"
  230          b <- input "B"
  231          output =<< shl sc a b
  232     CellTypeShr ->
  233       -- If output size is larger, we must extend before shifting to
  234       -- compute high bits correctly.
  235       -- If output size is smaller, we must shift before truncating.
  236       do let w = max (connWidthNat "A") (connWidthNat "Y")
  237          a <- extTrunc sc w =<< input "A"
  238          b <- input "B"
  239          output =<< shr sc a b
  240     CellTypeSshl ->
  241       -- If output size is larger, then we must extend before shifting
  242       -- to avoid losing high bits.
  243       -- If output size is smaller, truncating first is safe.
  244       do a <- extTrunc sc (connWidthNat "Y") =<< input "A"
  245          b <- input "B"
  246          output =<< shl sc a b
  247     CellTypeSshr ->
  248       -- If output size is larger, we must extend before shifting to
  249       -- compute high bits correctly.
  250       -- If output size is smaller, we must shift before truncating.
  251       do let w = max (connWidthNat "A") (connWidthNat "Y")
  252          a <- extTrunc sc w =<< input "A"
  253          b <- input "B"
  254          output =<< sshr sc a b
  255     CellTypeShift ->
  256       -- If output size is larger, we must extend before shifting to
  257       -- compute high bits correctly.
  258       -- If output size is smaller, we must shift before truncating.
  259       do let w = max (connWidthNat "A") (connWidthNat "Y")
  260          a <- extTrunc sc w =<< input "A"
  261          b <- input "B"
  262          output =<< shift sc a b
  263     CellTypeShiftx ->
  264       -- $shiftx is like $shift, but shifts in x bits instead of 0s.
  265       -- For SAW, we model x bits as 0 bits, so we translate the two
  266       -- cell types identically.
  267       do let w = max (connWidthNat "A") (connWidthNat "Y")
  268          a <- extTrunc sc w =<< input "A"
  269          b <- input "B"
  270          output =<< shift sc a b
  271     CellTypeLt -> bvBinaryCmp . liftBinaryCmp sc $ SC.scBvULt sc
  272     CellTypeLe -> bvBinaryCmp . liftBinaryCmp sc $ SC.scBvULe sc
  273     CellTypeGt -> bvBinaryCmp . liftBinaryCmp sc $ SC.scBvUGt sc
  274     CellTypeGe -> bvBinaryCmp . liftBinaryCmp sc $ SC.scBvUGe sc
  275     CellTypeEq -> bvBinaryCmp . liftBinaryCmp sc $ SC.scBvEq sc
  276     CellTypeNe ->
  277       bvBinaryCmp $ liftBinaryCmp sc $ \w x y ->
  278       do r <- SC.scBvEq sc w x y
  279          SC.scNot sc r
  280     CellTypeEqx -> bvBinaryCmp . liftBinaryCmp sc $ SC.scBvEq sc
  281     CellTypeNex ->
  282       bvBinaryCmp $ liftBinaryCmp sc $ \w x y ->
  283       do r <- SC.scBvEq sc w x y
  284          SC.scNot sc r
  285     CellTypeAdd -> bvBinaryOp . liftBinary sc $ SC.scBvAdd sc
  286     CellTypeSub -> bvBinaryOp . liftBinary sc $ SC.scBvSub sc
  287     CellTypeMul -> bvBinaryOp . liftBinary sc $ SC.scBvMul sc
  288     CellTypeDiv -> bvBinaryOp . liftBinary sc $ SC.scBvUDiv sc
  289     CellTypeMod -> bvBinaryOp . liftBinary sc $ SC.scBvURem sc
  290     -- "$modfloor" -> _
  291     CellTypeLogicNot ->
  292       do w <- connWidth "A"
  293          ta <- cellTermTerm <$> input "A"
  294          anz <- SC.scBvNonzero sc w ta
  295          res <- SC.scNot sc anz
  296          outputBit res
  297     CellTypeLogicAnd ->
  298       bvBinaryCmp $ liftBinaryCmp sc $ \w x y ->
  299       do xnz <- SC.scBvNonzero sc w x
  300          ynz <- SC.scBvNonzero sc w y
  301          SC.scAnd sc xnz ynz
  302     CellTypeLogicOr ->
  303       bvBinaryCmp $ liftBinaryCmp sc $ \w x y ->
  304       do xnz <- SC.scBvNonzero sc w x
  305          ynz <- SC.scBvNonzero sc w y
  306          SC.scOr sc xnz ynz
  307     CellTypeMux ->
  308       do ta <- cellTermTerm <$> input "A"
  309          tb <- cellTermTerm <$> input "B"
  310          ts <- cellTermTerm <$> input "S"
  311          swidth <- connWidth "S"
  312          snz <- SC.scBvNonzero sc swidth ts
  313          let width = connWidthNat "Y"
  314          ty <- SC.scBitvector sc width
  315          res <- SC.scIte sc ty snz tb ta
  316          output $ CellTerm res (connWidthNat "A") (connSigned "A")
  317     CellTypePmux ->
  318       do ta <- cellTermTerm <$> input "A"
  319          tb <- cellTermTerm <$> input "B"
  320          ts <- cellTermTerm <$> input "S"
  321          width <- connWidth "A"
  322          widthBv <- SC.scBitvector sc $ connWidthNat "A"
  323          swidth <- connWidth "S"
  324          bool <- SC.scBoolType sc
  325          splitb <- SC.scSplit sc swidth width bool tb
  326          scPmux <- SC.scGlobalDef sc $ SC.mkIdent SC.preludeName "pmux"
  327          res <- SC.scApplyAll sc scPmux [swidth, widthBv, ts, splitb, ta]
  328          output $ CellTerm res (connWidthNat "A") (connSigned "Y")
  329     CellTypeBmux ->
  330       do ia <- input "A"
  331          is <- input "S"
  332          let swidth = cellTermWidth is
  333          let ywidth = connWidthNat "Y"
  334          -- Split input A into chunks
  335          chunks <- SC.scNat sc (2 ^ swidth)
  336          ywidth' <- SC.scNat sc ywidth
  337          bool <- SC.scBoolType sc
  338          splitA <- SC.scSplit sc chunks ywidth' bool (cellTermTerm ia)
  339          -- reverse to put index 0 on the left
  340          outputType <- SC.scBitvector sc ywidth
  341          revA <- SC.scGlobalApply sc "Prelude.reverse" [chunks, outputType, splitA]
  342          -- Select chunk from output
  343          ixWidth <- SC.scNat sc swidth
  344          elt <- SC.scBvAt sc chunks outputType ixWidth revA (cellTermTerm is)
  345          output (CellTerm elt ywidth (connSigned "Y"))
  346     -- "$demux" -> _
  347     -- "$lut" -> _
  348     -- "$slice" -> _
  349     -- "$concat" -> _
  350     CellTypeDff -> pure Nothing
  351     CellTypeFf -> pure Nothing
  352     CellTypeBUF ->
  353       do res <- input "A"
  354          output res
  355     CellTypeUnsupportedPrimitive _ -> pure Nothing
  356     CellTypeUserType _ -> pure Nothing
  357   where
  358     nm :: Text
  359     nm = Text.pack $ show $ c ^. cellType
  360 
  361     textBinNat :: Text -> Natural
  362     textBinNat = fromIntegral . Text.foldl' (\a x -> digitToInt x + a * 2) 0
  363     connSigned :: Text -> Bool
  364     connSigned onm =
  365       case Map.lookup (onm <> "_SIGNED") $ c ^. cellParameters of
  366         Just (Aeson.Number n) -> n > 0
  367         Just (Aeson.String t) -> textBinNat t > 0
  368         Just v ->
  369           -- XXX This should not be a panic, as it is possible to trigger this
  370           -- with a malformed input file.
  371           panic "cellToTerm"
  372             [ "Expected SIGNED parameter to be a number or a string,"
  373             , "but encountered " <> Text.pack (show v)
  374             ]
  375         Nothing -> False
  376     connWidthNat :: Text -> Natural
  377     connWidthNat onm =
  378       case Map.lookup onm $ c ^. cellConnections of
  379         -- XXX This should not be a panic, as it is possible to trigger this
  380         -- with a malformed input file.
  381         Nothing -> panic "cellToTerm" ["Missing expected output name for " <> nm <> " cell"]
  382         Just bits -> fromIntegral $ length bits
  383     connWidth :: Text -> IO SC.Term
  384     connWidth onm = SC.scNat sc $ connWidthNat onm
  385 
  386     input :: Text -> IO CellTerm
  387     input inpNm =
  388       case Map.lookup inpNm args of
  389         -- XXX This should not be a panic, as it is possible to trigger this
  390         -- with a malformed input file.
  391         Nothing -> panic "cellToTerm" [nm <> " missing input " <> inpNm]
  392         Just a -> pure $ CellTerm a (connWidthNat inpNm) (connSigned inpNm)
  393 
  394     -- | Extend or truncate a cell term as needed to fit the output @Y@ port.
  395     output :: CellTerm -> IO (Maybe (Map Text SC.Term))
  396     output res =
  397       do CellTerm t _ _ <- extTrunc sc (connWidthNat "Y") res
  398          pure (Just (Map.singleton "Y" t))
  399 
  400     outputBit :: SC.Term -> IO (Maybe (Map Text SC.Term))
  401     outputBit res =
  402       do bool <- SC.scBoolType sc
  403          vres <- SC.scSingle sc bool res
  404          output $ CellTerm vres 1 False
  405 
  406     -- convert input to big endian
  407     bvUnaryOp :: (CellTerm -> IO CellTerm) -> IO (Maybe (Map Text SC.Term))
  408     bvUnaryOp f =
  409       do t <- input "A"
  410          res <- f t
  411          output res
  412     -- extend inputs to output width
  413     bvBinaryOp :: (CellTerm -> CellTerm -> IO CellTerm) -> IO (Maybe (Map Text SC.Term))
  414     bvBinaryOp f =
  415       do let w = connWidthNat "Y"
  416          ta <- extTrunc sc w =<< input "A"
  417          tb <- extTrunc sc w =<< input "B"
  418          res <- f ta tb
  419          output res
  420     -- extend inputs to max input width, output is a single bit extended to the output width
  421     bvBinaryCmp :: (CellTerm -> CellTerm -> IO SC.Term) -> IO (Maybe (Map Text SC.Term))
  422     bvBinaryCmp f =
  423       do ta <- input "A"
  424          tb <- input "B"
  425          res <- uncurry f =<< extMax sc ta tb
  426          outputBit res
  427     bvReduce :: Bool -> SC.Term -> IO (Maybe (Map Text SC.Term))
  428     bvReduce boolIdentity boolFun =
  429       do CellTerm t _ _ <- input "A"
  430          w <- connWidth "A"
  431          boolTy <- SC.scBoolType sc
  432          identity <- SC.scBool sc boolIdentity
  433          scFoldr <- SC.scGlobalDef sc $ SC.mkIdent SC.preludeName "foldr"
  434          bit <- SC.scApplyAll sc scFoldr [boolTy, boolTy, w, boolFun, identity, t]
  435          outputBit bit
