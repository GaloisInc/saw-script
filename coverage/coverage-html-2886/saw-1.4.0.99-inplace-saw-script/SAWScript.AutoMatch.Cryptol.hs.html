   31 -- | Parse a Cryptol module into a list of declarations
   32 --   Yields an Interaction so that we can talk to the user about what went wrong
   33 getDeclsCryptol :: FilePath -> IO (Interaction (Maybe [Decl]))
   34 getDeclsCryptol path = do
   35    let evalOpts = EvalOpts quietLogger defaultPPOpts
   36    modEnv <- M.initialModuleEnv
   37    let minp solver = M.ModuleInput {
   38            minpCallStacks = True,
   39            minpSaveRenamed = False,
   40            minpEvalOpts = pure evalOpts,
   41            minpByteReader = BS.readFile,
   42            minpModuleEnv = modEnv,
   43            minpTCSolver = solver
   44        }
   45    (result, warnings) <-
   46      SMT.withSolver (return ()) (meSolverConfig modEnv) $ \solver ->
   47          M.loadModuleByPath path (minp solver)
   48    return $ do
   49       forM_ warnings $ liftF . flip Warning () . pretty
   50       case result of
   51          Left err -> liftF $ Failure True (pretty err) Nothing
   52          Right ((_, top), _) ->
   53             case top of
   54               AST.TCTopSignature {} ->
   55                  liftF $ Failure True "Expected a module but found an intreface" Nothing
   56               AST.TCTopModule AST.Module { mDecls } ->
   57                 let stdDecls = catMaybes . for (concatMap flattenDeclGroup mDecls) $
   58                       \(AST.Decl{dName, dSignature, dDefinition}) -> do
   59                          funcName <- sourceName dName
   60                          argNames <- mapM sourceName =<< tupleLambdaBindings =<< declDefExpr dDefinition
   61                          (argTypes, retType) <- tupleFunctionType (AST.sType dSignature)
   62                          return $ Decl funcName retType (zipWith Arg argNames argTypes)
   63                 in return $ Just (stdDecls :: [Decl])
   64 
   65 -- All this is just sifting through the Cryptol typechecker's AST to get the information we want:
   66 
   67 -- Things will break if Cryptol's internals shift the way they desugar tuple bindings in declarations:
   68 -- Currently, a declaration like @foo (a,b,c) = ...@ turns into something
   69 -- like @foo = \x -> ... where a = x.0.; b = x.1; c = x.2@
   70 -- We rely on this behavior to parse out the information we need.
   71 
   72 -- | We don't care about recursive bindings, so we flatten them out
   73 flattenDeclGroup :: AST.DeclGroup -> [AST.Decl]
   74 flattenDeclGroup (AST.Recursive decls)   = decls
   75 flattenDeclGroup (AST.NonRecursive decl) = [decl]
   76 
   77 -- | If the expression is a tuple projection, get the name of the tuple and the index projected
   78 tupleSelInfo :: AST.Expr -> Maybe (AST.Name, Int)
   79 tupleSelInfo (AST.ESel (AST.EVar name) (AST.TupleSel i _)) = return (name, i)
   80 tupleSelInfo _                                             = Nothing
   81 
   82 -- | If an expression is a where binding, get all its local declarations
   83 whereBindings :: AST.Expr -> Maybe [AST.Decl]
   84 whereBindings (AST.EWhere _ bindings) = return $ concatMap flattenDeclGroup bindings
   85 whereBindings _                       = Nothing
   86 
   87 -- | Find the expression inside a definition
   88 --   We can't handle primitives currently
   89 declDefExpr :: AST.DeclDef -> Maybe AST.Expr
   90 declDefExpr = \case
   91    AST.DPrim            -> Nothing
   92    AST.DForeign _ mexpr -> mexpr
   93    AST.DExpr expr       -> Just expr
   94 
   95 -- | If a lambda is of the form @\(a,b,...,z) -> ...)@ then give the list of names bound in the tuple
   96 tupleLambdaBindings :: AST.Expr -> Maybe [AST.Name]
   97 tupleLambdaBindings (AST.EAbs tupleName _ whereClause) = do
   98    bindings <- whereBindings whereClause
   99    return . map snd . sortBy (comparing fst) . catMaybes . for bindings $
  100       \AST.Decl{dDefinition, dName} -> do
  101          (from, index) <- tupleSelInfo =<< declDefExpr dDefinition
  102          guard (from == tupleName)
  103          return (index, dName)
  104 tupleLambdaBindings (AST.ETAbs _ expr)     = tupleLambdaBindings expr   --   \
  105 tupleLambdaBindings (AST.EProofAbs _ expr) = tupleLambdaBindings expr   --    >- drill down past casts & type abstractions
  106 tupleLambdaBindings _ = Nothing
  107 
  108 -- | If the type given is of the form @(a,b,...,y) -> z@ then give the pair of arguments and result
  109 tupleFunctionType :: AST.Type -> Maybe ([AST.Type], AST.Type)
  110 tupleFunctionType (AST.TCon (AST.TC AST.TCFun) [AST.TCon (AST.TC (AST.TCTuple _)) inputs, output]) = return (inputs, output)
  111 tupleFunctionType _                                                                                = Nothing
  112 
  113 -- | Find the name from the source if one exists
  114 sourceName :: Name -> Maybe Text
  115 sourceName nm = Just (identText (nameIdent nm))
