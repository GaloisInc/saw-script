  255 constMap :: forall sym. Sym sym => sym -> Map Ident (SPrim sym)
  256 constMap sym =
  257   Map.union (Prims.constMap (prims sym)) $
  258   Map.fromList
  259   [
  260   -- Shifts
  261     ("Prelude.bvShl" , bvShLOp sym)
  262   , ("Prelude.bvShr" , bvShROp sym)
  263   , ("Prelude.bvSShr", bvSShROp sym)
  264   -- Integers
  265   , ("Prelude.intToNat", intToNatOp sym)
  266   , ("Prelude.natToInt", natToIntOp sym)
  267   , ("Prelude.intToBv" , intToBvOp sym)
  268   , ("Prelude.bvToInt" , bvToIntOp sym)
  269   , ("Prelude.sbvToInt", sbvToIntOp sym)
  270   -- Integers mod n
  271   , ("Prelude.toIntMod"  , toIntModOp)
  272   , ("Prelude.fromIntMod", fromIntModOp sym)
  273   , ("Prelude.intModEq"  , intModEqOp sym)

  288 swBvWidth :: SWord sym -> Int
  289 swBvWidth x
  290   | w <= toInteger (maxBound :: Int) = fromInteger w
  291   | otherwise = croak
  292  where w = SW.bvWidth x
  293        croak = panic "swBvWidth" [
  294            "Bitvector too long: width " <> Text.pack (show w) <>
  295                ", value " <> Text.pack (show x)
  296         ]

  302 toWord :: forall sym.
  303   Sym sym => sym -> SValue sym -> IO (SWord sym)

  325   IsExpr (SymExpr sym) =>
  326   W.BaseTypeRepr tp ->

  329 symExprToValue tp expr = case tp of
  330   BaseBoolRepr -> Just $ VBool expr

  340 -- primitive intToNat : Integer -> Nat;
  341 -- intToNat x == max 0 x
  342 intToNatOp :: forall sym. Sym sym => sym -> SPrim sym
  343 intToNatOp sym =
  344   Prims.intFun $ \i ->
  345   Prims.Prim $
  346     case W.asInteger i of
  347       Just i'
  348         | 0 <= i'   -> pure (VNat (fromInteger i'))
  349         | otherwise -> pure (VNat 0)
  350       Nothing ->
  351         do z <- W.intLit sym 0
  352            pneg <- W.intLt sym i z
  353            i' <- W.intIte sym pneg z i
  354            pure (VIntToNat (VInt i'))
  355 
  356 -- primitive natToInt :: Nat -> Integer;
  357 natToIntOp :: forall sym. Sym sym => sym -> SPrim sym
  358 natToIntOp sym =
  359   Prims.natFun $ \n ->
  360     Prims.Prim (VInt <$> W.intLit sym (toInteger n))

  386 --
  387 -- Shifts
  388 --
  389 
  390 -- | Shift left, shifting in copies of the given bit
  391 bvShl :: IsExprBuilder sym => sym -> Pred sym -> SWord sym -> SWord sym -> IO (SWord sym)
  392 bvShl sym z w i =
  393   W.iteM SW.bvIte sym z
  394     (do w' <- SW.bvNot sym w
  395         SW.bvNot sym =<< SW.bvShl sym w' i)
  396     (SW.bvShl sym w i)
  397 
  398 -- | Shift right, shifting in copies of the given bit
  399 bvShr :: IsExprBuilder sym => sym -> Pred sym -> SWord sym -> SWord sym -> IO (SWord sym)
  400 bvShr sym z w i =
  401   W.iteM SW.bvIte sym z
  402     (do w' <- SW.bvNot sym w
  403         SW.bvNot sym =<< SW.bvLshr sym w' i)
  404     (SW.bvLshr sym w i)
  405 
  406 liftShift :: IsExprBuilder sym =>
  407   sym ->
  408   (SWord sym -> SWord sym -> IO (SWord sym)) ->
  409   SWord sym -> Integer -> IO (SWord sym)
  410 liftShift sym f w i =
  411   f w =<< SW.bvLit sym (SW.bvWidth w) (i `min` SW.bvWidth w)
  412 
  413 liftRotate :: IsExprBuilder sym =>
  414   sym ->
  415   (SWord sym -> SWord sym -> IO (SWord sym)) ->
  416   SWord sym -> Integer -> IO (SWord sym)
  417 liftRotate sym f w i =
  418   f w =<< SW.bvLit sym (SW.bvWidth w) (i `mod` SW.bvWidth w)

  421 -- | op : (n : Nat) -> Vec n Bool -> Nat -> Vec n Bool
  422 bvShiftOp :: Sym sym => sym ->
  423              (SWord sym -> SWord sym -> IO (SWord sym)) ->
  424              (SWord sym -> Integer   -> IO (SWord sym)) -> SPrim sym
  425 bvShiftOp sym bvOp natOp =
  426   Prims.constFun  $                  -- additional argument? the size?
  427   wordFun sym $ \x ->                -- word to shift
  428   Prims.strictFun $ \y ->            -- amount to shift as a nat
  429   Prims.Prim $
  430     case y of
  431       VNat i   -> VWord <$> natOp x j
  432         where j = toInteger i `min` SW.bvWidth x
  433       VBVToNat _ v -> VWord <$> (bvOp x =<< toWord sym v)
  434       _        -> error $ unwords ["SAWCoreWhat4.What4.bvShiftOp", show y]
  435 
  436 -- bvShl : (w : Nat) -> Vec w Bool -> Nat -> Vec w Bool;
  437 bvShLOp :: forall sym. Sym sym => sym -> SPrim sym
  438 bvShLOp sym = bvShiftOp sym (SW.bvShl sym)
  439                     (liftShift sym (SW.bvShl sym))

  441 -- bvShR : (w : Nat) -> Vec w Bool -> Nat -> Vec w Bool;
  442 bvShROp :: forall sym. Sym sym => sym -> SPrim sym
  443 bvShROp sym = bvShiftOp sym (SW.bvLshr sym)
  444                     (liftShift sym (SW.bvLshr sym))

  446 -- bvSShR : (w : Nat) -> Vec w Bool -> Nat -> Vec w Bool;
  447 bvSShROp :: forall sym. Sym sym => sym -> SPrim sym
  448 bvSShROp sym = bvShiftOp sym (SW.bvAshr sym)

  451 bvForall :: W.IsSymExprBuilder sym =>
  452   sym -> Natural -> (SWord sym -> IO (Pred sym)) -> IO (Pred sym)
  453 bvForall sym n f =
  454   do let indexSymbol = W.safeSymbol "i"
  455      case mkNatRepr n of
  456         Some w
  457           | Just LeqProof <- testLeq (knownNat @1) w ->
  458             withKnownNat w $ do
  459               i <- W.freshBoundVar sym indexSymbol $ W.BaseBVRepr w
  460               body <- f . DBV $ W.varExpr sym i
  461               W.forallPred sym i body
  462           | otherwise -> f ZBV

  493 intModEqOp :: IsExprBuilder sym => sym -> SPrim sym
  494 intModEqOp sym =
  495   Prims.natFun $ \n ->
  496   Prims.intModFun $ \x ->
  497   Prims.intModFun $ \y -> Prims.Prim $
  498   do modulus <- W.intLit sym (toInteger n)
  499      d <- W.intSub sym x y
  500      r <- W.intMod sym d modulus
  501      z <- W.intLit sym 0
  502      VBool <$> W.intEq sym r z

  504 intModBinOp ::
  505   IsExprBuilder sym => sym ->
  506   (sym -> SInt sym -> SInt sym -> IO (SInt sym)) -> SPrim sym
  507 intModBinOp sym f =
  508   Prims.natFun $ \n ->
  509   Prims.intModFun $ \x ->
  510   Prims.intModFun $ \y -> Prims.Prim
  511     (VIntMod n <$> (normalizeIntMod sym n =<< f sym x y))

  513 intModUnOp ::
  514   IsExprBuilder sym => sym ->
  515   (sym -> SInt sym -> IO (SInt sym)) -> SPrim sym
  516 intModUnOp sym f =
  517   Prims.natFun $ \n ->
  518   Prims.intModFun $ \x ->
  519     Prims.Prim (VIntMod n <$> (normalizeIntMod sym n =<< f sym x))

  521 normalizeIntMod :: IsExprBuilder sym => sym -> Natural -> SInt sym -> IO (SInt sym)
  522 normalizeIntMod sym n x =
  523   case W.asInteger x of
  524     Nothing -> return x
  525     Just i -> W.intLit sym (i `mod` toInteger n)

  530 -- MkStream :: (a :: sort 0) -> (Nat -> a) -> Stream a;
  531 mkStreamOp :: SPrim sym
  532 mkStreamOp =
  533   Prims.constFun $
  534   Prims.strictFun $ \f -> Prims.Prim $
  535     do r <- newIORef Map.empty
  536        return $ VExtra (SStream (\n -> apply f (ready (VNat n))) r)

  538 -- streamGet :: (a :: sort 0) -> Stream a -> Nat -> a;
  539 streamGetOp :: forall sym. Sym sym => sym -> SPrim sym
  540 streamGetOp sym =
  541   Prims.tvalFun   $ \tp ->
  542   Prims.strictFun $ \xs ->
  543   Prims.strictFun $ \ix ->
  544   Prims.Prim $
  545     case ix of
  546       VNat n -> lookupSStream xs n
  547       VBVToNat _ w ->
  548         do ilv <- toWord sym w
  549            selectV sym (lazyMux @sym (muxBVal sym tp)) ((2 ^ SW.bvWidth ilv) - 1) (lookupSStream xs) ilv
  550       v -> panic "streamGetOp" ["Expected Nat value, found: " <> Text.pack (show v)]
  551 
  552 lookupSStream :: SValue sym -> Natural -> IO (SValue sym)
  553 lookupSStream (VExtra (SStream f r)) n = do
  554    m <- readIORef r
  555    case Map.lookup n m of
  556      Just v  -> return v
  557      Nothing -> do v <- f n
  558                    writeIORef r (Map.insert n v m)
  559                    return v
  560 lookupSStream _ _ = fail "SAWCoreWhat4.What4.lookupSStream: expected Stream"

  591 -- @selectV sym merger maxValue valueFn vx@ treats @vx@ as an index, represented
  592 -- as a big-endian list of bits. It does a binary lookup, using @merger@ as an
  593 -- if-then-else operator. If the index is greater than @maxValue@, then it
  594 -- returns @valueFn maxValue@.
  595 selectV :: forall sym b.
  596   Sym sym =>
  597   sym ->
  598   (SBool sym -> IO b -> IO b -> IO b) -> Natural -> (Natural -> IO b) -> SWord sym -> IO b
  599 selectV sym merger maxValue valueFn vx =
  600   case SW.bvAsUnsignedInteger vx of
  601     Just i  -> valueFn (fromInteger i :: Natural)
  602     Nothing -> impl (swBvWidth vx) 0
  603   where
  604     impl :: Int -> Natural -> IO b
  605     impl _ x | x > maxValue || x < 0 = valueFn maxValue
  606     impl 0 y = valueFn y
  607     impl i y = do
  608       -- NB: `i` counts down in each iteration, so we use bvAtLE (a
  609       -- little-endian indexing function) to ensure that the bits are processed
  610       -- in big-endian order. Alternatively, we could have `i` count up and use
  611       -- bvAtBE (a big-endian indexing function), but we use bvAtLE as it is
  612       -- slightly cheaper to compute.
  613       p <- SW.bvAtLE sym vx (toInteger j)
  614       merger p (impl j (y `setBit` j)) (impl j y) where j = i - 1

  824 w4SolveBasic ::
  825   forall sym. IsSymExprBuilder sym =>
  826   sym ->

  829   Map VarIndex (SValue sym) {- ^ bindings for ExtCns values -} ->

  857   SymFnWrapper :: !(W.SymFn sym args ret) -> SymFnWrapper sym (args Ctx.::> ret)

  861 lookupSymFn ::
  862   W.SolverSymbol -> Assignment BaseTypeRepr args -> BaseTypeRepr ty ->
  863   SymFnCache sym -> Maybe (W.SymFn sym args ty)
  864 lookupSymFn s args ty cache =
  865   do m <- Map.lookup s cache
  866      SymFnWrapper fn <- MapF.lookup (Ctx.extend args ty) m
  867      return fn

  869 insertSymFn ::
  870   W.SolverSymbol -> Assignment BaseTypeRepr args -> BaseTypeRepr ty ->
  871   W.SymFn sym args ty -> SymFnCache sym -> SymFnCache sym
  872 insertSymFn s args ty fn = Map.alter upd s
  873   where
  874     upd Nothing = Just (MapF.singleton (Ctx.extend args ty) (SymFnWrapper fn))
  875     upd (Just m) = Just (MapF.insert (Ctx.extend args ty) (SymFnWrapper fn) m)

  877 mkSymFn ::
  878   forall sym args ret. (IsSymExprBuilder sym) =>
  879   sym -> IORef (SymFnCache sym) ->
  880   String -> Assignment BaseTypeRepr args -> BaseTypeRepr ret ->
  881   IO (W.SymFn sym args ret)
  882 mkSymFn sym ref nm args ret =
  883   do let s = W.safeSymbol nm
  884      cache <- readIORef ref
  885      case lookupSymFn s args ret cache of
  886        Just fn -> return fn
  887        Nothing ->
  888          do fn <- W.freshTotalUninterpFn sym s args ret
  889             writeIORef ref (insertSymFn s args ret fn cache)
  890             return fn

  893 -- Given a constant nm of (saw-core) type ty, construct an uninterpreted
  894 -- constant with that type.
  895 -- Importantly: The types of these uninterpreted values are *not*
  896 -- limited to What4 BaseTypes or FirstOrderTypes.
  897 
  898 parseUninterpreted ::
  899   forall sym.
  900   (IsSymExprBuilder sym) =>
  901   sym -> IORef (SymFnCache sym) ->
  902   UnintApp (SymExpr sym) ->
  903   TValue (What4 sym) -> IO (SValue sym)
  904 parseUninterpreted sym ref app ty =
  905   case ty of
  906     VPiType nm _ body
  907       -> pure $ VFun nm $ \x ->
  908            do x' <- force x
  909               app' <- applyUnintApp sym app x'
  910               t2 <- applyPiBody body (ready x')
  911               parseUninterpreted sym ref app' t2
  912 
  913     VBoolType
  914       -> VBool <$> mkUninterpreted sym ref app BaseBoolRepr
  915 
  916     VIntType
  917       -> VInt  <$> mkUninterpreted sym ref app BaseIntegerRepr
  918 
  919     VIntModType n
  920       -> VIntMod n <$> mkUninterpreted sym ref app BaseIntegerRepr
  921 
  922     -- 0 width bitvector is a constant
  923     VVecType 0 VBoolType
  924       -> return $ VWord ZBV
  925 
  926     VVecType n VBoolType
  927       | Just (Some (PosNat w)) <- somePosNat n
  928       -> (VWord . DBV) <$> mkUninterpreted sym ref app (BaseBVRepr w)
  929 
  930     VVecType n ety
  931       ->  do xs <- sequence $
  932                   [ parseUninterpreted sym ref (suffixUnintApp ("_a" ++ show i) app) ety
  933                   | i <- [0 .. n-1] ]
  934              return (VVector (V.fromList (map ready xs)))
  935 
  936     VArrayType ity ety
  937       | Just (Some idx_repr) <- valueAsBaseType ity
  938       , Just (Some elm_repr) <- valueAsBaseType ety
  939       -> (VArray . SArray) <$>
  940           mkUninterpreted sym ref app (BaseArrayRepr (Ctx.Empty Ctx.:> idx_repr) elm_repr)
  941 
  942     VUnitType
  943       -> return VUnit
  944 
  945     VPairType ty1 ty2
  946       -> do x1 <- parseUninterpreted sym ref (suffixUnintApp "_L" app) ty1
  947             x2 <- parseUninterpreted sym ref (suffixUnintApp "_R" app) ty2
  948             return (VPair (ready x1) (ready x2))
  949 
  950     VRecordType elem_tps
  951       -> (VRecordValue <$>
  952           mapM (\(f,tp) ->
  953                  (f,) <$> ready <$>
  954                  parseUninterpreted sym ref (suffixUnintApp ("_" ++ Text.unpack f) app) tp) elem_tps)
  955 
  956     _ -> fail $ "could not create uninterpreted symbol of type " ++ show ty

  969 -- | A value of type @UnintApp f@ represents an uninterpreted function
  970 -- with the given 'String' name, applied to a list of argument values
  971 -- paired with a representation of their types. The context of
  972 -- argument types is existentially quantified.
  973 data UnintApp f =
  974   forall args. UnintApp String (Assignment f args) (Assignment BaseTypeRepr args)

  976 -- | Extract the string from an 'UnintApp'.
  977 stringOfUnintApp :: UnintApp f -> String
  978 stringOfUnintApp (UnintApp s _ _) = s

  980 -- | Make an 'UnintApp' with the given name and no arguments.
  981 mkUnintApp :: String -> UnintApp f
  982 mkUnintApp nm = UnintApp nm Ctx.empty Ctx.empty

  984 -- | Add a suffix to the function name of an 'UnintApp'.
  985 suffixUnintApp :: String -> UnintApp f -> UnintApp f
  986 suffixUnintApp s (UnintApp nm args tys) = UnintApp (nm ++ s) args tys

  988 -- | Extend an 'UnintApp' with an additional argument.
  989 extendUnintApp :: UnintApp f -> f ty -> BaseTypeRepr ty -> UnintApp f
  990 extendUnintApp (UnintApp nm xs tys) x ty =
  991   UnintApp nm (Ctx.extend xs x) (Ctx.extend tys ty)

  993 -- | Flatten an 'SValue' to a sequence of components, each of which is
  994 -- a symbolic value of a base type (e.g. word or boolean), and add
  995 -- them to the list of arguments of the given 'UnintApp'. If the
  996 -- 'SValue' contains any values built from data constructors, then
  997 -- encode them as suffixes on the function name of the 'UnintApp'.
  998 applyUnintApp ::
  999   forall sym.
 1000   (W.IsExprBuilder sym) =>
 1001   sym ->
 1002   UnintApp (SymExpr sym) ->
 1003   SValue sym ->
 1004   IO (UnintApp (SymExpr sym))
 1005 applyUnintApp sym app0 v =
 1006   case v of
 1007     VUnit                     -> return app0
 1008     VPair x y                 -> do app1 <- applyUnintApp sym app0 =<< force x
 1009                                     app2 <- applyUnintApp sym app1 =<< force y
 1010                                     return app2
 1011     VRecordValue elems        -> foldM (applyUnintApp sym) app0 =<< traverse (force . snd) elems
 1012     VVector xv                -> foldM (applyUnintApp sym) app0 =<< traverse force xv
 1013     VBool sb                  -> return (extendUnintApp app0 sb BaseBoolRepr)
 1014     VInt si                   -> return (extendUnintApp app0 si BaseIntegerRepr)
 1015     VIntMod 0 si              -> return (extendUnintApp app0 si BaseIntegerRepr)
 1016     VIntMod n si              -> do n' <- W.intLit sym (toInteger n)
 1017                                     si' <- W.intMod sym si n'
 1018                                     return (extendUnintApp app0 si' BaseIntegerRepr)
 1019     VWord (DBV sw)            -> return (extendUnintApp app0 sw (W.exprType sw))
 1020     VArray (SArray sa)        -> return (extendUnintApp app0 sa (W.exprType sa))
 1021     VWord ZBV                 -> return app0
 1022     VCtorApp i ps xv          -> foldM (applyUnintApp sym) app' =<< traverse force (ps++xv)
 1023                                    where app' = suffixUnintApp ("_" ++ (Text.unpack (identBaseName (primName i)))) app0
 1024     VNat n                    -> return (suffixUnintApp ("_" ++ show n) app0)
 1025     VBVToNat w v'             -> applyUnintApp sym app' v'
 1026                                    where app' = suffixUnintApp ("_" ++ show w) app0
 1027     TValue (suffixTValue -> Just s)
 1028                               -> return (suffixUnintApp s app0)
 1029     VFun _ _ ->
 1030       fail $
 1031       "Cannot create uninterpreted higher-order function " ++
 1032       show (stringOfUnintApp app0)
 1033     _ ->
 1034       fail $
 1035       "Cannot create uninterpreted function " ++
 1036       show (stringOfUnintApp app0) ++
 1037       " with argument " ++ show v

 1090 -- | Given a list of external constants with first-order types,
 1091 --   descend in to the structure of those types (as needed) and construct
 1092 --   corresponding What4 bound variables so we can bind them using
 1093 --   a forall quantifier. At the same time construct @SValue@s containing
 1094 --   those variables suitable for passing to the term evaluator as substituions
 1095 --   for the given @ExtCns@ values. The length of the @SValue@ list returned
 1096 --   will match the list of the input @ExtCns@ list, but the list of What4
 1097 --   @BoundVar@s might not.
 1098 --
 1099 --   This procedure it capable of handling most first-order types, execpt
 1100 --   that Array types must have base types as index and result types rather
 1101 --   than more general first-order types. (TODO? should we actually restrict the
 1102 --   @FirstOrderType@ in the same way?)
 1103 boundFOTs :: forall sym.
 1104   IsSymExprBuilder sym =>
 1105   sym ->
 1106   [(ExtCns Term, FirstOrderType)] ->
 1107   IO ([SValue sym], [Some (BoundVar sym)])
 1108 boundFOTs sym vars =
 1109   do (svals,(bndvars,_)) <- runStateT (mapM (uncurry handleVar) vars) ([], 0)
 1110      return (svals, bndvars)
 1111 
 1112  where
 1113    freshBnd :: ExtCns Term -> W.BaseTypeRepr tp -> StateT ([Some (BoundVar sym)],Integer) IO (SymExpr sym tp)
 1114    freshBnd ec tpr =
 1115      do (vs,n) <- get
 1116         let nm = Text.unpack (toShortName (ecName ec)) ++ "." ++ show n
 1117         bvar <- lift (W.freshBoundVar sym (W.safeSymbol nm) tpr)
 1118         put (Some bvar : vs, n+1)
 1119         return (W.varExpr sym bvar)
 1120 
 1121    handleVar :: ExtCns Term -> FirstOrderType -> StateT ([Some (BoundVar sym)], Integer) IO (SValue sym)
 1122    handleVar ec fot =
 1123      case fot of
 1124        FOTBit -> VBool <$> freshBnd ec BaseBoolRepr
 1125        FOTInt -> VInt  <$> freshBnd ec BaseIntegerRepr
 1126        FOTIntMod m -> VIntMod m <$> freshBnd ec BaseIntegerRepr
 1127 
 1128        FOTVec n FOTBit ->
 1129          case somePosNat n of
 1130            Nothing -> -- n == 0
 1131              return (VWord ZBV)
 1132            Just (Some (PosNat nr)) ->
 1133              VWord . DBV <$> freshBnd ec (BaseBVRepr nr)
 1134 
 1135        FOTVec n tp -> -- NB, not Bit
 1136          do vs  <- V.replicateM (fromIntegral n) (handleVar ec tp)
 1137             vs' <- traverse (return . ready) vs
 1138             return (VVector vs')
 1139 
 1140        FOTRec tm ->
 1141          do vs  <- traverse (handleVar ec) tm
 1142             vs' <- traverse (return . ready) vs
 1143             return (vRecord vs')
 1144 
 1145        FOTTuple ts ->
 1146          do vs  <- traverse (handleVar ec) ts
 1147             vs' <- traverse (return . ready) vs
 1148             return (vTuple vs')
 1149 
 1150        FOTArray idx res
 1151          | Just (Some idx_repr) <- fotToBaseType idx
 1152          , Just (Some res_repr) <- fotToBaseType res
 1153 
 1154          -> VArray . SArray <$> freshBnd ec (BaseArrayRepr (Ctx.Empty Ctx.:> idx_repr) res_repr)

 1156          | otherwise -> fail ("boundFOTs: cannot handle " ++ show fot)

 1334   forall n st fs.

 1337   SharedContext ->

 1339   Set VarIndex ->

 1341   IO Term

 1431 -- | Simplify a saw-core term by evaluating it through the saw backend
 1432 -- of what4.
 1433 w4EvalAny ::
 1434   forall n st fs.
 1435   B.ExprBuilder n st fs ->
 1436   SAWCoreState n ->
 1437   SharedContext ->
 1438   Map Ident (SPrim (B.ExprBuilder n st fs)) ->
 1439   Set VarIndex ->
 1440   Term ->
 1441   IO ([String],
 1442       [SValue (B.ExprBuilder n st fs)],
 1443       [Maybe (Labeler (B.ExprBuilder n st fs))],
 1444       SValue (B.ExprBuilder n st fs))
 1445 w4EvalAny sym st sc ps unintSet t =
 1446   do modmap <- scGetModuleMap sc
 1447      ref <- newIORef Map.empty
 1448      let eval = w4EvalBasic sym st sc modmap ps ref unintSet
 1449      ty <- eval =<< scTypeOf sc t
 1450 
 1451      -- get the names of the arguments to the function
 1452      let lamNames = map (Text.unpack . fst) (fst (R.asLambdaList t))
 1453      let varNames = [ "var" ++ show (i :: Integer) | i <- [0 ..] ]
 1454      let argNames = zipWith (++) varNames (map ("_" ++) lamNames ++ repeat "")
 1455 
 1456      -- and their types
 1457      argTs <- argTypes ty
 1458 
 1459      -- construct symbolic expressions for the variables
 1460      vars' <-
 1461        flip evalStateT 0 $
 1462        sequence (zipWith (newVarsForType sym ref) argTs argNames)
 1463 
 1464      -- symbolically evaluate
 1465      bval <- eval t
 1466 
 1467      -- apply and existentially quantify
 1468      let (bvs, vars) = unzip vars'
 1469      let vars'' = fmap ready vars
 1470      bval' <- applyAll bval vars''
 1471 
 1472      return (argNames, vars, bvs, bval')

 1560 -- | Given a constant nm of (saw-core) type ty, construct an
 1561 -- uninterpreted constant with that type. The 'Term' argument should
 1562 -- be an open term, which should have the designated return type when
 1563 -- the local variables have the corresponding types from the
 1564 -- 'Assignment'.
 1565 parseUninterpretedSAW ::
 1566   forall n st fs.
 1567   B.ExprBuilder n st fs ->
 1568   SAWCoreState n ->
 1569   SharedContext ->
 1570   IORef (SymFnCache (B.ExprBuilder n st fs)) ->
 1571   ArgTerm {- ^ representation of function applied to arguments -} ->
 1572   UnintApp (SymExpr (B.ExprBuilder n st fs)) ->
 1573   TValue (What4 (B.ExprBuilder n st fs)) {- ^ return type -} ->
 1574   IO (SValue (B.ExprBuilder n st fs))
 1575 parseUninterpretedSAW sym st sc ref trm app ty =
 1576   case ty of
 1577     VPiType nm t1 body
 1578       -> pure $ VFun nm $ \x ->
 1579            do x' <- force x
 1580               app' <- applyUnintApp sym app x'
 1581               arg <- mkArgTerm sc t1 x'
 1582               let trm' = ArgTermApply trm arg
 1583               t2 <- applyPiBody body (ready x')
 1584               parseUninterpretedSAW sym st sc ref trm' app' t2
 1585 
 1586     VBoolType
 1587       -> VBool <$> mkUninterpretedSAW sym st sc ref trm app BaseBoolRepr
 1588 
 1589     VIntType
 1590       -> VInt  <$> mkUninterpretedSAW sym st sc ref trm app BaseIntegerRepr
 1591 
 1592     VIntModType n
 1593       -> VIntMod n <$> mkUninterpretedSAW sym st sc ref (ArgTermFromIntMod n trm) app BaseIntegerRepr
 1594 
 1595     -- 0 width bitvector is a constant
 1596     VVecType 0 VBoolType
 1597       -> return $ VWord ZBV
 1598 
 1599     VVecType n VBoolType
 1600       | Just (Some (PosNat w)) <- somePosNat n
 1601       -> (VWord . DBV) <$> mkUninterpretedSAW sym st sc ref trm app (BaseBVRepr w)
 1602 
 1603     VVecType n ety | n >= 0
 1604       ->  do ety' <- termOfTValue sc ety
 1605              let mkElem i =
 1606                    do let trm' = ArgTermAt n ety' trm i
 1607                       let app' = suffixUnintApp ("_a" ++ show i) app
 1608                       parseUninterpretedSAW sym st sc ref trm' app' ety
 1609              xs <- traverse mkElem (genericTake n [0 ..])
 1610              return (VVector (V.fromList (map ready xs)))
 1611 
 1612     VArrayType ity ety
 1613       | Just (Some idx_repr) <- valueAsBaseType ity
 1614       , Just (Some elm_repr) <- valueAsBaseType ety
 1615       -> (VArray . SArray) <$>
 1616           mkUninterpretedSAW sym st sc ref trm app (BaseArrayRepr (Ctx.Empty Ctx.:> idx_repr) elm_repr)
 1617 
 1618     VUnitType
 1619       -> return VUnit
 1620 
 1621     VPairType ty1 ty2
 1622       -> do let trm1 = ArgTermPairLeft trm
 1623             let trm2 = ArgTermPairRight trm
 1624             x1 <- parseUninterpretedSAW sym st sc ref trm1 (suffixUnintApp "_L" app) ty1
 1625             x2 <- parseUninterpretedSAW sym st sc ref trm2 (suffixUnintApp "_R" app) ty2
 1626             return (VPair (ready x1) (ready x2))
 1627 
 1628     _ -> fail $ "could not create uninterpreted symbol of type " ++ show ty

 1650 data ArgTerm
 1651   = ArgTermVar
 1652   | ArgTermBVZero -- ^ scBvNat 0 0
 1653   | ArgTermToIntMod Natural ArgTerm -- ^ toIntMod n x
 1654   | ArgTermFromIntMod Natural ArgTerm -- ^ fromIntMod n x
 1655   | ArgTermVector Term [ArgTerm] -- ^ element type, elements
 1656   | ArgTermUnit
 1657   | ArgTermPair ArgTerm ArgTerm
 1658   | ArgTermRecord [(FieldName, ArgTerm)]
 1659   | ArgTermConst Term
 1660   | ArgTermApply ArgTerm ArgTerm
 1661   | ArgTermAt Natural Term ArgTerm Natural
 1662     -- ^ length, element type, list, index
 1663   | ArgTermPairLeft ArgTerm
 1664   | ArgTermPairRight ArgTerm
 1665   | ArgTermBVToNat Natural ArgTerm

 1667 -- | Reassemble a saw-core term from an 'ArgTerm' and a list of parts.
 1668 -- The length of the list should be equal to the number of
 1669 -- 'ArgTermVar' constructors in the 'ArgTerm'.
 1670 reconstructArgTerm :: ArgTerm -> SharedContext -> [Term] -> IO Term
 1671 reconstructArgTerm atrm sc ts =
 1672   do (t, unused) <- parse atrm ts
 1673      unless (null unused) $ fail "reconstructArgTerm: too many function arguments"
 1674      return t
 1675   where
 1676     parse :: ArgTerm -> [Term] -> IO (Term, [Term])
 1677     parse at ts0 =
 1678       case at of
 1679         ArgTermVar ->
 1680           case ts0 of
 1681             x : ts1 -> return (x, ts1)
 1682             [] -> fail "reconstructArgTerm: too few function arguments"
 1683         ArgTermBVZero ->
 1684           do z <- scNat sc 0
 1685              x <- scBvNat sc z z
 1686              return (x, ts0)
 1687         ArgTermToIntMod n at1 ->
 1688           do n' <- scNat sc n
 1689              (x1, ts1) <- parse at1 ts0
 1690              x <- scToIntMod sc n' x1
 1691              pure (x, ts1)
 1692         ArgTermFromIntMod n at1 ->
 1693           do n' <- scNat sc n
 1694              (x1, ts1) <- parse at1 ts0
 1695              x <- scFromIntMod sc n' x1
 1696              pure (x, ts1)
 1697         ArgTermVector ty ats ->
 1698           do (xs, ts1) <- parseList ats ts0
 1699              x <- scVectorReduced sc ty xs
 1700              return (x, ts1)
 1701         ArgTermUnit ->
 1702           do x <- scUnitValue sc
 1703              return (x, ts0)
 1704         ArgTermPair at1 at2 ->
 1705           do (x1, ts1) <- parse at1 ts0
 1706              (x2, ts2) <- parse at2 ts1
 1707              x <- scPairValue sc x1 x2
 1708              return (x, ts2)
 1709         ArgTermRecord flds ->
 1710           do let (tags, ats) = unzip flds
 1711              (xs, ts1) <- parseList ats ts0
 1712              x <- scRecord sc (Map.fromList (zip tags xs))
 1713              return (x, ts1)
 1714         ArgTermConst x ->
 1715           do return (x, ts0)
 1716         ArgTermApply at1 at2 ->
 1717           do (x1, ts1) <- parse at1 ts0
 1718              (x2, ts2) <- parse at2 ts1
 1719              x <- scApply sc x1 x2
 1720              return (x, ts2)
 1721         ArgTermAt n ty at1 i ->
 1722           do n' <- scNat sc n
 1723              (x1, ts1) <- parse at1 ts0
 1724              i' <- scNat sc i
 1725              x <- scAt sc n' ty x1 i'
 1726              return (x, ts1)
 1727         ArgTermPairLeft at1 ->
 1728           do (x1, ts1) <- parse at1 ts0
 1729              x <- scPairLeft sc x1
 1730              return (x, ts1)
 1731         ArgTermPairRight at1 ->
 1732           do (x1, ts1) <- parse at1 ts0
 1733              x <- scPairRight sc x1
 1734              return (x, ts1)
 1735         ArgTermBVToNat w at1 ->
 1736           do (x1, ts1) <- parse at1 ts0
 1737              x <- scBvToNat sc w x1
 1738              pure (x, ts1)
 1739 
 1740     parseList :: [ArgTerm] -> [Term] -> IO ([Term], [Term])
 1741     parseList [] ts0 = return ([], ts0)
 1742     parseList (at : ats) ts0 =
 1743       do (x, ts1) <- parse at ts0
 1744          (xs, ts2) <- parseList ats ts1
 1745          return (x : xs, ts2)

 1747 -- | Given a type and value encoded as 'SValue's, construct an
 1748 -- 'ArgTerm' that builds a term of that type from local variables with
 1749 -- base types. The number of 'ArgTermVar' constructors should match
 1750 -- the number of arguments appended by 'applyUnintApp'.
 1751 mkArgTerm :: SharedContext -> TValue (What4 sym) -> SValue sym -> IO ArgTerm
 1752 mkArgTerm sc ty val =
 1753   case (ty, val) of
 1754     (VBoolType, VBool _) -> return ArgTermVar
 1755     (VIntType, VInt _)   -> return ArgTermVar
 1756     (_, VWord ZBV)       -> return ArgTermBVZero     -- 0-width bitvector is a constant
 1757     (_, VWord (DBV _))   -> return ArgTermVar
 1758     (_, VArray{})        -> return ArgTermVar
 1759     (VUnitType, VUnit)   -> return ArgTermUnit
 1760     (VIntModType n, VIntMod _ _) -> pure (ArgTermToIntMod n ArgTermVar)
 1761 
 1762     (VVecType _ ety, VVector vv) ->
 1763       do vs <- traverse force (V.toList vv)
 1764          xs <- traverse (mkArgTerm sc ety) vs
 1765          ety' <- termOfTValue sc ety
 1766          return (ArgTermVector ety' xs)
 1767 
 1768     (VPairType ty1 ty2, VPair v1 v2) ->
 1769       do x1 <- mkArgTerm sc ty1 =<< force v1
 1770          x2 <- mkArgTerm sc ty2 =<< force v2
 1771          return (ArgTermPair x1 x2)
 1772 
 1773     (VRecordType tys, VRecordValue flds) | map fst tys == map fst flds ->
 1774       do let tags = map fst tys
 1775          vs <- traverse (force . snd) flds
 1776          xs <- sequence [ mkArgTerm sc t v | (t, v) <- zip (map snd tys) vs ]
 1777          return (ArgTermRecord (zip tags xs))
 1778 
 1779     (_, VCtorApp i ps vv) ->
 1780       do ctor <- scRequireCtor sc (primName i)
 1781          ps' <- traverse (termOfSValue sc <=< force) ps
 1782          vv' <- traverse (termOfSValue sc <=< force) vv
 1783          x   <- scCtorAppParams sc (ctorPrimName ctor) ps' vv'
 1784          return (ArgTermConst x)
 1785 
 1786     (_, TValue tval) ->
 1787       do x <- termOfTValue sc tval
 1788          pure (ArgTermConst x)
 1789 
 1790     (_, VNat n) ->
 1791       do x <- scNat sc n
 1792          pure (ArgTermConst x)
 1793 
 1794     (_, VBVToNat w v) ->
 1795       do let w' = fromIntegral w -- FIXME: make w :: Natural to avoid fromIntegral
 1796          x <- mkArgTerm sc (VVecType w' VBoolType) v
 1797          pure (ArgTermBVToNat w' x)
 1798 
 1799     _ -> fail $ "could not create uninterpreted function argument of type " ++ show ty
 1800 
 1801 termOfTValue :: SharedContext -> TValue (What4 sym) -> IO Term
 1802 termOfTValue sc val =
 1803   case val of
 1804     VBoolType -> scBoolType sc
 1805     VIntType -> scIntegerType sc
 1806     VUnitType -> scUnitType sc
 1807     VVecType n a ->
 1808       do n' <- scNat sc n
 1809          a' <- termOfTValue sc a
 1810          scVecType sc n' a'
 1811     VPairType a b
 1812       -> do a' <- termOfTValue sc a
 1813             b' <- termOfTValue sc b
 1814             scPairType sc a' b'
 1815     VRecordType flds
 1816       -> do flds' <- traverse (traverse (termOfTValue sc)) flds
 1817             scRecordType sc flds'
 1818     _ -> fail $ "termOfTValue: " ++ show val
 1819 
 1820 termOfSValue :: SharedContext -> SValue sym -> IO Term
 1821 termOfSValue sc val =
 1822   case val of
 1823     VUnit -> scUnitValue sc
 1824     VNat n
 1825       -> scNat sc n
 1826     TValue tv -> termOfTValue sc tv
 1827     _ -> fail $ "termOfSValue: " ++ show val
