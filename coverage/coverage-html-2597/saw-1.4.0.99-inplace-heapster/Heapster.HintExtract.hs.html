   46 -- | The monad we use for extracting hints, which just has 'String' errors
   47 type ExtractM = Except String
   48 
   49 -- | Extract block hints from calls to @heapster.require@ in the Crucible CFG.
   50 extractHints ::
   51   forall ghosts args outs blocks init ret.
   52   PermEnv ->
   53   [L.Module] {- ^ The original source modules: used for finding constant values (i.e. spec strings) -} ->
   54   FunPerm ghosts args outs ret {- ^ The FunPerm corresponding to the CFG we are scanning -} ->
   55   CFG LLVM blocks init ret {- ^ The Crucible CFG for which to build the block hint map -} ->
   56   Either String (Ctx.Assignment (Constant (Maybe Hint)) blocks)
   57 extractHints env modules perm cfg =
   58   runExcept $ traverseFC extractHint (cfgBlockMap cfg)
   59   where
   60     globals =
   61       Map.fromList
   62         [ (globalSym g, str) | m <- modules,
   63                                g <- modGlobals m,
   64                                ValString chars <- maybeToList (globalValue g),
   65                                let str = chr . fromEnum <$> chars ]
   66 
   67     extractHint :: Block LLVM blocks ret ctx ->
   68                    ExtractM (Constant (Maybe Hint) ctx)
   69     extractHint block =
   70       extractBlockHints env globals (funPermTops perm) block >>= \case
   71       Just (SomeHintSpec ghosts valuePerms) ->
   72         return $ Constant $ Just (mkBlockEntryHint
   73                                   cfg
   74                                   (blockID block)
   75                                   (funPermTops perm)
   76                                   ghosts
   77                                   valuePerms)
   78       _ -> return $ Constant Nothing
   79 
   80 -- | Packs up the ghosts in a parsed hint permission spec
   81 data SomeHintSpec tops ctx where
   82   SomeHintSpec ::
   83     CruCtx ghosts ->
   84     MbValuePerms ((tops :++: CtxToRList ctx) :++: ghosts) ->
   85     SomeHintSpec tops ctx

   87 -- | Try to find a hint in a Block
   88 extractBlockHints ::
   89   forall blocks ret ctx tops.
   90   PermEnv ->
   91   Map.Map L.Symbol String {- ^ Globals map -} ->
   92   CruCtx tops {- ^ top context derived from current function's perm -} ->
   93   Block LLVM blocks ret ctx ->
   94   ExtractM (Maybe (SomeHintSpec tops ctx))
   95 extractBlockHints env globals tops block =
   96   extractStmtsHint who env globals tops inputs stmts
   97   where
   98     stmts = block ^. blockStmts
   99     inputs = blockInputs block
  100     who = show (blockID block)

  102 -- | Test if a sequence of statements starts with the Crucible representation of
  103 -- a call to the dummy function @heapster.require@
  104 extractStmtsHint ::
  105   forall blocks ret ctx tops.
  106   String ->
  107   PermEnv ->
  108   Map.Map L.Symbol String {- ^ globals -} ->
  109   CruCtx tops {- ^ top context derived from current function's perm -} ->
  110   CtxRepr ctx {- ^ block arguments -} ->
  111   StmtSeq LLVM blocks ret ctx ->
  112   ExtractM (Maybe (SomeHintSpec tops ctx))
  113 extractStmtsHint who env globals tops inputs = loop Ctx.zeroSize
  114   where
  115     loop ::
  116       forall rest.
  117       Ctx.Size rest ->
  118       StmtSeq LLVM blocks ret (ctx Ctx.<+> rest) ->
  119       ExtractM (Maybe (SomeHintSpec tops ctx))
  120     loop sz_rest s =
  121       extractHintFromSequence who env globals tops inputs sz_rest s >>= \case
  122       Just p -> return $ Just p
  123       _ | ConsStmt _ s' rest <- s ->
  124           let inc_rest :: forall tp. Ctx.Size (rest Ctx.::> tp)
  125               inc_rest = Ctx.incSize sz_rest in
  126           case s' of
  127             SetReg{} -> loop inc_rest rest
  128             ExtendAssign{} -> loop inc_rest rest
  129             CallHandle{} -> loop inc_rest rest
  130             Print{} -> loop sz_rest rest
  131             ReadGlobal{} -> loop inc_rest rest
  132             WriteGlobal{} -> loop sz_rest rest
  133             FreshConstant {} -> loop inc_rest rest
  134             FreshFloat {} -> loop inc_rest rest
  135             FreshNat {} -> loop inc_rest rest
  136             NewRefCell {} -> loop inc_rest rest
  137             NewEmptyRefCell {} -> loop inc_rest rest
  138             ReadRefCell {} -> loop inc_rest rest
  139             WriteRefCell {} -> loop sz_rest rest
  140             DropRefCell {} -> loop sz_rest rest
  141             Assert {} -> loop sz_rest rest
  142             Assume {} -> loop sz_rest rest
  143       _ -> return Nothing
  144 
  145 -- | Try to recognize the sequence of Crucible instructions leading up to
  146 -- a call to heapster.require. If found, build a hint by parsing the provided
  147 -- (global) ghost context string and spec string by looking them up
  148 -- in the global map.
  149 --
  150 -- Will throw an error if the @require@ is malformed (malformed spec strings
  151 -- or references out-of-scope values)
  152 extractHintFromSequence ::
  153   forall tops ctx rest blocks ret.
  154   String ->
  155   PermEnv ->
  156   Map.Map L.Symbol String {- ^ globals -} ->
  157   CruCtx tops {- ^ toplevel context -} ->
  158   CtxRepr ctx {- ^ block arguments -} ->
  159   Ctx.Size rest {- ^ keeps track of how deep we are into the current block -} ->
  160   StmtSeq LLVM blocks ret (ctx Ctx.<+> rest) ->
  161   ExtractM (Maybe (SomeHintSpec tops ctx))
  162 extractHintFromSequence who env globals tops blockIns sz s =
  163   case s of
  164     ConsStmt _ (ExtendAssign (LLVM_ResolveGlobal _ _ f))
  165       (ConsStmt _ (ExtendAssign (LLVM_ResolveGlobal _ _ ghosts))
  166         (ConsStmt _ (ExtendAssign (LLVM_ResolveGlobal _ _ spec))
  167           (ConsStmt _ (ExtendAssign (LLVM_LoadHandle _ _ ptr _ _))
  168             (ConsStmt _ (CallHandle _ fh _ actuals) _))))
  169           | globalSymbolName f == heapsterRequireName
  170           , Just Refl <- testEquality ptr fnPtrReg
  171           , Just Refl <- testEquality fh fnHdlReg
  172           , Just ghosts_str <- Map.lookup (fromString (globalSymbolName ghosts)) globals
  173           , Just spec_str <- Map.lookup (fromString (globalSymbolName spec)) globals ->
  174             -- The first two arguments are the ghost/spec strings.
  175             -- we can't "demote" their contexts to block args since they're globals
  176             -- and hence loaded in this block
  177             let (_, _, args) = expectLengthAtLeastTwo $ toListFC Some actuals in
  178             -- "demote" the context of each reg to the block input context,
  179             -- proving that each arg is in fact defined in a previous block
  180             -- (and is thus valid for use in this spec)
  181             case sequence (toBlockArg (Ctx.size blockIns) sizeAtCall <$> args) of
  182               Just demoted ->
  183                 Just <$> requireArgsToHint who env blockIns tops demoted ghosts_str spec_str
  184               Nothing ->
  185                 throwError (who ++ ": spec refers to block-defined expressions")
  186 
  187     _ -> return Nothing
  188 
  189   where
  190     fnPtrReg :: forall a b tp. Reg (ctx Ctx.<+> rest Ctx.::> tp Ctx.::> a Ctx.::> b) tp
  191     fnPtrReg = Reg (Ctx.skipIndex (Ctx.skipIndex (Ctx.nextIndex (Ctx.addSize (Ctx.size blockIns) sz))))
  192 
  193     fnHdlReg :: forall a b c tp. Reg ((ctx Ctx.<+> rest) Ctx.::> a Ctx.::> b Ctx.::> c Ctx.::> tp) tp
  194     fnHdlReg = Reg (Ctx.lastIndex (Ctx.addSize (Ctx.size blockIns) sizeAtCall))
  195 
  196     sizeAtCall :: forall a b c d. Ctx.Size (rest Ctx.::> a Ctx.::> b Ctx.::> c Ctx.::> d)
  197     sizeAtCall = Ctx.incSize (Ctx.incSize (Ctx.incSize (Ctx.incSize sz)))
  198 
  199 -- | Assemble a Hint
  200 --
  201 -- Will throw an error if the @require@ is malformed (malformed spec strings
  202 -- or references out-of-scope values)
  203 requireArgsToHint ::
  204   String {-^ A string representing the block in which this call appears (for errors) -} ->
  205   PermEnv ->
  206   CtxRepr ctx {-^ The block arguments -}  ->
  207   CruCtx tops {-^ Toplevel arguments -} ->
  208   [Some (Reg ctx)] {-^ The actual arguments to the require, demoted to block args -} ->
  209   String {-^ The ghost ctx to parse -} ->
  210   String {-^ The permissions to parse -} ->
  211   ExtractM (SomeHintSpec tops ctx)
  212 requireArgsToHint who env blockIns tops args ghostString specString =
  213   case parseParsedCtxString who env ghostString of
  214     Just (Some ghost_ctx) ->
  215       let full_ctx = appendParsedCtx (appendParsedCtx top_ctx ctx_rename) ghost_ctx
  216           sub = buildHintSub blockIns args
  217           ctx = mkArgsParsedCtx (mkCruCtx blockIns)
  218           top_ctx = mkTopParsedCtx tops
  219           ctx_rename = renameParsedCtx sub ctx
  220       in maybe (throwError (who ++ ": error parsing permissions"))
  221                (return . SomeHintSpec (parsedCtxCtx ghost_ctx))
  222                (parsePermsString who env full_ctx specString)
  223     Nothing ->
  224       throwError (who ++ ": error parsing ghost context")
  225 
  226 -- | Apply a substitution to the names in a ParsedCtx
  227 renameParsedCtx :: [(String, String)] -> ParsedCtx ctx -> ParsedCtx ctx
  228 renameParsedCtx sub ctx = ctx { parsedCtxNames = renamed }
  229   where
  230     renamed = mapRAssign (\(Constant x) ->
  231                            Constant (substNames x)) (parsedCtxNames ctx)
  232     substNames x = fromMaybe x (lookup x sub)
  233 
  234 -- | Build a susbstitution to apply to block arguments based on the actual
  235 -- arguments provided to a @requires@ call, i.e. given
  236 -- @heapster.require(..., ..., %11, %50)@
  237 -- if @%11@ corresponds to block argument 1 and @%50@ to block argument 0,
  238 -- with block arg 2 unused, then return the substitution
  239 -- @[("arg1", "arg0"), ("arg1, arg0"), ("arg2", "arg2")]@
  240 buildHintSub ::
  241   forall block_args.
  242   CtxRepr block_args ->
  243   [Some (Reg block_args)] ->
  244   [(String, String)]
  245 buildHintSub blockArgs args = usedSub
  246   where
  247     argNames = someRegName <$> args
  248     unusedNames = argNamei <$> [length argNames .. (Ctx.sizeInt (Ctx.size blockArgs))]
  249     usedSub  = [ (a, argNamei i) | i <- [0..] | a <- argNames ++ unusedNames ]
  250 
  251 -- | Check if we can use a register in a smaller context, and
  252 -- return the register indexed by the new context if so.
  253 toBlockArg ::
  254   Ctx.Size block_args ->
  255   Ctx.Size rest ->
  256   Some (Reg (block_args Ctx.<+> rest)) ->
  257   Maybe (Some (Reg block_args))
  258 toBlockArg argsSz _restSz reg =
  259   case reg of
  260     Some (Reg idx) ->
  261       do Some idx' <- Ctx.intIndex (Ctx.indexVal idx) argsSz
  262          pure $ Some $ Reg idx'

  264 -- | Constructor for block entry hints
  265 mkBlockEntryHint ::
  266   forall blocks init ret top_args ghosts args.
  267   CFG LLVM blocks init ret ->
  268   BlockID blocks args ->
  269   CruCtx top_args ->
  270   CruCtx ghosts ->
  271   MbValuePerms ((top_args :++: CtxToRList args) :++: ghosts) ->
  272   Hint
  273 mkBlockEntryHint cfg blockId tops ghosts valPerms  =
  274   Hint_Block $ BlockHint h blocks blockId entryHint
  275   where
  276     entryHint = BlockEntryHintSort tops ghosts valPerms
  277     h = cfgHandle cfg
  278     blocks = fmapFC blockInputs $ cfgBlockMap cfg
  279 
  280 -- | Like mkArgParsedContext, but with all of the names
  281 -- set to \"topi\" instead of \"argi\"
  282 mkTopParsedCtx :: CruCtx ctx -> ParsedCtx ctx
  283 mkTopParsedCtx = mkPrefixParsedCtx "top"

  285 someRegName :: Some (Reg ctx) -> String
  286 someRegName (Some (Reg i)) = argNamei (Ctx.indexVal i)

  288 argNamei :: Int -> String
  289 argNamei i = "arg" ++ show i
