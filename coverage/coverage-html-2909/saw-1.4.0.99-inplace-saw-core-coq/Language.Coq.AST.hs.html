   10 module Language.Coq.AST where
   11 
   12 import Data.String (IsString(..))
   13 
   14 -- | An 'Ident' is a Coq qualified identifier represented as a string,
   15 -- with the invariant that it is lexically valid.
   16 -- A valid Coq identifier is a sequence of letters, digits,
   17 -- underscores and primes that starts with a letter or underscore.
   18 -- A /qualified/ identifier is a sequence of one or more identifiers
   19 -- separated by periods.
   20 newtype Ident = Ident String
   21   deriving (Eq, Ord)
   22 
   23 instance Show Ident where
   24   show (Ident s) = show s
   25 
   26 instance IsString Ident where
   27   fromString s = Ident s
   28 
   29 data Sort
   30   = Prop
   31   | Set
   32   | Type
   33   deriving (Show)
   34 
   35 data Term
   36   = Lambda [Binder] Term
   37   | Fix Ident [Binder] Term Term
   38   | Pi [PiBinder] Term
   39   | Let Ident [Binder] (Maybe Type) Term Term
   40   | If Term Term Term
   41   | App Term [Term]
   42   | Sort Sort
   43   | Var Ident
   44     -- | A variable that needs to be printed with a leading at sign in order to
   45     -- make all arguments explicit
   46   | ExplVar Ident
   47     -- | A ascription @tm : tp@ of a type to a term
   48   | Ascription Term Term
   49   | NatLit Integer
   50   | ZLit Integer
   51   | List [Term]
   52   | StringLit String
   53   | Scope Term String
   54   | Ltac String
   55   deriving (Show)
   56 
   57 -- | Type synonym useful for indicating when a term is used as a type.
   58 type Type = Term
   59 
   60 -- | Is this an implicit or explicit binder?
   61 type IsImplicit = Bool
   62 
   63 -- | An 'Ident' with an optional 'Type', which may be explicit or implicit.
   64 -- For use representing the bound variables in 'Lambda's, 'Let's, etc.
   65 data Binder
   66   = Binder IsImplicit Ident (Maybe Type)
   67     deriving (Show)
   68 
   69 -- | An 'Type' with an optional 'Ident', which may be explicit or implicit.
   70 -- For use representing arguments in 'Pi' types.
   71 data PiBinder
   72   = PiBinder IsImplicit (Maybe Ident) Type
   73     deriving (Show)
   74 
   75 -- Because saw-core does not give very helpful access to the parameters and
   76 -- indices, we just follow their style and define the constructor by its fully
   77 -- applied return type.
   78 data Constructor = Constructor
   79   { constructorName    :: Ident
   80   -- ^ NOTE: The constructor name must be an /unqualified/ identifier.
   81   , constructorType    :: Term
   82   }
   83   deriving (Show)
   84 
   85 data Inductive = Inductive
   86   { inductiveName         :: Ident
   87   , inductiveParameters   :: [Binder]
   88   , inductiveIndices      :: [PiBinder]
   89   , inductiveSort         :: Sort
   90   , inductiveConstructors :: [Constructor]
   91   }
   92   deriving (Show)
   93 
   94 data Decl
   95   = Axiom Ident Type
   96   | Comment String
   97   | Definition Ident [Binder] (Maybe Type) Term
   98   | Parameter Ident Type
   99   | Variable Ident Type
  100   | InductiveDecl Inductive
  101   | Section Ident [Decl]
  102   | Snippet String
  103   deriving (Show)
