    3 -- |
    4 -- Module      : SAWSupport.ScopedMap
    5 -- Description : Scoped-aware finite map library
    6 -- License     : BSD3
    7 -- Maintainer  : saw@galois.com
    8 -- Stability   : provisional
    9 --
   10 -- Scope-aware map for use as an interpreter or typechecker
   11 -- environment.
   12 --
   13 -- The `ScopedMap` type is a stack of maps; each layer corresponds
   14 -- to a scope as the client moves into and out of nested naming
   15 -- environments.
   16 --
   17 -- The current (innermost) scope is at the top of the stack / head of
   18 -- the list.
   19 --
   20 -- Update operations act on the current scope. Lookup operations
   21 -- inspect the whole stack, starting at the innermost layer and
   22 -- working out.
   23 --
   24 -- The `push` and `pop` operations add and drop scopes respectively,
   25 -- and should be used when the client enters and leaves a nested
   26 -- naming environment.
   27 --
   28 -- The `flatten` operation produces a single-layer Data.Map that
   29 -- contains all the visible mappings. (That is, duplicate keys in
   30 -- outer scopes are suppressed.)
   31 --
   32 -- Other operations correspond to the ordinary operations on
   33 -- Data.Map. Operations that generate collections (e.g. @keys@) come
   34 -- in two versions: one that preserves the scope structure
   35 -- (generally called @scoped@), and one that flattens it (generally
   36 -- called @all@).
   37 --
   38 -- In principle we should provide all the operations on Data.Map.
   39 -- However, the implementation has been lazily evaluated and the
   40 -- operations present are the ones that have been needed so far.
   41 -- Add more as desired.
   42 
   43 module SAWSupport.ScopedMap (
   44     ScopedMap, -- opaque
   45     push,
   46     pop,
   47     empty,

   56   ) where

   71 -- | The type of scoped maps.
   72 --
   73 --   The list of scopes is a `NonEmpty` to avoid unnecessary
   74 --   degenerate cases.
   75 newtype ScopedMap k v = ScopedMap (NonEmpty (Map k v))
   76 
   77 -- | Push a new scope.
   78 push :: Ord k => ScopedMap k v -> ScopedMap k v
   79 push (ScopedMap scopes) =
   80     ScopedMap $ N.cons Map.empty scopes
   81 
   82 -- | Pop and discard the most recent scope.
   83 --
   84 --   Popping the last scope is an error and will panic.
   85 pop :: Ord k => ScopedMap k v -> ScopedMap k v
   86 pop (ScopedMap (_ :| more)) = case more of
   87     [] ->
   88         panic "pop" ["Popping topmost scope"]
   89     s : more' ->
   90         ScopedMap (s :| more')
   91 
   92 -- | An empty ScopedMap, with a single empty scope.
   93 empty :: Ord k => ScopedMap k v
   94 empty =
   95     ScopedMap (Map.empty :| [])
   96 
   97 -- | Initialize a ScopedMap from a Map, which becomes the current (and
   98 --   only) scope.
   99 seed :: Ord k => Map k v -> ScopedMap k v
  100 seed m =
  101     ScopedMap (m :| [])
  102 
  103 -- | Insert into a ScopedMap. Always inserts in the innermost (most
  104 --   recent) scope.
  105 insert :: Ord k => k -> v -> ScopedMap k v -> ScopedMap k v
  106 insert k v (ScopedMap (s :| more)) =
  107     let s' = Map.insert k v s in
  108     ScopedMap (s' :| more)
  109 
  110 -- | Look up a key in a ScopedMap. Checks all the scopes with the most
  111 --   recent first.
  112 lookup :: Ord k => k -> ScopedMap k v -> Maybe v
  113 lookup k (ScopedMap (s0 :| more0)) =
  114     let visit s more = case Map.lookup k s of
  115           Just result -> Just result
  116           Nothing -> case more of
  117               [] -> Nothing
  118               s' : more' -> visit s' more'
  119     in
  120     visit s0 more0
  121 
  122 -- | Drop entries that don't match a predicate.
  123 filter :: Ord k => (v -> Bool) -> ScopedMap k v -> ScopedMap k v
  124 filter keep (ScopedMap (s :| more)) =
  125     ScopedMap (Map.filter keep s :| map (Map.filter keep) more)
  126 
  127 -- | Return Map.assocs for each scope, preserving the scope boundaries.
  128 --   The head of the returned list is the most recent scope.
  129 scopedAssocs :: Ord k => ScopedMap k v -> [[(k, v)]]
  130 scopedAssocs (ScopedMap scopes) =
  131     map Map.assocs $ toList scopes
  132

  135 -- | Convert to a plain map by folding the scopes together. Duplicate
  136 --   bindings in more recent scopes take priority.
  137 flatten :: Ord k => ScopedMap k v -> Map k v
  138 flatten (ScopedMap (s0 :| more0)) =
  139     let visit s more =
  140           let result = case more of
  141                 [] -> Map.empty
  142                 s' : more' -> visit s' more'
  143           in
  144           -- Map.union gives its left argument priority given a
  145           -- duplicate key.
  146           Map.union s result
  147     in
  148     visit s0 more0
  149

  152 -- | Return all keys, ignoring scope.
  153 allKeys :: Ord k => ScopedMap k v -> [k]
  154 allKeys m =
  155     Map.keys $ flatten m
  156 
  157 -- | Return all keys as a set, ignoring scope.
  158 allKeysSet :: Ord k => ScopedMap k v -> Set k
  159 allKeysSet m =
  160     Map.keysSet $ flatten m
