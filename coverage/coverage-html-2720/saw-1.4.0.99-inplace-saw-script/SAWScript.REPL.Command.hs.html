   85 -- | Commands.
   86 data Command
   87   = Command (REPL ())         -- ^ Successfully parsed command
   88   | Ambiguous String [String] -- ^ Ambiguous command, list of conflicting
   89                               --   commands
   90   | Unknown String            -- ^ The unknown command

   92 -- | Command builder.
   93 data CommandDescr = CommandDescr
   94   { cName :: String
   95   , cAliases :: [String]
   96   , cBody :: CommandBody
   97   , cHelp :: String
   98   }
   99 
  100 instance Show CommandDescr where
  101   show = cName
  102 
  103 instance Eq CommandDescr where
  104   (==) = (==) `on` cName
  105 
  106 instance Ord CommandDescr where
  107   compare = compare `on` cName
  108 
  109 data CommandBody
  110   = ExprArg     (String   -> REPL ())
  111   | TypeArg     (String   -> REPL ())
  112   | FilenameArg (FilePath -> REPL ())
  113   | ShellArg    (String   -> REPL ())
  114   | NoArg       (REPL ())
  115

  117 -- | Convert the command list to a Trie, expanding aliases.
  118 makeCommands :: [CommandDescr] -> CommandMap
  119 makeCommands list  = foldl insert emptyTrie (concatMap expandAliases list)
  120   where
  121   insert m (name, d) = insertTrie name d m
  122   expandAliases :: CommandDescr -> [(String, CommandDescr)]
  123   expandAliases d = (cName d, d) : zip (cAliases d) (repeat d)

  161 genHelp :: [CommandDescr] -> [String]
  162 genHelp cs = map cmdHelp cs
  163   where
  164   cmdHelp cmd = concat [ "  ", cName cmd, pad (cName cmd), cHelp cmd ]
  165   padding     = 2 + maximum (map (length . cName) cs)
  166   pad n       = replicate (max 0 (padding - length n)) ' '

  171 -- | Run a command.
  172 runCommand :: Command -> REPL ()
  173 runCommand c = case c of
  174 
  175   Command cmd -> exceptionProtect cmd
  176 
  177   Unknown cmd -> io (putStrLn ("Unknown command: " ++ cmd))
  178 
  179   Ambiguous cmd cmds -> io $ do
  180     putStrLn (cmd ++ " is ambiguous, it could mean one of:")
  181     putStrLn ("\t" ++ intercalate ", " cmds)

  184 lexSAW :: String -> Text.Text -> REPL [Token Pos]
  185 lexSAW fileName str = do
  186   -- XXX wrap printing of positions in the message-printing infrastructure
  187   case SAWScript.Lexer.lexSAW fileName str of
  188     Left (_, pos, msg) ->
  189          fail $ show pos ++ ": " ++ Text.unpack msg
  190     Right (tokens', Nothing) -> pure tokens'
  191     Right (_, Just (SAWCentral.Options.Error, pos, msg)) ->
  192          fail $ show pos ++ ": " ++ Text.unpack msg
  193     Right (tokens', Just _) -> pure tokens'

  195 typeOfCmd :: String -> REPL ()
  196 typeOfCmd str
  197   | null str = do io $ putStrLn "[error] :type requires an argument"
  198   | otherwise =
  199   do tokens <- lexSAW replFileName (Text.pack str)
  200      expr <- case SAWScript.Parser.parseExpression tokens of
  201        Left err -> fail (show err)
  202        Right expr -> return expr
  203      let pos = getPos expr
  204          decl = SS.Decl pos (SS.PWild pos Nothing) Nothing expr
  205      rw <- getTopLevelRW
  206      decl' <- do
  207        let primsAvail = rwPrimsAvail rw
  208            -- XXX it should not be necessary to do this munging
  209            Environ varenv tyenv _cryenvs = rwEnviron rw
  210            squash (defpos, lc, ty, _val, _doc) = (defpos, lc, ReadOnlyVar, ty)
  211            varenv' = Map.map squash $ ScopedMap.flatten varenv
  212            tyenv' = ScopedMap.flatten tyenv
  213            rbenv = rwRebindables rw
  214            rbsquash (defpos, ty, _val) = (defpos, Current, RebindableVar, ty)
  215            rbenv' = Map.map rbsquash rbenv
  216            varenv'' = Map.union varenv' rbenv'
  217        let (errs_or_results, warns) = checkDecl primsAvail varenv'' tyenv' decl
  218        let issueWarning (msgpos, msg) =
  219              -- XXX the print functions should be what knows how to show positions...
  220              putStrLn (show msgpos ++ ": Warning: " ++ msg)
  221        io $ mapM_ issueWarning warns
  222        either failTypecheck return errs_or_results
  223      let ~(SS.Decl _pos _ (Just schema) _expr') = decl'
  224      io $ TextIO.putStrLn $ PPS.pShowText schema

  226 searchCmd :: String -> REPL ()
  227 searchCmd str
  228   | null str = do io $ putStrLn $ "[error] :search requires at least one argument"
  229   | otherwise =
  230   do tokens <- lexSAW replFileName (Text.pack str)
  231      pat <- case SAWScript.Parser.parseSchemaPattern tokens of
  232        Left err -> fail (show err)
  233        Right pat -> return pat
  234      rw <- getTopLevelRW
  235 
  236      -- Always search the entire environment and recognize all type
  237      -- names in the user's pattern, regardless of whether
  238      -- enable_experimental or enable_deprecated is in effect. It is
  239      -- definitely confusing to search for a hidden type and have
  240      -- search treat that as a free type variable and match all kinds
  241      -- of things you didn't intend. It is also better to retrieve
  242      -- invisible/deprecated items and report their existence than to
  243      -- hide them from the search.
  244 
  245      -- FUTURE: it would be nice to be able to use the words
  246      -- "experimental" and "deprecated" in the search term to match
  247      -- against the lifecycle, to allow doing stuff like searching
  248      -- for deprecated functions that take Terms.
  249 
  250      let primsAvail = rwPrimsAvail rw
  251          -- XXX it should not be necessary to do this munging
  252          Environ varenv tyenv _cryenv = rwEnviron rw
  253          rbenv = rwRebindables rw
  254          squash (pos, lc, ty, _val, _doc) = (pos, lc, ReadOnlyVar, ty)
  255          varenv' = Map.map squash $ ScopedMap.flatten varenv
  256          tyenv' = ScopedMap.flatten tyenv
  257          rbsquash (pos, ty, _val) = (pos, Current, RebindableVar, ty)
  258          rbenv' = Map.map rbsquash rbenv
  259          varenv'' = Map.union varenv' rbenv'
  260          (errs_or_results, warns) = checkSchemaPattern everythingAvailable varenv'' tyenv' pat
  261      let issueWarning (msgpos, msg) =
  262            -- XXX the print functions should be what knows how to show positions...
  263            putStrLn (show msgpos ++ ": Warning: " ++ msg)
  264      io $ mapM_ issueWarning warns
  265      pat' <- either failTypecheck return $ errs_or_results
  266      let search = compileSearchPattern tyenv pat'
  267          keep (_pos, _lc, _rb, ty) = matchSearchPattern search ty
  268          allMatches = Map.filter keep varenv'
  269 
  270          -- Divide the results into visible, experimental-not-visible,
  271          -- and deprecated-not-visible.
  272          inspect ::
  273              SS.Name ->
  274              (SS.Pos, PrimitiveLifecycle, SS.Rebindable, SS.Schema) ->
  275              (Map.Map SS.Name (PrimitiveLifecycle, SS.Schema),
  276               Map.Map SS.Name (PrimitiveLifecycle, SS.Schema),
  277               Map.Map SS.Name (PrimitiveLifecycle, SS.Schema)) ->
  278              (Map.Map SS.Name (PrimitiveLifecycle, SS.Schema),
  279               Map.Map SS.Name (PrimitiveLifecycle, SS.Schema),
  280               Map.Map SS.Name (PrimitiveLifecycle, SS.Schema))
  281          inspect name (_pos, lc, _rb, ty) (vis, ex, dep) =
  282              if Set.member lc primsAvail then
  283                  (Map.insert name (lc, ty) vis, ex, dep)
  284              else case lc of
  285                Current -> oops
  286                WarnDeprecated -> oops
  287                HideDeprecated ->
  288                  (vis, ex, Map.insert name (lc, ty) dep)
  289                Experimental ->
  290                  (vis, Map.insert name (lc, ty) ex, dep)
  291              where
  292                oops =
  293                  panic "searchCmd" [
  294                      "Found non-visible object " <> Text.pack (show name) <>
  295                      " with unexpected lifecycle " <> Text.pack (show lc)
  296                  ]
  297          (visMatches, expMatches, depMatches) =
  298              Map.foldrWithKey inspect (Map.empty, Map.empty, Map.empty) allMatches
  299 
  300          printMatch (name, (lc, ty)) = do
  301            let ty' = PPS.pShowText ty
  302                lc' = case lc of
  303                    Current -> ""
  304                    WarnDeprecated -> "  (DEPRECATED AND WILL WARN)"
  305                    HideDeprecated -> "  (DEPRECATED AND UNAVAILABLE BY DEFAULT)"
  306                    Experimental -> "  (EXPERIMENTAL)"
  307            TextIO.putStrLn (name <> " : " <> ty' <> lc')
  308          printMatches matches =
  309            io $ mapM_ printMatch (Map.assocs matches)
  310 
  311          moreMatches matches =
  312              let n = Map.size matches in
  313              if n == 1 then "1 more match"
  314              else show n ++ " more matches"
  315          alsoExperimental =
  316              if not (Map.null expMatches) then
  317                  io $ putStrLn $ moreMatches expMatches ++ " tagged " ++
  318                                  "experimental; use enable_experimental to " ++
  319                                  "see them"
  320              else
  321                  pure ()
  322          alsoDeprecated =
  323              if not (Map.null depMatches) then
  324                  io $ putStrLn $ moreMatches depMatches ++ " tagged " ++
  325                                  "deprecated; use enable_deprecated to " ++
  326                                  "see them"
  327              else
  328                  pure ()
  329 
  330      if not (Map.null visMatches) then do
  331          printMatches visMatches
  332          alsoExperimental
  333          alsoDeprecated
  334      else do
  335          io $ putStrLn "No matches."
  336          if not (Map.null expMatches) then do
  337              io $ putStrLn $ "The following experimental matches require " ++
  338                              "enable_experimental:"
  339              printMatches expMatches
  340              alsoDeprecated
  341          else if not (Map.null depMatches) then do
  342              io $ putStrLn $ "The following deprecated matches require " ++
  343                              "enable_deprecated:"
  344              printMatches depMatches
  345          else
  346              pure ()

  353 envCmd :: REPL ()
  354 envCmd = do
  355   rw <- getTopLevelRW
  356   let Environ varenv _tyenv _cryenv = rwEnviron rw
  357       rbenv = rwRebindables rw
  358       avail = rwPrimsAvail rw
  359 
  360   -- print the rebindable globals first, if any
  361   unless (Map.null rbenv) $ do
  362       let rbsay (x, (_pos, ty, _v)) = do
  363               let ty' = PPS.pShowText ty
  364               TextIO.putStrLn (x <> " : rebindable " <> ty')
  365       io $ mapM_ rbsay $ Map.assocs rbenv
  366       io $ TextIO.putStrLn ""
  367 
  368   -- print the normal environment
  369   let say (x, (_pos, _lc, ty, _v, _doc)) = do
  370           let ty' = PPS.pShowText ty
  371           TextIO.putStrLn (x <> " : " <> ty')
  372       -- Print only the visible objects
  373       keep (_x, (_pos, lc, _ty, _v, _doc)) = Set.member lc avail
  374       -- Insert a blank line in the output where there's a scope boundary
  375       printScope mItems = case mItems of
  376           Nothing -> TextIO.putStrLn ""
  377           Just items -> mapM_ say $ filter keep items
  378       -- Reverse the list of scopes so the innermost prints last,
  379       -- because that's what people will expect to see.
  380       itemses = reverse $ ScopedMap.scopedAssocs varenv
  381   io $ mapM_ printScope $ intersperse Nothing $ map Just itemses

  402 helpCmd :: String -> REPL ()
  403 helpCmd cmd
  404   | null cmd = io (mapM_ putStrLn (genHelp commandList))
  405   | otherwise =
  406     do rw <- getTopLevelRW
  407        -- Note: there's no rebindable builtins and thus no way to
  408        -- attach help text to anything rebindable, so we can ignore
  409        -- the rebindables.
  410        let Environ varenv _tyenv _cryenvs = rwEnviron rw
  411        case ScopedMap.lookup (Text.pack cmd) varenv of
  412          Just (_pos, _lc, _ty, _v, Just doc) ->
  413            io $ mapM_ TextIO.putStrLn doc
  414          Just (_pos, _lc, _ty, _v, Nothing) -> do
  415            io $ putStrLn $ "// No documentation is available."
  416            typeOfCmd cmd
  417          Nothing ->
  418            io $ putStrLn "// No such command."

  435 -- SAWScript commands ----------------------------------------------------------

  437 {- Evaluation is fairly straightforward; however, there are a few important
  438 caveats:
  439 
  440   1. 'return' is type-polymorphic.  This means that 'return "foo"' will produce
  441      a type-polymorphic function 'm -> m String', for any monad 'm'.  It also
  442      means that if you type 'return "foo"' into a naively-written interpreter,
  443      you won't see 'foo'!  The solution is to force each statement that comes
  444      into the REPL to have type 'TopLevel t' ('TopLevel' is the SAWScript
  445      version of 'IO').  This gets done as soon as the statement is parsed.
  446 
  447   2. Handling binding variables to values is somewhat tricky.  When you're
  448      allowed to bind variables in the REPL, you're basically working in the
  449      context of a partially evaluated module: all the results of all the old
  450      computations are in scope, and you're allowed to add new computations that
  451      depend on them.  It's insufficient to merely hang onto the AST for the
  452      old computations, as that could cause them to be evaluated multiple times;
  453      it could also cause their type to be inferred differently several times,
  454      which is bad.  Instead, we hang onto the inferred types of previous
  455      computations and use them to seed the name resolver and the typechecker;
  456      we also hang onto the results and use them to seed the interpreter. -}
  457 sawScriptCmd :: String -> REPL ()
  458 sawScriptCmd str = do
  459   tokens <- lexSAW replFileName (Text.pack str)
  460   case SAWScript.Parser.parseStmtSemi tokens of
  461     Left err -> io $ print err
  462     Right stmt ->
  463       do mr <- getProofStateRef
  464          case mr of
  465            Nothing -> void $ liftTopLevel (interpretTopStmt True stmt)
  466            Just r  -> void $ liftProofScript (interpretTopStmt True stmt) r
  467

  469 replFileName = "<stdin>"

  491 -- | Strip leading space.
  492 sanitize :: String -> String
  493 sanitize  = dropWhile isSpace
  494 
  495 -- | Strip trailing space.
  496 sanitizeEnd :: String -> String
  497 sanitizeEnd = reverse . sanitize . reverse
  498 
  499 -- | Split at the first word boundary.
  500 splitCommand :: String -> Maybe (String,String)
  501 splitCommand txt =
  502   case sanitize txt of
  503     ':' : more
  504       | (as,bs) <- span (\x -> isPunctuation x || isSymbol x) more
  505       , not (null as) -> Just (':' : as, sanitize bs)
  506 
  507       | (as,bs) <- break isSpace more
  508       , not (null as) -> Just (':' : as, sanitize bs)
  509 
  510       | otherwise -> Nothing
  511 
  512     expr -> guard (not (null expr)) >> return (expr,[])
  513 
  514 -- | Look up a string in a command list. If given a string that's both
  515 -- itself a command and a prefix of something else, choose that
  516 -- command; otherwise such commands are inaccessible. Also, deduplicate
  517 -- the list of results to avoid silliness with command aliases.
  518 findSomeCommand :: String -> CommandMap -> [CommandDescr]
  519 findSomeCommand str commandTable = nub $ lookupTrieWithExact str commandTable
  520 
  521 -- | Look up a string in the command list.
  522 findCommand :: String -> [CommandDescr]
  523 findCommand str = findSomeCommand str commands
  524 
  525 -- | Look up a string in the notebook-safe command list.
  526 findNbCommand :: String -> [CommandDescr]
  527 findNbCommand str = findSomeCommand str nbCommands
  528 
  529 -- | Parse a line as a command.
  530 parseCommand :: (String -> [CommandDescr]) -> String -> Maybe Command
  531 parseCommand findCmd line = do
  532   (cmd,args) <- splitCommand line
  533   let args' = sanitizeEnd args
  534   case findCmd cmd of
  535     -- nothing matched; if it doesn't begin with a colon, eval it
  536     [] -> case cmd of
  537       []      -> Nothing
  538       ':' : _ -> Just (Unknown cmd)
  539       _       -> Just (Command (sawScriptCmd line))
  540 
  541     -- matched exactly one command; run it
  542     [c] -> case cBody c of
  543       ExprArg     body -> Just (Command (body args'))
  544       TypeArg     body -> Just (Command (body args'))
  545       FilenameArg body -> Just (Command (body =<< expandHome args'))
  546       ShellArg    body -> Just (Command (body args'))
  547       NoArg       body -> Just (Command  body)
  548 
  549     -- matched several things; complain
  550     cs -> Just (Ambiguous cmd (map cName cs))
  551 
  552   where
  553   expandHome path =
  554     case path of
  555       '~' : c : more | isPathSeparator c -> do dir <- io getHomeDirectory
  556                                                return (dir </> more)
  557       _ -> return path
