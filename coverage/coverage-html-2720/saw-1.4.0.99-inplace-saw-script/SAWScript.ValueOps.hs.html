   76 isVUnit :: Value -> Bool
   77 isVUnit (VTuple []) = True
   78 isVUnit _ = False

   80 indexValue :: SS.Pos -> Value -> Value -> Value
   81 indexValue pos (VArray vs) (VInteger x)
   82     | i < length vs = vs !! i
   83     | otherwise = error $ show pos ++ ": Array index out of bounds"
   84     where i = fromInteger x
   85 indexValue pos v1 v2 =
   86     panic "indexValue" [
   87         "Type error that escaped the typechecker",
   88         "Source position: " <> Text.pack (show pos),
   89         "Array value: " <> Text.pack (show v1),
   90         "Index value: " <> Text.pack (show v2)
   91     ]
   92 
   93 lookupValue :: SS.Pos -> Value -> Text -> Value
   94 lookupValue pos (VRecord vm) name =
   95     case M.lookup name vm of
   96       Nothing -> error $ show pos ++ ": No such record field: " ++ Text.unpack name
   97       Just x -> x
   98 lookupValue pos v1 v2 =
   99     panic "lookupValue" [
  100         "Type error that escaped the typechecker",
  101         "Source position: " <> Text.pack (show pos),
  102         "Array value: " <> Text.pack (show v1),
  103         "Index value: " <> Text.pack (show v2)
  104     ]

  106 tupleLookupValue :: SS.Pos -> Value -> Integer -> Value
  107 tupleLookupValue pos (VTuple vs) i
  108   | 0 <= i && fromIntegral i < length vs = vs !! fromIntegral i
  109   | otherwise = error $ show pos ++ ": No such tuple index: " ++ show i
  110 tupleLookupValue pos v1 v2 =
  111     panic "tupleLookupValue" [

  113         "Source position: " <> Text.pack (show pos),
  114         "Array value: " <> Text.pack (show v1),

  118 -- | A version of 'Control.Exception.bracket' specialized to 'TopLevel'. We
  119 -- can't use 'Control.Monad.Catch.bracket' because it requires 'TopLevel' to
  120 -- implement 'Control.Monad.Catch.MonadMask', which it can't do.
  121 bracketTopLevel :: TopLevel a -> (a -> TopLevel b) -> (a -> TopLevel c) -> TopLevel c
  122 bracketTopLevel acquire release action =
  123   do  resource <- acquire
  124       try (action resource) >>= \case
  125         Left (bad :: X.SomeException) -> release resource >> throwM bad
  126         Right good -> release resource >> pure good

  128 -- Note: this is used only by restoreCheckpoint and restoreCheckpoint is
  129 -- not really expected to work.
  130 --
  131 -- This is good, because whatever this function is doing does not seem
  132 -- to make a great deal of sense. (Which has become more overt with recent
  133 -- reorgs and cleanup, but hasn't itself changed.)
  134 combineRW :: TopLevelCheckpoint -> TopLevelRW -> IO TopLevelRW
  135 combineRW (TopLevelCheckpoint chk scc) rw = do
  136      let CryptolEnvStack chk'cenv chk'cenvs = rwGetCryptolEnvStack chk
  137          CryptolEnvStack rw'cenv rw'cenvs = rwGetCryptolEnvStack rw
  138      -- Caution: this merge may have unexpected results if the
  139      -- number of scopes doesn't match. But, it doesn't make sense
  140      -- to do that in the first place. Caveat emptor...
  141      cenv' <- CEnv.combineCryptolEnv chk'cenv rw'cenv
  142      cenvs' <- zipWithM CEnv.combineCryptolEnv chk'cenvs rw'cenvs
  143      sc' <- restoreSharedContext scc (rwSharedContext rw)
  144      let cryenv' = CryptolEnvStack cenv' cenvs'
  145      let chk' = rwSetCryptolEnvStack cryenv' chk
  146      return chk' { rwSharedContext = sc' }

  148 -- | Represents the mutable state of the TopLevel monad
  149 --   that can later be restored.
  150 data TopLevelCheckpoint =
  151   TopLevelCheckpoint
  152     TopLevelRW
  153     SharedContextCheckpoint
  154 
  155 makeCheckpoint :: TopLevelRW -> IO TopLevelCheckpoint
  156 makeCheckpoint rw =
  157   do scc <- checkpointSharedContext (rwSharedContext rw)
  158      return (TopLevelCheckpoint rw scc)
  159 
  160 restoreCheckpoint :: TopLevelCheckpoint -> TopLevel ()
  161 restoreCheckpoint chk =
  162   do rw <- getTopLevelRW
  163      rw' <- io (combineRW chk rw)
  164      putTopLevelRW rw'
  165 
  166 -- | Capture the current state of the TopLevel monad
  167 --   and return an action that, if invoked, resets
  168 --   the state back to that point.
  169 checkpoint :: TopLevel (() -> TopLevel ())
  170 checkpoint = TopLevel_ $
  171   do chk <- liftIO . makeCheckpoint =<< get
  172      return $ \_ ->
  173        do printOutLnTop Info "Restoring state from checkpoint"
  174           restoreCheckpoint chk
  175 
  176 -- | Capture the current proof state and return an
  177 --   action that, if invoked, resets the state back to that point.
  178 proof_checkpoint :: ProofScript (() -> ProofScript ())
  179 proof_checkpoint =
  180   do ps <- get
  181      return $ \_ ->
  182        do scriptTopLevel (printOutLnTop Info "Restoring proof state from checkpoint")
  183           put ps

  185 withEnviron :: Environ -> TopLevel a -> TopLevel a
  186 withEnviron env m = do
  187   prevEnv <- gets rwEnviron
  188   modify (\rw -> rw { rwEnviron = env })
  189   result <- m
  190   modify (\rw -> rw { rwEnviron = prevEnv })
  191   return result
