   43 -- | Internal wrapper for 'TypeRep' that we provide an 'MapF.OrdF'
   44 --   instance for.
   45 newtype TypeRep' a = TypeRep' (TypeRep a)
   46   deriving (TestEquality)
   47 
   48 instance MapF.OrdF TypeRep' where
   49   compareF (TypeRep' tr1) (TypeRep' tr2) =
   50     case testEquality tr1 tr2 of
   51       Just Refl -> MapF.EQF
   52       Nothing -> case compare (SomeTypeRep tr1) (SomeTypeRep tr2) of
   53         LT -> MapF.LTF
   54         GT -> MapF.GTF
   55         EQ -> error "inconsistent TypeRep equality"
   56 
   57 -- | Simple wrapper to intelligently use IntMap if the keys
   58 --   are 'Int'.
   59 data MapPI k v where
   60   MapP :: Ord k => Map k v -> MapPI k v
   61   MapI :: IntMap v -> MapPI Int v
   62 
   63 lookupMapPI :: k -> MapPI k v -> Maybe v
   64 lookupMapPI k = \case
   65   MapP m -> Map.lookup k m
   66   MapI m -> IntMap.lookup k m
   67 
   68 emptyMapPI :: forall k v. (Ord k, Typeable k) => MapPI k v
   69 emptyMapPI = case testEquality (TypeRep @Int) (TypeRep @k) of
   70   Just Refl -> MapI IntMap.empty
   71   Nothing -> MapP Map.empty
   72 
   73 alterMapPI :: (Maybe v -> Maybe v) -> k -> MapPI k v -> MapPI k v
   74 alterMapPI f k = \case
   75   MapP m -> MapP (Map.alter f k m)
   76   MapI m -> MapI (IntMap.alter f k m)
   77 
   78 -- | A map that associates keys of type 'k' with values of any
   79 --   'Typeable' type. Each stored type is treated as a distinct map:
   80 --   adding/modifying/deleting entries for a given key and type will
   81 --   not affect entries of other types.
   82 data TypedMap k = (Ord k, Typeable k) =>
   83   TypedMap (MapF TypeRep' (MapPI k))
   84 
   85 -- | An empty 'TypedMap'. Uses 'Typeable' keys to pick the
   86 --   underlying storage datatype based on the type of 'k'.
   87 empty :: forall k. (Ord k, Typeable k) => TypedMap k
   88 empty = TypedMap MapF.empty
   89 
   90 -- | Lookup the value of type 'a' associated with the given key.
   91 lookup :: forall a k. Typeable a => k -> TypedMap k -> Maybe a
   92 lookup k (TypedMap tm) = do
   93   m <- MapF.lookup (TypeRep' typeRep) tm
   94   lookupMapPI k m
   95 
   96 -- | Alter the data of type 'a' associated with this key, inserting or
   97 --   deleting based on the result of 'f'. Values of types other than
   98 --   'a' associated with this key are unmodified.
   99 alter :: forall a k.
  100   Typeable a =>
  101   (Maybe a -> Maybe a) ->
  102   k ->
  103   TypedMap k ->
  104   TypedMap k
  105 alter f k (TypedMap tm) = TypedMap (MapF.updatedValue tm')
  106     where
  107       rep = TypeRep' (TypeRep @a)
  108 
  109       go :: MapPI k a -> Identity (MapF.UpdateRequest (MapPI k a))
  110       go m = pure $ MapF.Set (alterMapPI f k m)
  111 
  112       tm' = runIdentity $
  113         MapF.updateAtKey rep (pure $ Just emptyMapPI) go tm
  114 
  115 
  116 -- | Insert a value of type 'a' at the given key.
  117 insert :: forall a k. Typeable a => k -> a -> TypedMap k -> TypedMap k
  118 insert k a = alter (\_ -> Just a) k
  119 
  120 -- | Delete the value of type 'a' for the given key, if it exists.
  121 delete :: forall a k. Typeable a => k -> TypedMap k -> TypedMap k
  122 delete k tm = alter @a (\_ -> Nothing) k tm
  123 
  124 lookup' :: forall a k. Typeable a => TypedMap k -> MapPI k a
  125 lookup' (TypedMap tm) = case MapF.lookup rep tm of
  126   Just m -> m
  127   Nothing -> emptyMapPI
  128   where
  129     rep = TypeRep' (TypeRep @a)
  130 
  131 elems :: forall a k. Typeable a => TypedMap k -> [a]
  132 elems tm = case lookup' tm of
  133   MapP m -> Map.elems m
  134   MapI m -> IntMap.elems m
  135 
  136 toMap :: forall a k. Typeable a => TypedMap k -> Map k a
  137 toMap tm = case lookup' tm of
  138   MapP m -> m
  139   MapI m -> Map.fromAscList $ IntMap.toAscList m
  140 
  141 toIntMap :: forall a. Typeable a => TypedMap Int -> IntMap a
  142 toIntMap tm = case lookup' @a @Int tm of
  143   MapP m -> IntMap.fromAscList $ Map.toAscList m
  144   MapI m -> m
  145 
  146 toList :: forall a k. Typeable a => TypedMap k -> [(k, a)]
  147 toList tm = case lookup' tm of
  148   MapP m -> Map.toList m
  149   MapI m -> IntMap.toList m
