   45 -- | Internal wrapper for 'TypeRep' that we provide an 'MapF.OrdF'
   46 --   instance for.
   47 newtype TypeRep' a = TypeRep' (TypeRep a)
   48   deriving (TestEquality)
   49 
   50 withRep :: TypeRep' a -> (Typeable a => b) -> b
   51 withRep (TypeRep' tr) f = withTypeable tr f
   52 
   53 instance MapF.OrdF TypeRep' where
   54   compareF (TypeRep' tr1) (TypeRep' tr2) =
   55     case testEquality tr1 tr2 of
   56       Just Refl -> MapF.EQF
   57       Nothing -> case compare (SomeTypeRep tr1) (SomeTypeRep tr2) of
   58         LT -> MapF.LTF
   59         GT -> MapF.GTF
   60         EQ -> error "inconsistent TypeRep equality"
   61 
   62 newtype Wrapped t = Wrapped { unwrap :: t }
   63 
   64 -- | A store of arbitrary 'Typeable' data, indexed by type.
   65 newtype TypedStore = TypedStore (MapF TypeRep' Wrapped)
   66 
   67 -- | An empty 'TypedStore'.
   68 empty :: TypedStore
   69 empty = TypedStore MapF.empty
   70 
   71 -- | Lookup the value of type 'a' from the store.
   72 lookup :: forall a. Typeable a => TypedStore -> Maybe a
   73 lookup (TypedStore tm) = fmap unwrap $ MapF.lookup (TypeRep' typeRep) tm
   74 
   75 -- | Alter the data of type 'a' in the store, inserting or
   76 --   deleting based on the result of 'f'.
   77 alter :: forall a.
   78   Typeable a =>
   79   (Maybe a -> Maybe a) ->
   80   TypedStore ->
   81   TypedStore
   82 alter f (TypedStore tm) = TypedStore (MapF.updatedValue tm')
   83   where
   84     rep = TypeRep' (TypeRep @a)
   85 
   86     go :: Wrapped a -> Identity (MapF.UpdateRequest (Wrapped a))
   87     go (Wrapped a) = case f (Just a) of
   88       Nothing -> pure $ MapF.Delete
   89       Just a' -> pure $ MapF.Set (Wrapped a')
   90 
   91     tm' = runIdentity $
   92       MapF.updateAtKey rep (pure $ fmap Wrapped $ f Nothing) go tm
   93 
   94 
   95 -- | Insert a value of type 'a' in the store, overwriting existing
   96 --   data.
   97 insert :: forall a. Typeable a => a -> TypedStore -> TypedStore
   98 insert a = alter (\_ -> Just a)
   99 
  100 singleton :: Typeable a => a -> TypedStore
  101 singleton a = TypedStore (MapF.singleton (TypeRep' typeRep) (Wrapped a))
  102 
  103 size :: TypedStore -> Int
  104 size (TypedStore ts) = MapF.size ts
  105 
  106 -- | Delete the value of type 'a' from the store.
  107 delete :: forall a. Typeable a => TypedStore -> TypedStore
  108 delete tm = alter @a (\_ -> Nothing) tm
  109 
  110 map :: (forall a. Typeable a => a -> a) -> TypedStore -> TypedStore
  111 map f (TypedStore ts) = TypedStore $ 
  112   MapF.mapWithKey (\rep (Wrapped x) -> withRep rep $ Wrapped (f x)) ts
  113 
  114 traverse :: 
  115   Applicative m => 
  116   (forall a. Typeable a => a -> m a) -> 
  117   TypedStore -> 
  118   m TypedStore
  119 traverse f (TypedStore ts) = TypedStore <$>
  120   MapF.traverseWithKey (\rep (Wrapped x) -> withRep rep $ Wrapped <$> f x) ts
  121 
  122 liftWrap2 :: 
  123   (Typeable a => a -> a -> Maybe a) ->
  124   TypeRep' a -> Wrapped a -> Wrapped a -> Maybe (Wrapped a)
  125 liftWrap2 f rep (Wrapped l) (Wrapped r) = withRep rep $
  126   case f l r of
  127     Just a -> Just (Wrapped a)
  128     Nothing -> Nothing
  129 
  130 merge :: 
  131   -- keys in both
  132   (forall a. Typeable a => a -> a -> Maybe a) ->
  133   -- keys only in left
  134   (forall a. Typeable a => a -> Maybe a) ->
  135   -- keys only in right
  136   (forall a. Typeable a => a -> Maybe a) ->
  137   TypedStore ->
  138   TypedStore ->
  139   TypedStore
  140 merge f g1 g2 (TypedStore ts1) (TypedStore ts2) =
  141   TypedStore (MapF.mergeWithKey (liftWrap2 f) (do_filter g1) (do_filter g2) ts1 ts2)
  142     where
  143       do_filter :: 
  144         (forall a. Typeable a => a -> Maybe a) -> 
  145         MapF TypeRep' Wrapped -> 
  146         MapF TypeRep' Wrapped
  147       do_filter g = MapF.mapMaybeWithKey $ 
  148         \rep (Wrapped a) -> withRep rep $
  149           case g a of
  150             Just b -> Just (Wrapped b)
  151             Nothing -> Nothing
  152 
  153 -- | Combine two 'TypedStore's, using the given function to merge
  154 --   entries of the same type.
  155 union :: 
  156   (forall a. Typeable a => a -> a -> Maybe a) ->
  157   TypedStore ->
  158   TypedStore ->
  159   TypedStore
  160 union f (TypedStore ts1) (TypedStore ts2) =
  161   TypedStore (MapF.mergeWithKey (liftWrap2 f) id id ts1 ts2)
  162 
  163 -- | Intersect two 'TypedStore's, using the given function to merge
  164 --   entries of the same type, and discarding types which only are
  165 --   only present in one store.
  166 intersection :: 
  167   (forall a. Typeable a => a -> a -> Maybe a) ->
  168   TypedStore ->
  169   TypedStore ->
  170   TypedStore
  171 intersection f (TypedStore ts1) (TypedStore ts2) =
  172   TypedStore (MapF.mergeWithKey (liftWrap2 f) (const MapF.empty) (const MapF.empty) ts1 ts2)
  173 
  174 toList :: (forall a. Typeable a => a -> b) -> TypedStore -> [b]
  175 toList f (TypedStore ts) = 
  176   List.map (\(MapF.Pair rep (Wrapped x)) -> withRep rep $ f x) $ MapF.toList ts
