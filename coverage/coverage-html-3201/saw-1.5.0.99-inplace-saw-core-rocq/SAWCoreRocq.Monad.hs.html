   37 data TranslationError
   38   = NotSupported Term
   39   | NotExpr Term
   40   | NotType Term
   41   | LocalVarOutOfBounds Term
   42   | BadTerm Term
   43   | CannotCreateDefaultValue Term
   44 
   45 ppTranslationError :: SharedContext -> TranslationError -> IO Text
   46 ppTranslationError sc err = do
   47   let (msg, tm) = case err of
   48         NotSupported t -> ("Not supported:", t)
   49         NotExpr t      -> ("Expecting an expression term:", t)
   50         NotType t      -> ("Expecting a type term: ", t)
   51         LocalVarOutOfBounds t ->
   52             ("Local variable reference is out of bounds:", t)
   53         BadTerm t      -> ("Malformed term:", t)
   54         CannotCreateDefaultValue t ->
   55             ("Unable to generate a default value of the given type:", t)
   56   ppopts <- scGetPPOpts sc
   57   tm' <- prettyTerm sc tm
   58   pure $ PPS.renderText ppopts $ msg <+> tm'

   60 data TranslationConfiguration = TranslationConfiguration
   61   { constantRenaming :: [(String, String)]
   62   -- ^ A map from 'ImportedName's of constants to names that should be used to
   63   -- realize them in Rocq; this is generally used to map cryptol operators like
   64   -- @||@ or @&&@ to nicer names in Rocq, but works on any imported names
   65   , constantSkips :: [String]
   66   -- ^ A list of 'ImportedName's to skip, i.e., not translate when encountered
   67   , monadicTranslation :: Bool
   68   -- ^ Whether to wrap everything in a free monad construction.
   69   -- - Advantage: fixpoints can be readily represented
   70   -- - Disadvantage: pure computations look more gnarly
   71   , postPreamble :: String
   72   -- ^ Text to be concatenated at the end of the Rocq preamble, before the code
   73   -- generated by the translation.  Usually consists of extra file imports,
   74   -- module imports, scopes openings.
   75   , vectorModule       :: String
   76   -- ^ all vector operations will be prepended with this module name, i.e.
   77   -- "<VectorModule>.append", etc.  So that it can be retargeted easily.
   78   -- Current provided options are:
   79   -- - SAWCoreVectorsAsRocqLists
   80   -- - SAWCoreVectorsAsRocqVectors
   81   -- Currently considering adding:
   82   -- - SAWCoreVectorsAsSSReflectSeqs
   83   }
   84 
   85 -- | The functional dependency of 'MonadReader' makes it not compositional, so
   86 -- we have to jam together different structures that want to be in the 'Reader'
   87 -- into a single datatype.  This type allows adding extra configuration on top
   88 -- of the translation configuration.
   89 data WithTranslationConfiguration r = WithTranslationConfiguration
   90   { translationConfiguration :: TranslationConfiguration
   91   , otherConfiguration :: r
   92   }
   93 
   94 -- | Some computations will rely solely on access to the configuration, so we
   95 -- provide it separately.
   96 type TranslationConfigurationMonad r m =
   97   ( MonadReader (WithTranslationConfiguration r) m
   98   )
   99 
  100 type TranslationMonad r s m =
  101   ( Except.MonadError TranslationError  m
  102   , TranslationConfigurationMonad r m
  103   , MonadState s m
  104   )
  105 
  106 runTranslationMonad ::
  107   TranslationConfiguration ->
  108   r ->
  109   s ->
  110   (forall m. TranslationMonad r s m => m a) ->
  111   Either TranslationError (a, s)
  112 runTranslationMonad configuration r s m =
  113   runStateT (runReaderT m (WithTranslationConfiguration configuration r)) s
