   47 -- | Declare a particular name as opaque
   48 declare_unint ::
   49   String {- ^ Name of command for error messages -} ->
   50   Setter' (MS.CrucibleContext ext) (Set VarIndex) {-^ Keep uninterpreted names here -} ->
   51   [Text] {- ^ Names for the things to be left uninterpreted -} ->
   52   CrucibleSetup ext ()
   53 declare_unint cmd unintF names = 
   54   do
   55     prePost <- use csPrePost
   56     case prePost of
   57       MS.PreState ->
   58         do
   59           ns <- crucibleSetupTopLevel (resolveNames names)
   60           csCrucibleContext . unintF %= Set.union ns
   61       MS.PostState ->
   62         fail ("`" ++ cmd ++ "` works only in the pre-condition of a specification.")

   64 crucible_precond ::
   65   W4.ProgramLoc ->
   66   TypedTerm ->
   67   CrucibleSetup ext ()
   68 crucible_precond loc p = do
   69   st <- get
   70   tags <- view croTags
   71   let md = MS.ConditionMetadata
   72            { MS.conditionLoc = loc
   73            , MS.conditionTags = tags
   74            , MS.conditionType = "specification assertion"
   75            , MS.conditionContext = ""
   76            }
   77   when (st ^. csPrePost == MS.PostState) $
   78     fail "attempt to use `crucible_precond` in post state"
   79   addCondition (MS.SetupCond_Pred md p)
   80

   98 crucible_return ::
   99   MS.SetupValue ext ->
  100   CrucibleSetup ext ()
  101 crucible_return retval = do
  102   ret <- use (csMethodSpec . MS.csRetValue)
  103   case ret of
  104     Just _ -> fail "crucible_return: duplicate return value specification"
  105     Nothing -> csMethodSpec . MS.csRetValue .= Just retval
  106 
  107 crucible_execute_func ::
  108   [MS.SetupValue ext] ->
  109   CrucibleSetup ext ()
  110 crucible_execute_func args = do
  111   st <- get
  112   tps <- use (csMethodSpec . MS.csArgs)
  113   when (st ^. csPrePost == MS.PostState) $
  114     fail "duplicate execute_func declaration"
  115   csPrePost .= MS.PostState
  116   csMethodSpec . MS.csArgBindings .= Map.fromList [ (i, (t,a))
  117                                                   | i <- [0..]
  118                                                   | a <- args
  119                                                   | t <- tps
  120                                                   ]
  121 
  122 crucible_equal ::
  123   W4.ProgramLoc ->
  124   MS.SetupValue ext ->
  125   MS.SetupValue ext ->
  126   CrucibleSetup ext ()
  127 crucible_equal loc val1 val2 = do
  128   tags <- view croTags
  129   let md = MS.ConditionMetadata
  130            { MS.conditionLoc = loc
  131            , MS.conditionTags = tags
  132            , MS.conditionType = "equality specification"
  133            , MS.conditionContext = ""
  134            }
  135   addCondition (MS.SetupCond_Equal md val1 val2)

  140 -- | When invoking a points-to command, against what should SAW check the type
  141 -- of the RHS value?
  142 data CheckPointsToType ty
  143   = CheckAgainstPointerType
  144     -- ^ Check the type of the RHS value against the type that the LHS points to.
  145     --   Used by commands such as @llvm_{conditional_}points_to@ and
  146     --   @mir_points_to@.
  147   | CheckAgainstCastedType ty
  148     -- ^ Check the type of the RHS value against the provided @ty@, which
  149     --   the LHS pointer is casted to.
  150     --   This is currently used by @llvm_{conditional_}points_to_at_type@ only.
  151   deriving (Functor, Foldable, Traversable)
