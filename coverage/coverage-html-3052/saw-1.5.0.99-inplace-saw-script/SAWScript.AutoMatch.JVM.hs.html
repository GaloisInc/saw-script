   18 -- | Parse a Java class into a list of declarations
   19 --   Yields an Interaction so that we can talk to the user about what went wrong
   20 getDeclsJVM :: JVM.Class -> IO (Interaction (Maybe [Decl]))
   21 getDeclsJVM cls = return $ do
   22 
   23    let (untranslateable, translations) =
   24          partitionEithers . for (JVM.classMethods cls) $ \method ->
   25             maybe (Left $ JVM.methodName method) Right $ do
   26                returnType <- jssTypeToStdType =<< JVM.methodReturnType method
   27                argTypes <- mapM jssTypeToStdType $ JVM.methodParameterTypes method
   28                let argIndices = map (JVM.localIndexOfParameter method) [0 .. length argTypes - 1]
   29                tableEntries <- forM argIndices $ JVM.lookupLocalVariableByIdx method 0
   30                let getname name = Text.pack $ JVM.localName name
   31                    args = zipWith Arg (map getname tableEntries) argTypes
   32                return $ Decl (Text.pack $ JVM.methodName method) returnType args
   33 
   34    when (not . null $ untranslateable) $ do
   35       separator ThinSep
   36       liftF . flip Warning () $ "No translation for the following signatures in " ++ JVM.unClassName (JVM.className cls) ++ ".class:"
   37       bulleted $ map (("'" ++) . (++ "'")) untranslateable
   38 
   39    return $ Just translations
   40 
   41    where
   42 
   43       jssTypeToStdType = \case
   44          JVM.ArrayType _    -> Nothing -- TODO: Handle array types
   45          JVM.BooleanType    -> Just $ TCon (TC TCBit) []
   46          JVM.ByteType       -> Just $ bitSeqType 8
   47          JVM.CharType       -> Just $ bitSeqType 16 -- TODO: Not sure about this equivalence
   48          JVM.ClassType _str -> Nothing -- TODO: Support classes as records?
   49          JVM.DoubleType     -> Nothing -- We don't support floating point types
   50          JVM.FloatType      -> Nothing -- We don't support floating point types
   51          JVM.IntType        -> Just $ bitSeqType 32
   52          JVM.LongType       -> Just $ bitSeqType 64
   53          JVM.ShortType      -> Just $ bitSeqType 16
