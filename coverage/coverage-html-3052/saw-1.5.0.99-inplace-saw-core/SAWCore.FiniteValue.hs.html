   38 -- | Finite types that can be encoded as bits for a SAT/SMT solver.
   39 data FiniteType
   40   = FTBit
   41   | FTVec Natural FiniteType
   42   | FTTuple [FiniteType]
   43   | FTRec (Map FieldName FiniteType)
   44   deriving (Eq, Show)
   45 
   46 -- | Values inhabiting those finite types.
   47 data FiniteValue
   48   = FVBit Bool
   49   | FVWord Natural Integer -- ^ a more efficient special case for 'FVVec FTBit _'.
   50   | FVVec FiniteType [FiniteValue]
   51   | FVTuple [FiniteValue]
   52   | FVRec (Map FieldName FiniteValue)
   53   deriving Eq
   54 
   55 -- | First-order types that can be encoded in an SMT solver.
   56 -- NB: The JSON encoding of this type, used for saw-script solver result caching,
   57 -- assumes constructor names and argument orders will not change (though the
   58 -- order and number of constructors may change) - see 'firstOrderJSONOptions'
   59 data FirstOrderType
   60   = FOTBit
   61   | FOTInt
   62   | FOTIntMod Natural
   63   | FOTVec Natural FirstOrderType
   64   | FOTArray FirstOrderType FirstOrderType
   65   | FOTTuple [FirstOrderType]
   66   | FOTRec (Map FieldName FirstOrderType)
   67   deriving (Eq, Ord, Show, Generic)
   68 
   69 -- | Values inhabiting those first-order types.
   70 -- NB: The JSON encoding of this type, used for saw-script solver result caching,
   71 -- assumes constructor names and argument orders will not change (though the
   72 -- order and number of constructors may change) - see 'firstOrderJSONOptions'
   73 --
   74 -- The type argument of FOVArray is the key type; the value type is
   75 -- derivable from the default value, which is the second argument. The third
   76 -- argument is an assignment for all the entries that have non-default values.
   77 --
   78 -- FOVOpaqueArray is for arrays we get back from the solver only as a
   79 -- function call that one can use for lookups. We can't do anything
   80 -- with that (see Note [FOVArray] below) so we just treat it as an
   81 -- opaque blob. The arguments are the key and value types, since we
   82 -- do at least have that info.
   83 data FirstOrderValue
   84   = FOVBit Bool
   85   | FOVInt Integer
   86   | FOVIntMod Natural Integer
   87   | FOVWord Natural Integer -- ^ a more efficient special case for 'FOVVec FOTBit _'.
   88   | FOVVec FirstOrderType [FirstOrderValue]
   89   | FOVArray FirstOrderType FirstOrderValue (Map FirstOrderValue FirstOrderValue)
   90   | FOVOpaqueArray FirstOrderType FirstOrderType
   91   | FOVTuple [FirstOrderValue]
   92   | FOVRec (Map FieldName FirstOrderValue)
   93   deriving (Eq, Ord, Generic)
   94 
   95 --
   96 -- Note [FOVArray]
   97 -- ~~~~~~~~~~~~~~~
   98 --
   99 -- We only handle arrays that are:
  100 --    - unidimensional
  101 --    - made up of explicit concrete values
  102 --
  103 -- We could handle multidimensional arrays easily enough (the key type
  104 -- and assignment keys just need to become lists) but for the moment
  105 -- there's no use case.
  106 --
  107 -- The What4 interface can sometimes return an array that isn't made
  108 -- up of explicit concrete values but is instead represented as a
  109 -- function you can call to get values out for given keys. It is not
  110 -- clear how we'd use this, since the primary thing we do with array
  111 -- values that come back from the solver is print them as part of
  112 -- models and without a way to know what keys are present a function
  113 -- that just extracts values is fairly useless. If one of these pops
  114 -- up, it comes back as FOVOpaqueArray and we treat it as an opaque
  115 -- blob. Unfortuately this does happen (maybe we can improve What4 so
  116 -- it happens less or maybe not at all, but that's not happening right
  117 -- away) so we need to be able to handle it rather than erroring out
  118 -- or crashing.
  119 --
  120 -- Note that trying to retain the function at this level is
  121 -- problematic for a number of reasons:
  122 --
  123 --    1. It's not actually a first-order value, and calling it one is
  124 --       borrowing trouble.
  125 --
  126 --    2. We need ToJSON and FromJSON instances for solver caching, and
  127 --       there's no way to do that. Now, trying to stick one of these
  128 --       objects in the solver cache is just not going to work no
  129 --       matter how we handle it; but things are not structured so
  130 --       that we can e.g. decline to cache values that contain
  131 --       function-based arrays so that it's safe to stub out the
  132 --       ToJSON and FromJSON instances with an error invocation. That
  133 --       could doubtless be arranged, but it'll take work, possibly a
  134 --       lot of work. As things stand, we _can_ stuff FOVOpaqueArray
  135 --       into the solver cache, and it won't do anything useful when
  136 --       we unstuff it later, but it won't crash.
  137 --
  138 --    3. FirstOrderValue needs Eq and Ord instances, at least but not
  139 --       necessarily only for use by maps. At least one of those maps
  140 --       (the one used in FOVArray) is restricted to not have array
  141 --       values as keys; however, that's not an _explicit_ restriction
  142 --       (e.g. in typechecking), it's a consequence of the things
  143 --       What4 allows as array keys. To be halfway safe we'd need to
  144 --       make it an explicit restriction, and in the current state of
  145 --       SAW maintenance it's not really clear enough where we'd need
  146 --       to enforce that to keep it safe. Also, there may be other
  147 --       such maps about where arrays _can_ be keys, or for that
  148 --       matter other comparisons. In principle you could have the
  149 --       compiler find you all the uses by disabling the Eq and Ord
  150 --       instances. But that doesn't work because the ToJSON and
  151 --       FromJSON classes need them. The compiler then just fails on
  152 --       those before telling you anything else.
  153 --
  154 -- Besides all the above, restrictions in What4 mean that array values
  155 -- coming back from the solver via that interface are indexed only by
  156 -- integers or bitvectors. At this layer, though, we can support any
  157 -- FirstOrderValue.
  158 --
  159 
  160 toFirstOrderType :: FiniteType -> FirstOrderType
  161 toFirstOrderType ft =
  162   case ft of
  163     FTBit      -> FOTBit
  164     FTVec n t  -> FOTVec n (toFirstOrderType t)
  165     FTTuple ts -> FOTTuple (map toFirstOrderType ts)
  166     FTRec tm   -> FOTRec (fmap toFirstOrderType tm)
  167 
  168 toFirstOrderValue :: FiniteValue -> FirstOrderValue
  169 toFirstOrderValue fv =
  170   case fv of
  171     FVBit b    -> FOVBit b
  172     FVWord w i -> FOVWord w i
  173     FVVec t vs -> FOVVec (toFirstOrderType t) (map toFirstOrderValue vs)
  174     FVTuple vs -> FOVTuple (map toFirstOrderValue vs)
  175     FVRec vm   -> FOVRec (fmap toFirstOrderValue vm)

  178 toFiniteType :: FirstOrderType -> Maybe FiniteType
  179 toFiniteType FOTBit        = pure FTBit
  180 toFiniteType (FOTVec n t)  = FTVec n <$> toFiniteType t
  181 toFiniteType (FOTTuple ts) = FTTuple <$> traverse toFiniteType ts
  182 toFiniteType (FOTRec fs)   = FTRec   <$> traverse toFiniteType fs
  183 toFiniteType FOTInt{}      = Nothing
  184 toFiniteType FOTIntMod{}   = Nothing
  185 toFiniteType FOTArray{}    = Nothing

  187 instance Show FiniteValue where
  188   showsPrec p fv = showsPrec p (toFirstOrderValue fv)
  189 
  190 instance Show FirstOrderValue where
  191   showsPrec _ fv =
  192     case fv of
  193       FOVBit b    -> shows b
  194       FOVInt i    -> shows i
  195       FOVIntMod _ i -> shows i
  196       FOVWord _ x -> shows x
  197       FOVVec _ vs -> showString "[" . commaSep (map shows vs) . showString "]"
  198       FOVArray _kty d vs ->
  199         let vs' = map showEntry $ Map.toAscList vs
  200             d' = showEntry ("<default>", d)
  201         in
  202         showString "[" . commaSep (vs' ++ [d']) . showString "]"
  203       FOVOpaqueArray _kty _vty ->
  204         showString "[ opaque array, sorry ]"
  205       FOVTuple vs -> showString "(" . commaSep (map shows vs) . showString ")"
  206       FOVRec vm   -> showString "{" . commaSep (map showField (Map.assocs vm)) . showString "}"
  207     where
  208       commaSep ss = foldr (.) id (intersperse (showString ",") ss)
  209       showEntry (k, v) = shows k . showString " := " . shows v
  210       showField (field, v) = showString (Text.unpack field) . showString " = " . shows v

  212 prettyFiniteValue :: PPS.Opts -> FiniteValue -> PPS.Doc
  213 prettyFiniteValue opts fv = prettyFirstOrderValue opts (toFirstOrderValue fv)

  215 prettyFirstOrderValue :: PPS.Opts -> FirstOrderValue -> PPS.Doc
  216 prettyFirstOrderValue opts = loop
  217  where
  218  loop fv = case fv of
  219    FOVBit b
  220      | b         -> pretty "True"
  221      | otherwise -> pretty "False"
  222    FOVInt i      -> pretty i
  223    FOVIntMod _ i -> pretty i
  224    FOVWord _w i  -> prettyNat opts i
  225    FOVVec _ xs   -> brackets (sep (punctuate comma (map loop xs)))
  226    FOVArray _kty d vs ->
  227       let ppEntry' k' v = k' <+> pretty ":=" <+> loop v
  228           ppEntry (k, v) = ppEntry' (loop k) v
  229           d' = ppEntry' (pretty "<default>") d
  230           vs' = map ppEntry $ Map.toAscList vs
  231       in
  232       brackets (nest 4 (sep (punctuate comma (vs' ++ [d']))))
  233    FOVOpaqueArray _kty _vty ->
  234       pretty "[ opaque array, sorry ]"
  235    FOVTuple xs   -> parens (nest 4 (sep (punctuate comma (map loop xs))))
  236    FOVRec xs     -> braces (sep (punctuate comma (map ppField (Map.toList xs))))
  237       where ppField (f,x) = pretty f <+> pretty '=' <+> loop x

  239 -- | The options for JSON-serializing 'FirstOrderType's and 'FirstOrderValue's:
  240 -- remove the @FOT@/@FOV@ prefixes and encode the different constructors as
  241 -- two-element arrays. Thus, this encoding assumes constructor names and
  242 -- argument orders will not change (though the order and number of constructors
  243 -- may change).
  244 firstOrderJSONOptions :: JSON.Options
  245 firstOrderJSONOptions =
  246   JSON.defaultOptions { JSON.sumEncoding = JSON.TwoElemArray
  247                       , JSON.constructorTagModifier = dropFO }
  248   where dropFO ('F':'O':tv:cs) | tv `elem` ['T', 'V'] = cs
  249         dropFO cs = cs
  250 
  251 instance FromJSON FirstOrderType where
  252   parseJSON = JSON.genericParseJSON firstOrderJSONOptions
  253 instance FromJSON FirstOrderValue where
  254   parseJSON = JSON.genericParseJSON firstOrderJSONOptions
  255 instance FromJSONKey FirstOrderValue
  256 
  257 instance ToJSON FirstOrderType where
  258   toJSON = JSON.genericToJSON firstOrderJSONOptions
  259   toEncoding = JSON.genericToEncoding firstOrderJSONOptions
  260 instance ToJSON FirstOrderValue where
  261   toJSON = JSON.genericToJSON firstOrderJSONOptions
  262   toEncoding = JSON.genericToEncoding firstOrderJSONOptions
  263 instance ToJSONKey FirstOrderValue

  266 -- | Smart constructor
  267 fvVec :: FiniteType -> [FiniteValue] -> FiniteValue
  268 fvVec t vs =
  269   case (t, traverse toBit vs) of
  270     (FTBit, Just bs) -> FVWord (fromIntegral (length bs)) (fromBits bs)
  271     _ -> FVVec t vs
  272   where
  273     toBit :: FiniteValue -> Maybe Bool
  274     toBit (FVBit b) = Just b
  275     toBit _ = Nothing
  276 
  277     fromBits :: [Bool] -> Integer
  278     fromBits = foldl (\n b -> 2*n + if b then 1 else 0) 0
  279 
  280 -- | Smart constructor
  281 fovVec :: FirstOrderType -> [FirstOrderValue] -> FirstOrderValue
  282 fovVec t vs =
  283   case (t, traverse toBit vs) of
  284     (FOTBit, Just bs) -> FOVWord (fromIntegral (length bs)) (fromBits bs)
  285     _ -> FOVVec t vs
  286   where
  287     toBit :: FirstOrderValue -> Maybe Bool
  288     toBit (FOVBit b) = Just b
  289     toBit _ = Nothing
  290 
  291     fromBits :: [Bool] -> Integer
  292     fromBits = foldl (\n b -> 2*n + if b then 1 else 0) 0

  294 finiteTypeOf :: FiniteValue -> FiniteType
  295 finiteTypeOf fv =
  296   case fv of
  297     FVBit _    -> FTBit
  298     FVWord n _ -> FTVec n FTBit
  299     FVVec t vs -> FTVec (fromIntegral (length vs)) t
  300     FVTuple vs -> FTTuple (map finiteTypeOf vs)
  301     FVRec vm   -> FTRec (fmap finiteTypeOf vm)
  302 
  303 firstOrderTypeOf :: FirstOrderValue -> FirstOrderType
  304 firstOrderTypeOf fv =
  305   case fv of
  306     FOVBit _    -> FOTBit
  307     FOVInt _    -> FOTInt
  308     FOVIntMod n _ -> FOTIntMod n
  309     FOVWord n _ -> FOTVec n FOTBit
  310     FOVVec t vs -> FOTVec (fromIntegral (length vs)) t
  311     FOVArray tk d _vs -> FOTArray tk (firstOrderTypeOf d)
  312     FOVOpaqueArray tk tv -> FOTArray tk tv
  313     FOVTuple vs -> FOTTuple (map firstOrderTypeOf vs)
  314     FOVRec vm   -> FOTRec (fmap firstOrderTypeOf vm)

  316 -- | Compute the number of bits in the type
  317 sizeFiniteType :: FiniteType -> Int
  318 sizeFiniteType x =
  319   case x of
  320     FTBit      -> 1
  321     FTVec n xs -> fromIntegral n * sizeFiniteType xs
  322     FTTuple xs -> sum (map sizeFiniteType xs)
  323     FTRec xm   -> sum (map sizeFiniteType (Map.elems xm))

  325 asFiniteType :: SharedContext -> Term -> IO FiniteType
  326 asFiniteType sc t = do
  327   t' <- scWhnf sc t
  328   case t' of
  329     (R.asBoolType -> Just ())
  330       -> return FTBit
  331     (R.isVecType return -> Just (n R.:*: tp))
  332       -> FTVec n <$> asFiniteType sc tp
  333     (R.asTupleType -> Just ts)
  334       -> FTTuple <$> traverse (asFiniteType sc) ts
  335     (R.asRecordType -> Just fs)
  336       -> FTRec <$> traverse (asFiniteType sc) (Map.fromList fs)
  337     _ -> do
  338         t'' <- ppTerm sc PPS.defaultOpts t'
  339         fail $ "asFiniteType: unsupported argument type: " ++ t''

  341 asFirstOrderType :: SharedContext -> Term -> IO FirstOrderType
  342 asFirstOrderType sc t = maybe err pure =<< runMaybeT (asFirstOrderTypeMaybe sc t)
  343   where
  344     err =
  345       do t' <- scWhnf sc t
  346          t'' <- ppTerm sc PPS.defaultOpts t'
  347          fail $ "asFirstOrderType: unsupported argument type: " ++ t''
  348 
  349 asFirstOrderTypeMaybe :: SharedContext -> Term -> MaybeT IO FirstOrderType
  350 asFirstOrderTypeMaybe sc t =
  351   do t' <- lift (scWhnf sc t)
  352      case t' of
  353        (R.asBoolType -> Just ())
  354          -> return FOTBit
  355        (R.asIntegerType -> Just ())
  356          -> return FOTInt
  357        (R.asIntModType -> Just n)
  358          -> return (FOTIntMod n)
  359        (R.isVecType return -> Just (n R.:*: tp))
  360          -> FOTVec n <$> asFirstOrderTypeMaybe sc tp
  361        (R.asArrayType -> Just (tp1 R.:*: tp2)) -> do
  362          tp1' <- asFirstOrderTypeMaybe sc tp1
  363          tp2' <- asFirstOrderTypeMaybe sc tp2
  364          return $ FOTArray tp1' tp2'
  365        (R.asTupleType -> Just ts)
  366          -> FOTTuple <$> traverse (asFirstOrderTypeMaybe sc) ts
  367        (R.asRecordType -> Just fs)
  368          -> FOTRec <$> traverse (asFirstOrderTypeMaybe sc) (Map.fromList fs)
  369        _ -> mzero

  386 -- | Convert a finite type to a Term.
  387 scFiniteType :: SharedContext -> FiniteType -> IO Term
  388 scFiniteType sc ft = scFirstOrderType sc (toFirstOrderType ft)
  389 
  390 -- | Convert a finite type to a Term.
  391 scFirstOrderType :: SharedContext -> FirstOrderType -> IO Term
  392 scFirstOrderType sc ft =
  393   case ft of
  394     FOTBit      -> scBoolType sc
  395     FOTInt      -> scIntegerType sc
  396     FOTIntMod n -> scIntModType sc =<< scNat sc n
  397     FOTVec n t  -> do n' <- scNat sc n
  398                       t' <- scFirstOrderType sc t
  399                       scVecType sc n' t'
  400     FOTArray t1 t2 -> do t1' <- scFirstOrderType sc t1
  401                          t2' <- scFirstOrderType sc t2
  402                          scArrayType sc t1' t2'
  403     FOTTuple ts -> scTupleType sc =<< traverse (scFirstOrderType sc) ts
  404     FOTRec tm   ->
  405       scRecordType sc =<< (Map.assocs <$> traverse (scFirstOrderType sc) tm)
  406 
  407 -- | Convert a finite value to a SharedTerm.
  408 scFiniteValue :: SharedContext -> FiniteValue -> IO Term
  409 scFiniteValue sc fv = scFirstOrderValue sc (toFirstOrderValue fv)
  410 
  411 -- | Convert a finite value to a SharedTerm.
  412 scFirstOrderValue :: SharedContext -> FirstOrderValue -> IO Term
  413 scFirstOrderValue sc fv =
  414   case fv of
  415     FOVBit b    -> scBool sc b
  416     FOVInt i
  417       | i >= 0  -> scNatToInt sc =<< scNat sc (fromInteger i)
  418       | True    -> scIntNeg sc =<< scNatToInt sc =<< scNat sc (fromInteger (- i))
  419     FOVIntMod 0 i ->
  420       do n' <- scNat sc 0
  421          scToIntMod sc n' =<< scFirstOrderValue sc (FOVInt i)
  422     FOVIntMod n i ->
  423       do n' <- scNat sc n
  424          i' <- scNatToInt sc =<< scNat sc (fromInteger (i `mod` toInteger n))
  425          scToIntMod sc n' i'
  426     FOVWord n x -> scBvConst sc n x
  427     FOVVec t vs -> do t' <- scFirstOrderType sc t
  428                       vs' <- traverse (scFirstOrderValue sc) vs
  429                       scVector sc t' vs'
  430     FOVArray tk d vs -> do
  431         -- first get the key and value types
  432         tk' <- scFirstOrderType sc tk
  433         tv' <- scFirstOrderType sc $ firstOrderTypeOf d
  434         -- convert the default value and make an array
  435         d' <- scFirstOrderValue sc d
  436         arr0 <- scArrayConstant sc tk' tv' d'
  437         -- now add each update to the array (monadic fold)
  438         let visit k v arr = do
  439               k' <- scFirstOrderValue sc k
  440               v' <- scFirstOrderValue sc v
  441               scArrayUpdate sc tk' tv' arr k' v'
  442         ifoldrM visit arr0 vs
  443     FOVOpaqueArray _tk _tv -> do
  444         fail "Cannot convert opaque array to SharedTerm"
  445     FOVTuple vs -> scTuple sc =<< traverse (scFirstOrderValue sc) vs
  446     FOVRec vm   -> scRecord sc =<< traverse (scFirstOrderValue sc) vm

  448 -- Parsing values from lists of booleans ---------------------------------------
  449 
  450 data Endianness
  451   = BigEndian
  452   | LittleEndian
  453 
  454 readFiniteValue' :: Endianness -> FiniteType -> S.StateT [Bool] Maybe FiniteValue
  455 readFiniteValue' en ft =
  456   case ft of
  457     FTBit      -> do bs <- S.get
  458                      case bs of
  459                        []      -> S.lift Nothing
  460                        b : bs' -> S.put bs' >> return (FVBit b)
  461     FTVec n t  -> (fvVec t . fixup) <$> replicateM (fromIntegral n) (readFiniteValue' en t)
  462                     where fixup = case (t, en) of
  463                                     (FTBit, LittleEndian) -> reverse
  464                                     _ -> id
  465     FTTuple ts -> FVTuple <$> traverse (readFiniteValue' en) ts
  466     FTRec tm   -> FVRec <$> traverse (readFiniteValue' en) tm
  467 
  468 readFiniteValues :: [FiniteType] -> [Bool] -> Maybe [FiniteValue]
  469 readFiniteValues ts bs = do
  470   (vs, rest) <- S.runStateT (traverse (readFiniteValue' BigEndian) ts) bs
  471   case rest of
  472     [] -> return vs
  473     _ -> Nothing
  474 
  475 readFiniteValuesLE :: [FiniteType] -> [Bool] -> Maybe [FiniteValue]
  476 readFiniteValuesLE ts bs = do
  477   (vs, rest) <- S.runStateT (traverse (readFiniteValue' LittleEndian) ts) bs
  478   case rest of
  479     [] -> return vs
  480     _ -> Nothing

  482 readFiniteValue :: FiniteType -> [Bool] -> Maybe FiniteValue
  483 readFiniteValue t bs = do
  484   (v, rest) <- S.runStateT (readFiniteValue' BigEndian t) bs
  485   case rest of
  486     [] -> return v
  487     _ -> Nothing
