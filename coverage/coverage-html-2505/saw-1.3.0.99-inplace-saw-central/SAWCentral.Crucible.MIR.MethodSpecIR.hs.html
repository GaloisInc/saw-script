  117 mutIso :: Iso' M.Mutability Bool
  118 mutIso =
  119   iso
  120     (\case
  121       M.Mut -> True
  122       M.Immut -> False)
  123     (\case
  124       True -> M.Mut
  125       False -> M.Immut)
  126 
  127 isMut :: Lens' (MirAllocSpec tp) Bool
  128 isMut = maMutbl . mutIso

  130 -- | Converts a 'MirPointerKind' to a 'M.Ty' constructor.
  131 ptrKindToTy :: MirPointerKind -> M.Ty -> M.Mutability -> M.Ty
  132 ptrKindToTy MirPointerRef = M.TyRef
  133 ptrKindToTy MirPointerRaw = M.TyRawPtr
  134 
  135 -- | Gets the 'MirPointerKind' for a reference or pointer type. Panics if the
  136 -- given 'M.Ty' is not a reference or pointer.
  137 tyToPtrKind :: M.Ty -> MirPointerKind
  138 tyToPtrKind (M.TyRef _ _) = MirPointerRef
  139 tyToPtrKind (M.TyRawPtr _ _) = MirPointerRaw
  140 tyToPtrKind ty = panic "tyToPtrKind"
  141   ["Type is not a reference or pointer: " <> Text.pack (show ty)]

  145 initialDefCrucibleMethodSpecIR ::
  146   CollectionState ->
  147   M.Fn ->
  148   ProgramLoc ->
  149   MS.CrucibleMethodSpecIR MIR
  150 initialDefCrucibleMethodSpecIR rm fn loc =
  151   let fname = fn ^. M.fname
  152       fsig = fn ^. M.fsig
  153       argTys = fsig ^. M.fsarg_tys
  154       retTy = case fsig ^. M.fsreturn_ty of
  155                 M.TyTuple [] -> Nothing
  156                 ty -> Just ty in
  157   MS.makeCrucibleMethodSpecIR fname argTys retTy loc rm
  158 
  159 initialCrucibleSetupState ::
  160   MIRCrucibleContext ->
  161   M.Fn ->
  162   ProgramLoc ->
  163   Setup.CrucibleSetupState MIR
  164 initialCrucibleSetupState cc fn loc =
  165   Setup.makeCrucibleSetupState () cc $
  166     initialDefCrucibleMethodSpecIR
  167       (cc ^. mccRustModule ^. rmCS)
  168       fn
  169       loc
