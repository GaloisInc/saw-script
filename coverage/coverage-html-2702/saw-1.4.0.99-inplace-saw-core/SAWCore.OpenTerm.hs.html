   11 Module      : SAWCore.OpenTerm
   12 Copyright   : Galois, Inc. 2018

   14 Stability   : experimental
   15 Portability : non-portable (language extensions)
   16 
   17 This module defines an interface to building SAW core terms in an incrementally
   18 type-checked way, meaning that type-checking is performed as the terms are
   19 built. The interface provides a convenient DSL for building terms in a pure way,
   20 where sub-terms can be freely composed and combined into super-terms without
   21 monadic sequencing or 'IO' computations; the 'IO' computation is only run at the
   22 top level when all the term-building is complete. Users of this interface can
   23 also build binding constructs like lambda- and pi-abstractions without worrying
   24 about deBruijn indices, lifting, and free variables. Instead, a key feature of
   25 this interface is that it uses higher-order abstract syntax for lambda- and
   26 pi-abstractions, meaning that the bodies of these term constructs are specified
   27 as Haskell functions that take in terms for the bound variables. The library
   28 takes care of all the deBruijn indices under the hood.
   29 
   30 To use the 'OpenTerm' API, the caller builds up 'OpenTerm's using a variety of
   31 combinators that mirror the SAW core 'Term' structure. As some useful examples
   32 of 'OpenTerm' operations, 'applyOpenTerm' applies one 'OpenTerm' to another,
   33 'globalOpenTerm' builds an 'OpenTerm' for a global identifier, and
   34 'lambdaOpenTerm' builds a lambda-abstraction. For instance, the SAW core term
   35 
   36 > \ (f : Bool -> Bool) (x : Bool) -> f x
   37 
   38 can be built with the 'OpenTerm' expression
   39 
   40 > let bool = globalOpenTerm "Prelude.Bool" in
   41 > lambdaOpenTerm "f" (arrowOpenTerm bool bool) $ \f ->
   42 > lambdaOpenTerm "x" (globalOpenTerm "Prelude.Bool") $ \x ->
   43 > applyOpenTerm f x
   44 
   45 Existing SAW core 'Term's can be used in 'OpenTerm' by applying 'closedOpenTerm'
   46 if the 'Term' is closed (meaning it has no free variables) or 'openOpenTerm' if
   47 it does, where the latter requires the context of free variables to be
   48 specified. At the top level, 'completeOpenTerm' then "completes" an 'OpenTerm'
   49 by running its underlying 'IO' computation to build and type-check the resulting
   50 SAW core 'Term'.
   51 -}

  105 -- | An open term is represented as a type-checking computation that computes a
  106 -- SAW core term and its type
  107 newtype OpenTerm = OpenTerm { unOpenTerm :: TCM SC.Term }
  108 
  109 -- | \"Complete\" an 'OpenTerm' to a closed term or 'fail' on type-checking
  110 -- error
  111 completeOpenTerm :: SharedContext -> OpenTerm -> IO Term
  112 completeOpenTerm sc (OpenTerm termM) =
  113   either (fail . show) return =<<
  114   runTCM (SC.ctermVal <$> termM) sc mempty
  115 
  116 -- | \"Complete\" an 'OpenTerm' to a closed term for its type
  117 completeOpenTermType :: SharedContext -> OpenTerm -> IO Term
  118 completeOpenTermType sc (OpenTerm termM) =
  119   either (fail . show) return =<<
  120   runTCM (SC.ctermType <$> termM) sc mempty
  121 
  122 -- | Embed a closed 'Term' into an 'OpenTerm'
  123 closedOpenTerm :: Term -> OpenTerm
  124 closedOpenTerm t = OpenTerm $ typeInferComplete t
  125 
  126 -- | Embed a 'Term' in the given typing context into an 'OpenTerm'
  127 openOpenTerm :: [(VarName, Term)] -> Term -> OpenTerm
  128 openOpenTerm ctx t =
  129   -- Extend the local type-checking context, wherever this OpenTerm gets used,
  130   -- by appending ctx to the end, so that variables 0..length ctx-1 all get
  131   -- type-checked with ctx. If these are really the only free variables, then it
  132   -- won't matter what the rest of the ambient context is.
  133   --
  134   -- FIXME: we should check that the free variables of t are all < length ctx
  135   OpenTerm $ withCtx ctx $ typeInferComplete t
  136 
  137 -- | Build an 'OpenTerm' that 'fail's in the underlying monad when completed
  138 failOpenTerm :: String -> OpenTerm
  139 failOpenTerm str = OpenTerm $ fail str
  140 
  141 -- | Bind the result of a type-checking computation in building an 'OpenTerm'.
  142 -- NOTE: this operation should be considered \"unsafe\" because it can create
  143 -- malformed 'OpenTerm's if the result of the 'TCM' computation is used as part
  144 -- of the resulting 'OpenTerm'. For instance, @a@ should not be 'OpenTerm'.
  145 bindTCMOpenTerm :: TCM a -> (a -> OpenTerm) -> OpenTerm
  146 bindTCMOpenTerm m f = OpenTerm (m >>= unOpenTerm . f)
  147 
  148 -- | Bind the result of pretty-printing an 'OpenTerm' while building another
  149 bindPPOpenTerm :: OpenTerm -> (String -> OpenTerm) -> OpenTerm
  150 bindPPOpenTerm (OpenTerm m) f =
  151   OpenTerm $
  152   do t <- SC.ctermVal <$> m
  153      -- XXX: this could use scPrettyTermInCtx (which builds in the call to
  154      -- PPS.render) except that it's slightly different under the covers
  155      -- (in its use of the "global" flag, and it isn't entirely clear what
  156      -- that actually does)
  157      unOpenTerm $ f $ PPS.render PPS.defaultOpts $
  158        ppTermInCtx PPS.defaultOpts [] t

  160 -- | Return type type of an 'OpenTerm' as an 'OpenTerm
  161 openTermType :: OpenTerm -> OpenTerm
  162 openTermType (OpenTerm m) =
  163   OpenTerm $ do t <- m
  164                 liftTCM SC.scTypeOf t

  248 -- | Build a right-nested tuple as an 'OpenTerm'
  249 tupleOpenTerm :: [OpenTerm] -> OpenTerm
  250 tupleOpenTerm = foldr pairOpenTerm unitOpenTerm
  251 
  252 -- | Build a right-nested tuple type as an 'OpenTerm'
  253 tupleTypeOpenTerm :: [OpenTerm] -> OpenTerm
  254 tupleTypeOpenTerm = foldr pairTypeOpenTerm unitTypeOpenTerm

  256 -- | Project the @n@th element of a right-nested tuple type
  257 projTupleOpenTerm :: Integer -> OpenTerm -> OpenTerm
  258 projTupleOpenTerm 0 t = pairLeftOpenTerm t
  259 projTupleOpenTerm i t = projTupleOpenTerm (i-1) (pairRightOpenTerm t)
  260 
  261 -- | Build a right-nested tuple as an 'OpenTerm' but without adding a final unit
  262 -- as the right-most element
  263 tupleOpenTerm' :: [OpenTerm] -> OpenTerm
  264 tupleOpenTerm' [] = unitOpenTerm
  265 tupleOpenTerm' ts = foldr1 pairOpenTerm ts
  266 
  267 -- | Build a right-nested tuple type as an 'OpenTerm' but without adding a final
  268 -- unit type as the right-most element
  269 tupleTypeOpenTerm' :: [OpenTerm] -> OpenTerm
  270 tupleTypeOpenTerm' [] = unitTypeOpenTerm
  271 tupleTypeOpenTerm' ts = foldr1 pairTypeOpenTerm ts
  272 
  273 -- | Project the @i@th element from a term of a right-nested tuple term that
  274 -- does not have a final unit type as the right-most type. The first argument is
  275 -- the number of types used to make the tuple type and the second is the index.
  276 projTupleOpenTerm' :: Natural -> Natural -> OpenTerm -> OpenTerm
  277 projTupleOpenTerm' 0 _ _ =
  278   panic "projTupleOpenTerm'" ["Projection of empty tuple!"]
  279 projTupleOpenTerm' 1 0 tup = tup
  280 projTupleOpenTerm' _ 0 tup = pairLeftOpenTerm tup
  281 projTupleOpenTerm' len i tup =
  282   projTupleOpenTerm' (len-1) (i-1) $ pairRightOpenTerm tup
  283 
  284 -- | Build a record value as an 'OpenTerm'
  285 recordOpenTerm :: [(FieldName, OpenTerm)] -> OpenTerm
  286 recordOpenTerm flds_ts =
  287   OpenTerm $ do let (flds,ots) = unzip flds_ts
  288                 ts <- mapM unOpenTerm ots
  289                 typeInferComplete $ RecordValue $ zip flds ts
  290 
  291 -- | Build a record type as an 'OpenTerm'
  292 recordTypeOpenTerm :: [(FieldName, OpenTerm)] -> OpenTerm
  293 recordTypeOpenTerm flds_ts =
  294   OpenTerm $ do let (flds,ots) = unzip flds_ts
  295                 ts <- mapM unOpenTerm ots
  296                 typeInferComplete $ RecordType $ zip flds ts

  298 -- | Project a field from a record
  299 projRecordOpenTerm :: OpenTerm -> FieldName -> OpenTerm
  300 projRecordOpenTerm (OpenTerm m) f =
  301   OpenTerm $ do t <- m
  302                 typeInferComplete $ RecordProj t f

  304 -- | Build an 'OpenTerm' for a constructor applied to its arguments
  305 ctorOpenTerm :: Ident -> [OpenTerm] -> OpenTerm
  306 ctorOpenTerm c all_args = applyOpenTermMulti ctor_open_term all_args
  307   where
  308     ctor_open_term =
  309       OpenTerm $
  310       do maybe_ctor <- liftTCM scFindCtor c
  311          ctor <- case maybe_ctor of
  312                    Just ctor -> pure ctor
  313                    Nothing -> throwTCError $ NoSuchCtor (ModuleIdentifier c)
  314          typeInferComplete (Constant (ctorName ctor) :: TermF Term)
  315 
  316 -- | Build an 'OpenTerm' for a datatype applied to its arguments
  317 dataTypeOpenTerm :: Ident -> [OpenTerm] -> OpenTerm
  318 dataTypeOpenTerm d all_args = applyOpenTermMulti dt_open_term all_args
  319   where
  320     dt_open_term =
  321       OpenTerm $
  322       do maybe_dt <- liftTCM scFindDataType d
  323          dt <- case maybe_dt of
  324                  Just dt -> pure dt
  325                  Nothing -> throwTCError $ NoSuchDataType (ModuleIdentifier d)
  326          typeInferComplete (Constant (dtName dt) :: TermF Term)
  327 
  328 -- | Build an 'OpenTerm' for a global name with a definition

  343 -- | Apply an 'OpenTerm' to another
  344 applyOpenTerm :: OpenTerm -> OpenTerm -> OpenTerm
  345 applyOpenTerm (OpenTerm f) (OpenTerm arg) =
  346   OpenTerm ((App <$> f <*> arg) >>= typeInferComplete)
  347 
  348 -- | Apply an 'OpenTerm' to 0 or more arguments
  349 applyOpenTermMulti :: OpenTerm -> [OpenTerm] -> OpenTerm
  350 applyOpenTermMulti = foldl applyOpenTerm
  351 
  352 -- | Apply a named global to 0 or more arguments
  353 applyGlobalOpenTerm :: Ident -> [OpenTerm] -> OpenTerm
  354 applyGlobalOpenTerm ident = applyOpenTermMulti (globalOpenTerm ident)
  355 
  356 -- | Compute the output type of applying a function of a given type to an
  357 -- argument. That is, given @tp@ and @arg@, compute the type of applying any @f@
  358 -- of type @tp@ to @arg@.
  359 applyPiOpenTerm :: OpenTerm -> OpenTerm -> OpenTerm
  360 applyPiOpenTerm (OpenTerm m_f) (OpenTerm m_arg) =
  361   OpenTerm $
  362   do f <- m_f
  363      arg <- m_arg
  364      ret <- applyPiTyped (NotFuncTypeInApp f arg) (SC.ctermVal f) arg
  365      typeInferComplete ret
  366 
  367 -- | Get the argument type of a function type, 'fail'ing if the input term is
  368 -- not a function type
  369 piArgOpenTerm :: OpenTerm -> OpenTerm
  370 piArgOpenTerm (OpenTerm m) =
  371   OpenTerm $ m >>= \case
  372   (unwrapTermF . SC.ctermVal -> Pi _ tp _) -> typeInferComplete tp
  373   t ->
  374     fail ("piArgOpenTerm: not a pi type: " ++
  375           scPrettyTermInCtx PPS.defaultOpts [] (SC.ctermVal t))
  376 
  377 -- | Build a lambda abstraction as an 'OpenTerm'
  378 lambdaOpenTerm :: LocalName -> OpenTerm -> (OpenTerm -> OpenTerm) -> OpenTerm
  379 lambdaOpenTerm x (OpenTerm tpM) body_f = OpenTerm $
  380   do tp <- tpM
  381      vn <- liftTCM scFreshVarName x
  382      var <- typeInferComplete $ Variable vn tp
  383      body <- unOpenTerm (body_f (OpenTerm (pure var)))
  384      typeInferComplete $ Lambda vn tp body
  385 
  386 -- | Build a nested sequence of lambda abstractions as an 'OpenTerm'
  387 lambdaOpenTermMulti :: [(LocalName, OpenTerm)] -> ([OpenTerm] -> OpenTerm) ->
  388                        OpenTerm
  389 lambdaOpenTermMulti xs_tps body_f =
  390   foldr (\(x,tp) rest_f xs ->
  391           lambdaOpenTerm x tp (rest_f . (:xs))) (body_f . reverse) xs_tps []
  392 
  393 -- | Build a Pi abstraction as an 'OpenTerm'
  394 piOpenTerm :: LocalName -> OpenTerm -> (OpenTerm -> OpenTerm) -> OpenTerm
  395 piOpenTerm x (OpenTerm tpM) body_f = OpenTerm $
  396   do tp <- tpM
  397      nm <- liftTCM scFreshVarName x
  398      var <- typeInferComplete $ Variable nm tp
  399      body <- unOpenTerm (body_f (OpenTerm (pure var)))
  400      typeInferComplete $ Pi nm tp body
  401 
  402 -- | Build a non-dependent function type.
  403 arrowOpenTerm :: LocalName -> OpenTerm -> OpenTerm -> OpenTerm
  404 arrowOpenTerm x tp body = piOpenTerm x tp (const body)
  405 
  406 -- | Build a nested sequence of Pi abstractions as an 'OpenTerm'
  407 piOpenTermMulti :: [(LocalName, OpenTerm)] -> ([OpenTerm] -> OpenTerm) ->
  408                    OpenTerm
  409 piOpenTermMulti xs_tps body_f =
  410   foldr (\(x,tp) rest_f xs ->
  411           piOpenTerm x tp (rest_f . (:xs))) (body_f . reverse) xs_tps []

  413 -- | Build a let expression as an 'OpenTerm'. This is equivalent to
  414 -- > 'applyOpenTerm' ('lambdaOpenTerm' x tp body) rhs
  415 letOpenTerm :: LocalName -> OpenTerm -> OpenTerm -> (OpenTerm -> OpenTerm) ->
  416                OpenTerm
  417 letOpenTerm x tp rhs body_f = applyOpenTerm (lambdaOpenTerm x tp body_f) rhs
  418 
  419 -- | Build a let expression as an 'OpenTerm'. This is equivalent to
  420 -- > 'applyOpenTerm' ('lambdaOpenTerm' x tp body) rhs
  421 sawLetOpenTerm :: LocalName -> OpenTerm -> OpenTerm -> OpenTerm ->
  422                   (OpenTerm -> OpenTerm) -> OpenTerm
  423 sawLetOpenTerm x tp tp_ret rhs body_f =
  424   applyOpenTermMulti (globalOpenTerm "Prelude.sawLet")
  425   [tp, tp_ret, rhs, lambdaOpenTerm x tp body_f]
  426 
  427 -- | Build a bitvector type with the given length
  428 bitvectorTypeOpenTerm :: OpenTerm -> OpenTerm
  429 bitvectorTypeOpenTerm w =
  430   applyGlobalOpenTerm "Prelude.Vec" [w, globalOpenTerm "Prelude.Bool"]
  431 
  432 -- | Build a SAW core term for a list with the given element type
  433 listOpenTerm :: OpenTerm -> [OpenTerm] -> OpenTerm
  434 listOpenTerm tp elems =
  435   foldr (\x l -> ctorOpenTerm "Prelude.Cons" [tp, x, l])
  436   (ctorOpenTerm "Prelude.Nil" [tp]) elems

  438 -- | Build the type @Either a b@ from types @a@ and @b@
  439 eitherTypeOpenTerm :: OpenTerm -> OpenTerm -> OpenTerm

  445 class OpenTermLike t where
  446   -- | Convert an 'OpenTerm' to a @t@
  447   openTermLike :: OpenTerm -> t
  448   -- | Get the type of a @t@
  449   typeOfTermLike :: t -> t
  450   -- | Build a @t@ from a 'FlatTermF'
  451   flatTermLike :: FlatTermF t -> t
  452   -- | Apply a @t@ to another @t@
  453   applyTermLike :: t -> t -> t
  454   -- | Build a lambda abstraction as a @t@
  455   lambdaTermLike :: LocalName -> t -> (t -> t) -> t
  456   -- | Build a pi abstraction as a @t@
  457   piTermLike :: LocalName -> t -> (t -> t) -> t
  458   -- | Build a @t@ for a constructor applied to its arguments
  459   ctorTermLike :: Ident -> [t] -> t
  460   -- | Build a @t@ for a datatype applied to its arguments
  461   dataTypeTermLike :: Ident -> [t] -> t
  462 
  463 instance OpenTermLike OpenTerm where
  464   openTermLike = id
  465   typeOfTermLike = openTermType
  466   flatTermLike = flatOpenTerm
  467   applyTermLike = applyOpenTerm
  468   lambdaTermLike = lambdaOpenTerm
  469   piTermLike = piOpenTerm
  470   ctorTermLike = ctorOpenTerm
  471   dataTypeTermLike = dataTypeOpenTerm
  472 
  473 -- Lift an OpenTermLike instance from t to functions from some type a to t,
  474 -- where the OpenTermLike methods pass the same input a argument to all subterms
  475 instance OpenTermLike t => OpenTermLike (a -> t) where
  476   openTermLike t = const $ openTermLike t
  477   typeOfTermLike t = \x -> typeOfTermLike (t x)
  478   flatTermLike ftf = \x -> flatTermLike (fmap ($ x) ftf)
  479   applyTermLike f arg = \x -> applyTermLike (f x) (arg x)
  480   lambdaTermLike nm tp bodyF =
  481     \x -> lambdaTermLike nm (tp x) (\y -> bodyF (const y) x)
  482   piTermLike nm tp bodyF =
  483     \x -> piTermLike nm (tp x) (\y -> bodyF (const y) x)
  484   ctorTermLike c args = \x -> ctorTermLike c (map ($ x) args)
  485   dataTypeTermLike d args = \x -> dataTypeTermLike d (map ($ x) args)
  486 
  487 -- This is the same as the function instance above
  488 instance OpenTermLike t => OpenTermLike (Reader r t) where
  489   openTermLike t = reader $ openTermLike t
  490   typeOfTermLike t = reader $ typeOfTermLike $ runReader t
  491   flatTermLike ftf = reader $ flatTermLike $ fmap runReader ftf
  492   applyTermLike f arg = reader $ applyTermLike (runReader f) (runReader arg)
  493   lambdaTermLike x tp body =
  494     reader $ lambdaTermLike x (runReader tp) (runReader . body . reader)
  495   piTermLike x tp body =
  496     reader $ piTermLike x (runReader tp) (runReader . body . reader)
  497   ctorTermLike c args = reader $ ctorTermLike c $ map runReader args
  498   dataTypeTermLike d args = reader $ dataTypeTermLike d $ map runReader args
  499 
  500 -- | Build a nested sequence of lambda abstractions
  501 lambdaTermLikeMulti :: OpenTermLike t => [(LocalName, t)] -> ([t] -> t) -> t
  502 lambdaTermLikeMulti xs_tps body_f =
  503   foldr (\(x,tp) rest_f xs ->
  504           lambdaTermLike x tp (rest_f . (:xs))) (body_f . reverse) xs_tps []

  608 -- | Build a let expression as a term. This is equivalent to
  609 -- > 'applyTermLike' ('lambdaTermLike' x tp body) rhs
  610 letTermLike :: OpenTermLike t => LocalName -> t -> t -> (t -> t) -> t
  611 letTermLike x tp rhs body_f = applyTermLike (lambdaTermLike x tp body_f) rhs

  613 -- | Build a let expression as a term using the @sawLet@ combinator. This
  614 -- is equivalent to the term @sawLet tp tp_ret rhs (\ (x : tp) -> body_f)@
  615 sawLetTermLike :: OpenTermLike t => LocalName -> t -> t -> t -> (t -> t) -> t
  616 sawLetTermLike x tp tp_ret rhs body_f =
  617   applyTermLikeMulti (globalTermLike "Prelude.sawLet")
  618   [tp, tp_ret, rhs, lambdaTermLike x tp body_f]
