    9 module SAWScript.Token where
   10 
   11 import SAWScript.Position (Positioned(..))
   12 
   13 data Token p = TVar      { tokPos :: p, tokStr :: String                               }
   14              | TQVar     { tokPos :: p, tokStr :: String, tokVars :: ([String],String) }
   15              | TLit      { tokPos :: p, tokStr :: String                               }
   16              | TCode     { tokPos :: p, tokStr :: String                               }
   17              | TCType    { tokPos :: p, tokStr :: String                               }
   18              | TUnknown  { tokPos :: p, tokStr :: String                               }
   19              | TPunct    { tokPos :: p, tokStr :: String                               }
   20              | TReserved { tokPos :: p, tokStr :: String                               }
   21              | TOp       { tokPos :: p, tokStr :: String                               }
   22              | TNum      { tokPos :: p, tokStr :: String, tokNum :: Integer            }
   23              | TCmntS    { tokPos :: p, tokStr :: String                               }
   24              | TCmntE    { tokPos :: p, tokStr :: String                               }
   25              | TEOF      { tokPos :: p, tokStr :: String                               }
   26              deriving (Show, Functor)
   27 
   28 instance Positioned p => Positioned (Token p) where
   29   getPos = getPos . tokPos
